19__rte_experimental void
20rte_bpf_destroy(struct rte_bpf *bpf)
21{
22 if (bpf != NULL) {
23 if (bpf->jit.func != NULL)
24 munmap(bpf->jit.func, bpf->jit.sz);
25 munmap(bpf, bpf->sz);
26 }
27}
28
29__rte_experimental int
30rte_bpf_get_jit(const struct rte_bpf *bpf, struct rte_bpf_jit *jit)
31{
32 if (bpf == NULL || jit == NULL)
33 return -EINVAL;
34
35 jit[0] = bpf->jit;
36 return 0;
37}
38
39int
40bpf_jit(struct rte_bpf *bpf)
41{
42 int32_t rc;
43
44#ifdef RTE_ARCH_X86_64
45 rc = bpf_jit_x86(bpf);
46#else
47 rc = -ENOTSUP;
48#endif
49
50 if (rc != 0)
51 RTE_BPF_LOG(WARNING, "%s(%p) failed, error code: %d;\n",
52 __func__, bpf, rc);
53 return rc;
54}

58static void
59rte_bpf_init_log(void)
60{
61 rte_bpf_logtype = rte_log_register("lib.bpf");
62 if (rte_bpf_logtype >= 0)
63 rte_log_set_level(rte_bpf_logtype, RTE_LOG_INFO);
64}
