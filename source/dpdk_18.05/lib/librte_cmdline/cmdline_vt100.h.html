68extern const char *cmdline_vt100_commands[];

70enum cmdline_vt100_parser_state {
71 CMDLINE_VT100_INIT,
72 CMDLINE_VT100_ESCAPE,
73 CMDLINE_VT100_ESCAPE_CSI
74};
75
76#define CMDLINE_VT100_BUF_SIZE 8
77struct cmdline_vt100 {
78 uint8_t bufpos;
79 char buf[CMDLINE_VT100_BUF_SIZE];
80 enum cmdline_vt100_parser_state state;
81};
82
83/**
84 * Init
85 */
86void vt100_init(struct cmdline_vt100 *vt);
87
88/**
89 * Input a new character.
90 * Return -1 if the character is not part of a control sequence
91 * Return -2 if c is not the last char of a control sequence
92 * Else return the index in vt100_commands[]
93 */
94int vt100_parser(struct cmdline_vt100 *vt, char c);
