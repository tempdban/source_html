171 struct {

176 struct {

191 } hi_dword;
192 } lower;
193 struct {
194 __le32 status_error; /* ext status/error */
195 __le16 length; /* Packet length */
196 __le16 vlan; /* VLAN tag */
197 } upper;
198 } wb; /* writeback */
199};
