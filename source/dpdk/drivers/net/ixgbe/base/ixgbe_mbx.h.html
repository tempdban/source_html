34#ifndef _IXGBE_MBX_H_
35#define _IXGBE_MBX_H_

37#include "ixgbe_type.h"

39#define IXGBE_VFMAILBOX_SIZE 16 /* 16 32 bit words - 64 bytes */
40#define IXGBE_ERR_MBX -100
41
42#define IXGBE_VFMAILBOX 0x002FC
43#define IXGBE_VFMBMEM 0x00200
44
45/* Define mailbox register bits */
46#define IXGBE_VFMAILBOX_REQ 0x00000001 /* Request for PF Ready bit */
47#define IXGBE_VFMAILBOX_ACK 0x00000002 /* Ack PF message received */
48#define IXGBE_VFMAILBOX_VFU 0x00000004 /* VF owns the mailbox buffer */
49#define IXGBE_VFMAILBOX_PFU 0x00000008 /* PF owns the mailbox buffer */
50#define IXGBE_VFMAILBOX_PFSTS 0x00000010 /* PF wrote a message in the MB */
51#define IXGBE_VFMAILBOX_PFACK 0x00000020 /* PF ack the previous VF msg */
52#define IXGBE_VFMAILBOX_RSTI 0x00000040 /* PF has reset indication */
53#define IXGBE_VFMAILBOX_RSTD 0x00000080 /* PF has indicated reset done */
54#define IXGBE_VFMAILBOX_R2C_BITS 0x000000B0 /* All read to clear bits */
55
56#define IXGBE_PFMAILBOX_STS 0x00000001 /* Initiate message send to VF */
57#define IXGBE_PFMAILBOX_ACK 0x00000002 /* Ack message recv'd from VF */
58#define IXGBE_PFMAILBOX_VFU 0x00000004 /* VF owns the mailbox buffer */
59#define IXGBE_PFMAILBOX_PFU 0x00000008 /* PF owns the mailbox buffer */
60#define IXGBE_PFMAILBOX_RVFU 0x00000010 /* Reset VFU - used when VF stuck */
61
62#define IXGBE_MBVFICR_VFREQ_MASK 0x0000FFFF /* bits for VF messages */
63#define IXGBE_MBVFICR_VFREQ_VF1 0x00000001 /* bit for VF 1 message */
64#define IXGBE_MBVFICR_VFACK_MASK 0xFFFF0000 /* bits for VF acks */
65#define IXGBE_MBVFICR_VFACK_VF1 0x00010000 /* bit for VF 1 ack */
66
67
68/* If it's a IXGBE_VF_* msg then it originates in the VF and is sent to the
69 * PF. The reverse is true if it is IXGBE_PF_*.
70 * Message ACK's are the value or'd with 0xF0000000
71 */
72#define IXGBE_VT_MSGTYPE_ACK 0x80000000 /* Messages below or'd with
73 * this are the ACK */
74#define IXGBE_VT_MSGTYPE_NACK 0x40000000 /* Messages below or'd with
75 * this are the NACK */
76#define IXGBE_VT_MSGTYPE_CTS 0x20000000 /* Indicates that VF is still
77 * clear to send requests */
78#define IXGBE_VT_MSGINFO_SHIFT 16
79/* bits 23:16 are used for extra info for certain messages */
80#define IXGBE_VT_MSGINFO_MASK (0xFF << IXGBE_VT_MSGINFO_SHIFT)
81
82/* definitions to support mailbox API version negotiation */
83
84/*
85 * each element denotes a version of the API; existing numbers may not
86 * change; any additions must go at the end
87 */
88enum ixgbe_pfvf_api_rev {
89 ixgbe_mbox_api_10, /* API version 1.0, linux/freebsd VF driver */
90 ixgbe_mbox_api_20, /* API version 2.0, solaris Phase1 VF driver */
91 ixgbe_mbox_api_11, /* API version 1.1, linux/freebsd VF driver */
92 ixgbe_mbox_api_12, /* API version 1.2, linux/freebsd VF driver */
93 ixgbe_mbox_api_13, /* API version 1.3, linux/freebsd VF driver */
94 /* This value should always be last */
95 ixgbe_mbox_api_unknown, /* indicates that API version is not known */
96};
97
98/* mailbox API, legacy requests */
99#define IXGBE_VF_RESET 0x01 /* VF requests reset */
100#define IXGBE_VF_SET_MAC_ADDR 0x02 /* VF requests PF to set MAC addr */
101#define IXGBE_VF_SET_MULTICAST 0x03 /* VF requests PF to set MC addr */
102#define IXGBE_VF_SET_VLAN 0x04 /* VF requests PF to set VLAN */
103
104/* mailbox API, version 1.0 VF requests */
105#define IXGBE_VF_SET_LPE 0x05 /* VF requests PF to set VMOLR.LPE */
106#define IXGBE_VF_SET_MACVLAN 0x06 /* VF requests PF for unicast filter */
107#define IXGBE_VF_API_NEGOTIATE 0x08 /* negotiate API version */
108
109/* mailbox API, version 1.1 VF requests */
110#define IXGBE_VF_GET_QUEUES 0x09 /* get queue configuration */
111
112/* mailbox API, version 1.2 VF requests */
113#define IXGBE_VF_GET_RETA 0x0a /* VF request for RETA */
114#define IXGBE_VF_GET_RSS_KEY 0x0b /* get RSS key */
115#define IXGBE_VF_UPDATE_XCAST_MODE 0x0c
116
117/* mode choices for IXGBE_VF_UPDATE_XCAST_MODE */
118enum ixgbevf_xcast_modes {
119 IXGBEVF_XCAST_MODE_NONE = 0,
120 IXGBEVF_XCAST_MODE_MULTI,
121 IXGBEVF_XCAST_MODE_ALLMULTI,
122 IXGBEVF_XCAST_MODE_PROMISC,
123};
124
125/* GET_QUEUES return data indices within the mailbox */
126#define IXGBE_VF_TX_QUEUES 1 /* number of Tx queues supported */
127#define IXGBE_VF_RX_QUEUES 2 /* number of Rx queues supported */
128#define IXGBE_VF_TRANS_VLAN 3 /* Indication of port vlan */
129#define IXGBE_VF_DEF_QUEUE 4 /* Default queue offset */
130
131/* length of permanent address message returned from PF */
132#define IXGBE_VF_PERMADDR_MSG_LEN 4
133/* word in permanent address message with the current multicast type */
134#define IXGBE_VF_MC_TYPE_WORD 3
135
136#define IXGBE_PF_CONTROL_MSG 0x0100 /* PF control message */
137
138/* mailbox API, version 2.0 VF requests */
139#define IXGBE_VF_API_NEGOTIATE 0x08 /* negotiate API version */
140#define IXGBE_VF_GET_QUEUES 0x09 /* get queue configuration */
141#define IXGBE_VF_ENABLE_MACADDR 0x0A /* enable MAC address */
142#define IXGBE_VF_DISABLE_MACADDR 0x0B /* disable MAC address */
143#define IXGBE_VF_GET_MACADDRS 0x0C /* get all configured MAC addrs */
144#define IXGBE_VF_SET_MCAST_PROMISC 0x0D /* enable multicast promiscuous */
145#define IXGBE_VF_GET_MTU 0x0E /* get bounds on MTU */
146#define IXGBE_VF_SET_MTU 0x0F /* set a specific MTU */
147
148/* mailbox API, version 2.0 PF requests */
149#define IXGBE_PF_TRANSPARENT_VLAN 0x0101 /* enable transparent vlan */
150
