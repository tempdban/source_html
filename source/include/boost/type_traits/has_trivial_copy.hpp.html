23namespace boost {
24
25namespace detail {
26
27template <typename T>
28struct has_trivial_copy_impl
29{
30#ifdef BOOST_HAS_TRIVIAL_COPY
31 BOOST_STATIC_CONSTANT(bool, value = BOOST_HAS_TRIVIAL_COPY(T));
32#else
33 BOOST_STATIC_CONSTANT(bool, value =
34 (::boost::type_traits::ice_and<
35 ::boost::is_pod<T>::value,
36 ::boost::type_traits::ice_not< ::boost::is_volatile<T>::value >::value
37 >::value));
38#endif
39};
40
41} // namespace detail
42
43BOOST_TT_AUX_BOOL_TRAIT_DEF1(has_trivial_copy,T,::boost::detail::has_trivial_copy_impl<T>::value)
44BOOST_TT_AUX_BOOL_TRAIT_DEF1(has_trivial_copy_constructor,T,::boost::detail::has_trivial_copy_impl<T>::value)
45
46BOOST_TT_AUX_BOOL_TRAIT_SPEC1(has_trivial_copy,void,false)
47#ifndef BOOST_NO_CV_VOID_SPECIALIZATIONS
48BOOST_TT_AUX_BOOL_TRAIT_SPEC1(has_trivial_copy,void const,false)
49BOOST_TT_AUX_BOOL_TRAIT_SPEC1(has_trivial_copy,void const volatile,false)
50BOOST_TT_AUX_BOOL_TRAIT_SPEC1(has_trivial_copy,void volatile,false)
51#endif
52
53BOOST_TT_AUX_BOOL_TRAIT_SPEC1(has_trivial_copy_constructor,void,false)
54#ifndef BOOST_NO_CV_VOID_SPECIALIZATIONS
55BOOST_TT_AUX_BOOL_TRAIT_SPEC1(has_trivial_copy_constructor,void const,false)
56BOOST_TT_AUX_BOOL_TRAIT_SPEC1(has_trivial_copy_constructor,void const volatile,false)
57BOOST_TT_AUX_BOOL_TRAIT_SPEC1(has_trivial_copy_constructor,void volatile,false)
58#endif
59
60} // namespace boost
