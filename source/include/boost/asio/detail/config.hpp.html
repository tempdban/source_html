61// If BOOST_ASIO_MOVE_CAST isn't defined, and move support is available, define
62// BOOST_ASIO_MOVE_ARG and BOOST_ASIO_MOVE_CAST to take advantage of rvalue
63// references and perfect forwarding.
64#if defined(BOOST_ASIO_HAS_MOVE) && !defined(BOOST_ASIO_MOVE_CAST)
65# define BOOST_ASIO_MOVE_ARG(type) type&&
66# define BOOST_ASIO_MOVE_CAST(type) static_cast<type&&>
67#endif // defined(BOOST_ASIO_HAS_MOVE) && !defined(BOOST_ASIO_MOVE_CAST)
68
69// If BOOST_ASIO_MOVE_CAST still isn't defined, default to a C++03-compatible
70// implementation. Note that older g++ and MSVC versions don't like it when you
71// pass a non-member function through a const reference, so for most compilers
72// we'll play it safe and stick with the old approach of passing the handler by
73// value.
74#if !defined(BOOST_ASIO_MOVE_CAST)
75# if defined(__GNUC__)
76# if ((__GNUC__ == 4) && (__GNUC_MINOR__ >= 1)) || (__GNUC__ > 4)
77# define BOOST_ASIO_MOVE_ARG(type) const type&
78# else // ((__GNUC__ == 4) && (__GNUC_MINOR__ >= 1)) || (__GNUC__ > 4)
79# define BOOST_ASIO_MOVE_ARG(type) type
80# endif // ((__GNUC__ == 4) && (__GNUC_MINOR__ >= 1)) || (__GNUC__ > 4)
81# elif defined(BOOST_MSVC)
82# if (_MSC_VER >= 1400)
83# define BOOST_ASIO_MOVE_ARG(type) const type&
84# else // (_MSC_VER >= 1400)
85# define BOOST_ASIO_MOVE_ARG(type) type
86# endif // (_MSC_VER >= 1400)
87# else
88# define BOOST_ASIO_MOVE_ARG(type) type
89# endif
90# define BOOST_ASIO_MOVE_CAST(type) static_cast<const type&>
91#endif // !defined_BOOST_ASIO_MOVE_CAST
