82socket_type accept(socket_type s, socket_addr_type* addr,
83 std::size_t* addrlen, boost::system::error_code& ec)
84{
85 if (s == invalid_socket)
86 {
87 ec = boost::asio::error::bad_descriptor;
88 return invalid_socket;
89 }
90
91 clear_last_error();
92
93 socket_type new_s = error_wrapper(call_accept(
94 &msghdr::msg_namelen, s, addr, addrlen), ec);
95 if (new_s == invalid_socket)
96 return new_s;
97
98#if defined(__MACH__) && defined(__APPLE__) || defined(__FreeBSD__)
99 int optval = 1;
100 int result = error_wrapper(::setsockopt(new_s,
101 SOL_SOCKET, SO_NOSIGPIPE, &optval, sizeof(optval)), ec);
102 if (result != 0)
103 {
104 ::close(new_s);
105 return invalid_socket;
106 }
107#endif
108
109 ec = boost::system::error_code();
110 return new_s;
111}

113socket_type sync_accept(socket_type s, state_type state,
114 socket_addr_type* addr, std::size_t* addrlen, boost::system::error_code& ec)
115{
116 // Accept a socket.
117 for (;;)
118 {
119 // Try to complete the operation without blocking.
120 socket_type new_socket = socket_ops::accept(s, addr, addrlen, ec);
121
122 // Check if operation succeeded.
123 if (new_socket != invalid_socket)
124 return new_socket;
125
126 // Operation failed.
127 if (ec == boost::asio::error::would_block
128 || ec == boost::asio::error::try_again)
129 {
130 if (state & user_set_non_blocking)
131 return invalid_socket;
132 // Fall through to retry operation.
133 }
134 else if (ec == boost::asio::error::connection_aborted)
135 {
136 if (state & enable_connection_aborted)
137 return invalid_socket;
138 // Fall through to retry operation.
139 }
140#if defined(EPROTO)
141 else if (ec.value() == EPROTO)
142 {
143 if (state & enable_connection_aborted)
144 return invalid_socket;
145 // Fall through to retry operation.
146 }
147#endif // defined(EPROTO)
148 else
149 return invalid_socket;
150
151 // Wait for socket to become ready.
152 if (socket_ops::poll_read(s, 0, ec) < 0)
153 return invalid_socket;
154 }
155}

159void complete_iocp_accept(socket_type s,
160 void* output_buffer, DWORD address_length,
161 socket_addr_type* addr, std::size_t* addrlen,
162 socket_type new_socket, boost::system::error_code& ec)
163{
164 // Map non-portable errors to their portable counterparts.
165 if (ec.value() == ERROR_NETNAME_DELETED)
166 ec = boost::asio::error::connection_aborted;
167
168 if (!ec)
169 {
170 // Get the address of the peer.
171 if (addr && addrlen)
172 {
173 LPSOCKADDR local_addr = 0;
174 int local_addr_length = 0;
175 LPSOCKADDR remote_addr = 0;
176 int remote_addr_length = 0;
177 GetAcceptExSockaddrs(output_buffer, 0, address_length,
178 address_length, &local_addr, &local_addr_length,
179 &remote_addr, &remote_addr_length);
180 if (static_cast<std::size_t>(remote_addr_length) > *addrlen)
181 {
182 ec = boost::asio::error::invalid_argument;
183 }
184 else
185 {
186 using namespace std; // For memcpy.
187 memcpy(addr, remote_addr, remote_addr_length);
188 *addrlen = static_cast<std::size_t>(remote_addr_length);
189 }
190 }
191
192 // Need to set the SO_UPDATE_ACCEPT_CONTEXT option so that getsockname
193 // and getpeername will work on the accepted socket.
194 SOCKET update_ctx_param = s;
195 socket_ops::state_type state = 0;
196 socket_ops::setsockopt(new_socket, state,
197 SOL_SOCKET, SO_UPDATE_ACCEPT_CONTEXT,
198 &update_ctx_param, sizeof(SOCKET), ec);
199 }
200}

204bool non_blocking_accept(socket_type s,
205 state_type state, socket_addr_type* addr, std::size_t* addrlen,
206 boost::system::error_code& ec, socket_type& new_socket)
207{
208 for (;;)
209 {
210 // Accept the waiting connection.
211 new_socket = socket_ops::accept(s, addr, addrlen, ec);
212
213 // Check if operation succeeded.
214 if (new_socket != invalid_socket)
215 return true;
216
217 // Retry operation if interrupted by signal.
218 if (ec == boost::asio::error::interrupted)
219 continue;
220
221 // Operation failed.
222 if (ec == boost::asio::error::would_block
223 || ec == boost::asio::error::try_again)
224 {
225 if (state & user_set_non_blocking)
226 return true;
227 // Fall through to retry operation.
228 }
229 else if (ec == boost::asio::error::connection_aborted)
230 {
231 if (state & enable_connection_aborted)
232 return true;
233 // Fall through to retry operation.
234 }
235#if defined(EPROTO)
236 else if (ec.value() == EPROTO)
237 {
238 if (state & enable_connection_aborted)
239 return true;
240 // Fall through to retry operation.
241 }
242#endif // defined(EPROTO)
243 else
244 return true;
245
246 return false;
247 }
248}

276int close(socket_type s, state_type& state,
277 bool destruction, boost::system::error_code& ec)
278{
279 int result = 0;
280 if (s != invalid_socket)
281 {
282 // We don't want the destructor to block, so set the socket to linger in
283 // the background. If the user doesn't like this behaviour then they need
284 // to explicitly close the socket.
285 if (destruction && (state & user_set_linger))
286 {
287 ::linger opt;
288 opt.l_onoff = 0;
289 opt.l_linger = 0;
290 boost::system::error_code ignored_ec;
291 socket_ops::setsockopt(s, state, SOL_SOCKET,
292 SO_LINGER, &opt, sizeof(opt), ignored_ec);
293 }
294
295 clear_last_error();
296#if defined(BOOST_WINDOWS) || defined(__CYGWIN__)
297 result = error_wrapper(::closesocket(s), ec);
298#else // defined(BOOST_WINDOWS) || defined(__CYGWIN__)
299 result = error_wrapper(::close(s), ec);
300#endif // defined(BOOST_WINDOWS) || defined(__CYGWIN__)
301
302 if (result != 0
303 && (ec == boost::asio::error::would_block
304 || ec == boost::asio::error::try_again))
305 {
306 // According to UNIX Network Programming Vol. 1, it is possible for
307 // close() to fail with EWOULDBLOCK under certain circumstances. What
308 // isn't clear is the state of the descriptor after this error. The one
309 // current OS where this behaviour is seen, Windows, says that the socket
310 // remains open. Therefore we'll put the descriptor back into blocking
311 // mode and have another attempt at closing it.
312#if defined(BOOST_WINDOWS) || defined(__CYGWIN__)
313 ioctl_arg_type arg = 0;
314 ::ioctlsocket(s, FIONBIO, &arg);
315#else // defined(BOOST_WINDOWS) || defined(__CYGWIN__)
316# if defined(__SYMBIAN32__)
317 int flags = ::fcntl(s, F_GETFL, 0);
318 if (flags >= 0)
319 ::fcntl(s, F_SETFL, flags & ~O_NONBLOCK);
320# else // defined(__SYMBIAN32__)
321 ioctl_arg_type arg = 0;
322 ::ioctl(s, FIONBIO, &arg);
323# endif // defined(__SYMBIAN32__)
324#endif // defined(BOOST_WINDOWS) || defined(__CYGWIN__)
325 state &= ~non_blocking;
326
327 clear_last_error();
328#if defined(BOOST_WINDOWS) || defined(__CYGWIN__)
329 result = error_wrapper(::closesocket(s), ec);
330#else // defined(BOOST_WINDOWS) || defined(__CYGWIN__)
331 result = error_wrapper(::close(s), ec);
332#endif // defined(BOOST_WINDOWS) || defined(__CYGWIN__)
333 }
334 }
335
336 if (result == 0)
337 ec = boost::system::error_code();
338 return result;
339}

341bool set_user_non_blocking(socket_type s,
342 state_type& state, bool value, boost::system::error_code& ec)
343{
344 if (s == invalid_socket)
345 {
346 ec = boost::asio::error::bad_descriptor;
347 return false;
348 }
349
350 clear_last_error();
351#if defined(BOOST_WINDOWS) || defined(__CYGWIN__)
352 ioctl_arg_type arg = (value ? 1 : 0);
353 int result = error_wrapper(::ioctlsocket(s, FIONBIO, &arg), ec);
354#elif defined(__SYMBIAN32__)
355 int result = error_wrapper(::fcntl(s, F_GETFL, 0), ec);
356 if (result >= 0)
357 {
358 clear_last_error();
359 int flag = (value ? (result | O_NONBLOCK) : (result & ~O_NONBLOCK));
360 result = error_wrapper(::fcntl(s, F_SETFL, flag), ec);
361 }
362#else
363 ioctl_arg_type arg = (value ? 1 : 0);
364 int result = error_wrapper(::ioctl(s, FIONBIO, &arg), ec);
365#endif
366
367 if (result >= 0)
368 {
369 ec = boost::system::error_code();
370 if (value)
371 state |= user_set_non_blocking;
372 else
373 {
374 // Clearing the user-set non-blocking mode always overrides any
375 // internally-set non-blocking flag. Any subsequent asynchronous
376 // operations will need to re-enable non-blocking I/O.
377 state &= ~(user_set_non_blocking | internal_non_blocking);
378 }
379 return true;
380 }
381
382 return false;
383}

385bool set_internal_non_blocking(socket_type s,
386 state_type& state, bool value, boost::system::error_code& ec)
387{
388 if (s == invalid_socket)
389 {
390 ec = boost::asio::error::bad_descriptor;
391 return false;
392 }
393
394 if (!value && (state & user_set_non_blocking))
395 {
396 // It does not make sense to clear the internal non-blocking flag if the
397 // user still wants non-blocking behaviour. Return an error and let the
398 // caller figure out whether to update the user-set non-blocking flag.
399 ec = boost::asio::error::invalid_argument;
400 return false;
401 }
402
403 clear_last_error();
404#if defined(BOOST_WINDOWS) || defined(__CYGWIN__)
405 ioctl_arg_type arg = (value ? 1 : 0);
406 int result = error_wrapper(::ioctlsocket(s, FIONBIO, &arg), ec);
407#elif defined(__SYMBIAN32__)
408 int result = error_wrapper(::fcntl(s, F_GETFL, 0), ec);
409 if (result >= 0)
410 {
411 clear_last_error();
412 int flag = (value ? (result | O_NONBLOCK) : (result & ~O_NONBLOCK));
413 result = error_wrapper(::fcntl(s, F_SETFL, flag), ec);
414 }
415#else
416 ioctl_arg_type arg = (value ? 1 : 0);
417 int result = error_wrapper(::ioctl(s, FIONBIO, &arg), ec);
418#endif
419
420 if (result >= 0)
421 {
422 ec = boost::system::error_code();
423 if (value)
424 state |= internal_non_blocking;
425 else
426 state &= ~internal_non_blocking;
427 return true;
428 }
429
430 return false;
431}

476void sync_connect(socket_type s, const socket_addr_type* addr,
477 std::size_t addrlen, boost::system::error_code& ec)
478{
479 // Perform the connect operation.
480 socket_ops::connect(s, addr, addrlen, ec);
481 if (ec != boost::asio::error::in_progress
482 && ec != boost::asio::error::would_block)
483 {
484 // The connect operation finished immediately.
485 return;
486 }
487
488 // Wait for socket to become ready.
489 if (socket_ops::poll_connect(s, ec) < 0)
490 return;
491
492 // Get the error code from the connect operation.
493 int connect_error = 0;
494 size_t connect_error_len = sizeof(connect_error);
495 if (socket_ops::getsockopt(s, 0, SOL_SOCKET, SO_ERROR,
496 &connect_error, &connect_error_len, ec) == socket_error_retval)
497 return;
498
499 // Return the result of the connect operation.
500 ec = boost::system::error_code(connect_error,
501 boost::asio::error::get_system_category());
502}

504bool non_blocking_connect(socket_type s, boost::system::error_code& ec)
505{
506 // Get the error code from the connect operation.
507 int connect_error = 0;
508 size_t connect_error_len = sizeof(connect_error);
509 if (socket_ops::getsockopt(s, 0, SOL_SOCKET, SO_ERROR,
510 &connect_error, &connect_error_len, ec) == 0)
511 {
512 if (connect_error)
513 {
514 ec = boost::system::error_code(connect_error,
515 boost::asio::error::get_system_category());
516 }
517 else
518 ec = boost::system::error_code();
519 }
520
521 return true;
522}

568#endif // defined(SIOCATMARK)
569
570 return ec ? false : value != 0;

673signed_size_type recv(socket_type s, buf* bufs, size_t count,
674 int flags, boost::system::error_code& ec)
675{
676 clear_last_error();
677#if defined(BOOST_WINDOWS) || defined(__CYGWIN__)
678 // Receive some data.
679 DWORD recv_buf_count = static_cast<DWORD>(count);
680 DWORD bytes_transferred = 0;
681 DWORD recv_flags = flags;
682 int result = error_wrapper(::WSARecv(s, bufs,
683 recv_buf_count, &bytes_transferred, &recv_flags, 0, 0), ec);
684 if (ec.value() == ERROR_NETNAME_DELETED)
685 ec = boost::asio::error::connection_reset;
686 else if (ec.value() == ERROR_PORT_UNREACHABLE)
687 ec = boost::asio::error::connection_refused;
688 if (result != 0)
689 return socket_error_retval;
690 ec = boost::system::error_code();
691 return bytes_transferred;
692#else // defined(BOOST_WINDOWS) || defined(__CYGWIN__)
693 msghdr msg = msghdr();
694 msg.msg_iov = bufs;
695 msg.msg_iovlen = static_cast<int>(count);
696 signed_size_type result = error_wrapper(::recvmsg(s, &msg, flags), ec);
697 if (result >= 0)
698 ec = boost::system::error_code();
699 return result;
700#endif // defined(BOOST_WINDOWS) || defined(__CYGWIN__)
701}

703size_t sync_recv(socket_type s, state_type state, buf* bufs,
704 size_t count, int flags, bool all_empty, boost::system::error_code& ec)
705{
706 if (s == invalid_socket)
707 {
708 ec = boost::asio::error::bad_descriptor;
709 return 0;
710 }
711
712 // A request to read 0 bytes on a stream is a no-op.
713 if (all_empty && (state & stream_oriented))
714 {
715 ec = boost::system::error_code();
716 return 0;
717 }
718
719 // Read some data.
720 for (;;)
721 {
722 // Try to complete the operation without blocking.
723 signed_size_type bytes = socket_ops::recv(s, bufs, count, flags, ec);
724
725 // Check if operation succeeded.
726 if (bytes > 0)

1342
1343int setsockopt(socket_type s, state_type& state, int level, int optname,
1344 const void* optval, std::size_t optlen, boost::system::error_code& ec)
1345{
1346 if (s == invalid_socket)
1347 {
1348 ec = boost::asio::error::bad_descriptor;
1349 return socket_error_retval;
1350 }
1351
1352 if (level == custom_socket_option_level && optname == always_fail_option)
1353 {
1354 ec = boost::asio::error::invalid_argument;
1355 return socket_error_retval;
1356 }
1357
1358 if (level == custom_socket_option_level
1359 && optname == enable_connection_aborted_option)
1360 {
1361 if (optlen != sizeof(int))
1362 {
1363 ec = boost::asio::error::invalid_argument;
1364 return socket_error_retval;
1365 }
1366
1367 if (*static_cast<const int*>(optval))
1368 state |= enable_connection_aborted;
1369 else
1370 state &= ~enable_connection_aborted;
1371 ec = boost::system::error_code();
1372 return 0;
1373 }
1374
1375 if (level == SOL_SOCKET && optname == SO_LINGER)
1376 state |= user_set_linger;
1377
1378#if defined(__BORLANDC__)
1379 // Mysteriously, using the getsockopt and setsockopt functions directly with
1380 // Borland C++ results in incorrect values being set and read. The bug can be
1381 // worked around by using function addresses resolved with GetProcAddress.
1382 if (HMODULE winsock_module = ::GetModuleHandleA("ws2_32"))
1383 {
1384 typedef int (WSAAPI *sso_t)(SOCKET, int, int, const char*, int);
1385 if (sso_t sso = (sso_t)::GetProcAddress(winsock_module, "setsockopt"))
1386 {
1387 clear_last_error();
1388 return error_wrapper(sso(s, level, optname,
1389 reinterpret_cast<const char*>(optval),
1390 static_cast<int>(optlen)), ec);
1391 }
1392 }
1393 ec = boost::asio::error::fault;
1394 return socket_error_retval;
1395#else // defined(__BORLANDC__)
1396 clear_last_error();
1397 int result = error_wrapper(call_setsockopt(&msghdr::msg_namelen,
1398 s, level, optname, optval, optlen), ec);
1399 if (result == 0)
1400 {
1401 ec = boost::system::error_code();
1402
1403#if defined(__MACH__) && defined(__APPLE__) \
1404 || defined(__NetBSD__) || defined(__FreeBSD__) || defined(__OpenBSD__)
1405 // To implement portable behaviour for SO_REUSEADDR with UDP sockets we
1406 // need to also set SO_REUSEPORT on BSD-based platforms.
1407 if ((state & datagram_oriented)
1408 && level == SOL_SOCKET && optname == SO_REUSEADDR)
1409 {
1410 call_setsockopt(&msghdr::msg_namelen, s,
1411 SOL_SOCKET, SO_REUSEPORT, optval, optlen);
1412 }
1413#endif
1414 }
1415
1416 return result;
1417#endif // defined(__BORLANDC__)
1418}

1609int ioctl(socket_type s, state_type& state, int cmd,
1610 ioctl_arg_type* arg, boost::system::error_code& ec)

1613 {

1616 }
1617
1618 clear_last_error();
1619#if defined(BOOST_WINDOWS) || defined(__CYGWIN__)
1620 int result = error_wrapper(::ioctlsocket(s, cmd, arg), ec);
1621#elif defined(__MACH__) && defined(__APPLE__) \
1622 || defined(__NetBSD__) || defined(__FreeBSD__) || defined(__OpenBSD__)

1624 static_cast<unsigned int>(cmd), arg), ec);
1625#else

1627#endif
1628 if (result >= 0)
1629 {
1630 ec = boost::system::error_code();
1631
1632 // When updating the non-blocking mode we always perform the ioctl syscall,
1633 // even if the flags would otherwise indicate that the socket is already in
1634 // the correct state. This ensures that the underlying socket is put into
1635 // the state that has been requested by the user. If the ioctl syscall was
1636 // successful then we need to update the flags to match.
1637 if (cmd == static_cast<int>(FIONBIO))
1638 {
1639 if (*arg)
1640 {
1641 state |= user_set_non_blocking;
1642 }
1643 else
1644 {
1645 // Clearing the non-blocking mode always overrides any internally-set
1646 // non-blocking flag. Any subsequent asynchronous operations will need
1647 // to re-enable non-blocking I/O.
1648 state &= ~(user_set_non_blocking | internal_non_blocking);
1649 }
1650 }
1651 }
1652
1653 return result;
1654}

1656int select(int nfds, fd_set* readfds, fd_set* writefds,
1657 fd_set* exceptfds, timeval* timeout, boost::system::error_code& ec)
1658{
1659 clear_last_error();
1660#if defined(BOOST_WINDOWS) || defined(__CYGWIN__)
1661 if (!readfds && !writefds && !exceptfds && timeout)
1662 {
1663 DWORD milliseconds = timeout->tv_sec * 1000 + timeout->tv_usec / 1000;
1664 if (milliseconds == 0)
1665 milliseconds = 1; // Force context switch.
1666 ::Sleep(milliseconds);
1667 ec = boost::system::error_code();
1668 return 0;
1669 }
1670
1671 // The select() call allows timeout values measured in microseconds, but the
1672 // system clock (as wrapped by boost::posix_time::microsec_clock) typically
1673 // has a resolution of 10 milliseconds. This can lead to a spinning select
1674 // reactor, meaning increased CPU usage, when waiting for the earliest
1675 // scheduled timeout if it's less than 10 milliseconds away. To avoid a tight
1676 // spin we'll use a minimum timeout of 1 millisecond.
1677 if (timeout && timeout->tv_sec == 0
1678 && timeout->tv_usec > 0 && timeout->tv_usec < 1000)
1679 timeout->tv_usec = 1000;
1680#endif // defined(BOOST_WINDOWS) || defined(__CYGWIN__)
1681
1682#if defined(__hpux) && defined(__SELECT)
1683 timespec ts;
1684 ts.tv_sec = timeout ? timeout->tv_sec : 0;
1685 ts.tv_nsec = timeout ? timeout->tv_usec * 1000 : 0;
1686 return error_wrapper(::pselect(nfds, readfds,
1687 writefds, exceptfds, timeout ? &ts : 0, 0), ec);
1688#else
1689 int result = error_wrapper(::select(nfds, readfds,
1690 writefds, exceptfds, timeout), ec);
1691 if (result >= 0)
1692 ec = boost::system::error_code();
1693 return result;
1694#endif
1695}

2794#else
2795 sprintf(serv, "%u", ntohs(port));
2796#endif
2797 }

2804#endif // defined(BOOST_HAS_THREADS) && defined(BOOST_HAS_PTHREADS)

2808 {

2810 }

2816 }
2817#if BOOST_WORKAROUND(BOOST_MSVC, >= 1400) && !defined(UNDER_CE)
2818 sprintf_s(serv, servlen, "%u", ntohs(port));

2822 }

2826#endif // defined(BOOST_HAS_THREADS) && defined(BOOST_HAS_PTHREADS)
2827 // && !defined(BOOST_ASIO_DISABLE_THREADS)
2828 }
2829 }
2830
2831 ec = boost::system::error_code();

2855#if defined(EAI_ADDRFAMILY)
2856 case EAI_ADDRFAMILY:

2858#if defined(EAI_NODATA) && (EAI_NODATA != EAI_NONAME)
2859 case EAI_NODATA:
2860#endif
2861 return boost::asio::error::host_not_found;
2862 case EAI_SERVICE:
2863 return boost::asio::error::service_not_found;
2864 case EAI_SOCKTYPE:
2865 return boost::asio::error::socket_type_not_supported;
2866 default: // Possibly the non-portable EAI_SYSTEM.
2867#if defined(BOOST_WINDOWS) || defined(__CYGWIN__)
2868 return boost::system::error_code(
2869 WSAGetLastError(), boost::asio::error::get_system_category());
