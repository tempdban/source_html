26/*! \file
27 Defines find iterator classes. Find iterator repeatedly applies a Finder
28 to the specified input string to search for matches. Dereferencing
29 the iterator yields the current match or a range between the last and the current
30 match depending on the iterator used.
31*/
32
33namespace boost {
34 namespace algorithm {
35
36// find_iterator -----------------------------------------------//
37
38 //! find_iterator
39 /*!
40 Find iterator encapsulates a Finder and allows
41 for incremental searching in a string.
42 Each increment moves the iterator to the next match.
43
44 Find iterator is a readable forward traversal iterator.
45
46 Dereferencing the iterator yields an iterator_range delimiting
47 the current match.
48 */
49 template<typename IteratorT>
50 class find_iterator :
51 public iterator_facade<
52 find_iterator<IteratorT>,
53 const iterator_range<IteratorT>,
54 forward_traversal_tag >,
55 private detail::find_iterator_base<IteratorT>
56 {
57 private:
58 // facade support
59 friend class ::boost::iterator_core_access;
60
61 private:
62 // typedefs
63
64 typedef detail::find_iterator_base<IteratorT> base_type;
65 typedef BOOST_STRING_TYPENAME
66 base_type::input_iterator_type input_iterator_type;
67 typedef BOOST_STRING_TYPENAME
68 base_type::match_type match_type;
69
70 public:
71 //! Default constructor
72 /*!
73 Construct null iterator. All null iterators are equal.
74
75 \post eof()==true
76 */
77 find_iterator() {}
78
79 //! Copy constructor
80 /*!
81 Construct a copy of the find_iterator
82 */
83 find_iterator( const find_iterator& Other ) :
84 base_type(Other),
85 m_Match(Other.m_Match),
86 m_End(Other.m_End) {}
87
88 //! Constructor
89 /*!
90 Construct new find_iterator for a given finder
91 and a range.
92 */
93 template<typename FinderT>
94 find_iterator(
95 IteratorT Begin,
96 IteratorT End,
97 FinderT Finder ) :
98 detail::find_iterator_base<IteratorT>(Finder,0),
99 m_Match(Begin,Begin),
100 m_End(End)
101 {
102 increment();
103 }
104
105 //! Constructor
106 /*!
107 Construct new find_iterator for a given finder
108 and a range.
109 */
110 template<typename FinderT, typename RangeT>
111 find_iterator(
112 RangeT& Col,
113 FinderT Finder ) :
114 detail::find_iterator_base<IteratorT>(Finder,0)
115 {
116 iterator_range<BOOST_STRING_TYPENAME range_iterator<RangeT>::type> lit_col(::boost::as_literal(Col));
117 m_Match=::boost::make_iterator_range(::boost::begin(lit_col), ::boost::begin(lit_col));
118 m_End=::boost::end(lit_col);
119
120 increment();
121 }
122
123 private:
124 // iterator operations
125
126 // dereference
127 const match_type& dereference() const
128 {
129 return m_Match;
130 }
131
132 // increment
133 void increment()
134 {
135 if(m_Match.begin() == m_Match.end())
136 m_Match=this->do_find(m_Match.end(),m_End);
137 else
138 m_Match=this->do_find(m_Match.begin()+1,m_End);
139 }
140
141 // comparison
142 bool equal( const find_iterator& Other ) const
143 {
144 bool bEof=eof();
145 bool bOtherEof=Other.eof();
146
147 return bEof || bOtherEof ? bEof==bOtherEof :
148 (
149 m_Match==Other.m_Match &&
150 m_End==Other.m_End
151 );
152 }
153
154 public:
155 // operations
156
157 //! Eof check
158 /*!
159 Check the eof condition. Eof condition means that
160 there is nothing more to be searched i.e. find_iterator
161 is after the last match.
162 */
163 bool eof() const
164 {
165 return
166 this->is_null() ||
167 (
168 m_Match.begin() == m_End &&
169 m_Match.end() == m_End
170 );
171 }
172
173 private:
174 // Attributes
175 match_type m_Match;
176 input_iterator_type m_End;
177 };
178
179 //! find iterator construction helper
180 /*!
181 * Construct a find iterator to iterate through the specified string
182 */
183 template<typename RangeT, typename FinderT>
184 inline find_iterator<
185 BOOST_STRING_TYPENAME range_iterator<RangeT>::type>
186 make_find_iterator(
187 RangeT& Collection,
188 FinderT Finder)
189 {
190 return find_iterator<BOOST_STRING_TYPENAME range_iterator<RangeT>::type>(
191 Collection, Finder);
192 }

194// split iterator -----------------------------------------------//
195
196 //! split_iterator
197 /*!
198 Split iterator encapsulates a Finder and allows
199 for incremental searching in a string.
200 Unlike the find iterator, split iterator iterates
201 through gaps between matches.
202
203 Find iterator is a readable forward traversal iterator.
204
205 Dereferencing the iterator yields an iterator_range delimiting
206 the current match.
207 */
208 template<typename IteratorT>
209 class split_iterator :
210 public iterator_facade<
211 split_iterator<IteratorT>,
212 const iterator_range<IteratorT>,
213 forward_traversal_tag >,
214 private detail::find_iterator_base<IteratorT>
215 {
216 private:
217 // facade support
218 friend class ::boost::iterator_core_access;
219
220 private:
221 // typedefs
222
223 typedef detail::find_iterator_base<IteratorT> base_type;
224 typedef BOOST_STRING_TYPENAME
225 base_type::input_iterator_type input_iterator_type;
226 typedef BOOST_STRING_TYPENAME
227 base_type::match_type match_type;
228
229 public:
230 //! Default constructor
231 /*!
232 Construct null iterator. All null iterators are equal.
233
234 \post eof()==true
235 */
236 split_iterator() {}
237 //! Copy constructor
238 /*!
239 Construct a copy of the split_iterator
240 */
241 split_iterator( const split_iterator& Other ) :
242 base_type(Other),
243 m_Match(Other.m_Match),
244 m_Next(Other.m_Next),
245 m_End(Other.m_End),
246 m_bEof(Other.m_bEof)
247 {}
248
249 //! Constructor
250 /*!
251 Construct new split_iterator for a given finder
252 and a range.
253 */
254 template<typename FinderT>
255 split_iterator(
256 IteratorT Begin,
257 IteratorT End,
258 FinderT Finder ) :
259 detail::find_iterator_base<IteratorT>(Finder,0),
260 m_Match(Begin,Begin),
261 m_Next(Begin),
262 m_End(End),
263 m_bEof(false)
264 {
265 // force the correct behavior for empty sequences and yield at least one token
266 if(Begin!=End)
267 {
268 increment();
269 }
270 }
271 //! Constructor
272 /*!
273 Construct new split_iterator for a given finder
274 and a collection.
275 */
276 template<typename FinderT, typename RangeT>
277 split_iterator(
278 RangeT& Col,
279 FinderT Finder ) :
280 detail::find_iterator_base<IteratorT>(Finder,0),
281 m_bEof(false)
282 {
283 iterator_range<BOOST_STRING_TYPENAME range_iterator<RangeT>::type> lit_col(::boost::as_literal(Col));
284 m_Match=make_iterator_range(::boost::begin(lit_col), ::boost::begin(lit_col));
285 m_Next=::boost::begin(lit_col);
286 m_End=::boost::end(lit_col);
287
288 // force the correct behavior for empty sequences and yield at least one token
289 if(m_Next!=m_End)
290 {
291 increment();
292 }
293 }
294
295
296 private:
297 // iterator operations
298
299 // dereference
300 const match_type& dereference() const
301 {
302 return m_Match;
303 }
304
305 // increment
306 void increment()
307 {
308 match_type FindMatch=this->do_find( m_Next, m_End );
309
310 if(FindMatch.begin()==m_End && FindMatch.end()==m_End)
311 {
312 if(m_Match.end()==m_End)
313 {
314 // Mark iterator as eof
315 m_bEof=true;
316 }
317 }
318
319 m_Match=match_type( m_Next, FindMatch.begin() );
320 m_Next=FindMatch.end();
321 }
322
323 // comparison
324 bool equal( const split_iterator& Other ) const
325 {
326 bool bEof=eof();
327 bool bOtherEof=Other.eof();
328
329 return bEof || bOtherEof ? bEof==bOtherEof :
330 (
331 m_Match==Other.m_Match &&
332 m_Next==Other.m_Next &&
333 m_End==Other.m_End
334 );
335 }
336
337 public:
338 // operations
339
340 //! Eof check
341 /*!
342 Check the eof condition. Eof condition means that
343 there is nothing more to be searched i.e. find_iterator
344 is after the last match.
345 */
346 bool eof() const
347 {
348 return this->is_null() || m_bEof;
349 }
350
351 private:
352 // Attributes
353 match_type m_Match;
354 input_iterator_type m_Next;
355 input_iterator_type m_End;
356 bool m_bEof;
357 };
358
359 //! split iterator construction helper
360 /*!
361 * Construct a split iterator to iterate through the specified collection
362 */
363 template<typename RangeT, typename FinderT>
364 inline split_iterator<
365 BOOST_STRING_TYPENAME range_iterator<RangeT>::type>
366 make_split_iterator(
367 RangeT& Collection,
368 FinderT Finder)
369 {
370 return split_iterator<BOOST_STRING_TYPENAME range_iterator<RangeT>::type>(
371 Collection, Finder);
372 }
