94# define DSA_FLAG_CACHE_MONT_P 0x01
95/*
96 * new with 0.9.7h; the built-in DSA implementation now uses constant time
97 * modular exponentiation for secret exponents by default. This flag causes
98 * the faster variable sliding window method to be used for all exponents.
99 */
100# define DSA_FLAG_NO_EXP_CONSTTIME 0x02
101
102/*
103 * If this flag is set the DSA method is FIPS compliant and can be used in
104 * FIPS mode. This is set in the validated module method. If an application
105 * sets this flag in its own methods it is its reposibility to ensure the
106 * result is compliant.
107 */
108
109# define DSA_FLAG_FIPS_METHOD 0x0400
110
111/*
112 * If this flag is set the operations normally disabled in FIPS mode are
113 * permitted it is then the applications responsibility to ensure that the
114 * usage is compliant.
115 */
116
117# define DSA_FLAG_NON_FIPS_ALLOW 0x0400

127typedef struct DSA_SIG_st {
128 BIGNUM *r;
129 BIGNUM *s;

132struct dsa_method {
133 const char *name;
134 DSA_SIG *(*dsa_do_sign) (const unsigned char *dgst, int dlen, DSA *dsa);
135 int (*dsa_sign_setup) (DSA *dsa, BN_CTX *ctx_in, BIGNUM **kinvp,
136 BIGNUM **rp);
137 int (*dsa_do_verify) (const unsigned char *dgst, int dgst_len,
138 DSA_SIG *sig, DSA *dsa);
139 int (*dsa_mod_exp) (DSA *dsa, BIGNUM *rr, BIGNUM *a1, BIGNUM *p1,
140 BIGNUM *a2, BIGNUM *p2, BIGNUM *m, BN_CTX *ctx,
141 BN_MONT_CTX *in_mont);
142 /* Can be null */
143 int (*bn_mod_exp) (DSA *dsa, BIGNUM *r, BIGNUM *a, const BIGNUM *p,
144 const BIGNUM *m, BN_CTX *ctx, BN_MONT_CTX *m_ctx);
145 int (*init) (DSA *dsa);
146 int (*finish) (DSA *dsa);
147 int flags;
148 char *app_data;
149 /* If this is non-NULL, it is used to generate DSA parameters */
150 int (*dsa_paramgen) (DSA *dsa, int bits,
151 const unsigned char *seed, int seed_len,
152 int *counter_ret, unsigned long *h_ret,
153 BN_GENCB *cb);
154 /* If this is non-NULL, it is used to generate DSA keys */
155 int (*dsa_keygen) (DSA *dsa);
156};
157
158struct dsa_st {
159 /*
160 * This first variable is used to pick up errors where a DSA is passed
161 * instead of of a EVP_PKEY
162 */
163 int pad;
164 long version;
165 int write_params;
166 BIGNUM *p;
167 BIGNUM *q; /* == 20 */
168 BIGNUM *g;
169 BIGNUM *pub_key; /* y public key */
170 BIGNUM *priv_key; /* x private key */
171 BIGNUM *kinv; /* Signing pre-calc */
172 BIGNUM *r; /* Signing pre-calc */
173 int flags;
174 /* Normally used to cache montgomery values */
175 BN_MONT_CTX *method_mont_p;
176 int references;
177 CRYPTO_EX_DATA ex_data;
178 const DSA_METHOD *meth;
179 /* functional reference if 'meth' is ENGINE-provided */
180 ENGINE *engine;
181};
