85/*
86 * new with 0.9.7h; the built-in DH
87 * implementation now uses constant time
88 * modular exponentiation for secret exponents
89 * by default. This flag causes the
90 * faster variable sliding window method to
91 * be used for all exponents.
92 */
93# define DH_FLAG_NO_EXP_CONSTTIME 0x02
94
95/*
96 * If this flag is set the DH method is FIPS compliant and can be used in
97 * FIPS mode. This is set in the validated module method. If an application
98 * sets this flag in its own methods it is its reposibility to ensure the
99 * result is compliant.
100 */
101
102# define DH_FLAG_FIPS_METHOD 0x0400
103
104/*
105 * If this flag is set the operations normally disabled in FIPS mode are
106 * permitted it is then the applications responsibility to ensure that the
107 * usage is compliant.
108 */
109
110# define DH_FLAG_NON_FIPS_ALLOW 0x0400

120struct dh_method {
121 const char *name;
122 /* Methods here */
123 int (*generate_key) (DH *dh);
124 int (*compute_key) (unsigned char *key, const BIGNUM *pub_key, DH *dh);
125 /* Can be null */
126 int (*bn_mod_exp) (const DH *dh, BIGNUM *r, const BIGNUM *a,
127 const BIGNUM *p, const BIGNUM *m, BN_CTX *ctx,
128 BN_MONT_CTX *m_ctx);
129 int (*init) (DH *dh);
130 int (*finish) (DH *dh);
131 int flags;
132 char *app_data;
133 /* If this is non-NULL, it will be used to generate parameters */
134 int (*generate_params) (DH *dh, int prime_len, int generator,
135 BN_GENCB *cb);
136};
137
138struct dh_st {
139 /*
140 * This first argument is used to pick up errors when a DH is passed
141 * instead of a EVP_PKEY
142 */
143 int pad;
144 int version;
145 BIGNUM *p;
146 BIGNUM *g;
147 long length; /* optional */
148 BIGNUM *pub_key; /* g^x % p */
149 BIGNUM *priv_key; /* x */
150 int flags;
151 BN_MONT_CTX *method_mont_p;
152 /* Place holders if we want to do X9.42 DH */
153 BIGNUM *q;
154 BIGNUM *j;
155 unsigned char *seed;
156 int seedlen;
157 BIGNUM *counter;
158 int references;
159 CRYPTO_EX_DATA ex_data;
160 const DH_METHOD *meth;
161 ENGINE *engine;
162};
