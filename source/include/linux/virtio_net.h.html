34/* The feature bitmap for virtio net */
35#define VIRTIO_NET_F_CSUM 0 /* Host handles pkts w/ partial csum */
36#define VIRTIO_NET_F_GUEST_CSUM 1 /* Guest handles pkts w/ partial csum */
37#define VIRTIO_NET_F_MTU 3 /* Initial MTU advice */
38#define VIRTIO_NET_F_MAC 5 /* Host has given MAC address. */
39#define VIRTIO_NET_F_GUEST_TSO4 7 /* Guest can handle TSOv4 in. */
40#define VIRTIO_NET_F_GUEST_TSO6 8 /* Guest can handle TSOv6 in. */
41#define VIRTIO_NET_F_GUEST_ECN 9 /* Guest can handle TSO[6] w/ ECN in. */
42#define VIRTIO_NET_F_GUEST_UFO 10 /* Guest can handle UFO in. */
43#define VIRTIO_NET_F_HOST_TSO4 11 /* Host can handle TSOv4 in. */
44#define VIRTIO_NET_F_HOST_TSO6 12 /* Host can handle TSOv6 in. */
45#define VIRTIO_NET_F_HOST_ECN 13 /* Host can handle TSO[6] w/ ECN in. */
46#define VIRTIO_NET_F_HOST_UFO 14 /* Host can handle UFO in. */
47#define VIRTIO_NET_F_MRG_RXBUF 15 /* Host can merge receive buffers. */
48#define VIRTIO_NET_F_STATUS 16 /* virtio_net_config.status available */
49#define VIRTIO_NET_F_CTRL_VQ 17 /* Control channel available */
50#define VIRTIO_NET_F_CTRL_RX 18 /* Control channel RX mode support */
51#define VIRTIO_NET_F_CTRL_VLAN 19 /* Control channel VLAN filtering */
52#define VIRTIO_NET_F_CTRL_RX_EXTRA 20 /* Extra RX mode control support */
53#define VIRTIO_NET_F_GUEST_ANNOUNCE 21 /* Guest can announce device on the
54 * network */
55#define VIRTIO_NET_F_MQ 22 /* Device supports Receive Flow
56 * Steering */
57#define VIRTIO_NET_F_CTRL_MAC_ADDR 23 /* Set MAC address */
58
59#ifndef VIRTIO_NET_NO_LEGACY
60#define VIRTIO_NET_F_GSO 6 /* Host handles pkts w/ any GSO type */
61#endif /* VIRTIO_NET_NO_LEGACY */

63#define VIRTIO_NET_S_LINK_UP 1 /* Link is up */
64#define VIRTIO_NET_S_ANNOUNCE 2 /* Announcement is needed */

66struct virtio_net_config {
67 /* The config defining mac address (if VIRTIO_NET_F_MAC) */
68 __u8 mac[6];
69 /* See VIRTIO_NET_F_STATUS and VIRTIO_NET_S_* above */
70 __u16 status;
71 /* Maximum number of each of transmit and receive queues;
72 * see VIRTIO_NET_F_MQ and VIRTIO_NET_CTRL_MQ.
73 * Legal values are between 1 and 0x8000
74 */
75 __u16 max_virtqueue_pairs;
76 /* Default maximum transmit unit advice */
77 __u16 mtu;
78} __attribute__((packed));

80/*
81 * This header comes first in the scatter-gather list. If you don't
82 * specify GSO or CSUM features, you can simply ignore the header.
83 *
84 * This is bitwise-equivalent to the legacy struct virtio_net_hdr_mrg_rxbuf,
85 * only flattened.
86 */
87struct virtio_net_hdr_v1 {
88#define VIRTIO_NET_HDR_F_NEEDS_CSUM 1 /* Use csum_start, csum_offset */
89#define VIRTIO_NET_HDR_F_DATA_VALID 2 /* Csum is valid */
90 __u8 flags;
91#define VIRTIO_NET_HDR_GSO_NONE 0 /* Not a GSO frame */
92#define VIRTIO_NET_HDR_GSO_TCPV4 1 /* GSO frame, IPv4 TCP (TSO) */
93#define VIRTIO_NET_HDR_GSO_UDP 3 /* GSO frame, IPv4 UDP (UFO) */
94#define VIRTIO_NET_HDR_GSO_TCPV6 4 /* GSO frame, IPv6 TCP */
95#define VIRTIO_NET_HDR_GSO_ECN 0x80 /* TCP has ECN set */
96 __u8 gso_type;
97 __virtio16 hdr_len; /* Ethernet + IP + tcp/udp hdrs */
98 __virtio16 gso_size; /* Bytes to append to hdr_len per frame */
99 __virtio16 csum_start; /* Position to start checksumming from */
100 __virtio16 csum_offset; /* Offset after that to place checksum */
101 __virtio16 num_buffers; /* Number of merged rx buffers */
102};

104#ifndef VIRTIO_NET_NO_LEGACY
105/* This header comes first in the scatter-gather list.
106 * For legacy virtio, if VIRTIO_F_ANY_LAYOUT is not negotiated, it must
107 * be the first element of the scatter-gather list. If you don't
108 * specify GSO or CSUM features, you can simply ignore the header. */
109struct virtio_net_hdr {
110 /* See VIRTIO_NET_HDR_F_* */
111 __u8 flags;
112 /* See VIRTIO_NET_HDR_GSO_* */
113 __u8 gso_type;
114 __virtio16 hdr_len; /* Ethernet + IP + tcp/udp hdrs */
115 __virtio16 gso_size; /* Bytes to append to hdr_len per frame */
116 __virtio16 csum_start; /* Position to start checksumming from */
117 __virtio16 csum_offset; /* Offset after that to place checksum */
118};

128/*
129 * Control virtqueue data structures
130 *
131 * The control virtqueue expects a header in the first sg entry
132 * and an ack/status response in the last entry. Data for the
133 * command goes in between.
134 */
135struct virtio_net_ctrl_hdr {
136 __u8 class;
137 __u8 cmd;
138} __attribute__((packed));
139
140typedef __u8 virtio_net_ctrl_ack;

142#define VIRTIO_NET_OK 0
143#define VIRTIO_NET_ERR 1

145/*
146 * Control the RX mode, ie. promisucous, allmulti, etc...
147 * All commands require an "out" sg entry containing a 1 byte
148 * state value, zero = disable, non-zero = enable. Commands
149 * 0 and 1 are supported with the VIRTIO_NET_F_CTRL_RX feature.
150 * Commands 2-5 are added with VIRTIO_NET_F_CTRL_RX_EXTRA.
151 */
152#define VIRTIO_NET_CTRL_RX 0
153 #define VIRTIO_NET_CTRL_RX_PROMISC 0
154 #define VIRTIO_NET_CTRL_RX_ALLMULTI 1
155 #define VIRTIO_NET_CTRL_RX_ALLUNI 2
156 #define VIRTIO_NET_CTRL_RX_NOMULTI 3
157 #define VIRTIO_NET_CTRL_RX_NOUNI 4
158 #define VIRTIO_NET_CTRL_RX_NOBCAST 5
159
160/*
161 * Control the MAC
162 *
163 * The MAC filter table is managed by the hypervisor, the guest should
164 * assume the size is infinite. Filtering should be considered
165 * non-perfect, ie. based on hypervisor resources, the guest may
166 * received packets from sources not specified in the filter list.
167 *
168 * In addition to the class/cmd header, the TABLE_SET command requires
169 * two out scatterlists. Each contains a 4 byte count of entries followed
170 * by a concatenated byte stream of the ETH_ALEN MAC addresses. The
171 * first sg list contains unicast addresses, the second is for multicast.
172 * This functionality is present if the VIRTIO_NET_F_CTRL_RX feature
173 * is available.
174 *
175 * The ADDR_SET command requests one out scatterlist, it contains a
176 * 6 bytes MAC address. This functionality is present if the
177 * VIRTIO_NET_F_CTRL_MAC_ADDR feature is available.
178 */
179struct virtio_net_ctrl_mac {
180 __virtio32 entries;
181 __u8 macs[][ETH_ALEN];
182} __attribute__((packed));
183
184#define VIRTIO_NET_CTRL_MAC 1
185 #define VIRTIO_NET_CTRL_MAC_TABLE_SET 0
186 #define VIRTIO_NET_CTRL_MAC_ADDR_SET 1
187
188/*
189 * Control VLAN filtering
190 *
191 * The VLAN filter table is controlled via a simple ADD/DEL interface.
192 * VLAN IDs not added may be filterd by the hypervisor. Del is the
193 * opposite of add. Both commands expect an out entry containing a 2
194 * byte VLAN ID. VLAN filterting is available with the
195 * VIRTIO_NET_F_CTRL_VLAN feature bit.
196 */
197#define VIRTIO_NET_CTRL_VLAN 2
198 #define VIRTIO_NET_CTRL_VLAN_ADD 0
199 #define VIRTIO_NET_CTRL_VLAN_DEL 1
200
201/*
202 * Control link announce acknowledgement
203 *
204 * The command VIRTIO_NET_CTRL_ANNOUNCE_ACK is used to indicate that
205 * driver has recevied the notification; device would clear the
206 * VIRTIO_NET_S_ANNOUNCE bit in the status field after it receives
207 * this command.
208 */
209#define VIRTIO_NET_CTRL_ANNOUNCE 3
210 #define VIRTIO_NET_CTRL_ANNOUNCE_ACK 0
211
212/*
213 * Control Receive Flow Steering
214 *
215 * The command VIRTIO_NET_CTRL_MQ_VQ_PAIRS_SET
216 * enables Receive Flow Steering, specifying the number of the transmit and
217 * receive queues that will be used. After the command is consumed and acked by
218 * the device, the device will not steer new packets on receive virtqueues
219 * other than specified nor read from transmit virtqueues other than specified.
220 * Accordingly, driver should not transmit new packets on virtqueues other than
221 * specified.
222 */
223struct virtio_net_ctrl_mq {
224 __virtio16 virtqueue_pairs;
225};
226
227#define VIRTIO_NET_CTRL_MQ 4
228 #define VIRTIO_NET_CTRL_MQ_VQ_PAIRS_SET 0
229 #define VIRTIO_NET_CTRL_MQ_VQ_PAIRS_MIN 1
230 #define VIRTIO_NET_CTRL_MQ_VQ_PAIRS_MAX 0x8000
