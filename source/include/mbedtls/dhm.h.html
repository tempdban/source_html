1/**
2 * \file dhm.h
3 *
4 * \brief Diffie-Hellman-Merkle key exchange.
5 *
6 * <em>RFC-3526: More Modular Exponential (MODP) Diffie-Hellman groups for
7 * Internet Key Exchange (IKE)</em> defines a number of standardized
8 * Diffie-Hellman groups for IKE.
9 *
10 * <em>RFC-5114: Additional Diffie-Hellman Groups for Use with IETF
11 * Standards</em> defines a number of standardized Diffie-Hellman
12 * groups that can be used.
13 *
14 * \warning The security of the DHM key exchange relies on the proper choice
15 * of prime modulus - optimally, it should be a safe prime. The usage
16 * of non-safe primes both decreases the difficulty of the underlying
17 * discrete logarithm problem and can lead to small subgroup attacks
18 * leaking private exponent bits when invalid public keys are used
19 * and not detected. This is especially relevant if the same DHM
20 * parameters are reused for multiple key exchanges as in static DHM,
21 * while the criticality of small-subgroup attacks is lower for
22 * ephemeral DHM.
23 *
24 * \warning For performance reasons, the code does neither perform primality
25 * nor safe primality tests, nor the expensive checks for invalid
26 * subgroups. Moreover, even if these were performed, non-standardized
27 * primes cannot be trusted because of the possibility of backdoors
28 * that can't be effectively checked for.
29 *
30 * \warning Diffie-Hellman-Merkle is therefore a security risk when not using
31 * standardized primes generated using a trustworthy ("nothing up
32 * my sleeve") method, such as the RFC 3526 / 7919 primes. In the TLS
33 * protocol, DH parameters need to be negotiated, so using the default
34 * primes systematically is not always an option. If possible, use
35 * Elliptic Curve Diffie-Hellman (ECDH), which has better performance,
36 * and for which the TLS protocol mandates the use of standard
37 * parameters.
38 *
39 */

89/**
90 * \brief The DHM context structure.
91 */
92typedef struct
93{
94 size_t len; /*!< The size of \p P in Bytes. */
95 mbedtls_mpi P; /*!< The prime modulus. */
96 mbedtls_mpi G; /*!< The generator. */
97 mbedtls_mpi X; /*!< Our secret value. */
98 mbedtls_mpi GX; /*!< Our public key = \c G^X mod \c P. */
99 mbedtls_mpi GY; /*!< The public key of the peer = \c G^Y mod \c P. */
100 mbedtls_mpi K; /*!< The shared secret = \c G^(XY) mod \c P. */
101 mbedtls_mpi RP; /*!< The cached value = \c R^2 mod \c P. */
102 mbedtls_mpi Vi; /*!< The blinding value. */
103 mbedtls_mpi Vf; /*!< The unblinding value. */
104 mbedtls_mpi pX; /*!< The previous \c X. */
105}
106mbedtls_dhm_context;

135/**
136 * \brief This function sets up and writes the ServerKeyExchange
137 * parameters.
138 *
139 * \param ctx The DHM context.
140 * \param x_size The private value size in Bytes.
141 * \param olen The number of characters written.
142 * \param output The destination buffer.
143 * \param f_rng The RNG function.
144 * \param p_rng The RNG parameter.
145 *
146 * \note The destination buffer must be large enough to hold
147 * the reduced binary presentation of the modulus, the generator
148 * and the public key, each wrapped with a 2-byte length field.

150 * space is available. Refer to \c mbedtls_mpi_size to computing
151 * the byte-size of an MPI.
152 *
153 * \note This function assumes that \c ctx->P and \c ctx->G
154 * have already been properly set. For that, use
155 * mbedtls_dhm_set_group() below in conjunction with
156 * mbedtls_mpi_read_binary() and mbedtls_mpi_read_string().
157 *
158 * \return \c 0 on success, or an \c MBEDTLS_ERR_DHM_XXX error code
159 * on failure.
160 */
161int mbedtls_dhm_make_params( mbedtls_dhm_context *ctx, int x_size,
162 unsigned char *output, size_t *olen,
163 int (*f_rng)(void *, unsigned char *, size_t),
164 void *p_rng );

221/**
222 * \brief This function derives and exports the shared secret
223 * \c (G^Y)^X mod \c P.
224 *
225 * \param ctx The DHM context.
226 * \param output The destination buffer.
227 * \param output_size The size of the destination buffer. Must be at least
228 * the size of ctx->len.
229 * \param olen On exit, holds the actual number of Bytes written.
230 * \param f_rng The RNG function, for blinding purposes.
231 * \param p_rng The RNG parameter.
232 *
233 * \return \c 0 on success, or an \c MBEDTLS_ERR_DHM_XXX error code

308/**
309 * RFC 3526, RFC 5114 and RFC 7919 standardize a number of
310 * Diffie-Hellman groups, some of which are included here
311 * for use within the SSL/TLS module and the user's convenience
312 * when configuring the Diffie-Hellman parameters by hand
313 * through \c mbedtls_ssl_conf_dh_param.
314 *
315 * The following lists the source of the above groups in the standards:
316 * - RFC 5114 section 2.2: 2048-bit MODP Group with 224-bit Prime Order Subgroup
317 * - RFC 3526 section 3: 2048-bit MODP Group
318 * - RFC 3526 section 4: 3072-bit MODP Group
319 * - RFC 3526 section 5: 4096-bit MODP Group
320 * - RFC 7919 section A.1: ffdhe2048
321 * - RFC 7919 section A.2: ffdhe3072
322 * - RFC 7919 section A.3: ffdhe4096
323 * - RFC 7919 section A.4: ffdhe6144
324 * - RFC 7919 section A.5: ffdhe8192
325 *
326 * The constants with suffix "_p" denote the chosen prime moduli, while
327 * the constants with suffix "_g" denote the chosen generator
328 * of the associated prime field.
329 *
330 * The constants further suffixed with "_bin" are provided in binary format,
331 * while all other constants represent null-terminated strings holding the
332 * hexadecimal presentation of the respective numbers.
333 *
334 * The primes from RFC 3526 and RFC 7919 have been generating by the following
335 * trust-worthy procedure:
336 * - Fix N in { 2048, 3072, 4096, 6144, 8192 } and consider the N-bit number
337 * the first and last 64 bits are all 1, and the remaining N - 128 bits of
338 * which are 0x7ff...ff.
339 * - Add the smallest multiple of the first N - 129 bits of the binary expansion
340 * of pi (for RFC 5236) or e (for RFC 7919) to this intermediate bit-string
341 * such that the resulting integer is a safe-prime.
342 * - The result is the respective RFC 3526 / 7919 prime, and the corresponding
343 * generator is always chosen to be 2 (which is a square for these prime,
344 * hence the corresponding subgroup has order (p-1)/2 and avoids leaking a
345 * bit in the private exponent).
346 *
347 */
