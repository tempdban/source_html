46/**
47 * \brief The SHA-512 context structure.
48 *
49 * The structure is used both for SHA-384 and for SHA-512
50 * checksum calculations. The choice between these two is
51 * made in the call to mbedtls_sha512_starts_ret().
52 */
53typedef struct
54{
55 uint64_t total[2]; /*!< The number of Bytes processed. */
56 uint64_t state[8]; /*!< The intermediate digest state. */
57 unsigned char buffer[128]; /*!< The data block being processed. */
58 int is384; /*!< Determines which function to use.
59 * <ul><li>0: Use SHA-512.</li>
60 * <li>1: Use SHA-384.</li></ul> */
61}
62mbedtls_sha512_context;
63
64/**
65 * \brief This function initializes a SHA-512 context.
66 *
67 * \param ctx The SHA-512 context to initialize.
68 */
69void mbedtls_sha512_init( mbedtls_sha512_context *ctx );
70
71/**
72 * \brief This function clears a SHA-512 context.
73 *
74 * \param ctx The SHA-512 context to clear.
75 */
76void mbedtls_sha512_free( mbedtls_sha512_context *ctx );
77
78/**
79 * \brief This function clones the state of a SHA-512 context.
80 *
81 * \param dst The destination context.
82 * \param src The context to clone.
83 */
84void mbedtls_sha512_clone( mbedtls_sha512_context *dst,
85 const mbedtls_sha512_context *src );
86
87/**
88 * \brief This function starts a SHA-384 or SHA-512 checksum
89 * calculation.
90 *
91 * \param ctx The SHA-512 context to initialize.
92 * \param is384 Determines which function to use.
93 * <ul><li>0: Use SHA-512.</li>
94 * <li>1: Use SHA-384.</li></ul>
95 *
96 * \return \c 0 on success.
97 */
98int mbedtls_sha512_starts_ret( mbedtls_sha512_context *ctx, int is384 );
99
100/**
101 * \brief This function feeds an input buffer into an ongoing
102 * SHA-512 checksum calculation.
103 *
104 * \param ctx The SHA-512 context.
105 * \param input The buffer holding the input data.
106 * \param ilen The length of the input data.
107 *
108 * \return \c 0 on success.
109 */
110int mbedtls_sha512_update_ret( mbedtls_sha512_context *ctx,
111 const unsigned char *input,
112 size_t ilen );
113
114/**
115 * \brief This function finishes the SHA-512 operation, and writes
116 * the result to the output buffer. This function is for
117 * internal use only.
118 *
119 * \param ctx The SHA-512 context.
120 * \param output The SHA-384 or SHA-512 checksum result.
121 *
122 * \return \c 0 on success.
123 */
124int mbedtls_sha512_finish_ret( mbedtls_sha512_context *ctx,
125 unsigned char output[64] );
126
127/**
128 * \brief This function processes a single data block within
129 * the ongoing SHA-512 computation.
130 *
131 * \param ctx The SHA-512 context.
132 * \param data The buffer holding one block of data.
133 *
134 * \return \c 0 on success.
135 */
136int mbedtls_internal_sha512_process( mbedtls_sha512_context *ctx,
137 const unsigned char data[128] );

213/**
214 * \brief This function calculates the SHA-512 or SHA-384
215 * checksum of a buffer.
216 *
217 * The function allocates the context, performs the
218 * calculation, and frees the context.
219 *
220 * The SHA-512 result is calculated as
221 * output = SHA-512(input buffer).
222 *
223 * \param input The buffer holding the input data.
224 * \param ilen The length of the input data.
225 * \param output The SHA-384 or SHA-512 checksum result.
226 * \param is384 Determines which function to use.
227 * <ul><li>0: Use SHA-512.</li>
228 * <li>1: Use SHA-384.</li></ul>
229 *
230 * \return \c 0 on success.
231 */
232int mbedtls_sha512_ret( const unsigned char *input,
233 size_t ilen,
234 unsigned char output[64],
235 int is384 );
