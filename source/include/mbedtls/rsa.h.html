87/**
88 * \brief The RSA context structure.
89 *
90 * \note Direct manipulation of the members of this structure
91 * is deprecated. All manipulation should instead be done through
92 * the public interface functions.
93 */
94typedef struct
95{
96 int ver; /*!< Always 0.*/
97 size_t len; /*!< The size of \p N in Bytes. */
98
99 mbedtls_mpi N; /*!< The public modulus. */
100 mbedtls_mpi E; /*!< The public exponent. */
101
102 mbedtls_mpi D; /*!< The private exponent. */
103 mbedtls_mpi P; /*!< The first prime factor. */
104 mbedtls_mpi Q; /*!< The second prime factor. */
105
106 mbedtls_mpi DP; /*!< \p D % (P - 1) */
107 mbedtls_mpi DQ; /*!< \p D % (Q - 1) */
108 mbedtls_mpi QP; /*!< 1 / (Q % P) */
109
110 mbedtls_mpi RN; /*!< cached R^2 mod \p N */
111
112 mbedtls_mpi RP; /*!< cached R^2 mod \p P */
113 mbedtls_mpi RQ; /*!< cached R^2 mod \p Q */
114
115 mbedtls_mpi Vi; /*!< The cached blinding value. */
116 mbedtls_mpi Vf; /*!< The cached un-blinding value. */
117
118 int padding; /*!< Selects padding mode:
119 #MBEDTLS_RSA_PKCS_V15 for 1.5 padding and
120 #MBEDTLS_RSA_PKCS_V21 for OAEP or PSS. */
121 int hash_id; /*!< Hash identifier of mbedtls_md_type_t type,
122 as specified in md.h for use in the MGF
123 mask generating function used in the
124 EME-OAEP and EMSA-PSS encodings. */
125#if defined(MBEDTLS_THREADING_C)
126 mbedtls_threading_mutex_t mutex; /*!< Thread-safety mutex. */
127#endif
128}
129mbedtls_rsa_context;

131/**
132 * \brief This function initializes an RSA context.
133 *
134 * \note Set padding to #MBEDTLS_RSA_PKCS_V21 for the RSAES-OAEP
135 * encryption scheme and the RSASSA-PSS signature scheme.
136 *
137 * \param ctx The RSA context to initialize.
138 * \param padding Selects padding mode: #MBEDTLS_RSA_PKCS_V15 or
139 * #MBEDTLS_RSA_PKCS_V21.
140 * \param hash_id The hash identifier of #mbedtls_md_type_t type, if
141 * \p padding is #MBEDTLS_RSA_PKCS_V21.
142 *
143 * \note The \p hash_id parameter is ignored when using
144 * #MBEDTLS_RSA_PKCS_V15 padding.
145 *
146 * \note The choice of padding mode is strictly enforced for private key
147 * operations, since there might be security concerns in
148 * mixing padding modes. For public key operations it is
149 * a default value, which can be overriden by calling specific
150 * \c rsa_rsaes_xxx or \c rsa_rsassa_xxx functions.
151 *
152 * \note The hash selected in \p hash_id is always used for OEAP
153 * encryption. For PSS signatures, it is always used for
154 * making signatures, but can be overriden for verifying them.
155 * If set to #MBEDTLS_MD_NONE, it is always overriden.
156 */
157void mbedtls_rsa_init( mbedtls_rsa_context *ctx,
158 int padding,
159 int hash_id);

161/**
162 * \brief This function imports a set of core parameters into an
163 * RSA context.
164 *
165 * \param ctx The initialized RSA context to store the parameters in.
166 * \param N The RSA modulus, or NULL.
167 * \param P The first prime factor of \p N, or NULL.
168 * \param Q The second prime factor of \p N, or NULL.
169 * \param D The private exponent, or NULL.
170 * \param E The public exponent, or NULL.
171 *
172 * \note This function can be called multiple times for successive
173 * imports, if the parameters are not simultaneously present.
174 *
175 * Any sequence of calls to this function should be followed
176 * by a call to mbedtls_rsa_complete(), which checks and
177 * completes the provided information to a ready-for-use
178 * public or private RSA key.
179 *
180 * \note See mbedtls_rsa_complete() for more information on which
181 * parameters are necessary to set up a private or public
182 * RSA key.
183 *
184 * \note The imported parameters are copied and need not be preserved
185 * for the lifetime of the RSA context being set up.
186 *
187 * \return \c 0 on success, or a non-zero error code on failure.

192 const mbedtls_mpi *D, const mbedtls_mpi *E );

198 * \param ctx The initialized RSA context to store the parameters in.
199 * \param N The RSA modulus, or NULL.
200 * \param N_len The Byte length of \p N, ignored if \p N == NULL.
201 * \param P The first prime factor of \p N, or NULL.
202 * \param P_len The Byte length of \p P, ignored if \p P == NULL.
203 * \param Q The second prime factor of \p N, or NULL.
204 * \param Q_len The Byte length of \p Q, ignored if \p Q == NULL.
205 * \param D The private exponent, or NULL.
206 * \param D_len The Byte length of \p D, ignored if \p D == NULL.
207 * \param E The public exponent, or NULL.
208 * \param E_len The Byte length of \p E, ignored if \p E == NULL.
209 *
210 * \note This function can be called multiple times for successive
211 * imports, if the parameters are not simultaneously present.
212 *
213 * Any sequence of calls to this function should be followed
214 * by a call to mbedtls_rsa_complete(), which checks and
215 * completes the provided information to a ready-for-use
216 * public or private RSA key.
217 *
218 * \note See mbedtls_rsa_complete() for more information on which
219 * parameters are necessary to set up a private or public
220 * RSA key.
221 *
222 * \note The imported parameters are copied and need not be preserved
223 * for the lifetime of the RSA context being set up.
224 *
225 * \return \c 0 on success, or a non-zero error code on failure.
226 */
227int mbedtls_rsa_import_raw( mbedtls_rsa_context *ctx,
228 unsigned char const *N, size_t N_len,

230 unsigned char const *Q, size_t Q_len,
231 unsigned char const *D, size_t D_len,
232 unsigned char const *E, size_t E_len );

234/**
235 * \brief This function completes an RSA context from
236 * a set of imported core parameters.
237 *
238 * To setup an RSA public key, precisely \p N and \p E
239 * must have been imported.
240 *
241 * To setup an RSA private key, sufficient information must
242 * be present for the other parameters to be derivable.
243 *
244 * The default implementation supports the following:
245 * <ul><li>Derive \p P, \p Q from \p N, \p D, \p E.</li>
246 * <li>Derive \p N, \p D from \p P, \p Q, \p E.</li></ul>
247 * Alternative implementations need not support these.
248 *
249 * If this function runs successfully, it guarantees that
250 * the RSA context can be used for RSA operations without
251 * the risk of failure or crash.
252 *
253 * \param ctx The initialized RSA context holding imported parameters.
254 *
255 * \return \c 0 on success, or #MBEDTLS_ERR_RSA_BAD_INPUT_DATA if the
256 * attempted derivations failed.
257 *
258 * \warning This function need not perform consistency checks
259 * for the imported parameters. In particular, parameters that
260 * are not needed by the implementation might be silently
261 * discarded and left unchecked. To check the consistency
262 * of the key material, see mbedtls_rsa_check_privkey().
263 *
264 */
265int mbedtls_rsa_complete( mbedtls_rsa_context *ctx );
266
267/**
268 * \brief This function exports the core parameters of an RSA key.
269 *
270 * If this function runs successfully, the non-NULL buffers
271 * pointed to by \p N, \p P, \p Q, \p D, and \p E are fully
272 * written, with additional unused space filled leading by
273 * zero Bytes.
274 *
275 * Possible reasons for returning
276 * #MBEDTLS_ERR_RSA_UNSUPPORTED_OPERATION:<ul>
277 * <li>An alternative RSA implementation is in use, which
278 * stores the key externally, and either cannot or should
279 * not export it into RAM.</li>
280 * <li>A SW or HW implementation might not support a certain
281 * deduction. For example, \p P, \p Q from \p N, \p D,
282 * and \p E if the former are not part of the
283 * implementation.</li></ul>
284 *
285 * If the function fails due to an unsupported operation,
286 * the RSA context stays intact and remains usable.
287 *
288 * \param ctx The initialized RSA context.
289 * \param N The MPI to hold the RSA modulus, or NULL.
290 * \param P The MPI to hold the first prime factor of \p N, or NULL.
291 * \param Q The MPI to hold the second prime factor of \p N, or NULL.
292 * \param D The MPI to hold the private exponent, or NULL.
293 * \param E The MPI to hold the public exponent, or NULL.
294 *
295 * \return \c 0 on success,
296 * #MBEDTLS_ERR_RSA_UNSUPPORTED_OPERATION if exporting the
297 * requested parameters cannot be done due to missing
298 * functionality or because of security policies,
299 * or a non-zero return code on any other failure.
300 *
301 */
302int mbedtls_rsa_export( const mbedtls_rsa_context *ctx,
303 mbedtls_mpi *N, mbedtls_mpi *P, mbedtls_mpi *Q,
304 mbedtls_mpi *D, mbedtls_mpi *E );
305

433/**
434 * \brief This function checks if a context contains an RSA private key
435 * and perform basic consistency checks.
436 *
437 * \param ctx The RSA context to check.
438 *
439 * \return \c 0 on success, or an \c MBEDTLS_ERR_RSA_XXX error code on
440 * failure.
441 *
442 * \note The consistency checks performed by this function not only
443 * ensure that mbedtls_rsa_private() can be called successfully
444 * on the given context, but that the various parameters are
445 * mutually consistent with high probability, in the sense that
446 * mbedtls_rsa_public() and mbedtls_rsa_private() are inverses.
447 *
448 * \warning This function should catch accidental misconfigurations
449 * like swapping of parameters, but it cannot establish full
450 * trust in neither the quality nor the consistency of the key
451 * material that was used to setup the given RSA context:
452 * <ul><li>Consistency: Imported parameters that are irrelevant
453 * for the implementation might be silently dropped. If dropped,
454 * the current function does not have access to them,
455 * and therefore cannot check them. See mbedtls_rsa_complete().
456 * If you want to check the consistency of the entire
457 * content of an PKCS1-encoded RSA private key, for example, you
458 * should use mbedtls_rsa_validate_params() before setting
459 * up the RSA context.
460 * Additionally, if the implementation performs empirical checks,
461 * these checks substantiate but do not guarantee consistency.</li>
462 * <li>Quality: This function is not expected to perform
463 * extended quality assessments like checking that the prime
464 * factors are safe. Additionally, it is the responsibility of the
465 * user to ensure the trustworthiness of the source of his RSA
466 * parameters, which goes beyond what is effectively checkable
467 * by the library.</li></ul>
468 */
469int mbedtls_rsa_check_privkey( const mbedtls_rsa_context *ctx );

507/**
508 * \brief This function performs an RSA private key operation.
509 *
510 * \param ctx The RSA context.
511 * \param f_rng The RNG function. Needed for blinding.
512 * \param p_rng The RNG parameter.
513 * \param input The input buffer.
514 * \param output The output buffer.
515 *
516 * \return \c 0 on success, or an \c MBEDTLS_ERR_RSA_XXX error code
517 * on failure.
518 *
519 * \note The input and output buffers must be large
520 * enough. For example, 128 Bytes if RSA-1024 is used.
521 *
522 * \note Blinding is used if and only if a PRNG is provided.
523 *
524 * \note If blinding is used, both the base of exponentation
525 * and the exponent are blinded, providing protection
526 * against some side-channel attacks.
527 *
528 * \warning It is deprecated and a security risk to not provide
529 * a PRNG here and thereby prevent the use of blinding.
530 * Future versions of the library may enforce the presence
531 * of a PRNG.
532 *
533 */

535 int (*f_rng)(void *, unsigned char *, size_t),
536 void *p_rng,

538 unsigned char *output );

540/**
541 * \brief This function adds the message padding, then performs an RSA
542 * operation.
543 *
544 * It is the generic wrapper for performing a PKCS#1 encryption
545 * operation using the \p mode from the context.
546 *
547 *
548 * \param ctx The RSA context.
549 * \param f_rng The RNG function. Needed for padding, PKCS#1 v2.1
550 * encoding, and #MBEDTLS_RSA_PRIVATE.
551 * \param p_rng The RNG parameter.
552 * \param mode #MBEDTLS_RSA_PUBLIC or #MBEDTLS_RSA_PRIVATE.
553 * \param ilen The length of the plaintext.
554 * \param input The buffer holding the data to encrypt.
555 * \param output The buffer used to hold the ciphertext.
556 *
557 * \deprecated It is deprecated and discouraged to call this function
558 * in #MBEDTLS_RSA_PRIVATE mode. Future versions of the library
559 * are likely to remove the \p mode argument and have it

561 *
562 * \note Alternative implementations of RSA need not support
563 * mode being set to #MBEDTLS_RSA_PRIVATE and might instead
564 * return #MBEDTLS_ERR_RSA_UNSUPPORTED_OPERATION.
565 *
566 * \return \c 0 on success, or an \c MBEDTLS_ERR_RSA_XXX error code

568 *

570 * of \p ctx->N. For example, 128 Bytes if RSA-1024 is used.

614/**
615 * \brief This function performs a PKCS#1 v2.1 OAEP encryption
616 * operation (RSAES-OAEP-ENCRYPT).
617 *
618 * \param ctx The RSA context.
