420/**
421 * \brief Callback type: send data on the network.
422 *
423 * \note That callback may be either blocking or non-blocking.
424 *
425 * \param ctx Context for the send callback (typically a file descriptor)
426 * \param buf Buffer holding the data to send
427 * \param len Length of the data to send
428 *
429 * \return The callback must return the number of bytes sent if any,
430 * or a non-zero error code.
431 * If performing non-blocking I/O, \c MBEDTLS_ERR_SSL_WANT_WRITE
432 * must be returned when the operation would block.
433 *
434 * \note The callback is allowed to send fewer bytes than requested.
435 * It must always return the number of bytes actually sent.
436 */
437typedef int mbedtls_ssl_send_t( void *ctx,
438 const unsigned char *buf,
439 size_t len );
440
441/**
442 * \brief Callback type: receive data from the network.
443 *
444 * \note That callback may be either blocking or non-blocking.
445 *
446 * \param ctx Context for the receive callback (typically a file
447 * descriptor)
448 * \param buf Buffer to write the received data to
449 * \param len Length of the receive buffer
450 *
451 * \return The callback must return the number of bytes received,
452 * or a non-zero error code.
453 * If performing non-blocking I/O, \c MBEDTLS_ERR_SSL_WANT_READ
454 * must be returned when the operation would block.
455 *
456 * \note The callback may receive fewer bytes than the length of the
457 * buffer. It must always return the number of bytes actually
458 * received and written to the buffer.
459 */
460typedef int mbedtls_ssl_recv_t( void *ctx,
461 unsigned char *buf,
462 size_t len );

464/**
465 * \brief Callback type: receive data from the network, with timeout
466 *
467 * \note That callback must block until data is received, or the
468 * timeout delay expires, or the operation is interrupted by a
469 * signal.
470 *
471 * \param ctx Context for the receive callback (typically a file descriptor)
472 * \param buf Buffer to write the received data to
473 * \param len Length of the receive buffer
474 * \param timeout Maximum nomber of millisecondes to wait for data
475 * 0 means no timeout (potentially waiting forever)
476 *
477 * \return The callback must return the number of bytes received,
478 * or a non-zero error code:
479 * \c MBEDTLS_ERR_SSL_TIMEOUT if the operation timed out,
480 * \c MBEDTLS_ERR_SSL_WANT_READ if interrupted by a signal.
481 *
482 * \note The callback may receive fewer bytes than the length of the

2228 * when the request is emitted. In order to increase
2229 * reliability, we can accept a number of records before the
2230 * expected handshake records.

2233 * scenario.

2235 * \note With DTLS and server-initiated renegotiation, the
2236 * HelloRequest is retransmited every time mbedtls_ssl_read() times
2237 * out or receives Application Data, until:
2238 * - max_records records have beens seen, if it is >= 0, or
2239 * - the number of retransmits that would happen during an
2240 * actual handshake has been reached.
2241 * Please remember the request might be lost a few times
2242 * if you consider setting max_records to a really low value.
2243 *
2244 * \warning On client, the grace period can only happen during
2245 * mbedtls_ssl_read(), as opposed to mbedtls_ssl_write() and mbedtls_ssl_renegotiate()
2246 * which always behave as if max_record was 0. The reason is,
2247 * if we receive application data from the server, we need a
2248 * place to write it, which only happens during mbedtls_ssl_read().
2249 *
2250 * \param conf SSL configuration
2251 * \param max_records Use MBEDTLS_SSL_RENEGOTIATION_NOT_ENFORCED if you don't want to
2252 * enforce renegotiation, or a non-negative value to enforce
2253 * it but allow for a grace period of max_records records.
