161/** Supported cipher padding types. */

163 MBEDTLS_PADDING_PKCS7 = 0, /**< PKCS7 padding (default). */

203/**
204 * Cipher information. Allows calling cipher functions
205 * in a generic way.
206 */
207typedef struct {
208 /** Full cipher identifier. For example,
209 * MBEDTLS_CIPHER_AES_256_CBC.
210 */
211 mbedtls_cipher_type_t type;
212
213 /** The cipher mode. For example, MBEDTLS_MODE_CBC. */
214 mbedtls_cipher_mode_t mode;
215
216 /** The cipher key length, in bits. This is the
217 * default length for variable sized ciphers.
218 * Includes parity bits for ciphers like DES.
219 */
220 unsigned int key_bitlen;
221
222 /** Name of the cipher. */
223 const char * name;
224
225 /** IV or nonce size, in Bytes.
226 * For ciphers that accept variable IV sizes,
227 * this is the recommended size.
228 */
229 unsigned int iv_size;
230
231 /** Flags to set. For example, if the cipher supports variable IV sizes or variable key sizes. */
232 int flags;
233
234 /** The block size, in Bytes. */
235 unsigned int block_size;
236
237 /** Struct for base cipher information and functions. */
238 const mbedtls_cipher_base_t *base;
239
240} mbedtls_cipher_info_t;
241
242/**
243 * Generic cipher context.
244 */
245typedef struct {
246 /** Information about the associated cipher. */
247 const mbedtls_cipher_info_t *cipher_info;
248
249 /** Key length to use. */
250 int key_bitlen;
251
252 /** Operation that the key of the context has been
253 * initialized for.
254 */
255 mbedtls_operation_t operation;
256
257#if defined(MBEDTLS_CIPHER_MODE_WITH_PADDING)
258 /** Padding functions to use, if relevant for
259 * the specific cipher mode.
260 */
261 void (*add_padding)( unsigned char *output, size_t olen, size_t data_len );
262 int (*get_padding)( unsigned char *input, size_t ilen, size_t *data_len );
263#endif
264
265 /** Buffer for input that has not been processed yet. */
266 unsigned char unprocessed_data[MBEDTLS_MAX_BLOCK_LENGTH];
267
268 /** Number of Bytes that have not been processed yet. */
269 size_t unprocessed_len;
270
271 /** Current IV or NONCE_COUNTER for CTR-mode. */
272 unsigned char iv[MBEDTLS_MAX_IV_LENGTH];
273
274 /** IV size in Bytes, for ciphers with variable-length IVs. */
275 size_t iv_size;
276
277 /** The cipher-specific context. */
278 void *cipher_ctx;
279
280#if defined(MBEDTLS_CMAC_C)
281 /** CMAC-specific context. */
282 mbedtls_cmac_context_t *cmac_ctx;
283#endif
284} mbedtls_cipher_context_t;

366/**
367 * \brief This function returns the block size of the given cipher.
368 *
369 * \param ctx The context of the cipher. Must be initialized.
370 *
371 * \return The size of the blocks of the cipher, or zero if \p ctx
372 * has not been initialized.
373 */
374static inline unsigned int mbedtls_cipher_get_block_size( const mbedtls_cipher_context_t *ctx )
375{
376 if( NULL == ctx || NULL == ctx->cipher_info )
377 return 0;
378
379 return ctx->cipher_info->block_size;
380}

382/**
383 * \brief This function returns the mode of operation for
384 * the cipher. For example, MBEDTLS_MODE_CBC.
385 *
386 * \param ctx The context of the cipher. Must be initialized.
387 *
388 * \return The mode of operation, or #MBEDTLS_MODE_NONE if
389 * \p ctx has not been initialized.
390 */
391static inline mbedtls_cipher_mode_t mbedtls_cipher_get_cipher_mode( const mbedtls_cipher_context_t *ctx )

394 return MBEDTLS_MODE_NONE;

399/**
400 * \brief This function returns the size of the IV or nonce
401 * of the cipher, in Bytes.
402 *
403 * \param ctx The context of the cipher. Must be initialized.
404 *
405 * \return <ul><li>If no IV has been set: the recommended IV size.
406 * 0 for ciphers not using IV or nonce.</li>
407 * <li>If IV has already been set: the actual size.</li></ul>
408 */
409static inline int mbedtls_cipher_get_iv_size( const mbedtls_cipher_context_t *ctx )
410{
411 if( NULL == ctx || NULL == ctx->cipher_info )
412 return 0;
413
414 if( ctx->iv_size != 0 )
415 return (int) ctx->iv_size;
416
417 return (int) ctx->cipher_info->iv_size;
418}

566/**
567 * \brief The generic cipher update function. It encrypts or
568 * decrypts using the given cipher context. Writes as
569 * many block-sized blocks of data as possible to output.
570 * Any data that cannot be written immediately is either
571 * added to the next block, or flushed when
572 * mbedtls_cipher_finish() is called.
573 * Exception: For MBEDTLS_MODE_ECB, expects a single block
574 * in size. For example, 16 Bytes for AES.
575 *
576 * \param ctx The generic cipher context.
577 * \param input The buffer holding the input data.
578 * \param ilen The length of the input data.
579 * \param output The buffer for the output data. Must be able to hold at
580 * least \p ilen + block_size. Must not be the same buffer
581 * as input.
582 * \param olen The length of the output data, to be updated with the
583 * actual number of Bytes written.
584 *
585 * \returns \c 0 on success, #MBEDTLS_ERR_CIPHER_BAD_INPUT_DATA if
586 * parameter verification fails,
587 * #MBEDTLS_ERR_CIPHER_FEATURE_UNAVAILABLE on an
588 * unsupported mode for a cipher, or a cipher-specific
589 * error code.
590 *
591 * \note If the underlying cipher is GCM, all calls to this
592 * function, except the last one before
593 * mbedtls_cipher_finish(). Must have \p ilen as a
594 * multiple of the block_size.
595 */
596int mbedtls_cipher_update( mbedtls_cipher_context_t *ctx, const unsigned char *input,
597 size_t ilen, unsigned char *output, size_t *olen );
598
599/**
600 * \brief The generic cipher finalization function. If data still
601 * needs to be flushed from an incomplete block, the data
602 * contained in it is padded to the size of
603 * the last block, and written to the \p output buffer.
604 *
605 * \param ctx The generic cipher context.
606 * \param output The buffer to write data to. Needs block_size available.
607 * \param olen The length of the data written to the \p output buffer.
608 *
609 * \returns \c 0 on success, #MBEDTLS_ERR_CIPHER_BAD_INPUT_DATA if
610 * parameter verification fails,
611 * #MBEDTLS_ERR_CIPHER_FULL_BLOCK_EXPECTED if decryption
612 * expected a full block but was not provided one,
613 * #MBEDTLS_ERR_CIPHER_INVALID_PADDING on invalid padding
614 * while decrypting, or a cipher-specific error code
615 * on failure for any other reason.
616 */
617int mbedtls_cipher_finish( mbedtls_cipher_context_t *ctx,
618 unsigned char *output, size_t *olen );

651 * \brief The generic all-in-one encryption/decryption function,
652 * for all ciphers except AEAD constructs.
653 *
654 * \param ctx The generic cipher context.
655 * \param iv The IV to use, or NONCE_COUNTER for CTR-mode ciphers.
656 * \param iv_len The IV length for ciphers with variable-size IV.
657 * This parameter is discarded by ciphers with fixed-size
658 * IV.
659 * \param input The buffer holding the input data.
660 * \param ilen The length of the input data.
661 * \param output The buffer for the output data. Must be able to hold at
662 * least \p ilen + block_size. Must not be the same buffer
663 * as input.
664 * \param olen The length of the output data, to be updated with the
665 * actual number of Bytes written.
666 *
667 * \note Some ciphers do not use IVs nor nonce. For these
668 * ciphers, use \p iv = NULL and \p iv_len = 0.
669 *
670 * \returns \c 0 on success, or
671 * #MBEDTLS_ERR_CIPHER_BAD_INPUT_DATA, or
672 * #MBEDTLS_ERR_CIPHER_FULL_BLOCK_EXPECTED if decryption
673 * expected a full block but was not provided one, or
674 * #MBEDTLS_ERR_CIPHER_INVALID_PADDING on invalid padding
675 * while decrypting, or a cipher-specific error code on
676 * failure for any other reason.
677 */
678int mbedtls_cipher_crypt( mbedtls_cipher_context_t *ctx,
679 const unsigned char *iv, size_t iv_len,
680 const unsigned char *input, size_t ilen,
681 unsigned char *output, size_t *olen );

692 * \param ad_len The length of \p ad.
693 * \param input The buffer holding the input data.
694 * \param ilen The length of the input data.
695 * \param output The buffer for the output data.

697 * \param olen The length of the output data, to be updated with the
698 * actual number of Bytes written.

703 * #MBEDTLS_ERR_CIPHER_BAD_INPUT_DATA, or

706int mbedtls_cipher_auth_encrypt( mbedtls_cipher_context_t *ctx,
