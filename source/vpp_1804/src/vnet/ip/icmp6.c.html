44static u8 *
45format_ip6_icmp_type_and_code (u8 * s, va_list * args)
46{
47 icmp6_type_t type = va_arg (*args, int);
48 u8 code = va_arg (*args, int);
49 char *t = 0;
50
51#define _(n,f) case n: t = #f; break;
52
53 switch (type)
54 {
55 foreach_icmp6_type;
56
57 default:
58 break;
59 }
60
61#undef _
62
63 if (!t)
64 return format (s, "unknown 0x%x", type);
65
66 s = format (s, "%s", t);
67
68 t = 0;
69 switch ((type << 8) | code)
70 {
71#define _(a,n,f) case (ICMP6_##a << 8) | (n): t = #f; break;
72
73 foreach_icmp6_code;
74
75#undef _
76 }
77
78 if (t)
79 s = format (s, " %s", t);
80
81 return s;
82}

84static u8 *
85format_icmp6_header (u8 * s, va_list * args)
86{
87 icmp46_header_t *icmp = va_arg (*args, icmp46_header_t *);
88 u32 max_header_bytes = va_arg (*args, u32);
89
90 /* Nothing to do. */
91 if (max_header_bytes < sizeof (icmp[0]))
92 return format (s, "ICMP header truncated");
93
94 s = format (s, "ICMP %U checksum 0x%x",
95 format_ip6_icmp_type_and_code, icmp->type, icmp->code,
96 clib_net_to_host_u16 (icmp->checksum));
97
98 if (max_header_bytes >=
99 sizeof (icmp6_neighbor_solicitation_or_advertisement_header_t) &&
100 (icmp->type == ICMP6_neighbor_solicitation ||
101 icmp->type == ICMP6_neighbor_advertisement))
102 {
103 icmp6_neighbor_solicitation_or_advertisement_header_t *icmp6_nd =
104 (icmp6_neighbor_solicitation_or_advertisement_header_t *) icmp;
105 s = format (s, "\n target address %U",
106 format_ip6_address, &icmp6_nd->target_address);
107 }
108
109 return s;
110}

112u8 *
113format_icmp6_input_trace (u8 * s, va_list * va)
114{
115 CLIB_UNUSED (vlib_main_t * vm) = va_arg (*va, vlib_main_t *);
116 CLIB_UNUSED (vlib_node_t * node) = va_arg (*va, vlib_node_t *);
117 icmp6_input_trace_t *t = va_arg (*va, icmp6_input_trace_t *);
118
119 s = format (s, "%U",
120 format_ip6_header, t->packet_data, sizeof (t->packet_data));
121
122 return s;
123}

137typedef struct
138{
139 uword *type_and_code_by_name;
140
141 uword *type_by_name;
142
143 /* Vector dispatch table indexed by [icmp type]. */
144 u8 input_next_index_by_type[256];
145
146 /* Max valid code indexed by icmp type. */
147 u8 max_valid_code_by_type[256];
148
149 /* hop_limit must be >= this value for this icmp type. */
150 u8 min_valid_hop_limit_by_type[256];
151
152 u8 min_valid_length_by_type[256];
153} icmp6_main_t;

157static uword
158ip6_icmp_input (vlib_main_t * vm,
159 vlib_node_runtime_t * node, vlib_frame_t * frame)
160{
161 icmp6_main_t *im = &icmp6_main;
162 u32 *from, *to_next;
163 u32 n_left_from, n_left_to_next, next_index;
164
165 from = vlib_frame_vector_args (frame);
166 n_left_from = frame->n_vectors;
167 next_index = node->cached_next_index;
168
169 if (node->flags & VLIB_NODE_FLAG_TRACE)
170 vlib_trace_frame_buffers_only (vm, node, from, frame->n_vectors,
171 /* stride */ 1,
172 sizeof (icmp6_input_trace_t));
173
174 while (n_left_from > 0)
175 {
176 vlib_get_next_frame (vm, node, next_index, to_next, n_left_to_next);
177
178 while (n_left_from > 0 && n_left_to_next > 0)
179 {
180 vlib_buffer_t *b0;
181 ip6_header_t *ip0;
182 icmp46_header_t *icmp0;
183 icmp6_type_t type0;
184 u32 bi0, next0, error0, len0;
185
186 bi0 = to_next[0] = from[0];
187
188 from += 1;
189 n_left_from -= 1;
190 to_next += 1;
191 n_left_to_next -= 1;
192
193 b0 = vlib_get_buffer (vm, bi0);
194 ip0 = vlib_buffer_get_current (b0);
195 icmp0 = ip6_next_header (ip0);
196 type0 = icmp0->type;
197
198 error0 = ICMP6_ERROR_NONE;
199
200 next0 = im->input_next_index_by_type[type0];
201 error0 =
202 next0 == ICMP_INPUT_NEXT_DROP ? ICMP6_ERROR_UNKNOWN_TYPE : error0;
203
204 /* Check code is valid for type. */
205 error0 =
206 icmp0->code >
207 im->max_valid_code_by_type[type0] ?
208 ICMP6_ERROR_INVALID_CODE_FOR_TYPE : error0;
209
210 /* Checksum is already validated by ip6_local node so we don't need to check that. */
211
212 /* Check that hop limit == 255 for certain types. */
213 error0 =
214 ip0->hop_limit <
215 im->min_valid_hop_limit_by_type[type0] ?
216 ICMP6_ERROR_INVALID_HOP_LIMIT_FOR_TYPE : error0;
217
218 len0 = clib_net_to_host_u16 (ip0->payload_length);
219 error0 =
220 len0 <
221 im->min_valid_length_by_type[type0] ?
222 ICMP6_ERROR_LENGTH_TOO_SMALL_FOR_TYPE : error0;
223
224 b0->error = node->errors[error0];
225
226 next0 = error0 != ICMP6_ERROR_NONE ? ICMP_INPUT_NEXT_DROP : next0;
227
228 vlib_validate_buffer_enqueue_x1 (vm, node, next_index,
229 to_next, n_left_to_next,
230 bi0, next0);
231 }
232
233 vlib_put_next_frame (vm, node, next_index, n_left_to_next);
234 }
235
236 return frame->n_vectors;
237}

265static uword
266ip6_icmp_echo_request (vlib_main_t * vm,
267 vlib_node_runtime_t * node, vlib_frame_t * frame)
268{
269 u32 *from, *to_next;
270 u32 n_left_from, n_left_to_next, next_index;
271 ip6_main_t *im = &ip6_main;
272
273 from = vlib_frame_vector_args (frame);
274 n_left_from = frame->n_vectors;
275 next_index = node->cached_next_index;
276
277 if (node->flags & VLIB_NODE_FLAG_TRACE)
278 vlib_trace_frame_buffers_only (vm, node, from, frame->n_vectors,
279 /* stride */ 1,
280 sizeof (icmp6_input_trace_t));
281
282 while (n_left_from > 0)
283 {
284 vlib_get_next_frame (vm, node, next_index, to_next, n_left_to_next);
285
286 while (n_left_from > 2 && n_left_to_next > 2)
287 {
288 vlib_buffer_t *p0, *p1;
289 ip6_header_t *ip0, *ip1;
290 icmp46_header_t *icmp0, *icmp1;
291 ip6_address_t tmp0, tmp1;
292 ip_csum_t sum0, sum1;
293 u32 bi0, bi1;
294 u32 fib_index0, fib_index1;
295 u32 next0 = ICMP6_ECHO_REQUEST_NEXT_LOOKUP;
296 u32 next1 = ICMP6_ECHO_REQUEST_NEXT_LOOKUP;
297
298 bi0 = to_next[0] = from[0];
299 bi1 = to_next[1] = from[1];
300
301 from += 2;
302 n_left_from -= 2;
303 to_next += 2;
304 n_left_to_next -= 2;
305
306 p0 = vlib_get_buffer (vm, bi0);
307 p1 = vlib_get_buffer (vm, bi1);
308 ip0 = vlib_buffer_get_current (p0);
309 ip1 = vlib_buffer_get_current (p1);
310 icmp0 = ip6_next_header (ip0);
311 icmp1 = ip6_next_header (ip1);
312
313 /* Check icmp type to echo reply and update icmp checksum. */
314 sum0 = icmp0->checksum;
315 sum1 = icmp1->checksum;
316
317 ASSERT (icmp0->type == ICMP6_echo_request);
318 ASSERT (icmp1->type == ICMP6_echo_request);
319 sum0 = ip_csum_update (sum0, ICMP6_echo_request, ICMP6_echo_reply,
320 icmp46_header_t, type);
321 sum1 = ip_csum_update (sum1, ICMP6_echo_request, ICMP6_echo_reply,
322 icmp46_header_t, type);
323
324 icmp0->checksum = ip_csum_fold (sum0);
325 icmp1->checksum = ip_csum_fold (sum1);
326
327 icmp0->type = ICMP6_echo_reply;
328 icmp1->type = ICMP6_echo_reply;
329
330 /* Swap source and destination address. */
331 tmp0 = ip0->src_address;
332 tmp1 = ip1->src_address;
333
334 ip0->src_address = ip0->dst_address;
335 ip1->src_address = ip1->dst_address;
336
337 ip0->dst_address = tmp0;
338 ip1->dst_address = tmp1;
339
340 /* New hop count. */
341 ip0->hop_limit = im->host_config.ttl;
342 ip1->hop_limit = im->host_config.ttl;
343
344 /* Determine the correct lookup fib indices... */
345 fib_index0 = vec_elt (im->fib_index_by_sw_if_index,
346 vnet_buffer (p0)->sw_if_index[VLIB_RX]);
347 vnet_buffer (p0)->sw_if_index[VLIB_TX] = fib_index0;
348 /* Determine the correct lookup fib indices... */
349 fib_index1 = vec_elt (im->fib_index_by_sw_if_index,
350 vnet_buffer (p1)->sw_if_index[VLIB_RX]);
351 vnet_buffer (p1)->sw_if_index[VLIB_TX] = fib_index1;
352
353 /* verify speculative enqueues, maybe switch current next frame */
354 /* if next0==next1==next_index then nothing special needs to be done */
355 vlib_validate_buffer_enqueue_x2 (vm, node, next_index,
356 to_next, n_left_to_next,
357 bi0, bi1, next0, next1);
358 }
359
360 while (n_left_from > 0 && n_left_to_next > 0)
361 {
362 vlib_buffer_t *p0;
363 ip6_header_t *ip0;
364 icmp46_header_t *icmp0;
365 u32 bi0;
366 ip6_address_t tmp0;
367 ip_csum_t sum0;
368 u32 fib_index0;
369 u32 next0 = ICMP6_ECHO_REQUEST_NEXT_LOOKUP;
370
371 bi0 = to_next[0] = from[0];
372
373 from += 1;
374 n_left_from -= 1;
375 to_next += 1;
376 n_left_to_next -= 1;
377
378 p0 = vlib_get_buffer (vm, bi0);
379 ip0 = vlib_buffer_get_current (p0);
380 icmp0 = ip6_next_header (ip0);
381
382 /* Check icmp type to echo reply and update icmp checksum. */
383 sum0 = icmp0->checksum;
384
385 ASSERT (icmp0->type == ICMP6_echo_request);
386 sum0 = ip_csum_update (sum0, ICMP6_echo_request, ICMP6_echo_reply,
387 icmp46_header_t, type);
388
389 icmp0->checksum = ip_csum_fold (sum0);
390
391 icmp0->type = ICMP6_echo_reply;
392
393 /* Swap source and destination address. */
394 tmp0 = ip0->src_address;
395 ip0->src_address = ip0->dst_address;
396 ip0->dst_address = tmp0;
397
398 ip0->hop_limit = im->host_config.ttl;
399
400 /* if the packet is link local, we'll bounce through the link-local
401 * table with the RX interface correctly set */
402 fib_index0 = vec_elt (im->fib_index_by_sw_if_index,
403 vnet_buffer (p0)->sw_if_index[VLIB_RX]);
404 vnet_buffer (p0)->sw_if_index[VLIB_TX] = fib_index0;
405
406 /* Verify speculative enqueue, maybe switch current next frame */
407 vlib_validate_buffer_enqueue_x1 (vm, node, next_index,
408 to_next, n_left_to_next,
409 bi0, next0);
410 }
411
412 vlib_put_next_frame (vm, node, next_index, n_left_to_next);
413 }
414
415 vlib_error_count (vm, ip6_icmp_input_node.index,
416 ICMP6_ERROR_ECHO_REPLIES_SENT, frame->n_vectors);
417
418 return frame->n_vectors;
419}

453static u8
454icmp6_icmp_type_to_error (u8 type)
455{
456 switch (type)
457 {
458 case ICMP6_destination_unreachable:
459 return ICMP6_ERROR_DEST_UNREACH_SENT;
460 case ICMP6_packet_too_big:
461 return ICMP6_ERROR_PACKET_TOO_BIG_SENT;
462 case ICMP6_time_exceeded:
463 return ICMP6_ERROR_TTL_EXPIRE_SENT;
464 case ICMP6_parameter_problem:
465 return ICMP6_ERROR_PARAM_PROBLEM_SENT;
466 default:
467 return ICMP6_ERROR_DROP;
468 }
469}

471static uword
472ip6_icmp_error (vlib_main_t * vm,
473 vlib_node_runtime_t * node, vlib_frame_t * frame)
474{
475 u32 *from, *to_next;
476 uword n_left_from, n_left_to_next;
477 ip6_icmp_error_next_t next_index;
478 ip6_main_t *im = &ip6_main;
479 ip_lookup_main_t *lm = &im->lookup_main;
480
481 from = vlib_frame_vector_args (frame);
482 n_left_from = frame->n_vectors;
483 next_index = node->cached_next_index;
484
485 if (node->flags & VLIB_NODE_FLAG_TRACE)
486 vlib_trace_frame_buffers_only (vm, node, from, frame->n_vectors,
487 /* stride */ 1,
488 sizeof (icmp6_input_trace_t));
489
490 while (n_left_from > 0)
491 {
492 vlib_get_next_frame (vm, node, next_index, to_next, n_left_to_next);
493
494 while (n_left_from > 0 && n_left_to_next > 0)
495 {
496 u32 pi0 = from[0];
497 u32 next0 = IP6_ICMP_ERROR_NEXT_LOOKUP;
498 u8 error0 = ICMP6_ERROR_NONE;
499 vlib_buffer_t *p0;
500 ip6_header_t *ip0, *out_ip0;
501 icmp46_header_t *icmp0;
502 u32 sw_if_index0, if_add_index0;
503 int bogus_length;
504
505 /* Speculatively enqueue p0 to the current next frame */
506 to_next[0] = pi0;
507 from += 1;
508 to_next += 1;
509 n_left_from -= 1;
510 n_left_to_next -= 1;
511
512 p0 = vlib_get_buffer (vm, pi0);
513 ip0 = vlib_buffer_get_current (p0);
514 sw_if_index0 = vnet_buffer (p0)->sw_if_index[VLIB_RX];
515
516 /* RFC4443 says to keep as much of the original packet as possible
517 * within the minimum MTU. We cheat "a little" here by keeping whatever fits
518 * in the first buffer, to be more efficient */
519 if (PREDICT_FALSE (p0->total_length_not_including_first_buffer))
520 { /* clear current_length of all other buffers in chain */
521 vlib_buffer_t *b = p0;
522 p0->total_length_not_including_first_buffer = 0;
523 while (b->flags & VLIB_BUFFER_NEXT_PRESENT)
524 {
525 b = vlib_get_buffer (vm, b->next_buffer);
526 b->current_length = 0;
527 }
528 }
529 p0->current_length =
530 p0->current_length > 1280 ? 1280 : p0->current_length;
531
532 /* Add IP header and ICMPv6 header including a 4 byte data field */
533 vlib_buffer_advance (p0,
534 -sizeof (ip6_header_t) -
535 sizeof (icmp46_header_t) - 4);
536 out_ip0 = vlib_buffer_get_current (p0);
537 icmp0 = (icmp46_header_t *) & out_ip0[1];
538
539 /* Fill ip header fields */
540 out_ip0->ip_version_traffic_class_and_flow_label =
541 clib_host_to_net_u32 (0x6 << 28);
542
543 out_ip0->payload_length =
544 clib_host_to_net_u16 (p0->current_length - sizeof (ip6_header_t));
545 out_ip0->protocol = IP_PROTOCOL_ICMP6;
546 out_ip0->hop_limit = 0xff;
547 out_ip0->dst_address = ip0->src_address;
548 if_add_index0 =
549 lm->if_address_pool_index_by_sw_if_index[sw_if_index0];
550 if (PREDICT_TRUE (if_add_index0 != ~0))
551 {
552 ip_interface_address_t *if_add =
553 pool_elt_at_index (lm->if_address_pool, if_add_index0);
554 ip6_address_t *if_ip =
555 ip_interface_address_get_address (lm, if_add);
556 out_ip0->src_address = *if_ip;
557 }
558 else /* interface has no IP6 address - should not happen */
559 {
560 next0 = IP6_ICMP_ERROR_NEXT_DROP;
561 error0 = ICMP6_ERROR_DROP;
562 }
563
564 /* Fill icmp header fields */
565 icmp0->type = vnet_buffer (p0)->ip.icmp.type;
566 icmp0->code = vnet_buffer (p0)->ip.icmp.code;
567 *((u32 *) (icmp0 + 1)) =
568 clib_host_to_net_u32 (vnet_buffer (p0)->ip.icmp.data);
569 icmp0->checksum = 0;
570 icmp0->checksum =
571 ip6_tcp_udp_icmp_compute_checksum (vm, p0, out_ip0,
572 &bogus_length);
573
574
575
576 /* Update error status */
577 if (error0 == ICMP6_ERROR_NONE)
578 error0 = icmp6_icmp_type_to_error (icmp0->type);
579 vlib_error_count (vm, node->node_index, error0, 1);
580
581 /* Verify speculative enqueue, maybe switch current next frame */
582 vlib_validate_buffer_enqueue_x1 (vm, node, next_index,
583 to_next, n_left_to_next,
584 pi0, next0);
585 }
586 vlib_put_next_frame (vm, node, next_index, n_left_to_next);
587 }
588
589 return frame->n_vectors;
590}

612static uword
613unformat_icmp_type_and_code (unformat_input_t * input, va_list * args)
614{
615 icmp46_header_t *h = va_arg (*args, icmp46_header_t *);
616 icmp6_main_t *cm = &icmp6_main;
617 u32 i;
618
619 if (unformat_user (input, unformat_vlib_number_by_name,
620 cm->type_and_code_by_name, &i))
621 {
622 h->type = (i >> 8) & 0xff;
623 h->code = (i >> 0) & 0xff;
624 }
625 else if (unformat_user (input, unformat_vlib_number_by_name,
626 cm->type_by_name, &i))
627 {
628 h->type = i;
629 h->code = 0;
630 }
631 else
632 return 0;
633
634 return 1;
635}

637static void
638icmp6_pg_edit_function (pg_main_t * pg,
639 pg_stream_t * s,
640 pg_edit_group_t * g, u32 * packets, u32 n_packets)
641{
642 vlib_main_t *vm = vlib_get_main ();
643 u32 ip_offset, icmp_offset;
644 int bogus_length;
645
646 icmp_offset = g->start_byte_offset;
647 ip_offset = (g - 1)->start_byte_offset;
648
649 while (n_packets >= 1)
650 {
651 vlib_buffer_t *p0;
652 ip6_header_t *ip0;
653 icmp46_header_t *icmp0;
654
655 p0 = vlib_get_buffer (vm, packets[0]);
656 n_packets -= 1;
657 packets += 1;
658
659 ASSERT (p0->current_data == 0);
660 ip0 = (void *) (p0->data + ip_offset);
661 icmp0 = (void *) (p0->data + icmp_offset);
662
663 icmp0->checksum = ip6_tcp_udp_icmp_compute_checksum (vm, p0, ip0,
664 &bogus_length);
665 ASSERT (bogus_length == 0);
666 }
667}

670{
671 pg_edit_t type, code;
672 pg_edit_t checksum;
673} pg_icmp46_header_t;

675always_inline void
676pg_icmp_header_init (pg_icmp46_header_t * p)
677{
678 /* Initialize fields that are not bit fields in the IP header. */
679#define _(f) pg_edit_init (&p->f, icmp46_header_t, f);
680 _(type);
681 _(code);
682 _(checksum);
683#undef _
684}

686static uword
687unformat_pg_icmp_header (unformat_input_t * input, va_list * args)
688{
689 pg_stream_t *s = va_arg (*args, pg_stream_t *);
690 pg_icmp46_header_t *p;
691 u32 group_index;
692
693 p = pg_create_edit_group (s, sizeof (p[0]), sizeof (icmp46_header_t),
694 &group_index);
695 pg_icmp_header_init (p);
696
697 p->checksum.type = PG_EDIT_UNSPECIFIED;
698
699 {
700 icmp46_header_t tmp;
701
702 if (!unformat (input, "ICMP %U", unformat_icmp_type_and_code, &tmp))
703 goto error;
704
705 pg_edit_set_fixed (&p->type, tmp.type);
706 pg_edit_set_fixed (&p->code, tmp.code);
707 }
708
709 /* Parse options. */
710 while (1)
711 {
712 if (unformat (input, "checksum %U",
713 unformat_pg_edit, unformat_pg_number, &p->checksum))
714 ;
715
716 /* Can't parse input: try next protocol level. */
717 else
718 break;
719 }
720
721 if (!unformat_user (input, unformat_pg_payload, s))
722 goto error;
723
724 if (p->checksum.type == PG_EDIT_UNSPECIFIED)
725 {
726 pg_edit_group_t *g = pg_stream_get_group (s, group_index);
727 g->edit_function = icmp6_pg_edit_function;
728 g->edit_function_opaque = 0;
729 }
730
731 return 1;
732
733error:
734 /* Free up any edits we may have added. */
735 pg_free_edit_group (s);
736 return 0;
737}

749static clib_error_t *
750icmp6_init (vlib_main_t * vm)
751{
752 ip_main_t *im = &ip_main;
753 ip_protocol_info_t *pi;
754 icmp6_main_t *cm = &icmp6_main;
755 clib_error_t *error;
756
757 error = vlib_call_init_function (vm, ip_main_init);
758
759 if (error)
760 return error;
761
762 pi = ip_get_protocol_info (im, IP_PROTOCOL_ICMP6);
763 pi->format_header = format_icmp6_header;
764 pi->unformat_pg_edit = unformat_pg_icmp_header;
765
766 cm->type_by_name = hash_create_string (0, sizeof (uword));
767#define _(n,t) hash_set_mem (cm->type_by_name, #t, (n));
768 foreach_icmp6_type;
769#undef _
770
771 cm->type_and_code_by_name = hash_create_string (0, sizeof (uword));
772#define _(a,n,t) hash_set_mem (cm->type_by_name, #t, (n) | (ICMP6_##a << 8));
773 foreach_icmp6_code;
774#undef _
775
776 memset (cm->input_next_index_by_type,
777 ICMP_INPUT_NEXT_DROP, sizeof (cm->input_next_index_by_type));
778 memset (cm->max_valid_code_by_type, 0, sizeof (cm->max_valid_code_by_type));
779
780#define _(a,n,t) cm->max_valid_code_by_type[ICMP6_##a] = clib_max (cm->max_valid_code_by_type[ICMP6_##a], n);
781 foreach_icmp6_code;
782#undef _
783
784 memset (cm->min_valid_hop_limit_by_type, 0,
785 sizeof (cm->min_valid_hop_limit_by_type));
786 cm->min_valid_hop_limit_by_type[ICMP6_router_solicitation] = 255;
787 cm->min_valid_hop_limit_by_type[ICMP6_router_advertisement] = 255;
788 cm->min_valid_hop_limit_by_type[ICMP6_neighbor_solicitation] = 255;
789 cm->min_valid_hop_limit_by_type[ICMP6_neighbor_advertisement] = 255;
790 cm->min_valid_hop_limit_by_type[ICMP6_redirect] = 255;
791
792 memset (cm->min_valid_length_by_type, sizeof (icmp46_header_t),
793 sizeof (cm->min_valid_length_by_type));
794 cm->min_valid_length_by_type[ICMP6_router_solicitation] =
795 sizeof (icmp6_neighbor_discovery_header_t);
796 cm->min_valid_length_by_type[ICMP6_router_advertisement] =
797 sizeof (icmp6_router_advertisement_header_t);
798 cm->min_valid_length_by_type[ICMP6_neighbor_solicitation] =
799 sizeof (icmp6_neighbor_solicitation_or_advertisement_header_t);
800 cm->min_valid_length_by_type[ICMP6_neighbor_advertisement] =
801 sizeof (icmp6_neighbor_solicitation_or_advertisement_header_t);
802 cm->min_valid_length_by_type[ICMP6_redirect] =
803 sizeof (icmp6_redirect_header_t);
804
805 icmp6_register_type (vm, ICMP6_echo_request,
806 ip6_icmp_echo_request_node.index);
807
808 return vlib_call_init_function (vm, ip6_neighbor_init);
809}
