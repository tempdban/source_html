40#include <vnet/ip/ip.h>
41
42/* Format IP protocol. */
43u8 *
44format_ip_protocol (u8 * s, va_list * args)
45{
46 ip_protocol_t protocol = va_arg (*args, ip_protocol_t);
47 ip_main_t *im = &ip_main;
48 ip_protocol_info_t *pi = ip_get_protocol_info (im, protocol);
49
50 if (pi)
51 return format (s, "%s", pi->name);
52 else
53 return format (s, "unknown %d", protocol);
54}
55
56uword
57unformat_ip_protocol (unformat_input_t * input, va_list * args)
58{
59 u8 *result = va_arg (*args, u8 *);
60 ip_main_t *im = &ip_main;
61 ip_protocol_info_t *pi;
62 int i;
63
64 if (!unformat_user (input, unformat_vlib_number_by_name,
65 im->protocol_info_by_name, &i))
66 return 0;
67
68 pi = vec_elt_at_index (im->protocol_infos, i);
69 *result = pi->protocol;
70 return 1;
71}
72
73u8 *
74format_tcp_udp_port (u8 * s, va_list * args)
75{
76 int port = va_arg (*args, int);
77 ip_main_t *im = &ip_main;
78 tcp_udp_port_info_t *pi;
79
80 pi = ip_get_tcp_udp_port_info (im, port);
81 if (pi)
82 s = format (s, "%s", pi->name);
83 else
84 s = format (s, "%d", clib_net_to_host_u16 (port));
85
86 return s;
87}
88
89uword
90unformat_tcp_udp_port (unformat_input_t * input, va_list * args)
91{
92 u16 *result = va_arg (*args, u16 *);
93 ip_main_t *im = &ip_main;
94 tcp_udp_port_info_t *pi;
95 u32 i, port;
96
97
98 if (unformat_user (input, unformat_vlib_number_by_name,
99 im->port_info_by_name, &i))
100 {
101 pi = vec_elt_at_index (im->port_infos, i);
102 port = pi->port;
103 }
104 else if (unformat_user (input, unformat_vlib_number, &port)
105 && port < (1 << 16))
106 port = clib_host_to_net_u16 (port);
107
108 else
109 return 0;
110
111 *result = port;
112 return 1;
113}
