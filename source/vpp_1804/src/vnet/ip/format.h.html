40#ifndef included_ip_format_h
41#define included_ip_format_h
42
43/* IP4 or IP6. */
44
45format_function_t format_ip_protocol;
46unformat_function_t unformat_ip_protocol;
47
48format_function_t format_tcp_udp_port;
49unformat_function_t unformat_tcp_udp_port;
50
51typedef enum format_ip_adjacency_flags_t_
52{
53 FORMAT_IP_ADJACENCY_NONE,
54 FORMAT_IP_ADJACENCY_BRIEF = FORMAT_IP_ADJACENCY_NONE,
55 FORMAT_IP_ADJACENCY_DETAIL = (1 << 0),
56} format_ip_adjacency_flags_t;
57
58format_function_t format_ip_adjacency;
59format_function_t format_ip_adjacency_packet_data;
60
61format_function_t format_ip46_address;
62
63typedef enum
64{
65 IP46_TYPE_ANY,
66 IP46_TYPE_IP4,
67 IP46_TYPE_IP6
68} ip46_type_t;
69/* unformat_ip46_address expects arguments (ip46_address_t *, ip46_type_t)
70 * The type argument is used to enforce a particular IP version. */
71unformat_function_t unformat_ip46_address;
72
73/* IP4 */
74
75/* Parse an IP4 address %d.%d.%d.%d. */
76unformat_function_t unformat_ip4_address;
77
78/* Format an IP4 address. */
79format_function_t format_ip4_address;
80format_function_t format_ip4_address_and_length;
81
82/* Parse an IP4 header. */
83unformat_function_t unformat_ip4_header;
84
85/* Format an IP4 header. */
86format_function_t format_ip4_header;
87
88/* Parse an IP packet matching pattern. */
89unformat_function_t unformat_ip4_match;
90
91unformat_function_t unformat_pg_ip4_header;
92
93/* IP6 */
94unformat_function_t unformat_ip6_address;
95format_function_t format_ip6_address;
96format_function_t format_ip6_address_and_length;
97unformat_function_t unformat_ip6_header;
98format_function_t format_ip6_header;
99unformat_function_t unformat_pg_ip6_header;
100
101/* Format a TCP/UDP headers. */
102format_function_t format_tcp_header, format_udp_header, format_sctp_header;
103
104unformat_function_t unformat_pg_tcp_header, unformat_pg_udp_header,
105 unformat_pg_sctp_header;
106
107#endif /* included_ip_format_h */
