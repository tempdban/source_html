60/** Flow hash configuration */
61#define IP_FLOW_HASH_SRC_ADDR (1<<0)
62#define IP_FLOW_HASH_DST_ADDR (1<<1)
63#define IP_FLOW_HASH_PROTO (1<<2)
64#define IP_FLOW_HASH_SRC_PORT (1<<3)
65#define IP_FLOW_HASH_DST_PORT (1<<4)
66#define IP_FLOW_HASH_REVERSE_SRC_DST (1<<5)
67
68/** Default: 5-tuple without the "reverse" bit */
69#define IP_FLOW_HASH_DEFAULT (0x1F)
70
71#define foreach_flow_hash_bit \
72_(src, IP_FLOW_HASH_SRC_ADDR) \
73_(dst, IP_FLOW_HASH_DST_ADDR) \
74_(sport, IP_FLOW_HASH_SRC_PORT) \
75_(dport, IP_FLOW_HASH_DST_PORT) \
76_(proto, IP_FLOW_HASH_PROTO) \
77_(reverse, IP_FLOW_HASH_REVERSE_SRC_DST)
78

80 * A flow hash configuration is a mask of the flow hash options

84/* An all zeros address */

88typedef struct
89{
90 /* Key for mhash; in fact, just a byte offset into mhash key vector. */
91 u32 address_key;
92
93 /* Interface which has this address. */
94 u32 sw_if_index;
95
96 /* Address (prefix) length for this interface. */
97 u16 address_length;
98
99 /* Will be used for something eventually. Primary vs. secondary? */
100 u16 flags;
101
102 /* Next and previous pointers for doubly linked list of
103 addresses per software interface. */
104 u32 next_this_sw_interface;
105 u32 prev_this_sw_interface;
106} ip_interface_address_t;

109{
110 IP_LOCAL_NEXT_DROP,
111 IP_LOCAL_NEXT_PUNT,
112 IP_LOCAL_NEXT_UDP_LOOKUP,
113 IP_LOCAL_NEXT_ICMP,
114 IP_LOCAL_N_NEXT,
115} ip_local_next_t;

117struct ip_lookup_main_t;
118
119typedef struct ip_lookup_main_t
120{
121 /** Pool of addresses that are assigned to interfaces. */
122 ip_interface_address_t *if_address_pool;
123
124 /** Hash table mapping address to index in interface address pool. */
125 mhash_t address_to_if_address_index;
126
127 /** Head of doubly linked list of interface addresses for each software interface.
128 ~0 means this interface has no address. */
129 u32 *if_address_pool_index_by_sw_if_index;
130
131 /** First table index to use for this interface, ~0 => none */
132 u32 *classify_table_index_by_sw_if_index;
133
134 /** Feature arc indices */
135 u8 mcast_feature_arc_index;
136 u8 ucast_feature_arc_index;
137 u8 output_feature_arc_index;
138
139 /** Number of bytes in a fib result. Must be at least
140 sizeof (uword). First word is always adjacency index. */
141 u32 fib_result_n_bytes, fib_result_n_words;
142
143 /** 1 for ip6; 0 for ip4. */
144 u32 is_ip6;
145
146 /** Either format_ip4_address_and_length or format_ip6_address_and_length. */
147 format_function_t *format_address_and_length;
148
149 /** Table mapping ip protocol to ip[46]-local node next index. */
150 u8 local_next_by_ip_protocol[256];
151
152 /** IP_BUILTIN_PROTOCOL_{TCP,UDP,ICMP,OTHER} by protocol in IP header. */
153 u8 builtin_protocol_by_ip_protocol[256];
154} ip_lookup_main_t;

164always_inline ip_interface_address_t *
165ip_get_interface_address (ip_lookup_main_t * lm, void *addr_fib)
166{
167 uword *p = mhash_get (&lm->address_to_if_address_index, addr_fib);
168 return p ? pool_elt_at_index (lm->if_address_pool, p[0]) : 0;
169}
170
171always_inline void *
172ip_interface_address_get_address (ip_lookup_main_t * lm,
173 ip_interface_address_t * a)
174{
175 return mhash_key_to_mem (&lm->address_to_if_address_index, a->address_key);
176}

178/* *INDENT-OFF* */
179#define foreach_ip_interface_address(lm,a,sw_if_index,loop,body) \
180do { \
181 vnet_main_t *_vnm = vnet_get_main(); \
182 u32 _sw_if_index = sw_if_index; \
183 vnet_sw_interface_t *_swif; \
184 _swif = vnet_get_sw_interface (_vnm, _sw_if_index); \
185 \
186 /* \
187 * Loop => honor unnumbered interface addressing. \
188 */ \
189 if (_swif->flags & VNET_SW_INTERFACE_FLAG_UNNUMBERED) \
190 { \
191 if (loop) \
192 _sw_if_index = _swif->unnumbered_sw_if_index; \
193 else \
194 /* the interface is unnumbered, by the caller does not want \
195 * unnumbered interfaces considered/honoured */ \
196 break; \
197 } \
198 u32 _ia = ((vec_len((lm)->if_address_pool_index_by_sw_if_index) \
199 > (_sw_if_index)) ? \
200 vec_elt ((lm)->if_address_pool_index_by_sw_if_index, \
201 (_sw_if_index)) : \
202 (u32)~0); \
203 ip_interface_address_t * _a; \
204 while (_ia != ~0) \
205 { \
206 _a = pool_elt_at_index ((lm)->if_address_pool, _ia); \
207 _ia = _a->next_this_sw_interface; \
208 (a) = _a; \
209 body; \
210 } \
211} while (0)
