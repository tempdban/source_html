16/*
17 * IPv4 and IPv6 Fragmentation Nodes
18 *
19 * A packet sent to those nodes require the following
20 * buffer attributes to be set:
21 * ip_frag.header_offset :
22 * Where to find the IPv4 (or IPv6) header in the packet. Previous
23 * bytes are left untouched and copied in every fragment. The fragments
24 * are then appended. This option is used for fragmented packets
25 * that are encapsulated.
26 * ip_frag.mtu :
27 * Maximum size of IP packets, header included, but ignoring
28 * the 'ip_frag.header_offset' copied bytes.
29 * ip_frag.next_index :
30 * One of ip_frag_next_t, indicating to which exit node the fragments
31 * should be sent to.
32 *
33 */
34
35#ifndef IP_FRAG_H
36#define IP_FRAG_H
37
38#include <vnet/vnet.h>
39
40#define IP_FRAG_FLAG_IP4_HEADER 0x01 //Encapsulating IPv4 header
41#define IP_FRAG_FLAG_IP6_HEADER 0x02 //Encapsulating IPv6 header
42
43#define IP4_FRAG_NODE_NAME "ip4-frag"
44#define IP6_FRAG_NODE_NAME "ip6-frag"
45
46extern vlib_node_registration_t ip4_frag_node;
47extern vlib_node_registration_t ip6_frag_node;
48
49typedef enum
50{
51 IP4_FRAG_NEXT_IP4_LOOKUP,
52 IP4_FRAG_NEXT_IP6_LOOKUP,
53 IP4_FRAG_NEXT_ICMP_ERROR,
54 IP4_FRAG_NEXT_DROP,
55 IP4_FRAG_N_NEXT
56} ip4_frag_next_t;
57
58typedef enum
59{
60 IP6_FRAG_NEXT_IP4_LOOKUP,
61 IP6_FRAG_NEXT_IP6_LOOKUP,
62 IP6_FRAG_NEXT_DROP,
63 IP6_FRAG_N_NEXT
64} ip6_frag_next_t;
65
66#define foreach_ip_frag_error \
67 /* Must be first. */ \
68 _(NONE, "packet fragmented") \
69 _(SMALL_PACKET, "packet smaller than MTU") \
70 _(FRAGMENT_SENT, "number of sent fragments") \
71 _(CANT_FRAGMENT_HEADER, "can't fragment header") \
72 _(DONT_FRAGMENT_SET, "can't fragment this packet") \
73 _(MALFORMED, "malformed packet") \
74 _(MEMORY, "could not allocate buffer") \
75 _(UNKNOWN, "unknown error")
76
77typedef enum
78{
79#define _(sym,str) IP_FRAG_ERROR_##sym,
80 foreach_ip_frag_error
81#undef _
82 IP_FRAG_N_ERROR,
83} ip_frag_error_t;
84
85void ip_frag_set_vnet_buffer (vlib_buffer_t * b, u16 offset, u16 mtu,
86 u8 next_index, u8 flags);
87
88#endif /* ifndef IP_FRAG_H */
