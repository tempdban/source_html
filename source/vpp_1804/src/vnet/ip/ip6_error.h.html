40#ifndef included_ip_ip6_error_h
41#define included_ip_ip6_error_h
42
43#define foreach_ip6_error \
44 /* Must be first. */ \
45 _ (NONE, "valid ip6 packets") \
46 \
47 /* Errors signalled by ip6-input */ \
48 _ (TOO_SHORT, "ip6 length < 40 bytes") \
49 _ (BAD_LENGTH, "ip6 length > l2 length") \
50 _ (VERSION, "ip6 version != 6") \
51 _ (TIME_EXPIRED, "ip6 ttl <= 1") \
52 \
53 /* Errors signalled by ip6-rewrite. */ \
54 _ (MTU_EXCEEDED, "ip6 MTU exceeded") \
55 _ (DST_LOOKUP_MISS, "ip6 destination lookup miss") \
56 _ (SRC_LOOKUP_MISS, "ip6 source lookup miss") \
57 _ (DROP, "ip6 drop") \
58 _ (PUNT, "ip6 punt") \
59 \
60 /* Errors signalled by ip6-local. */ \
61 _ (UNKNOWN_PROTOCOL, "unknown ip protocol") \
62 _ (UDP_CHECKSUM, "bad udp checksum") \
63 _ (ICMP_CHECKSUM, "bad icmp checksum") \
64 _ (UDP_LENGTH, "inconsistent udp/ip lengths") \
65 \
66 /* Errors signalled by udp6-lookup. */ \
67 _ (UNKNOWN_UDP_PORT, "no listener for udp port") \
68 \
69 /* Spoofed packets in ip6-rewrite-local */ \
70 _(SPOOFED_LOCAL_PACKETS, "ip4 spoofed local-address packet drops") \
71 \
72 /* Erros singalled by ip6-inacl */ \
73 _ (INACL_TABLE_MISS, "input ACL table-miss drops") \
74 _ (INACL_SESSION_DENY, "input ACL session deny drops") \
75 /* Erros singalled by ip6-outacl */ \
76 _ (OUTACL_TABLE_MISS, "output ACL table-miss drops") \
77 _ (OUTACL_SESSION_DENY, "output ACL session deny drops") \
78 \
79 /* Errors signalled by ip6-reassembly */ \
80 _ (REASS_MISSING_UPPER, "missing-upper layer drops") \
81 _ (REASS_DUPLICATE_FRAGMENT, "duplicate fragments") \
82 _ (REASS_OVERLAPPING_FRAGMENT, "overlapping fragments") \
83 _ (REASS_LIMIT_REACHED, "drops due to concurrent reassemblies limit") \
84 _ (REASS_TIMEOUT, "fragments dropped due to reassembly timeout")
85
86typedef enum
87{
88#define _(sym,str) IP6_ERROR_##sym,
89 foreach_ip6_error
90#undef _
91 IP6_N_ERROR,
92} ip6_error_t;
93
94#endif /* included_ip_ip6_error_h */
