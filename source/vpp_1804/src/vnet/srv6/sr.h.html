74/* SR policy types */
75#define SR_POLICY_TYPE_DEFAULT 0
76#define SR_POLICY_TYPE_SPRAY 1
77/**
78 * @brief SR Policy
79 */
80typedef struct
81{
82 u32 *segments_lists; /**< SID lists indexes (vector) */
83
84 ip6_address_t bsid; /**< BindingSID (key) */
85
86 u8 type; /**< Type (default is 0) */
87 /* SR Policy specific DPO */
88 /* IF Type = DEFAULT Then Load Balancer DPO among SID lists */
89 /* IF Type = SPRAY then Spray DPO with all SID lists */
90 dpo_id_t bsid_dpo; /**< SR Policy specific DPO - BSID */
91 dpo_id_t ip4_dpo; /**< SR Policy specific DPO - IPv6 */
92 dpo_id_t ip6_dpo; /**< SR Policy specific DPO - IPv4 */
93
94 u32 fib_table; /**< FIB table */
95
96 u8 is_encap; /**< Mode (0 is SRH insert, 1 Encaps) */
97} ip6_sr_policy_t;

281/**
282 * @brief SR rewrite string computation for SRH insertion (inline)
283 *
284 * @param sl is a vector of IPv6 addresses composing the Segment List
285 *
286 * @return precomputed rewrite string for SRH insertion
287 */
288static inline u8 *
289ip6_sr_compute_rewrite_string_insert (ip6_address_t * sl)
290{
291 ip6_sr_header_t *srh;
292 ip6_address_t *addrp, *this_address;
293 u32 header_length = 0;
294 u8 *rs = NULL;
295
296 header_length = 0;
297 header_length += sizeof (ip6_sr_header_t);
298 header_length += (vec_len (sl) + 1) * sizeof (ip6_address_t);
299
300 vec_validate (rs, header_length - 1);
301
302 srh = (ip6_sr_header_t *) rs;
303 srh->type = ROUTING_HEADER_TYPE_SR;
304 srh->segments_left = vec_len (sl);
305 srh->first_segment = vec_len (sl);
306 srh->length = ((sizeof (ip6_sr_header_t) +
307 ((vec_len (sl) + 1) * sizeof (ip6_address_t))) / 8) - 1;
308 srh->flags = 0x00;
309 srh->reserved = 0x0000;
310 addrp = srh->segments + vec_len (sl);
311 vec_foreach (this_address, sl)
312 {
313 clib_memcpy (addrp->as_u8, this_address->as_u8, sizeof (ip6_address_t));
314 addrp--;
315 }
316 return rs;
317}
