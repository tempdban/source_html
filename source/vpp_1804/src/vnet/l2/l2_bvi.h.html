27#define TO_BVI_ERR_OK 0
28#define TO_BVI_ERR_BAD_MAC 1
29#define TO_BVI_ERR_ETHERTYPE 2

31/**
32 * Send a packet from L2 processing to L3 via the BVI interface.
33 * Set next0 to the proper L3 input node.
34 * Return an error if the packet isn't what we expect.
35 */
36
37static_always_inline u32
38l2_to_bvi (vlib_main_t * vlib_main,
39 vnet_main_t * vnet_main,
40 vlib_buffer_t * b0,
41 u32 bvi_sw_if_index, next_by_ethertype_t * l3_next, u32 * next0)
42{
43 u8 l2_len;
44 u16 ethertype;
45 u8 *l3h;
46 ethernet_header_t *e0;
47 vnet_hw_interface_t *hi;
48
49 e0 = vlib_buffer_get_current (b0);
50 hi = vnet_get_sup_hw_interface (vnet_main, bvi_sw_if_index);
51
52 /* Perform L3 my-mac filter */
53 if ((!ethernet_address_cast (e0->dst_address)) &&
54 (!eth_mac_equal ((u8 *) e0, hi->hw_address)))
55 {
56 return TO_BVI_ERR_BAD_MAC;
57 }
58
59 /* Save L2 header position which may be changed due to packet replication */
60 vnet_buffer (b0)->l2_hdr_offset = b0->current_data;
61
62 /* Strip L2 header */
63 l2_len = vnet_buffer (b0)->l2.l2_len;
64 vlib_buffer_advance (b0, l2_len);
65
66 l3h = vlib_buffer_get_current (b0);
67 ethertype = clib_net_to_host_u16 (*(u16 *) (l3h - 2));
68
69 /* Set the input interface to be the BVI interface */
70 vnet_buffer (b0)->sw_if_index[VLIB_RX] = bvi_sw_if_index;
71 vnet_buffer (b0)->sw_if_index[VLIB_TX] = ~0;
72
73 /* Go to appropriate L3 input node */
74 if (ethertype == ETHERNET_TYPE_IP4)
75 {
76 *next0 = l3_next->input_next_ip4;
77 }
78 else if (ethertype == ETHERNET_TYPE_IP6)
79 {
80 *next0 = l3_next->input_next_ip6;
81 }
82 else
83 {
84 /* uncommon ethertype, check table */
85 u32 i0;
86
87 i0 = sparse_vec_index (l3_next->input_next_by_type, ethertype);
88 *next0 = vec_elt (l3_next->input_next_by_type, i0);
89
90 if (i0 == SPARSE_VEC_INVALID_INDEX)
91 {
92 return TO_BVI_ERR_ETHERTYPE;
93 }
94 }
95
96 /* increment BVI RX interface stat */
97 vlib_increment_combined_counter
98 (vnet_main->interface_main.combined_sw_if_counters
99 + VNET_INTERFACE_COUNTER_RX,
100 vlib_main->thread_index,
101 vnet_buffer (b0)->sw_if_index[VLIB_RX],
102 1, vlib_buffer_length_in_chain (vlib_main, b0));
103 return TO_BVI_ERR_OK;
104}
