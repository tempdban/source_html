52static uword
53vnet_local_interface_tx (vlib_main_t * vm,
54 vlib_node_runtime_t * node, vlib_frame_t * f)
55{
56 ASSERT (0);
57 return f->n_vectors;
58}

73clib_error_t *
74vnet_main_init (vlib_main_t * vm)
75{
76 vnet_main_t *vnm = vnet_get_main ();
77 clib_error_t *error;
78 u32 hw_if_index;
79 vnet_hw_interface_t *hw;
80
81 if ((error = vlib_call_init_function (vm, vnet_interface_init)))
82 return error;
83
84 if ((error = vlib_call_init_function (vm, fib_module_init)))
85 return error;
86
87 if ((error = vlib_call_init_function (vm, mfib_module_init)))
88 return error;
89
90 if ((error = vlib_call_init_function (vm, ip_main_init)))
91 return error;
92
93 if ((error = vlib_call_init_function (vm, ip4_lookup_init)))
94 return error;
95
96 if ((error = vlib_call_init_function (vm, ip6_lookup_init)))
97 return error;
98
99 if ((error = vlib_call_init_function (vm, mpls_init)))
100 return error;
101
102 vnm->vlib_main = vm;
103
104 hw_if_index = vnet_register_interface
105 (vnm, vnet_local_interface_device_class.index, /* instance */ 0,
106 vnet_local_interface_hw_class.index, /* instance */ 0);
107 hw = vnet_get_hw_interface (vnm, hw_if_index);
108
109 vnm->local_interface_hw_if_index = hw_if_index;
110 vnm->local_interface_sw_if_index = hw->sw_if_index;
111
112 /* the local interface is used as an input interface when decapping from
113 * an IPSEC tunnel. so it needs to be IP enabled */
114 ip4_sw_interface_enable_disable (hw->sw_if_index, 1);
115 ip6_sw_interface_enable_disable (hw->sw_if_index, 1);
116
117 return 0;
118}

120VLIB_INIT_FUNCTION (vnet_main_init);
