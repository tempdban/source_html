147typedef enum

149#define _(a,b,c) VNET_API_ERROR_##a = (b),

151#undef _
152 VNET_API_N_ERROR,
153} vnet_api_error_t;
154
155/* *INDENT-OFF* */
156static inline u8 *
157format_vnet_api_errno (u8 * s, va_list * args)
158{
159 vnet_api_error_t api_error = va_arg (*args, vnet_api_error_t);
160#ifdef _
161#undef _
162#endif
163#define _(a, b, c) \
164 case b: \
165 s = format (s, "%s", c); \
166 break;
167 switch (api_error)
168 {
169 foreach_vnet_api_error
170 default:
171 s = format (s, "UNKNOWN");
172 break;
173 }
174 return s;
175#undef _
176}
