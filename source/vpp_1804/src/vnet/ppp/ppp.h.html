40#ifndef included_ppp_h
41#define included_ppp_h
42
43#include <vnet/vnet.h>
44#include <vnet/ppp/packet.h>
45#include <vnet/pg/pg.h>
46
47extern vnet_hw_interface_class_t ppp_hw_interface_class;
48
49typedef enum
50{
51#define ppp_error(n,s) PPP_ERROR_##n,
52#include <vnet/ppp/error.def>
53#undef ppp_error
54 PPP_N_ERROR,
55} ppp_error_t;
56
57typedef struct
58{
59 /* Name (a c string). */
60 char *name;
61
62 /* PPP protocol type in host byte order. */
63 ppp_protocol_t protocol;
64
65 /* Node which handles this type. */
66 u32 node_index;
67
68 /* Next index for this type. */
69 u32 next_index;
70} ppp_protocol_info_t;
71
72typedef struct
73{
74 vlib_main_t *vlib_main;
75
76 ppp_protocol_info_t *protocol_infos;
77
78 /* Hash tables mapping name/protocol to protocol info index. */
79 uword *protocol_info_by_name, *protocol_info_by_protocol;
80} ppp_main_t;
81
82always_inline ppp_protocol_info_t *
83ppp_get_protocol_info (ppp_main_t * em, ppp_protocol_t protocol)
84{
85 uword *p = hash_get (em->protocol_info_by_protocol, protocol);
86 return p ? vec_elt_at_index (em->protocol_infos, p[0]) : 0;
87}
88
89extern ppp_main_t ppp_main;
90
91/* Register given node index to take input for given ppp type. */
92void
93ppp_register_input_type (vlib_main_t * vm,
94 ppp_protocol_t protocol, u32 node_index);
95
96format_function_t format_ppp_protocol;
97format_function_t format_ppp_header;
98format_function_t format_ppp_header_with_length;
99
100/* Parse ppp protocol as 0xXXXX or protocol name.
101 In either host or network byte order. */
102unformat_function_t unformat_ppp_protocol_host_byte_order;
103unformat_function_t unformat_ppp_protocol_net_byte_order;
104
105/* Parse ppp header. */
106unformat_function_t unformat_ppp_header;
107unformat_function_t unformat_pg_ppp_header;
108
109always_inline void
110ppp_setup_node (vlib_main_t * vm, u32 node_index)
111{
112 vlib_node_t *n = vlib_get_node (vm, node_index);
113 pg_node_t *pn = pg_get_node (node_index);
114
115 n->format_buffer = format_ppp_header_with_length;
116 n->unformat_buffer = unformat_ppp_header;
117 pn->unformat_edit = unformat_pg_ppp_header;
118}
119
120void
121ppp_register_input_protocol (vlib_main_t * vm,
122 ppp_protocol_t protocol, u32 node_index);
123
124#endif /* included_ppp_h */
