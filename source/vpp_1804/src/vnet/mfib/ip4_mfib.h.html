59{
60 return (&(pool_elt_at_index(ip4_main.mfibs, index)->v4));

79static inline
80u32 ip4_mfib_index_from_table_id (u32 table_id)
81{
82 ip4_main_t * im = &ip4_main;
83 uword * p;
84
85 p = hash_get (im->mfib_index_by_table_id, table_id);
86 if (!p)
87 return ~0;
88
89 return p[0];
90}
