87static inline
88u32 ip6_mfib_index_from_table_id (u32 table_id)
89{
90 ip6_main_t * im = &ip6_main;
91 uword * p;
92
93 p = hash_get (im->mfib_index_by_table_id, table_id);
94 if (!p)
95 return ~0;
96
97 return p[0];
98}
