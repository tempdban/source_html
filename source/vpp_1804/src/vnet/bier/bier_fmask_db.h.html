18
19#ifndef __BIER_FMASK_DB_H__
20#define __BIER_FMASK_DB_H__
21
22#include <vnet/ip/ip.h>
23
24#include <vnet/fib/fib_types.h>
25
26/**
27 * BIER header encapulsation types
28 */
29typedef enum bier_hdr_type_t_ {
30 /**
31 * BIER Header in MPLS networks
32 */
33 BIER_HDR_O_MPLS,
34
35 /**
36 * BIER header in non-MPLS networks
37 */
38 BIER_HDR_O_OTHER,
39} bier_hdr_type_t;
40
41/**
42 * A key/ID for a BIER forwarding Mas (FMask).
43 * This is a simplified version of a fib_route_path_t.
44 */
45typedef struct bier_fmask_id_t_ {
46 /**
47 * Type of BIER header this fmask supports
48 */
49 bier_hdr_type_t bfmi_hdr_type;
50
51 /**
52 * The BIER table this fmask is in
53 */
54 index_t bfmi_bti;
55
56 union {
57 /**
58 * next-hop of the peer
59 */
60 ip46_address_t bfmi_nh;
61
62 /**
63 * ID of the next-hop object, e.g. a UDP-encap
64 */
65 u32 bfmi_id;
66 };
67
68 /**
69 * Software interface index
70 */
71 u32 bfmi_sw_if_index;
72} __attribute__((packed)) bier_fmask_id_t;
73
74extern index_t
75bier_fmask_db_find_or_create_and_lock(index_t bti,
76 const fib_route_path_t *rpath);
77extern index_t bier_fmask_db_find (index_t bti,
78 const fib_route_path_t *rpath);
79
80extern void bier_fmask_db_remove (const bier_fmask_id_t *fmid);
81
82
83#endif
