38/* *INDENT-OFF* */
39typedef CLIB_PACKED (struct tls_cxt_id_
40{
41 u32 parent_app_index;
42 session_handle_t app_session_handle;
43 session_handle_t tls_session_handle;
44 u32 listener_ctx_index;
45 u8 tcp_is_ip4;
46 u8 tls_engine_id;
47}) tls_ctx_id_t;
48/* *INDENT-ON* */
49
50STATIC_ASSERT (sizeof (tls_ctx_id_t) <= 42, "ctx id must be less than 42");
51
52typedef struct tls_ctx_
53{
54 union
55 {
56 transport_connection_t connection;
57 tls_ctx_id_t c_tls_ctx_id;
58 };
59#define parent_app_index c_tls_ctx_id.parent_app_index
60#define app_session_handle c_tls_ctx_id.app_session_handle
61#define tls_session_handle c_tls_ctx_id.tls_session_handle
62#define listener_ctx_index c_tls_ctx_id.listener_ctx_index
63#define tcp_is_ip4 c_tls_ctx_id.tcp_is_ip4
64#define tls_ctx_engine c_tls_ctx_id.tls_engine_id
65#define tls_ctx_handle c_c_index
66 /* Temporary storage for session open opaque. Overwritten once
67 * underlying tcp connection is established */
68#define parent_app_api_context c_s_index
69
70 u8 is_passive_close;
71 u8 *srv_hostname;
72} tls_ctx_t;
73
74typedef struct tls_main_
75{
76 u32 app_index;
77 tls_ctx_t *listener_ctx_pool;
78 tls_ctx_t *half_open_ctx_pool;
79 clib_rwlock_t half_open_rwlock;
80 u8 **rx_bufs;
81 u8 **tx_bufs;
82
83 /*
84 * Config
85 */
86 u8 use_test_cert_in_ca;
87 char *ca_cert_path;
88} tls_main_t;
89
90typedef struct tls_engine_vft_
91{
92 u32 (*ctx_alloc) (void);
93 void (*ctx_free) (tls_ctx_t * ctx);
94 tls_ctx_t *(*ctx_get) (u32 ctx_index);
95 tls_ctx_t *(*ctx_get_w_thread) (u32 ctx_index, u8 thread_index);
96 int (*ctx_init_client) (tls_ctx_t * ctx);
97 int (*ctx_init_server) (tls_ctx_t * ctx);
98 int (*ctx_read) (tls_ctx_t * ctx, stream_session_t * tls_session);
99 int (*ctx_write) (tls_ctx_t * ctx, stream_session_t * app_session);
100 u8 (*ctx_handshake_is_over) (tls_ctx_t * ctx);
101} tls_engine_vft_t;
102
103typedef enum tls_engine_type_
104{
105 TLS_ENGINE_NONE,
106 TLS_ENGINE_MBEDTLS,
107 TLS_ENGINE_OPENSSL,
108 TLS_N_ENGINES
109} tls_engine_type_t;
110
111tls_main_t *vnet_tls_get_main (void);
112void tls_register_engine (const tls_engine_vft_t * vft,

114int tls_add_vpp_q_evt (svm_fifo_t * f, u8 evt_type);
115int tls_notify_app_accept (tls_ctx_t * ctx);
116int tls_notify_app_connected (tls_ctx_t * ctx, u8 is_failed);
117void tls_notify_app_enqueue (tls_ctx_t * ctx, stream_session_t * app_session);
