222always_inline u8
223stream_session_is_valid (u32 si, u8 thread_index)
224{
225 stream_session_t *s;
226 s = pool_elt_at_index (session_manager_main.sessions[thread_index], si);
227 if (s->thread_index != thread_index || s->session_index != si
228 /* || s->server_rx_fifo->master_session_index != si
229 || s->server_tx_fifo->master_session_index != si
230 || s->server_rx_fifo->master_thread_index != thread_index
231 || s->server_tx_fifo->master_thread_index != thread_index */ )
232 return 0;
233 return 1;
234}

247always_inline stream_session_t *
248session_get_if_valid (u64 si, u32 thread_index)
249{
250 if (thread_index >= vec_len (session_manager_main.sessions))
251 return 0;
252
253 if (pool_is_free_index (session_manager_main.sessions[thread_index], si))
254 return 0;
255
256 ASSERT (stream_session_is_valid (si, thread_index));
257 return pool_elt_at_index (session_manager_main.sessions[thread_index], si);
258}

348/**
349 * Acquires a lock that blocks a session pool from expanding.
350 *
351 * This is typically used for safely peeking into other threads'
352 * pools in order to clone elements. Lock should be dropped as soon
353 * as possible by calling @ref session_pool_remove_peeker.
354 *
355 * NOTE: Avoid using pool_elt_at_index while the lock is held because
356 * it may lead to free elt bitmap expansion/contraction!
357 */
358always_inline void
359session_pool_add_peeker (u32 thread_index)
360{
361 session_manager_main_t *smm = &session_manager_main;
362 if (thread_index == vlib_get_thread_index ())
363 return;
364 clib_rwlock_reader_lock (&smm->peekers_rw_locks[thread_index]);
365}
366
367always_inline void
368session_pool_remove_peeker (u32 thread_index)
369{
370 session_manager_main_t *smm = &session_manager_main;
371 if (thread_index == vlib_get_thread_index ())
372 return;
373 clib_rwlock_reader_unlock (&smm->peekers_rw_locks[thread_index]);
374}
375
376/**
377 * Get session from handle and 'lock' pool resize if not in same thread
378 *
379 * Caller should drop the peek 'lock' as soon as possible.
380 */
381always_inline stream_session_t *
382session_get_from_handle_safe (u64 handle)
383{
384 session_manager_main_t *smm = &session_manager_main;
385 u32 thread_index = session_thread_from_handle (handle);
386 if (thread_index == vlib_get_thread_index ())
387 {
388 return pool_elt_at_index (smm->sessions[thread_index],
389 session_index_from_handle (handle));
390 }
391 else
392 {
393 session_pool_add_peeker (thread_index);
394 /* Don't use pool_elt_at index. See @ref session_pool_add_peeker */
395 return smm->sessions[thread_index] + session_index_from_handle (handle);
396 }
397}

434always_inline stream_session_t *
435session_clone_safe (u32 session_index, u32 thread_index)
436{
437 stream_session_t *old_s, *new_s;
438 u32 current_thread_index = vlib_get_thread_index ();
439
440 /* If during the memcpy pool is reallocated AND the memory allocator
441 * decides to give the old chunk of memory to somebody in a hurry to
442 * scribble something on it, we have a problem. So add this thread as
443 * a session pool peeker.
444 */
445 session_pool_add_peeker (thread_index);
446 new_s = session_alloc (current_thread_index);
447 old_s = session_manager_main.sessions[thread_index] + session_index;
448 clib_memcpy (new_s, old_s, sizeof (*new_s));
449 session_pool_remove_peeker (thread_index);
450 new_s->thread_index = current_thread_index;
451 new_s->session_index = session_get_index (new_s);
452 return new_s;
453}

519always_inline stream_session_t *
520listen_session_get_from_handle (session_handle_t handle)
521{
522 session_manager_main_t *smm = &session_manager_main;
523 stream_session_t *s;
524 u32 type, index;
525 type = handle >> 32;
526 index = handle & 0xFFFFFFFF;
527
528 if (pool_is_free_index (smm->listen_sessions[type], index))
529 return 0;
530
531 s = pool_elt_at_index (smm->listen_sessions[type], index);
532 ASSERT (s->session_state == SESSION_STATE_LISTENING);
533 return s;
534}

543always_inline stream_session_t *
544listen_session_new (session_type_t type)
545{
546 stream_session_t *s;
547 pool_get_aligned (session_manager_main.listen_sessions[type], s,
548 CLIB_CACHE_LINE_BYTES);
549 memset (s, 0, sizeof (*s));
550
551 s->session_type = type;
552 s->session_state = SESSION_STATE_LISTENING;
553 s->session_index = s - session_manager_main.listen_sessions[type];
554
555 return s;
556}
