40#include <vlib/vlib.h>
41#include <vnet/pg/pg.h>
42#include <vnet/hdlc/hdlc.h>
43
44typedef struct
45{
46 pg_edit_t address;
47 pg_edit_t control;
48 pg_edit_t protocol;
49} pg_hdlc_header_t;
50
51static inline void
52pg_hdlc_header_init (pg_hdlc_header_t * e)
53{
54 pg_edit_init (&e->address, hdlc_header_t, address);
55 pg_edit_init (&e->control, hdlc_header_t, control);
56 pg_edit_init (&e->protocol, hdlc_header_t, protocol);
57}
58
59uword
60unformat_pg_hdlc_header (unformat_input_t * input, va_list * args)
61{
62 pg_stream_t *s = va_arg (*args, pg_stream_t *);
63 pg_hdlc_header_t *h;
64 u32 group_index, error;
65
66 h = pg_create_edit_group (s, sizeof (h[0]), sizeof (hdlc_header_t),
67 &group_index);
68 pg_hdlc_header_init (h);
69
70 pg_edit_set_fixed (&h->address, 0x0f);
71 pg_edit_set_fixed (&h->control, 0x00);
72
73 error = 1;
74 if (!unformat (input, "%U",
75 unformat_pg_edit,
76 unformat_hdlc_protocol_net_byte_order, &h->protocol))
77 goto done;
78
79 {
80 hdlc_main_t *pm = &hdlc_main;
81 hdlc_protocol_info_t *pi = 0;
82 pg_node_t *pg_node = 0;
83
84 if (h->protocol.type == PG_EDIT_FIXED)
85 {
86 u16 t = *(u16 *) h->protocol.values[PG_EDIT_LO];
87 pi = hdlc_get_protocol_info (pm, clib_net_to_host_u16 (t));
88 if (pi && pi->node_index != ~0)
89 pg_node = pg_get_node (pi->node_index);
90 }
91
92 if (pg_node && pg_node->unformat_edit
93 && unformat_user (input, pg_node->unformat_edit, s))
94 ;
95
96 else if (!unformat_user (input, unformat_pg_payload, s))
97 goto done;
98 }
99
100 error = 0;
101done:
102 if (error)
103 pg_free_edit_group (s);
104 return error == 0;
105}
