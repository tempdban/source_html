40#include <vlib/vlib.h>
41#include <vnet/pg/pg.h>
42#include <vnet/ethernet/ethernet.h>
43
44typedef struct
45{
46 pg_edit_t type;
47 pg_edit_t src_address;
48 pg_edit_t dst_address;
49} pg_ethernet_header_t;
50
51static inline void
52pg_ethernet_header_init (pg_ethernet_header_t * e)
53{
54 pg_edit_init (&e->type, ethernet_header_t, type);
55 pg_edit_init (&e->src_address, ethernet_header_t, src_address);
56 pg_edit_init (&e->dst_address, ethernet_header_t, dst_address);
57}
58
59typedef struct
60{
61 pg_edit_t type;
62 pg_edit_t id;
63 pg_edit_t cfi;
64 pg_edit_t priority;
65} pg_ethernet_vlan_header_t;
66
67static inline void
68pg_ethernet_vlan_header_init (pg_ethernet_vlan_header_t * v, int vlan_index)
69{
70 ASSERT (vlan_index < ARRAY_LEN (((ethernet_max_header_t *) 0)->vlan));
71 pg_edit_init (&v->type, ethernet_max_header_t, vlan[vlan_index].type);
72
73 pg_edit_init_bitfield (&v->id, ethernet_max_header_t,
74 vlan[vlan_index].priority_cfi_and_id, 0, 12);
75 pg_edit_init_bitfield (&v->cfi, ethernet_max_header_t,
76 vlan[vlan_index].priority_cfi_and_id, 12, 1);
77 pg_edit_init_bitfield (&v->priority, ethernet_max_header_t,
78 vlan[vlan_index].priority_cfi_and_id, 13, 3);
79}
80
81uword
82unformat_pg_ethernet_header (unformat_input_t * input, va_list * args)
83{
84 pg_stream_t *s = va_arg (*args, pg_stream_t *);
85 pg_ethernet_header_t *e;
86 pg_ethernet_vlan_header_t *v;
87 pg_edit_t *ether_type_edit;
88 u32 n_vlan, error, group_index;
89
90 e = pg_create_edit_group (s, sizeof (e[0]), sizeof (ethernet_header_t),
91 &group_index);
92 pg_ethernet_header_init (e);
93 error = 1;
94
95 if (!unformat (input, "%U: %U -> %U",
96 unformat_pg_edit,
97 unformat_ethernet_type_net_byte_order, &e->type,
98 unformat_pg_edit,
99 unformat_ethernet_address, &e->src_address,
100 unformat_pg_edit,
101 unformat_ethernet_address, &e->dst_address))
102 goto done;
103
104 n_vlan = 0;
105 while (unformat (input, "vlan"))
106 {
107 v = pg_add_edits (s, sizeof (v[0]), sizeof (ethernet_vlan_header_t),
108 group_index);
109 pg_ethernet_vlan_header_init (v, n_vlan);
110
111 if (!unformat_user (input, unformat_pg_edit,
112 unformat_pg_number, &v->id))
113 goto done;
114
115 if (!unformat (input, "priority %U", unformat_pg_edit,
116 unformat_pg_number, &v->priority))
117 pg_edit_set_fixed (&v->priority, 0);
118
119 if (!unformat (input, "cfi %U", unformat_pg_edit,
120 unformat_pg_number, &v->cfi))
121 pg_edit_set_fixed (&v->cfi, 0);
122
123 /* Too many vlans given. */
124 if (n_vlan >= 2)
125 goto done;
126
127 n_vlan++;
128 }
129
130 /* Address of e may have changed due to vlan edits being added */
131 e = pg_get_edit_group (s, group_index);
132 v = (void *) (e + 1);
133
134 /* Correct types for vlan packets. */
135 ether_type_edit = &e->type;
136 if (n_vlan > 0)
137 {
138 int i;
139
140 ether_type_edit = &v[n_vlan - 1].type;
141 pg_edit_copy_type_and_values (ether_type_edit, &e->type);
142 pg_edit_set_fixed (&e->type, ETHERNET_TYPE_VLAN);
143
144 for (i = 0; i < n_vlan - 1; i++)
145 pg_edit_set_fixed (&v[i].type, ETHERNET_TYPE_VLAN);
146 }
147
148 {
149 ethernet_main_t *em = &ethernet_main;
150 ethernet_type_info_t *ti = 0;
151 pg_node_t *pg_node = 0;
152
153 if (ether_type_edit->type == PG_EDIT_FIXED)
154 {
155 u16 t = *(u16 *) ether_type_edit->values[PG_EDIT_LO];
156 ti = ethernet_get_type_info (em, clib_net_to_host_u16 (t));
157 if (ti && ti->node_index != ~0)
158 pg_node = pg_get_node (ti->node_index);
159 }
160
161 if (pg_node && pg_node->unformat_edit
162 && unformat_user (input, pg_node->unformat_edit, s))
163 ;
164 else if (!unformat_user (input, unformat_pg_payload, s))
165 goto done;
166 }
167
168 error = 0;
169
170done:
171 if (error)
172 pg_free_edit_group (s);
173 return error == 0;
174}
