16#ifndef __span_h__
17#define __span_h__
18
19#include <vnet/vnet.h>
20#include <vnet/ip/ip.h>
21#include <vnet/l2/l2_output.h>
22
23typedef enum
24{
25 SPAN_FEAT_DEVICE,
26 SPAN_FEAT_L2,
27 SPAN_FEAT_N
28} span_feat_t;
29
30typedef struct
31{
32 clib_bitmap_t *mirror_ports;
33 u32 num_mirror_ports;
34} span_mirror_t;
35
36typedef struct
37{
38 span_mirror_t mirror_rxtx[SPAN_FEAT_N][VLIB_N_RX_TX];
39} span_interface_t;
40
41typedef struct
42{
43 /* l2 feature Next nodes */
44 u32 l2_input_next[32];
45 u32 l2_output_next[32];
46
47 /* per-interface vector of span instances */
48 span_interface_t *interfaces;
49
50 /* biggest sw_if_index used so far */
51 u32 max_sw_if_index;
52
53 /* convenience */
54 vlib_main_t *vlib_main;
55 vnet_main_t *vnet_main;
56} span_main_t;
57
58extern span_main_t span_main;
59
60typedef struct
61{
62 u32 src_sw_if_index; /* mirrored interface index */
63 u32 mirror_sw_if_index; /* output interface index */
64} span_trace_t;
65
66#endif /* __span_h__ */

70 u32 dst_sw_if_index, u8 state, span_feat_t sf);
