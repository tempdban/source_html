348always_inline void
349ipsec_alloc_empty_buffers (vlib_main_t * vm, ipsec_main_t * im)
350{
351 u32 thread_index = vlib_get_thread_index ();
352 uword l = vec_len (im->empty_buffers[thread_index]);
353 uword n_alloc = 0;
354
355 if (PREDICT_FALSE (l < VLIB_FRAME_SIZE))
356 {
357 if (!im->empty_buffers[thread_index])
358 {
359 vec_alloc (im->empty_buffers[thread_index], 2 * VLIB_FRAME_SIZE);
360 }
361
362 n_alloc = vlib_buffer_alloc (vm, im->empty_buffers[thread_index] + l,
363 2 * VLIB_FRAME_SIZE - l);
364
365 _vec_len (im->empty_buffers[thread_index]) = l + n_alloc;
366 }
367}

369static_always_inline u32
370get_next_output_feature_node_index (vlib_buffer_t * b,
371 vlib_node_runtime_t * nr)
372{
373 u32 next;
374 u32 sw_if_index = vnet_buffer (b)->sw_if_index[VLIB_TX];
375 vlib_main_t *vm = vlib_get_main ();
376 vlib_node_t *node = vlib_get_node (vm, nr->node_index);
377
378 vnet_feature_next (sw_if_index, &next, b);
379 return node->next_nodes[next];
380}
