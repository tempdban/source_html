18#include <vnet/vnet.h>
19#include <vnet/api_errno.h>
20#include <vnet/ipsec/esp.h>
21
22u8 *
23format_esp_header (u8 * s, va_list * args)
24{
25 esp_header_t *esp = va_arg (*args, esp_header_t *);
26
27 s = format (s, "ESP: spi %u, seq %u",
28 clib_net_to_host_u32 (esp->spi),
29 clib_net_to_host_u32 (esp->seq));
30 return s;
31}
