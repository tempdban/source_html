69/* packet trace format function */
70static u8 *
71format_esp_decrypt_trace (u8 * s, va_list * args)
72{
73 CLIB_UNUSED (vlib_main_t * vm) = va_arg (*args, vlib_main_t *);
74 CLIB_UNUSED (vlib_node_t * node) = va_arg (*args, vlib_node_t *);
75 esp_decrypt_trace_t *t = va_arg (*args, esp_decrypt_trace_t *);
76
77 s = format (s, "esp: crypto %U integrity %U",
78 format_ipsec_crypto_alg, t->crypto_alg,
79 format_ipsec_integ_alg, t->integ_alg);
80 return s;
81}

83always_inline void
84esp_decrypt_cbc (ipsec_crypto_alg_t alg,
85 u8 * in, u8 * out, size_t in_len, u8 * key, u8 * iv)
86{
87 ipsec_proto_main_t *em = &ipsec_proto_main;
88 u32 thread_index = vlib_get_thread_index ();
89#if OPENSSL_VERSION_NUMBER >= 0x10100000L
90 EVP_CIPHER_CTX *ctx = em->per_thread_data[thread_index].decrypt_ctx;
91#else
92 EVP_CIPHER_CTX *ctx = &(em->per_thread_data[thread_index].decrypt_ctx);
93#endif
94 const EVP_CIPHER *cipher = NULL;
95 int out_len;
96
97 ASSERT (alg < IPSEC_CRYPTO_N_ALG);
98
99 if (PREDICT_FALSE (em->ipsec_proto_main_crypto_algs[alg].type == 0))
100 return;
101
102 if (PREDICT_FALSE
103 (alg != em->per_thread_data[thread_index].last_decrypt_alg))
104 {
105 cipher = em->ipsec_proto_main_crypto_algs[alg].type;
106 em->per_thread_data[thread_index].last_decrypt_alg = alg;
107 }
108
109 EVP_DecryptInit_ex (ctx, cipher, NULL, key, iv);
110
111 EVP_DecryptUpdate (ctx, out, &out_len, in, in_len);
112 EVP_DecryptFinal_ex (ctx, out + out_len, &out_len);
113}

115static uword
116esp_decrypt_node_fn (vlib_main_t * vm,
117 vlib_node_runtime_t * node, vlib_frame_t * from_frame)
118{
119 u32 n_left_from, *from, next_index, *to_next;
120 ipsec_main_t *im = &ipsec_main;
121 ipsec_proto_main_t *em = &ipsec_proto_main;
122 u32 *recycle = 0;
123 from = vlib_frame_vector_args (from_frame);
124 n_left_from = from_frame->n_vectors;
125 u32 thread_index = vlib_get_thread_index ();
126
127 ipsec_alloc_empty_buffers (vm, im);
128
129 u32 *empty_buffers = im->empty_buffers[thread_index];
130
131 if (PREDICT_FALSE (vec_len (empty_buffers) < n_left_from))
132 {
133 vlib_node_increment_counter (vm, esp_decrypt_node.index,
134 ESP_DECRYPT_ERROR_NO_BUFFER, n_left_from);
135 goto free_buffers_and_exit;
136 }
137
138 next_index = node->cached_next_index;
139
140 while (n_left_from > 0)
141 {
142 u32 n_left_to_next;
143
144 vlib_get_next_frame (vm, node, next_index, to_next, n_left_to_next);
145
146 while (n_left_from > 0 && n_left_to_next > 0)
147 {
148 u32 i_bi0, o_bi0 = (u32) ~ 0, next0;
149 vlib_buffer_t *i_b0;
150 vlib_buffer_t *o_b0 = 0;
151 esp_header_t *esp0;
152 ipsec_sa_t *sa0;
153 u32 sa_index0 = ~0;
154 u32 seq;
155 ip4_header_t *ih4 = 0, *oh4 = 0;
156 ip6_header_t *ih6 = 0, *oh6 = 0;
157 u8 tunnel_mode = 1;
158 u8 transport_ip6 = 0;
159
160
161 i_bi0 = from[0];
162 from += 1;
163 n_left_from -= 1;
164 n_left_to_next -= 1;
165
166 next0 = ESP_DECRYPT_NEXT_DROP;
167
168 i_b0 = vlib_get_buffer (vm, i_bi0);
169 esp0 = vlib_buffer_get_current (i_b0);
170
171 sa_index0 = vnet_buffer (i_b0)->ipsec.sad_index;
172 sa0 = pool_elt_at_index (im->sad, sa_index0);
173
174 seq = clib_host_to_net_u32 (esp0->seq);
175
176 /* anti-replay check */
177 if (sa0->use_anti_replay)
178 {
179 int rv = 0;
180
181 if (PREDICT_TRUE (sa0->use_esn))
182 rv = esp_replay_check_esn (sa0, seq);
183 else
184 rv = esp_replay_check (sa0, seq);
185
186 if (PREDICT_FALSE (rv))
187 {
188 clib_warning ("anti-replay SPI %u seq %u", sa0->spi, seq);
189 vlib_node_increment_counter (vm, esp_decrypt_node.index,
190 ESP_DECRYPT_ERROR_REPLAY, 1);
191 o_bi0 = i_bi0;
192 to_next[0] = o_bi0;
193 to_next += 1;
194 goto trace;
195 }
196 }
197
198 sa0->total_data_size += i_b0->current_length;
199
200 if (PREDICT_TRUE (sa0->integ_alg != IPSEC_INTEG_ALG_NONE))
201 {
202 u8 sig[64];
203 int icv_size =
204 em->ipsec_proto_main_integ_algs[sa0->integ_alg].trunc_size;
205 memset (sig, 0, sizeof (sig));
206 u8 *icv =
207 vlib_buffer_get_current (i_b0) + i_b0->current_length -
208 icv_size;
209 i_b0->current_length -= icv_size;
210
211 hmac_calc (sa0->integ_alg, sa0->integ_key, sa0->integ_key_len,
212 (u8 *) esp0, i_b0->current_length, sig, sa0->use_esn,
213 sa0->seq_hi);
214
215 if (PREDICT_FALSE (memcmp (icv, sig, icv_size)))
216 {
217 vlib_node_increment_counter (vm, esp_decrypt_node.index,
218 ESP_DECRYPT_ERROR_INTEG_ERROR,
219 1);
220 o_bi0 = i_bi0;
221 to_next[0] = o_bi0;
222 to_next += 1;
223 goto trace;
224 }
225 }
226
227 if (PREDICT_TRUE (sa0->use_anti_replay))
228 {
229 if (PREDICT_TRUE (sa0->use_esn))
230 esp_replay_advance_esn (sa0, seq);
231 else
232 esp_replay_advance (sa0, seq);
233 }
234
235 /* grab free buffer */
236 uword last_empty_buffer = vec_len (empty_buffers) - 1;
237 o_bi0 = empty_buffers[last_empty_buffer];
238 to_next[0] = o_bi0;
239 to_next += 1;
240 o_b0 = vlib_get_buffer (vm, o_bi0);
241 vlib_prefetch_buffer_with_index (vm,
242 empty_buffers[last_empty_buffer -
243 1], STORE);
244 _vec_len (empty_buffers) = last_empty_buffer;
245
246 /* add old buffer to the recycle list */
247 vec_add1 (recycle, i_bi0);
248
249 if ((sa0->crypto_alg >= IPSEC_CRYPTO_ALG_AES_CBC_128 &&
250 sa0->crypto_alg <= IPSEC_CRYPTO_ALG_AES_CBC_256) ||
251 (sa0->crypto_alg >= IPSEC_CRYPTO_ALG_DES_CBC &&
252 sa0->crypto_alg <= IPSEC_CRYPTO_ALG_3DES_CBC))
253 {
254 const int BLOCK_SIZE =
255 em->ipsec_proto_main_crypto_algs[sa0->crypto_alg].block_size;;
256 const int IV_SIZE =
257 em->ipsec_proto_main_crypto_algs[sa0->crypto_alg].iv_size;
258 esp_footer_t *f0;
259 u8 ip_hdr_size = 0;
260
261 int blocks =
262 (i_b0->current_length - sizeof (esp_header_t) -
263 IV_SIZE) / BLOCK_SIZE;
264
265 o_b0->current_data = sizeof (ethernet_header_t);
266
267 /* transport mode */
268 if (PREDICT_FALSE (!sa0->is_tunnel && !sa0->is_tunnel_ip6))
269 {
270 tunnel_mode = 0;
271 ih4 =
272 (ip4_header_t *) (i_b0->data +
273 sizeof (ethernet_header_t));
274 if (PREDICT_TRUE
275 ((ih4->ip_version_and_header_length & 0xF0) != 0x40))
276 {
277 if (PREDICT_TRUE
278 ((ih4->ip_version_and_header_length & 0xF0) ==
279 0x60))
280 {
281 transport_ip6 = 1;
282 ip_hdr_size = sizeof (ip6_header_t);
283 ih6 =
284 (ip6_header_t *) (i_b0->data +
285 sizeof (ethernet_header_t));
286 oh6 = vlib_buffer_get_current (o_b0);
287 }
288 else
289 {
290 vlib_node_increment_counter (vm,
291 esp_decrypt_node.index,
292 ESP_DECRYPT_ERROR_NOT_IP,
293 1);
294 o_b0 = 0;
295 goto trace;
296 }
297 }
298 else
299 {
300 oh4 = vlib_buffer_get_current (o_b0);
301 ip_hdr_size = sizeof (ip4_header_t);
302 }
303 }
304
305 esp_decrypt_cbc (sa0->crypto_alg,
306 esp0->data + IV_SIZE,
307 (u8 *) vlib_buffer_get_current (o_b0) +
308 ip_hdr_size, BLOCK_SIZE * blocks,
309 sa0->crypto_key, esp0->data);
310
311 o_b0->current_length = (blocks * BLOCK_SIZE) - 2 + ip_hdr_size;
312 o_b0->flags = VLIB_BUFFER_TOTAL_LENGTH_VALID;
313 f0 =
314 (esp_footer_t *) ((u8 *) vlib_buffer_get_current (o_b0) +
315 o_b0->current_length);
316 o_b0->current_length -= f0->pad_length;
317
318 /* tunnel mode */
319 if (PREDICT_TRUE (tunnel_mode))
320 {
321 if (PREDICT_TRUE (f0->next_header == IP_PROTOCOL_IP_IN_IP))
322 {
323 next0 = ESP_DECRYPT_NEXT_IP4_INPUT;
324 oh4 = vlib_buffer_get_current (o_b0);
325 }
326 else if (f0->next_header == IP_PROTOCOL_IPV6)
327 next0 = ESP_DECRYPT_NEXT_IP6_INPUT;
328 else
329 {
330 clib_warning ("next header: 0x%x", f0->next_header);
331 vlib_node_increment_counter (vm, esp_decrypt_node.index,
332 ESP_DECRYPT_ERROR_DECRYPTION_FAILED,
333 1);
334 o_b0 = 0;
335 goto trace;
336 }
337 }
338 /* transport mode */
339 else
340 {
341 if (PREDICT_FALSE (transport_ip6))
342 {
343 next0 = ESP_DECRYPT_NEXT_IP6_INPUT;
344 oh6->ip_version_traffic_class_and_flow_label =
345 ih6->ip_version_traffic_class_and_flow_label;
346 oh6->protocol = f0->next_header;
347 oh6->hop_limit = ih6->hop_limit;
348 oh6->src_address.as_u64[0] = ih6->src_address.as_u64[0];
349 oh6->src_address.as_u64[1] = ih6->src_address.as_u64[1];
350 oh6->dst_address.as_u64[0] = ih6->dst_address.as_u64[0];
351 oh6->dst_address.as_u64[1] = ih6->dst_address.as_u64[1];
352 oh6->payload_length =
353 clib_host_to_net_u16 (vlib_buffer_length_in_chain
354 (vm,
355 o_b0) - sizeof (ip6_header_t));
356 }
357 else
358 {
359 next0 = ESP_DECRYPT_NEXT_IP4_INPUT;
360 oh4->ip_version_and_header_length = 0x45;
361 oh4->tos = ih4->tos;
362 oh4->fragment_id = 0;
363 oh4->flags_and_fragment_offset = 0;
364 oh4->ttl = ih4->ttl;
365 oh4->protocol = f0->next_header;
366 oh4->src_address.as_u32 = ih4->src_address.as_u32;
367 oh4->dst_address.as_u32 = ih4->dst_address.as_u32;
368 oh4->length =
369 clib_host_to_net_u16 (vlib_buffer_length_in_chain
370 (vm, o_b0));
371 oh4->checksum = ip4_header_checksum (oh4);
372 }
373 }
374
375 /* for IPSec-GRE tunnel next node is ipsec-gre-input */
376 if (PREDICT_FALSE
377 ((vnet_buffer (i_b0)->ipsec.flags) &
378 IPSEC_FLAG_IPSEC_GRE_TUNNEL))
379 next0 = ESP_DECRYPT_NEXT_IPSEC_GRE_INPUT;
380
381 vnet_buffer (o_b0)->sw_if_index[VLIB_TX] = (u32) ~ 0;
382 vnet_buffer (o_b0)->sw_if_index[VLIB_RX] =
383 vnet_buffer (i_b0)->sw_if_index[VLIB_RX];
384 }
385
386 trace:
387 if (PREDICT_FALSE (i_b0->flags & VLIB_BUFFER_IS_TRACED))
388 {
389 if (o_b0)
390 {
391 o_b0->flags |= VLIB_BUFFER_IS_TRACED;
392 o_b0->trace_index = i_b0->trace_index;
393 esp_decrypt_trace_t *tr =
394 vlib_add_trace (vm, node, o_b0, sizeof (*tr));
395 tr->crypto_alg = sa0->crypto_alg;
396 tr->integ_alg = sa0->integ_alg;
397 }
398 }
399
400 vlib_validate_buffer_enqueue_x1 (vm, node, next_index, to_next,
401 n_left_to_next, o_bi0, next0);
402 }
403 vlib_put_next_frame (vm, node, next_index, n_left_to_next);
404 }
405 vlib_node_increment_counter (vm, esp_decrypt_node.index,
406 ESP_DECRYPT_ERROR_RX_PKTS,
407 from_frame->n_vectors);
408
409free_buffers_and_exit:
410 if (recycle)
411 vlib_buffer_free (vm, recycle, vec_len (recycle));
412 vec_free (recycle);
413 return from_frame->n_vectors;
414}
