38#ifndef included_clib_cache_h
39#define included_clib_cache_h
40
41#include <vppinfra/error_bootstrap.h>
42
43/*
44 * Allow CFLAGS to override the arch-specific cache line size
45 */
46#ifndef CLIB_LOG2_CACHE_LINE_BYTES
47
48/* Default cache line size of 64 bytes. */
49#ifndef CLIB_LOG2_CACHE_LINE_BYTES
50#define CLIB_LOG2_CACHE_LINE_BYTES 6
51#endif
52
53#endif /* CLIB_LOG2_CACHE_LINE_BYTES defined */
54
55#if (CLIB_LOG2_CACHE_LINE_BYTES >= 9)
56#error Cache line size 512 bytes or greater
57#endif
58
59#define CLIB_CACHE_LINE_BYTES (1 << CLIB_LOG2_CACHE_LINE_BYTES)
60#define CLIB_CACHE_LINE_ALIGN_MARK(mark) u8 mark[0] __attribute__((aligned(CLIB_CACHE_LINE_BYTES)))
61
62/* Read/write arguments to __builtin_prefetch. */
63#define CLIB_PREFETCH_READ 0
64#define CLIB_PREFETCH_LOAD 0 /* alias for read */
65#define CLIB_PREFETCH_WRITE 1
66#define CLIB_PREFETCH_STORE 1 /* alias for write */
67
68#define _CLIB_PREFETCH(n,size,type) \
69 if ((size) > (n)*CLIB_CACHE_LINE_BYTES) \
70 __builtin_prefetch (_addr + (n)*CLIB_CACHE_LINE_BYTES, \
71 CLIB_PREFETCH_##type, \
72 /* locality */ 3);
73
74#define CLIB_PREFETCH(addr,size,type) \
75do { \
76 void * _addr = (addr); \
77 \
78 ASSERT ((size) <= 4*CLIB_CACHE_LINE_BYTES); \
79 _CLIB_PREFETCH (0, size, type); \
80 _CLIB_PREFETCH (1, size, type); \
81 _CLIB_PREFETCH (2, size, type); \
82 _CLIB_PREFETCH (3, size, type); \
83} while (0)
84
85#undef _
86
87#endif /* included_clib_cache_h */
