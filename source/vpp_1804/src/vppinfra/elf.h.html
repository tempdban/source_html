957always_inline void *
958elf_get_section_contents (elf_main_t * em,
959 uword section_index, uword elt_size)
960{
961 elf_section_t *s;
962 void *result;
963
964 s = vec_elt_at_index (em->sections, section_index);
965
966 result = 0;
967 if (vec_len (s->contents) > 0)
968 {
969 /* Make vector copy of contents with given element size. */
970 result = _vec_resize (result,
971 vec_len (s->contents) / elt_size,
972 vec_len (s->contents),
973 /* header_bytes */ 0,
974 /* align */ 0);
975 clib_memcpy (result, s->contents, vec_len (s->contents));
976 }
977
978 return result;
979}
