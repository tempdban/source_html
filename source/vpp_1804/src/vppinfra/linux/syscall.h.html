16#ifndef included_linux_syscall_h
17#define included_linux_syscall_h
18
19#include <unistd.h>
20#include <sys/syscall.h>
21
22static inline long
23set_mempolicy (int mode, const unsigned long *nodemask, unsigned long maxnode)
24{
25 return syscall (__NR_set_mempolicy, mode, nodemask, maxnode);
26}
27
28static inline int
29get_mempolicy (int *mode, unsigned long *nodemask, unsigned long maxnode,
30 void *addr, unsigned long flags)
31{
32 return syscall (__NR_get_mempolicy, mode, nodemask, maxnode, addr, flags);
33}
34
35static inline long
36move_pages (int pid, unsigned long count, void **pages, const int *nodes,
37 int *status, int flags)
38{
39 return syscall (__NR_move_pages, pid, count, pages, nodes, status, flags);
40}

42#ifndef HAVE_MEMFD_CREATE
43static inline int
44memfd_create (const char *name, unsigned int flags)
45{
46 return syscall (__NR_memfd_create, name, flags);
47}
48#endif
