51#ifndef included_random_isaac_h
52#define included_random_isaac_h
53
54#include <vppinfra/clib.h> /* for u32/u64 */
55#include <vppinfra/format.h> /* for unformat_input_t */
56
57/* Bob recommends 8 for crypto, 4 for simulations */
58#define ISAAC_LOG2_SIZE (4)
59#define ISAAC_SIZE (1 << ISAAC_LOG2_SIZE)
60
61typedef struct
62{
63 uword memory[ISAAC_SIZE];
64 uword a, b, c;
65} isaac_t;
66
67void isaac (isaac_t * ctx, uword * results);
68void isaac2 (isaac_t * ctx, uword * results);
69void isaac_init (isaac_t * ctx, uword * results);
70
71int test_isaac_main (unformat_input_t * input);
72
73#endif /* included_random_isaac_h */
