38#ifndef included_clib_smp_h
39#define included_clib_smp_h
40
41#include <vppinfra/cache.h>
42#include <vppinfra/os.h> /* for os_panic */
43
44#define clib_smp_compare_and_swap(addr,new,old) __sync_val_compare_and_swap(addr,old,new)
45#define clib_smp_swap(addr,new) __sync_lock_test_and_set(addr,new)
46#define clib_smp_atomic_add(addr,increment) __sync_fetch_and_add(addr,increment)
47
48#if defined (i386) || defined (__x86_64__)
49#define clib_smp_pause() do { asm volatile ("pause"); } while (0)
50#elif defined (__aarch64__) || defined (__arm__)
51#define clib_smp_pause() do { asm volatile ("isb" ::: "memory"); } while (0)
52#endif
53
54#ifndef clib_smp_pause
55#define clib_smp_pause() do { } while (0)
56#endif
57
58#ifdef CLIB_UNIX
59#include <sched.h>
60
61always_inline void
62os_sched_yield (void)
63{
64 sched_yield ();
65}
66#else
67always_inline void
68os_sched_yield (void)
69{
70 clib_smp_pause ();
71}
72#endif

75#endif /* included_clib_smp_h */
