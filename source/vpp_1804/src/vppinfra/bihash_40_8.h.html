16#undef BIHASH_TYPE
17#undef BIHASH_KVP_CACHE_SIZE
18#undef BIHASH_KVP_PER_PAGE
19
20#define BIHASH_TYPE _40_8
21#define BIHASH_KVP_PER_PAGE 4
22#define BIHASH_KVP_CACHE_SIZE 2
23
24#ifndef __included_bihash_40_8_h__
25#define __included_bihash_40_8_h__
26
27#include <vppinfra/crc32.h>
28#include <vppinfra/heap.h>
29#include <vppinfra/format.h>
30#include <vppinfra/pool.h>
31#include <vppinfra/xxhash.h>
32
33typedef struct
34{
35 u64 key[5];
36 u64 value;
37} clib_bihash_kv_40_8_t;
38
39static inline int
40clib_bihash_is_free_40_8 (const clib_bihash_kv_40_8_t * v)
41{
42 /* Free values are memset to 0xff, check a bit... */
43 if (v->key[0] == ~0ULL && v->value == ~0ULL)
44 return 1;
45 return 0;
46}
47
48static inline u64
49clib_bihash_hash_40_8 (const clib_bihash_kv_40_8_t * v)
50{
51#ifdef clib_crc32c_uses_intrinsics
52 return clib_crc32c ((u8 *) v->key, 40);
53#else
54 u64 tmp = v->key[0] ^ v->key[1] ^ v->key[2] ^ v->key[3] ^ v->key[4];
55 return clib_xxhash (tmp);
56#endif
57}
58
59static inline u8 *
60format_bihash_kvp_40_8 (u8 * s, va_list * args)
61{
62 clib_bihash_kv_40_8_t *v = va_arg (*args, clib_bihash_kv_40_8_t *);
63
64 s = format (s, "key %llu %llu %llu %llu %llu value %llu", v->key[0],
65 v->key[1], v->key[2], v->key[3], v->key[4], v->value);
66 return s;
67}
68
69static inline int
70clib_bihash_key_compare_40_8 (const u64 * a, const u64 * b)
71{
72 return ((a[0] ^ b[0]) | (a[1] ^ b[1]) | (a[2] ^ b[2]) | (a[3] ^ b[3])
73 | (a[4] ^ b[4])) == 0;
74}
