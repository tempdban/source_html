41#include <vppinfra/format.h>
42#include <vppinfra/hash.h>
43#include <vppinfra/heap.h>
44
45/* Hash table plus vector of keys. */
46typedef struct
47{
48 /* Vector or heap used to store keys. Hash table stores keys as byte
49 offsets into this vector. */
50 u8 *key_vector_or_heap;
51
52 /* Byte offsets of free keys in vector (used to store free keys when
53 n_key_bytes > 1). */
54 u32 *key_vector_free_indices;
55
56 u8 **key_tmps;
57
58 /* Possibly fixed size of key.
59 0 means keys are vectors of u8's.
60 1 means keys are null terminated c strings. */
61#define MHASH_VEC_STRING_KEY 0
62#define MHASH_C_STRING_KEY 1
63 u32 n_key_bytes;
64
65 /* Seed value for Jenkins hash. */
66 u32 hash_seed;
67
68 /* Hash table mapping key -> value. */
69 uword *hash;
70
71 /* Format function for keys. */
72 format_function_t *format_key;
73} mhash_t;
74
75void mhash_init (mhash_t * h, uword n_value_bytes, uword n_key_bytes);
76
77always_inline void
78mhash_init_c_string (mhash_t * h, uword n_value_bytes)
79{
80 mhash_init (h, n_value_bytes, MHASH_C_STRING_KEY);
81}
82
83always_inline void
84mhash_init_vec_string (mhash_t * h, uword n_value_bytes)
85{
86 mhash_init (h, n_value_bytes, MHASH_VEC_STRING_KEY);
87}
88
89always_inline void *
90mhash_key_to_mem (mhash_t * h, uword key)
91{
92 if (key == ~0)
93 {
94 u8 *key_tmp;
95
96 int my_cpu = os_get_thread_index ();
97 vec_validate (h->key_tmps, my_cpu);
98 key_tmp = h->key_tmps[my_cpu];
99 return key_tmp;
100 }
101 return vec_elt_at_index (h->key_vector_or_heap, key);
102}
103
104hash_pair_t *mhash_get_pair (mhash_t * h, const void *key);
105uword mhash_set_mem (mhash_t * h, void *key, uword * new_value,
106 uword * old_value);
107uword mhash_unset (mhash_t * h, void *key, uword * old_value);
108
109always_inline uword *
110mhash_get (mhash_t * h, const void *key)
111{
112 hash_pair_t *p = mhash_get_pair (h, key);
113 return p ? &p->value[0] : 0;
114}
115
116always_inline uword
117mhash_set (mhash_t * h, void *key, uword new_value, uword * old_value)
118{
119 return mhash_set_mem (h, key, &new_value, old_value);
120}
121
122always_inline uword
123mhash_unset_key (mhash_t * h, uword key, uword * old_value)
124{
125 void *k = mhash_key_to_mem (h, key);
126 return mhash_unset (h, k, old_value);
127}
128
129always_inline uword
130mhash_value_bytes (mhash_t * m)
131{
132 hash_t *h = hash_header (m->hash);
133 return hash_value_bytes (h);
134}
135
136always_inline uword
137mhash_elts (mhash_t * m)
138{
139 return hash_elts (m->hash);
140}
141
142always_inline uword
143mhash_key_vector_is_heap (mhash_t * h)
144{
145 return h->n_key_bytes <= 1;
146}
147
148always_inline void
149mhash_free (mhash_t * h)
150{
151 if (mhash_key_vector_is_heap (h))
152 heap_free (h->key_vector_or_heap);
153 else
154 vec_free (h->key_vector_or_heap);
155 vec_free (h->key_vector_free_indices);
156 hash_free (h->hash);
157}

159#define mhash_foreach(k,v,mh,body) \
160do { \
161 hash_pair_t * _mhash_foreach_p; \
162 hash_foreach_pair (_mhash_foreach_p, (mh)->hash, ({ \
163 (k) = mhash_key_to_mem ((mh), _mhash_foreach_p->key); \
164 (v) = &_mhash_foreach_p->value[0]; \
165 body; \
166 })); \
167} while (0)
