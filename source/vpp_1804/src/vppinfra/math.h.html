38#ifndef included_math_h
39#define included_math_h
40
41#include <vppinfra/clib.h>
42
43always_inline f64
44sqrt (f64 x)
45{
46 return __builtin_sqrt (x);
47}
48
49always_inline f64
50fabs (f64 x)
51{
52 return __builtin_fabs (x);
53}
54
55#ifndef isnan
56#define isnan(x) __builtin_isnan(x)
57#endif
58
59#ifndef isinf
60#define isinf(x) __builtin_isinf(x)
61#endif
62
63#endif /* included_math_h */
