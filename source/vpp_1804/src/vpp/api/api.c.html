98static clib_error_t *
99memclnt_delete_callback (u32 client_index)
100{
101 vpe_api_main_t *vam = &vpe_api_main;
102 vpe_client_registration_t *rp;
103 uword *p;
104 int stats_memclnt_delete_callback (u32 client_index);
105
106 stats_memclnt_delete_callback (client_index);
107
108#define _(a) \
109 p = hash_get (vam->a##_registration_hash, client_index); \
110 if (p) { \
111 rp = pool_elt_at_index (vam->a##_registrations, p[0]); \
112 pool_put (vam->a##_registrations, rp); \
113 hash_unset (vam->a##_registration_hash, client_index); \
114 }
115 foreach_registration_hash;
116#undef _
117 return 0;
118}

136static void
137shmem_cli_output (uword arg, u8 * buffer, uword buffer_bytes)
138{
139 u8 **shmem_vecp = (u8 **) arg;
140 u8 *shmem_vec;
141 void *oldheap;
142 api_main_t *am = &api_main;
143 u32 offset;
144
145 shmem_vec = *shmem_vecp;
146
147 offset = vec_len (shmem_vec);
148
149 pthread_mutex_lock (&am->vlib_rp->mutex);
150 oldheap = svm_push_data_heap (am->vlib_rp);
151
152 vec_validate (shmem_vec, offset + buffer_bytes - 1);
153
154 clib_memcpy (shmem_vec + offset, buffer, buffer_bytes);
155
156 svm_pop_heap (oldheap);
157 pthread_mutex_unlock (&am->vlib_rp->mutex);
158
159 *shmem_vecp = shmem_vec;
160}

163static void
164vl_api_cli_t_handler (vl_api_cli_t * mp)
165{
166 vl_api_cli_reply_t *rp;
167 vl_api_registration_t *reg;
168 vlib_main_t *vm = vlib_get_main ();
169 api_main_t *am = &api_main;
170 unformat_input_t input;
171 u8 *shmem_vec = 0;
172 void *oldheap;
173
174 reg = vl_api_client_index_to_registration (mp->client_index);
175 if (!reg)
176 return;;
177
178 rp = vl_msg_api_alloc (sizeof (*rp));
179 rp->_vl_msg_id = ntohs (VL_API_CLI_REPLY);
180 rp->context = mp->context;
181
182 unformat_init_vector (&input, (u8 *) (uword) mp->cmd_in_shmem);
183
184 vlib_cli_input (vm, &input, shmem_cli_output, (uword) & shmem_vec);
185
186 pthread_mutex_lock (&am->vlib_rp->mutex);
187 oldheap = svm_push_data_heap (am->vlib_rp);
188
189 vec_add1 (shmem_vec, 0);
190
191 svm_pop_heap (oldheap);
192 pthread_mutex_unlock (&am->vlib_rp->mutex);
193
194 rp->reply_in_shmem = (uword) shmem_vec;
195
196 vl_api_send_msg (reg, (u8 *) rp);
197}

281static void
282vl_api_get_next_index_t_handler (vl_api_get_next_index_t * mp)
283{
284 vlib_main_t *vm = vlib_get_main ();
285 vl_api_get_next_index_reply_t *rmp;
286 vlib_node_t *node, *next_node;
287 int rv = 0;
288 u32 next_node_index = ~0, next_index = ~0;
289 uword *p;
290
291 node = vlib_get_node_by_name (vm, mp->node_name);
292
293 if (node == 0)
294 {
295 rv = VNET_API_ERROR_NO_SUCH_NODE;
296 goto out;
297 }
298
299 next_node = vlib_get_node_by_name (vm, mp->next_name);
300
301 if (next_node == 0)
302 {
303 rv = VNET_API_ERROR_NO_SUCH_NODE2;
304 goto out;
305 }
306 else
307 next_node_index = next_node->index;
308
309 p = hash_get (node->next_slot_by_node, next_node_index);
310
311 if (p == 0)
312 {
313 rv = VNET_API_ERROR_NO_SUCH_ENTRY;
314 goto out;
315 }
316 else
317 next_index = p[0];
318
319out:
320 /* *INDENT-OFF* */
321 REPLY_MACRO2(VL_API_GET_NEXT_INDEX_REPLY,
322 ({
323 rmp->next_index = ntohl(next_index);
324 }));
325 /* *INDENT-ON* */
326}

398#define BOUNCE_HANDLER(nn) \
399static void vl_api_##nn##_t_handler ( \
400 vl_api_##nn##_t *mp) \
401{ \
402 vpe_client_registration_t *reg; \
403 vpe_api_main_t * vam = &vpe_api_main; \
404 svm_queue_t * q; \
405 \
406 /* One registration only... */ \
407 pool_foreach(reg, vam->nn##_registrations, \
408 ({ \
409 q = vl_api_client_index_to_input_queue (reg->client_index); \
410 if (q) { \
411 /* \
412 * If the queue is stuffed, turf the msg and complain \
413 * It's unlikely that the intended recipient is \
414 * alive; avoid deadlock at all costs. \
415 */ \
416 if (q->cursize == q->maxsize) { \
417 clib_warning ("ERROR: receiver queue full, drop msg"); \
418 vl_msg_api_free (mp); \
419 return; \
420 } \
421 vl_msg_api_send_shmem (q, (u8 *)&mp); \
422 return; \
423 } \
424 })); \
425 vl_msg_api_free (mp); \
426}

428static void setup_message_id_table (api_main_t * am);
429
430/*
431 * vpe_api_hookup
432 * Add vpe's API message handlers to the table.
433 * vlib has alread mapped shared memory and
434 * added the client registration handlers.
435 * See .../open-repo/vlib/memclnt_vlib.c:memclnt_process()
436 */
437static clib_error_t *
438vpe_api_hookup (vlib_main_t * vm)
439{
440 api_main_t *am = &api_main;
441
442#define _(N,n) \
443 vl_msg_api_set_handlers(VL_API_##N, #n, \
444 vl_api_##n##_t_handler, \
445 vl_noop_handler, \
446 vl_api_##n##_t_endian, \
447 vl_api_##n##_t_print, \
448 sizeof(vl_api_##n##_t), 1);
449 foreach_vpe_api_msg;
450#undef _
451
452 /*
453 * Trace space for classifier mask+match
454 */
455 am->api_trace_cfg[VL_API_CLASSIFY_ADD_DEL_TABLE].size += 5 * sizeof (u32x4);
456 am->api_trace_cfg[VL_API_CLASSIFY_ADD_DEL_SESSION].size
457 += 5 * sizeof (u32x4);
458
459 /*
460 * Thread-safe API messages
461 */
462 am->is_mp_safe[VL_API_IP_ADD_DEL_ROUTE] = 1;
463 am->is_mp_safe[VL_API_GET_NODE_GRAPH] = 1;
464
465 /*
466 * Set up the (msg_name, crc, message-id) table
467 */
468 setup_message_id_table (am);
469
470 return 0;
471}

496static clib_error_t *
497api_segment_config (vlib_main_t * vm, unformat_input_t * input)
498{
499 u8 *chroot_path;
500 u64 baseva, size, pvt_heap_size;
501 int uid, gid, rv;
502 const int max_buf_size = 4096;
503 char *s, *buf;
504 struct passwd _pw, *pw;
505 struct group _grp, *grp;
506 clib_error_t *e;
507 buf = vec_new (char, 128);
508 while (unformat_check_input (input) != UNFORMAT_END_OF_INPUT)
509 {
510 if (unformat (input, "prefix %s", &chroot_path))
511 {
512 vec_add1 (chroot_path, 0);
513 vl_set_memory_root_path ((char *) chroot_path);
514 }
515 else if (unformat (input, "uid %d", &uid))
516 vl_set_memory_uid (uid);
517 else if (unformat (input, "gid %d", &gid))
518 vl_set_memory_gid (gid);
519 else if (unformat (input, "baseva %llx", &baseva))
520 vl_set_global_memory_baseva (baseva);
521 else if (unformat (input, "global-size %lldM", &size))
522 vl_set_global_memory_size (size * (1ULL << 20));
523 else if (unformat (input, "global-size %lldG", &size))
524 vl_set_global_memory_size (size * (1ULL << 30));
525 else if (unformat (input, "global-size %lld", &size))
526 vl_set_global_memory_size (size);
527 else if (unformat (input, "global-pvt-heap-size %lldM", &pvt_heap_size))
528 vl_set_global_pvt_heap_size (pvt_heap_size * (1ULL << 20));
529 else if (unformat (input, "global-pvt-heap-size size %lld",
530 &pvt_heap_size))
531 vl_set_global_pvt_heap_size (pvt_heap_size);
532 else if (unformat (input, "api-pvt-heap-size %lldM", &pvt_heap_size))
533 vl_set_api_pvt_heap_size (pvt_heap_size * (1ULL << 20));
534 else if (unformat (input, "api-pvt-heap-size size %lld",
535 &pvt_heap_size))
536 vl_set_api_pvt_heap_size (pvt_heap_size);
537 else if (unformat (input, "api-size %lldM", &size))
538 vl_set_api_memory_size (size * (1ULL << 20));
539 else if (unformat (input, "api-size %lldG", &size))
540 vl_set_api_memory_size (size * (1ULL << 30));
541 else if (unformat (input, "api-size %lld", &size))
542 vl_set_api_memory_size (size);
543 else if (unformat (input, "uid %s", &s))
544 {
545 /* lookup the username */
546 pw = NULL;
547 while (((rv =
548 getpwnam_r (s, &_pw, buf, vec_len (buf), &pw)) == ERANGE)
549 && (vec_len (buf) <= max_buf_size))
550 {
551 vec_resize (buf, vec_len (buf) * 2);
552 }
553 if (rv < 0)
554 {
555 e = clib_error_return_code (0, rv,
556 CLIB_ERROR_ERRNO_VALID |
557 CLIB_ERROR_FATAL,
558 "cannot fetch username %s", s);
559 vec_free (s);
560 vec_free (buf);
561 return e;
562 }
563 if (pw == NULL)
564 {
565 e =
566 clib_error_return_fatal (0, "username %s does not exist", s);
567 vec_free (s);
568 vec_free (buf);
569 return e;
570 }
571 vec_free (s);
572 vl_set_memory_uid (pw->pw_uid);
573 }
574 else if (unformat (input, "gid %s", &s))
575 {
576 /* lookup the group name */
577 grp = NULL;
578 while (((rv =
579 getgrnam_r (s, &_grp, buf, vec_len (buf), &grp)) == ERANGE)
580 && (vec_len (buf) <= max_buf_size))
581 {
582 vec_resize (buf, vec_len (buf) * 2);
583 }
584 if (rv != 0)
585 {
586 e = clib_error_return_code (0, rv,
587 CLIB_ERROR_ERRNO_VALID |
588 CLIB_ERROR_FATAL,
589 "cannot fetch group %s", s);
590 vec_free (s);
591 vec_free (buf);
592 return e;
593 }
594 if (grp == NULL)
595 {
596 e = clib_error_return_fatal (0, "group %s does not exist", s);
597 vec_free (s);
598 vec_free (buf);
599 return e;
600 }
601 vec_free (s);
602 vec_free (buf);
603 vl_set_memory_gid (grp->gr_gid);
604 }
605 else
606 return clib_error_return (0, "unknown input `%U'",
607 format_unformat_error, input);
608 }
609 return 0;
610}

640static clib_error_t *
641show_ip_arp_nd_events_fn (vlib_main_t * vm,
642 unformat_input_t * input, vlib_cli_command_t * cmd)
643{
644 vpe_api_main_t *am = &vpe_api_main;
645 vl_api_ip4_arp_event_t *arp_event;
646 vl_api_ip6_nd_event_t *nd_event;
647
648 if (pool_elts (am->arp_events) == 0 && pool_elts (am->nd_events) == 0 &&
649 pool_elts (am->wc_ip4_arp_events_registrations) == 0 &&
650 pool_elts (am->wc_ip6_nd_events_registrations) == 0)
651 {
652 vlib_cli_output (vm, "No active arp or nd event registrations");
653 return 0;
654 }
655
656 /* *INDENT-OFF* */
657 pool_foreach (arp_event, am->arp_events,
658 ({
659 vlib_cli_output (vm, "%U", format_arp_event, arp_event);
660 }));
661
662 vpe_client_registration_t *reg;
663 pool_foreach(reg, am->wc_ip4_arp_events_registrations,
664 ({
665 vlib_cli_output (vm, "pid %d: bd mac/ip4 binding events",
666 ntohl (reg->client_pid));
667 }));
668
