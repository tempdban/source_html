18/** \file
19
20 2 x CDP graph nodes: an "interior" node to process
21 incoming announcements, and a "process" node to periodically
22 send announcements.
23
24 The interior node is neither pipelined nor dual-looped, because
25 it would be very unusual to see more than one CDP packet in
26 a given input frame. So, it's a very simple / straighforward
27 example.
28*/
29
30/*
31 * packet counter strings
32 * Dump these counters via the "show error" CLI command
33 */
34static char *cdp_error_strings[] = {
35#define _(sym,string) string,
36 foreach_cdp_error
37#undef _
38};
39
40/*
41 * We actually send all cdp pkts to the "error" node after scanning
42 * them, so the graph node has only one next-index. The "error-drop"
43 * node automatically bumps our per-node packet counters for us.
44 */
45typedef enum
46{
47 CDP_INPUT_NEXT_NORMAL,
48 CDP_INPUT_N_NEXT,
49} cdp_next_t;
50
51/*
52 * Process a frame of cdp packets
53 * Expect 1 packet / frame
54 */
55static uword
56cdp_node_fn (vlib_main_t * vm,
57 vlib_node_runtime_t * node, vlib_frame_t * frame)
58{
59 u32 n_left_from, *from;
60 cdp_input_trace_t *t0;
61
62 from = vlib_frame_vector_args (frame); /* array of buffer indices */
63 n_left_from = frame->n_vectors; /* number of buffer indices */
64
65 while (n_left_from > 0)
66 {
67 u32 bi0;
68 vlib_buffer_t *b0;
69 u32 next0, error0;
70
71 bi0 = from[0];
72 b0 = vlib_get_buffer (vm, bi0);
73
74 next0 = CDP_INPUT_NEXT_NORMAL;
75
76 /* scan this cdp pkt. error0 is the counter index to bump */
77 error0 = cdp_input (vm, b0, bi0);
78 b0->error = node->errors[error0];
79
80 /* If this pkt is traced, snapshoot the data */
81 if (b0->flags & VLIB_BUFFER_IS_TRACED)
82 {
83 int len;
84 t0 = vlib_add_trace (vm, node, b0, sizeof (*t0));
85 len = (b0->current_length < sizeof (t0->data))
86 ? b0->current_length : sizeof (t0->data);
87 t0->len = len;
88 clib_memcpy (t0->data, vlib_buffer_get_current (b0), len);
89 }
90 /* push this pkt to the next graph node, always error-drop */
91 vlib_set_next_frame_buffer (vm, node, next0, bi0);
92
93 from += 1;
94 n_left_from -= 1;
95 }
96
97 return frame->n_vectors;
98}

101 * cdp input graph node declaration

103/* *INDENT-OFF* */
104VLIB_REGISTER_NODE (cdp_input_node, static) = {
105 .function = cdp_node_fn,
106 .name = "cdp-input",
107 .vector_size = sizeof (u32),
108 .type = VLIB_NODE_TYPE_INTERNAL,
109
110 .n_errors = CDP_N_ERROR,
111 .error_strings = cdp_error_strings,
112
113 .format_trace = cdp_input_format_trace,
114
115 .n_next_nodes = CDP_INPUT_N_NEXT,
116 .next_nodes = {
117 [CDP_INPUT_NEXT_NORMAL] = "error-drop",
118 },
119};

122/*
123 * cdp periodic function
124 */
125static uword
126cdp_process (vlib_main_t * vm, vlib_node_runtime_t * rt, vlib_frame_t * f)
127{
128 cdp_main_t *cm = &cdp_main;
129 f64 poll_time_remaining;
130 uword event_type, *event_data = 0;
131
132 /* So we can send events to the cdp process */
133 cm->cdp_process_node_index = cdp_process_node.index;
134
135 /* Dynamically register the cdp input node with the snap classifier */
136 snap_register_input_protocol (vm, "cdp-input", 0xC /* ieee_oui, Cisco */ ,
137 0x2000 /* protocol CDP */ ,
138 cdp_input_node.index);
139
140 snap_register_input_protocol (vm, "cdp-input", 0xC /* ieee_oui, Cisco */ ,
141 0x2004 /* protocol CDP */ ,
142 cdp_input_node.index);
143
144#if 0 /* retain for reference */
145 /* with the hdlc classifier */
146 hdlc_register_input_protocol (vm, HDLC_PROTOCOL_cdp, cdp_input_node.index);
147#endif
148
149 /* with ethernet input (for SRP) */
150 ethernet_register_input_type (vm, ETHERNET_TYPE_CDP /* CDP */ ,
151 cdp_input_node.index);
152
153 /* Start w/ cdp effectively disabled */
154 poll_time_remaining = 86400.0;
155
156 while (1)
157 {
158 /* sleep until next poll time, or msg serialize event occurs */
159 poll_time_remaining =
160 vlib_process_wait_for_event_or_clock (vm, poll_time_remaining);
161
162 event_type = vlib_process_get_events (vm, &event_data);
163 switch (event_type)
164 {
165 case ~0: /* no events => timeout */
166 break;
167
168 case CDP_EVENT_ENABLE:
169 poll_time_remaining = 10.0;
170 break;
171
172 case CDP_EVENT_DISABLE:
173 poll_time_remaining = 86400.0;
174 break;
175
176 default:
177 clib_warning ("BUG: event type 0x%wx", event_type);
178 break;
179 }
180 vec_reset_length (event_data);
181
182 /* peer timeout scan, send announcements */
183 if (vlib_process_suspend_time_is_zero (poll_time_remaining))
184 {
185 cdp_periodic (vm);
186 poll_time_remaining = 10.0;
187 }
188 }
189
190 return 0;
191}
