104static uword
105unformat_memif_queues (unformat_input_t * input, va_list * args)
106{
107 u32 *rx_queues = va_arg (*args, u32 *);
108 u32 *tx_queues = va_arg (*args, u32 *);
109
110 if (unformat (input, "rx-queues %u", rx_queues))
111 ;
112 if (unformat (input, "tx-queues %u", tx_queues))
113 ;
114
115 return 1;
116}

118/* memif_socket_filename_add_del API */
119static int
120api_memif_socket_filename_add_del (vat_main_t * vam)
121{
122 unformat_input_t *i = vam->input;
123 vl_api_memif_socket_filename_add_del_t *mp;
124 u8 is_add;
125 u32 socket_id;
126 u8 *socket_filename;
127 int ret;
128
129 is_add = 1;
130 socket_id = ~0;
131 socket_filename = 0;
132
133 while (unformat_check_input (i) != UNFORMAT_END_OF_INPUT)
134 {
135 if (unformat (i, "id %u", &socket_id))
136 ;
137 else if (unformat (i, "filename %s", &socket_filename))
138 ;
139 else if (unformat (i, "del"))
140 is_add = 0;
141 else if (unformat (i, "add"))
142 is_add = 1;
143 else
144 {
145 vec_free (socket_filename);
146 clib_warning ("unknown input `%U'", format_unformat_error, i);
147 return -99;
148 }
149 }
150
151 if (socket_id == 0 || socket_id == ~0)
152 {
153 vec_free (socket_filename);
154 errmsg ("Invalid socket id");
155 return -99;
156 }
157
158 if (is_add && (!socket_filename || *socket_filename == 0))
159 {
160 vec_free (socket_filename);
161 errmsg ("Invalid socket filename");
162 return -99;
163 }
164
165 M (MEMIF_SOCKET_FILENAME_ADD_DEL, mp);
166
167 mp->is_add = is_add;
168 mp->socket_id = htonl (socket_id);
169 strncpy ((char *) mp->socket_filename,
170 (char *) socket_filename, sizeof (mp->socket_filename) - 1);
171
172 vec_free (socket_filename);
173
174 S (mp);
175 W (ret);
176
177 return ret;
178}

192/* memif-create API */
193static int
194api_memif_create (vat_main_t * vam)
195{
196 unformat_input_t *i = vam->input;
197 vl_api_memif_create_t *mp;
198 u32 id = 0;
199 u32 socket_id = 0;
200 u8 *secret = 0;
201 u8 role = 1;
202 u32 ring_size = 0;
203 u32 buffer_size = 0;
204 u8 hw_addr[6] = { 0 };
205 u32 rx_queues = MEMIF_DEFAULT_RX_QUEUES;
206 u32 tx_queues = MEMIF_DEFAULT_TX_QUEUES;
207 int ret;
208 u8 mode = MEMIF_INTERFACE_MODE_ETHERNET;
209
210 while (unformat_check_input (i) != UNFORMAT_END_OF_INPUT)
211 {
212 if (unformat (i, "id %u", &id))
213 ;
214 else if (unformat (i, "socket-id %u", &socket_id))
215 ;
216 else if (unformat (i, "secret %s", &secret))
217 ;
218 else if (unformat (i, "ring_size %u", &ring_size))
219 ;
220 else if (unformat (i, "buffer_size %u", &buffer_size))
221 ;
222 else if (unformat (i, "master"))
223 role = 0;
224 else if (unformat (i, "slave %U",
225 unformat_memif_queues, &rx_queues, &tx_queues))
226 role = 1;
227 else if (unformat (i, "mode ip"))
228 mode = MEMIF_INTERFACE_MODE_IP;
229 else if (unformat (i, "hw_addr %U", unformat_ethernet_address, hw_addr))
230 ;
231 else
232 {
233 clib_warning ("unknown input '%U'", format_unformat_error, i);
234 return -99;
235 }
236 }
237
238 if (socket_id == ~0)
239 {
240 errmsg ("invalid socket-id\n");
241 return -99;
242 }
243
244 if (!is_pow2 (ring_size))
245 {
246 errmsg ("ring size must be power of 2\n");
247 return -99;
248 }
249
250 if (rx_queues > 255 || rx_queues < 1)
251 {
252 errmsg ("rx queue must be between 1 - 255\n");
253 return -99;
254 }
255
256 if (tx_queues > 255 || tx_queues < 1)
257 {
258 errmsg ("tx queue must be between 1 - 255\n");
259 return -99;
260 }
261
262 M (MEMIF_CREATE, mp);
263
264 mp->mode = mode;
265 mp->id = clib_host_to_net_u32 (id);
266 mp->role = role;
267 mp->ring_size = clib_host_to_net_u32 (ring_size);
268 mp->buffer_size = clib_host_to_net_u16 (buffer_size & 0xffff);
269 mp->socket_id = clib_host_to_net_u32 (socket_id);
270 if (secret != 0)
271 {
272 strncpy ((char *) mp->secret, (char *) secret, 16);
273 vec_free (secret);
274 }
275 memcpy (mp->hw_addr, hw_addr, 6);
276 mp->rx_queues = rx_queues;
277 mp->tx_queues = tx_queues;
278
279 S (mp);
280 W (ret);
281 return ret;
282}

302/* memif-delete API */
303static int
304api_memif_delete (vat_main_t * vam)
305{
306 unformat_input_t *i = vam->input;
307 vl_api_memif_delete_t *mp;
308 u32 sw_if_index = 0;
309 u8 index_defined = 0;
310 int ret;
311
312 while (unformat_check_input (i) != UNFORMAT_END_OF_INPUT)
313 {
314 if (unformat (i, "sw_if_index %u", &sw_if_index))
315 index_defined = 1;
316 else
317 {
318 clib_warning ("unknown input '%U'", format_unformat_error, i);
319 return -99;
320 }
321 }
322
323 if (!index_defined)
324 {
325 errmsg ("missing sw_if_index\n");
326 return -99;
327 }
328
329 M (MEMIF_DELETE, mp);
330
331 mp->sw_if_index = clib_host_to_net_u32 (sw_if_index);
332
333 S (mp);
334 W (ret);
335 return ret;
336}

338/* memif-dump API */
339static int
340api_memif_dump (vat_main_t * vam)
341{
342 memif_test_main_t *mm = &memif_test_main;
343 vl_api_memif_dump_t *mp;
344 vl_api_control_ping_t *mp_ping;
345 int ret;
346
347 if (vam->json_output)
348 {
349 clib_warning ("JSON output not supported for memif_dump");
350 return -99;
351 }
352
353 M (MEMIF_DUMP, mp);
354 S (mp);
355
356 /* Use a control ping for synchronization */
357 mp_ping = vl_msg_api_alloc_as_if_client (sizeof (*mp_ping));
358 mp_ping->_vl_msg_id = htons (mm->ping_id);
359 mp_ping->client_index = vam->my_client_index;
360
361 fformat (vam->ofp, "Sending ping id=%d\n", mm->ping_id);
362
363 vam->result_ready = 0;
364 S (mp_ping);
365
366 W (ret);
367 return ret;
368}

370/* memif-details message handler */
371static void
372vl_api_memif_details_t_handler (vl_api_memif_details_t * mp)
373{
374 vat_main_t *vam = memif_test_main.vat_main;
375
376 fformat (vam->ofp, "%s: sw_if_index %u mac %U\n"
377 " id %u socket-id %u role %s\n"
378 " ring_size %u buffer_size %u\n"
379 " state %s link %s\n",
380 mp->if_name, ntohl (mp->sw_if_index), format_ethernet_address,
381 mp->hw_addr, clib_net_to_host_u32 (mp->id),
382 clib_net_to_host_u32 (mp->socket_id),
383 mp->role ? "slave" : "master",
384 ntohl (mp->ring_size), ntohs (mp->buffer_size),
385 mp->admin_up_down ? "up" : "down",
386 mp->link_up_down ? "up" : "down");
387}

433/*
434 * List of messages that the api test plugin sends,
435 * and that the data plane plugin processes
436 */
437#define foreach_vpe_api_msg \
438_(memif_create, "[id <id>] [socket-id <id>] [ring_size <size>] " \
439 "[buffer_size <size>] [hw_addr <mac_address>] " \
440 "[secret <string>] [mode ip] <master|slave>") \
441_(memif_delete, "<sw_if_index>") \
442_(memif_dump, "") \
443_(memif_socket_filename_dump, "") \
444_(memif_socket_filename_add_del, "[add|del] id <id> filename <file>")
445
446static void
447memif_vat_api_hookup (vat_main_t * vam)
448{
449 memif_test_main_t *mm __attribute__ ((unused)) = &memif_test_main;
450 /* Hook up handlers for replies from the data plane plug-in */
451#define _(N,n) \
452 vl_msg_api_set_handlers((VL_API_##N + mm->msg_id_base), \
453 #n, \
454 vl_api_##n##_t_handler, \
455 vl_noop_handler, \
456 vl_api_##n##_t_endian, \
457 vl_api_##n##_t_print, \
458 sizeof(vl_api_##n##_t), 1);
459 foreach_vpe_api_reply_msg;
460#undef _
461
462 /* API messages we can send */
463#define _(n,h) \
464 hash_set_mem (vam->function_by_name, #n, api_##n);
465 foreach_vpe_api_msg;
466#undef _
467
468 /* Help strings */
469#define _(n,h) hash_set_mem (vam->help_by_name, #n, h);
470 foreach_vpe_api_msg;
471#undef _
472}

474clib_error_t *
475vat_plugin_register (vat_main_t * vam)
476{
477 memif_test_main_t *mm = &memif_test_main;
478 u8 *name;
479
480 mm->vat_main = vam;
481
482 /* Ask the vpp engine for the first assigned message-id */
483 name = format (0, "memif_%08x%c", api_version, 0);
484 mm->msg_id_base = vl_client_get_first_plugin_msg_id ((char *) name);
485
486 /* Get the control ping ID */
487#define _(id,n,crc) \
488 const char *id ## _CRC __attribute__ ((unused)) = #n "_" #crc;
489 foreach_vl_msg_name_crc_vpe;
490#undef _
491 mm->ping_id = vl_msg_api_get_msg_index ((u8 *) (VL_API_CONTROL_PING_CRC));
492
493 if (mm->msg_id_base != (u16) ~ 0)
494 memif_vat_api_hookup (vam);
495
496 vec_free (name);
497
498 return 0;
499}
