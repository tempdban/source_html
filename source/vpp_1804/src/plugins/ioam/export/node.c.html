24typedef struct
25{
26 u32 next_index;
27 u32 flow_label;

30/* packet trace format function */
31static u8 *
32format_export_trace (u8 * s, va_list * args)
33{
34 CLIB_UNUSED (vlib_main_t * vm) = va_arg (*args, vlib_main_t *);
35 CLIB_UNUSED (vlib_node_t * node) = va_arg (*args, vlib_node_t *);
36 export_trace_t *t = va_arg (*args, export_trace_t *);
37
38 s = format (s, "EXPORT: flow_label %d, next index %d",
39 t->flow_label, t->next_index);
40 return s;
41}

125static void
126ip6_export_fixup_func (vlib_buffer_t * export_buf, vlib_buffer_t * pak_buf)
127{
128 ip6_header_t *ip6_temp =
129 (ip6_header_t *) (export_buf->data + export_buf->current_length);
130 u32 flow_label_temp =
131 clib_net_to_host_u32(ip6_temp->ip_version_traffic_class_and_flow_label)
132 & 0xFFF00000;
133 flow_label_temp |=
134 IOAM_MASK_DECAP_BIT((vnet_buffer(pak_buf)->l2_classify.opaque_index));
135 ip6_temp->ip_version_traffic_class_and_flow_label =
136 clib_host_to_net_u32(flow_label_temp);
137}
138
139static uword
140ip6_export_node_fn (vlib_main_t * vm,
141 vlib_node_runtime_t * node, vlib_frame_t * frame)
142{
143 ioam_export_main_t *em = &ioam_export_main;
144 ioam_export_node_common(em, vm, node, frame, ip6_header_t, payload_length,
145 ip_version_traffic_class_and_flow_label,
146 EXPORT_NEXT_POP_HBYH, ip6_export_fixup_func);
147 return frame->n_vectors;
148}
