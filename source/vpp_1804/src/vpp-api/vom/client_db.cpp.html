16#include "vom/client_db.hpp"
17
18namespace VOM {
19object_ref_list&
20client_db::find(const client_db::key_t& k)
21{
22 return (m_objs[k]);
23}
24
25void
26client_db::flush(const client_db::key_t& k)
27{
28 auto found = m_objs.find(k);
29
30 if (found != m_objs.end())
31 m_objs.erase(found);
32}
33
34void
35client_db::dump(const key_t& key, std::ostream& os)
36{
37 object_ref_list& orlist = find(key);
38
39 for (auto entry : orlist) {
40 os << " " << entry.obj()->to_string() << std::endl;
41 }
42}
43
44void
45client_db::dump(std::ostream& os)
46{
47 for (auto entry : m_objs) {
48 os << " key:[" << entry.first << "]" << std::endl;
49 }
50}
51}
