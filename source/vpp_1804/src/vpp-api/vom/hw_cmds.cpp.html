16#include "vom/hw_cmds.hpp"
17
18namespace VOM {
19namespace hw_cmds {
20
21poll::poll(HW::item<bool>& item)
22 : rpc_cmd(item)
23{
24}
25
26rc_t
27poll::issue(connection& con)
28{
29 msg_t req(con.ctx(), std::ref(*this));
30
31 VAPI_CALL(req.execute());
32
33 m_hw_item.set(wait());
34
35 return (rc_t::OK);
36}
37
38std::string
39poll::to_string() const
40{
41 std::ostringstream s;
42
43 s << "poll: " << m_hw_item.to_string();
44
45 return (s.str());
46}
47}
48};
