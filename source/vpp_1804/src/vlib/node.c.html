43/* Query node given name. */
44vlib_node_t *
45vlib_get_node_by_name (vlib_main_t * vm, u8 * name)
46{
47 vlib_node_main_t *nm = &vm->node_main;
48 uword *p;
49 u8 *key = name;
50 if (!clib_mem_is_heap_object (key))
51 key = format (0, "%s", key);
52 p = hash_get (nm->node_by_name, key);
53 if (key != name)
54 vec_free (key);
55 return p ? vec_elt (nm->nodes, p[0]) : 0;
56}

58static void
59node_set_elog_name (vlib_main_t * vm, uword node_index)
60{
61 vlib_node_t *n = vlib_get_node (vm, node_index);
62 elog_event_type_t *t;
63
64 t = vec_elt_at_index (vm->node_call_elog_event_types, node_index);
65 vec_free (t->format);
66 t->format = (char *) format (0, "%v-call: %%d%c", n->name, 0);
67
68 t = vec_elt_at_index (vm->node_return_elog_event_types, node_index);
69 vec_free (t->format);
70 t->format = (char *) format (0, "%v-return: %%d%c", n->name, 0);
71
72 n->name_elog_string = elog_string (&vm->elog_main, "%v%c", n->name, 0);
73}

75static void
76vlib_worker_thread_node_rename (u32 node_index)
77{
78 int i;
79 vlib_main_t *vm;
80 vlib_node_t *n;
81
82 if (vec_len (vlib_mains) == 1)
83 return;
84
85 vm = vlib_mains[0];
86 n = vlib_get_node (vm, node_index);
87
88 ASSERT (vlib_get_thread_index () == 0);
89 ASSERT (*vlib_worker_threads->wait_at_barrier == 1);
90
91 for (i = 1; i < vec_len (vlib_mains); i++)
92 {
93 vlib_main_t *vm_worker = vlib_mains[i];
94 vlib_node_t *n_worker = vlib_get_node (vm_worker, node_index);
95
96 n_worker->name = n->name;
97 n_worker->name_elog_string = n->name_elog_string;
98 }
99}

101void
102vlib_node_rename (vlib_main_t * vm, u32 node_index, char *fmt, ...)
103{
104 va_list va;
105 vlib_node_main_t *nm = &vm->node_main;
106 vlib_node_t *n = vlib_get_node (vm, node_index);
107
108 va_start (va, fmt);
109 hash_unset (nm->node_by_name, n->name);
110 vec_free (n->name);
111 n->name = va_format (0, fmt, &va);
112 va_end (va);
113 hash_set (nm->node_by_name, n->name, n->index);
114
115 node_set_elog_name (vm, node_index);
116
117 /* Propagate the change to all worker threads */
118 vlib_worker_thread_node_rename (node_index);
119}

121static void
122vlib_node_runtime_update (vlib_main_t * vm, u32 node_index, u32 next_index)
123{
124 vlib_node_main_t *nm = &vm->node_main;
125 vlib_node_runtime_t *r, *s;
126 vlib_node_t *node, *next_node;
127 vlib_next_frame_t *nf;
128 vlib_pending_frame_t *pf;
129 i32 i, j, n_insert;
130
131 ASSERT (vlib_get_thread_index () == 0);
132
133 vlib_worker_thread_barrier_sync (vm);
134
135 node = vec_elt (nm->nodes, node_index);
136 r = vlib_node_get_runtime (vm, node_index);
137
138 n_insert = vec_len (node->next_nodes) - r->n_next_nodes;
139 if (n_insert > 0)
140 {
141 i = r->next_frame_index + r->n_next_nodes;
142 vec_insert (nm->next_frames, n_insert, i);
143
144 /* Initialize newly inserted next frames. */
145 for (j = 0; j < n_insert; j++)
146 vlib_next_frame_init (nm->next_frames + i + j);
147
148 /* Relocate other next frames at higher indices. */
149 for (j = 0; j < vec_len (nm->nodes); j++)
150 {
151 s = vlib_node_get_runtime (vm, j);
152 if (j != node_index && s->next_frame_index >= i)
153 s->next_frame_index += n_insert;
154 }
155
156 /* Pending frames may need to be relocated also. */
157 vec_foreach (pf, nm->pending_frames)
158 {
159 if (pf->next_frame_index != VLIB_PENDING_FRAME_NO_NEXT_FRAME
160 && pf->next_frame_index >= i)
161 pf->next_frame_index += n_insert;
162 }
163 /* *INDENT-OFF* */
164 pool_foreach (pf, nm->suspended_process_frames, ({
165 if (pf->next_frame_index != ~0 && pf->next_frame_index >= i)
166 pf->next_frame_index += n_insert;
167 }));
168 /* *INDENT-ON* */
169
170 r->n_next_nodes = vec_len (node->next_nodes);
171 }
172
173 /* Set frame's node runtime index. */
174 next_node = vlib_get_node (vm, node->next_nodes[next_index]);
175 nf = nm->next_frames + r->next_frame_index + next_index;
176 nf->node_runtime_index = next_node->runtime_index;
177
178 vlib_worker_thread_node_runtime_update ();
179
180 vlib_worker_thread_barrier_release (vm);
181}

183uword
184vlib_node_get_next (vlib_main_t * vm, uword node_index, uword next_node_index)
185{
186 vlib_node_main_t *nm = &vm->node_main;
187 vlib_node_t *node;
188 uword *p;
189
190 node = vec_elt (nm->nodes, node_index);
191
192 /* Runtime has to be initialized. */
193 ASSERT (nm->flags & VLIB_NODE_MAIN_RUNTIME_STARTED);
194
195 if ((p = hash_get (node->next_slot_by_node, next_node_index)))
196 {
197 return p[0];
198 }
199
200 return (~0);
201}

203/* Add next node to given node in given slot. */
204uword
205vlib_node_add_next_with_slot (vlib_main_t * vm,
206 uword node_index,
207 uword next_node_index, uword slot)
208{
209 vlib_node_main_t *nm = &vm->node_main;
210 vlib_node_t *node, *next;
211 uword *p;
212
213 node = vec_elt (nm->nodes, node_index);
214 next = vec_elt (nm->nodes, next_node_index);
215
216 /* Runtime has to be initialized. */
217 ASSERT (nm->flags & VLIB_NODE_MAIN_RUNTIME_STARTED);
218
219 if ((p = hash_get (node->next_slot_by_node, next_node_index)))
220 {
221 /* Next already exists: slot must match. */
222 if (slot != ~0)
223 ASSERT (slot == p[0]);
224 return p[0];
225 }
226
227 if (slot == ~0)
228 slot = vec_len (node->next_nodes);
229
230 vec_validate_init_empty (node->next_nodes, slot, ~0);
231 vec_validate (node->n_vectors_by_next_node, slot);
232
233 node->next_nodes[slot] = next_node_index;
234 hash_set (node->next_slot_by_node, next_node_index, slot);
235
236 vlib_node_runtime_update (vm, node_index, slot);
237
238 next->prev_node_bitmap = clib_bitmap_ori (next->prev_node_bitmap,
239 node_index);
240
241 /* Siblings all get same node structure. */
242 {
243 uword sib_node_index, sib_slot;
244 vlib_node_t *sib_node;
245 /* *INDENT-OFF* */
246 clib_bitmap_foreach (sib_node_index, node->sibling_bitmap, ({
247 sib_node = vec_elt (nm->nodes, sib_node_index);
248 if (sib_node != node)
249 {
250 sib_slot = vlib_node_add_next_with_slot (vm, sib_node_index, next_node_index, slot);
251 ASSERT (sib_slot == slot);
252 }
253 }));
254 /* *INDENT-ON* */
255 }
256
257 return slot;
258}

260/* Add named next node to given node in given slot. */
261uword
262vlib_node_add_named_next_with_slot (vlib_main_t * vm,
263 uword node, char *name, uword slot)
264{
265 vlib_node_main_t *nm;
266 vlib_node_t *n, *n_next;
267
268 nm = &vm->node_main;
269 n = vlib_get_node (vm, node);
270
271 n_next = vlib_get_node_by_name (vm, (u8 *) name);
272 if (!n_next)
273 {
274 if (nm->flags & VLIB_NODE_MAIN_RUNTIME_STARTED)
275 return ~0;
276
277 if (slot == ~0)
278 slot = clib_max (vec_len (n->next_node_names),
279 vec_len (n->next_nodes));
280 vec_validate (n->next_node_names, slot);
281 n->next_node_names[slot] = name;
282 return slot;
283 }
284
285 return vlib_node_add_next_with_slot (vm, node, n_next->index, slot);
286}

288static void
289node_elog_init (vlib_main_t * vm, uword ni)
290{
291 elog_event_type_t t;
292
293 memset (&t, 0, sizeof (t));
294
295 /* 2 event types for this node: one when node function is called.
296 One when it returns. */
297 vec_validate (vm->node_call_elog_event_types, ni);
298 vm->node_call_elog_event_types[ni] = t;
299
300 vec_validate (vm->node_return_elog_event_types, ni);
301 vm->node_return_elog_event_types[ni] = t;
302
303 node_set_elog_name (vm, ni);
304}

312static void
313register_node (vlib_main_t * vm, vlib_node_registration_t * r)
314{
315 vlib_node_main_t *nm = &vm->node_main;
316 vlib_node_t *n;
317 u32 page_size = clib_mem_get_page_size ();
318 int i;
319
320 if (CLIB_DEBUG > 0)
321 {
322 /* Default (0) type should match INTERNAL. */
323 vlib_node_t zero = { 0 };
324 ASSERT (VLIB_NODE_TYPE_INTERNAL == zero.type);
325 }
326
327 ASSERT (r->function != 0);
328
329 n = clib_mem_alloc_no_fail (sizeof (n[0]));
330 memset (n, 0, sizeof (n[0]));
331 n->index = vec_len (nm->nodes);
332
333 vec_add1 (nm->nodes, n);
334
335 /* Name is always a vector so it can be formatted with %v. */
336 if (clib_mem_is_heap_object (vec_header (r->name, 0)))
337 n->name = vec_dup ((u8 *) r->name);
338 else
339 n->name = format (0, "%s", r->name);
340
341 if (!nm->node_by_name)
342 nm->node_by_name = hash_create_vec ( /* size */ 32,
343 sizeof (n->name[0]), sizeof (uword));
344
345 /* Node names must be unique. */
346 {
347 vlib_node_t *o = vlib_get_node_by_name (vm, n->name);
348 if (o)
349 clib_error ("more than one node named `%v'", n->name);
350 }
351
352 hash_set (nm->node_by_name, n->name, n->index);
353
354 r->index = n->index; /* save index in registration */
355 n->function = r->function;
356
357 /* Node index of next sibling will be filled in by vlib_node_main_init. */
358 n->sibling_of = r->sibling_of;
359 if (r->sibling_of && r->n_next_nodes > 0)
360 clib_error ("sibling node should not have any next nodes `%v'", n->name);
361
362 if (r->type == VLIB_NODE_TYPE_INTERNAL)
363 ASSERT (r->vector_size > 0);
364
365#define _(f) n->f = r->f
366
367 _(type);
368 _(flags);
369 _(state);
370 _(scalar_size);
371 _(vector_size);
372 _(format_buffer);
373 _(unformat_buffer);
374 _(format_trace);
375 _(validate_frame);
376
377 /* Register error counters. */
378 vlib_register_errors (vm, n->index, r->n_errors, r->error_strings);
379 node_elog_init (vm, n->index);
380
381 _(runtime_data_bytes);
382 if (r->runtime_data_bytes > 0)
383 {
384 vec_resize (n->runtime_data, r->runtime_data_bytes);
385 if (r->runtime_data)
386 clib_memcpy (n->runtime_data, r->runtime_data, r->runtime_data_bytes);
387 }
388
389 vec_resize (n->next_node_names, r->n_next_nodes);
390 for (i = 0; i < r->n_next_nodes; i++)
391 n->next_node_names[i] = r->next_nodes[i];
392
393 vec_validate_init_empty (n->next_nodes, r->n_next_nodes - 1, ~0);
394 vec_validate (n->n_vectors_by_next_node, r->n_next_nodes - 1);
395
396 n->owner_node_index = n->owner_next_index = ~0;
397
398 /* Initialize node runtime. */
399 {
400 vlib_node_runtime_t *rt;
401 u32 i;
402
403 if (n->type == VLIB_NODE_TYPE_PROCESS)
404 {
405 vlib_process_t *p;
406 uword log2_n_stack_bytes;
407
408 log2_n_stack_bytes = clib_max (r->process_log2_n_stack_bytes, 15);
409
410#ifdef CLIB_UNIX
411 /*
412 * Bump the stack size if running over a kernel with a large page size,
413 * and the stack isn't any too big to begin with. Otherwise, we'll
414 * trip over the stack guard page for sure.
415 */
416 if ((page_size > (4 << 10)) && log2_n_stack_bytes < 19)
417 {
418 if ((1 << log2_n_stack_bytes) <= page_size)
419 log2_n_stack_bytes = min_log2 (page_size) + 1;
420 else
421 log2_n_stack_bytes++;
422 }
423#endif
424
425 p = clib_mem_alloc_aligned_at_offset
426 (sizeof (p[0]) + (1 << log2_n_stack_bytes),
427 STACK_ALIGN, STRUCT_OFFSET_OF (vlib_process_t, stack),
428 0 /* no, don't call os_out_of_memory */ );
429 if (p == 0)
430 clib_panic ("failed to allocate process stack (%d bytes)",
431 1 << log2_n_stack_bytes);
432
433 memset (p, 0, sizeof (p[0]));
434 p->log2_n_stack_bytes = log2_n_stack_bytes;
435
436 /* Process node's runtime index is really index into process
437 pointer vector. */
438 n->runtime_index = vec_len (nm->processes);
439
440 vec_add1 (nm->processes, p);
441
442 /* Paint first stack word with magic number so we can at least
443 detect process stack overruns. */
444 p->stack[0] = VLIB_PROCESS_STACK_MAGIC;
445
446 /* Node runtime is stored inside of process. */
447 rt = &p->node_runtime;
448
449#ifdef CLIB_UNIX
450 /*
451 * Disallow writes to the bottom page of the stack, to
452 * catch stack overflows.
453 */
454 if (mprotect (p->stack, page_size, PROT_READ) < 0)
455 clib_unix_warning ("process stack");
456#endif
457
458 }
459 else
460 {
461 vec_add2_aligned (nm->nodes_by_type[n->type], rt, 1,
462 /* align */ CLIB_CACHE_LINE_BYTES);
463 n->runtime_index = rt - nm->nodes_by_type[n->type];
464 }
465
466 if (n->type == VLIB_NODE_TYPE_INPUT)
467 nm->input_node_counts_by_state[n->state] += 1;
468
469 rt->function = n->function;
470 rt->flags = n->flags;
471 rt->state = n->state;
472 rt->node_index = n->index;
473
474 rt->n_next_nodes = r->n_next_nodes;
475 rt->next_frame_index = vec_len (nm->next_frames);
476
477 vec_resize (nm->next_frames, rt->n_next_nodes);
478 for (i = 0; i < rt->n_next_nodes; i++)
479 vlib_next_frame_init (nm->next_frames + rt->next_frame_index + i);
480
481 vec_resize (rt->errors, r->n_errors);
482 for (i = 0; i < vec_len (rt->errors); i++)
483 rt->errors[i] = vlib_error_set (n->index, i);
484
485 STATIC_ASSERT_SIZEOF (vlib_node_runtime_t, 128);
486 ASSERT (vec_len (n->runtime_data) <= VLIB_NODE_RUNTIME_DATA_SIZE);
487
488 if (vec_len (n->runtime_data) > 0)
489 clib_memcpy (rt->runtime_data, n->runtime_data,
490 vec_len (n->runtime_data));
491
492 vec_free (n->runtime_data);
493 }
494}

504static uword
505null_node_fn (vlib_main_t * vm,
506 vlib_node_runtime_t * node, vlib_frame_t * frame)
507{
508 u16 n_vectors = frame->n_vectors;
509
510 vlib_node_increment_counter (vm, node->node_index, 0, n_vectors);
511 vlib_buffer_free (vm, vlib_frame_args (frame), n_vectors);
512 vlib_frame_free (vm, node, frame);
513
514 return n_vectors;
515}

517void
518vlib_register_all_static_nodes (vlib_main_t * vm)
519{
520 vlib_node_registration_t *r;
521
522 static char *null_node_error_strings[] = {
523 "blackholed packets",
524 };
525
526 static vlib_node_registration_t null_node_reg = {
527 .function = null_node_fn,
528 .vector_size = sizeof (u32),
529 .name = "null-node",
530 .n_errors = 1,
531 .error_strings = null_node_error_strings,
532 };
533
534 /* make sure that node index 0 is not used by
535 real node */
536 register_node (vm, &null_node_reg);
537
538 r = vm->node_main.node_registrations;
539 while (r)
540 {
541 register_node (vm, r);
542 r = r->next_registration;
543 }
544}

546vlib_node_t ***
547vlib_node_get_nodes (vlib_main_t * vm, u32 max_threads, int include_stats)
548{
549 vlib_node_main_t *nm = &vm->node_main;
550 vlib_node_t *n;
551 static vlib_node_t ***node_dups;
552 vlib_node_t **nodes;
553 static vlib_main_t **stat_vms;
554 vlib_main_t *stat_vm;
555 uword i, j;
556 u32 threads_to_serialize;
557
558 vec_reset_length (node_dups);
559
560 if (vec_len (stat_vms) == 0)
561 {
562 for (i = 0; i < vec_len (vlib_mains); i++)
563 {
564 stat_vm = vlib_mains[i];
565 if (stat_vm)
566 vec_add1 (stat_vms, stat_vm);
567 }
568 }
569
570 threads_to_serialize = clib_min (max_threads, vec_len (stat_vms));
571
572 /*
573 * Barrier sync across stats scraping.
574 * Otherwise, the counts will be grossly inaccurate.
575 */
576 vlib_worker_thread_barrier_sync (vm);
577
578 for (j = 0; j < threads_to_serialize; j++)
579 {
580 stat_vm = stat_vms[j];
581 nm = &stat_vm->node_main;
582
583 if (include_stats)
584 {
585 for (i = 0; i < vec_len (nm->nodes); i++)
586 {
587 n = nm->nodes[i];
588 vlib_node_sync_stats (stat_vm, n);
589 }
590 }
591
592 nodes = vec_dup (nm->nodes);
593 vec_add1 (node_dups, nodes);
594 }
595 vlib_worker_thread_barrier_release (vm);
596
597 return node_dups;
598}

600clib_error_t *
601vlib_node_main_init (vlib_main_t * vm)
602{
603 vlib_node_main_t *nm = &vm->node_main;
604 clib_error_t *error = 0;
605 vlib_node_t *n;
606 uword ni;
607
608 nm->frame_size_hash = hash_create (0, sizeof (uword));
609 nm->flags |= VLIB_NODE_MAIN_RUNTIME_STARTED;
610
611 /* Generate sibling relationships */
612 {
613 vlib_node_t *n, *sib;
614 uword si;
615
616 for (ni = 0; ni < vec_len (nm->nodes); ni++)
617 {
618 n = vec_elt (nm->nodes, ni);
619
620 if (!n->sibling_of)
621 continue;
622
623 sib = vlib_get_node_by_name (vm, (u8 *) n->sibling_of);
624 if (!sib)
625 {
626 error = clib_error_create ("sibling `%s' not found for node `%v'",
627 n->sibling_of, n->name);
628 goto done;
629 }
630
631 /* *INDENT-OFF* */
632 clib_bitmap_foreach (si, sib->sibling_bitmap, ({
633 vlib_node_t * m = vec_elt (nm->nodes, si);
634
635 /* Connect all of sibling's siblings to us. */
636 m->sibling_bitmap = clib_bitmap_ori (m->sibling_bitmap, n->index);
637
638 /* Connect us to all of sibling's siblings. */
639 n->sibling_bitmap = clib_bitmap_ori (n->sibling_bitmap, si);
640 }));
641 /* *INDENT-ON* */
642
643 /* Connect sibling to us. */
644 sib->sibling_bitmap = clib_bitmap_ori (sib->sibling_bitmap, n->index);
645
646 /* Connect us to sibling. */
647 n->sibling_bitmap = clib_bitmap_ori (n->sibling_bitmap, sib->index);
648 }
649 }
650
651 /* Resolve next names into next indices. */
652 for (ni = 0; ni < vec_len (nm->nodes); ni++)
653 {
654 uword i;
655
656 n = vec_elt (nm->nodes, ni);
657
658 for (i = 0; i < vec_len (n->next_node_names); i++)
659 {
660 char *a = n->next_node_names[i];
661
662 if (!a)
663 continue;
664
665 if (~0 == vlib_node_add_named_next_with_slot (vm, n->index, a, i))
666 {
667 error = clib_error_create
668 ("node `%v' refers to unknown node `%s'", n->name, a);
669 goto done;
670 }
671 }
672
673 vec_free (n->next_node_names);
674 }
675
676 /* Set previous node pointers. */
677 for (ni = 0; ni < vec_len (nm->nodes); ni++)
678 {
679 vlib_node_t *n_next;
680 uword i;
681
682 n = vec_elt (nm->nodes, ni);
683
684 for (i = 0; i < vec_len (n->next_nodes); i++)
685 {
686 if (n->next_nodes[i] >= vec_len (nm->nodes))
687 continue;
688
689 n_next = vec_elt (nm->nodes, n->next_nodes[i]);
690 n_next->prev_node_bitmap =
691 clib_bitmap_ori (n_next->prev_node_bitmap, n->index);
692 }
693 }
694
695 {
696 vlib_next_frame_t *nf;
697 vlib_node_runtime_t *r;
698 vlib_node_t *next;
699 uword i;
700
701 vec_foreach (r, nm->nodes_by_type[VLIB_NODE_TYPE_INTERNAL])
702 {
703 if (r->n_next_nodes == 0)
704 continue;
705
706 n = vlib_get_node (vm, r->node_index);
707 nf = vec_elt_at_index (nm->next_frames, r->next_frame_index);
708
709 for (i = 0; i < vec_len (n->next_nodes); i++)
710 {
711 next = vlib_get_node (vm, n->next_nodes[i]);
712
713 /* Validate node runtime indices are correctly initialized. */
714 ASSERT (nf[i].node_runtime_index == next->runtime_index);
715
716 nf[i].flags = 0;
717 if (next->flags & VLIB_NODE_FLAG_FRAME_NO_FREE_AFTER_DISPATCH)
718 nf[i].flags |= VLIB_FRAME_NO_FREE_AFTER_DISPATCH;
719 }
720 }
721 }
722
723done:
724 return error;
725}
