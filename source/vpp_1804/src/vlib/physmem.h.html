40#ifndef included_vlib_physmem_h
41#define included_vlib_physmem_h
42
43typedef u8 vlib_physmem_region_index_t;
44
45typedef struct
46{
47 vlib_physmem_region_index_t index;
48 void *mem;
49 uword size;
50 int fd;
51 u8 log2_page_size;
52 u16 n_pages;
53 u32 page_mask;
54
55 void *heap;
56 u32 flags;
57#define VLIB_PHYSMEM_F_INIT_MHEAP (1 << 0)
58#define VLIB_PHYSMEM_F_HUGETLB (1 << 1)
59#define VLIB_PHYSMEM_F_SHARED (1 << 2)
60
61 u8 numa_node;
62 u64 *page_table;
63 u8 *name;
64} vlib_physmem_region_t;
65
66
67
68typedef struct
69{
70 u32 flags;
71#define VLIB_PHYSMEM_MAIN_F_HAVE_PAGEMAP (1 << 0)
72#define VLIB_PHYSMEM_MAIN_F_HAVE_IOMMU (1 << 1)
73 vlib_physmem_region_t *regions;
74} vlib_physmem_main_t;
75
76extern vlib_physmem_main_t physmem_main;
77
78#endif /* included_vlib_physmem_h */
