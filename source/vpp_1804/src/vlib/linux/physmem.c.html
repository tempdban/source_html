56static void *
57unix_physmem_alloc_aligned (vlib_main_t * vm, vlib_physmem_region_index_t idx,
58 uword n_bytes, uword alignment)
59{
60 vlib_physmem_region_t *pr = vlib_physmem_get_region (vm, idx);
61 uword lo_offset, hi_offset;
62 uword *to_free = 0;
63
64 if (pr->heap == 0)
65 return 0;
66
67 /* IO memory is always at least cache aligned. */
68 alignment = clib_max (alignment, CLIB_CACHE_LINE_BYTES);
69
70 while (1)
71 {
72 mheap_get_aligned (pr->heap, n_bytes,
73 /* align */ alignment,
74 /* align offset */ 0,
75 &lo_offset);
76
77 /* Allocation failed? */
78 if (lo_offset == ~0)
79 break;
80
81 /* Make sure allocation does not span DMA physical chunk boundary. */
82 hi_offset = lo_offset + n_bytes - 1;
83
84 if (((pointer_to_uword (pr->heap) + lo_offset) >> pr->log2_page_size) ==
85 ((pointer_to_uword (pr->heap) + hi_offset) >> pr->log2_page_size))
86 break;
87
88 /* Allocation would span chunk boundary, queue it to be freed as soon as
89 we find suitable chunk. */
90 vec_add1 (to_free, lo_offset);
91 }
92
93 if (to_free != 0)
94 {
95 uword i;
96 for (i = 0; i < vec_len (to_free); i++)
97 mheap_put (pr->heap, to_free[i]);
98 vec_free (to_free);
99 }
100
101 return lo_offset != ~0 ? pr->heap + lo_offset : 0;
102}

104static void
105unix_physmem_free (vlib_main_t * vm, vlib_physmem_region_index_t idx, void *x)
106{
107 vlib_physmem_region_t *pr = vlib_physmem_get_region (vm, idx);
108 /* Return object to region's heap. */
109 mheap_put (pr->heap, x - pr->heap);
110}

112static clib_error_t *
113unix_physmem_region_alloc (vlib_main_t * vm, char *name, u32 size,
114 u8 numa_node, u32 flags,
115 vlib_physmem_region_index_t * idx)
116{
117 vlib_physmem_main_t *vpm = &physmem_main;
118 vlib_physmem_region_t *pr;
119 clib_error_t *error = 0;
120 clib_mem_vm_alloc_t alloc = { 0 };
121 int i;
122
123 pool_get (vpm->regions, pr);
124
125 if ((pr - vpm->regions) >= 256)
126 {
127 error = clib_error_return (0, "maximum number of regions reached");
128 goto error;
129 }
130
131 alloc.name = name;
132 alloc.size = size;
133 alloc.numa_node = numa_node;
134
135 alloc.flags = (flags & VLIB_PHYSMEM_F_SHARED) ?
136 CLIB_MEM_VM_F_SHARED : CLIB_MEM_VM_F_LOCKED;
137
138 if ((flags & VLIB_PHYSMEM_F_HUGETLB))
139 {
140 alloc.flags |= CLIB_MEM_VM_F_HUGETLB;
141 alloc.flags |= CLIB_MEM_VM_F_HUGETLB_PREALLOC;
142 alloc.flags |= CLIB_MEM_VM_F_NUMA_FORCE;
143 }
144 else
145 {
146 alloc.flags |= CLIB_MEM_VM_F_NUMA_PREFER;
147 }
148
149 error = clib_mem_vm_ext_alloc (&alloc);
150 if (error)
151 goto error;
152
153 pr->index = pr - vpm->regions;
154 pr->flags = flags;
155 pr->fd = alloc.fd;
156 pr->mem = alloc.addr;
157 pr->log2_page_size = alloc.log2_page_size;
158 pr->n_pages = alloc.n_pages;
159 pr->size = (u64) pr->n_pages << (u64) pr->log2_page_size;
160 pr->page_mask = (1 << pr->log2_page_size) - 1;
161 pr->numa_node = numa_node;
162 pr->name = format (0, "%s", name);
163
164 for (i = 0; i < pr->n_pages; i++)
165 {
166 void *ptr = pr->mem + ((u64) i << pr->log2_page_size);
167 int node;
168 if ((move_pages (0, 1, &ptr, 0, &node, 0) == 0) && (numa_node != node))
169 {
170 clib_warning ("physmem page for region \'%s\' allocated on the"
171 " wrong numa node (requested %u actual %u)",
172 pr->name, pr->numa_node, node, i);
173 break;
174 }
175 }
176
177 pr->page_table = clib_mem_vm_get_paddr (pr->mem, pr->log2_page_size,
178 pr->n_pages);
179
180 linux_vfio_dma_map_regions (vm);
181
182 if (flags & VLIB_PHYSMEM_F_INIT_MHEAP)
183 {
184 pr->heap = mheap_alloc_with_flags (pr->mem, pr->size,
185 /* Don't want mheap mmap/munmap with IO memory. */
186 MHEAP_FLAG_DISABLE_VM |
187 MHEAP_FLAG_THREAD_SAFE);
188 }
189
190 *idx = pr->index;
191
192 goto done;
193
194error:
195 memset (pr, 0, sizeof (*pr));
196 pool_put (vpm->regions, pr);
197
198done:
199 return error;
200}

202static void
203unix_physmem_region_free (vlib_main_t * vm, vlib_physmem_region_index_t idx)
204{
205 vlib_physmem_main_t *vpm = &physmem_main;
206 vlib_physmem_region_t *pr = vlib_physmem_get_region (vm, idx);
207
208 if (pr->fd > 0)
209 close (pr->fd);
210 munmap (pr->mem, pr->size);
211 vec_free (pr->name);
212 pool_put (vpm->regions, pr);
213}

215clib_error_t *
216unix_physmem_init (vlib_main_t * vm)
217{
218 vlib_physmem_main_t *vpm = &physmem_main;
219 clib_error_t *error = 0;
220 u64 *pt = 0;
221
222 /* Avoid multiple calls. */
223 if (vm->os_physmem_alloc_aligned)
224 return error;
225
226 /* check if pagemap is accessible */
227 pt = clib_mem_vm_get_paddr (&pt, min_log2 (sysconf (_SC_PAGESIZE)), 1);
228 if (pt[0])
229 vpm->flags |= VLIB_PHYSMEM_MAIN_F_HAVE_PAGEMAP;
230 vec_free (pt);
231
232 if ((error = linux_vfio_init (vm)))
233 return error;
234
235 vm->os_physmem_alloc_aligned = unix_physmem_alloc_aligned;
236 vm->os_physmem_free = unix_physmem_free;
237 vm->os_physmem_region_alloc = unix_physmem_region_alloc;
238 vm->os_physmem_region_free = unix_physmem_region_free;
239
240 return error;
241}

243static clib_error_t *
244show_physmem (vlib_main_t * vm,
245 unformat_input_t * input, vlib_cli_command_t * cmd)
246{
247 vlib_physmem_main_t *vpm = &physmem_main;
248 vlib_physmem_region_t *pr;
249
250 /* *INDENT-OFF* */
251 pool_foreach (pr, vpm->regions, (
252 {
253 vlib_cli_output (vm, "index %u name '%s' page-size %uKB num-pages %d "
254 "numa-node %u fd %d\n",
255 pr->index, pr->name, (1 << (pr->log2_page_size -10)),
256 pr->n_pages, pr->numa_node, pr->fd);
257 if (pr->heap)
258 vlib_cli_output (vm, " %U", format_mheap, pr->heap, /* verbose */ 1);
259 else
260 vlib_cli_output (vm, " no heap\n");
261 }));
262 /* *INDENT-ON* */
263 return 0;
264}

268 .path = "show physmem",
269 .short_help = "Show physical memory allocation",
270 .function = show_physmem,
271};
