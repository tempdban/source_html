164#define IXGBE_WRITE_REG_THEN_POLL_MASK(hw, reg, val, mask, poll_ms) \
165do { \
166 uint32_t cnt = poll_ms; \
167 IXGBE_WRITE_REG(hw, (reg), (val)); \
168 while (((IXGBE_READ_REG(hw, (reg))) & (mask)) && (cnt--)) \
169 rte_delay_ms(1); \
170} while (0)
