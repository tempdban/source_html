5#include <assert.h>

12static STAILQ_HEAD(, svf_entry) head = STAILQ_HEAD_INITIALIZER(head);
13
14void
15nicvf_bsvf_push(struct svf_entry *entry)
16{
17 assert(entry != NULL);
18 assert(entry->vf != NULL);
19
20 STAILQ_INSERT_TAIL(&head, entry, next);
21}
22
23struct svf_entry *
24nicvf_bsvf_pop(void)
25{
26 struct svf_entry *entry;
27
28 assert(!STAILQ_EMPTY(&head));
29
30 entry = STAILQ_FIRST(&head);
31
32 assert(entry != NULL);
33 assert(entry->vf != NULL);
34
35 STAILQ_REMOVE_HEAD(&head, next);
36
37 return entry;
38}
39
40int
41nicvf_bsvf_empty(void)
42{
43 return STAILQ_EMPTY(&head);
44}
