34#ifndef _AVF_ALLOC_H_
35#define _AVF_ALLOC_H_
36
37struct avf_hw;
38
39/* Memory allocation types */
40enum avf_memory_type {
41 avf_mem_arq_buf = 0, /* ARQ indirect command buffer */
42 avf_mem_asq_buf = 1,
43 avf_mem_atq_buf = 2, /* ATQ indirect command buffer */
44 avf_mem_arq_ring = 3, /* ARQ descriptor ring */
45 avf_mem_atq_ring = 4, /* ATQ descriptor ring */
46 avf_mem_pd = 5, /* Page Descriptor */
47 avf_mem_bp = 6, /* Backing Page - 4KB */
48 avf_mem_bp_jumbo = 7, /* Backing Page - > 4KB */
49 avf_mem_reserved
50};
51
52/* prototype for functions used for dynamic memory allocation */
53enum avf_status_code avf_allocate_dma_mem(struct avf_hw *hw,
54 struct avf_dma_mem *mem,
55 enum avf_memory_type type,
56 u64 size, u32 alignment);
57enum avf_status_code avf_free_dma_mem(struct avf_hw *hw,
58 struct avf_dma_mem *mem);
59enum avf_status_code avf_allocate_virt_mem(struct avf_hw *hw,
60 struct avf_virt_mem *mem,
61 u32 size);
62enum avf_status_code avf_free_virt_mem(struct avf_hw *hw,
63 struct avf_virt_mem *mem);
64
65#endif /* _AVF_ALLOC_H_ */
