6#ifndef _VNIC_ENIC_H_
7#define _VNIC_ENIC_H_
8
9/* Hardware intr coalesce timer is in units of 1.5us */
10#define INTR_COALESCE_USEC_TO_HW(usec) ((usec) * 2 / 3)
11#define INTR_COALESCE_HW_TO_USEC(usec) ((usec) * 3 / 2)
12
13/* Device-specific region: enet configuration */
14struct vnic_enet_config {
15 u32 flags;
16 u32 wq_desc_count;
17 u32 rq_desc_count;
18 u16 mtu;
19 u16 intr_timer_deprecated;
20 u8 intr_timer_type;
21 u8 intr_mode;
22 char devname[16];
23 u32 intr_timer_usec;
24 u16 loop_tag;
25 u16 vf_rq_count;
26 u16 num_arfs;
27 u64 mem_paddr;
28 u16 rdma_qp_id;
29 u16 rdma_qp_count;
30 u16 rdma_resgrp;
31 u32 rdma_mr_id;
32 u32 rdma_mr_count;
33 u32 max_pkt_size;
34};

36#define VENETF_TSO 0x1 /* TSO enabled */
37#define VENETF_LRO 0x2 /* LRO enabled */
38#define VENETF_RXCSUM 0x4 /* RX csum enabled */
39#define VENETF_TXCSUM 0x8 /* TX csum enabled */
40#define VENETF_RSS 0x10 /* RSS enabled */
41#define VENETF_RSSHASH_IPV4 0x20 /* Hash on IPv4 fields */
42#define VENETF_RSSHASH_TCPIPV4 0x40 /* Hash on TCP + IPv4 fields */
43#define VENETF_RSSHASH_IPV6 0x80 /* Hash on IPv6 fields */
44#define VENETF_RSSHASH_TCPIPV6 0x100 /* Hash on TCP + IPv6 fields */
45#define VENETF_RSSHASH_IPV6_EX 0x200 /* Hash on IPv6 extended fields */
46#define VENETF_RSSHASH_TCPIPV6_EX 0x400 /* Hash on TCP + IPv6 ext. fields */
47#define VENETF_LOOP 0x800 /* Loopback enabled */
48#define VENETF_FAILOVER 0x1000 /* Fabric failover enabled */
49#define VENETF_USPACE_NIC 0x2000 /* vHPC enabled */
50#define VENETF_VMQ 0x4000 /* VMQ enabled */
51#define VENETF_ARFS 0x8000 /* ARFS enabled */
52#define VENETF_VXLAN 0x10000 /* VxLAN offload */
53#define VENETF_NVGRE 0x20000 /* NVGRE offload */
54#define VENETF_GRPINTR 0x40000 /* group interrupt */
55
56#define VENET_INTR_TYPE_MIN 0 /* Timer specs min interrupt spacing */
57#define VENET_INTR_TYPE_IDLE 1 /* Timer specs idle time before irq */
58
59#define VENET_INTR_MODE_ANY 0 /* Try MSI-X, then MSI, then INTx */
60#define VENET_INTR_MODE_MSI 1 /* Try MSI then INTx */
61#define VENET_INTR_MODE_INTX 2 /* Try INTx only */
62
63#endif /* _VNIC_ENIC_H_ */
