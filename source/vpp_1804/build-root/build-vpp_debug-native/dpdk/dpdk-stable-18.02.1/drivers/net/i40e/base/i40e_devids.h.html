34#ifndef _I40E_DEVIDS_H_
35#define _I40E_DEVIDS_H_

37/* Vendor ID */
38#define I40E_INTEL_VENDOR_ID 0x8086
39
40/* Device IDs */
41#define I40E_DEV_ID_SFP_XL710 0x1572
42#define I40E_DEV_ID_QEMU 0x1574
43#define I40E_DEV_ID_KX_B 0x1580
44#define I40E_DEV_ID_KX_C 0x1581
45#define I40E_DEV_ID_QSFP_A 0x1583
46#define I40E_DEV_ID_QSFP_B 0x1584
47#define I40E_DEV_ID_QSFP_C 0x1585
48#define I40E_DEV_ID_10G_BASE_T 0x1586
49#define I40E_DEV_ID_20G_KR2 0x1587
50#define I40E_DEV_ID_20G_KR2_A 0x1588
51#define I40E_DEV_ID_10G_BASE_T4 0x1589
52#define I40E_DEV_ID_25G_B 0x158A
53#define I40E_DEV_ID_25G_SFP28 0x158B
54#if defined(INTEGRATED_VF) || defined(VF_DRIVER) || defined(I40E_NDIS_SUPPORT)
55#define I40E_DEV_ID_VF 0x154C
56#define I40E_DEV_ID_VF_HV 0x1571
57#define I40E_DEV_ID_ADAPTIVE_VF 0x1889
58#endif /* VF_DRIVER */
59#ifdef X722_A0_SUPPORT
60#define I40E_DEV_ID_X722_A0 0x374C
61#if defined(INTEGRATED_VF) || defined(VF_DRIVER)
62#define I40E_DEV_ID_X722_A0_VF 0x374D
63#endif
64#endif
65#define I40E_DEV_ID_KX_X722 0x37CE
66#define I40E_DEV_ID_QSFP_X722 0x37CF
67#define I40E_DEV_ID_SFP_X722 0x37D0
68#define I40E_DEV_ID_1G_BASE_T_X722 0x37D1
69#define I40E_DEV_ID_10G_BASE_T_X722 0x37D2
70#define I40E_DEV_ID_SFP_I_X722 0x37D3
71#if defined(INTEGRATED_VF) || defined(VF_DRIVER) || defined(I40E_NDIS_SUPPORT)
72#define I40E_DEV_ID_X722_VF 0x37CD
73#endif /* VF_DRIVER */
74
75#define i40e_is_40G_device(d) ((d) == I40E_DEV_ID_QSFP_A || \
76 (d) == I40E_DEV_ID_QSFP_B || \
77 (d) == I40E_DEV_ID_QSFP_C)
78
79#define i40e_is_25G_device(d) ((d) == I40E_DEV_ID_25G_B || \
80 (d) == I40E_DEV_ID_25G_SFP28)
