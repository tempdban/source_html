34#ifndef _RTE_ETH_VHOST_H_
35#define _RTE_ETH_VHOST_H_
36
37#ifdef __cplusplus
38extern "C" {
39#endif
40
41#include <stdint.h>
42#include <stdbool.h>
43
44#include <rte_vhost.h>
45
46/*
47 * Event description.
48 */
49struct rte_eth_vhost_queue_event {
50 uint16_t queue_id;
51 bool rx;
52 bool enable;
53};
54
55/**
56 * Get queue events from specified port.
57 * If a callback for below event is registered by
58 * rte_eth_dev_callback_register(), this function will describe what was
59 * changed.
60 * - RTE_ETH_EVENT_QUEUE_STATE
61 * Multiple events may cause only one callback kicking, so call this function
62 * while returning 0.
63 *
64 * @param port_id
65 * Port id.
66 * @param event
67 * Pointer to a rte_eth_vhost_queue_event structure.
68 * @return
69 * - On success, zero.
70 * - On failure, a negative value.
71 */
72int rte_eth_vhost_get_queue_event(uint16_t port_id,
73 struct rte_eth_vhost_queue_event *event);
74
75/**
76 * Get the 'vid' value associated with the specified port.
77 *
78 * @return
79 * - On success, the 'vid' associated with 'port_id'.
80 * - On failure, a negative value.
81 */
82int rte_eth_vhost_get_vid_from_port_id(uint16_t port_id);
