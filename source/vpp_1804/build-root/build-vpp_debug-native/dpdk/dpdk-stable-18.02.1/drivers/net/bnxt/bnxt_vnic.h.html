34#ifndef _BNXT_VNIC_H_
35#define _BNXT_VNIC_H_
36
37#include <sys/queue.h>
38#include <stdbool.h>
39
40struct bnxt_vnic_info {
41 STAILQ_ENTRY(bnxt_vnic_info) next;
42 uint8_t ff_pool_idx;
43
44 uint16_t fw_vnic_id; /* returned by Chimp during alloc */
45 uint16_t rss_rule;
46#define MAX_NUM_TRAFFIC_CLASSES 8
47#define MAX_NUM_RSS_QUEUES_PER_VNIC 16
48#define MAX_QUEUES_PER_VNIC (MAX_NUM_RSS_QUEUES_PER_VNIC + \
49 MAX_NUM_TRAFFIC_CLASSES)
50 uint16_t start_grp_id;
51 uint16_t end_grp_id;
52 uint16_t fw_grp_ids[MAX_QUEUES_PER_VNIC];
53 uint16_t dflt_ring_grp;
54 uint16_t mru;
55 uint16_t hash_type;
56 rte_iova_t rss_table_dma_addr;
57 uint16_t *rss_table;
58 rte_iova_t rss_hash_key_dma_addr;
59 void *rss_hash_key;
60 rte_iova_t mc_list_dma_addr;
61 char *mc_list;
62 uint32_t mc_addr_cnt;
63#define BNXT_MAX_MC_ADDRS 16
64 uint32_t flags;
65#define BNXT_VNIC_INFO_PROMISC (1 << 0)
66#define BNXT_VNIC_INFO_ALLMULTI (1 << 1)
67#define BNXT_VNIC_INFO_BCAST (1 << 2)
68#define BNXT_VNIC_INFO_UCAST (1 << 3)
69#define BNXT_VNIC_INFO_MCAST (1 << 4)
70#define BNXT_VNIC_INFO_TAGGED (1 << 5)
71#define BNXT_VNIC_INFO_UNTAGGED (1 << 6)
72
73 uint16_t cos_rule;
74 uint16_t lb_rule;
75 bool vlan_strip;
76 bool func_default;
77 bool bd_stall;
78 bool roce_dual;
79 bool roce_only;
80 bool rss_dflt_cr;
81
82 STAILQ_HEAD(, bnxt_filter_info) filter;
83 STAILQ_HEAD(, rte_flow) flow_list;
84};
85
86struct bnxt;
87void bnxt_init_vnics(struct bnxt *bp);
88int bnxt_free_vnic(struct bnxt *bp, struct bnxt_vnic_info *vnic,
89 int pool);
90struct bnxt_vnic_info *bnxt_alloc_vnic(struct bnxt *bp);
91void bnxt_free_all_vnics(struct bnxt *bp);
92void bnxt_free_vnic_attributes(struct bnxt *bp);
93int bnxt_alloc_vnic_attributes(struct bnxt *bp);
94void bnxt_free_vnic_mem(struct bnxt *bp);
95int bnxt_alloc_vnic_mem(struct bnxt *bp);
96#endif
