35#include <stdbool.h>

37#include "vhost.h"

39static __rte_always_inline void
40vhost_user_iotlb_rd_lock(struct vhost_virtqueue *vq)
41{
42 rte_rwlock_read_lock(&vq->iotlb_lock);
43}
44
45static __rte_always_inline void
46vhost_user_iotlb_rd_unlock(struct vhost_virtqueue *vq)
47{
48 rte_rwlock_read_unlock(&vq->iotlb_lock);
49}
50
51static __rte_always_inline void
52vhost_user_iotlb_wr_lock(struct vhost_virtqueue *vq)
53{
54 rte_rwlock_write_lock(&vq->iotlb_lock);
55}
56
57static __rte_always_inline void
58vhost_user_iotlb_wr_unlock(struct vhost_virtqueue *vq)
59{
60 rte_rwlock_write_unlock(&vq->iotlb_lock);
61}
