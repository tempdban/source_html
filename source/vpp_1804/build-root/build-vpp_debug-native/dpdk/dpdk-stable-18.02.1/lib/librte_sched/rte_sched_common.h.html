53/*
54 * Compute the Greatest Common Divisor (GCD) of two numbers.
55 * This implementation uses Euclid's algorithm:
56 * gcd(a, 0) = a
57 * gcd(a, b) = gcd(b, a mod b)
58 *
59 */
60static inline uint32_t
61rte_get_gcd(uint32_t a, uint32_t b)
62{
63 uint32_t c;
64
65 if (a == 0)
66 return b;
67 if (b == 0)
68 return a;
69
70 if (a < b) {
71 c = a;
72 a = b;
73 b = c;
74 }
75
76 while (b != 0) {
77 c = a % b;
78 a = b;
79 b = c;
80 }
81
82 return a;
83}
84
85/*
86 * Compute the Lowest Common Denominator (LCD) of two numbers.
87 * This implementation computes GCD first:
88 * LCD(a, b) = (a * b) / GCD(a, b)
89 *
90 */
91static inline uint32_t
92rte_get_lcd(uint32_t a, uint32_t b)
93{
94 return (a * b) / rte_get_gcd(a, b);
95}
