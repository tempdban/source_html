40/* ========================================================================== */
41/* AES out-of-order scheduler fields */
42typedef struct {
43 AES_ARGS_x8 args;
44 DECLARE_ALIGNED(UINT16 lens[8], 16);
45 /* each nibble is index (0...7) of an unused lane,
46 * the last nibble is set to F as a flag
47 */
48 UINT64 unused_lanes;
49 JOB_AES_HMAC *job_in_lane[8];
50} MB_MGR_AES_OOO;
51
52/* ========================================================================== */
53/* AES XCBC out-of-order scheduler fields */
54
55typedef struct {
56 DECLARE_ALIGNED(UINT8 final_block[2 * 16], 32);
57 JOB_AES_HMAC *job_in_lane;
58 UINT64 final_done;
59} XCBC_LANE_DATA;
60
61typedef struct {
62 AES_XCBC_ARGS_x8 args;
63 DECLARE_ALIGNED(UINT16 lens[8], 16);
64 /* each byte is index (0...3) of unused lanes
65 * byte 4 is set to FF as a flag
66 */
67 UINT64 unused_lanes;
68 XCBC_LANE_DATA ldata[8];
69} MB_MGR_AES_XCBC_OOO;

71/* ========================================================================== */
72/* CBC-MAC out-of-order scheduler structure */
73
74typedef struct {
75 AES_ARGS_x8 args; /* need to re-use AES arguments */
76 DECLARE_ALIGNED(UINT16 lens[8], 16);
77 DECLARE_ALIGNED(UINT16 init_done[8], 16);
78 /* each byte is index (0...3) of unused lanes
79 * byte 4 is set to FF as a flag
80 */
81 UINT64 unused_lanes;
82 JOB_AES_HMAC *job_in_lane[8];
83 DECLARE_ALIGNED(UINT8 init_blocks[8 * (4 * 16)], 32);
84} MB_MGR_CBCMAC_OOO;
85
86/* ========================================================================== */

91 /* each nibble is index (0...7) of unused lanes
92 * nibble 8 is set to F as a flag
93 */
94 UINT64 unused_lanes;
95 JOB_AES_HMAC *job_in_lane[16];
96 UINT32 num_lanes_inuse;
97} MB_MGR_DES_OOO;

99/* ========================================================================== */
100/* SHA-HMAC out-of-order scheduler fields */
101
102/* used for SHA1 and SHA256 */
103typedef struct {
104 /* YMM aligned access to extra_block */
105 DECLARE_ALIGNED(UINT8 extra_block[2 * SHA1_BLOCK_SIZE+8], 32);
106 JOB_AES_HMAC *job_in_lane;
107 UINT8 outer_block[64];
108 UINT32 outer_done;
109 UINT32 extra_blocks; /* num extra blocks (1 or 2) */
110 UINT32 size_offset; /* offset in extra_block to start of size field */
111 UINT32 start_offset; /* offset to start of data */
112} HMAC_SHA1_LANE_DATA;
113
114/* used for SHA512 */
115typedef struct {
116 DECLARE_ALIGNED(UINT8 extra_block[2 * SHA_512_BLOCK_SIZE + 16], 32);
117 UINT8 outer_block[SHA_512_BLOCK_SIZE];
118 JOB_AES_HMAC *job_in_lane;
119 UINT32 outer_done;
120 UINT32 extra_blocks; /* num extra blocks (1 or 2) */
121 UINT32 size_offset; /* offset in extra_block to start of size field */
122 UINT32 start_offset; /* offset to start of data */
123} HMAC_SHA512_LANE_DATA;

268/*
269 * JOBS() and ADV_JOBS() also used in mb_mgr_code.h
270 * index in JOBS array using byte offset rather than object index
271 */
272__forceinline
273JOB_AES_HMAC *JOBS(MB_MGR *state, const int offset)
274{
275 char *cp = (char *)state->jobs;
276
277 return (JOB_AES_HMAC *)(cp + offset);
278}
279
280__forceinline
281void ADV_JOBS(int *ptr)
282{
283 *ptr += sizeof(JOB_AES_HMAC);
284 if (*ptr >= (int) (MAX_JOBS * sizeof(JOB_AES_HMAC)))
285 *ptr = 0;
286}
287
288__forceinline
289JOB_AES_HMAC *
290get_completed_job_sse(MB_MGR *state)
291{
292 JOB_AES_HMAC *job;
293
294 if (state->earliest_job < 0)
295 return NULL;
296
297 job = JOBS(state, state->earliest_job);
298 if (job->status < STS_COMPLETED)
299 return NULL;
300
301 ADV_JOBS(&state->earliest_job);
302
303 if (state->earliest_job == state->next_job)
304 state->earliest_job = -1;
305
306 return job;
307}
308
309__forceinline
310JOB_AES_HMAC *
311get_next_job_sse(MB_MGR *state)
312{
313 return JOBS(state, state->next_job);
314}

316/*
317 * Wrapper macros to call arch API's set up
318 * at init phase of multi-buffer manager.
319 *
320 * For example, after calling init_mb_mgr_sse(&mgr)
321 * The 'mgr' structure be set up so that:
322 * mgr.get_next_job will point to get_next_job_sse(),
323 * mgr.submit_job will point to submit_job_sse(),
324 * mgr.submit_job_nocheck will point to submit_job_nocheck_sse(),
325 * mgr.get_completed_job will point to get_completed_job_sse(),
326 * mgr.flush_job will point to flush_job_sse(),
327 * mgr.queue_size will point to queue_size_sse()
328 * mgr.keyexp_128 will point to aes_keyexp_128_sse()
329 * mgr.keyexp_192 will point to aes_keyexp_192_sse()
330 * mgr.keyexp_256 will point to aes_keyexp_256_sse()
331 *
332 * Direct use of arch API's may result in better performance.
333 * Using below indirect interface may produce slightly worse performance but
334 * it can simplify application implementation.
335 * LibTestApp provides example of using the indirect interface.
336 */
337#define IMB_GET_NEXT_JOB(_mgr) ((_mgr)->get_next_job((_mgr)))
338#define IMB_SUBMIT_JOB(_mgr) ((_mgr)->submit_job((_mgr)))
339#define IMB_SUBMIT_JOB_NOCHECK(_mgr) ((_mgr)->submit_job_nocheck((_mgr)))
340#define IMB_GET_COMPLETED_JOB(_mgr) ((_mgr)->get_completed_job((_mgr)))
341#define IMB_FLUSH_JOB(_mgr) ((_mgr)->flush_job((_mgr)))
342#define IMB_QUEUE_SIZE(_mgr) ((_mgr)->queue_size((_mgr)))
343#define IMB_AES_KEYEXP_128(_mgr, _raw, _enc, _dec) \
344 ((_mgr)->keyexp_128((_raw), (_enc), (_dec)))
345#define IMB_AES_KEYEXP_192(_mgr, _raw, _enc, _dec) \
346 ((_mgr)->keyexp_192((_raw), (_enc), (_dec)))
347#define IMB_AES_KEYEXP_256(_mgr, _raw, _enc, _dec) \
348 ((_mgr)->keyexp_256((_raw), (_enc), (_dec)))
