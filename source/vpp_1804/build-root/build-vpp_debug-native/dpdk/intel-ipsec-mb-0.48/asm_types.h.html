28#ifndef IMB_ASM_TYPES_H
29#define IMB_ASM_TYPES_H
30
31#include "os.h"
32#include "constants.h"
33
34typedef struct {
35 const uint8_t *in[8];
36 uint8_t *out[8];
37 const uint32_t *keys[8];
38 DECLARE_ALIGNED(UINT128 IV[8], 32);
39} AES_ARGS_x8;
40
41/* AVX512 with its larger register sizes (vs AVX2) supports more parallelism
42 * It will be used to size objects
43 */
44
45#define AVX512_NUM_SHA1_LANES 16
46#define AVX512_NUM_SHA256_LANES 16
47#define AVX512_NUM_SHA512_LANES 8
48#define AVX512_NUM_MD5_LANES 32
49#define AVX512_NUM_DES_LANES 16
50
51/* AVX2 with its larger register sizes (vs SSE) supports more parallelism */
52
53#define AVX2_NUM_SHA1_LANES 8
54#define AVX2_NUM_SHA256_LANES 8
55#define AVX2_NUM_SHA512_LANES 4
56#define AVX2_NUM_MD5_LANES 16
57
58#define AVX_NUM_SHA1_LANES 4
59#define AVX_NUM_SHA256_LANES 4
60#define AVX_NUM_SHA512_LANES 2
61#define AVX_NUM_MD5_LANES 8
62
63#define SSE_NUM_SHA1_LANES AVX_NUM_SHA1_LANES
64#define SSE_NUM_SHA256_LANES AVX_NUM_SHA256_LANES
65#define SSE_NUM_SHA512_LANES AVX_NUM_SHA512_LANES
66#define SSE_NUM_MD5_LANES AVX_NUM_MD5_LANES
67
68/* Each row is sized to hold enough lanes for AVX2, AVX1 and SSE use a subset
69 * of each row. Thus one row is not adjacent in memory to its neighboring rows
70 * in the case of SSE and AVX1.
71 */
72#define MD5_DIGEST_SZ (NUM_MD5_DIGEST_WORDS * AVX512_NUM_MD5_LANES)
73#define SHA1_DIGEST_SZ (NUM_SHA_DIGEST_WORDS * AVX512_NUM_SHA1_LANES)
74#define SHA256_DIGEST_SZ (NUM_SHA_256_DIGEST_WORDS * AVX512_NUM_SHA256_LANES)
75#define SHA512_DIGEST_SZ (NUM_SHA_512_DIGEST_WORDS * AVX512_NUM_SHA512_LANES)
76
77typedef struct {
78 DECLARE_ALIGNED(uint32_t digest[SHA1_DIGEST_SZ], 32);
79 uint8_t *data_ptr[AVX512_NUM_SHA1_LANES];
80} SHA1_ARGS;
81
82typedef struct {
83 DECLARE_ALIGNED(uint32_t digest[SHA256_DIGEST_SZ], 32);
84 uint8_t *data_ptr[AVX512_NUM_SHA256_LANES];
85} SHA256_ARGS;
86
87typedef struct {
88 DECLARE_ALIGNED(uint64_t digest[SHA512_DIGEST_SZ], 32);
89 uint8_t *data_ptr[AVX512_NUM_SHA512_LANES];
90} SHA512_ARGS;
91
92typedef struct {
93 DECLARE_ALIGNED(uint32_t digest[MD5_DIGEST_SZ], 32);
94 uint8_t *data_ptr[AVX512_NUM_MD5_LANES];
95} MD5_ARGS;
96
97typedef struct {
98 const uint8_t *in[8];
99 const uint32_t *keys[8];
100 DECLARE_ALIGNED(UINT128 ICV[8], 32);
101} AES_XCBC_ARGS_x8;
102
103typedef struct {
104 const uint8_t *in[AVX512_NUM_DES_LANES];
105 uint8_t *out[AVX512_NUM_DES_LANES];
106 const uint8_t *keys[AVX512_NUM_DES_LANES];
107 uint32_t IV[AVX512_NUM_DES_LANES * 2]; /* uint32_t is more handy here */
108 uint32_t partial_len[AVX512_NUM_DES_LANES];
109 uint32_t block_len[AVX512_NUM_DES_LANES];
110 const uint8_t *last_in[AVX512_NUM_DES_LANES];
111 uint8_t *last_out[AVX512_NUM_DES_LANES];
112} DES_ARGS_x16;
113
114#endif /* ifdef IMB_ASM_TYPES_H */
