195struct cpuid_regs {
196 UINT32 eax;
197 UINT32 ebx;
198 UINT32 ecx;
199 UINT32 edx;
200};
201
202/*
203 * A C wrapper for CPUID opcode
204 *
205 * Parameters:
206 * [in] leaf - CPUID leaf number (EAX)
207 * [in] subleaf - CPUID sub-leaf number (ECX)
208 * [out] out - registers structure to store results of CPUID into
209 */
210static void
211__mbcpuid(const unsigned leaf, const unsigned subleaf,
212 struct cpuid_regs *out)
213{
214#ifdef _WIN32
215 /* Windows */
216 int regs[4];
217
218 __cpuidex(regs, leaf, subleaf);
219 out->eax = regs[0];
220 out->ebx = regs[1];
221 out->ecx = regs[2];
222 out->edx = regs[3];
223#else
224 /* Linux */
225#ifdef __x86_64__
226 asm volatile("mov %4, %%eax\n\t"
227 "mov %5, %%ecx\n\t"
228 "cpuid\n\t"
229 "mov %%eax, %0\n\t"
230 "mov %%ebx, %1\n\t"
231 "mov %%ecx, %2\n\t"
232 "mov %%edx, %3\n\t"
233 : "=g" (out->eax), "=g" (out->ebx), "=g" (out->ecx),
234 "=g" (out->edx)
235 : "g" (leaf), "g" (subleaf)
236 : "%eax", "%ebx", "%ecx", "%edx");
237#else
238 asm volatile("push %%ebx\n\t"
239 "mov %4, %%eax\n\t"
240 "mov %5, %%ecx\n\t"
241 "cpuid\n\t"
242 "mov %%eax, %0\n\t"
243 "mov %%ebx, %1\n\t"
244 "mov %%ecx, %2\n\t"
245 "mov %%edx, %3\n\t"
246 "pop %%ebx\n\t"
247 : "=g" (out->eax), "=g" (out->ebx), "=g" (out->ecx),
248 "=g" (out->edx)
249 : "g" (leaf), "g" (subleaf)
250 : "%eax", "%ecx", "%edx");
251#endif
252#endif /* Linux */
253}

255/*
256 * Uses CPUID instruction to detected presence of SHA extensions.
257 *
258 * Return value:
259 * 0 - SHA extensions not present
260 * 1 - SHA extensions present
261 */
262static int
263sha_extensions_supported(void)
264{
265 struct cpuid_regs r;
266
267 /* Check highest leaf number. If less then 7 then SHA not supported. */
268 __mbcpuid(0x0, 0x0, &r);
269 if (r.eax < 0x7)
270 return 0;
271
272 /* Check presence of SHA extensions in the extended feature flags */
273 __mbcpuid(0x7, 0x0, &r);
274 if (r.ebx & (1 << 29))
275 return 1;
276
277 return 0;
278}
