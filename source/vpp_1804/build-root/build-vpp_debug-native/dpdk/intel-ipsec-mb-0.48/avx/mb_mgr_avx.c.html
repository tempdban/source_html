251 /* Init HMAC/SHA1 out-of-order fields */
252 state->hmac_sha_1_ooo.lens[0] = 0;
253 state->hmac_sha_1_ooo.lens[1] = 0;
254 state->hmac_sha_1_ooo.lens[2] = 0;
255 state->hmac_sha_1_ooo.lens[3] = 0;
256 state->hmac_sha_1_ooo.lens[4] = 0xFFFF;
257 state->hmac_sha_1_ooo.lens[5] = 0xFFFF;
258 state->hmac_sha_1_ooo.lens[6] = 0xFFFF;
259 state->hmac_sha_1_ooo.lens[7] = 0xFFFF;
260 state->hmac_sha_1_ooo.unused_lanes = 0xFF03020100;
261 for (j = 0; j < AVX_NUM_SHA1_LANES; j++) {
262 state->hmac_sha_1_ooo.ldata[j].job_in_lane = NULL;
263 state->hmac_sha_1_ooo.ldata[j].extra_block[64] = 0x80;
264 memset(state->hmac_sha_1_ooo.ldata[j].extra_block + 65,
265 0x00,
266 64+7);
267 p = state->hmac_sha_1_ooo.ldata[j].outer_block;
268 memset(p + 5*4 + 1,
269 0x00,
270 64 - 5*4 - 1 - 2);
271 p[5*4] = 0x80;
272 p[64-2] = 0x02;
273 p[64-1] = 0xA0;
274 }
275 /* Init HMAC/SHA224 out-of-order fields */
276 state->hmac_sha_224_ooo.lens[0] = 0;
277 state->hmac_sha_224_ooo.lens[1] = 0;
278 state->hmac_sha_224_ooo.lens[2] = 0;
279 state->hmac_sha_224_ooo.lens[3] = 0;
280 state->hmac_sha_224_ooo.lens[4] = 0xFFFF;
281 state->hmac_sha_224_ooo.lens[5] = 0xFFFF;
282 state->hmac_sha_224_ooo.lens[6] = 0xFFFF;
283 state->hmac_sha_224_ooo.lens[7] = 0xFFFF;
284 state->hmac_sha_224_ooo.unused_lanes = 0xFF03020100;
285 for (j = 0; j < AVX_NUM_SHA256_LANES; j++) {
286 state->hmac_sha_224_ooo.ldata[j].job_in_lane = NULL;
287 state->hmac_sha_224_ooo.ldata[j].extra_block[64] = 0x80;
288 memset(state->hmac_sha_224_ooo.ldata[j].extra_block + 65,
289 0x00,
290 64+7);
291 p = state->hmac_sha_224_ooo.ldata[j].outer_block;
292 memset(p + 8*4 + 1,
293 0x00,
294 64 - 8*4 - 1 - 2);
295 p[7 * 4] = 0x80; /* digest 7 words long */
296 p[64 - 2] = 0x02; /* length in little endian = 0x02E0 */
297 p[64 - 1] = 0xE0;
298 }
299
300 /* Init HMAC/SHA256 out-of-order fields */
301 state->hmac_sha_256_ooo.lens[0] = 0;
302 state->hmac_sha_256_ooo.lens[1] = 0;

304 state->hmac_sha_256_ooo.lens[3] = 0;
305 state->hmac_sha_256_ooo.lens[4] = 0xFFFF;
306 state->hmac_sha_256_ooo.lens[5] = 0xFFFF;
307 state->hmac_sha_256_ooo.lens[6] = 0xFFFF;
308 state->hmac_sha_256_ooo.lens[7] = 0xFFFF;
309 state->hmac_sha_256_ooo.unused_lanes = 0xFF03020100;
310 for (j = 0; j < AVX_NUM_SHA256_LANES; j++) {
311 state->hmac_sha_256_ooo.ldata[j].job_in_lane = NULL;
312 state->hmac_sha_256_ooo.ldata[j].extra_block[64] = 0x80;
313 memset(state->hmac_sha_256_ooo.ldata[j].extra_block + 65,
314 0x00,
315 64+7);
316 p = state->hmac_sha_256_ooo.ldata[j].outer_block;
317 memset(p + 8*4 + 1,
318 0x00,
319 64 - 8*4 - 1 - 2);
320 p[8 * 4] = 0x80; /* 8 digest words */
321 p[64 - 2] = 0x03; /* length */
322 p[64 - 1] = 0x00;
323 }

325
326 /* Init HMAC/SHA384 out-of-order fields */
327 state->hmac_sha_384_ooo.lens[0] = 0;
328 state->hmac_sha_384_ooo.lens[1] = 0;
329 state->hmac_sha_384_ooo.lens[2] = 0xFFFF;
330 state->hmac_sha_384_ooo.lens[3] = 0xFFFF;
331 state->hmac_sha_384_ooo.lens[4] = 0xFFFF;
332 state->hmac_sha_384_ooo.lens[5] = 0xFFFF;
333 state->hmac_sha_384_ooo.lens[6] = 0xFFFF;
334 state->hmac_sha_384_ooo.lens[7] = 0xFFFF;
335 state->hmac_sha_384_ooo.unused_lanes = 0xFF0100;
336 for (j = 0; j < AVX_NUM_SHA512_LANES; j++) {
337 MB_MGR_HMAC_SHA_512_OOO *ctx = &state->hmac_sha_384_ooo;
338
339 ctx->ldata[j].job_in_lane = NULL;
340 ctx->ldata[j].extra_block[SHA_384_BLOCK_SIZE] = 0x80;
341 memset(ctx->ldata[j].extra_block + (SHA_384_BLOCK_SIZE + 1),
342 0x00, SHA_384_BLOCK_SIZE + 7);
343
344 p = ctx->ldata[j].outer_block;
345 memset(p + SHA384_DIGEST_SIZE_IN_BYTES + 1, 0x00,
346 /* special end point because this length is constant */
347 SHA_384_BLOCK_SIZE -
348 SHA384_DIGEST_SIZE_IN_BYTES - 1 - 2);
349 /* mark the end */
350 p[SHA384_DIGEST_SIZE_IN_BYTES] = 0x80;
351 /* hmac outer block length always of fixed size,
352 * it is OKey length, a whole message block length, 1024 bits,
353 * with padding plus the length of the inner digest,
354 * which is 384 bits, 1408 bits == 0x0580.
355 * The input message block needs to be converted to big endian
356 * within the sha implementation before use.
357 */
358 p[SHA_384_BLOCK_SIZE - 2] = 0x05;
359 p[SHA_384_BLOCK_SIZE - 1] = 0x80;
360 }
361

375 ctx->ldata[j].job_in_lane = NULL;

387 * hmac outer block length always of fixed size,

390 * which is 512 bits, 1536 bits == 0x600.
391 * The input message block needs to be converted to big endian
392 * within the sha implementation before use.
393 */
394 p[SHA_512_BLOCK_SIZE - 2] = 0x06;
395 p[SHA_512_BLOCK_SIZE - 1] = 0x00;
396 }
