28#ifndef __OS_H
29#define __OS_H
30
31#ifdef LINUX
32#define DECLARE_ALIGNED(decl, alignval) \
33 decl __attribute__((aligned(alignval)))
34#define __forceinline \
35 static inline __attribute__((always_inline))
36
37#if __GNUC__ >= 4
38#define IMB_DLL_EXPORT __attribute__((visibility("default")))
39#define IMB_DLL_LOCAL __attribute__((visibility("hidden")))
40#else /* GNU C 4.0 and later */
41#define IMB_DLL_EXPORT
42#define IMB_DLL_LOCAL
43#endif /* different C compiler */
44
45#else
46
47#define DECLARE_ALIGNED(decl, alignval) \
48 __declspec(align(alignval)) decl
49#define __forceinline \
50 static __forceinline
51
52/* Windows DLL export is done via DEF file */
53#define IMB_DLL_EXPORT
54#define IMB_DLL_LOCAL
55
56#endif
57
58#ifdef DEBUG
59#include <assert.h>
60#define IMB_ASSERT(x) assert(x)
61#else
62#define IMB_ASSERT(x)
63#endif
64
65#ifndef IMB_DIM
66#define IMB_DIM(x) (sizeof(x) / sizeof(x[0]))
67#endif
68
69#endif /* __OS_H */
