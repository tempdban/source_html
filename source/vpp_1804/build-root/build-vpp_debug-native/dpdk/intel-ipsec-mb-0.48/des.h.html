28#ifndef IMB_DES_H
29#define IMB_DES_H
30
31#include <stdint.h>
32
33#define DES_KEY_SCHED_SIZE (16 * 8) /* 16 rounds x 8 bytes */
34#define DES_BLOCK_SIZE 8
35
36/**
37 * @brief DES CBC encryption
38 *
39 * @param input source buffer with plain text
40 * @param output destination buffer for cipher text
41 * @param size number of bytes to encrypt (multiple of 8)
42 * @param ks pointer to key schedule structure
43 * @param ivec pointer to initialization vector
44 */
45void des_enc_cbc_basic(const void *input, void *output, const int size,
46 const uint64_t *ks, const uint64_t *ivec);
47
48/**
49 * @brief DES CBC decryption
50 *
51 * @param input source buffer with cipher text
52 * @param output destination buffer for plain text
53 * @param size number of bytes to decrypt (multiple of 8)
54 * @param ks pointer to key schedule structure
55 * @param ivec pointer to initialization vector
56 */
57void des_dec_cbc_basic(const void *input, void *output, const int size,
58 const uint64_t *ks, const uint64_t *ivec);
59
60/**
61 * @brief DOCSIS DES encryption
62 *
63 * @param input source buffer with plain text
64 * @param output destination buffer for cipher text
65 * @param size number of bytes to encrypt
66 * @param ks pointer to key schedule structure
67 * @param ivec pointer to initialization vector
68 */
69void docsis_des_enc_basic(const void *input, void *output, const int size,
70 const uint64_t *ks, const uint64_t *ivec);
71
72/**
73 * @brief DOCSIS DES decryption
74 *
75 * @param input source buffer with cipher text
76 * @param output destination buffer for plain text
77 * @param size number of bytes to decrypt
78 * @param ks pointer to key schedule structure
79 * @param ivec pointer to initialization vector
80 */
81void docsis_des_dec_basic(const void *input, void *output, const int size,
82 const uint64_t *ks, const uint64_t *ivec);
83
84/**
85 * @brief DES key schedule set up
86 *
87 * \a ks buffer needs to accomodate \a DES_KEY_SCHED_SIZE (128) bytes of data.
88 *
89 * @param ks destination buffer to accomodate DES key schedule
90 * @param key a pointer to an 8 byte DES key
91 *
92 * @return Operation status
93 * @retval 0 success
94 * @retval !0 error
95 */
96int des_key_schedule(uint64_t *ks, const void *key);
97
98#endif /* IMB_DES_H */
