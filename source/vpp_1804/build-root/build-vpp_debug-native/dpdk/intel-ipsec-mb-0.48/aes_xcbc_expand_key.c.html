42static UINT32 in[4*3] = {
43 0x01010101, 0x01010101, 0x01010101, 0x01010101,

48void
49aes_xcbc_expand_key_sse(const void *key, void *k1_exp, void *k2, void *k3)
50{
51 DECLARE_ALIGNED(UINT32 keys_exp_enc[11*4], 16);
52
53 aes_keyexp_128_enc_sse(key, keys_exp_enc);
54
55 aes128_ecbenc_x3_sse(in, keys_exp_enc, k1_exp, k2, k3);
56
57 aes_keyexp_128_enc_sse(k1_exp, k1_exp);
58}

68
69 aes_keyexp_128_enc_avx(k1_exp, k1_exp);
