39 /* Authenticated Tag Length in bytes.
40 * Valid values are 16 (most likely), 12 or 8. */
41#define MAX_TAG_LEN (16)
42 /*
43 * IV data is limited to 16 bytes as follows:
44 * 12 bytes is provided by an application -
45 * pre-counter block j0: 4 byte salt (from Security Association)
46 * concatenated with 8 byte Initialization Vector (from IPSec ESP
47 * Payload).
48 * 4 byte value 0x00000001 is padded automatically by the library -
49 * there is no need to add these 4 bytes on application side anymore.
50 */
51#define GCM_IV_DATA_LEN (12)
52
53#define LONGEST_TESTED_AAD_LENGTH (2 * 1024)
54
55 /* Key lengths of 128 and 256 supported */
56#define GCM_128_KEY_LEN (16)
57#define GCM_192_KEY_LEN (24)
58#define GCM_256_KEY_LEN (32)
59
60#define GCM_BLOCK_LEN 16
61#define GCM_ENC_KEY_LEN 16
62#define GCM_KEY_SETS (15) /*exp key + 14 exp round keys*/
63
64/**
65 * @brief holds intermediate key data needed to improve performance
66 *
67 * gcm_key_data hold internal key information used by gcm128, gcm192 and gcm256.
68 */
69#ifdef __WIN32
70__declspec(align(16))
71#endif /* WIN32 */
72struct gcm_key_data {
73 uint8_t expanded_keys[GCM_ENC_KEY_LEN * GCM_KEY_SETS];
74 /* storage for HashKey mod poly */
75 uint8_t shifted_hkey_1[GCM_ENC_KEY_LEN]; /* HashKey<<1 mod poly */
76 uint8_t shifted_hkey_2[GCM_ENC_KEY_LEN]; /* HashKey^2<<1 mod poly */
77 uint8_t shifted_hkey_3[GCM_ENC_KEY_LEN]; /* HashKey^3<<1 mod poly */
78 uint8_t shifted_hkey_4[GCM_ENC_KEY_LEN]; /* HashKey^4<<1 mod poly */
79 uint8_t shifted_hkey_5[GCM_ENC_KEY_LEN]; /* HashKey^5<<1 mod poly */
80 uint8_t shifted_hkey_6[GCM_ENC_KEY_LEN]; /* HashKey^6<<1 mod poly */
81 uint8_t shifted_hkey_7[GCM_ENC_KEY_LEN]; /* HashKey^7<<1 mod poly */
82 uint8_t shifted_hkey_8[GCM_ENC_KEY_LEN]; /* HashKey^8<<1 mod poly */
83 /*
84 * Storage for XOR of High 64 bits and low 64 bits of HashKey mod poly.
85 * This is needed for Karatsuba purposes.
86 */
87 uint8_t shifted_hkey_1_k[GCM_ENC_KEY_LEN]; /* HashKey<<1 mod poly */
88 uint8_t shifted_hkey_2_k[GCM_ENC_KEY_LEN]; /* HashKey^2<<1 mod poly */
89 uint8_t shifted_hkey_3_k[GCM_ENC_KEY_LEN]; /* HashKey^3<<1 mod poly */
90 uint8_t shifted_hkey_4_k[GCM_ENC_KEY_LEN]; /* HashKey^4<<1 mod poly */
91 uint8_t shifted_hkey_5_k[GCM_ENC_KEY_LEN]; /* HashKey^5<<1 mod poly */
92 uint8_t shifted_hkey_6_k[GCM_ENC_KEY_LEN]; /* HashKey^6<<1 mod poly */
93 uint8_t shifted_hkey_7_k[GCM_ENC_KEY_LEN]; /* HashKey^7<<1 mod poly */
94 uint8_t shifted_hkey_8_k[GCM_ENC_KEY_LEN]; /* HashKey^8<<1 mod poly */
95}
96#ifdef LINUX
97 __attribute__((aligned(16)));
98#else
99 ;

102/**
103 * @brief holds GCM operation context
104 */
105struct gcm_context_data {
106 /* init, update and finalize context data */
107 uint8_t aad_hash[GCM_BLOCK_LEN];
108 uint64_t aad_length;
109 uint64_t in_length;
110 uint8_t partial_block_enc_key[GCM_BLOCK_LEN];
111 uint8_t orig_IV[GCM_BLOCK_LEN];
112 uint8_t current_counter[GCM_BLOCK_LEN];
113 uint64_t partial_block_length;
114};
