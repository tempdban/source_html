28#ifndef IMB_JOB_AES_HMAC_H
29#define IMB_JOB_AES_HMAC_H
30
31#include "types.h"
32
33typedef enum {
34 STS_BEING_PROCESSED = 0,
35 STS_COMPLETED_AES = 1,
36 STS_COMPLETED_HMAC = 2,
37 STS_COMPLETED = 3, /* COMPLETED_AES | COMPLETED_HMAC */
38 STS_INVALID_ARGS = 4,
39 STS_INTERNAL_ERROR,
40 STS_ERROR
41} JOB_STS;
42
43typedef enum {
44 CBC = 1,
45 CNTR,
46 NULL_CIPHER,
47 DOCSIS_SEC_BPI,
48#ifndef NO_GCM
49 GCM,
50#endif /* !NO_GCM */
51 CUSTOM_CIPHER,
52 DES,
53 DOCSIS_DES,
54 CCM,
55} JOB_CIPHER_MODE;
56
57typedef enum {
58 ENCRYPT = 1,
59 DECRYPT
60} JOB_CIPHER_DIRECTION;
61
62typedef enum {
63 SHA1 = 1,
64 SHA_224,
65 SHA_256,
66 SHA_384,
67 SHA_512,
68 AES_XCBC,
69 MD5,
70 NULL_HASH,
71#ifndef NO_GCM
72 AES_GMAC,
73#endif /* !NO_GCM */
74 CUSTOM_HASH,
75 AES_CCM,
76} JOB_HASH_ALG;
77
78typedef enum {
79 CIPHER_HASH = 1,
80 HASH_CIPHER
81} JOB_CHAIN_ORDER;
82
83typedef enum {
84 AES_128_BYTES = 16,
85 AES_192_BYTES = 24,
86 AES_256_BYTES = 32
87} AES_KEY_SIZE_BYTES;
88
89typedef struct JOB_AES_HMAC {
90 const void *aes_enc_key_expanded; /* 16-byte aligned pointer. */
91 const void *aes_dec_key_expanded;
92 UINT64 aes_key_len_in_bytes; /* Only 16, 24, and 32 byte (128, 192 and
93 * 256-bit) keys supported at this time. */
94 const UINT8 *src; /* Input. May be cipher text or plaintext.
95 * In-place ciphering allowed. */
96 UINT8 *dst; /*Output. May be cipher text or plaintext.
97 * In-place ciphering allowed, i.e. dst = src. */
98 UINT64 cipher_start_src_offset_in_bytes;
99 UINT64 msg_len_to_cipher_in_bytes; /* Max len = 65472 bytes.
100 * IPSec case, the maximum cipher
101 * length would be:
102 * 65535 -
103 * 20 (outer IP header) -
104 * 24 (ESP header + IV) -
105 * 12 (supported ICV length) */
106 UINT64 hash_start_src_offset_in_bytes;
107 UINT64 msg_len_to_hash_in_bytes; /* Max len = 65496 bytes.
108 * (Max cipher len +
109 * 24 bytes ESP header) */
110 const UINT8 *iv; /* AES IV. */
111 UINT64 iv_len_in_bytes; /* AES IV length in bytes. */
112 UINT8 *auth_tag_output; /* HMAC Tag output. This may point to a location
113 * in the src buffer (for in place)*/
114 UINT64 auth_tag_output_len_in_bytes; /* HMAC Tag output length in bytes.
115 * (May be a truncated value)*/
116
117 /* Start algorithm-specific fields */
118 union {
119 struct _HMAC_specific_fields {
120 /* Hashed result of HMAC key xor'd with ipad (0x36). */
121 const UINT8 *_hashed_auth_key_xor_ipad;
122 /* Hashed result of HMAC key xor'd with opad (0x5c). */
123 const UINT8 *_hashed_auth_key_xor_opad;
124 } HMAC;
125 struct _AES_XCBC_specific_fields {
126 /* 16-byte aligned pointers */
127 const UINT32 *_k1_expanded;
128 const UINT8 *_k2;
129 const UINT8 *_k3;
130 } XCBC;
131 struct _AES_CCM_specific_fields {
132 /* Additional Authentication Data (AAD) */
133 const void *aad;
134 UINT64 aad_len_in_bytes; /* Length of AAD */
135 } CCM;
136#ifndef NO_GCM
137 struct _AES_GCM_specific_fields {
138 /* Additional Authentication Data (AAD) */
139 const void *aad;
140 UINT64 aad_len_in_bytes; /* Length of AAD */
141 } GCM;
142#endif /* !NO_GCM */
143 } u;
144
145 JOB_STS status;
146 JOB_CIPHER_MODE cipher_mode; /* CBC, CNTR, DES, GCM etc. */
147 JOB_CIPHER_DIRECTION cipher_direction; /* Encrypt/decrypt */
148 /* Ignored as the direction is implied by the chain _order field. */
149 JOB_HASH_ALG hash_alg; /* SHA-1 or others... */
150 JOB_CHAIN_ORDER chain_order; /* CIPHER_HASH or HASH_CIPHER */
151
152 void *user_data;
153 void *user_data2;
154
155 /*
156 * stateless custom cipher and hash
157 * Return:
158 * success: 0
159 * fail: other
160 */
161 int (*cipher_func)(struct JOB_AES_HMAC *);
162 int (*hash_func)(struct JOB_AES_HMAC *);
163} JOB_AES_HMAC;
164
165#define hashed_auth_key_xor_ipad u.HMAC._hashed_auth_key_xor_ipad
166#define hashed_auth_key_xor_opad u.HMAC._hashed_auth_key_xor_opad
167#define _k1_expanded u.XCBC._k1_expanded
168#define _k2 u.XCBC._k2
169#define _k3 u.XCBC._k3
170
171#endif /* IMB_JOB_AES_HMAC_H */
