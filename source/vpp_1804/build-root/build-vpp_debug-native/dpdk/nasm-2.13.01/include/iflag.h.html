14static inline unsigned int iflag_test(const iflag_t *f, unsigned int bit)
15{
16 unsigned int index = bit / 32;
17 return f->field[index] & (UINT32_C(1) << (bit - (index * 32)));

42static inline int iflag_cmp(const iflag_t *a, const iflag_t *b)
43{
44 int i;
45
46 for (i = sizeof(a->field) / sizeof(a->field[0]) - 1; i >= 0; i--) {
47 if (a->field[i] == b->field[i])
48 continue;
49
50 return (a->field[i] > b->field[i]) ? 1 : -1;
51 }
52
53 return 0;
54}

56static inline int iflag_cmp_cpu(const iflag_t *a, const iflag_t *b)
57{
58 if (a->field[3] < b->field[3])
59 return -1;
60 else if (a->field[3] > b->field[3])
61 return 1;
62 return 0;

65static inline unsigned int iflag_ffs(const iflag_t *a)
66{
67 unsigned int i;
68
69 for (i = 0; i < sizeof(a->field) / sizeof(a->field[0]); i++) {
70 if (a->field[i])
71 return ilog2_32(a->field[i]) + (i * 32);
72 }
73
74 return 0;
75}

123static inline int iflag_cmp_cpu_level(const iflag_t *a, const iflag_t *b)
124{
125 iflag_t v1 = *a;
126 iflag_t v2 = *b;
127
128 iflag_clear(&v1, IF_CYRIX);
129 iflag_clear(&v1, IF_AMD);
130
131 iflag_clear(&v2, IF_CYRIX);
132 iflag_clear(&v2, IF_AMD);
133
134 if (v1.field[3] < v2.field[3])
135 return -1;
136 else if (v1.field[3] > v2.field[3])
137 return 1;
138
139 return 0;
140}

142static inline iflag_t _iflag_pfmask(const iflag_t *a)
143{
144 iflag_t r;
145
146 iflag_clear_all(&r);
147
148 if (iflag_test(a, IF_CYRIX))
149 iflag_set(&r, IF_CYRIX);
150 if (iflag_test(a, IF_AMD))
151 iflag_set(&r, IF_AMD);
152
153 return r;
154}
