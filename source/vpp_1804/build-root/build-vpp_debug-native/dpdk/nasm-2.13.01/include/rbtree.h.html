39/* This structure should be embedded in a larger data structure;
40 the final output from rb_search() can then be converted back
41 to the larger data structure via container_of(). */
42struct rbtree {
43 uint64_t key;
44 struct rbtree *left, *right;
45 bool red;
46};
47
48struct rbtree *rb_insert(struct rbtree *, struct rbtree *);
49struct rbtree *rb_search(struct rbtree *, uint64_t);
50
51#endif /* NASM_RBTREE_H */
