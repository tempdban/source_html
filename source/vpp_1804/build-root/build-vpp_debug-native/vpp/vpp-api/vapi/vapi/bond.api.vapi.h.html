545static inline vapi_msg_bond_enslave* vapi_alloc_bond_enslave(struct vapi_ctx_s *ctx)
546{
547 vapi_msg_bond_enslave *msg = NULL;
548 const size_t size = sizeof(vapi_msg_bond_enslave);
549 /* cast here required to play nicely with C++ world ... */
550 msg = (vapi_msg_bond_enslave*)vapi_msg_alloc(ctx, size);
551 if (!msg) {
552 return NULL;
553 }
554 msg->header.client_index = vapi_get_client_index(ctx);
555 msg->header.context = 0;
