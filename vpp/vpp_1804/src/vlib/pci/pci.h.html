60typedef struct vlib_pci_device_info
61{
62 u32 flags;
63#define VLIB_PCI_DEVICE_INFO_F_NOIOMMU (1 << 0);
64
65 /* addr */
66 vlib_pci_addr_t addr;
67
68 /* Numa Node */
69 int numa_node;
70
71 /* Device data */

73 u16 vendor_id;
74 u16 device_id;
75
76 /* Vital Product Data */
77 u8 *product_name;
78 u8 *vpd_r;
79 u8 *vpd_w;

81 /* Driver name */

106static inline void
107vlib_pci_free_device_info (vlib_pci_device_info_t * di)
108{
109 if (!di)
110 return;
111 vec_free (di->product_name);
112 vec_free (di->vpd_r);
113 vec_free (di->vpd_w);
114 vec_free (di->driver_name);
115 clib_mem_free (di);
116}

203static inline clib_error_t *
204vlib_pci_intr_enable (vlib_pci_dev_handle_t h)
205{
206 u16 command;
207 clib_error_t *err;
208
209 err = vlib_pci_read_config_u16 (h, 4, &command);
210
211 if (err)
212 return err;
213
214 command &= ~PCI_COMMAND_INTX_DISABLE;
215
216 return vlib_pci_write_config_u16 (h, 4, &command);
217}
218
219static inline clib_error_t *
220vlib_pci_intr_disable (vlib_pci_dev_handle_t h)
221{
222 u16 command;
223 clib_error_t *err;
224
225 err = vlib_pci_read_config_u16 (h, 4, &command);
226
227 if (err)
228 return err;
229
230 command |= PCI_COMMAND_INTX_DISABLE;
231
232 return vlib_pci_write_config_u16 (h, 4, &command);
233}
234
235static inline clib_error_t *
236vlib_pci_bus_master_enable (vlib_pci_dev_handle_t h)
237{
238 clib_error_t *err;
239 u16 command;
240
241 /* Set bus master enable (BME) */
242 err = vlib_pci_read_config_u16 (h, 4, &command);
243
244 if (err)
245 return err;
246
247 if (command & PCI_COMMAND_BUS_MASTER)
248 return 0;
249
250 command |= PCI_COMMAND_BUS_MASTER;
251
252 return vlib_pci_write_config_u16 (h, 4, &command);
253}
