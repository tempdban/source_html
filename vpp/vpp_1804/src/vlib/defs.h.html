40#ifndef included_vlib_defs_h
41#define included_vlib_defs_h
42
43/* Receive or transmit. */
44typedef enum
45{
46 VLIB_RX,
47 VLIB_TX,
48 VLIB_N_RX_TX = 2, /* Used to size arrays. */
49} vlib_rx_or_tx_t;
50
51#define vlib_foreach_rx_tx(v) for (v = 0; v < VLIB_N_RX_TX; v++)
52
53/* Read/write. */
54typedef enum
55{
56 VLIB_READ,
57 VLIB_WRITE,
58} vlib_read_or_write_t;
59
60/* Up/down. */
61typedef enum
62{
63 VLIB_DOWN = 0,
64 VLIB_UP = 1,
65} vlib_up_or_down_t;
66
67/* Enable/disable. */
68typedef enum
69{
70 VLIB_DISABLE = 0,
71 VLIB_ENABLE = 1,
72} vlib_enable_or_disable_t;
73
74#endif /* included_vlib_defs_h */
