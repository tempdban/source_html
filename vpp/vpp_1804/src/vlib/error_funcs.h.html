45always_inline void
46vlib_error_elog_count (vlib_main_t * vm, uword counter, uword increment)
47{
48 if (VLIB_ELOG_MAIN_LOOP > 0 && increment > 0)
49 {
50 elog_main_t *em = &vm->elog_main;
51 elog (em, vec_elt_at_index (vm->error_elog_event_types, counter),
52 increment);
53 }
54}
55
56always_inline void
57vlib_error_count (vlib_main_t * vm, uword node_index,
58 uword counter, uword increment)
59{
60 vlib_node_t *n = vlib_get_node (vm, node_index);
61 vlib_error_main_t *em = &vm->error_main;
62
63 ASSERT (counter < n->n_errors);
64 counter += n->error_heap_index;
65
66 ASSERT (counter < vec_len (em->counters));
67 em->counters[counter] += increment;
68
69 vlib_error_elog_count (vm, counter, increment);
70}
