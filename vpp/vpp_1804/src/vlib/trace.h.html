40#ifndef included_vlib_trace_h
41#define included_vlib_trace_h
42
43#include <vppinfra/pool.h>
44
45typedef struct
46{
47 /* CPU time stamp trace was made. */
48 u64 time;
49
50 /* Node which generated this trace. */
51 u32 node_index;
52
53 /* Number of data words in this trace. */
54 u32 n_data;
55
56 /* Trace data follows. */
57 u8 data[0];
58} vlib_trace_header_t;
59
60typedef struct
61{
62 /* Current number of traces in buffer. */
63 u32 count;
64
65 /* Max. number of traces to be added to buffer. */
66 u32 limit;
67} vlib_trace_node_t;
68
69typedef struct
70{
71 /* Pool of trace buffers. */
72 vlib_trace_header_t **trace_buffer_pool;
73
74 u32 last_main_loop_count;
75 u32 filter_node_index;
76 u32 filter_flag;
77#define FILTER_FLAG_NONE 0
78#define FILTER_FLAG_INCLUDE 1
79#define FILTER_FLAG_EXCLUDE 2
80 u32 filter_count;
81
82 /* set on trace add, cleared on clear trace */
83 u32 trace_active_hint;
84
85 /* Per node trace counts. */
86 vlib_trace_node_t *nodes;
87
88 /* verbosity */
89 int verbose;
90} vlib_trace_main_t;
91
92#endif /* included_vlib_trace_h */
