22always_inline vlib_main_t *
23vlib_get_main (void)
24{
25 vlib_main_t *vm;
26 vm = vlib_mains[vlib_get_thread_index ()];
27 ASSERT (vm);
28 return vm;
29}
30
31always_inline vlib_thread_main_t *
32vlib_get_thread_main ()
33{
34 return &vlib_thread_main;
35}
