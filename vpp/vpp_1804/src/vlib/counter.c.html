40#include <vlib/vlib.h>
41
42void
43vlib_clear_simple_counters (vlib_simple_counter_main_t * cm)
44{
45 counter_t *my_counters;
46 uword i, j;
47
48 for (i = 0; i < vec_len (cm->counters); i++)
49 {
50 my_counters = cm->counters[i];
51
52 for (j = 0; j < vec_len (my_counters); j++)
53 {
54 my_counters[j] = 0;
55 }
56 }
57}
58
59void
60vlib_clear_combined_counters (vlib_combined_counter_main_t * cm)
61{
62 vlib_counter_t *my_counters;
63 uword i, j;
64
65 for (i = 0; i < vec_len (cm->counters); i++)
66 {
67 my_counters = cm->counters[i];
68
69 for (j = 0; j < vec_len (my_counters); j++)
70 {
71 my_counters[j].packets = 0;
72 my_counters[j].bytes = 0;
73 }
74 }
75}

77void
78vlib_validate_simple_counter (vlib_simple_counter_main_t * cm, u32 index)
79{
80 vlib_thread_main_t *tm = vlib_get_thread_main ();
81 int i;
82
83 vec_validate (cm->counters, tm->n_vlib_mains - 1);
84 for (i = 0; i < tm->n_vlib_mains; i++)
85 vec_validate_aligned (cm->counters[i], index, CLIB_CACHE_LINE_BYTES);
86}
87
88void
89vlib_validate_combined_counter (vlib_combined_counter_main_t * cm, u32 index)
90{
91 vlib_thread_main_t *tm = vlib_get_thread_main ();
92 int i;
93
94 vec_validate (cm->counters, tm->n_vlib_mains - 1);
95 for (i = 0; i < tm->n_vlib_mains; i++)
96 vec_validate_aligned (cm->counters[i], index, CLIB_CACHE_LINE_BYTES);
97}

99u32
100vlib_combined_counter_n_counters (const vlib_combined_counter_main_t * cm)
101{
102 ASSERT (cm->counters);
103 return (vec_len (cm->counters[0]));
104}
