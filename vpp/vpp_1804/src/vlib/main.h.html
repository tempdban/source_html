59typedef struct vlib_main_t
60{
61 CLIB_CACHE_LINE_ALIGN_MARK (cacheline0);
62 /* Instruction level timing state. */
63 clib_time_t clib_time;
64
65 /* Time stamp of last node dispatch. */
66 u64 cpu_time_last_node_dispatch;
67
68 /* Time stamp when main loop was entered (time 0). */
69 u64 cpu_time_main_loop_start;
70
71 /* Incremented once for each main loop. */
72 u32 main_loop_count;
73
74 /* Count of vectors processed this main loop. */
75 u32 main_loop_vectors_processed;
76 u32 main_loop_nodes_processed;
77
78 /* Circular buffer of input node vector counts.
79 Indexed by low bits of
80 (main_loop_count >> VLIB_LOG2_INPUT_VECTORS_PER_MAIN_LOOP). */
81 u32 vector_counts_per_main_loop[2];
82 u32 node_counts_per_main_loop[2];
83
84 /* Every so often we switch to the next counter. */
85#define VLIB_LOG2_MAIN_LOOPS_PER_STATS_UPDATE 7
86
87 /* Jump target to exit main loop with given code. */
88 u32 main_loop_exit_set;
89 /* Set e.g. in the SIGTERM signal handler, checked in a safe place... */
90 volatile u32 main_loop_exit_now;
91 clib_longjmp_t main_loop_exit;
92#define VLIB_MAIN_LOOP_EXIT_NONE 0
93#define VLIB_MAIN_LOOP_EXIT_PANIC 1

95#define VLIB_MAIN_LOOP_EXIT_CLI 2

97 /* Error marker to use when exiting main loop. */

238/* Busy wait for specified time. */
239always_inline void
240vlib_time_wait (vlib_main_t * vm, f64 wait)
241{
242 f64 t = vlib_time_now (vm);
243 f64 limit = t + wait;
244 while (t < limit)
245 t = vlib_time_now (vm);
246}
247
248/* Time a piece of code. */
249#define vlib_time_code(vm,body) \
250do { \
251 f64 _t[2]; \
252 _t[0] = vlib_time_now (vm); \
253 do { body; } while (0); \
254 _t[1] = vlib_time_now (vm); \
255 clib_warning ("%.7e", _t[1] - _t[0]); \
256} while (0)
257
258#define vlib_wait_with_timeout(vm,suspend_time,timeout_time,test) \
259({ \
260 uword __vlib_wait_with_timeout = 0; \
261 f64 __vlib_wait_time = 0; \
262 while (! (__vlib_wait_with_timeout = (test)) \
263 && __vlib_wait_time < (timeout_time)) \
264 { \
265 vlib_process_suspend (vm, suspend_time); \
266 __vlib_wait_time += suspend_time; \
267 } \
268 __vlib_wait_with_timeout; \
269})

271always_inline void
272vlib_panic_with_error (vlib_main_t * vm, clib_error_t * error)
273{
274 vm->main_loop_error = error;
275 clib_longjmp (&vm->main_loop_exit, VLIB_MAIN_LOOP_EXIT_PANIC);
276}

278#define vlib_panic_with_msg(vm,args...) \
279 vlib_panic_with_error (vm, clib_error_return (0, args))

282vlib_panic (vlib_main_t * vm)
283{
284 vlib_panic_with_error (vm, 0);

287always_inline u32
288vlib_vector_input_stats_index (vlib_main_t * vm, word delta)
289{
290 u32 i;
291 i = vm->main_loop_count >> VLIB_LOG2_MAIN_LOOPS_PER_STATS_UPDATE;
292 ASSERT (is_pow2 (ARRAY_LEN (vm->vector_counts_per_main_loop)));
293 return (i + delta) & (ARRAY_LEN (vm->vector_counts_per_main_loop) - 1);
294}
295
296/* Estimate input rate based on previous
297 2^VLIB_LOG2_MAIN_LOOPS_PER_STATS_UPDATE
298 samples. */
299always_inline u32
300vlib_last_vectors_per_main_loop (vlib_main_t * vm)
301{
302 u32 i = vlib_vector_input_stats_index (vm, -1);
303 u32 n = vm->vector_counts_per_main_loop[i];
304 return n >> VLIB_LOG2_MAIN_LOOPS_PER_STATS_UPDATE;
305}
306
307/* Total ave vector count per iteration of main loop. */
308always_inline f64
309vlib_last_vectors_per_main_loop_as_f64 (vlib_main_t * vm)
310{
311 u32 i = vlib_vector_input_stats_index (vm, -1);
312 u32 v = vm->vector_counts_per_main_loop[i];
313 return (f64) v / (f64) (1 << VLIB_LOG2_MAIN_LOOPS_PER_STATS_UPDATE);
314}
315
316/* Total ave vectors/node count per iteration of main loop. */
317always_inline f64
318vlib_last_vector_length_per_node (vlib_main_t * vm)
319{
320 u32 i = vlib_vector_input_stats_index (vm, -1);
321 u32 v = vm->vector_counts_per_main_loop[i];
322 u32 n = vm->node_counts_per_main_loop[i];
323 return n == 0 ? 0 : (f64) v / (f64) n;
324}

326extern u32 wraps;
327
328always_inline void
329vlib_increment_main_loop_counter (vlib_main_t * vm)
330{
331 u32 i, c, n, v, is_wrap;
332
333 c = vm->main_loop_count++;
334
335 is_wrap = (c & pow2_mask (VLIB_LOG2_MAIN_LOOPS_PER_STATS_UPDATE)) == 0;
336
337 if (is_wrap)
338 wraps++;
339
340 i = vlib_vector_input_stats_index (vm, /* delta */ is_wrap);
341
342 v = is_wrap ? 0 : vm->vector_counts_per_main_loop[i];
343 n = is_wrap ? 0 : vm->node_counts_per_main_loop[i];
344
345 v += vm->main_loop_vectors_processed;
346 n += vm->main_loop_nodes_processed;
347 vm->main_loop_vectors_processed = 0;
348 vm->main_loop_nodes_processed = 0;
349 vm->vector_counts_per_main_loop[i] = v;
350 vm->node_counts_per_main_loop[i] = n;
351
352 if (PREDICT_FALSE (vm->main_loop_exit_now))
353 clib_longjmp (&vm->main_loop_exit, VLIB_MAIN_LOOP_EXIT_CLI);
354}
