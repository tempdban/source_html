40#ifndef included_vlib_cli_h
41#define included_vlib_cli_h
42
43#include <vppinfra/format.h>
44
45struct vlib_cli_command_t;
46
47typedef struct
48{
49 u32 min_char;
50
51 /* Indexed by name[position] - min_char. */
52 uword **bitmaps;
53} vlib_cli_parse_position_t;
54
55typedef struct
56{
57 u8 *name;
58
59 u32 index;
60} vlib_cli_sub_command_t;
61
62typedef struct
63{
64 u8 *name;
65
66 u32 rule_index;
67
68 u32 command_index;
69} vlib_cli_sub_rule_t;
70
71typedef struct
72{
73 char *name;
74 char *short_help;
75 char *long_help;
76
77 /* Number of bytes in parsed data. Zero for vector. */
78 uword data_size;
79
80 unformat_function_t *unformat_function;
81
82 /* Opaque for unformat function. */
83 uword unformat_function_arg[2];
84} vlib_cli_parse_rule_t;
85
86/* CLI command callback function. */
87typedef clib_error_t *(vlib_cli_command_function_t)
88 (struct vlib_main_t * vm,
89 unformat_input_t * input, struct vlib_cli_command_t * cmd);
90
91typedef struct vlib_cli_command_t
92{
93 /* Command path (e.g. "show something").
94 Spaces delimit elements of path. */
95 char *path;
96
97 /* Short/long help strings. */
98 char *short_help;
99 char *long_help;
100
101 /* Callback function. */
102 vlib_cli_command_function_t *function;
103
104 /* Opaque. */
105 uword function_arg;
106
107 /* Known MP-safe? */
108 uword is_mp_safe;
109
110 /* Sub commands for this command. */
111 vlib_cli_sub_command_t *sub_commands;
112
113 /* Hash table mapping name (e.g. last path element) to sub command index. */
114 uword *sub_command_index_by_name;
115
116 /* bitmap[p][c][i] says whether sub-command i has character
117 c in position p. */
118 vlib_cli_parse_position_t *sub_command_positions;
119
120 /* Hash table mapping name (e.g. last path element) to sub rule index. */
121 uword *sub_rule_index_by_name;
122
123 /* Vector of possible parse rules for this path. */
124 vlib_cli_sub_rule_t *sub_rules;
125
126 /* List of CLI commands, built by constructors */
127 struct vlib_cli_command_t *next_cli_command;
128
129} vlib_cli_command_t;
130
131typedef void (vlib_cli_output_function_t) (uword arg,
132 u8 * buffer, uword buffer_bytes);
133typedef struct
134{
135 /* Vector of all known commands. */
136 vlib_cli_command_t *commands;
137
138 /* Hash table mapping normalized path to index into all_commands. */
139 uword *command_index_by_path;
140
141 /* Vector of all known parse rules. */
142 vlib_cli_parse_rule_t *parse_rules;
143
144 /* Hash table mapping parse rule name to index into parse_rule vector. */
145 uword *parse_rule_index_by_name;
146
147 /* Data parsed for rules. */
148 void **parse_rule_data;
149
150 /* registration list added by constructors */
151 vlib_cli_command_t *cli_command_registrations;
152} vlib_cli_main_t;
153
154#define VLIB_CLI_COMMAND(x,...) \
155 __VA_ARGS__ vlib_cli_command_t x; \
156static void __vlib_cli_command_registration_##x (void) \
157 __attribute__((__constructor__)) ; \
158static void __vlib_cli_command_registration_##x (void) \
159{ \
160 vlib_main_t * vm = vlib_get_main(); \
161 vlib_cli_main_t *cm = &vm->cli_main; \
162 x.next_cli_command = cm->cli_command_registrations; \
163 cm->cli_command_registrations = &x; \
164} \
165__VA_ARGS__ vlib_cli_command_t x
166#define VLIB_CLI_PARSE_RULE(x) \
167 vlib_cli_parse_rule_t x
168/* Output to current CLI connection. */
169void vlib_cli_output (struct vlib_main_t *vm, char *fmt, ...);
170
171/* Process CLI input. */
172void vlib_cli_input (struct vlib_main_t *vm,
173 unformat_input_t * input,
174 vlib_cli_output_function_t * function,
175 uword function_arg);
176
177clib_error_t *vlib_cli_register (struct vlib_main_t *vm,
178 vlib_cli_command_t * c);
179clib_error_t *vlib_cli_register_parse_rule (struct vlib_main_t *vm,
180 vlib_cli_parse_rule_t * c);
181
182uword unformat_vlib_cli_sub_input (unformat_input_t * i, va_list * args);
183
184/* Return an vector of strings consisting of possible auto-completions
185 * for a given input string */
186u8 **vlib_cli_get_possible_completions (u8 * input_str);
187
188#endif /* included_vlib_cli_h */
