48static clib_error_t *
49dpdk_launch_thread (void *fp, vlib_worker_thread_t * w, unsigned lcore_id)
50{
51 int r;
52 r = rte_eal_remote_launch (fp, (void *) w, lcore_id);
53 if (r)
54 return clib_error_return (0, "Failed to launch thread %u", lcore_id);
55 return 0;
56}
57
58static clib_error_t *
59dpdk_thread_set_lcore (u32 thread, u16 lcore)
60{
61 return 0;
62}
63
64static vlib_thread_callbacks_t callbacks = {
65 .vlib_launch_thread_cb = &dpdk_launch_thread,
66 .vlib_thread_set_lcore_cb = &dpdk_thread_set_lcore,
67};
68
69static clib_error_t *
70dpdk_thread_init (vlib_main_t * vm)
71{
72 vlib_thread_cb_register (vm, &callbacks);
73 return 0;
74}
75
76VLIB_INIT_FUNCTION (dpdk_thread_init);
