62static inline void
63dpdk_get_xstats (dpdk_device_t * xd)
64{
65 if (!(xd->flags & DPDK_DEVICE_FLAG_ADMIN_UP))
66 return;
67 int len;
68 if ((len = rte_eth_xstats_get (xd->device_index, NULL, 0)) > 0)
69 {
70 vec_validate (xd->xstats, len - 1);
71 vec_validate (xd->last_cleared_xstats, len - 1);
72
73 len =
74 rte_eth_xstats_get (xd->device_index, xd->xstats,
75 vec_len (xd->xstats));
76
77 ASSERT (vec_len (xd->xstats) == len);
78 ASSERT (vec_len (xd->last_cleared_xstats) == len);
79
80 _vec_len (xd->xstats) = len;
81 _vec_len (xd->last_cleared_xstats) = len;
82
83 }
84}

87static inline void
88dpdk_update_counters (dpdk_device_t * xd, f64 now)
89{
90 vlib_simple_counter_main_t *cm;
91 vnet_main_t *vnm = vnet_get_main ();
92 u32 thread_index = vlib_get_thread_index ();
93 u64 rxerrors, last_rxerrors;
94
95 /* only update counters for PMD interfaces */
96 if ((xd->flags & DPDK_DEVICE_FLAG_PMD) == 0)
97 return;
98
99 xd->time_last_stats_update = now ? now : xd->time_last_stats_update;
100 clib_memcpy (&xd->last_stats, &xd->stats, sizeof (xd->last_stats));
101 rte_eth_stats_get (xd->device_index, &xd->stats);
102
103 /* maybe bump interface rx no buffer counter */
104 if (PREDICT_FALSE (xd->stats.rx_nombuf != xd->last_stats.rx_nombuf))
105 {
106 cm = vec_elt_at_index (vnm->interface_main.sw_if_counters,
107 VNET_INTERFACE_COUNTER_RX_NO_BUF);
108
109 vlib_increment_simple_counter (cm, thread_index, xd->vlib_sw_if_index,
110 xd->stats.rx_nombuf -
111 xd->last_stats.rx_nombuf);
112 }
113
114 /* missed pkt counter */
115 if (PREDICT_FALSE (xd->stats.imissed != xd->last_stats.imissed))
116 {
117 cm = vec_elt_at_index (vnm->interface_main.sw_if_counters,
118 VNET_INTERFACE_COUNTER_RX_MISS);
119
120 vlib_increment_simple_counter (cm, thread_index, xd->vlib_sw_if_index,
121 xd->stats.imissed -
122 xd->last_stats.imissed);
123 }
124 rxerrors = xd->stats.ierrors;
125 last_rxerrors = xd->last_stats.ierrors;
126
127 if (PREDICT_FALSE (rxerrors != last_rxerrors))
128 {
129 cm = vec_elt_at_index (vnm->interface_main.sw_if_counters,
130 VNET_INTERFACE_COUNTER_RX_ERROR);
131
132 vlib_increment_simple_counter (cm, thread_index, xd->vlib_sw_if_index,
133 rxerrors - last_rxerrors);
134 }
135
136 dpdk_get_xstats (xd);
137}
