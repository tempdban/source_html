267static_always_inline void *
268memif_get_buffer (memif_if_t * mif, memif_ring_t * ring, u16 slot)
269{
270 u16 region = ring->desc[slot].region;
271 return mif->regions[region].shm + ring->desc[slot].offset;
272}
