21/*
22 * ported from an unspecified Cisco cdp implementation.
23 * Compute / return in HOST byte order. 0 => good checksum.
24 */
25u16
26cdp_checksum (void *p, int count)
27{
28 u32 sum;
29 u16 i, *data;
30
31 data = p;
32 sum = 0;
33 while (count > 1)
34 {
35 sum += ntohs (*data);
36 data++;
37 count -= 2;
38 }
39
40 if (count > 0)
41 sum += *(char *) data;
42
43 while (sum >> 16)
44 {
45 sum = (sum & 0xFFFF) + (sum >> 16);
46 }
47
48 i = (i16) sum;
49 return (~i);
50}

52/* TLV handler table */
53typedef struct
54{
55 char *name;
56 u32 tlv_id;
57 void *format;
58 void *process;
59} tlv_handler_t;

63/* Display a generic TLV as a set of hex bytes */
64static u8 *
65format_generic_tlv (u8 * s, va_list * va)
66{
67 cdp_tlv_t *t = va_arg (*va, cdp_tlv_t *);
68 tlv_handler_t *h = &tlv_handlers[t->t];
69
70 s = format (s, "%s(%d): %U\n", h->name,
71 t->t, format_hex_bytes, t->v, t->l - sizeof (*t));
72 return s;
73}

86/* print a text tlv */
87static u8 *
88format_text_tlv (u8 * s, va_list * va)
89{
90 cdp_tlv_t *t = va_arg (*va, cdp_tlv_t *);
91 tlv_handler_t *h = &tlv_handlers[t->t];
92 int i;
93
94 s = format (s, "%s(%d): ", h->name, t->t);
95
96 for (i = 0; i < (t->l - sizeof (*t)); i++)
97 vec_add1 (s, t->v[i]);
98
99 vec_add1 (s, '\n');
100 return s;
101}

224static u8 *
225format_cdp_hdr (u8 * s, va_list * va)
226{
227 cdp_hdr_t *h = va_arg (*va, cdp_hdr_t *);
228
229 s = format (s, "version %d, ttl %d(secs), cksum 0x%04x\n",
230 h->version, h->ttl, h->checksum);
231 return s;
232}
233
234static cdp_error_t
235process_cdp_hdr (cdp_main_t * cm, cdp_neighbor_t * n, cdp_hdr_t * h)
236{
237#if DEBUG_TLV_DUMP > 0
238 fformat (stdout, "%U", format_cdp_hdr, h);
239#endif
240
241 if (h->version != 1 && h->version != 2)
242 return CDP_ERROR_PROTOCOL_VERSION;
243
244 n->ttl_in_seconds = h->ttl;
245
246 return CDP_ERROR_NONE;
247}

249/* scan a cdp packet; header, then tlv's */
250static int
251cdp_packet_scan (cdp_main_t * cm, cdp_neighbor_t * n)
252{
253 u8 *cur = n->last_rx_pkt;
254 cdp_hdr_t *h;
255 cdp_tlv_t *tlv;
256 cdp_error_t e = CDP_ERROR_NONE;
257 tlv_handler_t *handler;
258 cdp_error_t (*fp) (cdp_main_t *, cdp_neighbor_t *, cdp_tlv_t *);
259 u16 computed_checksum;
260
261 computed_checksum = cdp_checksum (cur, vec_len (cur));
262
263 if (computed_checksum)
264 return CDP_ERROR_CHECKSUM;
265
266 h = (cdp_hdr_t *) cur;
267
268 e = process_cdp_hdr (cm, n, h);
269 if (e)
270 return e;
271
272 cur = (u8 *) (h + 1);
273
274 while (cur < n->last_rx_pkt + vec_len (n->last_rx_pkt) - 1)
275 {
276 tlv = (cdp_tlv_t *) cur;
277 tlv->t = ntohs (tlv->t);
278 tlv->l = ntohs (tlv->l);
279 if (tlv->t >= ARRAY_LEN (tlv_handlers))
280 return CDP_ERROR_BAD_TLV;
281 handler = &tlv_handlers[tlv->t];
282 fp = handler->process;
283 e = (*fp) (cm, n, tlv);
284 if (e)
285 return e;
286 /* tlv length includes (t, l) */
287 cur += tlv->l;
288 }
289
290 return CDP_ERROR_NONE;
291}

293/*
294 * cdp input routine
295 */
296cdp_error_t
297cdp_input (vlib_main_t * vm, vlib_buffer_t * b0, u32 bi0)
298{
299 cdp_main_t *cm = &cdp_main;
300 cdp_neighbor_t *n;
301 uword *p, nbytes;
302 cdp_error_t e;
303 uword last_packet_signature;
304
305 /* find or create a neighbor pool entry for the (sw) interface
306 upon which we received this pkt */
307 p = hash_get (cm->neighbor_by_sw_if_index,
308 vnet_buffer (b0)->sw_if_index[VLIB_RX]);
309
310 if (p == 0)
311 {
312 pool_get (cm->neighbors, n);
313 memset (n, 0, sizeof (*n));
314 n->sw_if_index = vnet_buffer (b0)->sw_if_index[VLIB_RX];
315 n->packet_template_index = (u8) ~ 0;
316 hash_set (cm->neighbor_by_sw_if_index, n->sw_if_index,
317 n - cm->neighbors);
318 }
319 else
320 {
321 n = pool_elt_at_index (cm->neighbors, p[0]);
322 }
323
324 /*
325 * typical clib idiom. Don't repeatedly allocate and free
326 * the per-neighbor rx buffer. Reset its apparent length to zero
327 * and reuse it.
328 */
329
330 if (n->last_rx_pkt)
331 _vec_len (n->last_rx_pkt) = 0;
332
333 /* cdp disabled on this interface, we're done */
334 if (n->disabled)
335 return CDP_ERROR_DISABLED;
336
337 /*
338 * Make sure the per-neighbor rx buffer is big enough to hold
339 * the data we're about to copy
340 */
341 vec_validate (n->last_rx_pkt, vlib_buffer_length_in_chain (vm, b0) - 1);
342
343 /*
344 * Coalesce / copy e the buffer chain into the per-neighbor
345 * rx buffer
346 */
347 nbytes = vlib_buffer_contents (vm, bi0, n->last_rx_pkt);
348 ASSERT (nbytes <= vec_len (n->last_rx_pkt));
349
350 /*
351 * Compute Jenkins hash of the new packet, decide if we need to
352 * actually parse through the TLV's. CDP packets are all identical,
353 * so unless we time out the peer, we don't need to process the packet.
354 */
355 last_packet_signature =
356 hash_memory (n->last_rx_pkt, vec_len (n->last_rx_pkt), 0xd00b);
357
358 if (n->last_packet_signature_valid &&
359 n->last_packet_signature == last_packet_signature)
360 {
361 e = CDP_ERROR_CACHE_HIT;
362 }
363 else
364 {
365 /* Actually scan the packet */
366 e = cdp_packet_scan (cm, n);
367 n->last_packet_signature_valid = 1;
368 n->last_packet_signature = last_packet_signature;
369 }
370
371 if (e == CDP_ERROR_NONE)
372 {
373 n->last_heard = vlib_time_now (vm);
374 }
375
376 return e;
377}

379/*
380 * setup neighbor hash table
381 */
382static clib_error_t *
383cdp_input_init (vlib_main_t * vm)
384{
385 clib_error_t *error;
386 cdp_main_t *cm = &cdp_main;
387 void vnet_cdp_node_reference (void);
388
389 vnet_cdp_node_reference ();
390
391 if ((error = vlib_call_init_function (vm, cdp_periodic_init)))
392 return error;
393
394 cm->vlib_main = vm;
395 cm->vnet_main = vnet_get_main ();
396 cm->neighbor_by_sw_if_index = hash_create (0, sizeof (uword));
397
398 return 0;
399}

404static u8 *
405format_cdp_neighbors (u8 * s, va_list * va)
406{
407 CLIB_UNUSED (vlib_main_t * vm) = va_arg (*va, vlib_main_t *);
408 cdp_main_t *cm = va_arg (*va, cdp_main_t *);
409 vnet_main_t *vnm = &vnet_main;
410 cdp_neighbor_t *n;
411 vnet_hw_interface_t *hw;
412
413 s = format (s,
414 "%=25s %=15s %=25s %=10s\n",
415 "Our Port", "Peer System", "Peer Port", "Last Heard");
416
417 /* *INDENT-OFF* */
418 pool_foreach (n, cm->neighbors,
419 ({
420 hw = vnet_get_sup_hw_interface (vnm, n->sw_if_index);
421
422 if (n->disabled == 0)
423 s = format (s, "%=25s %=15s %=25s %=10.1f\n",
424 hw->name, n->device_name, n->port_id,
425 n->last_heard);
426 }));
427 /* *INDENT-ON* */
428 return s;
429}

454/*
455 * packet trace format function, very similar to
456 * cdp_packet_scan except that we call the per TLV format
457 * functions instead of the per TLV processing functions
458 */
459u8 *
460cdp_input_format_trace (u8 * s, va_list * args)
461{
462 CLIB_UNUSED (vlib_main_t * vm) = va_arg (*args, vlib_main_t *);
463 CLIB_UNUSED (vlib_node_t * node) = va_arg (*args, vlib_node_t *);
464 cdp_input_trace_t *t = va_arg (*args, cdp_input_trace_t *);
465 u8 *cur;
466 cdp_hdr_t *h;
467 cdp_tlv_t *tlv;
468 tlv_handler_t *handler;
469 u8 *(*fp) (cdp_tlv_t *);
470
471 cur = t->data;
472
473 h = (cdp_hdr_t *) cur;
474 s = format (s, "%U", format_cdp_hdr, h);
475
476 cur = (u8 *) (h + 1);
477
478 while (cur < t->data + t->len)
479 {
480 tlv = (cdp_tlv_t *) cur;
481 tlv->t = ntohs (tlv->t);
482 tlv->l = ntohs (tlv->l);
483 if (tlv->t >= ARRAY_LEN (tlv_handlers))
484 {
485 s = format (s, "BAD_TLV\n");
486 break;
487 }
488 handler = &tlv_handlers[tlv->t];
489 fp = handler->format;
490 s = format (s, " %U", fp, tlv);
491 /* tlv length includes (t, l) */
492 cur += tlv->l;
493 }
494
495 return s;
496}
