33typedef enum
34{
35 CDP_PACKET_TEMPLATE_ETHERNET,
36 CDP_PACKET_TEMPLATE_HDLC,
37 CDP_PACKET_TEMPLATE_SRP,
38 CDP_N_PACKET_TEMPLATES,
39} cdp_packet_template_id_t;
40
41typedef struct
42{
43 /* neighbor's vlib software interface index */
44 u32 sw_if_index;
45
46 /* Timers */
47 f64 last_heard;
48 f64 last_sent;
49
50 /* Neighbor time-to-live (usually 180s) */
51 u8 ttl_in_seconds;
52
53 /* "no cdp run" or similar */
54 u8 disabled;
55
56 /* tx packet template id for this neighbor */
57 u8 packet_template_index;
58
59 /* Jenkins hash optimization: avoid tlv scan, send short keepalive msg */
60 u8 last_packet_signature_valid;
61 uword last_packet_signature;
62
63 /* Info we actually keep about each neighbor */
64 u8 *device_name;
65 u8 *version;
66 u8 *port_id;
67 u8 *platform;
68
69 /* last received packet, for the J-hash optimization */
70 u8 *last_rx_pkt;
71} cdp_neighbor_t;
72
73#define foreach_neighbor_string_field \
74_(device_name) \
75_(version) \
76_(port_id) \
77_(platform)
78
79typedef struct
80{
81 /* pool of cdp neighbors */
82 cdp_neighbor_t *neighbors;
83
84 /* plugin message id base */
85 u16 msg_id_base;
86
87 /* tx pcap debug enable */
88 u8 tx_pcap_debug;
89
90 /* rapidly find a neighbor by vlib software interface index */
91 uword *neighbor_by_sw_if_index;
92
93 /* Background process node index */
94 u32 cdp_process_node_index;
95
96 /* top-level state */
97 int enabled;
98
99 /* Packet templates for different encap types */
100 vlib_packet_template_t packet_templates[CDP_N_PACKET_TEMPLATES];
101
102 /* convenience variables */
103 vlib_main_t *vlib_main;
104 vnet_main_t *vnet_main;
105} cdp_main_t;
106
107extern cdp_main_t cdp_main;
108extern vlib_node_registration_t cdp_process_node;
109
110/* Packet counters */
111#define foreach_cdp_error \
112_ (NONE, "good cdp packets (processed)") \
113_ (CACHE_HIT, "good cdp packets (cache hit)") \
114_ (BAD_TLV, "cdp packets with bad TLVs") \
115_ (PROTOCOL_VERSION, "cdp packets with bad protocol versions") \
116_ (CHECKSUM, "cdp packets with bad checksums") \
117_ (DISABLED, "cdp packets received on disabled interfaces")
118
119typedef enum
120{
121#define _(sym,str) CDP_ERROR_##sym,
122 foreach_cdp_error
123#undef _
124 CDP_N_ERROR,
125} cdp_error_t;

134typedef enum
135{
136 CDP_EVENT_ENABLE,
137 CDP_EVENT_DISABLE,
138} cdp_process_event_t;

140cdp_error_t cdp_input (vlib_main_t * vm, vlib_buffer_t * b0, u32 bi0);

143u16 cdp_checksum (void *p, int count);
144u8 *cdp_input_format_trace (u8 * s, va_list * args);
