25#define foreach_nat64_tcp_ses_state \
26 _(0, CLOSED, "closed") \
27 _(1, V4_INIT, "v4-init") \
28 _(2, V6_INIT, "v6-init") \
29 _(3, ESTABLISHED, "established") \
30 _(4, V4_FIN_RCV, "v4-fin-rcv") \
31 _(5, V6_FIN_RCV, "v6-fin-rcv") \
32 _(6, V6_FIN_V4_FIN_RCV, "v6-fin-v4-fin-rcv") \
33 _(7, TRANS, "trans")
34
35typedef enum
36{
37#define _(v, N, s) NAT64_TCP_STATE_##N = v,
38 foreach_nat64_tcp_ses_state
39#undef _
40} nat64_tcp_ses_state_t;

68typedef struct
69{
70 /** Interface pool */
71 snat_interface_t *interfaces;
72
73 /** Address pool vector */

76 /** sw_if_indices whose interface addresses should be auto-added */
77 u32 *auto_add_sw_if_indices;

79 /** Pref64 vector */
80 nat64_prefix_t *pref64;

86 u32 fq_in2out_index;
87 u32 fq_out2in_index;
88
89 /** Pool of static BIB entries to be added/deleted in worker threads */

91
92 u32 error_node_index;

94 /** config parameters */
95 u32 bib_buckets;
96 u32 bib_memory_size;

105 u32 tcp_incoming_syn_timeout;
106
107 /* Total count of interfaces enabled */

110 u32 nat64_expire_walk_node_index;

120/**
121 * @brief Add/delete address to NAT64 pool.
122 *
123 * @param addr IPv4 address.
124 * @param vrf_id VRF id of tenant, ~0 means independent of VRF.
125 * @param is_add 1 if add, 0 if delete.
126 *
127 * @returns 0 on success, non-zero value otherwise.
128 */
129int nat64_add_del_pool_addr (ip4_address_t * addr, u32 vrf_id, u8 is_add);
130
131/**
132 * @brief Call back function when walking addresses in NAT64 pool, non-zero
133 * return value stop walk.
134 */
135typedef int (*nat64_pool_addr_walk_fn_t) (snat_address_t * addr, void *ctx);
136
137/**
138 * @brief Walk NAT64 pool.
139 *
140 * @param fn The function to invoke on each entry visited.
141 * @param ctx A context passed in the visit function.
142 */
143void nat64_pool_addr_walk (nat64_pool_addr_walk_fn_t fn, void *ctx);
144
145/**
146 * @brief NAT64 pool address from specific (DHCP addressed) interface.
147 *
148 * @param sw_if_index Index of the interface.
149 * @param is_add 1 if add, 0 if delete.
150 *
151 * @returns 0 on success, non-zero value otherwise.
152 */
153int nat64_add_interface_address (u32 sw_if_index, int is_add);
154
155/**
156 * @brief Enable/disable NAT64 feature on the interface.
157 *
158 * @param sw_if_index Index of the interface.
159 * @param is_inside 1 if inside, 0 if outside.
160 * @param is_add 1 if add, 0 if delete.
161 *
162 * @returns 0 on success, non-zero value otherwise.
163 */
164int nat64_add_del_interface (u32 sw_if_index, u8 is_inside, u8 is_add);
165
166/**
167 * @brief Call back function when walking interfaces with NAT64 feature,
168 * non-zero return value stop walk.
169 */
170typedef int (*nat64_interface_walk_fn_t) (snat_interface_t * i, void *ctx);
171
172/**
173 * @brief Walk NAT64 interfaces.
174 *
175 * @param fn The function to invoke on each entry visited.
176 * @param ctx A context passed in the visit function.
177 */
178void nat64_interfaces_walk (nat64_interface_walk_fn_t fn, void *ctx);

187clib_error_t *nat64_init (vlib_main_t * vm);

189/**
190 * @brief Add/delete static NAT64 BIB entry.
191 *
192 * @param in_addr Inside IPv6 address.
193 * @param out_addr Outside IPv4 address.
194 * @param in_port Inside port number.
195 * @param out_port Outside port number.
196 * @param proto L4 protocol.
197 * @param vrf_id VRF id of tenant.
198 * @param is_add 1 if add, 0 if delete.
199 *
200 * @returns 0 on success, non-zero value otherwise.
201 */
202int nat64_add_del_static_bib_entry (ip6_address_t * in_addr,
203 ip4_address_t * out_addr, u16 in_port,
204 u16 out_port, u8 proto, u32 vrf_id,
205 u8 is_add);
206
207/**
208 * @brief Alloce IPv4 address and port pair from NAT64 pool.
209 *
210 * @param fib_index FIB index of tenant.
211 * @param proto L4 protocol.
212 * @param addr Allocated IPv4 address.
213 * @param port Allocated port number.
214 * @param thread_index Thread index.
215 *
216 * @returns 0 on success, non-zero value otherwise.
217 */
218int nat64_alloc_out_addr_and_port (u32 fib_index, snat_protocol_t proto,
219 ip4_address_t * addr, u16 * port,
220 u32 thread_index);

255 * @brief Set TCP session timeouts.
