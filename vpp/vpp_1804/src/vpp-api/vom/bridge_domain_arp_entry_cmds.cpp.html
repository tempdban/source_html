16#include "vom/bridge_domain_arp_entry_cmds.hpp"
17
18namespace VOM {
19namespace bridge_domain_arp_entry_cmds {
20
21create_cmd::create_cmd(HW::item<bool>& item,
22 uint32_t bd,
23 const mac_address_t& mac,
24 const boost::asio::ip::address& ip_addr)
25 : rpc_cmd(item)
26 , m_bd(bd)
27 , m_mac(mac)
28 , m_ip_addr(ip_addr)
29{
30}
31
32bool
33create_cmd::operator==(const create_cmd& other) const
34{
35 return ((m_mac == other.m_mac) && (m_ip_addr == other.m_ip_addr) &&
36 (m_bd == other.m_bd));
37}
38
39rc_t
40create_cmd::issue(connection& con)
41{
42 msg_t req(con.ctx(), std::ref(*this));
43
44 auto& payload = req.get_request().get_payload();
45 payload.bd_id = m_bd;
46 payload.is_add = 1;
47 m_mac.to_bytes(payload.mac_address, 6);
48 to_bytes(m_ip_addr, &payload.is_ipv6, payload.ip_address);
49
50 VAPI_CALL(req.execute());
51
52 m_hw_item.set(wait());
53
54 return rc_t::OK;
55}
56
57std::string
58create_cmd::to_string() const
59{
60 std::ostringstream s;
61 s << "bridge-domain-arp-entry-create: " << m_hw_item.to_string()
62 << " bd:" << m_bd << " mac:" << m_mac.to_string()
63 << " ip:" << m_ip_addr.to_string();
64
65 return (s.str());
66}
67
68delete_cmd::delete_cmd(HW::item<bool>& item,
69 uint32_t bd,
70 const mac_address_t& mac,
71 const boost::asio::ip::address& ip_addr)
72 : rpc_cmd(item)
73 , m_bd(bd)
74 , m_mac(mac)
75 , m_ip_addr(ip_addr)
76{
77}
78
79bool
80delete_cmd::operator==(const delete_cmd& other) const
81{
82 return ((m_mac == other.m_mac) && (m_ip_addr == other.m_ip_addr) &&
83 (m_bd == other.m_bd));
84}
85
86rc_t
87delete_cmd::issue(connection& con)
88{
89 msg_t req(con.ctx(), std::ref(*this));
90
91 auto& payload = req.get_request().get_payload();
92 payload.bd_id = m_bd;
93 payload.is_add = 0;
94 m_mac.to_bytes(payload.mac_address, 6);
95 to_bytes(m_ip_addr, &payload.is_ipv6, payload.ip_address);
96
97 VAPI_CALL(req.execute());
98
99 wait();
100 m_hw_item.set(rc_t::NOOP);
101
102 return rc_t::OK;
103}
104
105std::string
106delete_cmd::to_string() const
107{
108 std::ostringstream s;
109 s << "bridge-domain-arp-entry-delete: " << m_hw_item.to_string()
110 << " bd:" << m_bd << " mac:" << m_mac.to_string()
111 << " ip:" << m_ip_addr.to_string();
112
113 return (s.str());
114}
