16#ifndef __VOM_INST_DB_FUNCS_H__
17#define __VOM_INST_DB_FUNCS_H__
18
19#include <ostream>
20
21#include "singular_db.hpp"
22
23/**
24 * A set of helper function to iterate over objects in the DB.
25 * These functions are delcared not as DB member functions so that
26 * the template instatiation of the DB does not require the definitions
27 * of the functions used to be declared.
28 */
29namespace VOM {
30/**
31 * Print each of the objects in the DB into the stream provided
32 */
33template <typename DB>
34void
35db_dump(const DB& db, std::ostream& os)
36{
37 for (const auto entry : db) {
38 os << "key: " << entry.first << std::endl;
39 os << " " << entry.second.lock()->to_string() << std::endl;
40 }
41}
42};
