41interface::type_t
42interface::type_t::from_string(const std::string& str)
43{
44 if ((str.find("Virtual") != std::string::npos) ||
45 (str.find("vhost") != std::string::npos) ||
46 (str.find("vhu") != std::string::npos)) {
47 return interface::type_t::VHOST;
48 } else if (str.find("Bond") != std::string::npos) {
49 return interface::type_t::BOND;
50 } else if (str.find("Ethernet") != std::string::npos) {
51 return interface::type_t::ETHERNET;
52 } else if (str.find("vxlan") != std::string::npos) {
53 return interface::type_t::VXLAN;
54 } else if (str.find("loop") != std::string::npos) {
55 return interface::type_t::LOOPBACK;
56 } else if (str.find("host-") != std::string::npos) {
57 return interface::type_t::AFPACKET;
58 } else if (str.find("local") != std::string::npos) {
59 return interface::type_t::LOCAL;
60 } else if (str.find("tap") != std::string::npos) {
61 return interface::type_t::TAP;
62 } else if (str.find("bvi") != std::string::npos) {
63 return interface::type_t::BVI;
64 }
65
66 return interface::type_t::UNKNOWN;
67}

89interface::admin_state_t
90interface::admin_state_t::from_int(uint8_t v)
91{
92 if (0 == v) {
93 return (interface::admin_state_t::DOWN);
94 }
95 return (interface::admin_state_t::UP);
96}
97
98interface::oper_state_t
99interface::oper_state_t::from_int(uint8_t v)
100{
101 if (0 == v) {
102 return (interface::oper_state_t::DOWN);
103 }
104 return (interface::oper_state_t::UP);
105}
