19#ifndef __included_plugin_h__
20#define __included_plugin_h__

22#include <sys/types.h>
23#include <sys/stat.h>
24#include <unistd.h>
25
26typedef struct
27{
28 u8 *name;
29 struct stat file_info;
30 void *handle;
31} plugin_info_t;
32
33typedef struct
34{
35 /* loaded plugin info */
36 plugin_info_t *plugin_info;
37 uword *plugin_by_name_hash;
38
39 /* path and name filter */
40 u8 *plugin_path;
41 u8 *plugin_name_filter;
42
43 /* convenience */
44 vat_main_t *vat_main;
45
46} plugin_main_t;
47
48extern plugin_main_t vat_plugin_main;
49
50int vat_plugin_init (vat_main_t * vam);
51int vat_load_new_plugins (plugin_main_t * pm);
