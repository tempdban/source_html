39static inline u32
40vl_msg_api_handle_get_epoch (u32 index)
41{
42 return (index & VL_API_EPOCH_MASK);
43}
44
45static inline u32
46vl_msg_api_handle_get_index (u32 index)
47{
48 return (index >> VL_API_EPOCH_SHIFT);
49}
50
51static inline u32
52vl_msg_api_handle_from_index_and_epoch (u32 index, u32 epoch)
53{
54 u32 handle;
55 ASSERT (index < 0x00FFFFFF);
56
57 handle = (index << VL_API_EPOCH_SHIFT) | (epoch & VL_API_EPOCH_MASK);
58 return handle;
59}
60
61static inline u8
62vl_msg_api_handle_is_valid (u32 handle, u32 restarts)
63{
64 u32 epoch = vl_msg_api_handle_get_epoch (handle);
65 return ((restarts & VL_API_EPOCH_MASK) == epoch);
66}
