38#ifndef included_clib_random_buffer_h
39#define included_clib_random_buffer_h
40
41#include <vppinfra/clib.h>
42#include <vppinfra/random_isaac.h>
43
44typedef struct
45{
46 /* Two parallel ISAAC contexts for speed. */
47 isaac_t ctx[2];
48
49 /* Random buffer. */
50 uword *buffer;
51
52 /* Cache up to 1 word worth of bytes for random data
53 less than one word at a time. */
54 uword n_cached_bytes;
55
56 union
57 {
58 u8 cached_bytes[sizeof (uword)];
59 uword cached_word;
60 };
61}
62clib_random_buffer_t;
63
64always_inline void
65clib_random_buffer_free (clib_random_buffer_t * b)
66{
67 vec_free (b->buffer);
68}
69
70/* Fill random buffer. */
71void clib_random_buffer_fill (clib_random_buffer_t * b, uword n_words);
72
73/* Initialize random buffer. */
74void clib_random_buffer_init (clib_random_buffer_t * b, uword seed);
75
76/* Returns word aligned random data, possibly filling buffer. */
77always_inline void *
78clib_random_buffer_get_data (clib_random_buffer_t * b, uword n_bytes)
79{
80 uword n_words, i, l;
81
82 l = b->n_cached_bytes;
83 if (n_bytes <= l)
84 {
85 b->n_cached_bytes = l - n_bytes;
86 return &b->cached_bytes[l];
87 }
88
89 n_words = n_bytes / sizeof (uword);
90 if (n_bytes % sizeof (uword))
91 n_words++;
92
93 /* Enough random words left? */
94 if (PREDICT_FALSE (n_words > vec_len (b->buffer)))
95 clib_random_buffer_fill (b, n_words);
96
97 i = vec_len (b->buffer) - n_words;
98 _vec_len (b->buffer) = i;
99
100 if (n_bytes < sizeof (uword))
101 {
102 b->cached_word = b->buffer[i];
103 b->n_cached_bytes = sizeof (uword) - n_bytes;
104 return b->cached_bytes;
105 }
106 else
107 return b->buffer + i;
108}
109
110#endif /* included_clib_random_buffer_h */
