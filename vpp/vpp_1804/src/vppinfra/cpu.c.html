72u8 *
73format_cpu_uarch (u8 * s, va_list * args)
74{
75#if __x86_64__
76 u32 __attribute__ ((unused)) eax, ebx, ecx, edx;
77 u8 model, family;
78
79 if (__get_cpuid (1, &eax, &ebx, &ecx, &edx) == 0)
80 return format (s, "unknown (missing cpuid)");
81
82 model = ((eax >> 4) & 0x0f) | ((eax >> 12) & 0xf0);
83 family = (eax >> 8) & 0x0f;
84
85#define _(f,m,a,c) if ((model == m) && (family == f)) return format(s, "%s (%s)", a, c);
86 foreach_x86_cpu_uarch
87#undef _
88 return format (s, "unknown (family 0x%02x model 0x%02x)", family, model);
89
90#elif __aarch64__
91 int fd;
92 unformat_input_t input;
93 u32 implementer, primary_part_number, variant, revision;
94
95 fd = open ("/proc/cpuinfo", 0);
96 if (fd < 0)
97 return format (s, "unknown");
98
99 unformat_init_clib_file (&input, fd);
100 while (unformat_check_input (&input) != UNFORMAT_END_OF_INPUT)
101 {
102 if (unformat (&input, "CPU implementer%_: 0x%x", &implementer))
103 ;
104 else if (unformat (&input, "CPU part%_: 0x%x", &primary_part_number))
105 ;
106 else if (unformat (&input, "CPU variant%_: 0x%x", &variant))
107 ;
108 else if (unformat (&input, "CPU revision%_: %u", &revision))
109 ;
110 else
111 unformat_skip_line (&input);
112 }
113 unformat_free (&input);
114 close (fd);
115
116 /* Note: Cavium starts counting variants from 1 instead of 0 */
117 if (implementer == 0x43)
118 variant++;
119
120#define _(i,p,a,c) if ((implementer == i) && (primary_part_number == p)) \
121 return format(s, "%s (%s PASS %u.%u)", a, c, variant, revision);
122 foreach_aarch64_cpu_uarch
123#undef _
124 return format (s, "unknown (implementer 0x%02x part 0x%03x PASS %u.%u)",
125 implementer, primary_part_number, variant, revision);
126
127#else /* ! __x86_64__ */
128 return format (s, "unknown");
129#endif
130}

132u8 *
133format_cpu_model_name (u8 * s, va_list * args)
134{
135#if __x86_64__
136 u32 __attribute__ ((unused)) eax, ebx, ecx, edx;
137 u8 *name = 0;
138 u32 *name_u32;
139
140 if (__get_cpuid (1, &eax, &ebx, &ecx, &edx) == 0)
141 return format (s, "unknown (missing cpuid)");
142
143 __get_cpuid (0x80000000, &eax, &ebx, &ecx, &edx);
144 if (eax < 0x80000004)
145 return format (s, "unknown (missing ext feature)");
146
147 vec_validate (name, 48);
148 name_u32 = (u32 *) name;
149
150 __get_cpuid (0x80000002, &eax, &ebx, &ecx, &edx);
151 name_u32[0] = eax;
152 name_u32[1] = ebx;
153 name_u32[2] = ecx;
154 name_u32[3] = edx;
155
156 __get_cpuid (0x80000003, &eax, &ebx, &ecx, &edx);
157 name_u32[4] = eax;
158 name_u32[5] = ebx;
159 name_u32[6] = ecx;
160 name_u32[7] = edx;
161
162 __get_cpuid (0x80000004, &eax, &ebx, &ecx, &edx);
163 name_u32[8] = eax;
164 name_u32[9] = ebx;
165 name_u32[10] = ecx;
166 name_u32[11] = edx;
167
168 s = format (s, "%s", name);
169 vec_free (name);
170 return s;
171
172#elif defined(__aarch64__)
173 return format (s, "armv8");
174#else /* ! __x86_64__ */
175 return format (s, "unknown");
176#endif
177}

180static inline char const *
181flag_skip_prefix (char const *flag)
182{
183 if (memcmp (flag, "x86_", sizeof ("x86_") - 1) == 0)
184 return flag + sizeof ("x86_") - 1;
185 if (memcmp (flag, "aarch64_", sizeof ("aarch64_") - 1) == 0)
186 return flag + sizeof ("aarch64_") - 1;
187 return flag;
188}

190u8 *
191format_cpu_flags (u8 * s, va_list * args)
192{
193#if defined(__x86_64__)
194#define _(flag, func, reg, bit) \
195 if (clib_cpu_supports_ ## flag()) \
196 s = format (s, "%s ", flag_skip_prefix(#flag));
197 foreach_x86_64_flags return s;
198#undef _
199#elif defined(__aarch64__)
200#define _(flag, bit) \
201 if (clib_cpu_supports_ ## flag()) \
202 s = format (s, "%s ", flag_skip_prefix(#flag));
203 foreach_aarch64_flags return s;
204#undef _
205#else /* ! ! __x86_64__ && ! __aarch64__ */
206 return format (s, "unknown");
207#endif
208}
