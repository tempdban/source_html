38#ifndef included_mheap_h
39#define included_mheap_h
40
41#include <vppinfra/vec.h>
42#include <vppinfra/error.h> /* clib_error_t */
43#include <vppinfra/mem.h> /* clib_mem_usage_t */
44#include <vppinfra/format.h> /* for unformat_input_t */
45
46/* Allocate size bytes. New heap and offset are returned.
47 offset == ~0 means allocation failed. */
48always_inline void *
49mheap_get (void *v, uword size, uword * offset_return)
50{
51 return mheap_get_aligned (v, size, 0, 0, offset_return);
52}
53
54/* Create allocation heap of given size.
55 * The actual usable size is smaller than the requested size.
56 * memory_bytes must be greater than mheap_page_size + sizeof (mheap_t) + 16.
57 * Otherwise, allocation may fail and return 0.
58 */
59void *mheap_alloc (void *memory, uword memory_bytes);
60void *mheap_alloc_with_flags (void *memory, uword memory_bytes, uword flags);
61
62#define mheap_free(v) (v) = _mheap_free(v)
63void *_mheap_free (void *v);
64
65void mheap_foreach (void *v,
66 uword (*func) (void *arg, void *v, void *elt_data,
67 uword elt_size), void *arg);
68
69/* Format mheap data structures as string. */
70u8 *format_mheap (u8 * s, va_list * va);
71
72/* Validate internal consistency. */
73void mheap_validate (void *h);
74
75/* Query bytes used. */
76uword mheap_bytes (void *v);
77
78void mheap_usage (void *v, clib_mem_usage_t * usage);
79
80/* Enable disable traceing. */
81void mheap_trace (void *v, int enable);
82
83/* Test routine. */
84int test_mheap_main (unformat_input_t * input);
85
86#endif /* included_mheap_h */
