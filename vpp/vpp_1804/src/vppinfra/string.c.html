41/* Exchanges source and destination. */
42void
43clib_memswap (void *_a, void *_b, uword bytes)
44{
45 uword pa = pointer_to_uword (_a);
46 uword pb = pointer_to_uword (_b);
47
48#define _(TYPE) \
49 if (0 == ((pa | pb) & (sizeof (TYPE) - 1))) \
50 { \
51 TYPE * a = uword_to_pointer (pa, TYPE *); \
52 TYPE * b = uword_to_pointer (pb, TYPE *); \
53 \
54 while (bytes >= 2*sizeof (TYPE)) \
55 { \
56 TYPE a0, a1, b0, b1; \
57 bytes -= 2*sizeof (TYPE); \
58 a += 2; \
59 b += 2; \
60 a0 = a[-2]; a1 = a[-1]; \
61 b0 = b[-2]; b1 = b[-1]; \
62 a[-2] = b0; a[-1] = b1; \
63 b[-2] = a0; b[-1] = a1; \
64 } \
65 pa = pointer_to_uword (a); \
66 pb = pointer_to_uword (b); \
67 }
68
69 if (BITS (uword) == BITS (u64))
70 _(u64);
71 _(u32);
72 _(u16);
73 _(u8);
74
75#undef _
76
77 ASSERT (bytes < 2);
78 if (bytes)
79 {
80 u8 *a = uword_to_pointer (pa, u8 *);
81 u8 *b = uword_to_pointer (pb, u8 *);
82 u8 a0 = a[0], b0 = b[0];
83 a[0] = b0;
84 b[0] = a0;
85 }
86}
