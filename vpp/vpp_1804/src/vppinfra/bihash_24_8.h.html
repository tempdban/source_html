15#undef BIHASH_TYPE
16#undef BIHASH_KVP_CACHE_SIZE
17#undef BIHASH_KVP_PER_PAGE
18
19#define BIHASH_TYPE _24_8
20#define BIHASH_KVP_PER_PAGE 4
21#define BIHASH_KVP_CACHE_SIZE 0
22
23#ifndef __included_bihash_24_8_h__
24#define __included_bihash_24_8_h__
25
26#include <vppinfra/crc32.h>
27#include <vppinfra/heap.h>
28#include <vppinfra/format.h>
29#include <vppinfra/pool.h>
30#include <vppinfra/xxhash.h>
31
32typedef struct
33{
34 u64 key[3];
35 u64 value;
36} clib_bihash_kv_24_8_t;
37
38static inline int
39clib_bihash_is_free_24_8 (const clib_bihash_kv_24_8_t * v)
40{
41 /* Free values are memset to 0xff, check a bit... */
42 if (v->key[0] == ~0ULL && v->value == ~0ULL)
43 return 1;
44 return 0;
45}
46
47static inline u64
48clib_bihash_hash_24_8 (const clib_bihash_kv_24_8_t * v)
49{
50#ifdef clib_crc32c_uses_intrinsics
51 return clib_crc32c ((u8 *) v->key, 24);
52#else
53 u64 tmp = v->key[0] ^ v->key[1] ^ v->key[2];
54 return clib_xxhash (tmp);
55#endif
56}
57
58static inline u8 *
59format_bihash_kvp_24_8 (u8 * s, va_list * args)
60{
61 clib_bihash_kv_24_8_t *v = va_arg (*args, clib_bihash_kv_24_8_t *);
62
63 s = format (s, "key %llu %llu %llu value %llu",
64 v->key[0], v->key[1], v->key[2], v->value);
65 return s;
66}
67
68static inline int
69clib_bihash_key_compare_24_8 (const u64 * a, const u64 * b)
70{
71 return ((a[0] ^ b[0]) | (a[1] ^ b[1]) | (a[2] ^ b[2])) == 0;
72}
