40#define F_ADD_SEALS (F_LINUX_SPECIFIC_BASE + 9)

49uword
50clib_mem_vm_get_page_size (int fd)
51{
52 struct stat st = { 0 };
53 if (fstat (fd, &st) == -1)
54 return 0;
55 return st.st_blksize;
56}

58int
59clib_mem_vm_get_log2_page_size (int fd)
60{
61 return min_log2 (clib_mem_vm_get_page_size (fd));
62}

64void
65clib_mem_vm_randomize_va (uword * requested_va, u32 log2_page_size)
66{
67 u8 bit_mask = 15;
68
69 if (log2_page_size <= 12)
70 bit_mask = 15;
71 else if (log2_page_size > 12 && log2_page_size <= 16)
72 bit_mask = 3;
73 else
74 bit_mask = 0;
75
76 *requested_va += (clib_cpu_time_now () & bit_mask) * (1 << log2_page_size);
77}

79clib_error_t *
80clib_mem_vm_ext_alloc (clib_mem_vm_alloc_t * a)
81{
82 int fd = -1;
83 clib_error_t *err = 0;
84 void *addr = 0;
85 u8 *filename = 0;
86 int mmap_flags = 0;
87 int log2_page_size;
88 int n_pages;
89 int old_mpol = -1;
90 u64 old_mask[16] = { 0 };
91
92 /* save old numa mem policy if needed */
93 if (a->flags & (CLIB_MEM_VM_F_NUMA_PREFER | CLIB_MEM_VM_F_NUMA_FORCE))
94 {
95 int rv;
96 rv = get_mempolicy (&old_mpol, old_mask, sizeof (old_mask) * 8 + 1,
97 0, 0);
98
99 if (rv == -1)
100 {
101 if (a->numa_node != 0 && (a->flags & CLIB_MEM_VM_F_NUMA_FORCE) != 0)
102 {
103 err = clib_error_return_unix (0, "get_mempolicy");
104 goto error;
105 }
106 else
107 old_mpol = -1;
108 }
109 }
110
111 if (a->flags & CLIB_MEM_VM_F_LOCKED)
112 mmap_flags |= MAP_LOCKED;
113
114 /* if we are creating shared segment, we need file descriptor */
115 if (a->flags & CLIB_MEM_VM_F_SHARED)
116 {
117 mmap_flags |= MAP_SHARED;
118 /* if hugepages are needed we need to create mount point */
119 if (a->flags & CLIB_MEM_VM_F_HUGETLB)
120 {
121 char *mount_dir;
122 char template[] = "/tmp/hugepage_mount.XXXXXX";
123
124 mount_dir = mkdtemp (template);
125 if (mount_dir == 0)
126 return clib_error_return_unix (0, "mkdtemp \'%s\'", template);
127
128 if (mount ("none", (char *) mount_dir, "hugetlbfs", 0, NULL))
129 {
130 err = clib_error_return_unix (0, "mount hugetlb directory '%s'",
131 mount_dir);
132 goto error;
133 }
134
135 filename = format (0, "%s/%s%c", mount_dir, a->name, 0);
136
137 if ((fd = open ((char *) filename, O_CREAT | O_RDWR, 0755)) == -1)
138 {
139 err = clib_error_return_unix (0, "open");
140 goto error;
141 }
142 umount2 ((char *) mount_dir, MNT_DETACH);
143 rmdir ((char *) mount_dir);
144 mmap_flags |= MAP_LOCKED;
145 }
146 else
147 {
148 if ((fd = memfd_create (a->name, MFD_ALLOW_SEALING)) == -1)
149 {
150 err = clib_error_return_unix (0, "memfd_create");
151 goto error;
152 }
153
154 if ((fcntl (fd, F_ADD_SEALS, F_SEAL_SHRINK)) == -1)
155 {
156 err = clib_error_return_unix (0, "fcntl (F_ADD_SEALS)");
157 goto error;
158 }
159 }
160
161 log2_page_size = clib_mem_vm_get_log2_page_size (fd);
162 if (log2_page_size == 0)
163 {
164 err = clib_error_return_unix (0, "cannot determine page size");
165 goto error;
166 }
167
168 if (a->requested_va)
169 {
170 clib_mem_vm_randomize_va (&a->requested_va, log2_page_size);
171 mmap_flags |= MAP_FIXED;
172 }
173 }
174 else /* not CLIB_MEM_VM_F_SHARED */
175 {
176 mmap_flags |= MAP_PRIVATE | MAP_ANONYMOUS;
177 if (a->flags & CLIB_MEM_VM_F_HUGETLB)
178 {
179 mmap_flags |= MAP_HUGETLB;
180 log2_page_size = 21;
181 }
182 else
183 {
184 log2_page_size = min_log2 (sysconf (_SC_PAGESIZE));
185 }
186 }
187
188 n_pages = ((a->size - 1) >> log2_page_size) + 1;
189
190 if (a->flags & CLIB_MEM_VM_F_HUGETLB_PREALLOC)
191 {
192 err = clib_sysfs_prealloc_hugepages (a->numa_node,
193 1 << (log2_page_size - 10),
194 n_pages);
195 if (err)
196 goto error;
197
198 }
199
200 if (fd != -1)
201 if ((ftruncate (fd, (u64) n_pages * (1 << log2_page_size))) == -1)
202 {
203 err = clib_error_return_unix (0, "ftruncate");
204 goto error;
205 }
206
207 if (old_mpol != -1)
208 {
209 int rv;
210 u64 mask[16] = { 0 };
211 mask[0] = 1 << a->numa_node;
212 rv = set_mempolicy (MPOL_BIND, mask, sizeof (mask) * 8 + 1);
213 if (rv)
214 {
215 err = clib_error_return_unix (0, "set_mempolicy");
216 goto error;
217 }
218 }
219
220 addr = mmap (uword_to_pointer (a->requested_va, void *), a->size,
221 (PROT_READ | PROT_WRITE), mmap_flags, fd, 0);
222 if (addr == MAP_FAILED)
223 {
224 err = clib_error_return_unix (0, "mmap");
225 goto error;
226 }
227
228 /* re-apply old numa memory policy */
229 if (old_mpol != -1 &&
230 set_mempolicy (old_mpol, old_mask, sizeof (old_mask) * 8 + 1) == -1)
231 {
232 err = clib_error_return_unix (0, "set_mempolicy");
233 goto error;
234 }
235
236 a->log2_page_size = log2_page_size;
237 a->n_pages = n_pages;
238 a->addr = addr;
239 a->fd = fd;
240 goto done;
241
242error:
243 if (fd != -1)
244 close (fd);
245
246done:
247 vec_free (filename);
248 return err;
249}

251u64 *
252clib_mem_vm_get_paddr (void *mem, int log2_page_size, int n_pages)
253{
254 int pagesize = sysconf (_SC_PAGESIZE);
255 int fd;
256 int i;
257 u64 *r = 0;
258
259 if ((fd = open ((char *) "/proc/self/pagemap", O_RDONLY)) == -1)
260 return 0;
261
262 for (i = 0; i < n_pages; i++)
263 {
264 u64 seek, pagemap = 0;
265 uword vaddr = pointer_to_uword (mem) + (((u64) i) << log2_page_size);
266 seek = ((u64) vaddr / pagesize) * sizeof (u64);
267 if (lseek (fd, seek, SEEK_SET) != seek)
268 goto done;
269
270 if (read (fd, &pagemap, sizeof (pagemap)) != (sizeof (pagemap)))
271 goto done;
272
273 if ((pagemap & (1ULL << 63)) == 0)
274 goto done;
275
276 pagemap &= pow2_mask (55);
277 vec_add1 (r, pagemap * pagesize);
278 }
279
280done:
281 close (fd);
282 if (vec_len (r) != n_pages)
283 {
284 vec_free (r);
285 return 0;
286 }
287 return r;
288}

290clib_error_t *
291clib_mem_vm_ext_map (clib_mem_vm_map_t * a)
292{
293 int mmap_flags = MAP_SHARED;
294 void *addr;
295
296 if (a->requested_va)
297 mmap_flags |= MAP_FIXED;
298
299 addr = (void *) mmap (uword_to_pointer (a->requested_va, void *), a->size,
300 PROT_READ | PROT_WRITE, mmap_flags, a->fd, 0);
301
302 if (addr == MAP_FAILED)
303 return clib_error_return_unix (0, "mmap");
304
305 a->addr = addr;
306 return 0;
307}
