67typedef struct
68{
69 ikev2_transform_type_t type;
70 union
71 {
72 u16 transform_id;
73 ikev2_transform_encr_type_t encr_type:16;
74 ikev2_transform_prf_type_t prf_type:16;
75 ikev2_transform_integ_type_t integ_type:16;
76 ikev2_transform_dh_type_t dh_type:16;
77 ikev2_transform_esn_type_t esn_type:16;
78 };
79 u8 *attrs;
80 u16 key_len;
81 u16 key_trunc;
82 u16 block_size;
83 u8 dh_group;
84 int nid;
85 const char *dh_p;
86 const char *dh_g;
87 const void *md;
88 const void *cipher;
89} ikev2_sa_transform_t;

197typedef struct
198{
199 ikev2_state_t state;
200 u8 unsupported_cp;

204 u64 ispi;
205 u64 rspi;
206 u8 *i_nonce;
207 u8 *r_nonce;
208
209 /* DH data */

213 u8 *i_dh_data;

224 u8 *sk_ei;

226 u8 *sk_pi;
227 u8 *sk_pr;

232
233 /* ID */
234 ikev2_id_t i_id;
235 ikev2_id_t r_id;
236
237 /* pending deletes */
238 ikev2_delete_t *del;
239
240 /* pending rekeyings */
241 ikev2_rekey_t *rekey;
242
243 /* packet data */
244 u8 *last_sa_init_req_packet_data;
245 u8 *last_sa_init_res_packet_data;
246
247 /* retransmit */
248 u32 last_msg_id;
249 u8 *last_res_packet_data;
250
251 u8 is_initiator;
252 u32 last_init_msg_id;

255 ikev2_child_sa_t *childs;
256} ikev2_sa_t;
