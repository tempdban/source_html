104//Structure for DHCPv6 payload from client
105typedef struct dhcpv6_hdr_
106{
107 union
108 {
109 u8 msg_type; //DHCP msg type
110 u32 xid; // transaction id

112 u8 data[0];
113} dhcpv6_header_t;
