32#define foreach_dhcpv6_proxy_to_server_input_next \
33 _ (DROP, "error-drop") \

63static u8 *
64format_dhcpv6_proxy_trace (u8 * s, va_list * args)
65{
66 CLIB_UNUSED (vlib_main_t * vm) = va_arg (*args, vlib_main_t *);
67 CLIB_UNUSED (vlib_node_t * node) = va_arg (*args, vlib_node_t *);
68 dhcpv6_proxy_trace_t *t = va_arg (*args, dhcpv6_proxy_trace_t *);
69
70 if (t->which == 0)
71 s = format (s, "DHCPV6 proxy: sent to server %U",
72 format_ip6_address, &t->packet_data, sizeof (ip6_address_t));
73 else
74 s = format (s, "DHCPV6 proxy: sent to client from %U",
75 format_ip6_address, &t->packet_data, sizeof (ip6_address_t));
76 if (t->error != (u32) ~ 0)
77 s = format (s, " error: %s\n", dhcpv6_proxy_error_strings[t->error]);
78
79 s = format (s, " original_sw_if_index: %d, sw_if_index: %d\n",
80 t->original_sw_if_index, t->sw_if_index);
81
82 return s;
83}

85static u8 *
86format_dhcpv6_proxy_header_with_length (u8 * s, va_list * args)
87{
88 dhcpv6_header_t *h = va_arg (*args, dhcpv6_header_t *);
89 u32 max_header_bytes = va_arg (*args, u32);
90 u32 header_bytes;
91
92 header_bytes = sizeof (h[0]);
93 if (max_header_bytes != 0 && header_bytes > max_header_bytes)
94 return format (s, "dhcpv6 header truncated");
95
96 s = format (s, "DHCPV6 Proxy");
97
98 return s;
99}

101/* get first interface address */
102static ip6_address_t *
103ip6_interface_first_global_or_site_address (ip6_main_t * im, u32 sw_if_index)
104{
105 ip_lookup_main_t *lm = &im->lookup_main;
106 ip_interface_address_t *ia = 0;
107 ip6_address_t *result = 0;
108
109 /* *INDENT-OFF* */
110 foreach_ip_interface_address (lm, ia, sw_if_index,
111 1 /* honor unnumbered */,
112 ({
113 ip6_address_t * a = ip_interface_address_get_address (lm, ia);
114 if ((a->as_u8[0] & 0xe0) == 0x20 ||
115 (a->as_u8[0] & 0xfe) == 0xfc) {
116 result = a;
117 break;
118 }
119 }));
120 /* *INDENT-ON* */
121 return result;
122}

124static inline void
125copy_ip6_address (ip6_address_t * dst, ip6_address_t * src)
126{
127 dst->as_u64[0] = src->as_u64[0];
128 dst->as_u64[1] = src->as_u64[1];
129}

131static uword
132dhcpv6_proxy_to_server_input (vlib_main_t * vm,
133 vlib_node_runtime_t * node,
134 vlib_frame_t * from_frame)
135{
136 u32 n_left_from, next_index, *from, *to_next;
137 dhcp_proxy_main_t *dpm = &dhcp_proxy_main;
138 from = vlib_frame_vector_args (from_frame);
139 n_left_from = from_frame->n_vectors;
140 u32 pkts_to_server = 0, pkts_to_client = 0, pkts_no_server = 0;
141 u32 pkts_no_interface_address = 0, pkts_no_exceeding_max_hop = 0;
142 u32 pkts_no_src_address = 0;
143 u32 pkts_wrong_msg_type = 0;
144 u32 pkts_too_big = 0;
145 ip6_main_t *im = &ip6_main;
146 ip6_address_t *src;
147 int bogus_length;
148 dhcp_proxy_t *proxy;
149 dhcp_server_t *server;
150 u32 rx_fib_idx = 0, server_fib_idx = 0;
151
152 next_index = node->cached_next_index;
153
154 while (n_left_from > 0)
155 {
156 u32 n_left_to_next;
157
158 vlib_get_next_frame (vm, node, next_index, to_next, n_left_to_next);
159
160 while (n_left_from > 0 && n_left_to_next > 0)
161 {
162 vnet_main_t *vnm = vnet_get_main ();
163 u32 sw_if_index = 0;
164 u32 rx_sw_if_index = 0;
165 vnet_sw_interface_t *swif;
166 u32 bi0;
167 vlib_buffer_t *b0;
168 udp_header_t *u0, *u1;
169 dhcpv6_header_t *h0; // client msg hdr
170 ip6_header_t *ip0, *ip1;
171 ip6_address_t _ia0, *ia0 = &_ia0;
172 u32 next0;
173 u32 error0 = (u32) ~ 0;
174 dhcpv6_option_t *fwd_opt;
175 dhcpv6_relay_hdr_t *r1;
176 u16 len;
177 dhcpv6_int_id_t *id1;
178 dhcpv6_vss_t *vss1;
179 dhcpv6_client_mac_t *cmac; // client mac
180 ethernet_header_t *e_h0;
181 u8 client_src_mac[6];
182 vlib_buffer_free_list_t *fl;
183 dhcp_vss_t *vss;
184 u8 is_solicit = 0;
185
186 bi0 = from[0];
187 from += 1;
188 n_left_from -= 1;
189
190 b0 = vlib_get_buffer (vm, bi0);
191
192 h0 = vlib_buffer_get_current (b0);
193
194 /*
195 * udp_local hands us the DHCPV6 header.
196 */
197 u0 = (void *) h0 - (sizeof (*u0));
198 ip0 = (void *) u0 - (sizeof (*ip0));
199 e_h0 = (void *) ip0 - ethernet_buffer_header_size (b0);
200
201 clib_memcpy (client_src_mac, e_h0->src_address, 6);
202
203 switch (h0->u.msg_type)
204 {
205 case DHCPV6_MSG_SOLICIT:
206 case DHCPV6_MSG_REQUEST:
207 case DHCPV6_MSG_CONFIRM:
208 case DHCPV6_MSG_RENEW:
209 case DHCPV6_MSG_REBIND:
210 case DHCPV6_MSG_RELEASE:
211 case DHCPV6_MSG_DECLINE:
212 case DHCPV6_MSG_INFORMATION_REQUEST:
213 case DHCPV6_MSG_RELAY_FORW:
214 /* send to server */
215 break;
216 case DHCPV6_MSG_RELAY_REPL:
217 /* send to client */
218 next0 = DHCPV6_PROXY_TO_SERVER_INPUT_NEXT_SEND_TO_CLIENT;
219 error0 = 0;
220 pkts_to_client++;
221 goto do_enqueue;
222 default:
223 /* drop the packet */
224 pkts_wrong_msg_type++;
225 error0 = DHCPV6_PROXY_ERROR_WRONG_MESSAGE_TYPE;
226 next0 = DHCPV6_PROXY_TO_SERVER_INPUT_NEXT_DROP;
227 goto do_trace;
228
229 }
230
231 /* Send to DHCPV6 server via the configured FIB */
232 rx_sw_if_index = sw_if_index =
233 vnet_buffer (b0)->sw_if_index[VLIB_RX];
234 rx_fib_idx = im->mfib_index_by_sw_if_index[rx_sw_if_index];
235 proxy = dhcp_get_proxy (dpm, rx_fib_idx, FIB_PROTOCOL_IP6);
236
237 if (PREDICT_FALSE (NULL == proxy))
238 {
239 error0 = DHCPV6_PROXY_ERROR_NO_SERVER;
240 next0 = DHCPV6_PROXY_TO_SERVER_INPUT_NEXT_DROP;
241 pkts_no_server++;
242 goto do_trace;
243 }
244
245 server = &proxy->dhcp_servers[0];
246 server_fib_idx = server->server_fib_index;
247 vnet_buffer (b0)->sw_if_index[VLIB_TX] = server_fib_idx;
248
249
250 /* relay-option header pointer */
251 vlib_buffer_advance (b0, -(sizeof (*fwd_opt)));
252 fwd_opt = vlib_buffer_get_current (b0);
253 /* relay message header pointer */
254 vlib_buffer_advance (b0, -(sizeof (*r1)));
255 r1 = vlib_buffer_get_current (b0);
256
257 vlib_buffer_advance (b0, -(sizeof (*u1)));
258 u1 = vlib_buffer_get_current (b0);
259
260 vlib_buffer_advance (b0, -(sizeof (*ip1)));
261 ip1 = vlib_buffer_get_current (b0);
262
263 /* fill in all that rubbish... */
264 len = clib_net_to_host_u16 (u0->length) - sizeof (udp_header_t);
265 copy_ip6_address (&r1->peer_addr, &ip0->src_address);
266
267 r1->msg_type = DHCPV6_MSG_RELAY_FORW;
268 fwd_opt->length = clib_host_to_net_u16 (len);
269 fwd_opt->option = clib_host_to_net_u16 (DHCPV6_OPTION_RELAY_MSG);
270
271 r1->hop_count++;
272 r1->hop_count =
273 (h0->u.msg_type != DHCPV6_MSG_RELAY_FORW) ? 0 : r1->hop_count;
274
275 if (PREDICT_FALSE (r1->hop_count >= HOP_COUNT_LIMIT))
276 {
277 error0 = DHCPV6_RELAY_PKT_DROP_MAX_HOPS;
278 next0 = DHCPV6_PROXY_TO_SERVER_INPUT_NEXT_DROP;
279 pkts_no_exceeding_max_hop++;
280 goto do_trace;
281 }
282
283
284 /* If relay-fwd and src address is site or global unicast address */
285 if (h0->u.msg_type == DHCPV6_MSG_RELAY_FORW &&
286 ((ip0->src_address.as_u8[0] & 0xe0) == 0x20 ||
287 (ip0->src_address.as_u8[0] & 0xfe) == 0xfc))
288 {
289 /* Set link address to zero */
290 r1->link_addr.as_u64[0] = 0;
291 r1->link_addr.as_u64[1] = 0;
292 goto link_address_set;
293 }
294
295 /* if receiving interface is unnumbered, use receiving interface
296 * IP address as link address, otherwise use the loopback interface
297 * IP address as link address.
298 */
299
300 swif = vnet_get_sw_interface (vnm, rx_sw_if_index);
301 if (swif->flags & VNET_SW_INTERFACE_FLAG_UNNUMBERED)
302 sw_if_index = swif->unnumbered_sw_if_index;
303
304 ia0 =
305 ip6_interface_first_global_or_site_address (&ip6_main,
306 sw_if_index);
307 if (ia0 == 0)
308 {
309 error0 = DHCPV6_PROXY_ERROR_NO_INTERFACE_ADDRESS;
310 next0 = DHCPV6_PROXY_TO_SERVER_INPUT_NEXT_DROP;
311 pkts_no_interface_address++;
312 goto do_trace;
313 }
314
315 copy_ip6_address (&r1->link_addr, ia0);
316
317 link_address_set:
318 fl =
319 vlib_buffer_get_free_list (vm,
320 vlib_buffer_get_free_list_index (b0));
321
322 if ((b0->current_length + sizeof (*id1) + sizeof (*vss1) +
323 sizeof (*cmac)) > fl->n_data_bytes)
324 {
325 error0 = DHCPV6_PROXY_ERROR_PKT_TOO_BIG;
326 next0 = DHCPV6_PROXY_TO_SERVER_INPUT_NEXT_DROP;
327 pkts_too_big++;
328 goto do_trace;
329 }
330
331 id1 = (dhcpv6_int_id_t *) (((uword) ip1) + b0->current_length);
332 b0->current_length += (sizeof (*id1));
333
334 id1->opt.option = clib_host_to_net_u16 (DHCPV6_OPTION_INTERFACE_ID);
335 id1->opt.length = clib_host_to_net_u16 (sizeof (rx_sw_if_index));
336 id1->int_idx = clib_host_to_net_u32 (rx_sw_if_index);
337
338 u1->length = 0;
339 if (h0->u.msg_type != DHCPV6_MSG_RELAY_FORW)
340 {
341 cmac =
342 (dhcpv6_client_mac_t *) (((uword) ip1) + b0->current_length);
343 b0->current_length += (sizeof (*cmac));
344 cmac->opt.length = clib_host_to_net_u16 (sizeof (*cmac) -
345 sizeof (cmac->opt));
346 cmac->opt.option =
347 clib_host_to_net_u16
348 (DHCPV6_OPTION_CLIENT_LINK_LAYER_ADDRESS);
349 cmac->link_type = clib_host_to_net_u16 (1); /* ethernet */
350 clib_memcpy (cmac->data, client_src_mac, 6);
351 u1->length += sizeof (*cmac);
352 }
353
354 vss = dhcp_get_vss_info (dpm, rx_fib_idx, FIB_PROTOCOL_IP6);
355
356 if (vss)
357 {
358 u16 id_len; /* length of VPN ID */
359 u16 type_len = sizeof (vss1->vss_type);
360
361 vss1 = (dhcpv6_vss_t *) (((uword) ip1) + b0->current_length);
362 vss1->vss_type = vss->vss_type;
363 if (vss->vss_type == VSS_TYPE_VPN_ID)
364 {
365 id_len = sizeof (vss->vpn_id); /* vpn_id is 7 bytes */
366 memcpy (vss1->data, vss->vpn_id, id_len);
367 }
368 else if (vss->vss_type == VSS_TYPE_ASCII)
369 {
370 id_len = vec_len (vss->vpn_ascii_id);
371 memcpy (vss1->data, vss->vpn_ascii_id, id_len);
372 }
373 else /* must be VSS_TYPE_DEFAULT, no VPN ID */
374 id_len = 0;
375
376 vss1->opt.option = clib_host_to_net_u16 (DHCPV6_OPTION_VSS);
377 vss1->opt.length = clib_host_to_net_u16 (type_len + id_len);
378 u1->length += type_len + id_len + sizeof (vss1->opt);
379 b0->current_length += type_len + id_len + sizeof (vss1->opt);
380 }
381
382 pkts_to_server++;
383 u1->checksum = 0;
384 u1->src_port = clib_host_to_net_u16 (UDP_DST_PORT_dhcpv6_to_client);
385 u1->dst_port = clib_host_to_net_u16 (UDP_DST_PORT_dhcpv6_to_server);
386
387 u1->length =
388 clib_host_to_net_u16 (clib_net_to_host_u16 (fwd_opt->length) +
389 sizeof (*r1) + sizeof (*fwd_opt) +
390 sizeof (*u1) + sizeof (*id1) + u1->length);
391
392 memset (ip1, 0, sizeof (*ip1));
393 ip1->ip_version_traffic_class_and_flow_label = 0x60;
394 ip1->payload_length = u1->length;
395 ip1->protocol = PROTO_UDP;
396 ip1->hop_limit = HOP_COUNT_LIMIT;
397 src = ((server->dhcp_server.ip6.as_u64[0] ||
398 server->dhcp_server.ip6.as_u64[1]) ?
399 &server->dhcp_server.ip6 : &all_dhcpv6_server_address);
400 copy_ip6_address (&ip1->dst_address, src);
401
402
403 ia0 = ip6_interface_first_global_or_site_address
404 (&ip6_main, vnet_buffer (b0)->sw_if_index[VLIB_RX]);
405
406 src = (proxy->dhcp_src_address.ip6.as_u64[0] ||
407 proxy->dhcp_src_address.ip6.as_u64[1]) ?
408 &proxy->dhcp_src_address.ip6 : ia0;
409 if (ia0 == 0)
410 {
411 error0 = DHCPV6_PROXY_ERROR_NO_SRC_ADDRESS;
412 next0 = DHCPV6_PROXY_TO_SERVER_INPUT_NEXT_DROP;
413 pkts_no_src_address++;
414 goto do_trace;
415 }
416
417 copy_ip6_address (&ip1->src_address, src);
418
419
420 u1->checksum = ip6_tcp_udp_icmp_compute_checksum (vm, b0, ip1,
421 &bogus_length);
422 ASSERT (bogus_length == 0);
423
424 next0 = DHCPV6_PROXY_TO_SERVER_INPUT_NEXT_LOOKUP;
425
426 is_solicit = (DHCPV6_MSG_SOLICIT == h0->u.msg_type);
427
428 /*
429 * If we have multiple servers configured and this is the
430 * client's discover message, then send copies to each of
431 * those servers
432 */
433 if (is_solicit && vec_len (proxy->dhcp_servers) > 1)
434 {
435 u32 ii;
436
437 for (ii = 1; ii < vec_len (proxy->dhcp_servers); ii++)
438 {
439 vlib_buffer_t *c0;
440 u32 ci0;
441
442 c0 = vlib_buffer_copy (vm, b0);
443 ci0 = vlib_get_buffer_index (vm, c0);
444 server = &proxy->dhcp_servers[ii];
445
446 ip0 = vlib_buffer_get_current (c0);
447
448 src = ((server->dhcp_server.ip6.as_u64[0] ||
449 server->dhcp_server.ip6.as_u64[1]) ?
450 &server->dhcp_server.ip6 :
451 &all_dhcpv6_server_address);
452 copy_ip6_address (&ip1->dst_address, src);
453
454 to_next[0] = ci0;
455 to_next += 1;
456 n_left_to_next -= 1;
457
458 vlib_validate_buffer_enqueue_x1 (vm, node, next_index,
459 to_next, n_left_to_next,
460 ci0, next0);
461
462 if (PREDICT_FALSE (b0->flags & VLIB_BUFFER_IS_TRACED))
463 {
464 dhcpv6_proxy_trace_t *tr;
465
466 tr = vlib_add_trace (vm, node, c0, sizeof (*tr));
467 tr->which = 0; /* to server */
468 tr->error = error0;
469 tr->original_sw_if_index = rx_sw_if_index;
470 tr->sw_if_index = sw_if_index;
471 if (next0 == DHCPV6_PROXY_TO_SERVER_INPUT_NEXT_LOOKUP)
472 copy_ip6_address ((ip6_address_t *) &
473 tr->packet_data[0],
474 &server->dhcp_server.ip6);
475 }
476
477 if (PREDICT_FALSE (0 == n_left_to_next))
478 {
479 vlib_put_next_frame (vm, node, next_index,
480 n_left_to_next);
481 vlib_get_next_frame (vm, node, next_index,
482 to_next, n_left_to_next);
483 }
484 }
485 }
486
487 do_trace:
488 if (PREDICT_FALSE (b0->flags & VLIB_BUFFER_IS_TRACED))
489 {
490 dhcpv6_proxy_trace_t *tr = vlib_add_trace (vm, node,
491 b0, sizeof (*tr));
492 tr->which = 0; /* to server */
493 tr->error = error0;
494 tr->original_sw_if_index = rx_sw_if_index;
495 tr->sw_if_index = sw_if_index;
496 if (DHCPV6_PROXY_TO_SERVER_INPUT_NEXT_LOOKUP == next0)
497 copy_ip6_address ((ip6_address_t *) & tr->packet_data[0],
498 &server->dhcp_server.ip6);
499 }
500
501 do_enqueue:
502 to_next[0] = bi0;
503 to_next += 1;
504 n_left_to_next -= 1;
505
506 vlib_validate_buffer_enqueue_x1 (vm, node, next_index,
507 to_next, n_left_to_next,
508 bi0, next0);
509 }
510
511 vlib_put_next_frame (vm, node, next_index, n_left_to_next);
512 }
513
514 vlib_node_increment_counter (vm, dhcpv6_proxy_to_server_node.index,
515 DHCPV6_PROXY_ERROR_RELAY_TO_CLIENT,
516 pkts_to_client);
517 vlib_node_increment_counter (vm, dhcpv6_proxy_to_server_node.index,
518 DHCPV6_PROXY_ERROR_RELAY_TO_SERVER,
519 pkts_to_server);
520 vlib_node_increment_counter (vm, dhcpv6_proxy_to_server_node.index,
521 DHCPV6_PROXY_ERROR_NO_INTERFACE_ADDRESS,
522 pkts_no_interface_address);
523 vlib_node_increment_counter (vm, dhcpv6_proxy_to_server_node.index,
524 DHCPV6_PROXY_ERROR_WRONG_MESSAGE_TYPE,
525 pkts_wrong_msg_type);
526 vlib_node_increment_counter (vm, dhcpv6_proxy_to_server_node.index,
527 DHCPV6_PROXY_ERROR_NO_SRC_ADDRESS,
528 pkts_no_src_address);
529 vlib_node_increment_counter (vm, dhcpv6_proxy_to_server_node.index,
530 DHCPV6_PROXY_ERROR_PKT_TOO_BIG, pkts_too_big);
531 return from_frame->n_vectors;
532}

559static uword
560dhcpv6_proxy_to_client_input (vlib_main_t * vm,
561 vlib_node_runtime_t * node,
562 vlib_frame_t * from_frame)
563{
564
565 u32 n_left_from, *from;
566 ethernet_main_t *em = vnet_get_ethernet_main ();
567 dhcp_proxy_main_t *dm = &dhcp_proxy_main;
568 dhcp_proxy_t *proxy;
569 dhcp_server_t *server;
570 vnet_main_t *vnm = vnet_get_main ();
571 int bogus_length;
572
573 from = vlib_frame_vector_args (from_frame);
574 n_left_from = from_frame->n_vectors;
575
576 while (n_left_from > 0)
577 {
578 u32 bi0;
579 vlib_buffer_t *b0;
580 udp_header_t *u0, *u1 = 0;
581 dhcpv6_relay_hdr_t *h0;
582 ip6_header_t *ip1 = 0, *ip0;
583 ip6_address_t _ia0, *ia0 = &_ia0;
584 ip6_address_t client_address;
585 ethernet_interface_t *ei0;
586 ethernet_header_t *mac0;
587 vnet_hw_interface_t *hi0;
588 vlib_frame_t *f0;
589 u32 *to_next0;
590 u32 sw_if_index = ~0;
591 u32 original_sw_if_index = ~0;
592 vnet_sw_interface_t *si0;
593 u32 error0 = (u32) ~ 0;
594 vnet_sw_interface_t *swif;
595 dhcpv6_option_t *r0 = 0, *o;
596 u16 len = 0;
597 u8 interface_opt_flag = 0;
598 u8 relay_msg_opt_flag = 0;
599 ip6_main_t *im = &ip6_main;
600 u32 server_fib_idx, client_fib_idx;
601
602 bi0 = from[0];
603 from += 1;
604 n_left_from -= 1;
605
606 b0 = vlib_get_buffer (vm, bi0);
607 h0 = vlib_buffer_get_current (b0);
608
609 if (DHCPV6_MSG_RELAY_REPL != h0->msg_type)
610 {
611 error0 = DHCPV6_PROXY_ERROR_WRONG_MESSAGE_TYPE;
612
613 drop_packet:
614 vlib_node_increment_counter (vm, dhcpv6_proxy_to_client_node.index,
615 error0, 1);
616
617 f0 = vlib_get_frame_to_node (vm, dm->error_drop_node_index);
618 to_next0 = vlib_frame_vector_args (f0);
619 to_next0[0] = bi0;
620 f0->n_vectors = 1;
621 vlib_put_frame_to_node (vm, dm->error_drop_node_index, f0);
622 goto do_trace;
623 }
624 /* hop count seems not need to be checked */
625 if (HOP_COUNT_LIMIT < h0->hop_count)
626 {
627 error0 = DHCPV6_RELAY_PKT_DROP_MAX_HOPS;
628 goto drop_packet;
629 }
630 u0 = (void *) h0 - (sizeof (*u0));
631 ip0 = (void *) u0 - (sizeof (*ip0));
632
633 vlib_buffer_advance (b0, sizeof (*h0));
634 o = vlib_buffer_get_current (b0);
635
636 /* Parse through TLVs looking for option 18 (DHCPV6_OPTION_INTERFACE_ID)
637 _and_ option 9 (DHCPV6_OPTION_RELAY_MSG) option which must be there.
638 Currently assuming no other options need to be processed
639 The interface-ID is the FIB number we need
640 to track down the client-facing interface */
641
642 while ((u8 *) o < (b0->data + b0->current_data + b0->current_length))
643 {
644 if (DHCPV6_OPTION_INTERFACE_ID == clib_net_to_host_u16 (o->option))
645 {
646 interface_opt_flag = 1;
647 if (clib_net_to_host_u16 (o->length) == sizeof (sw_if_index))
648 sw_if_index =
649 clib_net_to_host_u32 (((dhcpv6_int_id_t *) o)->int_idx);
650 if (sw_if_index >= vec_len (im->fib_index_by_sw_if_index))
651 {
652 error0 = DHCPV6_PROXY_ERROR_WRONG_INTERFACE_ID_OPTION;
653 goto drop_packet;
654 }
655 }
656 if (DHCPV6_OPTION_RELAY_MSG == clib_net_to_host_u16 (o->option))
657 {
658 relay_msg_opt_flag = 1;
659 r0 = vlib_buffer_get_current (b0);
660 }
661 if ((relay_msg_opt_flag == 1) && (interface_opt_flag == 1))
662 break;
663 vlib_buffer_advance (b0,
664 sizeof (*o) +
665 clib_net_to_host_u16 (o->length));
666 o =
667 (dhcpv6_option_t *) (((uword) o) +
668 clib_net_to_host_u16 (o->length) +
669 sizeof (*o));
670 }
671
672 if ((relay_msg_opt_flag == 0) || (r0 == 0))
673 {
674 error0 = DHCPV6_PROXY_ERROR_NO_RELAY_MESSAGE_OPTION;
675 goto drop_packet;
676 }
677
678 if ((u32) ~ 0 == sw_if_index)
679 {
680 error0 = DHCPV6_PROXY_ERROR_NO_CIRCUIT_ID_OPTION;
681 goto drop_packet;
682 }
683
684 //Advance buffer to start of encapsulated DHCPv6 message
685 vlib_buffer_advance (b0, sizeof (*r0));
686
687 client_fib_idx = im->mfib_index_by_sw_if_index[sw_if_index];
688 proxy = dhcp_get_proxy (dm, client_fib_idx, FIB_PROTOCOL_IP6);
689
690 if (NULL == proxy)
691 {
692 error0 = DHCPV6_PROXY_ERROR_NO_SERVER;
693 goto drop_packet;
694 }
695
696 server_fib_idx = im->fib_index_by_sw_if_index
697 [vnet_buffer (b0)->sw_if_index[VLIB_RX]];
698
699 vec_foreach (server, proxy->dhcp_servers)
700 {
701 if (server_fib_idx == server->server_fib_index &&
702 ip0->src_address.as_u64[0] == server->dhcp_server.ip6.as_u64[0] &&
703 ip0->src_address.as_u64[1] == server->dhcp_server.ip6.as_u64[1])
704 {
705 goto server_found;
706 }
707 }
708
709 //drop packet if not from server with configured address or FIB
710 error0 = DHCPV6_PROXY_ERROR_BAD_SVR_FIB_OR_ADDRESS;
711 goto drop_packet;
712
713 server_found:
714 vnet_buffer (b0)->sw_if_index[VLIB_TX] = original_sw_if_index
715 = sw_if_index;
716
717 swif = vnet_get_sw_interface (vnm, original_sw_if_index);
718 if (swif->flags & VNET_SW_INTERFACE_FLAG_UNNUMBERED)
719 sw_if_index = swif->unnumbered_sw_if_index;
720
721
722 /*
723 * udp_local hands us the DHCPV6 header, need udp hdr,
724 * ip hdr to relay to client
725 */
726 vlib_buffer_advance (b0, -(sizeof (*u1)));
727 u1 = vlib_buffer_get_current (b0);
728
729 vlib_buffer_advance (b0, -(sizeof (*ip1)));
730 ip1 = vlib_buffer_get_current (b0);
731
732 copy_ip6_address (&client_address, &h0->peer_addr);
733
734 ia0 = ip6_interface_first_address (&ip6_main, sw_if_index);
735 if (ia0 == 0)
736 {
737 error0 = DHCPV6_PROXY_ERROR_NO_INTERFACE_ADDRESS;
738 goto drop_packet;
739 }
740
741 len = clib_net_to_host_u16 (r0->length);
742 memset (ip1, 0, sizeof (*ip1));
743 copy_ip6_address (&ip1->dst_address, &client_address);
744 u1->checksum = 0;
745 u1->src_port = clib_net_to_host_u16 (UDP_DST_PORT_dhcpv6_to_server);
746 u1->dst_port = clib_net_to_host_u16 (UDP_DST_PORT_dhcpv6_to_client);
747 u1->length = clib_host_to_net_u16 (len + sizeof (udp_header_t));
748
749 ip1->ip_version_traffic_class_and_flow_label =
750 ip0->ip_version_traffic_class_and_flow_label & 0x00000fff;
751 ip1->payload_length = u1->length;
752 ip1->protocol = PROTO_UDP;
753 ip1->hop_limit = HOP_COUNT_LIMIT;
754 copy_ip6_address (&ip1->src_address, ia0);
755
756 u1->checksum = ip6_tcp_udp_icmp_compute_checksum (vm, b0, ip1,
757 &bogus_length);
758 ASSERT (bogus_length == 0);
759
760 vlib_buffer_advance (b0, -(sizeof (ethernet_header_t)));
761 si0 = vnet_get_sw_interface (vnm, original_sw_if_index);
762 if (si0->type == VNET_SW_INTERFACE_TYPE_SUB)
763 vlib_buffer_advance (b0, -4 /* space for VLAN tag */ );
764
765 mac0 = vlib_buffer_get_current (b0);
766
767 hi0 = vnet_get_sup_hw_interface (vnm, original_sw_if_index);
768 ei0 = pool_elt_at_index (em->interfaces, hi0->hw_instance);
769 clib_memcpy (mac0->src_address, ei0->address, sizeof (ei0->address));
770 memset (&mac0->dst_address, 0xff, sizeof (mac0->dst_address));
771 mac0->type = (si0->type == VNET_SW_INTERFACE_TYPE_SUB) ?
772 clib_net_to_host_u16 (0x8100) : clib_net_to_host_u16 (0x86dd);
773
774 if (si0->type == VNET_SW_INTERFACE_TYPE_SUB)
775 {
776 u32 *vlan_tag = (u32 *) (mac0 + 1);
777 u32 tmp;
778 tmp = (si0->sub.id << 16) | 0x0800;
779 *vlan_tag = clib_host_to_net_u32 (tmp);
780 }
781
782 /* $$$ consider adding a dynamic next to the graph node, for performance */
783 f0 = vlib_get_frame_to_node (vm, hi0->output_node_index);
784 to_next0 = vlib_frame_vector_args (f0);
785 to_next0[0] = bi0;
786 f0->n_vectors = 1;
787 vlib_put_frame_to_node (vm, hi0->output_node_index, f0);
788
789 do_trace:
790 if (PREDICT_FALSE (b0->flags & VLIB_BUFFER_IS_TRACED))
791 {
792 dhcpv6_proxy_trace_t *tr = vlib_add_trace (vm, node,
793 b0, sizeof (*tr));
794 tr->which = 1; /* to client */
795 if (ia0)
796 copy_ip6_address ((ip6_address_t *) tr->packet_data, ia0);
797 tr->error = error0;
798 tr->original_sw_if_index = original_sw_if_index;
799 tr->sw_if_index = sw_if_index;
800 }
801 }
802 return from_frame->n_vectors;
803
804}

854int
855dhcp6_proxy_set_server (ip46_address_t * addr,
856 ip46_address_t * src_addr,
857 u32 rx_table_id, u32 server_table_id, int is_del)
858{
859 u32 rx_fib_index = 0;
860 int rc = 0;
861
862 const mfib_prefix_t all_dhcp_servers = {
863 .fp_len = 128,
864 .fp_proto = FIB_PROTOCOL_IP6,
865 .fp_grp_addr = {
866 .ip6 = all_dhcpv6_server_relay_agent_address,
867 }
868 };
869
870 if (ip46_address_is_zero (addr))
871 return VNET_API_ERROR_INVALID_DST_ADDRESS;
872
873 if (ip46_address_is_zero (src_addr))
874 return VNET_API_ERROR_INVALID_SRC_ADDRESS;
875
876 rx_fib_index = mfib_table_find_or_create_and_lock (FIB_PROTOCOL_IP6,
877 rx_table_id,
878 MFIB_SOURCE_DHCP);
879
880 if (is_del)
881 {
882 if (dhcp_proxy_server_del (FIB_PROTOCOL_IP6, rx_fib_index,
883 addr, server_table_id))
884 {
885 mfib_table_entry_delete (rx_fib_index,
886 &all_dhcp_servers, MFIB_SOURCE_DHCP);
887 mfib_table_unlock (rx_fib_index, FIB_PROTOCOL_IP6,
888 MFIB_SOURCE_DHCP);
889 }
890 }
891 else
892 {
893 const fib_route_path_t path_for_us = {
894 .frp_proto = DPO_PROTO_IP6,
895 .frp_addr = zero_addr,
896 .frp_sw_if_index = 0xffffffff,
897 .frp_fib_index = ~0,
898 .frp_weight = 0,
899 .frp_flags = FIB_ROUTE_PATH_LOCAL,
900 };
901 if (dhcp_proxy_server_add (FIB_PROTOCOL_IP6, addr, src_addr,
902 rx_fib_index, server_table_id))
903 {
904 mfib_table_entry_path_update (rx_fib_index,
905 &all_dhcp_servers,
906 MFIB_SOURCE_DHCP,
907 &path_for_us, MFIB_ITF_FLAG_FORWARD);
908 /*
909 * Each interface that is enabled in this table, needs to be added
910 * as an accepting interface, but this is not easily doable in VPP.
911 * So we cheat. Add a flag to the entry that indicates accept form
912 * any interface.
913 * We will still only accept on v6 enabled interfaces, since the
914 * input feature ensures this.
915 */
916 mfib_table_entry_update (rx_fib_index,
917 &all_dhcp_servers,
918 MFIB_SOURCE_DHCP,
919 MFIB_RPF_ID_NONE,
920 MFIB_ENTRY_FLAG_ACCEPT_ALL_ITF);
921 mfib_table_lock (rx_fib_index, FIB_PROTOCOL_IP6, MFIB_SOURCE_DHCP);
922 }
923 }
924
925 mfib_table_unlock (rx_fib_index, FIB_PROTOCOL_IP6, MFIB_SOURCE_DHCP);
926
927 return (rc);
928}

930static clib_error_t *
931dhcpv6_proxy_set_command_fn (vlib_main_t * vm,
932 unformat_input_t * input,
933 vlib_cli_command_t * cmd)
934{
935 ip46_address_t addr, src_addr;
936 int set_server = 0, set_src_address = 0;
937 u32 rx_table_id = 0, server_table_id = 0;
938 int is_del = 0;
939
940 while (unformat_check_input (input) != UNFORMAT_END_OF_INPUT)
941 {
942 if (unformat (input, "server %U", unformat_ip6_address, &addr.ip6))
943 set_server = 1;
944 else if (unformat (input, "src-address %U",
945 unformat_ip6_address, &src_addr.ip6))
946 set_src_address = 1;
947 else if (unformat (input, "server-fib-id %d", &server_table_id))
948 ;
949 else if (unformat (input, "rx-fib-id %d", &rx_table_id))
950 ;
951 else if (unformat (input, "delete") || unformat (input, "del"))
952 is_del = 1;
953 else
954 break;
955 }
956
957 if (is_del || (set_server && set_src_address))
958 {
959 int rv;
960
961 rv = dhcp6_proxy_set_server (&addr, &src_addr, rx_table_id,
962 server_table_id, is_del);
963
964 //TODO: Complete the errors
965 switch (rv)
966 {
967 case 0:
968 return 0;
969
970 case VNET_API_ERROR_INVALID_DST_ADDRESS:
971 return clib_error_return (0, "Invalid server address");
972
973 case VNET_API_ERROR_INVALID_SRC_ADDRESS:
974 return clib_error_return (0, "Invalid src address");
975
976 case VNET_API_ERROR_NO_SUCH_ENTRY:
977 return clib_error_return
978 (0, "Fib id %d: no per-fib DHCP server configured", rx_table_id);
979
980 default:
981 return clib_error_return (0, "BUG: rv %d", rv);
982 }
983 }
984 else
985 return clib_error_return (0, "parse error`%U'",
986 format_unformat_error, input);
987}
