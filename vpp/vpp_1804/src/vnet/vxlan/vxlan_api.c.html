52static void
53 vl_api_sw_interface_set_vxlan_bypass_t_handler
54 (vl_api_sw_interface_set_vxlan_bypass_t * mp)
55{
56 vl_api_sw_interface_set_vxlan_bypass_reply_t *rmp;
57 int rv = 0;
58 u32 sw_if_index = ntohl (mp->sw_if_index);
59
60 VALIDATE_SW_IF_INDEX (mp);
61
62 vnet_int_vxlan_bypass_mode (sw_if_index, mp->is_ipv6, mp->enable);
63 BAD_SW_IF_INDEX_LABEL;
64
65 REPLY_MACRO (VL_API_SW_INTERFACE_SET_VXLAN_BYPASS_REPLY);
66}

68static void vl_api_vxlan_add_del_tunnel_t_handler
69 (vl_api_vxlan_add_del_tunnel_t * mp)
70{
71 vl_api_vxlan_add_del_tunnel_reply_t *rmp;
72 int rv = 0;
73 u32 fib_index;
74
75 fib_index = fib_table_find (fib_ip_proto (mp->is_ipv6),
76 ntohl (mp->encap_vrf_id));
77 if (fib_index == ~0)
78 {
79 rv = VNET_API_ERROR_NO_SUCH_FIB;
80 goto out;
81 }
82
83 vnet_vxlan_add_del_tunnel_args_t a = {
84 .is_add = mp->is_add,
85 .is_ip6 = mp->is_ipv6,
86 .instance = ntohl (mp->instance),
87 .mcast_sw_if_index = ntohl (mp->mcast_sw_if_index),
88 .encap_fib_index = fib_index,
89 .decap_next_index = ntohl (mp->decap_next_index),
90 .vni = ntohl (mp->vni),
91 .dst = to_ip46 (mp->is_ipv6, mp->dst_address),
92 .src = to_ip46 (mp->is_ipv6, mp->src_address),
93 };
94
95 /* Check src & dst are different */
96 if (ip46_address_cmp (&a.dst, &a.src) == 0)
97 {
98 rv = VNET_API_ERROR_SAME_SRC_DST;
99 goto out;
100 }
101 if (ip46_address_is_multicast (&a.dst) &&
102 !vnet_sw_if_index_is_api_valid (a.mcast_sw_if_index))
103 {
104 rv = VNET_API_ERROR_INVALID_SW_IF_INDEX;
105 goto out;
106 }
107
108 u32 sw_if_index = ~0;
109 rv = vnet_vxlan_add_del_tunnel (&a, &sw_if_index);
110
111out:
112 /* *INDENT-OFF* */
113 REPLY_MACRO2(VL_API_VXLAN_ADD_DEL_TUNNEL_REPLY,
114 ({
115 rmp->sw_if_index = ntohl (sw_if_index);
116 }));
117 /* *INDENT-ON* */
118}

120static void send_vxlan_tunnel_details
121 (vxlan_tunnel_t * t, vl_api_registration_t * reg, u32 context)
122{
123 vl_api_vxlan_tunnel_details_t *rmp;
124 ip4_main_t *im4 = &ip4_main;
125 ip6_main_t *im6 = &ip6_main;
126 u8 is_ipv6 = !ip46_address_is_ip4 (&t->dst);
127
128 rmp = vl_msg_api_alloc (sizeof (*rmp));
129 memset (rmp, 0, sizeof (*rmp));
130 rmp->_vl_msg_id = ntohs (VL_API_VXLAN_TUNNEL_DETAILS);
131 if (is_ipv6)
132 {
133 memcpy (rmp->src_address, t->src.ip6.as_u8, 16);
134 memcpy (rmp->dst_address, t->dst.ip6.as_u8, 16);
135 rmp->encap_vrf_id = htonl (im6->fibs[t->encap_fib_index].ft_table_id);
136 }
137 else
138 {
139 memcpy (rmp->src_address, t->src.ip4.as_u8, 4);
140 memcpy (rmp->dst_address, t->dst.ip4.as_u8, 4);
141 rmp->encap_vrf_id = htonl (im4->fibs[t->encap_fib_index].ft_table_id);
142 }
143
144 rmp->instance = htonl (t->user_instance);
145 rmp->mcast_sw_if_index = htonl (t->mcast_sw_if_index);
146 rmp->vni = htonl (t->vni);
147 rmp->decap_next_index = htonl (t->decap_next_index);
148 rmp->sw_if_index = htonl (t->sw_if_index);
149 rmp->is_ipv6 = is_ipv6;
150 rmp->context = context;
151
152 vl_api_send_msg (reg, (u8 *) rmp);
153}

155static void vl_api_vxlan_tunnel_dump_t_handler
156 (vl_api_vxlan_tunnel_dump_t * mp)
157{
158 vl_api_registration_t *reg;
159 vxlan_main_t *vxm = &vxlan_main;
160 vxlan_tunnel_t *t;
161 u32 sw_if_index;
162
163 reg = vl_api_client_index_to_registration (mp->client_index);
164 if (!reg)
165 return;
166
167 sw_if_index = ntohl (mp->sw_if_index);
168
169 if (~0 == sw_if_index)
170 {
171 /* *INDENT-OFF* */
172 pool_foreach (t, vxm->tunnels,
173 ({
174 send_vxlan_tunnel_details(t, reg, mp->context);
175 }));
176 /* *INDENT-ON* */
177 }
178 else
179 {
180 if ((sw_if_index >= vec_len (vxm->tunnel_index_by_sw_if_index)) ||
181 (~0 == vxm->tunnel_index_by_sw_if_index[sw_if_index]))
182 {
183 return;
184 }
185 t = &vxm->tunnels[vxm->tunnel_index_by_sw_if_index[sw_if_index]];
186 send_vxlan_tunnel_details (t, reg, mp->context);
187 }
188}

190/*
191 * vpe_api_hookup
192 * Add vpe's API message handlers to the table.
193 * vlib has alread mapped shared memory and
194 * added the client registration handlers.
195 * See .../vlib-api/vlibmemory/memclnt_vlib.c:memclnt_process()
196 */
197#define vl_msg_name_crc_list
198#include <vnet/vnet_all_api_h.h>
199#undef vl_msg_name_crc_list
200
201static void
202setup_message_id_table (api_main_t * am)
203{
204#define _(id,n,crc) vl_msg_api_add_msg_name_crc (am, #n "_" #crc, id);
205 foreach_vl_msg_name_crc_vxlan;
206#undef _
207}
208
209static clib_error_t *
210vxlan_api_hookup (vlib_main_t * vm)
211{
212 api_main_t *am = &api_main;
213
214#define _(N,n) \
215 vl_msg_api_set_handlers(VL_API_##N, #n, \
216 vl_api_##n##_t_handler, \
217 vl_noop_handler, \
218 vl_api_##n##_t_endian, \
219 vl_api_##n##_t_print, \
220 sizeof(vl_api_##n##_t), 1);
221 foreach_vpe_api_msg;
222#undef _
223
224 am->api_trace_cfg[VL_API_VXLAN_ADD_DEL_TUNNEL].size += 16 * sizeof (u32);
225
226 /*
227 * Set up the (msg_name, crc, message-id) table
228 */
229 setup_message_id_table (am);
230
231 return 0;
232}
