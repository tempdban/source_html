33typedef CLIB_PACKED (struct {
34 ip4_header_t ip4; /* 20 bytes */
35 udp_header_t udp; /* 8 bytes */
36 vxlan_header_t vxlan; /* 8 bytes */
37}) ip4_vxlan_header_t;
38
39typedef CLIB_PACKED (struct {
40 ip6_header_t ip6; /* 40 bytes */
41 udp_header_t udp; /* 8 bytes */
42 vxlan_header_t vxlan; /* 8 bytes */
43}) ip6_vxlan_header_t;
44
45typedef CLIB_PACKED(struct {
46 /*
47 * Key fields: ip src and vxlan vni on incoming VXLAN packet
48 * all fields in NET byte order
49 */
50 union {
51 struct {
52 u32 src;
53 u32 vni; /* shifted left 8 bits */
54 };
55 u64 as_u64;
56 };
57}) vxlan4_tunnel_key_t;
58
59typedef CLIB_PACKED(struct {
60 /*
61 * Key fields: ip src and vxlan vni on incoming VXLAN packet
62 * all fields in NET byte order
63 */
64 ip6_address_t src;
65 u32 vni; /* shifted left 8 bits */
66}) vxlan6_tunnel_key_t;
67
68typedef struct {
69 /* Required for pool_get_aligned */
70 CLIB_CACHE_LINE_ALIGN_MARK(cacheline0);
71
72 /* Rewrite string. $$$$ embed vnet_rewrite header */
73 u8 * rewrite;
74
75 /* FIB DPO for IP forwarding of VXLAN encap packet */
76 dpo_id_t next_dpo;
77
78 /* vxlan VNI in HOST byte order */
79 u32 vni;
80
81 /* tunnel src and dst addresses */
82 ip46_address_t src;
83 ip46_address_t dst;
84
85 /* mcast packet output intfc index (used only if dst is mcast) */
86 u32 mcast_sw_if_index;
87
88 /* decap next index */
89 u32 decap_next_index;
90
91 /* The FIB index for src/dst addresses */
92 u32 encap_fib_index;
93
94 /* vnet intfc index */
95 u32 sw_if_index;
96 u32 hw_if_index;
97
98 /**
99 * Linkage into the FIB object graph
100 */
101 fib_node_t node;
102
103 /*
104 * The FIB entry for (depending on VXLAN tunnel is unicast or mcast)
105 * sending unicast VXLAN encap packets or receiving mcast VXLAN packets
106 */
107 fib_node_index_t fib_entry_index;
108 adj_index_t mcast_adj_index;
109
110 /**
111 * The tunnel is a child of the FIB entry for its desintion. This is
112 * so it receives updates when the forwarding information for that entry
113 * changes.
114 * The tunnels sibling index on the FIB entry's dependency list.
115 */
116 u32 sibling_index;
117
118 u32 dev_instance; /* Real device instance in tunnel vector */
119 u32 user_instance; /* Instance name being shown to user */
120} vxlan_tunnel_t;
121
122#define foreach_vxlan_input_next \
123_(DROP, "error-drop") \
124_(L2_INPUT, "l2-input")
125
126typedef enum {
127#define _(s,n) VXLAN_INPUT_NEXT_##s,
128 foreach_vxlan_input_next
129#undef _
130 VXLAN_INPUT_N_NEXT,
131} vxlan_input_next_t;
132
133typedef enum {
134#define vxlan_error(n,s) VXLAN_ERROR_##n,
135#include <vnet/vxlan/vxlan_error.def>
136#undef vxlan_error
137 VXLAN_N_ERROR,
138} vxlan_input_error_t;
139
140typedef struct {
141 /* vector of encap tunnel instances */
142 vxlan_tunnel_t * tunnels;
143
144 /* lookup tunnel by key */
145 uword * vxlan4_tunnel_by_key; /* keyed on ipv4.dst + vni */
146 uword * vxlan6_tunnel_by_key; /* keyed on ipv6.dst + vni */
147
148 /* local VTEP IPs ref count used by vxlan-bypass node to check if
149 received VXLAN packet DIP matches any local VTEP address */
150 uword * vtep4; /* local ip4 VTEPs keyed on their ip4 addr */
151 uword * vtep6; /* local ip6 VTEPs keyed on their ip6 addr */
152
153 /* mcast shared info */
154 uword * mcast_shared; /* keyed on mcast ip46 addr */
155
156 /* Mapping from sw_if_index to tunnel index */
157 u32 * tunnel_index_by_sw_if_index;
158
159 /* convenience */
160 vlib_main_t * vlib_main;
161 vnet_main_t * vnet_main;
162
163 /* Record used instances */
164 uword *instance_used;
165} vxlan_main_t;
