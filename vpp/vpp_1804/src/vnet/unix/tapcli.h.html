25/** TAP CLI errors */
26#define foreach_tapcli_error \
27 /* Must be first. */ \
28 _(NONE, "no error") \
29 _(READ, "read error") \
30 _(BUFFER_ALLOC, "buffer allocation error") \
31 _(UNKNOWN, "unknown error")
32
33typedef enum
34{
35#define _(sym,str) TAPCLI_ERROR_##sym,
36 foreach_tapcli_error
37#undef _
