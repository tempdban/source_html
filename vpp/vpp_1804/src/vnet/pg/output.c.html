40#include <vppinfra/string.h>
41#include <vlib/vlib.h>
42#include <vnet/vnet.h>
43#include <vnet/pg/pg.h>
44#include <vnet/ethernet/ethernet.h>
45
46uword
47pg_output (vlib_main_t * vm, vlib_node_runtime_t * node, vlib_frame_t * frame)
48{
49 pg_main_t *pg = &pg_main;
50 u32 *buffers = vlib_frame_args (frame);
51 uword n_buffers = frame->n_vectors;
52 uword n_left = n_buffers;
53 vnet_interface_output_runtime_t *rd = (void *) node->runtime_data;
54 pg_interface_t *pif = pool_elt_at_index (pg->interfaces, rd->dev_instance);
55
56 if (PREDICT_FALSE (pif->lockp != 0))
57 while (__sync_lock_test_and_set (pif->lockp, 1))
58 ;
59
60 while (n_left > 0)
61 {
62 n_left--;
63 u32 bi0 = buffers[0];
64 vlib_buffer_t *b = vlib_get_buffer (vm, bi0);
65 buffers++;
66
67 if (b->flags & VLIB_BUFFER_IS_TRACED)
68 {
69 pg_output_trace_t *t = vlib_add_trace (vm, node, b, sizeof (*t));
70 t->buffer_index = bi0;
71 clib_memcpy (&t->buffer, b, sizeof (b[0]) - sizeof (b->pre_data));
72 clib_memcpy (t->buffer.pre_data, b->data + b->current_data,
73 sizeof (t->buffer.pre_data));
74 }
75
76 if (pif->pcap_file_name != 0)
77 pcap_add_buffer (&pif->pcap_main, vm, bi0, ETHERNET_MAX_PACKET_BYTES);
78 }
79 if (pif->pcap_file_name != 0)
80 pcap_write (&pif->pcap_main);
81
82
83 vlib_buffer_free (vm, vlib_frame_args (frame), n_buffers);
84 if (PREDICT_FALSE (pif->lockp != 0))
85 *pif->lockp = 0;
86 return n_buffers;
87}
