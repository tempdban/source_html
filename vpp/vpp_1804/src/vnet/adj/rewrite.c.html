43void
44vnet_rewrite_copy_slow_path (vnet_rewrite_data_t * p0,
45 vnet_rewrite_data_t * rw0,
46 word n_left, uword most_likely_size)
47{
48 uword n_done =
49 round_pow2 (most_likely_size, sizeof (rw0[0])) / sizeof (rw0[0]);
50
51 p0 -= n_done;
52 rw0 -= n_done;
53
54 /* As we enter the cleanup loop, p0 and rw0 point to the last chunk written
55 by the fast path. Hence, the constant 1, which the
56 vnet_rewrite_copy_one macro renders as p0[-1] = rw0[-1]. */
57
58 while (n_left > 0)
59 {
60 vnet_rewrite_copy_one (p0, rw0, 1);
61 p0--;
62 rw0--;
63 n_left--;
64 }
65}

67u8 *
68format_vnet_rewrite (u8 * s, va_list * args)
69{
70 vnet_rewrite_header_t *rw = va_arg (*args, vnet_rewrite_header_t *);
71 u32 max_data_bytes = va_arg (*args, u32);
72 CLIB_UNUSED (u32 indent) = va_arg (*args, u32);
73 vnet_main_t *vnm = vnet_get_main ();
74
75 if (rw->sw_if_index != ~0)
76 {
77 vnet_sw_interface_t *si;
78 si = vnet_get_sw_interface_safe (vnm, rw->sw_if_index);
79 if (NULL != si)
80 s = format (s, "%U:", format_vnet_sw_interface_name, vnm, si);
81 else
82 s = format (s, "DELETED:%d", rw->sw_if_index);
83 }
84
85 s = format (s, " mtu:%d", rw->max_l3_packet_bytes);
86
87 /* Format rewrite string. */
88 if (rw->data_bytes > 0)
89 s = format (s, " %U",
90 format_hex_bytes,
91 rw->data + max_data_bytes - rw->data_bytes, rw->data_bytes);
92
93 return s;
94}

103void
104vnet_rewrite_init (vnet_main_t * vnm,
105 u32 sw_if_index,
106 u32 this_node, u32 next_node, vnet_rewrite_header_t * rw)
107{
108 rw->sw_if_index = sw_if_index;
109 rw->next_index = vlib_node_add_next (vnm->vlib_main, this_node, next_node);
110 rw->max_l3_packet_bytes =
111 vnet_sw_interface_get_mtu (vnm, sw_if_index, VLIB_TX);
112}

114void
115vnet_rewrite_update_mtu (vnet_main_t * vnm, vnet_rewrite_header_t * rw)
116{
117 rw->max_l3_packet_bytes =
118 vnet_sw_interface_get_mtu (vnm, rw->sw_if_index, VLIB_TX);
119}

121void
122vnet_rewrite_for_sw_interface (vnet_main_t * vnm,
123 vnet_link_t link_type,
124 u32 sw_if_index,
125 u32 node_index,
126 void *dst_address,
127 vnet_rewrite_header_t * rw,
128 u32 max_rewrite_bytes)
129{
130
131 vnet_hw_interface_t *hw = vnet_get_sup_hw_interface (vnm, sw_if_index);
132 vnet_hw_interface_class_t *hc =
133 vnet_get_hw_interface_class (vnm, hw->hw_class_index);
134 u8 *rewrite = NULL;
135
136 vnet_rewrite_init (vnm, sw_if_index, node_index,
137 vnet_tx_node_index_for_sw_interface (vnm, sw_if_index),
138 rw);
139
140 ASSERT (hc->build_rewrite);
141 rewrite = hc->build_rewrite (vnm, sw_if_index, link_type, dst_address);
142
143 ASSERT (vec_len (rewrite) < max_rewrite_bytes);
144 vnet_rewrite_set_data_internal (rw, max_rewrite_bytes, rewrite,
145 vec_len (rewrite));
146 vec_free (rewrite);
147}

149void
150vnet_rewrite_for_tunnel (vnet_main_t * vnm,
151 u32 tx_sw_if_index,
152 u32 rewrite_node_index,
153 u32 post_rewrite_node_index,
154 vnet_rewrite_header_t * rw,
155 u8 * rewrite_data, u32 rewrite_length)
156{
157 ip_adjacency_t *adj = 0;
158 /*
159 * Installed into vnet_buffer(b)->sw_if_index[VLIB_TX] e.g.
160 * by ip4_rewrite_inline. If the post-rewrite node injects into
161 * ipX-forward, this will be interpreted as a FIB number.
162 */
163 rw->sw_if_index = tx_sw_if_index;
164 rw->next_index = vlib_node_add_next (vnm->vlib_main, rewrite_node_index,
165 post_rewrite_node_index);
166 rw->max_l3_packet_bytes = (u16) ~ 0; /* we can't know at this point */
167
168 ASSERT (rewrite_length < sizeof (adj->rewrite_data));
169 /* Leave room for ethernet + VLAN tag */
170 vnet_rewrite_set_data_internal (rw, sizeof (adj->rewrite_data),
171 rewrite_data, rewrite_length);
172}

174void
175serialize_vnet_rewrite (serialize_main_t * m, va_list * va)
176{
177 vnet_rewrite_header_t *rw = va_arg (*va, vnet_rewrite_header_t *);
178 u32 max_data_bytes = va_arg (*va, u32);
179 u8 *p;
180
181 serialize_integer (m, rw->sw_if_index, sizeof (rw->sw_if_index));
182 serialize_integer (m, rw->data_bytes, sizeof (rw->data_bytes));
183 serialize_integer (m, rw->max_l3_packet_bytes,
184 sizeof (rw->max_l3_packet_bytes));
185 p = serialize_get (m, rw->data_bytes);
186 clib_memcpy (p, vnet_rewrite_get_data_internal (rw, max_data_bytes),
187 rw->data_bytes);
188}
189
190void
191unserialize_vnet_rewrite (serialize_main_t * m, va_list * va)
192{
193 vnet_rewrite_header_t *rw = va_arg (*va, vnet_rewrite_header_t *);
194 u32 max_data_bytes = va_arg (*va, u32);
195 u8 *p;
196
197 /* It is up to user to fill these in. */
198 rw->next_index = ~0;
199
200 unserialize_integer (m, &rw->sw_if_index, sizeof (rw->sw_if_index));
201 unserialize_integer (m, &rw->data_bytes, sizeof (rw->data_bytes));
202 unserialize_integer (m, &rw->max_l3_packet_bytes,
203 sizeof (rw->max_l3_packet_bytes));
204 p = unserialize_get (m, rw->data_bytes);
205 clib_memcpy (vnet_rewrite_get_data_internal (rw, max_data_bytes), p,
206 rw->data_bytes);
207}
