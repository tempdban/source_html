18#ifndef _VNET_DEVICES_VIRTIO_TAP_H_
19#define _VNET_DEVICES_VIRTIO_TAP_H_
20
21#ifndef MIN
22#define MIN(x,y) (((x)<(y))?(x):(y))
23#endif
24
25typedef struct
26{
27 u32 id;
28 u8 mac_addr_set;
29 u8 mac_addr[6];
30 u16 rx_ring_sz;
31 u16 tx_ring_sz;
32 u8 *host_namespace;
33 u8 *host_if_name;
34 u8 host_mac_addr[6];
35 u8 *host_bridge;
36 ip4_address_t host_ip4_addr;
37 u8 host_ip4_prefix_len;
38 ip4_address_t host_ip4_gw;
39 u8 host_ip4_gw_set;
40 ip6_address_t host_ip6_addr;
41 u8 host_ip6_prefix_len;
42 ip6_address_t host_ip6_gw;
43 u8 host_ip6_gw_set;
44 /* return */
45 u32 sw_if_index;
46 int rv;
47 clib_error_t *error;
48} tap_create_if_args_t;
49
50/** TAP interface details struct */
51typedef struct
52{
53 u32 id;
54 u32 sw_if_index;
55 u8 dev_name[64];
56 u16 tx_ring_sz;
57 u16 rx_ring_sz;
58 u8 host_mac_addr[6];
59 u8 host_if_name[64];
60 u8 host_namespace[64];
61 u8 host_bridge[64];
62 u8 host_ip4_addr[4];
63 u8 host_ip4_prefix_len;
64 u8 host_ip6_addr[16];
65 u8 host_ip6_prefix_len;
66} tap_interface_details_t;
67
68typedef struct
69{
70 /* bit-map of in-use IDs */
71 uword *tap_ids;
72} tap_main_t;
73
74void tap_create_if (vlib_main_t * vm, tap_create_if_args_t * args);
75int tap_delete_if (vlib_main_t * vm, u32 sw_if_index);
76int tap_dump_ifs (tap_interface_details_t ** out_tapids);
77
78#endif /* _VNET_DEVICES_VIRTIO_TAP_H_ */
