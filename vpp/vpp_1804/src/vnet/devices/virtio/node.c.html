41typedef enum

50#define _(n,s) s,
51 foreach_virtio_input_error

64static u8 *
65format_virtio_input_trace (u8 * s, va_list * args)
66{
67 CLIB_UNUSED (vlib_main_t * vm) = va_arg (*args, vlib_main_t *);
68 CLIB_UNUSED (vlib_node_t * node) = va_arg (*args, vlib_node_t *);
69 virtio_input_trace_t *t = va_arg (*args, virtio_input_trace_t *);
70 u32 indent = format_get_indent (s);
71
72 s = format (s, "virtio: hw_if_index %d next-index %d vring %u len %u",
73 t->hw_if_index, t->next_index, t->ring, t->len);
74 s = format (s, "\n%Uhdr: flags 0x%02x gso_type 0x%02x hdr_len %u "
75 "gso_size %u csum_start %u csum_offset %u num_buffers %u",
76 format_white_space, indent + 2,
77 t->hdr.flags, t->hdr.gso_type, t->hdr.hdr_len, t->hdr.gso_size,
78 t->hdr.csum_start, t->hdr.csum_offset, t->hdr.num_buffers);
79 return s;
80}

82static_always_inline void
83virtio_refill_vring (vlib_main_t * vm, virtio_vring_t * vring)
84{
85 const int hdr_sz = sizeof (struct virtio_net_hdr_v1);
86 u16 used, next, avail, n_slots;
87 u16 sz = vring->size;
88 u16 mask = sz - 1;
89
90 used = vring->desc_in_use;
91
92 if (sz - used < sz / 8)
93 return;
94
95 n_slots = sz - used;
96 next = vring->desc_next;
97 avail = vring->avail->idx;
98 n_slots = vlib_buffer_alloc_to_ring (vm, vring->buffers, next, vring->size,
99 n_slots);
100
101 while (n_slots)
102 {
103 struct vring_desc *d = &vring->desc[next];;
104 vlib_buffer_t *b = vlib_get_buffer (vm, vring->buffers[next]);
105 d->addr = pointer_to_uword (vlib_buffer_get_current (b)) - hdr_sz;
106 d->len = VLIB_BUFFER_DATA_SIZE + hdr_sz;
107 d->flags = VRING_DESC_F_WRITE;
108 vring->avail->ring[avail & mask] = next;
109 avail++;
110 next = (next + 1) & mask;
111 n_slots--;
112 used++;
113 }
114 CLIB_MEMORY_STORE_BARRIER ();
115 vring->avail->idx = avail;
116 vring->desc_next = next;
117 vring->desc_in_use = used;
118
119 if ((vring->used->flags & VIRTIO_RING_FLAG_MASK_INT) == 0)
120 {
121 u64 b = 1;
122 CLIB_UNUSED (int r) = write (vring->kick_fd, &b, sizeof (b));
123 }
124}

126static_always_inline uword
127virtio_device_input_inline (vlib_main_t * vm, vlib_node_runtime_t * node,
128 vlib_frame_t * frame, virtio_if_t * vif, u16 qid)
129{
130 vnet_main_t *vnm = vnet_get_main ();
131 u32 thread_index = vlib_get_thread_index ();
132 uword n_trace = vlib_get_trace_count (vm, node);
133 virtio_vring_t *vring = vec_elt_at_index (vif->vrings, 0);
134 u32 next_index = VNET_DEVICE_INPUT_NEXT_ETHERNET_INPUT;
135 const int hdr_sz = sizeof (struct virtio_net_hdr_v1);
136 u32 *to_next = 0;
137 u32 n_rx_packets = 0;
138 u32 n_rx_bytes = 0;
139 u16 mask = vring->size - 1;
140 u16 last = vring->last_used_idx;
141 u16 n_left = vring->used->idx - last;
142
143 if (n_left == 0)
144 goto refill;
145
146 while (n_left)
147 {
148 u32 n_left_to_next;
149 u32 next0 = next_index;
150 vlib_get_next_frame (vm, node, next_index, to_next, n_left_to_next);
151
152 while (n_left && n_left_to_next)
153 {
154 u16 num_buffers;
155 struct vring_used_elem *e = &vring->used->ring[last & mask];
156 struct virtio_net_hdr_v1 *hdr;
157 u16 slot = e->id;
158 u16 len = e->len - hdr_sz;
159 u32 bi0 = vring->buffers[slot];
160 vlib_buffer_t *b0 = vlib_get_buffer (vm, bi0);
161 hdr = vlib_buffer_get_current (b0) - hdr_sz;
162 num_buffers = hdr->num_buffers;
163
164 b0->current_data = 0;
165 b0->current_length = len;
166 b0->total_length_not_including_first_buffer = 0;
167 b0->flags = VLIB_BUFFER_TOTAL_LENGTH_VALID;
168 vnet_buffer (b0)->sw_if_index[VLIB_RX] = vif->sw_if_index;
169 vnet_buffer (b0)->sw_if_index[VLIB_TX] = (u32) ~ 0;
170
171 /* if multisegment packet */
172 if (PREDICT_FALSE (num_buffers > 1))
173 {
174 vlib_buffer_t *pb, *cb;
175 pb = b0;
176 while (num_buffers > 1)
177 {
178 last++;
179 e = &vring->used->ring[last & mask];
180 u32 cbi = vring->buffers[e->id];
181 cb = vlib_get_buffer (vm, cbi);
182
183 /* current buffer */
184 cb->current_data = -hdr_sz;
185 cb->current_length = e->len;
186
187 /* previous buffer */
188 pb->next_buffer = cbi;
189 pb->flags |= VLIB_BUFFER_NEXT_PRESENT;
190
191 /* first buffer */
192 b0->total_length_not_including_first_buffer += e->len;
193
194 pb = cb;
195 vring->desc_in_use--;
196 num_buffers--;
197 n_left--;
198 }
199 }
200
201 if (PREDICT_FALSE (vif->per_interface_next_index != ~0))
202 next0 = vif->per_interface_next_index;
203 else
204 /* redirect if feature path enabled */
205 vnet_feature_start_device_input_x1 (vif->sw_if_index, &next0, b0);
206 /* trace */
207 VLIB_BUFFER_TRACE_TRAJECTORY_INIT (b0);
208
209 if (PREDICT_FALSE (n_trace > 0))
210 {
211 virtio_input_trace_t *tr;
212 vlib_trace_buffer (vm, node, next0, b0,
213 /* follow_chain */ 0);
214 vlib_set_trace_count (vm, node, --n_trace);
215 tr = vlib_add_trace (vm, node, b0, sizeof (*tr));
216 tr->next_index = next0;
217 tr->hw_if_index = vif->hw_if_index;
218 tr->len = len;
219 clib_memcpy (&tr->hdr, hdr, hdr_sz);
220 }
221
222 /* enqueue buffer */
223 to_next[0] = bi0;
224 vring->desc_in_use--;
225 to_next += 1;
226 n_left_to_next--;
227 n_left--;
228 last++;
229
230 /* enqueue */
231 vlib_validate_buffer_enqueue_x1 (vm, node, next_index, to_next,
232 n_left_to_next, bi0, next0);
233
234 /* next packet */
235 n_rx_packets++;
236 n_rx_bytes += len;
237 }
238 vlib_put_next_frame (vm, node, next_index, n_left_to_next);
239 }
240 vring->last_used_idx = last;
241
242 vlib_increment_combined_counter (vnm->interface_main.combined_sw_if_counters
243 + VNET_INTERFACE_COUNTER_RX, thread_index,
244 vif->hw_if_index, n_rx_packets,
245 n_rx_bytes);
246
247refill:
248 virtio_refill_vring (vm, vring);
249
250 return n_rx_packets;
251}
252
253static uword
254virtio_input_fn (vlib_main_t * vm, vlib_node_runtime_t * node,
255 vlib_frame_t * frame)
256{
257 u32 n_rx = 0;
258 virtio_main_t *nm = &virtio_main;
259 vnet_device_input_runtime_t *rt = (void *) node->runtime_data;
260 vnet_device_and_queue_t *dq;
261
262 foreach_device_and_queue (dq, rt->devices_and_queues)
263 {
264 virtio_if_t *mif;
265 mif = vec_elt_at_index (nm->interfaces, dq->dev_instance);
266 if (mif->flags & VIRTIO_IF_FLAG_ADMIN_UP)
267 {
268 n_rx += virtio_device_input_inline (vm, node, frame, mif,
269 dq->queue_id);
270 }
271 }
272
273 return n_rx;
274}
