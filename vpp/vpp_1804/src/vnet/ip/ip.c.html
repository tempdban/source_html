19u8
20ip_is_zero (ip46_address_t * ip46_address, u8 is_ip4)
21{
22 if (is_ip4)
23 return (ip46_address->ip4.as_u32 == 0);
24 else
25 return (ip46_address->as_u64[0] == 0 && ip46_address->as_u64[1] == 0);
26}

28u8
29ip_is_local_host (ip46_address_t * ip46_address, u8 is_ip4)
30{
31 if (is_ip4)
32 return (ip46_address->ip4.as_u8[0] == 127);
33 else
34 return (ip46_address->as_u64[0] == 0 &&
35 clib_net_to_host_u64 (ip46_address->as_u64[1]) == 1);
36}

38u8
39ip4_is_local_host (ip4_address_t * ip4_address)
40{
41 return (ip4_address->as_u8[0] == 127);
42}

44u8
45ip6_is_local_host (ip6_address_t * ip6_address)
46{
47 return (ip6_address->as_u64[0] == 0 &&
48 clib_net_to_host_u64 (ip6_address->as_u64[1]) == 1);
49}

51/**
52 * Checks that an ip is local to the requested fib
53 */
54u8
55ip_is_local (u32 fib_index, ip46_address_t * ip46_address, u8 is_ip4)
56{
57 fib_node_index_t fei;
58 fib_entry_flag_t flags;
59 fib_prefix_t prefix;
60
61 /* Check if requester is local */
62 if (is_ip4)
63 {
64 prefix.fp_len = 32;
65 prefix.fp_proto = FIB_PROTOCOL_IP4;
66 }
67 else
68 {
69 prefix.fp_len = 128;
70 prefix.fp_proto = FIB_PROTOCOL_IP6;
71 }
72
73 clib_memcpy (&prefix.fp_addr, ip46_address, sizeof (ip46_address_t));
74 fei = fib_table_lookup (fib_index, &prefix);
75 flags = fib_entry_get_flags (fei);
76
77 return (flags & FIB_ENTRY_FLAG_LOCAL);
78}

80void
81ip_copy (ip46_address_t * dst, ip46_address_t * src, u8 is_ip4)
82{
83 if (is_ip4)
84 dst->ip4.as_u32 = src->ip4.as_u32;
85 else
86 clib_memcpy (&dst->ip6, &src->ip6, sizeof (ip6_address_t));
87}

89void
90ip_set (ip46_address_t * dst, void *src, u8 is_ip4)
91{
92 if (is_ip4)
93 dst->ip4.as_u32 = ((ip4_address_t *) src)->as_u32;
94 else
95 clib_memcpy (&dst->ip6, (ip6_address_t *) src, sizeof (ip6_address_t));
96}

98u8
99ip_interface_has_address (u32 sw_if_index, ip46_address_t * ip, u8 is_ip4)
100{
101 ip_interface_address_t *ia = 0;
102
103 if (is_ip4)
104 {
105 ip_lookup_main_t *lm4 = &ip4_main.lookup_main;
106 ip4_address_t *ip4;
107 /* *INDENT-OFF* */
108 foreach_ip_interface_address (lm4, ia, sw_if_index, 1 /* unnumbered */ ,
109 ({
110 ip4 = ip_interface_address_get_address (lm4, ia);
111 if (ip4_address_compare (ip4, &ip->ip4) == 0)
112 return 1;
113 }));
114 /* *INDENT-ON* */
115 }
116 else
117 {
118 ip_lookup_main_t *lm6 = &ip6_main.lookup_main;
119 ip6_address_t *ip6;
120 /* *INDENT-OFF* */
121 foreach_ip_interface_address (lm6, ia, sw_if_index, 1 /* unnumbered */ ,
122 ({
123 ip6 = ip_interface_address_get_address (lm6, ia);
124 if (ip6_address_compare (ip6, &ip->ip6) == 0)
125 return 1;
126 }));
127 /* *INDENT-ON* */
128 }
129 return 0;
130}

132void *
133ip_interface_get_first_ip (u32 sw_if_index, u8 is_ip4)
134{
135 ip_lookup_main_t *lm4 = &ip4_main.lookup_main;
136 ip_lookup_main_t *lm6 = &ip6_main.lookup_main;
137 ip_interface_address_t *ia = 0;
138
139 if (is_ip4)
140 {
141 /* *INDENT-OFF* */
142 foreach_ip_interface_address (lm4, ia, sw_if_index, 1 /* unnumbered */ ,
143 ({
144 return ip_interface_address_get_address (lm4, ia);
145 }));
146 /* *INDENT-ON* */
147 }
148 else
149 {
150 /* *INDENT-OFF* */
151 foreach_ip_interface_address (lm6, ia, sw_if_index, 1 /* unnumbered */ ,
152 ({
153 ip6_address_t *rv;
154 rv = ip_interface_address_get_address (lm6, ia);
155 /* Trying to use a link-local ip6 src address is a fool's errand */
156 if (!ip6_address_is_link_local_unicast (rv))
157 return rv;
158 }));
159 /* *INDENT-ON* */
160 }
161
162 return 0;
163}

165void
166ip4_address_normalize (ip4_address_t * ip4, u8 preflen)
167{
168 ASSERT (preflen <= 32);
169 if (preflen == 0)
170 ip4->data_u32 = 0;
171 else
172 ip4->data_u32 &= clib_net_to_host_u32 (0xffffffff << (32 - preflen));
173}
174
175void
176ip6_address_normalize (ip6_address_t * ip6, u8 preflen)
177{
178 ASSERT (preflen <= 128);
179 if (preflen == 0)
180 {
181 ip6->as_u64[0] = 0;
182 ip6->as_u64[1] = 0;
183 }
184 else if (preflen <= 64)
185 {
186 ip6->as_u64[0] &=
187 clib_host_to_net_u64 (0xffffffffffffffffL << (64 - preflen));
188 ip6->as_u64[1] = 0;
189 }
190 else
191 ip6->as_u64[1] &=
192 clib_host_to_net_u64 (0xffffffffffffffffL << (128 - preflen));
193}

195void
196ip4_preflen_to_mask (u8 pref_len, ip4_address_t * ip)
197{
198 if (pref_len == 0)
199 ip->as_u32 = 0;
200 else
201 ip->as_u32 = clib_host_to_net_u32 (~((1 << (32 - pref_len)) - 1));
202}

204u32
205ip4_mask_to_preflen (ip4_address_t * mask)
206{
207 if (mask->as_u32 == 0)
208 return 0;
209 return (32 - log2_first_set (clib_net_to_host_u32 (mask->as_u32)));
210}

212void
213ip4_prefix_max_address_host_order (ip4_address_t * ip, u8 plen,
214 ip4_address_t * res)
215{
216 u32 not_mask;
217 not_mask = (1 << (32 - plen)) - 1;
218 res->as_u32 = clib_net_to_host_u32 (ip->as_u32) + not_mask;
219}

221void
222ip6_preflen_to_mask (u8 pref_len, ip6_address_t * mask)
223{
224 if (pref_len == 0)
225 {
226 mask->as_u64[0] = 0;
227 mask->as_u64[1] = 0;
228 }
229 else if (pref_len <= 64)
230 {
231 mask->as_u64[0] =
232 clib_host_to_net_u64 (0xffffffffffffffffL << (64 - pref_len));
233 mask->as_u64[1] = 0;
234 }
235 else
236 {
237 mask->as_u64[1] =
238 clib_host_to_net_u64 (0xffffffffffffffffL << (128 - pref_len));
239 }
240}
241
242void
243ip6_prefix_max_address_host_order (ip6_address_t * ip, u8 plen,
244 ip6_address_t * res)
245{
246 u64 not_mask;
247 if (plen == 0)
248 {
249 res->as_u64[0] = 0xffffffffffffffffL;
250 res->as_u64[1] = 0xffffffffffffffffL;
251 }
252 else if (plen <= 64)
253 {
254 not_mask = ((u64) 1 << (64 - plen)) - 1;
255 res->as_u64[0] = clib_net_to_host_u64 (ip->as_u64[0]) + not_mask;
256 res->as_u64[1] = 0xffffffffffffffffL;
257 }
258 else
259 {
260 not_mask = ((u64) 1 << (128 - plen)) - 1;
261 res->as_u64[1] = clib_net_to_host_u64 (ip->as_u64[1]) + not_mask;
262 }
263}

265u32
266ip6_mask_to_preflen (ip6_address_t * mask)
267{
268 u8 first1, first0;
269 if (mask->as_u64[0] == 0 && mask->as_u64[1] == 0)
270 return 0;
271 first1 = log2_first_set (clib_net_to_host_u64 (mask->as_u64[1]));
272 first0 = log2_first_set (clib_net_to_host_u64 (mask->as_u64[0]));
273
274 if (first1 != 0)
275 return 128 - first1;
276 else
277 return 64 - first0;
278}
