241always_inline uword
242ip6_destination_matches_route (const ip6_main_t * im,
243 const ip6_address_t * key,
244 const ip6_address_t * dest, uword dest_length)
245{
246 int i;
247 for (i = 0; i < ARRAY_LEN (key->as_uword); i++)
248 {
249 if ((key->as_uword[i] ^ dest->as_uword[i]) & im->
250 fib_masks[dest_length].as_uword[i])
251 return 0;
252 }
253 return 1;
254}
255
256always_inline uword
257ip6_destination_matches_interface (ip6_main_t * im,
258 ip6_address_t * key,
259 ip_interface_address_t * ia)
260{
261 ip6_address_t *a = ip_interface_address_get_address (&im->lookup_main, ia);
262 return ip6_destination_matches_route (im, key, a, ia->address_length);
263}
264
265/* As above but allows for unaligned destinations (e.g. works right from IP header of packet). */
266always_inline uword
267ip6_unaligned_destination_matches_route (ip6_main_t * im,
268 ip6_address_t * key,
269 ip6_address_t * dest,
270 uword dest_length)
271{
272 int i;
273 for (i = 0; i < ARRAY_LEN (key->as_uword); i++)
274 {
275 if ((clib_mem_unaligned (&key->as_uword[i], uword) ^ dest->as_uword[i])
276 & im->fib_masks[dest_length].as_uword[i])
277 return 0;
278 }
279 return 1;
280}

284always_inline int
285ip6_src_address_for_packet (ip_lookup_main_t * lm,
286 u32 sw_if_index,
287 const ip6_address_t * dst, ip6_address_t * src)
288{
289 if (ip6_address_is_link_local_unicast (dst))
290 {
291 return ip6_get_ll_address (sw_if_index, src);
292 }
293 else
294 {
295 u32 if_add_index =
296 lm->if_address_pool_index_by_sw_if_index[sw_if_index];
297 if (PREDICT_TRUE (if_add_index != ~0))
298 {
299 ip_interface_address_t *if_add =
300 pool_elt_at_index (lm->if_address_pool, if_add_index);
301 ip6_address_t *if_ip =
302 ip_interface_address_get_address (lm, if_add);
303 *src = *if_ip;
304 return (!0);
305 }
306 }
307
308 src->as_u64[0] = 0;
309 src->as_u64[1] = 0;
310
311 return (0);
312}

314/* Find interface address which matches destination. */
315always_inline ip6_address_t *
316ip6_interface_address_matching_destination (ip6_main_t * im,
317 ip6_address_t * dst,
318 u32 sw_if_index,
319 ip_interface_address_t **
320 result_ia)
321{
322 ip_lookup_main_t *lm = &im->lookup_main;
323 ip_interface_address_t *ia;
324 ip6_address_t *result = 0;
325
326 /* *INDENT-OFF* */
327 foreach_ip_interface_address (lm, ia, sw_if_index,
328 1 /* honor unnumbered */,
329 ({
330 ip6_address_t * a = ip_interface_address_get_address (lm, ia);
331 if (ip6_destination_matches_route (im, dst, a, ia->address_length))
332 {
333 result = a;
334 break;
335 }
336 }));
337 /* *INDENT-ON* */
338 if (result_ia)
339 *result_ia = result ? ia : 0;
340 return result;
341}

425/* Compute flow hash. We'll use it to select which Sponge to use for this
426 flow. And other things. */
427always_inline u32
428ip6_compute_flow_hash (const ip6_header_t * ip,
429 flow_hash_config_t flow_hash_config)
430{
431 tcp_header_t *tcp;
432 u64 a, b, c;
433 u64 t1, t2;
434 uword is_tcp_udp = 0;
435 u8 protocol = ip->protocol;
436
437 if (PREDICT_TRUE
438 ((ip->protocol == IP_PROTOCOL_TCP)
439 || (ip->protocol == IP_PROTOCOL_UDP)))
440 {
441 is_tcp_udp = 1;
442 tcp = (void *) (ip + 1);
443 }
444 else if (ip->protocol == IP_PROTOCOL_IP6_HOP_BY_HOP_OPTIONS)
445 {
446 ip6_hop_by_hop_header_t *hbh = (ip6_hop_by_hop_header_t *) (ip + 1);
447 if ((hbh->protocol == IP_PROTOCOL_TCP) ||
448 (hbh->protocol == IP_PROTOCOL_UDP))
449 {
450 is_tcp_udp = 1;
451 tcp = (tcp_header_t *) ((u8 *) hbh + ((hbh->length + 1) << 3));
452 }
453 protocol = hbh->protocol;
454 }
455
456 t1 = (ip->src_address.as_u64[0] ^ ip->src_address.as_u64[1]);
457 t1 = (flow_hash_config & IP_FLOW_HASH_SRC_ADDR) ? t1 : 0;
458
459 t2 = (ip->dst_address.as_u64[0] ^ ip->dst_address.as_u64[1]);
460 t2 = (flow_hash_config & IP_FLOW_HASH_DST_ADDR) ? t2 : 0;
461
462 a = (flow_hash_config & IP_FLOW_HASH_REVERSE_SRC_DST) ? t2 : t1;
463 b = (flow_hash_config & IP_FLOW_HASH_REVERSE_SRC_DST) ? t1 : t2;
464 b ^= (flow_hash_config & IP_FLOW_HASH_PROTO) ? protocol : 0;
465
466 t1 = is_tcp_udp ? tcp->src : 0;
467 t2 = is_tcp_udp ? tcp->dst : 0;
468
469 t1 = (flow_hash_config & IP_FLOW_HASH_SRC_PORT) ? t1 : 0;
470 t2 = (flow_hash_config & IP_FLOW_HASH_DST_PORT) ? t2 : 0;
471
472 c = (flow_hash_config & IP_FLOW_HASH_REVERSE_SRC_DST) ?
473 ((t1 << 16) | t2) : ((t2 << 16) | t1);
474
475 hash_mix64 (a, b, c);
476 return (u32) c;
477}

479/* ip6_locate_header
480 *
481 * This function is to search for the header specified by the protocol number
482 * in find_hdr_type.
483 * This is used to locate a specific IPv6 extension header
484 * or to find transport layer header.
485 * 1. If the find_hdr_type < 0 then it finds and returns the protocol number and
486 * offset stored in *offset of the transport or ESP header in the chain if
487 * found.
488 * 2. If a header with find_hdr_type > 0 protocol number is found then the
489 * offset is stored in *offset and protocol number of the header is
490 * returned.
491 * 3. If find_hdr_type is not found or packet is malformed or
492 * it is a non-first fragment -1 is returned.
493 */
494always_inline int
495ip6_locate_header (vlib_buffer_t * p0,
496 ip6_header_t * ip0, int find_hdr_type, u32 * offset)
497{
498 u8 next_proto = ip0->protocol;
499 u8 *next_header;
500 u8 done = 0;
501 u32 cur_offset;
502 u8 *temp_nxthdr = 0;
503 u32 exthdr_len = 0;
504
505 next_header = ip6_next_header (ip0);
506 cur_offset = sizeof (ip6_header_t);
507 while (1)
508 {
509 done = (next_proto == find_hdr_type);
510 if (PREDICT_FALSE
511 (next_header >=
512 (u8 *) vlib_buffer_get_current (p0) + p0->current_length))
513 {
514 //A malicious packet could set an extension header with a too big size
515 return (-1);
516 }
517 if (done)
518 break;
519 if ((!ip6_ext_hdr (next_proto)) || next_proto == IP_PROTOCOL_IP6_NONXT)
520 {
521 if (find_hdr_type < 0)
522 break;
523 return -1;
524 }
525 if (next_proto == IP_PROTOCOL_IPV6_FRAGMENTATION)
526 {
527 ip6_frag_hdr_t *frag_hdr = (ip6_frag_hdr_t *) next_header;
528 u16 frag_off = ip6_frag_hdr_offset (frag_hdr);
529 /* Non first fragment return -1 */
530 if (frag_off)
531 return (-1);
532 exthdr_len = sizeof (ip6_frag_hdr_t);
533 temp_nxthdr = next_header + exthdr_len;
534 }
535 else if (next_proto == IP_PROTOCOL_IPSEC_AH)
536 {
537 exthdr_len =
538 ip6_ext_authhdr_len (((ip6_ext_header_t *) next_header));
539 temp_nxthdr = next_header + exthdr_len;
540 }
541 else
542 {
543 exthdr_len =
544 ip6_ext_header_len (((ip6_ext_header_t *) next_header));
545 temp_nxthdr = next_header + exthdr_len;
546 }
547 next_proto = ((ip6_ext_header_t *) next_header)->next_hdr;
548 next_header = temp_nxthdr;
549 cur_offset += exthdr_len;
550 }
551
552 *offset = cur_offset;
553 return (next_proto);
554}

580/**
581 * Push IPv6 header to buffer
582 *
583 * @param vm - vlib_main
584 * @param b - buffer to write the header to
585 * @param src - source IP
586 * @param dst - destination IP
587 * @param prot - payload proto
588 *
589 * @return - pointer to start of IP header
590 */
591always_inline void *
592vlib_buffer_push_ip6 (vlib_main_t * vm, vlib_buffer_t * b,
593 ip6_address_t * src, ip6_address_t * dst, int proto)
594{
595 ip6_header_t *ip6h;
596 u16 payload_length;
597
598 /* make some room */
599 ip6h = vlib_buffer_push_uninit (b, sizeof (ip6_header_t));
600
601 ip6h->ip_version_traffic_class_and_flow_label =
602 clib_host_to_net_u32 (0x6 << 28);
603
604 /* calculate ip6 payload length */
605 payload_length = vlib_buffer_length_in_chain (vm, b);
606 payload_length -= sizeof (*ip6h);
607
608 ip6h->payload_length = clib_host_to_net_u16 (payload_length);
609
610 ip6h->hop_limit = 0xff;
611 ip6h->protocol = proto;
612 clib_memcpy (ip6h->src_address.as_u8, src->as_u8,
613 sizeof (ip6h->src_address));
614 clib_memcpy (ip6h->dst_address.as_u8, dst->as_u8,
615 sizeof (ip6h->src_address));
616 b->flags |= VNET_BUFFER_F_IS_IP6;
617
618 return ip6h;
619}
