239typedef CLIB_PACKED (struct
240 {
241 /* Option type. */
242 u8 type;
243 /* Length of this header plus option data in 8 byte units. */
244 u8 n_data_u64s;
245 /* Option data follows. */
246 u8 data[0];
247 }) icmp6_neighbor_discovery_option_header_t;
248
249typedef CLIB_PACKED (struct
250 {
251 icmp6_neighbor_discovery_option_header_t header;
252 u8 dst_address_length;
253 u8 flags;
254#define ICMP6_NEIGHBOR_DISCOVERY_PREFIX_INFORMATION_FLAG_ON_LINK (1 << 7)
255#define ICMP6_NEIGHBOR_DISCOVERY_PREFIX_INFORMATION_AUTO (1 << 6)
256 u32 valid_time;
257 u32 preferred_time;
258 u32 unused; ip6_address_t dst_address;
259 }) icmp6_neighbor_discovery_prefix_information_option_t;

313 }) icmp6_neighbor_discovery_header_t;
314
315/* Router advertisement packet formats. */
316typedef CLIB_PACKED (struct
317 {
318 icmp46_header_t icmp;
319 /* Current hop limit to use for outgoing packets. */
320 u8 current_hop_limit;
321 u8 flags;
322#define ICMP6_ROUTER_DISCOVERY_FLAG_ADDRESS_CONFIG_VIA_DHCP (1 << 7)
323#define ICMP6_ROUTER_DISCOVERY_FLAG_OTHER_CONFIG_VIA_DHCP (1 << 6)
324 /* Zero means unspecified. */
325 u16 router_lifetime_in_sec;
326 /* Zero means unspecified. */
327 u32 neighbor_reachable_time_in_msec;
328 /* Zero means unspecified. */
329 u32
330 time_in_msec_between_retransmitted_neighbor_solicitations;
331 /* Options that may follow: source_link_layer_address, mtu, prefix_information. */
332 }) icmp6_router_advertisement_header_t;
333
334/* Neighbor solicitation/advertisement header. */
335typedef CLIB_PACKED (struct
336 {
337 icmp46_header_t icmp;
338 /* Zero for solicitation; flags for advertisement. */
339 u32 advertisement_flags;
340 /* Set when sent by a router. */
341#define ICMP6_NEIGHBOR_ADVERTISEMENT_FLAG_ROUTER (1 << 31)
342 /* Set when response to solicitation. */
343#define ICMP6_NEIGHBOR_ADVERTISEMENT_FLAG_SOLICITED (1 << 30)
344#define ICMP6_NEIGHBOR_ADVERTISEMENT_FLAG_OVERRIDE (1 << 29)
345 ip6_address_t target_address;
346 /* Options that may follow: source_link_layer_address
347 (for solicitation) target_link_layer_address (for advertisement). */
348 }) icmp6_neighbor_solicitation_or_advertisement_header_t;
349
350typedef CLIB_PACKED (struct
351 {
352 icmp46_header_t icmp;
353 u32 reserved_must_be_zero;
354 /* Better next hop to use for given destination. */
355 ip6_address_t better_next_hop_address;
356 ip6_address_t dst_address;
357 /* Options that may follow: target_link_layer_address,
358 redirected_header. */
359 }) icmp6_redirect_header_t;
360

378 }) icmp6_router_solicitation_header_t;
379
380/* router advertisement packet format for ethernet. */
381typedef CLIB_PACKED (struct
382 {
383 ip6_header_t ip;
384 icmp6_router_advertisement_header_t router;
385 icmp6_neighbor_discovery_ethernet_link_layer_address_option_t

387 icmp6_neighbor_discovery_mtu_option_t mtu_option;
388 icmp6_neighbor_discovery_prefix_information_option_t
389 prefix[0];
390 }) icmp6_router_advertisement_packet_t;
391
