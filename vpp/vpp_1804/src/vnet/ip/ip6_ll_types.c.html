18#include <vnet/ip/ip.h>

20u8 *
21format_ip6_ll_prefix (u8 * s, va_list * args)
22{
23 ip6_ll_prefix_t *ilp = va_arg (*args, ip6_ll_prefix_t *);
24 vnet_main_t *vnm = vnet_get_main ();
25
26 s = format (s, "(%U, %U)",
27 format_ip6_address, &ilp->ilp_addr,
28 format_vnet_sw_interface_name,
29 vnm, vnet_get_sw_interface (vnm, ilp->ilp_sw_if_index));
30
31 return (s);
32}
