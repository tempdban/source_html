73static void
74serialize_vec_vnet_sw_hw_interface_state (serialize_main_t * m, va_list * va)
75{
76 vnet_sw_hw_interface_state_t *s =
77 va_arg (*va, vnet_sw_hw_interface_state_t *);

82 serialize_integer (m, s[i].sw_hw_if_index,
83 sizeof (s[i].sw_hw_if_index));
84 serialize_integer (m, s[i].flags, sizeof (s[i].flags));
85 }

125void
126vnet_hw_interface_set_mtu (vnet_main_t * vnm, u32 hw_if_index, u32 mtu)
127{
128 vnet_hw_interface_t *hi = vnet_get_hw_interface (vnm, hw_if_index);
129
130 if (hi->max_packet_bytes != mtu)
131 {
132 u16 l3_pad = hi->max_packet_bytes - hi->max_l3_packet_bytes[VLIB_TX];
133 hi->max_packet_bytes = mtu;
134 hi->max_l3_packet_bytes[VLIB_TX] =
135 hi->max_l3_packet_bytes[VLIB_RX] = mtu - l3_pad;
136 ethernet_set_flags (vnm, hw_if_index, ETHERNET_INTERFACE_FLAG_MTU);
137 adj_mtu_update (hw_if_index);
138 }
139}

166void
167serialize_vnet_interface_state (serialize_main_t * m, va_list * va)
168{
169 vnet_main_t *vnm = va_arg (*va, vnet_main_t *);
170 vnet_sw_hw_interface_state_t *sts = 0, *st;
171 vnet_sw_interface_t *sif;
172 vnet_hw_interface_t *hif;
173 vnet_interface_main_t *im = &vnm->interface_main;
174
175 /* Serialize hardware interface classes since they may have changed.
176 Must do this before sending up/down flags. */
177 /* *INDENT-OFF* */
178 pool_foreach (hif, im->hw_interfaces, ({
179 vnet_hw_interface_class_t * hw_class = vnet_get_hw_interface_class (vnm, hif->hw_class_index);
180 serialize_cstring (m, hw_class->name);
181 }));
182 /* *INDENT-ON* */
183
184 /* Send sw/hw interface state when non-zero. */
185 /* *INDENT-OFF* */
186 pool_foreach (sif, im->sw_interfaces, ({
187 if (sif->flags != 0)
188 {
189 vec_add2 (sts, st, 1);
190 st->sw_hw_if_index = sif->sw_if_index;
191 st->flags = sif->flags;
192 }
193 }));
194 /* *INDENT-ON* */
195
196 vec_serialize (m, sts, serialize_vec_vnet_sw_hw_interface_state);
197
198 if (sts)
199 _vec_len (sts) = 0;
200
201 /* *INDENT-OFF* */
202 pool_foreach (hif, im->hw_interfaces, ({
203 if (hif->flags != 0)
204 {
205 vec_add2 (sts, st, 1);
206 st->sw_hw_if_index = hif->hw_if_index;
207 st->flags = hif->flags;
208 }
209 }));
210 /* *INDENT-ON* */
211
212 vec_serialize (m, sts, serialize_vec_vnet_sw_hw_interface_state);
213
214 vec_free (sts);
215}

217void
218unserialize_vnet_interface_state (serialize_main_t * m, va_list * va)
219{
220 vnet_main_t *vnm = va_arg (*va, vnet_main_t *);
221 vnet_sw_hw_interface_state_t *sts = 0, *st;
222
223 /* First set interface hardware class. */
224 {
225 vnet_interface_main_t *im = &vnm->interface_main;
226 vnet_hw_interface_t *hif;
227 char *class_name;
228 uword *p;
229 clib_error_t *error;
230
231 /* *INDENT-OFF* */
232 pool_foreach (hif, im->hw_interfaces, ({
233 unserialize_cstring (m, &class_name);
234 p = hash_get_mem (im->hw_interface_class_by_name, class_name);
235 ASSERT (p != 0);
236 error = vnet_hw_interface_set_class_helper (vnm, hif->hw_if_index, p[0], /* redistribute */ 0);
237 if (error)
238 clib_error_report (error);
239 vec_free (class_name);
240 }));
241 /* *INDENT-ON* */
242 }
243
244 vec_unserialize (m, &sts, unserialize_vec_vnet_sw_hw_interface_state);
245 vec_foreach (st, sts)
246 vnet_sw_interface_set_flags_helper (vnm, st->sw_hw_if_index, st->flags,
247 /* no distribute */ 0);
248 vec_free (sts);
249
250 vec_unserialize (m, &sts, unserialize_vec_vnet_sw_hw_interface_state);
251 vec_foreach (st, sts)
252 vnet_hw_interface_set_flags_helper (vnm, st->sw_hw_if_index, st->flags,
253 /* no distribute */ 0);
254 vec_free (sts);
255}

257static clib_error_t *
258call_elf_section_interface_callbacks (vnet_main_t * vnm, u32 if_index,
259 u32 flags,
260 _vnet_interface_function_list_elt_t **
261 elts)
262{
263 _vnet_interface_function_list_elt_t *elt;
264 vnet_interface_function_priority_t prio;
265 clib_error_t *error = 0;
266
267 for (prio = VNET_ITF_FUNC_PRIORITY_LOW;
268 prio <= VNET_ITF_FUNC_PRIORITY_HIGH; prio++)
269 {
270 elt = elts[prio];
271
272 while (elt)
273 {
274 error = elt->fp (vnm, if_index, flags);
275 if (error)
276 return error;
277 elt = elt->next_interface_function;
278 }
279 }
280 return error;
281}

283static clib_error_t *
284call_hw_interface_add_del_callbacks (vnet_main_t * vnm, u32 hw_if_index,
285 u32 is_create)
286{
287 vnet_hw_interface_t *hi = vnet_get_hw_interface (vnm, hw_if_index);
288 vnet_hw_interface_class_t *hw_class =
289 vnet_get_hw_interface_class (vnm, hi->hw_class_index);
290 vnet_device_class_t *dev_class =
291 vnet_get_device_class (vnm, hi->dev_class_index);
292 clib_error_t *error = 0;
293
294 if (hw_class->interface_add_del_function
295 && (error =
296 hw_class->interface_add_del_function (vnm, hw_if_index, is_create)))
297 return error;
298
299 if (dev_class->interface_add_del_function
300 && (error =
301 dev_class->interface_add_del_function (vnm, hw_if_index,
302 is_create)))
303 return error;
304
305 error = call_elf_section_interface_callbacks
306 (vnm, hw_if_index, is_create, vnm->hw_interface_add_del_functions);
307
308 return error;
309}

319#define VNET_INTERFACE_SET_FLAGS_HELPER_IS_CREATE (1 << 0)
320#define VNET_INTERFACE_SET_FLAGS_HELPER_WANT_REDISTRIBUTE (1 << 1)
321
322static clib_error_t *
323vnet_hw_interface_set_flags_helper (vnet_main_t * vnm, u32 hw_if_index,
324 u32 flags, u32 helper_flags)
325{
326 vnet_hw_interface_t *hi = vnet_get_hw_interface (vnm, hw_if_index);
327 vnet_hw_interface_class_t *hw_class =
328 vnet_get_hw_interface_class (vnm, hi->hw_class_index);
329 vnet_device_class_t *dev_class =
330 vnet_get_device_class (vnm, hi->dev_class_index);
331 vlib_main_t *vm = vnm->vlib_main;
332 u32 mask;
333 clib_error_t *error = 0;
334 u32 is_create =
335 (helper_flags & VNET_INTERFACE_SET_FLAGS_HELPER_IS_CREATE) != 0;
336
337 mask =
338 (VNET_HW_INTERFACE_FLAG_LINK_UP | VNET_HW_INTERFACE_FLAG_DUPLEX_MASK |
339 VNET_HW_INTERFACE_FLAG_SPEED_MASK);
340 flags &= mask;
341
342 /* Call hardware interface add/del callbacks. */
343 if (is_create)
344 call_hw_interface_add_del_callbacks (vnm, hw_if_index, is_create);
345
346 /* Already in the desired state? */
347 if (!is_create && (hi->flags & mask) == flags)
348 goto done;
349
350 /* Some interface classes do not redistribute (e.g. are local). */
351 if (!dev_class->redistribute)
352 helper_flags &= ~VNET_INTERFACE_SET_FLAGS_HELPER_WANT_REDISTRIBUTE;
353
354 if (vm->mc_main
355 && (helper_flags & VNET_INTERFACE_SET_FLAGS_HELPER_WANT_REDISTRIBUTE))
356 {
357 vnet_sw_hw_interface_state_t s;
358 s.sw_hw_if_index = hw_if_index;
359 s.flags = flags;
360 mc_serialize (vm->mc_main, &vnet_hw_interface_set_flags_msg, &s);
361 }
362
363 if ((hi->flags & VNET_HW_INTERFACE_FLAG_LINK_UP) !=
364 (flags & VNET_HW_INTERFACE_FLAG_LINK_UP))
365 {
366 /* Do hardware class (e.g. ethernet). */
367 if (hw_class->link_up_down_function
368 && (error = hw_class->link_up_down_function (vnm, hw_if_index,
369 flags)))
370 goto done;
371
372 error = call_elf_section_interface_callbacks
373 (vnm, hw_if_index, flags, vnm->hw_interface_link_up_down_functions);
374
375 if (error)
376 goto done;
377 }
378
379 hi->flags &= ~mask;
380 hi->flags |= flags;
381
382done:
383 return error;
384}

386static clib_error_t *
387vnet_sw_interface_set_flags_helper (vnet_main_t * vnm, u32 sw_if_index,
388 u32 flags, u32 helper_flags)
389{
390 vnet_sw_interface_t *si = vnet_get_sw_interface (vnm, sw_if_index);
391 vlib_main_t *vm = vnm->vlib_main;
392 u32 mask;
393 clib_error_t *error = 0;
394 u32 is_create =
395 (helper_flags & VNET_INTERFACE_SET_FLAGS_HELPER_IS_CREATE) != 0;
396 u32 old_flags;
397
398 mask = VNET_SW_INTERFACE_FLAG_ADMIN_UP | VNET_SW_INTERFACE_FLAG_PUNT;
399 flags &= mask;
400
401 if (is_create)
402 {
403 error =
404 call_sw_interface_add_del_callbacks (vnm, sw_if_index, is_create);
405 if (error)
406 goto done;
407
408 if (flags & VNET_SW_INTERFACE_FLAG_ADMIN_UP)
409 {
410 /* Notify everyone when the interface is created as admin up */
411 error = call_elf_section_interface_callbacks (vnm, sw_if_index,
412 flags,
413 vnm->
414 sw_interface_admin_up_down_functions);
415 if (error)
416 goto done;
417 }
418 }
419 else
420 {
421 vnet_sw_interface_t *si_sup = si;
422
423 /* Check that super interface is in correct state. */
424 if (si->type == VNET_SW_INTERFACE_TYPE_SUB)
425 {
426 si_sup = vnet_get_sw_interface (vnm, si->sup_sw_if_index);
427
428 /* Check to see if we're bringing down the soft interface and if it's parent is up */
429 if ((flags != (si_sup->flags & mask)) &&
430 (!((flags == 0)
431 && ((si_sup->flags & mask) ==
432 VNET_SW_INTERFACE_FLAG_ADMIN_UP))))
433 {
434 error = clib_error_return (0, "super-interface %U must be %U",
435 format_vnet_sw_interface_name, vnm,
436 si_sup,
437 format_vnet_sw_interface_flags,
438 flags);
439 goto done;
440 }
441 }
442
443 /* Donot change state for slave link of bonded interfaces */
444 if (si->flags & VNET_SW_INTERFACE_FLAG_BOND_SLAVE)
445 {
446 error = clib_error_return
447 (0, "not allowed as %U belong to a BondEthernet interface",
448 format_vnet_sw_interface_name, vnm, si);
449 goto done;
450 }
451
452 /* Already in the desired state? */
453 if ((si->flags & mask) == flags)
454 goto done;
455
456 /* Sub-interfaces of hardware interfaces that do no redistribute,
457 do not redistribute themselves. */
458 if (si_sup->type == VNET_SW_INTERFACE_TYPE_HARDWARE)
459 {
460 vnet_hw_interface_t *hi =
461 vnet_get_hw_interface (vnm, si_sup->hw_if_index);
462 vnet_device_class_t *dev_class =
463 vnet_get_device_class (vnm, hi->dev_class_index);
464 if (!dev_class->redistribute)
465 helper_flags &=
466 ~VNET_INTERFACE_SET_FLAGS_HELPER_WANT_REDISTRIBUTE;
467 }
468
469 if (vm->mc_main
470 && (helper_flags &
471 VNET_INTERFACE_SET_FLAGS_HELPER_WANT_REDISTRIBUTE))
472 {
473 vnet_sw_hw_interface_state_t s;
474 s.sw_hw_if_index = sw_if_index;
475 s.flags = flags;
476 mc_serialize (vm->mc_main, &vnet_sw_interface_set_flags_msg, &s);
477 }
478
479 /* set the flags now before invoking the registered clients
480 * so that the state they query is consistent with the state here notified */
481 old_flags = si->flags;
482 si->flags &= ~mask;
483 si->flags |= flags;
484 if ((flags | old_flags) & VNET_SW_INTERFACE_FLAG_ADMIN_UP)
485 error = call_elf_section_interface_callbacks
486 (vnm, sw_if_index, flags,
487 vnm->sw_interface_admin_up_down_functions);
488 si->flags = old_flags;
489
490 if (error)
491 goto done;
492
493 if (si->type == VNET_SW_INTERFACE_TYPE_HARDWARE)
494 {
495 vnet_hw_interface_t *hi =
496 vnet_get_hw_interface (vnm, si->hw_if_index);
497 vnet_hw_interface_class_t *hw_class =
498 vnet_get_hw_interface_class (vnm, hi->hw_class_index);
499 vnet_device_class_t *dev_class =
500 vnet_get_device_class (vnm, hi->dev_class_index);
501
502 if ((flags & VNET_SW_INTERFACE_FLAG_ADMIN_UP) &&
503 (si->flags & VNET_SW_INTERFACE_FLAG_ERROR))
504 {
505 error = clib_error_return (0, "Interface in the error state");
506 goto done;
507 }
508
509 /* save the si admin up flag */
510 old_flags = si->flags;
511
512 /* update si admin up flag in advance if we are going admin down */
513 if (!(flags & VNET_SW_INTERFACE_FLAG_ADMIN_UP))
514 si->flags &= ~VNET_SW_INTERFACE_FLAG_ADMIN_UP;
515
516 if (dev_class->admin_up_down_function
517 && (error = dev_class->admin_up_down_function (vnm,
518 si->hw_if_index,
519 flags)))
520 {
521 /* restore si admin up flag to it's original state on errors */
522 si->flags = old_flags;
523 goto done;
524 }
525
526 if (hw_class->admin_up_down_function
527 && (error = hw_class->admin_up_down_function (vnm,
528 si->hw_if_index,
529 flags)))
530 {
531 /* restore si admin up flag to it's original state on errors */
532 si->flags = old_flags;
533 goto done;
534 }
535
536 /* Admin down implies link down. */
537 if (!(flags & VNET_SW_INTERFACE_FLAG_ADMIN_UP)
538 && (hi->flags & VNET_HW_INTERFACE_FLAG_LINK_UP))
539 vnet_hw_interface_set_flags_helper (vnm, si->hw_if_index,
540 hi->flags &
541 ~VNET_HW_INTERFACE_FLAG_LINK_UP,
542 helper_flags);
543 }
544 }
545
546 si->flags &= ~mask;
547 si->flags |= flags;
548
549done:
550 return error;
551}

569static u32
570vnet_create_sw_interface_no_callbacks (vnet_main_t * vnm,
571 vnet_sw_interface_t * template)
572{
573 vnet_interface_main_t *im = &vnm->interface_main;
574 vnet_sw_interface_t *sw;
575 u32 sw_if_index;
576
577 pool_get (im->sw_interfaces, sw);
578 sw_if_index = sw - im->sw_interfaces;
579
580 sw[0] = template[0];
581
582 sw->flags = 0;
583 sw->sw_if_index = sw_if_index;
584 if (sw->type == VNET_SW_INTERFACE_TYPE_HARDWARE)
585 sw->sup_sw_if_index = sw->sw_if_index;
586
587 /* Allocate counters for this interface. */
588 {
589 u32 i;
590
591 vnet_interface_counter_lock (im);
592
593 for (i = 0; i < vec_len (im->sw_if_counters); i++)
594 {
595 vlib_validate_simple_counter (&im->sw_if_counters[i], sw_if_index);
596 vlib_zero_simple_counter (&im->sw_if_counters[i], sw_if_index);
597 }
598
599 for (i = 0; i < vec_len (im->combined_sw_if_counters); i++)
600 {
601 vlib_validate_combined_counter (&im->combined_sw_if_counters[i],
602 sw_if_index);
603 vlib_zero_combined_counter (&im->combined_sw_if_counters[i],
604 sw_if_index);
605 }
606
607 vnet_interface_counter_unlock (im);
608 }
609
610 return sw_if_index;
611}

613clib_error_t *
614vnet_create_sw_interface (vnet_main_t * vnm, vnet_sw_interface_t * template,
615 u32 * sw_if_index)
616{
617 clib_error_t *error;
618 vnet_hw_interface_t *hi;
619 vnet_device_class_t *dev_class;
620
621 hi = vnet_get_sup_hw_interface (vnm, template->sup_sw_if_index);
622 dev_class = vnet_get_device_class (vnm, hi->dev_class_index);
623
624 if (template->type == VNET_SW_INTERFACE_TYPE_SUB &&
625 dev_class->subif_add_del_function)
626 {
627 error = dev_class->subif_add_del_function (vnm, hi->hw_if_index,
628 (struct vnet_sw_interface_t
629 *) template, 1);
630 if (error)
631 return error;
632 }
633
634 *sw_if_index = vnet_create_sw_interface_no_callbacks (vnm, template);
635 error = vnet_sw_interface_set_flags_helper
636 (vnm, *sw_if_index, template->flags,
637 VNET_INTERFACE_SET_FLAGS_HELPER_IS_CREATE);
638
639 if (error)
640 {

646 }
647
648 return error;

703/* Register an interface instance. */
704u32
705vnet_register_interface (vnet_main_t * vnm,
706 u32 dev_class_index,
707 u32 dev_instance,
708 u32 hw_class_index, u32 hw_instance)
709{
710 vnet_interface_main_t *im = &vnm->interface_main;
711 vnet_hw_interface_t *hw;
712 vnet_device_class_t *dev_class =
713 vnet_get_device_class (vnm, dev_class_index);
714 vnet_hw_interface_class_t *hw_class =
715 vnet_get_hw_interface_class (vnm, hw_class_index);
716 vlib_main_t *vm = vnm->vlib_main;
717 vnet_feature_config_main_t *fcm;
718 vnet_config_main_t *cm;
719 u32 hw_index, i;
720 char *tx_node_name = NULL, *output_node_name = NULL;
721
722 pool_get (im->hw_interfaces, hw);
723 memset (hw, 0, sizeof (*hw));
724
725 hw_index = hw - im->hw_interfaces;
726 hw->hw_if_index = hw_index;
727 hw->default_rx_mode = VNET_HW_INTERFACE_RX_MODE_POLLING;
728
729 if (dev_class->format_device_name)
730 hw->name = format (0, "%U", dev_class->format_device_name, dev_instance);
731 else if (hw_class->format_interface_name)
732 hw->name = format (0, "%U", hw_class->format_interface_name,
733 dev_instance);
734 else
735 hw->name = format (0, "%s%x", hw_class->name, dev_instance);
736
737 if (!im->hw_interface_by_name)
738 im->hw_interface_by_name = hash_create_vec ( /* size */ 0,
739 sizeof (hw->name[0]),
740 sizeof (uword));
741
742 hash_set_mem (im->hw_interface_by_name, hw->name, hw_index);
743
744 /* Make hardware interface point to software interface. */
745 {
746 vnet_sw_interface_t sw = {
747 .type = VNET_SW_INTERFACE_TYPE_HARDWARE,
748 .flood_class = VNET_FLOOD_CLASS_NORMAL,
749 .hw_if_index = hw_index
750 };
751 hw->sw_if_index = vnet_create_sw_interface_no_callbacks (vnm, &sw);
752 }
753
754 hw->dev_class_index = dev_class_index;
755 hw->dev_instance = dev_instance;
756 hw->hw_class_index = hw_class_index;
757 hw->hw_instance = hw_instance;
758
759 hw->max_rate_bits_per_sec = 0;
760 hw->min_packet_bytes = 0;
761 hw->per_packet_overhead_bytes = 0;
762 hw->max_l3_packet_bytes[VLIB_RX] = ~0;
763 hw->max_l3_packet_bytes[VLIB_TX] = ~0;
764
765 if (dev_class->tx_function == 0)
766 goto no_output_nodes; /* No output/tx nodes to create */
767
768 tx_node_name = (char *) format (0, "%v-tx", hw->name);
769 output_node_name = (char *) format (0, "%v-output", hw->name);
770
771 /* If we have previously deleted interface nodes, re-use them. */
772 if (vec_len (im->deleted_hw_interface_nodes) > 0)
773 {
774 vnet_hw_interface_nodes_t *hn;
775 vlib_node_t *node;
776 vlib_node_runtime_t *nrt;
777
778 hn = vec_end (im->deleted_hw_interface_nodes) - 1;
779
780 hw->tx_node_index = hn->tx_node_index;
781 hw->output_node_index = hn->output_node_index;
782
783 vlib_node_rename (vm, hw->tx_node_index, "%v", tx_node_name);
784 vlib_node_rename (vm, hw->output_node_index, "%v", output_node_name);
785
786 /* *INDENT-OFF* */
787 foreach_vlib_main ({
788 vnet_interface_output_runtime_t *rt;
789
790 rt = vlib_node_get_runtime_data (this_vlib_main, hw->output_node_index);
791 ASSERT (rt->is_deleted == 1);
792 rt->is_deleted = 0;
793 rt->hw_if_index = hw_index;
794 rt->sw_if_index = hw->sw_if_index;
795 rt->dev_instance = hw->dev_instance;
796
797 rt = vlib_node_get_runtime_data (this_vlib_main, hw->tx_node_index);
798 rt->hw_if_index = hw_index;
799 rt->sw_if_index = hw->sw_if_index;
800 rt->dev_instance = hw->dev_instance;
801 });
802 /* *INDENT-ON* */
803
804 /* The new class may differ from the old one.
805 * Functions have to be updated. */
806 node = vlib_get_node (vm, hw->output_node_index);
807 node->function = vnet_interface_output_node_multiarch_select ();
808 node->format_trace = format_vnet_interface_output_trace;
809 /* *INDENT-OFF* */
810 foreach_vlib_main ({
811 nrt = vlib_node_get_runtime (this_vlib_main, hw->output_node_index);
812 nrt->function = node->function;
813 });
814 /* *INDENT-ON* */
815
816 node = vlib_get_node (vm, hw->tx_node_index);
817 node->function = dev_class->tx_function;
818 node->format_trace = dev_class->format_tx_trace;
819 /* *INDENT-OFF* */
820 foreach_vlib_main ({
821 nrt = vlib_node_get_runtime (this_vlib_main, hw->tx_node_index);
822 nrt->function = node->function;
823 });
824 /* *INDENT-ON* */
825
826 _vec_len (im->deleted_hw_interface_nodes) -= 1;
827 }
828 else
829 {
830 vlib_node_registration_t r;
831 vnet_interface_output_runtime_t rt = {
832 .hw_if_index = hw_index,
833 .sw_if_index = hw->sw_if_index,
834 .dev_instance = hw->dev_instance,
835 .is_deleted = 0,
836 };
837
838 memset (&r, 0, sizeof (r));
839 r.type = VLIB_NODE_TYPE_INTERNAL;
840 r.runtime_data = &rt;
841 r.runtime_data_bytes = sizeof (rt);
842 r.scalar_size = 0;
843 r.vector_size = sizeof (u32);
844
845 r.flags = VLIB_NODE_FLAG_IS_OUTPUT;
846 r.name = tx_node_name;
847 r.function = dev_class->tx_function;
848
849 hw->tx_node_index = vlib_register_node (vm, &r);
850
851 vlib_node_add_named_next_with_slot (vm, hw->tx_node_index,
852 "error-drop",
853 VNET_INTERFACE_TX_NEXT_DROP);
854
855 r.flags = 0;
856 r.name = output_node_name;
857 r.function = vnet_interface_output_node_multiarch_select ();
858 r.format_trace = format_vnet_interface_output_trace;
859
860 {
861 static char *e[] = {
862 "interface is down",
863 "interface is deleted",
864 };
865
866 r.n_errors = ARRAY_LEN (e);
867 r.error_strings = e;
868 }
869 hw->output_node_index = vlib_register_node (vm, &r);
870
871 vlib_node_add_named_next_with_slot (vm, hw->output_node_index,
872 "error-drop",
873 VNET_INTERFACE_OUTPUT_NEXT_DROP);
874 vlib_node_add_next_with_slot (vm, hw->output_node_index,
875 hw->tx_node_index,
876 VNET_INTERFACE_OUTPUT_NEXT_TX);
877
878 /* add interface to the list of "output-interface" feature arc start nodes
879 and clone nexts from 1st interface if it exists */
880 fcm = vnet_feature_get_config_main (im->output_feature_arc_index);
881 cm = &fcm->config_main;
882 i = vec_len (cm->start_node_indices);
883 vec_validate (cm->start_node_indices, i);
884 cm->start_node_indices[i] = hw->output_node_index;
885 if (hw_index)
886 {
887 /* copy nexts from 1st interface */
888 vnet_hw_interface_t *first_hw;
889 vlib_node_t *first_node;
890
891 first_hw = vnet_get_hw_interface (vnm, /* hw_if_index */ 0);
892 first_node = vlib_get_node (vm, first_hw->output_node_index);
893
894 /* 1st 2 nexts are already added above */
895 for (i = 2; i < vec_len (first_node->next_nodes); i++)
896 vlib_node_add_next_with_slot (vm, hw->output_node_index,
897 first_node->next_nodes[i], i);
898 }
899 }
900
901 setup_output_node (vm, hw->output_node_index, hw_class);
902 setup_tx_node (vm, hw->tx_node_index, dev_class);
903
904no_output_nodes:
905 /* Call all up/down callbacks with zero flags when interface is created. */
906 vnet_sw_interface_set_flags_helper (vnm, hw->sw_if_index, /* flags */ 0,
907 VNET_INTERFACE_SET_FLAGS_HELPER_IS_CREATE);
908 vnet_hw_interface_set_flags_helper (vnm, hw_index, /* flags */ 0,
909 VNET_INTERFACE_SET_FLAGS_HELPER_IS_CREATE);
910 vec_free (tx_node_name);
911 vec_free (output_node_name);
912
913 return hw_index;
914}

916void
917vnet_delete_hw_interface (vnet_main_t * vnm, u32 hw_if_index)
918{
919 vnet_interface_main_t *im = &vnm->interface_main;
920 vnet_hw_interface_t *hw = vnet_get_hw_interface (vnm, hw_if_index);
921 vlib_main_t *vm = vnm->vlib_main;
922 vnet_device_class_t *dev_class = vnet_get_device_class (vnm,
923 hw->dev_class_index);
924 /* If it is up, mark it down. */
925 if (hw->flags != 0)
926 vnet_hw_interface_set_flags (vnm, hw_if_index, /* flags */ 0);
927
928 /* Call delete callbacks. */
929 call_hw_interface_add_del_callbacks (vnm, hw_if_index, /* is_create */ 0);
930
931 /* Delete any sub-interfaces. */
932 {
933 u32 id, sw_if_index;
934 /* *INDENT-OFF* */
935 hash_foreach (id, sw_if_index, hw->sub_interface_sw_if_index_by_id,
936 ({
937 vnet_sw_interface_t *si = vnet_get_sw_interface (vnm, sw_if_index);
938 u64 sup_and_sub_key =
939 ((u64) (si->sup_sw_if_index) << 32) | (u64) si->sub.id;
940 hash_unset_mem_free (&im->sw_if_index_by_sup_and_sub, &sup_and_sub_key);
941 vnet_delete_sw_interface (vnm, sw_if_index);
942 }));
943 hash_free (hw->sub_interface_sw_if_index_by_id);
944 /* *INDENT-ON* */
945 }
946
947 /* Delete software interface corresponding to hardware interface. */
948 vnet_delete_sw_interface (vnm, hw->sw_if_index);
949
950 if (dev_class->tx_function)
951 {
952 /* Put output/tx nodes into recycle pool */
953 vnet_hw_interface_nodes_t *dn;
954
955 /* *INDENT-OFF* */
956 foreach_vlib_main
957 ({
958 vnet_interface_output_runtime_t *rt =
959 vlib_node_get_runtime_data (this_vlib_main, hw->output_node_index);
960
961 /* Mark node runtime as deleted so output node (if called)
962 * will drop packets. */
963 rt->is_deleted = 1;
964 });
965 /* *INDENT-ON* */
966
967 vlib_node_rename (vm, hw->output_node_index,
968 "interface-%d-output-deleted", hw_if_index);
969 vlib_node_rename (vm, hw->tx_node_index, "interface-%d-tx-deleted",
970 hw_if_index);
971 vec_add2 (im->deleted_hw_interface_nodes, dn, 1);
972 dn->tx_node_index = hw->tx_node_index;
973 dn->output_node_index = hw->output_node_index;
974 }
975
976 hash_unset_mem (im->hw_interface_by_name, hw->name);
977 vec_free (hw->name);
978 vec_free (hw->hw_address);
979 vec_free (hw->input_node_thread_index_by_queue);
980 vec_free (hw->dq_runtime_index_by_queue);
981
982 pool_put (im->hw_interfaces, hw);
983}

1055static clib_error_t *
1056vnet_hw_interface_set_class_helper (vnet_main_t * vnm, u32 hw_if_index,
1057 u32 hw_class_index, u32 redistribute)
1058{
1059 vnet_hw_interface_t *hi = vnet_get_hw_interface (vnm, hw_if_index);
1060 vnet_sw_interface_t *si = vnet_get_sw_interface (vnm, hi->sw_if_index);
1061 vnet_hw_interface_class_t *old_class =
1062 vnet_get_hw_interface_class (vnm, hi->hw_class_index);
1063 vnet_hw_interface_class_t *new_class =
1064 vnet_get_hw_interface_class (vnm, hw_class_index);
1065 vnet_device_class_t *dev_class =
1066 vnet_get_device_class (vnm, hi->dev_class_index);
1067 clib_error_t *error = 0;
1068
1069 /* New class equals old class? Nothing to do. */
1070 if (hi->hw_class_index == hw_class_index)
1071 return 0;
1072
1073 /* No need (and incorrect since admin up flag may be set) to do error checking when
1074 receiving unserialize message. */
1075 if (redistribute)
1076 {
1077 if (si->flags & VNET_SW_INTERFACE_FLAG_ADMIN_UP)
1078 return clib_error_return (0,
1079 "%v must be admin down to change class from %s to %s",
1080 hi->name, old_class->name, new_class->name);
1081
1082 /* Make sure interface supports given class. */
1083 if ((new_class->is_valid_class_for_interface
1084 && !new_class->is_valid_class_for_interface (vnm, hw_if_index,
1085 hw_class_index))
1086 || (dev_class->is_valid_class_for_interface
1087 && !dev_class->is_valid_class_for_interface (vnm, hw_if_index,
1088 hw_class_index)))
1089 return clib_error_return (0,
1090 "%v class cannot be changed from %s to %s",
1091 hi->name, old_class->name, new_class->name);
1092
1093 if (vnm->vlib_main->mc_main)
1094 {
1095 mc_serialize (vnm->vlib_main->mc_main,
1096 &vnet_hw_interface_set_class_msg, hw_if_index,
1097 new_class->name);
1098 return 0;
1099 }
1100 }
1101
1102 if (old_class->hw_class_change)
1103 old_class->hw_class_change (vnm, hw_if_index, old_class->index,
1104 new_class->index);
1105
1106 vnet_hw_interface_init_for_class (vnm, hw_if_index, new_class->index,
1107 /* instance */ ~0);
1108
1109 if (new_class->hw_class_change)
1110 new_class->hw_class_change (vnm, hw_if_index, old_class->index,
1111 new_class->index);
1112
1113 if (dev_class->hw_class_change)
1114 dev_class->hw_class_change (vnm, hw_if_index, new_class->index);
1115
1116 return error;
1117}

1272 }
1273 }

1276 sizeof (uword));
1277
1278 im->sw_if_index_by_sup_and_sub = hash_create_mem (0, sizeof (u64),
1279 sizeof (uword));
1280 {
1281 vnet_hw_interface_class_t *c;
1282

1284
1285 while (c)
1286 {

1299
1300 if ((error = vlib_call_init_function (vm, vnet_interface_cli_init)))

1302
1303 vnm->interface_tag_by_sw_if_index = hash_create (0, sizeof (uword));
1304
1305#if VLIB_BUFFER_TRACE_TRAJECTORY > 0
1306 if ((error = vlib_call_init_function (vm, trajectory_trace_init)))
1307 return error;
1308#endif
1309
1310 return 0;
