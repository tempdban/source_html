20#ifndef __RECEIVE_DPO_H__
21#define __RECEIVE_DPO_H__
22
23#include <vnet/dpo/dpo.h>
24#include <vnet/ip/ip6.h>
25
26typedef struct receive_dpo_t_
27{
28 /**
29 * required for pool_get_aligned.
30 * memebers used in the switch path come first!
31 */
32 CLIB_CACHE_LINE_ALIGN_MARK(cacheline0);
33
34 /**
35 * The Software interface index on which traffic is received
36 */
37 u32 rd_sw_if_index;
38
39 /**
40 * The address on the receive interface. packet are destined to this address
41 */
42 ip46_address_t rd_addr;
43
44 /**
45 * number oflocks.
46 */
47 u16 rd_locks;
48} receive_dpo_t;
49

51 u32 sw_if_index,
52 const ip46_address_t *nh_addr,
53 dpo_id_t *dpo);

55extern void receive_dpo_module_init(void);
56
57/**
58 * @brief pool of all receive DPOs
59 */
60receive_dpo_t *receive_dpo_pool;
61
62static inline receive_dpo_t *
63receive_dpo_get (index_t index)
64{
65 return (pool_elt_at_index(receive_dpo_pool, index));
66}
