38/**
39 * @brief VXLAN GPE header struct
40 *
41 */
42/* *INDENT-OFF* */
43typedef CLIB_PACKED (struct {
44 /** 20 bytes */
45 ip4_header_t ip4;
46 /** 8 bytes */
47 udp_header_t udp;
48 /** 8 bytes */
49 vxlan_gpe_header_t vxlan;
50}) ip4_vxlan_gpe_header_t;
51/* *INDENT-ON* */
52
53/* *INDENT-OFF* */
54typedef CLIB_PACKED (struct {
55 /** 40 bytes */
56 ip6_header_t ip6;
57 /** 8 bytes */
58 udp_header_t udp;
59 /** 8 bytes */
60 vxlan_gpe_header_t vxlan;
61}) ip6_vxlan_gpe_header_t;
62/* *INDENT-ON* */
63
64/**
65 * @brief Key struct for IPv4 VXLAN GPE tunnel.
66 * Key fields: local remote, vni
67 * all fields in NET byte order
68 * VNI shifted 8 bits
69 */
70/* *INDENT-OFF* */
71typedef CLIB_PACKED(struct {
72 union {
73 struct {
74 u32 local;
75 u32 remote;
76
77 u32 vni;
78 u32 pad;
79 };
80 u64 as_u64[2];
81 };
82}) vxlan4_gpe_tunnel_key_t;
83/* *INDENT-ON* */
84
85/**
86 * @brief Key struct for IPv6 VXLAN GPE tunnel.
87 * Key fields: local remote, vni
88 * all fields in NET byte order
89 * VNI shifted 8 bits
90 */
91/* *INDENT-OFF* */
92typedef CLIB_PACKED(struct {
93 ip6_address_t local;
94 ip6_address_t remote;
95 u32 vni;
96}) vxlan6_gpe_tunnel_key_t;
97/* *INDENT-ON* */
98
99/**
100 * @brief Struct for VXLAN GPE tunnel
101 */
102typedef struct
103{
104 /* Required for pool_get_aligned */
105 CLIB_CACHE_LINE_ALIGN_MARK (cacheline0);
106
107 /** Rewrite string. $$$$ embed vnet_rewrite header */
108 u8 *rewrite;
109
110 /** encapsulated protocol */
111 u8 protocol;
112
113 /* FIB DPO for IP forwarding of VXLAN-GPE encap packet */
114 dpo_id_t next_dpo;
115 /** tunnel local address */
116 ip46_address_t local;
117 /** tunnel remote address */
118 ip46_address_t remote;
119
120 /* mcast packet output intfc index (used only if dst is mcast) */
121 u32 mcast_sw_if_index;
122
123 /** FIB indices - tunnel partner lookup here */
124 u32 encap_fib_index;
125 /** FIB indices - inner IP packet lookup here */
126 u32 decap_fib_index;
127
128 /** VXLAN GPE VNI in HOST byte order, shifted left 8 bits */
129 u32 vni;
130
131 /** vnet intfc hw_if_index */
132 u32 hw_if_index;
133 /** vnet intfc sw_if_index */
134 u32 sw_if_index;
135
136 /** flags */
137 u32 flags;
138
139 /** rewrite size for dynamic plugins like iOAM */
140 u8 rewrite_size;
141
142 /** Next node after VxLAN-GPE encap */
143 uword encap_next_node;
144
145 /**
146 * Linkage into the FIB object graph
147 */
148 fib_node_t node;
149
150 /*
151 * The FIB entry for (depending on VXLAN-GPE tunnel is unicast or mcast)
152 * sending unicast VXLAN-GPE encap packets or receiving mcast VXLAN-GPE packets
153 */
154 fib_node_index_t fib_entry_index;
155 adj_index_t mcast_adj_index;
156
157 /**
158 * The tunnel is a child of the FIB entry for its desintion. This is
159 * so it receives updates when the forwarding information for that entry
160 * changes.
161 * The tunnels sibling index on the FIB entry's dependency list.
162 */
163 u32 sibling_index;
164
165} vxlan_gpe_tunnel_t;
166
167/** Flags for vxlan_gpe_tunnel_t */
168#define VXLAN_GPE_TUNNEL_IS_IPV4 1

175_(L2_INPUT, "l2-input")

192 VXLAN_GPE_N_ERROR,

195/** Struct for VXLAN GPE node state */
196typedef struct
197{
198 /** vector of encap tunnel instances */
199 vxlan_gpe_tunnel_t *tunnels;
200
201 /** lookup IPv4 VXLAN GPE tunnel by key */
202 uword *vxlan4_gpe_tunnel_by_key;
203 /** lookup IPv6 VXLAN GPE tunnel by key */
204 uword *vxlan6_gpe_tunnel_by_key;
205
206 /* local VTEP IPs ref count used by vxlan-bypass node to check if
207 received VXLAN packet DIP matches any local VTEP address */
208 uword *vtep4; /* local ip4 VTEPs keyed on their ip4 addr */
209 uword *vtep6; /* local ip6 VTEPs keyed on their ip6 addr */
210 /* mcast shared info */
211 uword *mcast_shared; /* keyed on mcast ip46 addr */
212 /** Free vlib hw_if_indices */
213 u32 *free_vxlan_gpe_tunnel_hw_if_indices;
214
215 /** Mapping from sw_if_index to tunnel index */
216 u32 *tunnel_index_by_sw_if_index;
217
218 /** State convenience vlib_main_t */
219 vlib_main_t *vlib_main;
220 /** State convenience vnet_main_t */
221 vnet_main_t *vnet_main;
222
223 /** List of next nodes for the decap indexed on protocol */
224 uword decap_next_node_list[VXLAN_GPE_PROTOCOL_MAX];
225} vxlan_gpe_main_t;
