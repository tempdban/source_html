61typedef struct
62{
63 u32 domain_id;
64 u32 sequence_number;
65 u16 src_port;
66 u16 n_reports;
67 u16 next_template_no;

70typedef struct flow_report
71{
72 /* ipfix rewrite, set by callback */
73 u8 *rewrite;
74 u16 template_id;
75 u32 stream_index;
76 f64 last_template_sent;
77 int update_rewrite;
78
79 /* Bitmap of fields to send */
80 uword *fields_to_send;
81
82 /* Opaque data */
83 opaque_t opaque;
84
85 /* build-the-rewrite callback */
86 vnet_flow_rewrite_callback_t *rewrite_callback;
87
88 /* Send-flow-data callback */
89 vnet_flow_data_callback_t *flow_data_callback;
90} flow_report_t;
91
92typedef struct flow_report_main
93{
94 flow_report_t *reports;
95 flow_report_stream_t *streams;
96
97 /* ipfix collector ip address, port, our ip address, fib index */
98 ip4_address_t ipfix_collector;
99 u16 collector_port;
100 ip4_address_t src_address;
101 u32 fib_index;
102
103 /* Path MTU */
104 u32 path_mtu;
105
106 /* time interval in seconds after which to resend templates */
107 u32 template_interval;
108
109 /* UDP checksum calculation enable flag */
110 u8 udp_checksum;
111
112 /* time scale transform. Joy. */
113 u32 unix_time_0;
114 f64 vlib_time_0;
115
116 /* convenience variables */
117 vlib_main_t *vlib_main;
118 vnet_main_t *vnet_main;
119} flow_report_main_t;
