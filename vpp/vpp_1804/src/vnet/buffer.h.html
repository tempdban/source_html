48#define foreach_vnet_buffer_flag \
49 _( 1, L4_CHECKSUM_COMPUTED, "l4-cksum-computed") \
50 _( 2, L4_CHECKSUM_CORRECT, "l4-cksum-correct") \
51 _( 3, VLAN_2_DEEP, "vlan-2-deep") \
52 _( 4, VLAN_1_DEEP, "vlan-1-deep") \
53 _( 5, SPAN_CLONE, "span-clone") \
54 _( 6, HANDOFF_NEXT_VALID, "handoff-next-valid") \
55 _( 7, LOCALLY_ORIGINATED, "local") \
56 _( 8, IS_IP4, "ip4") \
57 _( 9, IS_IP6, "ip6") \
58 _(10, OFFLOAD_IP_CKSUM, "offload-ip-cksum") \
59 _(11, OFFLOAD_TCP_CKSUM, "offload-tcp-cksum") \
60 _(12, OFFLOAD_UDP_CKSUM, "offload-udp-cksum") \
61 _(13, IS_NATED, "nated") \
62 _(14, L2_HDR_OFFSET_VALID, 0) \
63 _(15, L3_HDR_OFFSET_VALID, 0) \
64 _(16, L4_HDR_OFFSET_VALID, 0) \
65 _(17, FLOW_REPORT, "flow-report") \
66 _(18, IS_DVR, "dvr") \
67 _(19, QOS_DATA_VALID, 0)

102/*
103 * vnet stack buffer opaque array overlay structure.
104 * The vnet_buffer_opaque_t *must* be the same size as the
105 * vlib_buffer_t "opaque" structure member, 32 bytes.
106 *
107 * When adding a union type, please add a stanza to
108 * foreach_buffer_opaque_union_subtype (directly above).
109 * Code in vnet_interface_init(...) verifies the size
110 * of the union, and will announce any deviations in an
111 * impossible-to-miss manner.
112 */
113typedef struct
114{
115 u32 sw_if_index[VLIB_N_RX_TX];
116 i16 l2_hdr_offset;
117 i16 l3_hdr_offset;
118 i16 l4_hdr_offset;
119 u16 dont_waste_me;
120
121 union
122 {
123 /* IP4/6 buffer opaque. */
124 struct
125 {
126 /* Adjacency from destination IP address lookup [VLIB_TX].
127 Adjacency from source IP address lookup [VLIB_RX].
128 This gets set to ~0 until source lookup is performed. */
129 u32 adj_index[VLIB_N_RX_TX];
130
131 union
132 {
133 struct
134 {
135 /* Flow hash value for this packet computed from IP src/dst address
136 protocol and ports. */
137 u32 flow_hash;
138
139 union
140 {
141 /* next protocol */
142 u32 save_protocol;
143
144 /* Hint for transport protocols */
145 u32 fib_index;
146 };
147
148 /* Rewrite length */
149 u32 save_rewrite_length;
150
151 /* MFIB RPF ID */
152 u32 rpf_id;
153 };
154
155 /* ICMP */
156 struct
157 {
158 u8 type;
159 u8 code;
160 u32 data;
161 } icmp;
162
163 /* reassembly */
164 union
165 {
166 /* in/out variables */
167 struct
168 {
169 u32 next_index; /* index of next node - ignored if "feature" node */
170 u16 estimated_mtu; /* estimated MTU calculated during reassembly */
171 };
172 /* internal variables used during reassembly */
173 struct
174 {
175 u16 fragment_first;
176 u16 fragment_last;
177 u16 range_first;
178 u16 range_last;
179 u32 next_range_bi;
180 u16 ip6_frag_hdr_offset;
181 };
182 } reass;
183 };
184
185 } ip;
186
187 /*
188 * MPLS:
189 * data copied from the MPLS header that was popped from the packet
190 * during the look-up.
191 */
192 struct
193 {
194 /* do not overlay w/ ip.adj_index[0,1] nor flow hash */
195 u32 pad[VLIB_N_RX_TX + 1];
196 u8 ttl;
197 u8 exp;
198 u8 first;
199 /* Rewrite length */
200 u32 save_rewrite_length;
201 /*
202 * BIER - the nubmer of bytes in the header.
203 * the len field inthe header is not authoritative. It's the
204 * value in the table that counts.
205 */
206 struct
207 {
208 u8 n_bytes;
209 } bier;
210 } mpls;
211
212 /* ip4-in-ip6 softwire termination, only valid there */
213 struct
214 {
215 u8 swt_disable;
216 u32 mapping_index;
217 } swt;
218
219 /* l2 bridging path, only valid there */
220 struct opaque_l2
221 {
222 u32 feature_bitmap;
223 u16 bd_index; /* bridge-domain index */
224 u8 l2_len; /* ethernet header length */
225 u8 shg; /* split-horizon group */
226 u16 l2fib_sn; /* l2fib bd/int seq_num */
227 u8 bd_age; /* aging enabled */
228 } l2;
229
230 /* l2tpv3 softwire encap, only valid there */
231 struct
232 {
233 u32 pad[4]; /* do not overlay w/ ip.adj_index[0,1] */
234 u8 next_index;
235 u32 session_index;
236 } l2t;
237
238 struct
239 {
240 u32 src, dst;
241 } gre;
242
243 /* L2 classify */
244 struct
245 {
246 struct opaque_l2 pad;
247 union
248 {
249 u32 table_index;
250 u32 opaque_index;
251 };
252 u64 hash;
253 } l2_classify;
254
255 /* IO - worker thread handoff */
256 struct
257 {
258 u32 next_index;
259 } handoff;
260
261 /* vnet policer */
262 struct
263 {
264 u32 pad[8 - VLIB_N_RX_TX - 1]; /* to end of opaque */
265 u32 index;
266 } policer;
267
268 /* interface output features */
269 struct
270 {
271 u32 flags;
272 u32 sad_index;
273 } ipsec;
274
275 /* MAP */
276 struct
277 {
278 u16 mtu;
279 } map;
280
281 /* MAP-T */
282 struct
283 {
284 u32 map_domain_index;
285 struct
286 {
287 u32 saddr, daddr;
288 u16 frag_offset; //Fragmentation header offset
289 u16 l4_offset; //L4 header overall offset
290 u8 l4_protocol; //The final protocol number
291 } v6; //Used by ip6_map_t only
292 u16 checksum_offset; //L4 checksum overall offset
293 u16 mtu; //Exit MTU
294 } map_t;
295
296 /* IP Fragmentation */
297 struct
298 {
299 u32 pad[2]; /* do not overlay w/ ip.adj_index[0,1] */
300 u16 header_offset;
301 u16 mtu;
302 u8 next_index;
303 u8 flags; //See ip_frag.h
304 } ip_frag;
305
306 /* COP - configurable junk filter(s) */
307 struct
308 {
309 /* Current configuration index. */
310 u32 current_config_index;
311 } cop;
312
313 /* LISP */
314 struct
315 {
316 /* overlay address family */
317 u16 overlay_afi;
318 } lisp;
319
320 /* Driver rx feature */
321 struct
322 {
323 u32 saved_next_index; /**< saved by drivers for short-cut */
324 u16 buffer_advance;
325 } device_input_feat;
326
327 /* TCP */
328 struct
329 {
330 u32 connection_index;
331 u32 seq_number;
332 u32 seq_end;
333 u32 ack_number;
334 u16 hdr_offset; /**< offset relative to ip hdr */
335 u16 data_offset; /**< offset relative to ip hdr */
336 u16 data_len; /**< data len */
337 u8 flags;
338 } tcp;
339
340 /* SCTP */
341 struct
342 {
343 u32 connection_index;
344 u16 sid; /**< Stream ID */
345 u16 ssn; /**< Stream Sequence Number */
346 u32 tsn; /**< Transmission Sequence Number */
347 u16 hdr_offset; /**< offset relative to ip hdr */
348 u16 data_offset; /**< offset relative to ip hdr */
349 u16 data_len; /**< data len */
350 u8 subconn_idx; /**< index of the sub_connection being used */
351 u8 flags;
352 } sctp;
353
354 /* SNAT */
355 struct
356 {
357 u32 flags;
358 } snat;
359
360 u32 unused[6];
361 };
362} vnet_buffer_opaque_t;
363
364/*
365 * The opaque field of the vlib_buffer_t is intepreted as a
366 * vnet_buffer_opaque_t. Hence it should be big enough to accommodate one.
367 */
368STATIC_ASSERT (sizeof (vnet_buffer_opaque_t) <=
369 STRUCT_SIZE_OF (vlib_buffer_t, opaque),
370 "VNET buffer meta-data too large for vlib_buffer");
371
372#define vnet_buffer(b) ((vnet_buffer_opaque_t *) (b)->opaque)
373
374/* Full cache line (64 bytes) of additional space */
375typedef struct
376{
377 /**
378 * QoS marking data that needs to persist from the recording nodes
379 * (nominally in the ingress path) to the marking node (in the
380 * egress path)
381 */
382 struct
383 {
384 u8 bits;
385 u8 source;
386 } qos;
387
388 u8 __unused[2];
389
390 union
391 {
392 struct
393 {
394#if VLIB_BUFFER_TRACE_TRAJECTORY > 0
395 /* buffer trajectory tracing */
396 u16 *trajectory_trace;
397#endif
398 };
399 u32 unused[11];
400 };
401} vnet_buffer_opaque2_t;
402
403#define vnet_buffer2(b) ((vnet_buffer_opaque2_t *) (b)->opaque2)
404
405/*
406 * The opaque2 field of the vlib_buffer_t is intepreted as a
407 * vnet_buffer_opaque2_t. Hence it should be big enough to accommodate one.
408 */
409STATIC_ASSERT (sizeof (vnet_buffer_opaque2_t) <=
410 STRUCT_SIZE_OF (vlib_buffer_t, opaque2),
411 "VNET buffer opaque2 meta-data too large for vlib_buffer");
