16#ifndef __MPLS_TUNNEL_H__
17#define __MPLS_TUNNEL_H__
18
19#include <vnet/mpls/mpls.h>
20#include <vnet/fib/fib_path_ext.h>
21
22typedef enum mpls_tunnel_attribute_t_
23{
24 MPLS_TUNNEL_ATTRIBUTE_FIRST = 0,
25 /**
26 * @brief The tunnel is L2 only
27 */
28 MPLS_TUNNEL_ATTRIBUTE_L2 = MPLS_TUNNEL_ATTRIBUTE_FIRST,
29 /**
30 * @brief The tunnel has an underlying multicast LSP
31 */
32 MPLS_TUNNEL_ATTRIBUTE_MCAST,
33 MPLS_TUNNEL_ATTRIBUTE_LAST = MPLS_TUNNEL_ATTRIBUTE_MCAST,
34} mpls_tunnel_attribute_t;
35
36#define MPLS_TUNNEL_ATTRIBUTES { \
37 [MPLS_TUNNEL_ATTRIBUTE_MCAST] = "multicast", \
38 [MPLS_TUNNEL_ATTRIBUTE_L2] = "L2", \
39}
40#define FOR_EACH_MPLS_TUNNEL_ATTRIBUTE(_item) \
41 for (_item = MPLS_TUNNEL_ATTRIBUTE_FIRST; \
42 _item <= MPLS_TUNNEL_ATTRIBUTE_LAST; \
43 _item++)
44
45typedef enum mpls_tunnel_flag_t_ {
46 MPLS_TUNNEL_FLAG_NONE = 0,
47 MPLS_TUNNEL_FLAG_L2 = (1 << MPLS_TUNNEL_ATTRIBUTE_L2),
48 MPLS_TUNNEL_FLAG_MCAST = (1 << MPLS_TUNNEL_ATTRIBUTE_MCAST),
49} __attribute__ ((packed)) mpls_tunnel_flags_t;
50
51
52/**
53 * @brief A uni-directional MPLS tunnel
54 */
55typedef struct mpls_tunnel_t_
56{
57 /**
58 * @brief The tunnel hooks into the FIB control plane graph.
59 */
60 fib_node_t mt_node;
61
62 /**
63 * @brief Tunnel flags
64 */
65 mpls_tunnel_flags_t mt_flags;
66
67 /**
68 * @brief If the tunnel is an L2 tunnel, this is the link type ETHERNET
69 * load-balance
70 */
71 dpo_id_t mt_l2_lb;
72
73 /**
74 * @brief The HW interface index of the tunnel interfaces
75 */
76 u32 mt_hw_if_index;
77
78 /**
79 * @brief The SW interface index of the tunnel interfaces
80 */
81 u32 mt_sw_if_index;
82
83 /**
84 * @brief The path-list over which the tunnel's destination is reachable
85 */
86 fib_node_index_t mt_path_list;
87
88 /**
89 * @brief sibling index on the path-list so notifications are received.
90 */
91 u32 mt_sibling_index;
92
93 /**
94 * A vector of path extensions o hold the label stack for each path
95 */
96 fib_path_ext_list_t mt_path_exts;
97} mpls_tunnel_t;
98
99/**
100 * @brief Create a new MPLS tunnel
101 * @return the SW Interface index of the newly created tuneel
102 */
103extern u32 vnet_mpls_tunnel_create (u8 l2_only,
104 u8 is_multicast);
105
106/**
107 * @brief Add a path to an MPLS tunnel
108 */
109extern void vnet_mpls_tunnel_path_add (u32 sw_if_index,
110 fib_route_path_t *rpath);
111
112/**
113 * @brief remove a path from a tunnel.
114 * @return the number of remaining paths. 0 implies the tunnel can be deleted
115 */
116extern int vnet_mpls_tunnel_path_remove (u32 sw_if_index,
117 fib_route_path_t *rpath);
118
119/**
120 * @brief Delete an MPLS tunnel
121 */
122extern void vnet_mpls_tunnel_del (u32 sw_if_index);
123
124extern const mpls_tunnel_t *mpls_tunnel_get(u32 index);
125
126/**
127 * @brief Callback function invoked while walking MPLS tunnels
128 */
129typedef void (*mpls_tunnel_walk_cb_t)(u32 index, void *ctx);
130
131/**
132 * @brief Walk all the MPLS tunnels
133 */
134extern void mpls_tunnel_walk(mpls_tunnel_walk_cb_t cb,
135 void *ctx);

137#endif
