54static void vl_api_sr_localsid_add_del_t_handler
55 (vl_api_sr_localsid_add_del_t * mp)
56{
57 vl_api_sr_localsid_add_del_reply_t *rmp;
58 int rv = 0;
59/*
60 * int sr_cli_localsid (char is_del, ip6_address_t *localsid_addr,
61 * char end_psp, u8 behavior, u32 sw_if_index, u32 vlan_index, u32 fib_table,
62 * ip46_address_t *nh_addr, void *ls_plugin_mem)
63 */
64
65 VALIDATE_SW_IF_INDEX (mp);
66
67 rv = sr_cli_localsid (mp->is_del,
68 (ip6_address_t *) & mp->localsid_addr,
69 mp->end_psp,
70 mp->behavior,
71 ntohl (mp->sw_if_index),
72 ntohl (mp->vlan_index),
73 ntohl (mp->fib_table),
74 (ip46_address_t *) & mp->nh_addr, NULL);
75
76 BAD_SW_IF_INDEX_LABEL;
77
78 REPLY_MACRO (VL_API_SR_LOCALSID_ADD_DEL_REPLY);
79}

81static void
82vl_api_sr_policy_add_t_handler (vl_api_sr_policy_add_t * mp)
83{
84 vl_api_sr_policy_add_reply_t *rmp;
85 ip6_address_t *segments = 0, *seg;
86 ip6_address_t *this_address = (ip6_address_t *) mp->segments;
87
88 int i;
89 for (i = 0; i < mp->n_segments; i++)
90 {
91 vec_add2 (segments, seg, 1);
92 clib_memcpy (seg->as_u8, this_address->as_u8, sizeof (*this_address));
93 this_address++;
94 }
95
96/*
97 * sr_policy_add (ip6_address_t *bsid, ip6_address_t *segments,
98 * u32 weight, u8 behavior, u32 fib_table, u8 is_encap)
99 */
100 int rv = 0;
101 rv = sr_policy_add ((ip6_address_t *) & mp->bsid_addr,
102 segments,
103 ntohl (mp->weight),
104 mp->type, ntohl (mp->fib_table), mp->is_encap);
105
106 REPLY_MACRO (VL_API_SR_POLICY_ADD_REPLY);
107}

109static void
110vl_api_sr_policy_mod_t_handler (vl_api_sr_policy_mod_t * mp)
111{
112 vl_api_sr_policy_mod_reply_t *rmp;
113
114 ip6_address_t *segments = 0, *seg;
115 ip6_address_t *this_address = (ip6_address_t *) mp->segments;
116
117 int i;
118 for (i = 0; i < mp->n_segments; i++)
119 {
120 vec_add2 (segments, seg, 1);
121 clib_memcpy (seg->as_u8, this_address->as_u8, sizeof (*this_address));
122 this_address++;
123 }
124
125 int rv = 0;
126/*
127 * int
128 * sr_policy_mod(ip6_address_t *bsid, u32 index, u32 fib_table,

130 * u32 weight, u8 is_encap)

132 rv = sr_policy_mod ((ip6_address_t *) & mp->bsid_addr,

134 ntohl (mp->fib_table),
135 mp->operation,
136 segments, ntohl (mp->sl_index), ntohl (mp->weight));
137
138 REPLY_MACRO (VL_API_SR_POLICY_MOD_REPLY);
139}

193static void send_sr_localsid_details
194 (ip6_sr_localsid_t * t, vl_api_registration_t * reg, u32 context)
195{
196 vl_api_sr_localsids_details_t *rmp;
197
198 rmp = vl_msg_api_alloc (sizeof (*rmp));
199 memset (rmp, 0, sizeof (*rmp));
200 rmp->_vl_msg_id = ntohs (VL_API_SR_LOCALSIDS_DETAILS);
201 memcpy (rmp->address, &t->localsid, sizeof (ip6_address_t));
202 rmp->end_psp = t->end_psp;
203 rmp->behavior = htons (t->behavior);
204 rmp->fib_table = htonl (t->fib_table);
205 memcpy (rmp->xconnect_next_hop, &t->next_hop, sizeof (ip6_address_t));
206 rmp->xconnect_iface_or_vrf_table = htonl (t->sw_if_index);
207 rmp->context = context;
208
209 vl_api_send_msg (reg, (u8 *) rmp);
210}

212static void vl_api_sr_localsids_dump_t_handler
213 (vl_api_sr_localsids_dump_t * mp)
214{
215 vl_api_registration_t *reg;
216 ip6_sr_main_t *sm = &sr_main;
217 ip6_sr_localsid_t *t;
218
219 reg = vl_api_client_index_to_registration (mp->client_index);
220 if (!reg)
221 return;
222
223 /* *INDENT-OFF* */
224 pool_foreach (t, sm->localsids,
225 ({
226 send_sr_localsid_details(t, reg, mp->context);
227 }));
228 /* *INDENT-ON* */
229}

243static void
244setup_message_id_table (api_main_t * am)
245{
246#define _(id,n,crc) vl_msg_api_add_msg_name_crc (am, #n "_" #crc, id);
247 foreach_vl_msg_name_crc_sr;
248#undef _
249}
250
251static clib_error_t *
252sr_api_hookup (vlib_main_t * vm)
253{
254 api_main_t *am = &api_main;
255
256#define _(N,n) \
257 vl_msg_api_set_handlers(VL_API_##N, #n, \
258 vl_api_##n##_t_handler, \
259 vl_noop_handler, \
260 vl_api_##n##_t_endian, \
261 vl_api_##n##_t_print, \
262 sizeof(vl_api_##n##_t), 1);
263 foreach_vpe_api_msg;
264#undef _
265
266 /*
267 * Manually register the sr policy add msg, so we trace
268 * enough bytes to capture a typical segment list
269 */
270 vl_msg_api_set_handlers (VL_API_SR_POLICY_ADD,
271 "sr_policy_add",
272 vl_api_sr_policy_add_t_handler,
273 vl_noop_handler,
274 vl_api_sr_policy_add_t_endian,
275 vl_api_sr_policy_add_t_print, 256, 1);
276
277 /*
278 * Manually register the sr policy mod msg, so we trace
279 * enough bytes to capture a typical segment list
280 */
281 vl_msg_api_set_handlers (VL_API_SR_POLICY_MOD,
282 "sr_policy_mod",
283 vl_api_sr_policy_mod_t_handler,
284 vl_noop_handler,
285 vl_api_sr_policy_mod_t_endian,
286 vl_api_sr_policy_mod_t_print, 256, 1);
287
288 /*
289 * Set up the (msg_name, crc, message-id) table
290 */
291 setup_message_id_table (am);
292
293 return 0;
294}
