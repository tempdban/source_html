27/* Per-subinterface L2 feature configuration */
28
29typedef struct
30{
31
32 union
33 {
34 u16 bd_index; /* bridge domain id */
35 u32 output_sw_if_index; /* for xconnect */
36 };
37
38 /* Interface mode. If both are 0, this interface is in L3 mode */
39 u8 xconnect;
40 u8 bridge;
41
42 /* this is the bvi interface for the bridge-domain */
43 u8 bvi;
44
45 /* config for which input features are configured on this interface */
46 u32 feature_bitmap;
47
48 /* some of these flags are also in the feature bitmap */
49 u8 learn_enable;
50 u8 fwd_enable;
51 u8 flood_enable;
52
53 /* split horizon group */
54 u8 shg;
55
56} l2_input_config_t;

80static_always_inline l2_bridge_domain_t *
81l2input_bd_config_from_index (l2input_main_t * l2im, u32 bd_index)
82{
83 l2_bridge_domain_t *bd_config;
84
85 bd_config = vec_elt_at_index (l2im->bd_configs, bd_index);
86 return bd_is_valid (bd_config) ? bd_config : NULL;
87}

89static_always_inline l2_bridge_domain_t *
90l2input_bd_config (u32 bd_index)
91{
92 l2input_main_t *mp = &l2input_main;
93 l2_bridge_domain_t *bd_config;
94
95 vec_validate (mp->bd_configs, bd_index);
96 bd_config = vec_elt_at_index (mp->bd_configs, bd_index);
97 return bd_config;
98}

123/* Feature bitmap positions */
124typedef enum
125{
126#define _(sym,str) L2INPUT_FEAT_##sym##_BIT,
127 foreach_l2input_feat
128#undef _
129 L2INPUT_N_FEAT
130} l2input_feat_t;
131
132STATIC_ASSERT (L2INPUT_N_FEAT <= 32, "too many l2 input features");
133
134/* Feature bit masks */
135typedef enum
136{
137#define _(sym,str) L2INPUT_FEAT_##sym = (1<<L2INPUT_FEAT_##sym##_BIT),
138 foreach_l2input_feat
139#undef _
140 L2INPUT_VALID_MASK =
141#define _(sym,str) L2INPUT_FEAT_##sym |
142 foreach_l2input_feat
143#undef _
144 0
145} l2input_feat_masks_t;
146
147STATIC_ASSERT ((u64) L2INPUT_VALID_MASK == (1ull << L2INPUT_N_FEAT) - 1, "");

216static inline void
217vnet_update_l2_len (vlib_buffer_t * b)
218{
219 ethernet_header_t *eth;
220 u16 ethertype;
221 u8 vlan_count = 0;
222
223 /* point at currrent l2 hdr */
224 eth = vlib_buffer_get_current (b);
225
226 /*
227 * l2-output pays no attention to this
228 * but the tag push/pop code on an l2 subif needs it.
229 *
230 * Determine l2 header len, check for up to 2 vlans
231 */
232 vnet_buffer (b)->l2.l2_len = sizeof (ethernet_header_t);
233 ethertype = clib_net_to_host_u16 (eth->type);
234 if (ethernet_frame_is_tagged (ethertype))
235 {
236 ethernet_vlan_header_t *vlan;
237 vnet_buffer (b)->l2.l2_len += sizeof (*vlan);
238 vlan_count = 1;
239 vlan = (void *) (eth + 1);
240 ethertype = clib_net_to_host_u16 (vlan->type);
241 if (ethertype == ETHERNET_TYPE_VLAN)
242 {
243 vnet_buffer (b)->l2.l2_len += sizeof (*vlan);
244 vlan_count = 2;
245 }
246 }
247 ethernet_buffer_set_vlan_count (b, vlan_count);
248}

250/*
251 * Compute flow hash of an ethernet packet, use 5-tuple hash if L3 packet
252 * is ip4 or ip6. Otherwise hash on smac/dmac/etype.
253 * The vlib buffer current pointer is expected to be at ethernet header
254 * and vnet l2.l2_len is exppected to be setup already.
255 */
256static inline u32
257vnet_l2_compute_flow_hash (vlib_buffer_t * b)
258{
259 ethernet_header_t *eh = vlib_buffer_get_current (b);
260 u8 *l3h = (u8 *) eh + vnet_buffer (b)->l2.l2_len;
261 u16 ethertype = clib_net_to_host_u16 (*(u16 *) (l3h - 2));
262
263 if (ethertype == ETHERNET_TYPE_IP4)
264 return ip4_compute_flow_hash ((ip4_header_t *) l3h, IP_FLOW_HASH_DEFAULT);
265 else if (ethertype == ETHERNET_TYPE_IP6)
266 return ip6_compute_flow_hash ((ip6_header_t *) l3h, IP_FLOW_HASH_DEFAULT);
267 else
268 {
269 u32 a, b, c;
270 u32 *ap = (u32 *) & eh->dst_address[2];
271 u32 *bp = (u32 *) & eh->src_address[2];
272 a = *ap;
273 b = *bp;
274 c = ethertype;
275 hash_v3_mix32 (a, b, c);
276 hash_v3_finalize32 (a, b, c);
277 return c;
278 }
279}
