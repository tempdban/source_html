25typedef struct _segment_manager_properties
26{
27 /** Session fifo sizes. */
28 u32 rx_fifo_size;
29 u32 tx_fifo_size;
30 u32 evt_q_size;
31
32 /** Configured additional segment size */
33 u32 add_segment_size;
34
35 /** Flag that indicates if additional segments should be created */
36 u8 add_segment;
37
38 /** Segment type: if set to SSVM_N_TYPES, private segments are used */
39 ssvm_segment_type_t segment_type;
40
41} segment_manager_properties_t;
42
43typedef struct _segment_manager
44{
45 /** Pool of segments allocated by this manager */
46 svm_fifo_segment_private_t *segments;
47
48 /** rwlock that protects the segments pool */
49 clib_rwlock_t segments_rwlock;
50
51 /** Owner app index */
52 u32 app_index;
53
54 /**
55 * First segment should not be deleted unless segment manger is deleted.
56 * This also indicates that the segment manager is the first to have been
57 * allocated for the app.
58 */
59 u8 first_is_protected;
60
61 /**
62 * App event queue allocated in first segment
63 */
64 svm_queue_t *event_queue;
65} segment_manager_t;
66
67#define segment_manager_foreach_segment_w_lock(VAR, SM, BODY) \
68do { \
69 clib_rwlock_reader_lock (&(SM)->segments_rwlock); \
70 pool_foreach((VAR), ((SM)->segments), (BODY)); \
71 clib_rwlock_reader_unlock (&(SM)->segments_rwlock); \
72} while (0)
73
74typedef struct segment_manager_main_
75{
76 /** Pool of segment managers */
77 segment_manager_t *segment_managers;
78
79 /** Virtual address allocator */
80 clib_valloc_main_t va_allocator;
81
82} segment_manager_main_t;

92#define SEGMENT_MANAGER_INVALID_APP_INDEX ((u32) ~0)
93
94/** Pool of segment managers */
95extern segment_manager_t *segment_managers;
96
97always_inline segment_manager_t *
98segment_manager_get (u32 index)
99{
100 return pool_elt_at_index (segment_manager_main.segment_managers, index);
101}
102
103always_inline segment_manager_t *
104segment_manager_get_if_valid (u32 index)
105{
106 if (pool_is_free_index (segment_manager_main.segment_managers, index))
107 return 0;
108 return pool_elt_at_index (segment_manager_main.segment_managers, index);
109}
110
111always_inline u32
112segment_manager_index (segment_manager_t * sm)
113{
114 return sm - segment_manager_main.segment_managers;
115}
116
117always_inline svm_queue_t *
118segment_manager_event_queue (segment_manager_t * sm)
119{
120 return sm->event_queue;
121}
