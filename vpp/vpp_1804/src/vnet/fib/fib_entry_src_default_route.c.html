16#include "fib_entry.h"
17#include "fib_entry_src.h"
18
19/**
20 * Source initialisation Function
21 */
22static void
23fib_entry_src_default_route_init (fib_entry_src_t *src)
24{
25 src->fes_flags = FIB_ENTRY_SRC_FLAG_NONE;
26}
27
28static void
29fib_entry_src_default_route_remove (fib_entry_src_t *src)
30{
31 src->fes_pl = FIB_NODE_INDEX_INVALID;
32}
33
34static void
35fib_entry_src_default_route_add (fib_entry_src_t *src,
36 const fib_entry_t *entry,
37 fib_entry_flag_t flags,
38 dpo_proto_t proto,
39 const dpo_id_t *dpo)
40{
41 src->fes_pl = fib_path_list_create_special(proto,
42 FIB_PATH_LIST_FLAG_DROP,
43 dpo);
44}
45
46const static fib_entry_src_vft_t interface_src_vft = {
47 .fesv_init = fib_entry_src_default_route_init,
48 .fesv_add = fib_entry_src_default_route_add,
49 .fesv_remove = fib_entry_src_default_route_remove,
50};
51
52void
53fib_entry_src_default_route_register (void)
54{
55 fib_entry_src_register(FIB_SOURCE_DEFAULT_ROUTE, &interface_src_vft);
56}
