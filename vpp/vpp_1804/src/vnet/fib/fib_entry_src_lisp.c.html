16#include "fib_entry.h"
17#include "fib_entry_src.h"
18#include "fib_path_list.h"
19
20/**
21 * Source initialisation Function
22 */
23static void
24fib_entry_src_lisp_init (fib_entry_src_t *src)
25{
26}
27
28/**
29 * Source deinitialisation Function
30 */
31static void
32fib_entry_src_lisp_deinit (fib_entry_src_t *src)
33{
34}
35
36static void
37fib_entry_src_lisp_path_swap (fib_entry_src_t *src,
38 const fib_entry_t *entry,
39 fib_path_list_flags_t pl_flags,
40 const fib_route_path_t *paths)
41{
42 src->fes_pl = fib_path_list_create((FIB_PATH_LIST_FLAG_SHARED | pl_flags),
43 paths);
44}
45
46static void
47fib_entry_src_lisp_path_add (fib_entry_src_t *src,
48 const fib_entry_t *entry,
49 fib_path_list_flags_t pl_flags,
50 const fib_route_path_t *paths)
51{
52 if (FIB_NODE_INDEX_INVALID == src->fes_pl)
53 {
54 src->fes_pl =
55 fib_path_list_create((FIB_PATH_LIST_FLAG_SHARED | pl_flags), paths);
56 }
57 else
58 {
59 src->fes_pl =
60 fib_path_list_copy_and_path_add(src->fes_pl,
61 (FIB_PATH_LIST_FLAG_SHARED | pl_flags),
62 paths);
63 }
64}
65
66static void
67fib_entry_src_lisp_path_remove (fib_entry_src_t *src,
68 fib_path_list_flags_t pl_flags,
69 const fib_route_path_t *paths)
70{
71 if (FIB_NODE_INDEX_INVALID != src->fes_pl)
72 {
73 src->fes_pl =
74 fib_path_list_copy_and_path_remove(src->fes_pl,
75 (FIB_PATH_LIST_FLAG_SHARED | pl_flags),
76 paths);
77 }
78}
79
80static void
81fib_entry_src_lisp_add (fib_entry_src_t *src,
82 const fib_entry_t *entry,
83 fib_entry_flag_t flags,
84 dpo_proto_t proto,
85 const dpo_id_t *dpo)
86{
87 if (FIB_ENTRY_FLAG_NONE != flags)
88 {
89 src->fes_pl = fib_path_list_create_special(
90 proto,
91 fib_entry_src_flags_2_path_list_flags(flags),
92 dpo);
93 }
94}
95
96static void
97fib_entry_src_lisp_remove (fib_entry_src_t *src)
98{
99 src->fes_pl = FIB_NODE_INDEX_INVALID;
100}
101
102static void
103fib_entry_src_lisp_set_data (fib_entry_src_t *src,
104 const fib_entry_t *entry,
105 const void *data)
106{
107 src->u.lisp.fesl_fib_index = *(u32*)data;
108}
109
110static const void*
111fib_entry_src_lisp_get_data (fib_entry_src_t *src,
112 const fib_entry_t *entry)
113{
114 return (&(src->u.lisp.fesl_fib_index));
115}

117const static fib_entry_src_vft_t api_src_vft = {
118 .fesv_init = fib_entry_src_lisp_init,
119 .fesv_deinit = fib_entry_src_lisp_deinit,
120 .fesv_add = fib_entry_src_lisp_add,
121 .fesv_remove = fib_entry_src_lisp_remove,
122 .fesv_path_add = fib_entry_src_lisp_path_add,
123 .fesv_path_swap = fib_entry_src_lisp_path_swap,
124 .fesv_path_remove = fib_entry_src_lisp_path_remove,
125 .fesv_set_data = fib_entry_src_lisp_set_data,
126 .fesv_get_data = fib_entry_src_lisp_get_data,
127};
128
129void
130fib_entry_src_lisp_register (void)
131{
132 fib_entry_src_register(FIB_SOURCE_LISP, &api_src_vft);
133}
