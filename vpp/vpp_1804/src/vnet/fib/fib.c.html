22static clib_error_t *
23fib_module_init (vlib_main_t * vm)
24{
25 clib_error_t * error;
26
27 if ((error = vlib_call_init_function (vm, dpo_module_init)))
28 return (error);
29 if ((error = vlib_call_init_function (vm, adj_module_init)))
30 return (error);
31
32 fib_entry_module_init();
33 fib_entry_src_module_init();
34 fib_path_module_init();
35 fib_path_list_module_init();
36 fib_walk_module_init();
37
38 return (NULL);
39}
40
41VLIB_INIT_FUNCTION (fib_module_init);
