496static inline vapi_error_e vapi_dhcp_client_config(struct vapi_ctx_s *ctx,
497 vapi_msg_dhcp_client_config *msg,
498 vapi_error_e (*callback)(struct vapi_ctx_s *ctx,
499 void *callback_ctx,
500 vapi_error_e rv,
501 bool is_last,
502 vapi_payload_dhcp_client_config_reply *reply),
503 void *callback_ctx)
504{
505 if (!msg || !callback) {
506 return VAPI_EINVAL;
507 }
508 if (vapi_is_nonblocking(ctx) && vapi_requests_full(ctx)) {
509 return VAPI_EAGAIN;
510 }
511 vapi_error_e rv;
512 if (VAPI_OK != (rv = vapi_producer_lock (ctx))) {
513 return rv;
514 }
515 u32 req_context = vapi_gen_req_context(ctx);
516 msg->header.context = req_context;
517 vapi_msg_dhcp_client_config_hton(msg);
518 if (VAPI_OK == (rv = vapi_send (ctx, msg))) {
519 vapi_store_request(ctx, req_context, false, (vapi_cb_t)callback, callback_ctx);
520 if (VAPI_OK != vapi_producer_unlock (ctx)) {
521 abort (); /* this really shouldn't happen */
522 }
523 if (vapi_is_nonblocking(ctx)) {
524 rv = VAPI_OK;
525 } else {
526 rv = vapi_dispatch(ctx);
527 }
528 } else {
529 vapi_msg_dhcp_client_config_ntoh(msg);
530 if (VAPI_OK != vapi_producer_unlock (ctx)) {
531 abort (); /* this really shouldn't happen */
532 }
533 }
534 return rv;
535}
