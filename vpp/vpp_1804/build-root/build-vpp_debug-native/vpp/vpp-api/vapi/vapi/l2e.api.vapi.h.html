124static inline vapi_error_e vapi_l2_emulation(struct vapi_ctx_s *ctx,
125 vapi_msg_l2_emulation *msg,
126 vapi_error_e (*callback)(struct vapi_ctx_s *ctx,
127 void *callback_ctx,
128 vapi_error_e rv,
129 bool is_last,
130 vapi_payload_l2_emulation_reply *reply),
131 void *callback_ctx)
132{
133 if (!msg || !callback) {
134 return VAPI_EINVAL;
135 }
136 if (vapi_is_nonblocking(ctx) && vapi_requests_full(ctx)) {
137 return VAPI_EAGAIN;
138 }
139 vapi_error_e rv;
140 if (VAPI_OK != (rv = vapi_producer_lock (ctx))) {
141 return rv;
142 }
143 u32 req_context = vapi_gen_req_context(ctx);
144 msg->header.context = req_context;
145 vapi_msg_l2_emulation_hton(msg);
146 if (VAPI_OK == (rv = vapi_send (ctx, msg))) {
147 vapi_store_request(ctx, req_context, false, (vapi_cb_t)callback, callback_ctx);
148 if (VAPI_OK != vapi_producer_unlock (ctx)) {
149 abort (); /* this really shouldn't happen */
150 }
151 if (vapi_is_nonblocking(ctx)) {
152 rv = VAPI_OK;
153 } else {
154 rv = vapi_dispatch(ctx);
155 }
156 } else {
157 vapi_msg_l2_emulation_ntoh(msg);
158 if (VAPI_OK != vapi_producer_unlock (ctx)) {
159 abort (); /* this really shouldn't happen */
160 }
161 }
162 return rv;
163}
