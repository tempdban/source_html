37__forceinline
38void permute_operation(uint32_t *pa, uint32_t *pb,
39 const uint32_t n, const uint32_t m)
40{
41 register uint32_t t = (*pb ^ (*pa >> n)) & m;
42
43 *pb ^= t;
44 *pa ^= (t << n);
45}
46
47/* inital permutation */
48__forceinline
49void ip_z(uint32_t *pl, uint32_t *pr)
50{
51 permute_operation(pr, pl, 4, 0x0f0f0f0f);
52 permute_operation(pl, pr, 16, 0x0000ffff);
53 permute_operation(pr, pl, 2, 0x33333333);
54 permute_operation(pl, pr, 8, 0x00ff00ff);
55 permute_operation(pr, pl, 1, 0x55555555);
56}
57
58/* final permuation */
59__forceinline
60void fp_z(uint32_t *pl, uint32_t *pr)
61{
62 permute_operation(pl, pr, 1, 0x55555555);
63 permute_operation(pr, pl, 8, 0x00ff00ff);
64 permute_operation(pl, pr, 2, 0x33333333);
65 permute_operation(pr, pl, 16, 0x0000ffff);
66 permute_operation(pl, pr, 4, 0x0f0f0f0f);
67}

69/* 1st part of DES round
70 * - permutes and exands R(32 bits) into 48 bits
71 */
72__forceinline
73uint64_t e_phase(const uint64_t R)
74{
75 /* E phase as in FIPS46-3 and also 8x6 to 8x8 expansion.
76 *
77 * Bit selection table for this operation looks as follows:
78 * 32, 1, 2, 3, 4, 5, X, X,
79 * 4, 5, 6, 7, 8, 9, X, X,
80 * 8, 9, 10, 11, 12, 13, X, X,
81 * 12, 13, 14, 15, 16, 17, X, X,
82 * 16, 17, 18, 19, 20, 21, X, X,
83 * 20, 21, 22, 23, 24, 25, X, X,
84 * 24, 25, 26, 27, 28, 29, X, X,
85 * 28, 29, 30, 31, 32, 1, X, X
86 * where 'X' is bit value 0.
87 */
88 return ((R << 1) & UINT64_C(0x3e)) | ((R >> 31) & UINT64_C(1)) |
89 ((R << 5) & UINT64_C(0x3f00)) |
90 ((R << 9) & UINT64_C(0x3f0000)) |
91 ((R << 13) & UINT64_C(0x3f000000)) |
92 ((R << 17) & UINT64_C(0x3f00000000)) |
93 ((R << 21) & UINT64_C(0x3f0000000000)) |
94 ((R << 25) & UINT64_C(0x3f000000000000)) |
95 ((R << 29) & UINT64_C(0x1f00000000000000)) |
96 ((R & UINT64_C(1)) << 61);
97}

379__forceinline
380uint32_t fRK(const uint32_t R, const uint64_t K)
381{
382 uint64_t x;
383
384 /* Combined e-phase and 8x6bits to 8x8bits expansion.
385 * 32 bits -> 48 bits permutation
386 */
387 x = e_phase((uint64_t) R) ^ K;
388
389 /* Combined s-box and p-phase.
390 * s-box: 48 bits -> 32 bits
391 * p-phase: 32 bits -> 32 bites permutation
392 */
393 return sbox0p[x & 0x3f] |
394 sbox1p[(x >> (8 * 1)) & 0x3f] |
395 sbox2p[(x >> (8 * 2)) & 0x3f] |
396 sbox3p[(x >> (8 * 3)) & 0x3f] |
397 sbox4p[(x >> (8 * 4)) & 0x3f] |
398 sbox5p[(x >> (8 * 5)) & 0x3f] |
399 sbox6p[(x >> (8 * 6)) & 0x3f] |
400 sbox7p[(x >> (8 * 7)) & 0x3f];
401}

403__forceinline
404uint64_t enc_dec_1(const uint64_t data, const uint64_t *ks, const int enc)
405{
406 uint32_t l, r;
407
408 r = (uint32_t) (data);
409 l = (uint32_t) (data >> 32);
410 ip_z(&r, &l);
411
412 if (enc) {
413 l ^= fRK(r, ks[0]);
414 r ^= fRK(l, ks[1]);
415 l ^= fRK(r, ks[2]);
416 r ^= fRK(l, ks[3]);
417 l ^= fRK(r, ks[4]);
418 r ^= fRK(l, ks[5]);
419 l ^= fRK(r, ks[6]);
420 r ^= fRK(l, ks[7]);
421 l ^= fRK(r, ks[8]);
422 r ^= fRK(l, ks[9]);
423 l ^= fRK(r, ks[10]);
424 r ^= fRK(l, ks[11]);
425 l ^= fRK(r, ks[12]);
426 r ^= fRK(l, ks[13]);
427 l ^= fRK(r, ks[14]);
428 r ^= fRK(l, ks[15]);
429 } else {
430 l ^= fRK(r, ks[15]); /* l: l0 -> r1/l2 */
431 r ^= fRK(l, ks[14]); /* r: r0 -> r2 */
432 l ^= fRK(r, ks[13]);
433 r ^= fRK(l, ks[12]);
434 l ^= fRK(r, ks[11]);
435 r ^= fRK(l, ks[10]);
436 l ^= fRK(r, ks[9]);
437 r ^= fRK(l, ks[8]);
438 l ^= fRK(r, ks[7]);
439 r ^= fRK(l, ks[6]);
440 l ^= fRK(r, ks[5]);
441 r ^= fRK(l, ks[4]);
442 l ^= fRK(r, ks[3]);
443 r ^= fRK(l, ks[2]);
444 l ^= fRK(r, ks[1]);
445 r ^= fRK(l, ks[0]);
446 }
447
448 fp_z(&r, &l);
449 return ((uint64_t) l) | (((uint64_t) r) << 32);
450}

504__forceinline
505void
506cfb_one_basic(const void *input, void *output, const int size,
507 const uint64_t *ks, const uint64_t *ivec)
508{
509 uint8_t *out = (uint8_t *) output;
510 const uint8_t *in = (const uint8_t *) input;
511 uint64_t t;
512
513 IMB_ASSERT(size <= 8 && size >= 0);
514 IMB_ASSERT(input != NULL);
515 IMB_ASSERT(output != NULL);
516 IMB_ASSERT(ks != NULL);
517 IMB_ASSERT(ivec != NULL);
518
519 t = enc_dec_1(*ivec, ks, 1 /* encrypt */);
520
521 /* XOR and copy in one go */
522 if (size & 1) {
523 *out++ = *in++ ^ ((uint8_t) t);
524 t >>= 8;
525 }
526
527 if (size & 2) {
528 uint16_t *out2 = (uint16_t *) out;
529 const uint16_t *in2 = (const uint16_t *) in;
530
531 *out2 = *in2 ^ ((uint16_t) t);
532 t >>= 16;
533 out += 2;
534 in += 2;
535 }
536
537 if (size & 4) {
538 uint32_t *out4 = (uint32_t *) out;
539 const uint32_t *in4 = (const uint32_t *) in;
540
541 *out4 = *in4 ^ ((uint32_t) t);
542 }
543}

546void
547docsis_des_enc_basic(const void *input, void *output, const int size,
548 const uint64_t *ks, const uint64_t *ivec)
549{
550 const uint64_t *in = input;
551 uint64_t *out = output;
552 const int nblocks = size / DES_BLOCK_SIZE;
553 const int partial = size & 7;
554 int n;
555 uint64_t iv = *ivec;
556
557 IMB_ASSERT(size >= 0);
558 IMB_ASSERT(input != NULL);
559 IMB_ASSERT(output != NULL);
560 IMB_ASSERT(ks != NULL);
561 IMB_ASSERT(ivec != NULL);
562
563 for (n = 0; n < nblocks; n++)
564 out[n] = iv = enc_dec_1(in[n] ^ iv, ks, 1 /* encrypt */);
565
566 if (partial) {
567 if (nblocks)
568 cfb_one_basic(&in[nblocks], &out[nblocks], partial,
569 ks, &out[nblocks - 1]);
570 else
571 cfb_one_basic(input, output, partial, ks, ivec);
572 }
573
574 /* *ivec = iv; */
575 iv = 0;
576}

578IMB_DLL_LOCAL
579void
580docsis_des_dec_basic(const void *input, void *output, const int size,
581 const uint64_t *ks, const uint64_t *ivec)
582{
583 const uint64_t *in = input;
584 uint64_t *out = output;
585 const int nblocks = size / DES_BLOCK_SIZE;
586 const int partial = size & 7;
587 int n;
588 uint64_t iv = *ivec;
589
590 IMB_ASSERT(size >= 0);
591 IMB_ASSERT(input != NULL);
592 IMB_ASSERT(output != NULL);
593 IMB_ASSERT(ks != NULL);
594 IMB_ASSERT(ivec != NULL);
595
596 if (partial) {
597 if (!nblocks) {
598 /* first block is the partial one */
599 cfb_one_basic(input, output, partial, ks, ivec);
600 iv = 0;
601 return;
602 }
603 /* last block is partial */
604 cfb_one_basic(&in[nblocks], &out[nblocks], partial,
605 ks, &in[nblocks - 1]);
606 }
607
608 for (n = 0; n < nblocks; n++) {
609 uint64_t in_block = in[n];
610
611 out[n] = enc_dec_1(in_block, ks, 0 /* decrypt */) ^ iv;
612 iv = in_block;
613 }
614
615 /* *ivec = iv; */
616 iv = 0;
617}
