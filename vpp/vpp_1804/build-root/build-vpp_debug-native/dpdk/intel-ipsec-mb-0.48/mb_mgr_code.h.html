28/*
29 * This contains the bulk of the mb_mgr code, with #define's to build
30 * an SSE, AVX, AVX2 or AVX512 version (see mb_mgr_sse.c, mb_mgr_avx.c, etc.)
31 *
32 * get_next_job() returns a job object. This must be filled in and returned
33 * via submit_job() before get_next_job() is called again.

36 * to be usable at the next call to get_next_job()

38 * Assume JOBS() and ADV_JOBS() from mb_mgr_code.h are available

132__forceinline
133JOB_AES_HMAC *
134submit_flush_job_aes_ccm(MB_MGR_CBCMAC_OOO *state, JOB_AES_HMAC *job,
135 const unsigned max_jobs, const int is_submit)
136{
137 const unsigned lane_blocks_size = 64;
138 const unsigned aad_len_size = 2;
139 unsigned lane, min_len, min_idx;
140 JOB_AES_HMAC *ret_job = NULL;
141 uint8_t *pb = NULL;
142 unsigned i;
143
144 if (is_submit) {
145 /*
146 * SUBMIT
147 * - get a free lane id
148 */
149 const unsigned L = AES_BLOCK_SIZE - 1 -
150 (unsigned) job->iv_len_in_bytes;
151
152 lane = state->unused_lanes & 15;
153 state->unused_lanes >>= 4;
154 pb = &state->init_blocks[lane * lane_blocks_size];
155
156 /*
157 * Build IV for AES-CTR-128.
158 * - byte 0: flags with L'
159 * - bytes 1 to 13: nonce
160 * - zero bytes after nonce (up to byte 15)
161 *
162 * First AES block of init_blocks will always hold this format
163 * throughtout job processing.
164 */
165 memset(&pb[8], 0, 8);
166 pb[0] = (uint8_t) L - 1; /* flags = L` = L - 1 */
167 /* nonce 7 to 13 */
168 memcpy(&pb[1], job->iv, job->iv_len_in_bytes);
169
170 if (job->cipher_direction != ENCRYPT) {
171 /* decrypt before authentication */
172 pb[15] = 1;
173 AES_CNTR_128(job->src +
174 job->cipher_start_src_offset_in_bytes,
175 pb, job->aes_enc_key_expanded, job->dst,
176 job->msg_len_to_cipher_in_bytes,
177 AES_BLOCK_SIZE);
178 }
179
180 /* copy job data in and set up inital blocks */
181 state->job_in_lane[lane] = job;
182 state->lens[lane] = AES_BLOCK_SIZE;
183 state->init_done[lane] = 0;
184 state->args.in[lane] = pb;
185 state->args.keys[lane] = job->aes_enc_key_expanded;
186 memset(&state->args.IV[lane], 0, sizeof(state->args.IV[0]));
187
188 /*
189 * Convert AES-CTR IV into BLOCK 0 for CBC-MAC-128:
190 * - correct flags by adding M' (AAD later)
191 * - put message length
192 */
193 pb[0] |= ((job->auth_tag_output_len_in_bytes - 2) >> 1) << 3;
194 pb[14] = (uint8_t) (job->msg_len_to_hash_in_bytes >> 8);
195 pb[15] = (uint8_t) job->msg_len_to_hash_in_bytes;
196
197 /* Make AAD correction and put together AAD blocks, if any */
198 if (job->u.CCM.aad_len_in_bytes != 0) {
199 /*
200 * - increment length by length of AAD and
201 * AAD length size
202 * - add AAD present flag
203 * - copy AAD to the lane initial blocks
204 * - zero trailing block bytes
205 */
206 const unsigned aadl =
207 (unsigned) job->u.CCM.aad_len_in_bytes +
208 aad_len_size;
209
210 state->lens[lane] +=
211 (aadl + AES_BLOCK_SIZE - 1) &
212 (~(AES_BLOCK_SIZE - 1));
213 pb[0] |= 0x40;
214 pb[AES_BLOCK_SIZE + 0] =
215 (uint8_t) (job->u.CCM.aad_len_in_bytes >> 8);
216 pb[AES_BLOCK_SIZE + 1] =
217 (uint8_t) job->u.CCM.aad_len_in_bytes;
218 memcpy(&pb[AES_BLOCK_SIZE + aad_len_size],
219 job->u.CCM.aad,
220 job->u.CCM.aad_len_in_bytes);
221 memset(&pb[AES_BLOCK_SIZE + aadl], 0,
222 state->lens[lane] - aadl);
223 }
224
225 /* enough jobs to start processing? */
226 if (state->unused_lanes != 0xf)
227 return NULL;
228 } else {
229 /*
230 * FLUSH
231 * - find 1st non null job
232 */
233 for (lane = 0; lane < max_jobs; lane++)
234 if (state->job_in_lane[lane] != NULL)
235 break;
236 if (lane >= max_jobs)
237 return NULL; /* no not null job */
238 }
239
240 ccm_round:
241 if (is_submit) {
242 /*
243 * SUBMIT
244 * - find min common length to process
245 */
246 min_idx = 0;
247 min_len = state->lens[0];
248
249 for (i = 1; i < max_jobs; i++) {
250 if (min_len > state->lens[i]) {
251 min_idx = i;
252 min_len = state->lens[i];
253 }
254 }
255 } else {
256 /*
257 * FLUSH
258 * - copy good (not null) lane onto empty lanes
259 * - find min common length to process across not null lanes
260 */
261 min_idx = lane;
262 min_len = state->lens[lane];
263
264 for (i = 0; i < max_jobs; i++) {
265 if (i == lane)
266 continue;
267
268 if (state->job_in_lane[i] != NULL) {
269 if (min_len > state->lens[i]) {
270 min_idx = i;
271 min_len = state->lens[i];
272 }
273 } else {
274 state->args.in[i] = state->args.in[lane];
275 state->args.keys[i] = state->args.keys[lane];
276 state->args.IV[i] = state->args.IV[lane];
277 state->lens[i] = UINT16_MAX;
278 state->init_done[i] = state->init_done[lane];
279 }
280 }
281 }
282
283 /* subtract min len from all lanes */
284 for (i = 0; i < max_jobs; i++)
285 state->lens[i] -= min_len;
286
287 /* run the algorythmic code on selected blocks */
288 if (min_len != 0)
289 AES128_CBC_MAC(&state->args, min_len);
290
291 ret_job = state->job_in_lane[min_idx];
292 pb = &state->init_blocks[min_idx * lane_blocks_size];
293
294 if (state->init_done[min_idx] == 0) {
295 /*
296 * First block and AAD blocks are done.
297 * Full message blocks are to do.
298 */
299 if (ret_job->cipher_direction == ENCRYPT)
300 state->args.in[min_idx] = ret_job->src +
301 ret_job->hash_start_src_offset_in_bytes;
302 else
303 state->args.in[min_idx] = ret_job->dst;
304
305 state->init_done[min_idx] = 1;
306
307 if (ret_job->msg_len_to_hash_in_bytes & (~15)) {
308 /* first block + AAD done - process message blocks */
309 state->lens[min_idx] =
310 ret_job->msg_len_to_hash_in_bytes & (~15);
311 goto ccm_round;
312 }
313 }
314
315 if (state->init_done[min_idx] == 1 &&
316 (ret_job->msg_len_to_hash_in_bytes & 15)) {
317 /*
318 * First block, AAD, message blocks are done.
319 * Partial message block is still to do.
320 */
321 state->init_done[min_idx] = 2;
322 state->lens[min_idx] = AES_BLOCK_SIZE;
323 memset(&pb[AES_BLOCK_SIZE], 0, AES_BLOCK_SIZE);
324 memcpy(&pb[AES_BLOCK_SIZE], state->args.in[min_idx],
325 (size_t) ret_job->msg_len_to_hash_in_bytes & 15);
326 state->args.in[min_idx] = &pb[AES_BLOCK_SIZE];
327 goto ccm_round;
328 }
329
330 /*
331 * Final XOR with AES-CNTR on B_0
332 * - remove M' and AAD presence bits from flags
333 * - set counter to 0
334 */
335 pb[0] = pb[0] & 7;
336 pb[14] = 0;
337 pb[15] = 0;
338
339 /*
340 * Clever use of AES-CTR mode saves a few ops here.
341 * What AES-CCM authentication requires us to do is:
342 * AES-CCM: E(KEY,B_0) XOR IV_CBC_MAC
343 *
344 * And what AES_CTR offers is:
345 * AES_CTR: E(KEY, NONCE|COUNTER) XOR PLAIN_TEXT
346 *
347 * So if:
348 * B_0 is passed instead of NONCE|COUNTER and IV instead of PLAIN_TESXT
349 * then AES_CTR function is doing pretty much what we need.
350 * On top of it can truncate the authentication tag and copy to
351 * destination.
352 */
353 AES_CNTR_128(&state->args.IV[min_idx] /* src = IV */,
354 pb /* nonce/iv = B_0 */,
355 state->args.keys[min_idx],
356 ret_job->auth_tag_output /* dst */,
357 ret_job->auth_tag_output_len_in_bytes /* num_bytes */,
358 AES_BLOCK_SIZE /* nonce/iv len */);
359
360 if (ret_job->cipher_direction == ENCRYPT) {
361 /* encrypt after authentication */
362 pb[15] = 1; /* start from counter 1, not 0 */
363 AES_CNTR_128(ret_job->src +
364 ret_job->cipher_start_src_offset_in_bytes,
365 pb, ret_job->aes_enc_key_expanded, ret_job->dst,
366 ret_job->msg_len_to_cipher_in_bytes,
367 AES_BLOCK_SIZE);
368 }
369
370 /* put back processed packet into unused lanes, set job as complete */
371 state->unused_lanes = (state->unused_lanes << 4) | min_idx;
372 ret_job = state->job_in_lane[min_idx];
373 ret_job->status |= (STS_COMPLETED_HMAC|STS_COMPLETED_AES);
374 state->job_in_lane[min_idx] = NULL;
375 return ret_job;
376}

529/* ========================================================================= */
530/* DOCSIS AES (AES128 CBC + AES128 CFB) */
531/* ========================================================================= */
532
533#define AES_BLOCK_SIZE 16
534
535/**
536 * @brief Encrypts/decrypts the last partial block for DOCSIS SEC v3.1 BPI
537 *
538 * The last partial block is encrypted/decrypted using AES CFB128.
539 * IV is always the next last ciphered block.
540 *
541 * @note It is assumed that length is bigger than one AES 128 block.
542 *
543 * @param job desriptor of performed crypto operation
544 * @return It always returns value passed in \a job
545 */
546__forceinline
547JOB_AES_HMAC *
548DOCSIS_LAST_BLOCK(JOB_AES_HMAC *job)
549{
550 const void *iv = NULL;
551 UINT64 offset = 0;
552 UINT64 partial_bytes = 0;
553
554 if (job == NULL)
555 return job;
556
557 IMB_ASSERT((job->cipher_direction == DECRYPT) ||
558 (job->status & STS_COMPLETED_AES));
559
560 partial_bytes = job->msg_len_to_cipher_in_bytes & (AES_BLOCK_SIZE - 1);
561 offset = job->msg_len_to_cipher_in_bytes & (~(AES_BLOCK_SIZE - 1));
562
563 if (!partial_bytes)
564 return job;
565
566 /* in either case IV has to be next last ciphered block */
567 if (job->cipher_direction == ENCRYPT)
568 iv = job->dst + offset - AES_BLOCK_SIZE;
569 else
570 iv = job->src + job->cipher_start_src_offset_in_bytes +
571 offset - AES_BLOCK_SIZE;
572
573 IMB_ASSERT(partial_bytes <= AES_BLOCK_SIZE);
574 AES_CFB_128_ONE(job->dst + offset,
575 job->src + job->cipher_start_src_offset_in_bytes +
576 offset,
577 iv, job->aes_enc_key_expanded, partial_bytes);
578
579 return job;
580}
581
582/**
583 * @brief Encrypts/decrypts the first and only partial block for
584 * DOCSIS SEC v3.1 BPI
585 *
586 * The first partial block is encrypted/decrypted using AES CFB128.
587 *
588 * @param job desriptor of performed crypto operation
589 * @return It always returns value passed in \a job
590 */
591__forceinline
592JOB_AES_HMAC *
593DOCSIS_FIRST_BLOCK(JOB_AES_HMAC *job)
594{
595 IMB_ASSERT(!(job->status & STS_COMPLETED_AES));
596 IMB_ASSERT(job->msg_len_to_cipher_in_bytes <= AES_BLOCK_SIZE);
597 AES_CFB_128_ONE(job->dst,
598 job->src + job->cipher_start_src_offset_in_bytes,
599 job->iv, job->aes_enc_key_expanded,
600 job->msg_len_to_cipher_in_bytes);
601 job->status |= STS_COMPLETED_AES;
602 return job;
603}

669/**
670 * @brief DES cipher decryption
671 *
672 * @param job desriptor of performed crypto operation
673 * @return It always returns value passed in \a job
674 */
675__forceinline
676JOB_AES_HMAC *
677DES_CBC_DEC(JOB_AES_HMAC *job)
678{
679 IMB_ASSERT(!(job->status & STS_COMPLETED_AES));
680 des_dec_cbc_basic(job->src + job->cipher_start_src_offset_in_bytes,
681 job->dst,
682 job->msg_len_to_cipher_in_bytes &
683 (~(DES_BLOCK_SIZE - 1)),
684 job->aes_dec_key_expanded, (const uint64_t *)job->iv);
685 job->status |= STS_COMPLETED_AES;
686 return job;
687}

693__forceinline
694JOB_AES_HMAC *
695SUBMIT_JOB_AES_ENC(MB_MGR *state, JOB_AES_HMAC *job)
696{
697 if (CBC == job->cipher_mode) {
698 if (16 == job->aes_key_len_in_bytes) {
699 return SUBMIT_JOB_AES128_ENC(&state->aes128_ooo, job);
700 } else if (24 == job->aes_key_len_in_bytes) {
701 return SUBMIT_JOB_AES192_ENC(&state->aes192_ooo, job);
702 } else { /* assume 32 */
703 return SUBMIT_JOB_AES256_ENC(&state->aes256_ooo, job);
704 }
705 } else if (CNTR == job->cipher_mode) {
706 if (16 == job->aes_key_len_in_bytes) {
707 return SUBMIT_JOB_AES128_CNTR(job);
708 } else if (24 == job->aes_key_len_in_bytes) {
709 return SUBMIT_JOB_AES192_CNTR(job);
710 } else { /* assume 32 */
711 return SUBMIT_JOB_AES256_CNTR(job);
712 }
713 } else if (DOCSIS_SEC_BPI == job->cipher_mode) {
714 if (job->msg_len_to_cipher_in_bytes >= AES_BLOCK_SIZE) {
715 JOB_AES_HMAC *tmp;
716
717 tmp = SUBMIT_JOB_AES128_ENC(&state->docsis_sec_ooo,
718 job);
719 return DOCSIS_LAST_BLOCK(tmp);
720 } else
721 return DOCSIS_FIRST_BLOCK(job);
722#ifndef NO_GCM
723 } else if (GCM == job->cipher_mode) {
724 return SUBMIT_JOB_AES_GCM_ENC(job);
725#endif /* NO_GCM */
726 } else if (CUSTOM_CIPHER == job->cipher_mode) {
727 return SUBMIT_JOB_CUSTOM_CIPHER(job);
728 } else if (DES == job->cipher_mode) {
729#ifdef SUBMIT_JOB_DES_CBC_ENC
730 return SUBMIT_JOB_DES_CBC_ENC(&state->des_enc_ooo, job);
731#else
732 return DES_CBC_ENC(job);
733#endif /* SUBMIT_JOB_DES_CBC_ENC */
734 } else if (DOCSIS_DES == job->cipher_mode) {
735#ifdef SUBMIT_JOB_DOCSIS_DES_ENC
736 return SUBMIT_JOB_DOCSIS_DES_ENC(&state->docsis_des_enc_ooo,
737 job);
738#else
739 return DOCSIS_DES_ENC(job);
740#endif /* SUBMIT_JOB_DOCSIS_DES_ENC */
741 } else { /* assume NUL_CIPHER or CCM */
742 job->status |= STS_COMPLETED_AES;
743 return job;
744 }

951#ifdef DEBUG
952#ifdef _WIN32
953#define INVALID_PRN(_fmt, ...) \
954 fprintf(stderr, "%s():%d: " _fmt, __FUNCTION__, __LINE__, __VA_ARGS__)
955#else
956#define INVALID_PRN(_fmt, ...) \
957 fprintf(stderr, "%s():%d: " _fmt, __func__, __LINE__, __VA_ARGS__)
958#endif
959#else
960#define INVALID_PRN(_fmt, ...)
961#endif
962
963__forceinline int
964is_job_invalid(const JOB_AES_HMAC *job)
965{
966 const UINT64 auth_tag_len_max[] = {
967 0, /* INVALID selection */
968 12, /* SHA1 */
969 14, /* SHA_224 */
970 16, /* SHA_256 */
971 24, /* SHA_384 */
972 32, /* SHA_512 */
973 12, /* AES_XCBC */
974 12, /* MD5 */
975 0, /* NULL_HASH */
976 16, /* AES_GMAC */
977 0, /* CUSTOM HASH */
978 0, /* AES_CCM */
979 };
980
981 switch (job->cipher_mode) {
982 case CBC:
983 if (job->aes_key_len_in_bytes != UINT64_C(16) &&
984 job->aes_key_len_in_bytes != UINT64_C(24) &&
985 job->aes_key_len_in_bytes != UINT64_C(32)) {
986 INVALID_PRN("cipher_mode:%d\n", job->cipher_mode);
987 return 1;
988 }
989 if (job->msg_len_to_cipher_in_bytes == 0) {
990 INVALID_PRN("cipher_mode:%d\n", job->cipher_mode);
991 return 1;
992 }
993 if (job->msg_len_to_cipher_in_bytes & UINT64_C(15)) {
994 INVALID_PRN("cipher_mode:%d\n", job->cipher_mode);
995 return 1;
996 }
997 if (job->iv_len_in_bytes != UINT64_C(16)) {
998 INVALID_PRN("cipher_mode:%d\n", job->cipher_mode);
999 return 1;
1000 }
1001 break;
1002 case CNTR:
1003 if (job->aes_key_len_in_bytes != UINT64_C(16) &&
1004 job->aes_key_len_in_bytes != UINT64_C(24) &&
1005 job->aes_key_len_in_bytes != UINT64_C(32)) {
1006 INVALID_PRN("cipher_mode:%d\n", job->cipher_mode);
1007 return 1;
1008 }
1009 if (job->iv_len_in_bytes != UINT64_C(16) &&
1010 job->iv_len_in_bytes != UINT64_C(12)) {
1011 INVALID_PRN("cipher_mode:%d\n", job->cipher_mode);
1012 return 1;
1013 }
1014 if (job->msg_len_to_cipher_in_bytes == 0) {
1015 INVALID_PRN("cipher_mode:%d\n", job->cipher_mode);
1016 return 1;
1017 }
1018 break;
1019 case NULL_CIPHER:
1020 /* NULL_CIPHER only allowed in HASH_CIPHER */
1021 if (job->chain_order != HASH_CIPHER)
1022 return 1;
1023 /* XXX: not copy src to dst */
1024 break;
1025 case DOCSIS_SEC_BPI:
1026 if (job->aes_key_len_in_bytes != UINT64_C(16)) {
1027 INVALID_PRN("cipher_mode:%d\n", job->cipher_mode);
1028 return 1;
1029 }
1030 if (job->iv_len_in_bytes != UINT64_C(16)) {
1031 INVALID_PRN("cipher_mode:%d\n", job->cipher_mode);
1032 return 1;
1033 }
1034 if (job->msg_len_to_cipher_in_bytes == 0) {
1035 INVALID_PRN("cipher_mode:%d\n", job->cipher_mode);
1036 return 1;
1037 }
1038 break;
1039#ifndef NO_GCM
1040 case GCM:
1041 if (job->aes_key_len_in_bytes != UINT64_C(16) &&
1042 job->aes_key_len_in_bytes != UINT64_C(24) &&
1043 job->aes_key_len_in_bytes != UINT64_C(32)) {
1044 INVALID_PRN("cipher_mode:%d\n", job->cipher_mode);
1045 return 1;
1046 }
1047 if (job->iv_len_in_bytes != UINT64_C(12)) {
1048 INVALID_PRN("cipher_mode:%d\n", job->cipher_mode);
1049 return 1;
1050 }
1051 if (job->hash_alg != AES_GMAC) {
1052 INVALID_PRN("cipher_mode:%d\n", job->cipher_mode);
1053 return 1;
1054 }
1055 if (job->msg_len_to_cipher_in_bytes == 0) {
1056 INVALID_PRN("cipher_mode:%d\n", job->cipher_mode);
1057 return 1;
1058 }
1059 break;
1060#endif /* !NO_GCM */
1061 case CUSTOM_CIPHER:
1062 /* no checks here */
1063 if (job->cipher_func == NULL) {
1064 INVALID_PRN("cipher_mode:%d\n", job->cipher_mode);
1065 return 1;
1066 }
1067 break;
1068 case DES:
1069 if (job->aes_key_len_in_bytes != UINT64_C(8)) {
1070 INVALID_PRN("cipher_mode:%d\n", job->cipher_mode);
1071 return 1;
1072 }
1073 if (job->msg_len_to_cipher_in_bytes == 0) {
1074 INVALID_PRN("cipher_mode:%d\n", job->cipher_mode);
1075 return 1;
1076 }
1077 if (job->msg_len_to_cipher_in_bytes & UINT64_C(7)) {
1078 INVALID_PRN("cipher_mode:%d\n", job->cipher_mode);
1079 return 1;
1080 }
1081 if (job->iv_len_in_bytes != UINT64_C(8)) {
1082 INVALID_PRN("cipher_mode:%d\n", job->cipher_mode);
1083 return 1;
1084 }
1085 break;
1086 case DOCSIS_DES:
1087 if (job->aes_key_len_in_bytes != UINT64_C(8)) {
1088 INVALID_PRN("cipher_mode:%d\n", job->cipher_mode);
1089 return 1;
1090 }
1091 if (job->msg_len_to_cipher_in_bytes == 0) {
1092 INVALID_PRN("cipher_mode:%d\n", job->cipher_mode);
1093 return 1;
1094 }
1095 if (job->iv_len_in_bytes != UINT64_C(8)) {
1096 INVALID_PRN("cipher_mode:%d\n", job->cipher_mode);
1097 return 1;
1098 }
1099 break;
1100 case CCM:
1101 /* currently only AES-CCM-128 is only supported */
1102 if (job->aes_key_len_in_bytes != UINT64_C(16)) {
1103 INVALID_PRN("cipher_mode:%d\n", job->cipher_mode);
1104 return 1;
1105 }
1106 /*
1107 * From RFC3610:
1108 * Nonce length = 15 - L
1109 * Valid L values are: 2 to 8
1110 * Then valid nonce lengths 13 to 7 (inclusive).
1111 */
1112 if (job->iv_len_in_bytes > UINT64_C(13) ||
1113 job->iv_len_in_bytes < UINT64_C(7)) {
1114 INVALID_PRN("cipher_mode:%d\n", job->cipher_mode);
1115 return 1;
1116 }
1117 if (job->msg_len_to_cipher_in_bytes == 0) {
1118 INVALID_PRN("cipher_mode:%d\n", job->cipher_mode);
1119 return 1;
1120 }
1121 if (job->hash_alg != AES_CCM) {
1122 INVALID_PRN("cipher_mode:%d\n", job->cipher_mode);
1123 return 1;
1124 }
1125 break;
1126 default:
1127 INVALID_PRN("cipher_mode:%d\n", job->cipher_mode);
1128 return 1;
1129 }
1130
1131 switch (job->hash_alg) {
1132 case SHA1:
1133 case AES_XCBC:
1134 case MD5:
1135 case SHA_224:
1136 case SHA_256:
1137 case SHA_384:
1138 case SHA_512:
1139 if (job->auth_tag_output_len_in_bytes !=
1140 auth_tag_len_max[job->hash_alg]) {
1141 INVALID_PRN("hash_alg:%d\n", job->hash_alg);
1142 return 1;
1143 }
1144 if (job->msg_len_to_hash_in_bytes == 0) {
1145 INVALID_PRN("hash_alg:%d\n", job->hash_alg);
1146 return 1;
1147 }
1148 break;
1149 case NULL_HASH:
1150 break;
1151#ifndef NO_GCM
1152 case AES_GMAC:
1153 if (job->auth_tag_output_len_in_bytes != UINT64_C(8) &&
1154 job->auth_tag_output_len_in_bytes != UINT64_C(12) &&
1155 job->auth_tag_output_len_in_bytes != UINT64_C(16)) {
1156 INVALID_PRN("hash_alg:%d\n", job->hash_alg);
1157 return 1;
1158 }
1159 if (job->cipher_mode != GCM) {
1160 INVALID_PRN("hash_alg:%d\n", job->hash_alg);
1161 return 1;
1162 }
1163 /*
1164 * msg_len_to_hash_in_bytes not checked against zero.
1165 * It is not used for AES-GCM & GMAC - see
1166 * SUBMIT_JOB_AES_GCM_ENC and SUBMIT_JOB_AES_GCM_DEC functions.
1167 */
1168 break;
1169#endif /* !NO_GCM */
1170 case CUSTOM_HASH:
1171 if (job->hash_func == NULL) {
1172 INVALID_PRN("hash_alg:%d\n", job->hash_alg);
1173 return 1;
1174 }
1175 break;
1176 case AES_CCM:
1177 if (job->u.CCM.aad_len_in_bytes > 46) {
1178 /* 3 x AES_BLOCK - 2 bytes for AAD len */
1179 INVALID_PRN("hash_alg:%d\n", job->hash_alg);
1180 return 1;
1181 }
1182 if ((job->u.CCM.aad_len_in_bytes > 0) &&
1183 (job->u.CCM.aad == NULL)) {
1184 INVALID_PRN("hash_alg:%d\n", job->hash_alg);
1185 return 1;
1186 }
1187 /* M can be any even number from 4 to 16 */
1188 if (job->auth_tag_output_len_in_bytes < UINT64_C(4) ||
1189 job->auth_tag_output_len_in_bytes > UINT64_C(16) ||
1190 ((job->auth_tag_output_len_in_bytes & 1) != 0)) {
1191 INVALID_PRN("hash_alg:%d\n", job->hash_alg);
1192 return 1;
1193 }
1194 if (job->cipher_mode != CCM) {
1195 INVALID_PRN("hash_alg:%d\n", job->hash_alg);
1196 return 1;
1197 }
1198 /*
1199 * AES-CCM allows for only one message for
1200 * cipher and uthentication.
1201 * AAD can be used to extend authentication over
1202 * clear text fields.
1203 */
1204 if (job->msg_len_to_cipher_in_bytes !=
1205 job->msg_len_to_hash_in_bytes) {
1206 INVALID_PRN("hash_alg:%d\n", job->hash_alg);
1207 return 1;
1208 }
1209 if (job->cipher_start_src_offset_in_bytes !=
1210 job->hash_start_src_offset_in_bytes) {
1211 INVALID_PRN("hash_alg:%d\n", job->hash_alg);
1212 return 1;
1213 }
1214 break;
1215 default:
1216 INVALID_PRN("hash_alg:%d\n", job->hash_alg);
1217 return 1;
1218 }
1219
1220 switch (job->chain_order) {
1221 case CIPHER_HASH:
1222 if (job->cipher_direction != ENCRYPT) {
1223 INVALID_PRN("chain_order:%d\n", job->chain_order);
1224 return 1;
1225 }
1226 break;
1227 case HASH_CIPHER:
1228 if (job->cipher_mode != NULL_CIPHER) {
1229 if (job->cipher_direction != DECRYPT) {
1230 INVALID_PRN("chain_order:%d\n",
1231 job->chain_order);
1232 return 1;
1233 }
1234 }
1235 break;
1236 default:
1237 INVALID_PRN("chain_order:%d\n", job->chain_order);
1238 return 1;
1239 }
1240
1241 return 0;
1242}

1244__forceinline
1245JOB_AES_HMAC *submit_new_job(MB_MGR *state, JOB_AES_HMAC *job)
1246{
1247 if (job->chain_order == CIPHER_HASH) {
1248 /* assume job->cipher_direction == ENCRYPT */
1249 job = SUBMIT_JOB_AES_ENC(state, job);
1250 if (job) {
1251 job = SUBMIT_JOB_HASH(state, job);
1252 if (job && (job->chain_order == HASH_CIPHER))
1253 SUBMIT_JOB_AES_DEC(state, job);
1254 } /* end if job */
1255 } else { /* job->chain_order == HASH_CIPHER */
1256 /* assume job->cipher_direction == DECRYPT */
1257 job = SUBMIT_JOB_HASH(state, job);
1258 if (job && (job->chain_order == HASH_CIPHER))
1259 SUBMIT_JOB_AES_DEC(state, job);
1260 }
1261 return job;
1262}

1264__forceinline
1265void complete_job(MB_MGR *state, JOB_AES_HMAC *job)
1266{
1267 JOB_AES_HMAC *tmp = NULL;
1268
1269 while (job->status < STS_COMPLETED) {
1270 if (job->chain_order == CIPHER_HASH) {
1271 /* assume job->cipher_direction == ENCRYPT */
1272 tmp = FLUSH_JOB_AES_ENC(state, job);
1273 if (tmp)
1274 tmp = SUBMIT_JOB_HASH(state, tmp);
1275 else
1276 tmp = FLUSH_JOB_HASH(state, job);
1277 if (tmp && (tmp->chain_order == HASH_CIPHER))
1278 SUBMIT_JOB_AES_DEC(state, tmp);
1279 } else { /* job->chain_order == HASH_CIPHER */
1280 /* assume job->cipher_direction == DECRYPT */
1281 tmp = FLUSH_JOB_HASH(state, job);
1282 if (tmp == NULL)
1283 tmp = FLUSH_JOB_AES_DEC(state, job);
1284 else
1285 if (tmp->chain_order == HASH_CIPHER)
1286 SUBMIT_JOB_AES_DEC(state, tmp);
1287 }
1288 }

1306 job->status = STS_INVALID_ARGS;

1308 job->status = STS_BEING_PROCESSED;
1309 job = submit_new_job(state, job);
1310 }
1311 } else {
1312 job->status = STS_BEING_PROCESSED;
1313 job = submit_new_job(state, job);
1314 }
1315
1316 if (state->earliest_job < 0) {
1317 /* state was previously empty */
1318 state->earliest_job = state->next_job;
1319 ADV_JOBS(&state->next_job);
1320#ifndef LINUX
1321 RESTORE_XMMS(xmm_save);
1322#endif
1323 return NULL; /* if we were empty, nothing to return */
1324 }
1325
1326 ADV_JOBS(&state->next_job);
1327
1328 if (state->earliest_job == state->next_job) {
1329 /* Full */
1330 job = JOBS(state, state->earliest_job);
1331 complete_job(state, job);
1332 ADV_JOBS(&state->earliest_job);
1333#ifndef LINUX

1338
1339 /* not full */

1343 job = JOBS(state, state->earliest_job);
