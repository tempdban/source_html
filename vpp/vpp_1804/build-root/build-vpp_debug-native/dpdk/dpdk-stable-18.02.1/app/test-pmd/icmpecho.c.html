38static const char *
39arp_op_name(uint16_t arp_op)
40{
41 switch (arp_op ) {
42 case ARP_OP_REQUEST:
43 return "ARP Request";
44 case ARP_OP_REPLY:
45 return "ARP Reply";
46 case ARP_OP_REVREQUEST:
47 return "Reverse ARP Request";
48 case ARP_OP_REVREPLY:
49 return "Reverse ARP Reply";
50 case ARP_OP_INVREQUEST:
51 return "Peer Identify Request";
52 case ARP_OP_INVREPLY:
53 return "Peer Identify Reply";
54 default:
55 break;
56 }
57 return "Unkwown ARP op";
58}

212static void
213ipv4_addr_to_dot(uint32_t be_ipv4_addr, char *buf)
214{
215 uint32_t ipv4_addr;
216
217 ipv4_addr = rte_be_to_cpu_32(be_ipv4_addr);
218 sprintf(buf, "%d.%d.%d.%d", (ipv4_addr >> 24) & 0xFF,
219 (ipv4_addr >> 16) & 0xFF, (ipv4_addr >> 8) & 0xFF,
220 ipv4_addr & 0xFF);
221}

245static uint16_t
246ipv4_hdr_cksum(struct ipv4_hdr *ip_h)
247{
248 uint16_t *v16_h;
249 uint32_t ip_cksum;
250
251 /*
252 * Compute the sum of successive 16-bit words of the IPv4 header,
253 * skipping the checksum field of the header.
254 */
255 v16_h = (unaligned_uint16_t *) ip_h;
256 ip_cksum = v16_h[0] + v16_h[1] + v16_h[2] + v16_h[3] +
257 v16_h[4] + v16_h[6] + v16_h[7] + v16_h[8] + v16_h[9];
258
259 /* reduce 32 bit checksum to 16 bits and complement it */
260 ip_cksum = (ip_cksum & 0xffff) + (ip_cksum >> 16);
261 ip_cksum = (ip_cksum & 0xffff) + (ip_cksum >> 16);
262 ip_cksum = (~ip_cksum) & 0x0000FFFF;
263 return (ip_cksum == 0) ? 0xFFFF : (uint16_t) ip_cksum;
264}

269/*
270 * Receive a burst of packets, lookup for ICMP echo requests, and, if any,
271 * send back ICMP echo replies.
272 */
273static void
274reply_to_icmp_echo_rqsts(struct fwd_stream *fs)
275{
276 struct rte_mbuf *pkts_burst[MAX_PKT_BURST];
277 struct rte_mbuf *pkt;
278 struct ether_hdr *eth_h;
279 struct vlan_hdr *vlan_h;
280 struct arp_hdr *arp_h;
281 struct ipv4_hdr *ip_h;
282 struct icmp_hdr *icmp_h;
283 struct ether_addr eth_addr;
284 uint32_t retry;
285 uint32_t ip_addr;
286 uint16_t nb_rx;
287 uint16_t nb_tx;
288 uint16_t nb_replies;
289 uint16_t eth_type;
290 uint16_t vlan_id;
291 uint16_t arp_op;
292 uint16_t arp_pro;
293 uint32_t cksum;
294 uint8_t i;
295 int l2_len;
296#ifdef RTE_TEST_PMD_RECORD_CORE_CYCLES
297 uint64_t start_tsc;
298 uint64_t end_tsc;
299 uint64_t core_cycles;
300#endif
301
302#ifdef RTE_TEST_PMD_RECORD_CORE_CYCLES
303 start_tsc = rte_rdtsc();
304#endif
305
306 /*
307 * First, receive a burst of packets.
308 */
309 nb_rx = rte_eth_rx_burst(fs->rx_port, fs->rx_queue, pkts_burst,
310 nb_pkt_per_burst);
311 if (unlikely(nb_rx == 0))
312 return;
313
314#ifdef RTE_TEST_PMD_RECORD_BURST_STATS
315 fs->rx_burst_stats.pkt_burst_spread[nb_rx]++;
316#endif
317 fs->rx_packets += nb_rx;
318 nb_replies = 0;
319 for (i = 0; i < nb_rx; i++) {
320 if (likely(i < nb_rx - 1))
321 rte_prefetch0(rte_pktmbuf_mtod(pkts_burst[i + 1],
322 void *));
323 pkt = pkts_burst[i];
324 eth_h = rte_pktmbuf_mtod(pkt, struct ether_hdr *);
325 eth_type = RTE_BE_TO_CPU_16(eth_h->ether_type);
326 l2_len = sizeof(struct ether_hdr);
327 if (verbose_level > 0) {
328 printf("\nPort %d pkt-len=%u nb-segs=%u\n",
329 fs->rx_port, pkt->pkt_len, pkt->nb_segs);
330 ether_addr_dump(" ETH: src=", &eth_h->s_addr);
331 ether_addr_dump(" dst=", &eth_h->d_addr);
332 }
333 if (eth_type == ETHER_TYPE_VLAN) {
334 vlan_h = (struct vlan_hdr *)
335 ((char *)eth_h + sizeof(struct ether_hdr));
336 l2_len += sizeof(struct vlan_hdr);
337 eth_type = rte_be_to_cpu_16(vlan_h->eth_proto);
338 if (verbose_level > 0) {
339 vlan_id = rte_be_to_cpu_16(vlan_h->vlan_tci)
340 & 0xFFF;
341 printf(" [vlan id=%u]", vlan_id);
342 }
343 }
344 if (verbose_level > 0) {
345 printf(" type=0x%04x\n", eth_type);
346 }
347
348 /* Reply to ARP requests */
349 if (eth_type == ETHER_TYPE_ARP) {
350 arp_h = (struct arp_hdr *) ((char *)eth_h + l2_len);
351 arp_op = RTE_BE_TO_CPU_16(arp_h->arp_op);
352 arp_pro = RTE_BE_TO_CPU_16(arp_h->arp_pro);
353 if (verbose_level > 0) {
354 printf(" ARP: hrd=%d proto=0x%04x hln=%d "
355 "pln=%d op=%u (%s)\n",
356 RTE_BE_TO_CPU_16(arp_h->arp_hrd),
357 arp_pro, arp_h->arp_hln,
358 arp_h->arp_pln, arp_op,
359 arp_op_name(arp_op));
360 }
361 if ((RTE_BE_TO_CPU_16(arp_h->arp_hrd) !=
362 ARP_HRD_ETHER) ||
363 (arp_pro != ETHER_TYPE_IPv4) ||
364 (arp_h->arp_hln != 6) ||
365 (arp_h->arp_pln != 4)
366 ) {
367 rte_pktmbuf_free(pkt);
368 if (verbose_level > 0)
369 printf("\n");
370 continue;
371 }
372 if (verbose_level > 0) {
373 ether_addr_copy(&arp_h->arp_data.arp_sha, &eth_addr);
374 ether_addr_dump(" sha=", &eth_addr);
375 ip_addr = arp_h->arp_data.arp_sip;
376 ipv4_addr_dump(" sip=", ip_addr);
377 printf("\n");
378 ether_addr_copy(&arp_h->arp_data.arp_tha, &eth_addr);
379 ether_addr_dump(" tha=", &eth_addr);
380 ip_addr = arp_h->arp_data.arp_tip;
381 ipv4_addr_dump(" tip=", ip_addr);
382 printf("\n");
383 }
384 if (arp_op != ARP_OP_REQUEST) {
385 rte_pktmbuf_free(pkt);
386 continue;
387 }
388
389 /*
390 * Build ARP reply.
391 */
392
393 /* Use source MAC address as destination MAC address. */
394 ether_addr_copy(&eth_h->s_addr, &eth_h->d_addr);
395 /* Set source MAC address with MAC address of TX port */
396 ether_addr_copy(&ports[fs->tx_port].eth_addr,
397 &eth_h->s_addr);
398
399 arp_h->arp_op = rte_cpu_to_be_16(ARP_OP_REPLY);
400 ether_addr_copy(&arp_h->arp_data.arp_tha, &eth_addr);
401 ether_addr_copy(&arp_h->arp_data.arp_sha, &arp_h->arp_data.arp_tha);
402 ether_addr_copy(&eth_h->s_addr, &arp_h->arp_data.arp_sha);
403
404 /* Swap IP addresses in ARP payload */
405 ip_addr = arp_h->arp_data.arp_sip;
406 arp_h->arp_data.arp_sip = arp_h->arp_data.arp_tip;
407 arp_h->arp_data.arp_tip = ip_addr;
408 pkts_burst[nb_replies++] = pkt;
409 continue;
410 }
411
412 if (eth_type != ETHER_TYPE_IPv4) {
413 rte_pktmbuf_free(pkt);
414 continue;
415 }
416 ip_h = (struct ipv4_hdr *) ((char *)eth_h + l2_len);
417 if (verbose_level > 0) {
418 ipv4_addr_dump(" IPV4: src=", ip_h->src_addr);
419 ipv4_addr_dump(" dst=", ip_h->dst_addr);
420 printf(" proto=%d (%s)\n",
421 ip_h->next_proto_id,
422 ip_proto_name(ip_h->next_proto_id));
423 }
424
425 /*
426 * Check if packet is a ICMP echo request.
427 */
428 icmp_h = (struct icmp_hdr *) ((char *)ip_h +
429 sizeof(struct ipv4_hdr));
430 if (! ((ip_h->next_proto_id == IPPROTO_ICMP) &&
431 (icmp_h->icmp_type == IP_ICMP_ECHO_REQUEST) &&
432 (icmp_h->icmp_code == 0))) {
433 rte_pktmbuf_free(pkt);
434 continue;
435 }
436
437 if (verbose_level > 0)
438 printf(" ICMP: echo request seq id=%d\n",
439 rte_be_to_cpu_16(icmp_h->icmp_seq_nb));
440
441 /*
442 * Prepare ICMP echo reply to be sent back.
443 * - switch ethernet source and destinations addresses,
444 * - use the request IP source address as the reply IP
445 * destination address,
446 * - if the request IP destination address is a multicast
447 * address:
448 * - choose a reply IP source address different from the
449 * request IP source address,
450 * - re-compute the IP header checksum.
451 * Otherwise:
452 * - switch the request IP source and destination
453 * addresses in the reply IP header,
454 * - keep the IP header checksum unchanged.
455 * - set IP_ICMP_ECHO_REPLY in ICMP header.
456 * ICMP checksum is computed by assuming it is valid in the
457 * echo request and not verified.
458 */
459 ether_addr_copy(&eth_h->s_addr, &eth_addr);
460 ether_addr_copy(&eth_h->d_addr, &eth_h->s_addr);
461 ether_addr_copy(&eth_addr, &eth_h->d_addr);
462 ip_addr = ip_h->src_addr;
463 if (is_multicast_ipv4_addr(ip_h->dst_addr)) {
464 uint32_t ip_src;
465
466 ip_src = rte_be_to_cpu_32(ip_addr);
467 if ((ip_src & 0x00000003) == 1)
468 ip_src = (ip_src & 0xFFFFFFFC) | 0x00000002;
469 else
470 ip_src = (ip_src & 0xFFFFFFFC) | 0x00000001;
471 ip_h->src_addr = rte_cpu_to_be_32(ip_src);
472 ip_h->dst_addr = ip_addr;
473 ip_h->hdr_checksum = ipv4_hdr_cksum(ip_h);
474 } else {
475 ip_h->src_addr = ip_h->dst_addr;
476 ip_h->dst_addr = ip_addr;
477 }
478 icmp_h->icmp_type = IP_ICMP_ECHO_REPLY;
479 cksum = ~icmp_h->icmp_cksum & 0xffff;
480 cksum += ~htons(IP_ICMP_ECHO_REQUEST << 8) & 0xffff;
481 cksum += htons(IP_ICMP_ECHO_REPLY << 8);
482 cksum = (cksum & 0xffff) + (cksum >> 16);
483 cksum = (cksum & 0xffff) + (cksum >> 16);
484 icmp_h->icmp_cksum = ~cksum;
485 pkts_burst[nb_replies++] = pkt;
486 }
487
488 /* Send back ICMP echo replies, if any. */
489 if (nb_replies > 0) {
490 nb_tx = rte_eth_tx_burst(fs->tx_port, fs->tx_queue, pkts_burst,
491 nb_replies);
492 /*
493 * Retry if necessary
494 */
495 if (unlikely(nb_tx < nb_replies) && fs->retry_enabled) {
496 retry = 0;
497 while (nb_tx < nb_replies &&
498 retry++ < burst_tx_retry_num) {
499 rte_delay_us(burst_tx_delay_time);
500 nb_tx += rte_eth_tx_burst(fs->tx_port,
501 fs->tx_queue,
502 &pkts_burst[nb_tx],
503 nb_replies - nb_tx);
504 }
505 }
506 fs->tx_packets += nb_tx;
507#ifdef RTE_TEST_PMD_RECORD_BURST_STATS
508 fs->tx_burst_stats.pkt_burst_spread[nb_tx]++;
509#endif
510 if (unlikely(nb_tx < nb_replies)) {
511 fs->fwd_dropped += (nb_replies - nb_tx);
512 do {
513 rte_pktmbuf_free(pkts_burst[nb_tx]);
514 } while (++nb_tx < nb_replies);
515 }
516 }
517
518#ifdef RTE_TEST_PMD_RECORD_CORE_CYCLES
519 end_tsc = rte_rdtsc();
520 core_cycles = (end_tsc - start_tsc);
521 fs->core_cycles = (uint64_t) (fs->core_cycles + core_cycles);
522#endif
523}
