298#define CRC32_UPD(crc, n) \
299 (crc32c_tables[(n)][(crc) & 0xFF] ^ \
300 crc32c_tables[(n)-1][((crc) >> 8) & 0xFF])

302static inline uint32_t
303crc32c_1byte(uint8_t data, uint32_t init_val)
304{
305 uint32_t crc;
306 crc = init_val;
307 crc ^= data;
308
309 return crc32c_tables[0][crc & 0xff] ^ (crc >> 8);
310}

324static inline uint32_t
325crc32c_1word(uint32_t data, uint32_t init_val)
326{
327 uint32_t crc, term1, term2;
328 crc = init_val;
329 crc ^= data;

332 term2 = crc >> 16;
333 crc = term1 ^ CRC32_UPD(term2, 1);

338static inline uint32_t
339crc32c_2words(uint64_t data, uint32_t init_val)
340{
341 union {
342 uint64_t u64;
343 uint32_t u32[2];
344 } d;
345 d.u64 = data;
346
347 uint32_t crc, term1, term2;
348
349 crc = init_val;
350 crc ^= d.u32[0];
351
352 term1 = CRC32_UPD(crc, 7);
353 term2 = crc >> 16;
354 crc = term1 ^ CRC32_UPD(term2, 5);
355 term1 = CRC32_UPD(d.u32[1], 3);
356 term2 = d.u32[1] >> 16;
357 crc ^= term1 ^ CRC32_UPD(term2, 1);
358
359 return crc;
360}

559/**
560 * Calculate CRC32 hash on user-supplied byte array.
561 *
562 * @param data
563 * Data to perform hash on.
564 * @param data_len
565 * How many bytes to use to calculate hash value.
566 * @param init_val
567 * Value to initialise hash generator.
568 * @return
569 * 32bit calculated hash value.
570 */
571static inline uint32_t
572rte_hash_crc(const void *data, uint32_t data_len, uint32_t init_val)
573{
574 unsigned i;
575 uintptr_t pd = (uintptr_t) data;
576
577 for (i = 0; i < data_len / 8; i++) {
578 init_val = rte_hash_crc_8byte(*(const uint64_t *)pd, init_val);
579 pd += 8;
580 }
581
582 if (data_len & 0x4) {
583 init_val = rte_hash_crc_4byte(*(const uint32_t *)pd, init_val);
584 pd += 4;
585 }
586
587 if (data_len & 0x2) {
588 init_val = rte_hash_crc_2byte(*(const uint16_t *)pd, init_val);
589 pd += 2;
590 }
591
592 if (data_len & 0x1)
593 init_val = rte_hash_crc_1byte(*(const uint8_t *)pd, init_val);
594
595 return init_val;
596}
