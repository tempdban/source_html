34#ifndef _RTE_ETHDEV_VDEV_H_
35#define _RTE_ETHDEV_VDEV_H_
36
37#include <rte_config.h>
38#include <rte_malloc.h>
39#include <rte_bus_vdev.h>
40#include <rte_ethdev_driver.h>
41
42/**
43 * @internal
44 * Allocates a new ethdev slot for an ethernet device and returns the pointer
45 * to that slot for the driver to use.
46 *
47 * @param dev
48 * Pointer to virtual device
49 *
50 * @param private_data_size
51 * Size of private data structure
52 *
53 * @return
54 * A pointer to a rte_eth_dev or NULL if allocation failed.
55 */
56static inline struct rte_eth_dev *
57rte_eth_vdev_allocate(struct rte_vdev_device *dev, size_t private_data_size)
58{
59 struct rte_eth_dev *eth_dev;
60 const char *name = rte_vdev_device_name(dev);
61
62 eth_dev = rte_eth_dev_allocate(name);
63 if (!eth_dev)
64 return NULL;
65
66 if (private_data_size) {
67 eth_dev->data->dev_private = rte_zmalloc_socket(name,
68 private_data_size, RTE_CACHE_LINE_SIZE,
69 dev->device.numa_node);
70 if (!eth_dev->data->dev_private) {
71 rte_eth_dev_release_port(eth_dev);
72 return NULL;
73 }
74 }
75
76 eth_dev->device = &dev->device;
77 eth_dev->intr_handle = NULL;
78
79 eth_dev->data->kdrv = RTE_KDRV_NONE;
80 eth_dev->data->numa_node = dev->device.numa_node;
81 return eth_dev;
82}
83
84#endif /* _RTE_ETHDEV_VDEV_H_ */
