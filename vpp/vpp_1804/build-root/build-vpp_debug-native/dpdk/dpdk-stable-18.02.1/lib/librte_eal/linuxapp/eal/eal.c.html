125/* parse a sysfs (or other) file containing one integer value */
126int
127eal_parse_sysfs_value(const char *filename, unsigned long *val)
128{
129 FILE *f;
130 char buf[BUFSIZ];
131 char *end = NULL;
132
133 if ((f = fopen(filename, "r")) == NULL) {
134 RTE_LOG(ERR, EAL, "%s(): cannot open sysfs value %s\n",
135 __func__, filename);
136 return -1;
137 }
138
139 if (fgets(buf, sizeof(buf), f) == NULL) {
140 RTE_LOG(ERR, EAL, "%s(): cannot read sysfs value %s\n",
141 __func__, filename);
142 fclose(f);
143 return -1;
144 }
145 *val = strtoul(buf, &end, 0);
146 if ((buf[0] == '\0') || (end == NULL) || (*end != '\n')) {
147 RTE_LOG(ERR, EAL, "%s(): cannot parse sysfs value %s\n",
148 __func__, filename);
149 fclose(f);
150 return -1;
151 }
152 fclose(f);
153 return 0;
154}

157/* create memory configuration in shared/mmap memory. Take out
158 * a write lock on the memsegs, so we can auto-detect primary/secondary.
159 * This means we never close the file while running (auto-close on exit).
160 * We also don't lock the whole file, so that in future we can use read-locks
161 * on other parts, e.g. memzones, to detect if there are running secondary
162 * processes. */
163static void
164rte_eal_config_create(void)
165{
166 void *rte_mem_cfg_addr;
167 int retval;
168
169 const char *pathname = eal_runtime_config_path();
170
171 if (internal_config.no_shconf)
172 return;
173
174 /* map the config before hugepage address so that we don't waste a page */
175 if (internal_config.base_virtaddr != 0)
176 rte_mem_cfg_addr = (void *)
177 RTE_ALIGN_FLOOR(internal_config.base_virtaddr -
178 sizeof(struct rte_mem_config), sysconf(_SC_PAGE_SIZE));
179 else
180 rte_mem_cfg_addr = NULL;
181
182 if (mem_cfg_fd < 0){
183 mem_cfg_fd = open(pathname, O_RDWR | O_CREAT, 0660);
184 if (mem_cfg_fd < 0)
185 rte_panic("Cannot open '%s' for rte_mem_config\n", pathname);
186 }
187
188 retval = ftruncate(mem_cfg_fd, sizeof(*rte_config.mem_config));
189 if (retval < 0){
190 close(mem_cfg_fd);
191 rte_panic("Cannot resize '%s' for rte_mem_config\n", pathname);
192 }
193
194 retval = fcntl(mem_cfg_fd, F_SETLK, &wr_lock);
195 if (retval < 0){
196 close(mem_cfg_fd);
197 rte_exit(EXIT_FAILURE, "Cannot create lock on '%s'. Is another primary "
198 "process running?\n", pathname);
199 }
200
201 rte_mem_cfg_addr = mmap(rte_mem_cfg_addr, sizeof(*rte_config.mem_config),
202 PROT_READ | PROT_WRITE, MAP_SHARED, mem_cfg_fd, 0);
203
204 if (rte_mem_cfg_addr == MAP_FAILED){
205 rte_panic("Cannot mmap memory for rte_config\n");
206 }
207 memcpy(rte_mem_cfg_addr, &early_mem_config, sizeof(early_mem_config));
208 rte_config.mem_config = rte_mem_cfg_addr;
209
210 /* store address of the config in the config itself so that secondary
211 * processes could later map the config into this exact location */
212 rte_config.mem_config->mem_cfg_addr = (uintptr_t) rte_mem_cfg_addr;
213
214}

216/* attach to an existing shared memory config */
217static void
218rte_eal_config_attach(void)
219{
220 struct rte_mem_config *mem_config;
221
222 const char *pathname = eal_runtime_config_path();
223
224 if (internal_config.no_shconf)
225 return;
226
227 if (mem_cfg_fd < 0){
228 mem_cfg_fd = open(pathname, O_RDWR);
229 if (mem_cfg_fd < 0)
230 rte_panic("Cannot open '%s' for rte_mem_config\n", pathname);
231 }
232
233 /* map it as read-only first */
234 mem_config = (struct rte_mem_config *) mmap(NULL, sizeof(*mem_config),
235 PROT_READ, MAP_SHARED, mem_cfg_fd, 0);
236 if (mem_config == MAP_FAILED)
237 rte_panic("Cannot mmap memory for rte_config! error %i (%s)\n",
238 errno, strerror(errno));
239
240 rte_config.mem_config = mem_config;
241}

243/* reattach the shared config at exact memory location primary process has it */
244static void
245rte_eal_config_reattach(void)
246{
247 struct rte_mem_config *mem_config;
248 void *rte_mem_cfg_addr;
249
250 if (internal_config.no_shconf)
251 return;
252
253 /* save the address primary process has mapped shared config to */
254 rte_mem_cfg_addr = (void *) (uintptr_t) rte_config.mem_config->mem_cfg_addr;
255
256 /* unmap original config */
257 munmap(rte_config.mem_config, sizeof(struct rte_mem_config));
258
259 /* remap the config at proper address */
260 mem_config = (struct rte_mem_config *) mmap(rte_mem_cfg_addr,
261 sizeof(*mem_config), PROT_READ | PROT_WRITE, MAP_SHARED,
262 mem_cfg_fd, 0);
263 if (mem_config == MAP_FAILED || mem_config != rte_mem_cfg_addr) {
264 if (mem_config != MAP_FAILED)
265 /* errno is stale, don't use */
266 rte_panic("Cannot mmap memory for rte_config at [%p], got [%p]"
267 " - please use '--base-virtaddr' option\n",
268 rte_mem_cfg_addr, mem_config);
269 else
270 rte_panic("Cannot mmap memory for rte_config! error %i (%s)\n",
271 errno, strerror(errno));
272 }
273 close(mem_cfg_fd);
274
275 rte_config.mem_config = mem_config;
276}

278/* Detect if we are a primary or a secondary process */
279enum rte_proc_type_t
280eal_proc_type_detect(void)
281{
282 enum rte_proc_type_t ptype = RTE_PROC_PRIMARY;
283 const char *pathname = eal_runtime_config_path();
284
285 /* if we can open the file but not get a write-lock we are a secondary
286 * process. NOTE: if we get a file handle back, we keep that open
287 * and don't close it to prevent a race condition between multiple opens */
288 if (((mem_cfg_fd = open(pathname, O_RDWR)) >= 0) &&
289 (fcntl(mem_cfg_fd, F_SETLK, &wr_lock) < 0))
290 ptype = RTE_PROC_SECONDARY;
291
292 RTE_LOG(INFO, EAL, "Auto-detected process type: %s\n",
293 ptype == RTE_PROC_PRIMARY ? "PRIMARY" : "SECONDARY");
294
295 return ptype;
296}

298/* Sets up rte_config structure with the pointer to shared memory config.*/
299static void
300rte_config_init(void)
301{
302 rte_config.process_type = internal_config.process_type;
303
304 switch (rte_config.process_type){
305 case RTE_PROC_PRIMARY:
306 rte_eal_config_create();
307 break;
308 case RTE_PROC_SECONDARY:
309 rte_eal_config_attach();
310 rte_eal_mcfg_wait_complete(rte_config.mem_config);
311 rte_eal_config_reattach();
312 break;
313 case RTE_PROC_AUTO:
314 case RTE_PROC_INVALID:
315 rte_panic("Invalid process type\n");
316 }
317}

319/* Unlocks hugepage directories that were locked by eal_hugepage_info_init */
320static void
321eal_hugedirs_unlock(void)
322{
323 int i;
324
325 for (i = 0; i < MAX_HUGEPAGE_SIZES; i++)
326 {
327 /* skip uninitialized */
328 if (internal_config.hugepage_info[i].lock_descriptor < 0)
329 continue;
330 /* unlock hugepage file */
331 flock(internal_config.hugepage_info[i].lock_descriptor, LOCK_UN);
332 close(internal_config.hugepage_info[i].lock_descriptor);
333 /* reset the field */
334 internal_config.hugepage_info[i].lock_descriptor = -1;
335 }
336}

372static int
373eal_parse_socket_mem(char *socket_mem)
374{
375 char * arg[RTE_MAX_NUMA_NODES];
376 char *end;
377 int arg_num, i, len;
378 uint64_t total_mem = 0;
379
380 len = strnlen(socket_mem, SOCKET_MEM_STRLEN);
381 if (len == SOCKET_MEM_STRLEN) {
382 RTE_LOG(ERR, EAL, "--socket-mem is too long\n");
383 return -1;
384 }
385
386 /* all other error cases will be caught later */
387 if (!isdigit(socket_mem[len-1]))
388 return -1;
389
390 /* split the optarg into separate socket values */
391 arg_num = rte_strsplit(socket_mem, len,
392 arg, RTE_MAX_NUMA_NODES, ',');
393
394 /* if split failed, or 0 arguments */
395 if (arg_num <= 0)
396 return -1;
397
398 internal_config.force_sockets = 1;
399
400 /* parse each defined socket option */
401 errno = 0;
402 for (i = 0; i < arg_num; i++) {
403 end = NULL;
404 internal_config.socket_mem[i] = strtoull(arg[i], &end, 10);
405
406 /* check for invalid input */
407 if ((errno != 0) ||
408 (arg[i][0] == '\0') || (end == NULL) || (*end != '\0'))
409 return -1;
410 internal_config.socket_mem[i] *= 1024ULL;
411 internal_config.socket_mem[i] *= 1024ULL;
412 total_mem += internal_config.socket_mem[i];
413 }
414
415 /* check if we have a positive amount of total memory */
416 if (total_mem == 0)
417 return -1;
418
419 return 0;
420}

422static int
423eal_parse_base_virtaddr(const char *arg)
424{
425 char *end;
426 uint64_t addr;
427
428 errno = 0;
429 addr = strtoull(arg, &end, 16);
430
431 /* check for errors */
432 if ((errno != 0) || (arg[0] == '\0') || end == NULL || (*end != '\0'))
433 return -1;
434
435 /* make sure we don't exceed 32-bit boundary on 32-bit target */
436#ifndef RTE_ARCH_64
437 if (addr >= UINTPTR_MAX)
438 return -1;
439#endif
440
441 /* align the addr on 16M boundary, 16MB is the minimum huge page
442 * size on IBM Power architecture. If the addr is aligned to 16MB,
443 * it can align to 2MB for x86. So this alignment can also be used
444 * on x86 */
445 internal_config.base_virtaddr =
446 RTE_PTR_ALIGN_CEIL((uintptr_t)addr, (size_t)RTE_PGSIZE_16M);
447
448 return 0;
449}

467 return 0;
468 }
469 }
470 return -1;
471}

473/* Parse the arguments for --log-level only */
474static void
475eal_log_level_parse(int argc, char **argv)
476{
477 int opt;
478 char **argvopt;
479 int option_index;
480 const int old_optind = optind;
481 const int old_optopt = optopt;
482 char * const old_optarg = optarg;
483
484 argvopt = argv;
485 optind = 1;
486
487 while ((opt = getopt_long(argc, argvopt, eal_short_options,
488 eal_long_options, &option_index)) != EOF) {
489
490 int ret;
491
492 /* getopt is not happy, stop right now */
493 if (opt == '?')
494 break;
495
496 ret = (opt == OPT_LOG_LEVEL_NUM) ?
497 eal_parse_common_option(opt, optarg, &internal_config) : 0;
498
499 /* common parser is not happy */
500 if (ret < 0)
501 break;
502 }
503
504 /* restore getopt lib */
505 optind = old_optind;
506 optopt = old_optopt;
507 optarg = old_optarg;
508}

510/* Parse the argument given in the command line of the application */
511static int
512eal_parse_args(int argc, char **argv)
513{
514 int opt, ret;
515 char **argvopt;
516 int option_index;
517 char *prgname = argv[0];
518 const int old_optind = optind;
519 const int old_optopt = optopt;
520 char * const old_optarg = optarg;
521
522 argvopt = argv;
523 optind = 1;
524
525 while ((opt = getopt_long(argc, argvopt, eal_short_options,
526 eal_long_options, &option_index)) != EOF) {
527
528 /* getopt is not happy, stop right now */
529 if (opt == '?') {
530 eal_usage(prgname);
531 ret = -1;
532 goto out;
533 }
534
535 ret = eal_parse_common_option(opt, optarg, &internal_config);
536 /* common parser is not happy */
537 if (ret < 0) {
538 eal_usage(prgname);
539 ret = -1;
540 goto out;
541 }
542 /* common parser handled this option */
543 if (ret == 0)
544 continue;
545
546 switch (opt) {
547 case 'h':
548 eal_usage(prgname);
549 exit(EXIT_SUCCESS);
550
551 case OPT_HUGE_DIR_NUM:
552 internal_config.hugepage_dir = strdup(optarg);
553 break;
554
555 case OPT_FILE_PREFIX_NUM:
556 internal_config.hugefile_prefix = strdup(optarg);
557 break;
558
559 case OPT_SOCKET_MEM_NUM:
560 if (eal_parse_socket_mem(optarg) < 0) {
561 RTE_LOG(ERR, EAL, "invalid parameters for --"
562 OPT_SOCKET_MEM "\n");
563 eal_usage(prgname);
564 ret = -1;
565 goto out;
566 }
567 break;
568
569 case OPT_BASE_VIRTADDR_NUM:
570 if (eal_parse_base_virtaddr(optarg) < 0) {
571 RTE_LOG(ERR, EAL, "invalid parameter for --"
572 OPT_BASE_VIRTADDR "\n");
573 eal_usage(prgname);
574 ret = -1;
575 goto out;
576 }
577 break;
578
579 case OPT_VFIO_INTR_NUM:
580 if (eal_parse_vfio_intr(optarg) < 0) {
581 RTE_LOG(ERR, EAL, "invalid parameters for --"
582 OPT_VFIO_INTR "\n");
583 eal_usage(prgname);
584 ret = -1;
585 goto out;
586 }
587 break;
588
589 case OPT_CREATE_UIO_DEV_NUM:
590 internal_config.create_uio_dev = 1;
591 break;
592
593 case OPT_MBUF_POOL_OPS_NAME_NUM:
594 internal_config.user_mbuf_pool_ops_name = optarg;
595 break;
596
597 default:
598 if (opt < OPT_LONG_MIN_NUM && isprint(opt)) {
599 RTE_LOG(ERR, EAL, "Option %c is not supported "
600 "on Linux\n", opt);
601 } else if (opt >= OPT_LONG_MIN_NUM &&
602 opt < OPT_LONG_MAX_NUM) {
603 RTE_LOG(ERR, EAL, "Option %s is not supported "
604 "on Linux\n",
605 eal_long_options[option_index].name);
606 } else {
607 RTE_LOG(ERR, EAL, "Option %d is not supported "
608 "on Linux\n", opt);
609 }
610 eal_usage(prgname);
611 ret = -1;
612 goto out;
613 }
614 }
615
616 if (eal_adjust_config(&internal_config) != 0) {
617 ret = -1;
618 goto out;
619 }
620
621 /* sanity checks */
622 if (eal_check_common_options(&internal_config) != 0) {
623 eal_usage(prgname);
624 ret = -1;
625 goto out;
626 }
627
628 if (optind >= 0)
629 argv[optind-1] = prgname;
630 ret = optind-1;
631
632out:
633 /* restore getopt lib */
634 optind = old_optind;
635 optopt = old_optopt;
636 optarg = old_optarg;
637
638 return ret;
639}

642eal_check_mem_on_local_socket(void)
643{

691{

707 vfio_mp_sync_setup() < 0)
708 return -1;
709 }
710
711 return 0;

721/* Launch threads, called at application init(). */
722int
723rte_eal_init(int argc, char **argv)
724{
725 int i, fctret, ret;
726 pthread_t thread_id;
727 static rte_atomic32_t run_once = RTE_ATOMIC32_INIT(0);
728 const char *logid;
729 char cpuset[RTE_CPU_AFFINITY_STR_LEN];
730 char thread_name[RTE_MAX_THREAD_NAME_LEN];
731
732 /* checks if the machine is adequate */
733 if (!rte_cpu_is_supported()) {
734 rte_eal_init_alert("unsupported cpu type.");
735 rte_errno = ENOTSUP;
736 return -1;
737 }
738
739 if (!rte_atomic32_test_and_set(&run_once)) {
740 rte_eal_init_alert("already called initialization.");
741 rte_errno = EALREADY;
742 return -1;
743 }
744
745 logid = strrchr(argv[0], '/');
746 logid = strdup(logid ? logid + 1: argv[0]);
747
748 thread_id = pthread_self();
749
750 eal_reset_internal_config(&internal_config);
751
752 /* set log level as early as possible */
753 eal_log_level_parse(argc, argv);
754
755 if (rte_eal_cpu_init() < 0) {
756 rte_eal_init_alert("Cannot detect lcores.");
757 rte_errno = ENOTSUP;
758 return -1;
759 }
760
761 fctret = eal_parse_args(argc, argv);
762 if (fctret < 0) {
763 rte_eal_init_alert("Invalid 'command line' arguments.");
764 rte_errno = EINVAL;
765 rte_atomic32_clear(&run_once);
766 return -1;
767 }
768
769 if (eal_plugins_init() < 0) {
770 rte_eal_init_alert("Cannot init plugins\n");
771 rte_errno = EINVAL;
772 rte_atomic32_clear(&run_once);
773 return -1;
774 }
775
776 if (eal_option_device_parse()) {
777 rte_errno = ENODEV;
778 rte_atomic32_clear(&run_once);
779 return -1;
780 }
781
782 if (rte_bus_scan()) {
783 rte_eal_init_alert("Cannot scan the buses for devices\n");
784 rte_errno = ENODEV;
785 rte_atomic32_clear(&run_once);
786 return -1;
787 }
788
789 /* autodetect the iova mapping mode (default is iova_pa) */
790 rte_eal_get_configuration()->iova_mode = rte_bus_get_iommu_class();
791
792 /* Workaround for KNI which requires physical address to work */
793 if (rte_eal_get_configuration()->iova_mode == RTE_IOVA_VA &&
794 rte_eal_check_module("rte_kni") == 1) {
795 rte_eal_get_configuration()->iova_mode = RTE_IOVA_PA;
796 RTE_LOG(WARNING, EAL,
797 "Some devices want IOVA as VA but PA will be used because.. "
798 "KNI module inserted\n");
799 }
800
801 if (internal_config.no_hugetlbfs == 0 &&
802 internal_config.process_type != RTE_PROC_SECONDARY &&
803 eal_hugepage_info_init() < 0) {
804 rte_eal_init_alert("Cannot get hugepage information.");
805 rte_errno = EACCES;
806 rte_atomic32_clear(&run_once);
807 return -1;
808 }
809
810 if (internal_config.memory == 0 && internal_config.force_sockets == 0) {
811 if (internal_config.no_hugetlbfs)
812 internal_config.memory = MEMSIZE_IF_NO_HUGE_PAGE;
813 }
814
815 if (internal_config.vmware_tsc_map == 1) {
816#ifdef RTE_LIBRTE_EAL_VMWARE_TSC_MAP_SUPPORT
817 rte_cycles_vmware_tsc_map = 1;
818 RTE_LOG (DEBUG, EAL, "Using VMWARE TSC MAP, "
819 "you must have monitor_control.pseudo_perfctr = TRUE\n");
820#else
821 RTE_LOG (WARNING, EAL, "Ignoring --vmware-tsc-map because "
822 "RTE_LIBRTE_EAL_VMWARE_TSC_MAP_SUPPORT is not set\n");
823#endif
824 }
825
826 rte_srand(rte_rdtsc());
827
828 rte_config_init();
829
830 if (rte_eal_log_init(logid, internal_config.syslog_facility) < 0) {
831 rte_eal_init_alert("Cannot init logging.");
832 rte_errno = ENOMEM;
833 rte_atomic32_clear(&run_once);
834 return -1;
835 }
836
837 if (rte_mp_channel_init() < 0) {
838 rte_eal_init_alert("failed to init mp channel\n");
839 if (rte_eal_process_type() == RTE_PROC_PRIMARY) {
840 rte_errno = EFAULT;
841 return -1;
842 }
843 }
844
845#ifdef VFIO_PRESENT
846 if (rte_eal_vfio_setup() < 0) {
847 rte_eal_init_alert("Cannot init VFIO\n");
848 rte_errno = EAGAIN;
849 rte_atomic32_clear(&run_once);
850 return -1;
851 }
852#endif
853
854 if (rte_eal_memory_init() < 0) {
855 rte_eal_init_alert("Cannot init memory\n");
856 rte_errno = ENOMEM;
857 return -1;
858 }
859
860 /* the directories are locked during eal_hugepage_info_init */
861 eal_hugedirs_unlock();
862
863 if (rte_eal_memzone_init() < 0) {
864 rte_eal_init_alert("Cannot init memzone\n");
865 rte_errno = ENODEV;
866 return -1;
867 }
868
869 if (rte_eal_tailqs_init() < 0) {
870 rte_eal_init_alert("Cannot init tail queues for objects\n");
871 rte_errno = EFAULT;
872 return -1;
873 }
874
875 if (rte_eal_alarm_init() < 0) {
876 rte_eal_init_alert("Cannot init interrupt-handling thread\n");
877 /* rte_eal_alarm_init sets rte_errno on failure. */
878 return -1;
879 }
880
881 if (rte_eal_timer_init() < 0) {
882 rte_eal_init_alert("Cannot init HPET or TSC timers\n");
883 rte_errno = ENOTSUP;
884 return -1;
885 }
886
887 eal_check_mem_on_local_socket();
888
889 eal_thread_init_master(rte_config.master_lcore);
890
891 ret = eal_thread_dump_affinity(cpuset, RTE_CPU_AFFINITY_STR_LEN);
892
893 RTE_LOG(DEBUG, EAL, "Master lcore %u is ready (tid=%x;cpuset=[%s%s])\n",
894 rte_config.master_lcore, (int)thread_id, cpuset,
895 ret == 0 ? "" : "...");
896
897 if (rte_eal_intr_init() < 0) {
898 rte_eal_init_alert("Cannot init interrupt-handling thread\n");
899 return -1;
900 }
901
902 RTE_LCORE_FOREACH_SLAVE(i) {
903
904 /*
905 * create communication pipes between master thread
906 * and children
907 */
908 if (pipe(lcore_config[i].pipe_master2slave) < 0)
909 rte_panic("Cannot create pipe\n");
910 if (pipe(lcore_config[i].pipe_slave2master) < 0)
911 rte_panic("Cannot create pipe\n");
912
913 lcore_config[i].state = WAIT;
914
915 /* create a thread for each lcore */
916 ret = pthread_create(&lcore_config[i].thread_id, NULL,
917 eal_thread_loop, NULL);
918 if (ret != 0)
919 rte_panic("Cannot create thread\n");
920
921 /* Set thread_name for aid in debugging. */
922 snprintf(thread_name, RTE_MAX_THREAD_NAME_LEN,
923 "lcore-slave-%d", i);
924 ret = rte_thread_setname(lcore_config[i].thread_id,
925 thread_name);
926 if (ret != 0)
927 RTE_LOG(DEBUG, EAL,
928 "Cannot set name for lcore thread\n");
929 }
930
931 /*
932 * Launch a dummy function on all slave lcores, so that master lcore
933 * knows they are all ready when this function returns.
934 */
935 rte_eal_mp_remote_launch(sync_func, NULL, SKIP_MASTER);
936 rte_eal_mp_wait_lcore();
937
938 /* initialize services so vdevs register service during bus_probe. */
939 ret = rte_service_init();
940 if (ret) {
941 rte_eal_init_alert("rte_service_init() failed\n");
942 rte_errno = ENOEXEC;
943 return -1;
944 }
945
946 /* Probe all the buses and devices/drivers on them */
947 if (rte_bus_probe()) {
948 rte_eal_init_alert("Cannot probe devices\n");
949 rte_errno = ENOTSUP;
950 return -1;
951 }
952
953 /* initialize default service/lcore mappings and start running. Ignore
954 * -ENOTSUP, as it indicates no service coremask passed to EAL.
955 */
956 ret = rte_service_start_with_defaults();
957 if (ret < 0 && ret != -ENOTSUP) {
958 rte_errno = ENOEXEC;
959 return -1;
960 }
961
962 rte_eal_mcfg_complete();
963
964 return fctret;
965}

971 return 0;

1008int
1009rte_eal_check_module(const char *module_name)
1010{
1011 char sysfs_mod_name[PATH_MAX];
1012 struct stat st;
1013 int n;
1014
1015 if (NULL == module_name)
1016 return -1;
1017
1018 /* Check if there is sysfs mounted */
1019 if (stat("/sys/module", &st) != 0) {
1020 RTE_LOG(DEBUG, EAL, "sysfs is not mounted! error %i (%s)\n",
1021 errno, strerror(errno));
1022 return -1;
1023 }
1024
1025 /* A module might be built-in, therefore try sysfs */
1026 n = snprintf(sysfs_mod_name, PATH_MAX, "/sys/module/%s", module_name);
1027 if (n < 0 || n > PATH_MAX) {
1028 RTE_LOG(DEBUG, EAL, "Could not format module path\n");
1029 return -1;
1030 }
1031
1032 if (stat(sysfs_mod_name, &st) != 0) {
1033 RTE_LOG(DEBUG, EAL, "Module %s not found! error %i (%s)\n",
1034 sysfs_mod_name, errno, strerror(errno));
1035 return 0;
1036 }
1037
1038 /* Module has been found */
1039 return 1;
1040}
