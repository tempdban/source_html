23/**
24 * ICMP Header
25 */
26struct icmp_hdr {
27 uint8_t icmp_type; /* ICMP packet type. */
28 uint8_t icmp_code; /* ICMP packet code. */
29 uint16_t icmp_cksum; /* ICMP packet checksum. */
30 uint16_t icmp_ident; /* ICMP packet identifier. */
31 uint16_t icmp_seq_nb; /* ICMP packet sequence number. */
32} __attribute__((__packed__));
