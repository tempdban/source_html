20#define PKO_DQ_KIND_BIT 49

24/* PKO VF register offsets from VF_BAR0 */
25#define PKO_VF_DQ_SW_XOFF(gdq) (0x000100 | (gdq) << 17)
26#define PKO_VF_DQ_WM_CTL(gdq) (0x000130 | (gdq) << 17)
27#define PKO_VF_DQ_WM_CNT(gdq) (0x000150 | (gdq) << 17)
28#define PKO_VF_DQ_FC_CONFIG (0x000160)
29#define PKO_VF_DQ_FC_STATUS(gdq) (0x000168 | (gdq) << 17)
30#define PKO_VF_DQ_OP_SEND(gdq, op) (0x001000 | (gdq) << 17 | (op) << 3)
31#define PKO_VF_DQ_OP_OPEN(gdq) (0x001100 | (gdq) << 17)
32#define PKO_VF_DQ_OP_CLOSE(gdq) (0x001200 | (gdq) << 17)
33#define PKO_VF_DQ_OP_QUERY(gdq) (0x001300 | (gdq) << 17)

35/* pko_send_hdr_s + pko_send_link */

37#define PKO_SEND_GATHER_SUBDC (0x0ull << 60)
38#define PKO_SEND_GATHER_LDTYPE(x) ((x) << 58)
39#define PKO_SEND_GATHER_GAUAR(x) ((x) << 24)
