33#ifndef _AVP_LOGS_H_
34#define _AVP_LOGS_H_
35
36#include <rte_log.h>
37
38#ifdef RTE_LIBRTE_AVP_DEBUG_RX
39#define PMD_RX_LOG(level, fmt, args...) \
40 RTE_LOG(level, PMD, "%s() rx: " fmt, __func__, ## args)
41#else
42#define PMD_RX_LOG(level, fmt, args...) do { } while (0)
43#endif
44
45#ifdef RTE_LIBRTE_AVP_DEBUG_TX
46#define PMD_TX_LOG(level, fmt, args...) \
47 RTE_LOG(level, PMD, "%s() tx: " fmt, __func__, ## args)
48#else
49#define PMD_TX_LOG(level, fmt, args...) do { } while (0)
50#endif
51
52extern int avp_logtype_driver;
53
54#define PMD_DRV_LOG(level, fmt, args...) \
55 rte_log(RTE_LOG_ ## level, avp_logtype_driver, \
56 "%s(): " fmt, __func__, ## args)
57
58#endif /* _AVP_LOGS_H_ */
