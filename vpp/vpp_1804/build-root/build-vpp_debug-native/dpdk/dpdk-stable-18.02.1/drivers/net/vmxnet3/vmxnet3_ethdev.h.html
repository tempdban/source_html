37/* RSS configuration structure - shared with device through GPA */
38typedef struct VMXNET3_RSSConf {
39 uint16_t hashType;
40 uint16_t hashFunc;
41 uint16_t hashKeySize;
42 uint16_t indTableSize;
43 uint8_t hashKey[VMXNET3_RSS_MAX_KEY_SIZE];
44 /*
45 * indTable is only element that can be changed without
46 * device quiesce-reset-update-activation cycle

48 uint8_t indTable[VMXNET3_RSS_MAX_IND_TABLE_SIZE];
49} VMXNET3_RSSConf;

112/* Config space read/writes */

114#define VMXNET3_PCI_REG(reg) rte_read32(reg)
115
116static inline uint32_t
117vmxnet3_read_addr(volatile void *addr)
118{
119 return VMXNET3_PCI_REG(addr);
120}
121
122#define VMXNET3_PCI_REG_WRITE(reg, value) rte_write32((value), (reg))
123
124#define VMXNET3_PCI_BAR0_REG_ADDR(hw, reg) \
125 ((volatile uint32_t *)((char *)(hw)->hw_addr0 + (reg)))
126#define VMXNET3_READ_BAR0_REG(hw, reg) \
127 vmxnet3_read_addr(VMXNET3_PCI_BAR0_REG_ADDR((hw), (reg)))
128#define VMXNET3_WRITE_BAR0_REG(hw, reg, value) \
129 VMXNET3_PCI_REG_WRITE(VMXNET3_PCI_BAR0_REG_ADDR((hw), (reg)), (value))
130
131#define VMXNET3_PCI_BAR1_REG_ADDR(hw, reg) \

138static inline uint8_t
139vmxnet3_get_ring_idx(struct vmxnet3_hw *hw, uint32 rqID)
140{
141 return (rqID >= hw->num_rx_queues &&
142 rqID < 2 * hw->num_rx_queues) ? 1 : 0;
143}
144
145static inline bool
146vmxnet3_rx_data_ring(struct vmxnet3_hw *hw, uint32 rqID)
147{
148 return (rqID >= 2 * hw->num_rx_queues &&
149 rqID < 3 * hw->num_rx_queues);
150}
