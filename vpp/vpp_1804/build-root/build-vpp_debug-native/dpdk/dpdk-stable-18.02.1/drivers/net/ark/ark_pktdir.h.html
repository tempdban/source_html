34#ifndef _ARK_PKTDIR_H_
35#define _ARK_PKTDIR_H_
36
37#include <stdint.h>
38
39#define ARK_PKTDIR_BASE_ADR 0xa0000
40
41typedef void *ark_pkt_dir_t;
42
43
44/* The packet director is an internal Arkville hardware module for
45 * directing packet data in non-typical flows, such as testing.
46 * This module is *not* intended for end-user manipulation, hence
47 * there is minimal documentation.
48 */
49
50/*
51 * This is an overlay structures to a memory mapped FPGA device. These
52 * structs will never be instantiated in ram memory
53 */
54struct ark_pkt_dir_regs {
55 uint32_t ctrl;
56 uint32_t status;
57 uint32_t stall_cnt;
58} __attribute__ ((packed));
59
60struct ark_pkt_dir_inst {
61 volatile struct ark_pkt_dir_regs *regs;
62};
63
64ark_pkt_dir_t ark_pktdir_init(void *base);
65void ark_pktdir_uninit(ark_pkt_dir_t handle);
66void ark_pktdir_setup(ark_pkt_dir_t handle, uint32_t v);
67uint32_t ark_pktdir_stall_cnt(ark_pkt_dir_t handle);
68uint32_t ark_pktdir_status(ark_pkt_dir_t handle);
69
70#endif
