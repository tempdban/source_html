37#define B_RX_DB(db, prod) \
38 (*(uint32_t *)db = (DB_KEY_RX | prod))

40#define BNXT_TPA_L4_SIZE(x) \
41 { \
42 typeof(x) hdr_info = (x); \
43 (((hdr_info) & 0xf8000000) ? ((hdr_info) >> 27) : 32) \
44 }
45
46#define BNXT_TPA_INNER_L3_OFF(hdr_info) \
47 (((hdr_info) >> 18) & 0x1ff)
48
49#define BNXT_TPA_INNER_L2_OFF(hdr_info) \
50 (((hdr_info) >> 9) & 0x1ff)
51
52#define BNXT_TPA_OUTER_L3_OFF(hdr_info) \
53 ((hdr_info) & 0x1ff)
54
55#define RX_CMP_L4_CS_BITS rte_cpu_to_le_32(RX_PKT_CMPL_FLAGS2_L4_CS_CALC)
56
57#define RX_CMP_L4_CS_ERR_BITS rte_cpu_to_le_32(RX_PKT_CMPL_ERRORS_L4_CS_ERROR)
58
59#define RX_CMP_L4_CS_OK(rxcmp1) \
60 (((rxcmp1)->flags2 & RX_CMP_L4_CS_BITS) && \
61 !((rxcmp1)->errors_v2 & RX_CMP_L4_CS_ERR_BITS))
62
63#define RX_CMP_IP_CS_ERR_BITS rte_cpu_to_le_32(RX_PKT_CMPL_ERRORS_IP_CS_ERROR)
64
65#define RX_CMP_IP_CS_BITS rte_cpu_to_le_32(RX_PKT_CMPL_FLAGS2_IP_CS_CALC)
66
67#define RX_CMP_IP_CS_OK(rxcmp1) \
68 (((rxcmp1)->flags2 & RX_CMP_IP_CS_BITS) && \
69 !((rxcmp1)->errors_v2 & RX_CMP_IP_CS_ERR_BITS))

71enum pkt_hash_types {
72 PKT_HASH_TYPE_NONE, /* Undefined type */
73 PKT_HASH_TYPE_L2, /* Input: src_MAC, dest_MAC */
74 PKT_HASH_TYPE_L3, /* Input: src_IP, dst_IP */
75 PKT_HASH_TYPE_L4, /* Input: src_IP, dst_IP, src_port, dst_port */
76};
77
78struct bnxt_tpa_info {
79 struct rte_mbuf *mbuf;
80 uint16_t len;
81 unsigned short gso_type;
82 uint32_t flags2;
83 uint32_t metadata;
84 enum pkt_hash_types hash_type;
85 uint32_t rss_hash;
86 uint32_t hdr_info;
87};
88
89struct bnxt_sw_rx_bd {
90 struct rte_mbuf *mbuf; /* data associated with RX descriptor */
91};
92
93struct bnxt_rx_ring_info {
94 uint16_t rx_prod;
95 uint16_t ag_prod;
96 void *rx_doorbell;
97 void *ag_doorbell;
98
99 struct rx_prod_pkt_bd *rx_desc_ring;
100 struct rx_prod_pkt_bd *ag_desc_ring;
101 struct bnxt_sw_rx_bd *rx_buf_ring; /* sw ring */
102 struct bnxt_sw_rx_bd *ag_buf_ring; /* sw ring */
103
104 rte_iova_t rx_desc_mapping;
105 rte_iova_t ag_desc_mapping;
106
107 struct bnxt_ring *rx_ring_struct;
108 struct bnxt_ring *ag_ring_struct;
109
110 /*
111 * To deal with out of order return from TPA, use free buffer indicator
112 */
113 struct rte_bitmap *ag_bitmap;
114
115 struct bnxt_tpa_info *tpa_info;
116};
