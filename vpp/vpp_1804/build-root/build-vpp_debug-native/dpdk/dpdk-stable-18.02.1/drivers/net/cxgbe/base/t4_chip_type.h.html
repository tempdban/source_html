34#ifndef __T4_CHIP_TYPE_H__
35#define __T4_CHIP_TYPE_H__
36
37/*
38 * All T4 and later chips have their PCI-E Device IDs encoded as 0xVFPP where:
39 *
40 * V = "4" for T4; "5" for T5, etc. or
41 * F = "0" for PF 0..3; "4".."7" for PF4..7; and "8" for VFs
42 * PP = adapter product designation
43 *
44 * We use the "version" (V) of the adpater to code the Chip Version above.
45 */
46#define CHELSIO_PCI_ID_VER(devid) ((devid) >> 12)
47#define CHELSIO_PCI_ID_FUNC(devid) (((devid) >> 8) & 0xf)
48#define CHELSIO_PCI_ID_PROD(devid) ((devid) & 0xff)
49
50#define CHELSIO_T4 0x4
51#define CHELSIO_T5 0x5
52#define CHELSIO_T6 0x6
53
54#define CHELSIO_CHIP_CODE(version, revision) (((version) << 4) | (revision))
55#define CHELSIO_CHIP_VERSION(code) (((code) >> 4) & 0xf)
56#define CHELSIO_CHIP_RELEASE(code) ((code) & 0xf)
57
58enum chip_type {
59 T4_A1 = CHELSIO_CHIP_CODE(CHELSIO_T4, 1),
60 T4_A2 = CHELSIO_CHIP_CODE(CHELSIO_T4, 2),
61 T4_FIRST_REV = T4_A1,
62 T4_LAST_REV = T4_A2,
63
64 T5_A0 = CHELSIO_CHIP_CODE(CHELSIO_T5, 0),
65 T5_A1 = CHELSIO_CHIP_CODE(CHELSIO_T5, 1),
66 T5_FIRST_REV = T5_A0,
67 T5_LAST_REV = T5_A1,
68
69 T6_A0 = CHELSIO_CHIP_CODE(CHELSIO_T6, 0),
70 T6_FIRST_REV = T6_A0,
71 T6_LAST_REV = T6_A0,
72};
73
74static inline int is_t4(enum chip_type chip)
75{
76 return (CHELSIO_CHIP_VERSION(chip) == CHELSIO_T4);
77}
78
79static inline int is_t5(enum chip_type chip)
80{
81 return (CHELSIO_CHIP_VERSION(chip) == CHELSIO_T5);
82}
83
84static inline int is_t6(enum chip_type chip)
85{
86 return (CHELSIO_CHIP_VERSION(chip) == CHELSIO_T6);
87}
88#endif /* __T4_CHIP_TYPE_H__ */
