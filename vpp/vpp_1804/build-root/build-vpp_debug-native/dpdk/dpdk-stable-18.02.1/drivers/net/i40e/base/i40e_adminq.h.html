46struct i40e_adminq_ring {
47 struct i40e_virt_mem dma_head; /* space for dma structures */
48 struct i40e_dma_mem desc_buf; /* descriptor ring memory */
49 struct i40e_virt_mem cmd_buf; /* command buffer memory */
50
51 union {
52 struct i40e_dma_mem *asq_bi;
53 struct i40e_dma_mem *arq_bi;
54 } r;
55
56 u16 count; /* Number of descriptors */
57 u16 rx_buf_len; /* Admin Receive Queue buffer length */
58
59 /* used for interrupt processing */
60 u16 next_to_use;
61 u16 next_to_clean;
62
63 /* used for queue tracking */
64 u32 head;
65 u32 tail;
66 u32 len;
67 u32 bah;
68 u32 bal;
69};

93/* Admin Queue information */
94struct i40e_adminq_info {
95 struct i40e_adminq_ring arq; /* receive queue */
96 struct i40e_adminq_ring asq; /* send queue */
97 u32 asq_cmd_timeout; /* send queue cmd write back timeout*/
98 u16 num_arq_entries; /* receive queue depth */
99 u16 num_asq_entries; /* send queue depth */
100 u16 arq_buf_size; /* receive queue buffer size */
101 u16 asq_buf_size; /* send queue buffer size */
102 u16 fw_maj_ver; /* firmware major version */
103 u16 fw_min_ver; /* firmware minor version */
104 u32 fw_build; /* firmware build number */
105 u16 api_maj_ver; /* api major version */
106 u16 api_min_ver; /* api minor version */
107
108 struct i40e_spinlock asq_spinlock; /* Send queue spinlock */
109 struct i40e_spinlock arq_spinlock; /* Receive queue spinlock */
110
111 /* last status values on send and receive queues */
112 enum i40e_admin_queue_err asq_last_status;
113 enum i40e_admin_queue_err arq_last_status;
114};

116/**
117 * i40e_aq_rc_to_posix - convert errors to user-land codes
118 * aq_ret: AdminQ handler error code can override aq_rc
119 * aq_rc: AdminQ firmware error code to convert
120 **/
121STATIC INLINE int i40e_aq_rc_to_posix(int aq_ret, int aq_rc)
122{
123 int aq_to_posix[] = {
124 0, /* I40E_AQ_RC_OK */
125 -EPERM, /* I40E_AQ_RC_EPERM */
126 -ENOENT, /* I40E_AQ_RC_ENOENT */
127 -ESRCH, /* I40E_AQ_RC_ESRCH */
128 -EINTR, /* I40E_AQ_RC_EINTR */
129 -EIO, /* I40E_AQ_RC_EIO */
130 -ENXIO, /* I40E_AQ_RC_ENXIO */
131 -E2BIG, /* I40E_AQ_RC_E2BIG */
132 -EAGAIN, /* I40E_AQ_RC_EAGAIN */
133 -ENOMEM, /* I40E_AQ_RC_ENOMEM */
134 -EACCES, /* I40E_AQ_RC_EACCES */
135 -EFAULT, /* I40E_AQ_RC_EFAULT */
136 -EBUSY, /* I40E_AQ_RC_EBUSY */
137 -EEXIST, /* I40E_AQ_RC_EEXIST */
138 -EINVAL, /* I40E_AQ_RC_EINVAL */
139 -ENOTTY, /* I40E_AQ_RC_ENOTTY */
140 -ENOSPC, /* I40E_AQ_RC_ENOSPC */
141 -ENOSYS, /* I40E_AQ_RC_ENOSYS */
142 -ERANGE, /* I40E_AQ_RC_ERANGE */
143 -EPIPE, /* I40E_AQ_RC_EFLUSHED */
144 -ESPIPE, /* I40E_AQ_RC_BAD_ADDR */
145 -EROFS, /* I40E_AQ_RC_EMODE */
146 -EFBIG, /* I40E_AQ_RC_EFBIG */
147 };
148
149 /* aq_rc is invalid if AQ timed out */
150 if (aq_ret == I40E_ERR_ADMIN_QUEUE_TIMEOUT)
151 return -EAGAIN;
152
153 if (!((u32)aq_rc < (sizeof(aq_to_posix) / sizeof((aq_to_posix)[0]))))
154 return -ERANGE;
155
156 return aq_to_posix[aq_rc];
157}
