14 __checkReturn efx_rc_t
15siena_nvram_partn_size(
16 __in efx_nic_t *enp,
17 __in uint32_t partn,
18 __out size_t *sizep)
19{
20 efx_rc_t rc;
21
22 if ((1 << partn) & ~enp->en_u.siena.enu_partn_mask) {
23 rc = ENOTSUP;
24 goto fail1;
25 }
26
27 if ((rc = efx_mcdi_nvram_info(enp, partn, sizep,
28 NULL, NULL, NULL)) != 0) {
29 goto fail2;
30 }
31
32 return (0);
33
34fail2:
35 EFSYS_PROBE(fail2);
36fail1:
37 EFSYS_PROBE1(fail1, efx_rc_t, rc);
38
39 return (rc);
40}

61 __checkReturn efx_rc_t
62siena_nvram_partn_read(
63 __in efx_nic_t *enp,
64 __in uint32_t partn,
65 __in unsigned int offset,
66 __out_bcount(size) caddr_t data,
67 __in size_t size)
68{
69 size_t chunk;
70 efx_rc_t rc;
71
72 while (size > 0) {
73 chunk = MIN(size, SIENA_NVRAM_CHUNK);
74
75 if ((rc = efx_mcdi_nvram_read(enp, partn, offset, data, chunk,
76 MC_CMD_NVRAM_READ_IN_V2_DEFAULT)) != 0) {
77 goto fail1;
78 }
79
80 size -= chunk;
81 data += chunk;
82 offset += chunk;
83 }
84
85 return (0);
86
87fail1:
88 EFSYS_PROBE1(fail1, efx_rc_t, rc);
89
90 return (rc);
91}

146 __checkReturn efx_rc_t
147siena_nvram_partn_unlock(
148 __in efx_nic_t *enp,
149 __in uint32_t partn,
150 __out_opt uint32_t *verify_resultp)
151{
152 boolean_t reboot;
153 efx_rc_t rc;
154
155 /*
156 * Reboot into the new image only for PHYs. The driver has to
157 * explicitly cope with an MC reboot after a firmware update.
158 */
159 reboot = (partn == MC_CMD_NVRAM_TYPE_PHY_PORT0 ||
160 partn == MC_CMD_NVRAM_TYPE_PHY_PORT1 ||
161 partn == MC_CMD_NVRAM_TYPE_DISABLED_CALLISTO);
162
163 rc = efx_mcdi_nvram_update_finish(enp, partn, reboot, verify_resultp);
164 if (rc != 0)
165 goto fail1;
166
167 return (0);
168
169fail1:
170 EFSYS_PROBE1(fail1, efx_rc_t, rc);
171
172 return (rc);
173}

210 __checkReturn efx_rc_t
211siena_nvram_type_to_partn(
212 __in efx_nic_t *enp,
213 __in efx_nvram_type_t type,
214 __out uint32_t *partnp)
215{
216 efx_mcdi_iface_t *emip = &(enp->en_mcdi.em_emip);
217 unsigned int i;
218
219 EFSYS_ASSERT3U(type, !=, EFX_NVRAM_INVALID);
220 EFSYS_ASSERT3U(type, <, EFX_NVRAM_NTYPES);
221 EFSYS_ASSERT(partnp != NULL);
222
223 for (i = 0; i < EFX_ARRAY_SIZE(siena_parttbl); i++) {
224 siena_parttbl_entry_t *entry = &siena_parttbl[i];
225
226 if (entry->port == emip->emi_port && entry->nvtype == type) {
227 *partnp = entry->partn;
228 return (0);
229 }
230 }
231
232 return (ENOTSUP);
233}

274#define SIENA_DYNAMIC_CFG_SIZE(_nitems) \
275 (sizeof (siena_mc_dynamic_config_hdr_t) + ((_nitems) * \
276 sizeof (((siena_mc_dynamic_config_hdr_t *)NULL)->fw_version[0])))
277
278 __checkReturn efx_rc_t
279siena_nvram_get_dynamic_cfg(
280 __in efx_nic_t *enp,
281 __in uint32_t partn,
282 __in boolean_t vpd,
283 __out siena_mc_dynamic_config_hdr_t **dcfgp,
284 __out size_t *sizep)
285{
286 siena_mc_dynamic_config_hdr_t *dcfg = NULL;
287 size_t size;
288 uint8_t cksum;
289 unsigned int vpd_offset;
290 unsigned int vpd_length;
291 unsigned int hdr_length;
292 unsigned int nversions;
293 unsigned int pos;
294 unsigned int region;
295 efx_rc_t rc;
296
297 EFSYS_ASSERT(partn == MC_CMD_NVRAM_TYPE_DYNAMIC_CFG_PORT0 ||
298 partn == MC_CMD_NVRAM_TYPE_DYNAMIC_CFG_PORT1);
299
300 /*
301 * Allocate sufficient memory for the entire dynamiccfg area, even
302 * if we're not actually going to read in the VPD.
303 */
304 if ((rc = siena_nvram_partn_size(enp, partn, &size)) != 0)
305 goto fail1;
306
307 EFSYS_KMEM_ALLOC(enp->en_esip, size, dcfg);
308 if (dcfg == NULL) {
309 rc = ENOMEM;
310 goto fail2;
311 }
312
313 if ((rc = siena_nvram_partn_read(enp, partn, 0,
314 (caddr_t)dcfg, SIENA_NVRAM_CHUNK)) != 0)
315 goto fail3;
316
317 /* Verify the magic */
318 if (EFX_DWORD_FIELD(dcfg->magic, EFX_DWORD_0)
319 != SIENA_MC_DYNAMIC_CONFIG_MAGIC)
320 goto invalid1;
321
322 /* All future versions of the structure must be backwards compatible */
323 EFX_STATIC_ASSERT(SIENA_MC_DYNAMIC_CONFIG_VERSION == 0);
324
325 hdr_length = EFX_WORD_FIELD(dcfg->length, EFX_WORD_0);
326 nversions = EFX_DWORD_FIELD(dcfg->num_fw_version_items, EFX_DWORD_0);
327 vpd_offset = EFX_DWORD_FIELD(dcfg->dynamic_vpd_offset, EFX_DWORD_0);
328 vpd_length = EFX_DWORD_FIELD(dcfg->dynamic_vpd_length, EFX_DWORD_0);
329
330 /* Verify the hdr doesn't overflow the partn size */
331 if (hdr_length > size || vpd_offset > size || vpd_length > size ||
332 vpd_length + vpd_offset > size)
333 goto invalid2;
334
335 /* Verify the header has room for all it's versions */
336 if (hdr_length < SIENA_DYNAMIC_CFG_SIZE(0) ||
337 hdr_length < SIENA_DYNAMIC_CFG_SIZE(nversions))
338 goto invalid3;
339
340 /*
341 * Read the remaining portion of the dcfg, either including
342 * the whole of VPD (there is no vpd length in this structure,
343 * so we have to parse each tag), or just the dcfg header itself
344 */
345 region = vpd ? vpd_offset + vpd_length : hdr_length;
346 if (region > SIENA_NVRAM_CHUNK) {
347 if ((rc = siena_nvram_partn_read(enp, partn, SIENA_NVRAM_CHUNK,
348 (caddr_t)dcfg + SIENA_NVRAM_CHUNK,
349 region - SIENA_NVRAM_CHUNK)) != 0)
350 goto fail4;
351 }
352
353 /* Verify checksum */
354 cksum = 0;
355 for (pos = 0; pos < hdr_length; pos++)
356 cksum += ((uint8_t *)dcfg)[pos];
357 if (cksum != 0)
358 goto invalid4;
359
360 goto done;
361
362invalid4:
363 EFSYS_PROBE(invalid4);
364invalid3:
365 EFSYS_PROBE(invalid3);
366invalid2:
367 EFSYS_PROBE(invalid2);
368invalid1:
369 EFSYS_PROBE(invalid1);
370
371 /*
372 * Construct a new "null" dcfg, with an empty version vector,
373 * and an empty VPD chunk trailing. This has the neat side effect
374 * of testing the exception paths in the write path.
375 */
376 EFX_POPULATE_DWORD_1(dcfg->magic,
377 EFX_DWORD_0, SIENA_MC_DYNAMIC_CONFIG_MAGIC);
378 EFX_POPULATE_WORD_1(dcfg->length, EFX_WORD_0, sizeof (*dcfg));
379 EFX_POPULATE_BYTE_1(dcfg->version, EFX_BYTE_0,
380 SIENA_MC_DYNAMIC_CONFIG_VERSION);
381 EFX_POPULATE_DWORD_1(dcfg->dynamic_vpd_offset,
382 EFX_DWORD_0, sizeof (*dcfg));
383 EFX_POPULATE_DWORD_1(dcfg->dynamic_vpd_length, EFX_DWORD_0, 0);
384 EFX_POPULATE_DWORD_1(dcfg->num_fw_version_items, EFX_DWORD_0, 0);
385
386done:
387 *dcfgp = dcfg;
388 *sizep = size;
389
390 return (0);
391
392fail4:
393 EFSYS_PROBE(fail4);
394fail3:
395 EFSYS_PROBE(fail3);
396
397 EFSYS_KMEM_FREE(enp->en_esip, size, dcfg);
398
399fail2:
400 EFSYS_PROBE(fail2);
401fail1:
402 EFSYS_PROBE1(fail1, efx_rc_t, rc);
403
404 return (rc);
405}

407 __checkReturn efx_rc_t
408siena_nvram_get_subtype(
409 __in efx_nic_t *enp,
410 __in uint32_t partn,
411 __out uint32_t *subtypep)
412{
413 efx_mcdi_req_t req;
414 uint8_t payload[MAX(MC_CMD_GET_BOARD_CFG_IN_LEN,
415 MC_CMD_GET_BOARD_CFG_OUT_LENMAX)];
416 efx_word_t *fw_list;
417 efx_rc_t rc;
418
419 (void) memset(payload, 0, sizeof (payload));
420 req.emr_cmd = MC_CMD_GET_BOARD_CFG;
421 req.emr_in_buf = payload;
422 req.emr_in_length = MC_CMD_GET_BOARD_CFG_IN_LEN;
423 req.emr_out_buf = payload;
424 req.emr_out_length = MC_CMD_GET_BOARD_CFG_OUT_LENMAX;
425
426 efx_mcdi_execute(enp, &req);
427
428 if (req.emr_rc != 0) {
429 rc = req.emr_rc;
430 goto fail1;
431 }
432
433 if (req.emr_out_length_used < MC_CMD_GET_BOARD_CFG_OUT_LENMIN) {
434 rc = EMSGSIZE;
435 goto fail2;
436 }
437
438 if (req.emr_out_length_used <
439 MC_CMD_GET_BOARD_CFG_OUT_FW_SUBTYPE_LIST_OFST +
440 (partn + 1) * sizeof (efx_word_t)) {
441 rc = ENOENT;
442 goto fail3;
443 }
444
445 fw_list = MCDI_OUT2(req, efx_word_t,
446 GET_BOARD_CFG_OUT_FW_SUBTYPE_LIST);
447 *subtypep = EFX_WORD_FIELD(fw_list[partn], EFX_WORD_0);
448
449 return (0);
450
451fail3:
452 EFSYS_PROBE(fail3);
453fail2:
454 EFSYS_PROBE(fail2);
455fail1:
456 EFSYS_PROBE1(fail1, efx_rc_t, rc);
457
458 return (rc);
459}

461 __checkReturn efx_rc_t
462siena_nvram_partn_get_version(
463 __in efx_nic_t *enp,
464 __in uint32_t partn,
465 __out uint32_t *subtypep,
466 __out_ecount(4) uint16_t version[4])
467{
468 siena_mc_dynamic_config_hdr_t *dcfg;
469 siena_parttbl_entry_t *entry;
470 uint32_t dcfg_partn;
471 unsigned int i;
472 efx_rc_t rc;
473
474 if ((1 << partn) & ~enp->en_u.siena.enu_partn_mask) {
475 rc = ENOTSUP;
476 goto fail1;
477 }
478
479 if ((rc = siena_nvram_get_subtype(enp, partn, subtypep)) != 0)
480 goto fail2;
481
482 /*
483 * Some partitions are accessible from both ports (for instance BOOTROM)
484 * Find the highest version reported by all dcfg structures on ports
485 * that have access to this partition.
486 */
487 version[0] = version[1] = version[2] = version[3] = 0;
488 for (i = 0; i < EFX_ARRAY_SIZE(siena_parttbl); i++) {
489 siena_mc_fw_version_t *verp;
490 unsigned int nitems;
491 uint16_t temp[4];
492 size_t length;
493
494 entry = &siena_parttbl[i];
495 if (entry->partn != partn)
496 continue;
497
498 dcfg_partn = (entry->port == 1)
499 ? MC_CMD_NVRAM_TYPE_DYNAMIC_CFG_PORT0
500 : MC_CMD_NVRAM_TYPE_DYNAMIC_CFG_PORT1;
501 /*
502 * Ingore missing partitions on port 2, assuming they're due
503 * to to running on a single port part.
504 */
505 if ((1 << dcfg_partn) & ~enp->en_u.siena.enu_partn_mask) {
506 if (entry->port == 2)
507 continue;
508 }
509
510 if ((rc = siena_nvram_get_dynamic_cfg(enp, dcfg_partn,
511 B_FALSE, &dcfg, &length)) != 0)
512 goto fail3;
513
514 nitems = EFX_DWORD_FIELD(dcfg->num_fw_version_items,
515 EFX_DWORD_0);
516 if (nitems < entry->partn)
517 goto done;
518
519 verp = &dcfg->fw_version[partn];
520 temp[0] = EFX_WORD_FIELD(verp->version_w, EFX_WORD_0);
521 temp[1] = EFX_WORD_FIELD(verp->version_x, EFX_WORD_0);
522 temp[2] = EFX_WORD_FIELD(verp->version_y, EFX_WORD_0);
523 temp[3] = EFX_WORD_FIELD(verp->version_z, EFX_WORD_0);
524 if (memcmp(version, temp, sizeof (temp)) < 0)
525 memcpy(version, temp, sizeof (temp));
526
527done:
528 EFSYS_KMEM_FREE(enp->en_esip, length, dcfg);
529 }
530
531 return (0);
532
533fail3:
534 EFSYS_PROBE(fail3);
535fail2:
536 EFSYS_PROBE(fail2);
537fail1:
538 EFSYS_PROBE1(fail1, efx_rc_t, rc);
539
540 return (rc);
541}

584 __checkReturn efx_rc_t
585siena_nvram_partn_set_version(
586 __in efx_nic_t *enp,
587 __in uint32_t partn,
588 __in_ecount(4) uint16_t version[4])
589{
590 efx_mcdi_iface_t *emip = &(enp->en_mcdi.em_emip);
591 siena_mc_dynamic_config_hdr_t *dcfg = NULL;
592 siena_mc_fw_version_t *fwverp;
593 uint32_t dcfg_partn;
594 size_t dcfg_size;
595 unsigned int hdr_length;
596 unsigned int vpd_length;
597 unsigned int vpd_offset;
598 unsigned int nitems;
599 unsigned int required_hdr_length;
600 unsigned int pos;
601 uint8_t cksum;
602 uint32_t subtype;
603 size_t length;
604 efx_rc_t rc;
605
606 dcfg_partn = (emip->emi_port == 1)
607 ? MC_CMD_NVRAM_TYPE_DYNAMIC_CFG_PORT0
608 : MC_CMD_NVRAM_TYPE_DYNAMIC_CFG_PORT1;
609
610 if ((rc = siena_nvram_partn_size(enp, dcfg_partn, &dcfg_size)) != 0)
611 goto fail1;
612
613 if ((rc = siena_nvram_partn_lock(enp, dcfg_partn)) != 0)
614 goto fail2;
615
616 if ((rc = siena_nvram_get_dynamic_cfg(enp, dcfg_partn,
617 B_TRUE, &dcfg, &length)) != 0)
618 goto fail3;
619
620 hdr_length = EFX_WORD_FIELD(dcfg->length, EFX_WORD_0);
621 nitems = EFX_DWORD_FIELD(dcfg->num_fw_version_items, EFX_DWORD_0);
622 vpd_length = EFX_DWORD_FIELD(dcfg->dynamic_vpd_length, EFX_DWORD_0);
623 vpd_offset = EFX_DWORD_FIELD(dcfg->dynamic_vpd_offset, EFX_DWORD_0);
624
625 /*
626 * NOTE: This function will blatt any fields trailing the version
627 * vector, or the VPD chunk.
628 */
629 required_hdr_length = SIENA_DYNAMIC_CFG_SIZE(partn + 1);
630 if (required_hdr_length + vpd_length > length) {
631 rc = ENOSPC;
632 goto fail4;
633 }
634
635 if (vpd_offset < required_hdr_length) {
636 (void) memmove((caddr_t)dcfg + required_hdr_length,
637 (caddr_t)dcfg + vpd_offset, vpd_length);
638 vpd_offset = required_hdr_length;
639 EFX_POPULATE_DWORD_1(dcfg->dynamic_vpd_offset,
640 EFX_DWORD_0, vpd_offset);
641 }
642
643 if (hdr_length < required_hdr_length) {
644 (void) memset((caddr_t)dcfg + hdr_length, 0,
645 required_hdr_length - hdr_length);
646 hdr_length = required_hdr_length;
647 EFX_POPULATE_WORD_1(dcfg->length,
648 EFX_WORD_0, hdr_length);
649 }
650
651 /* Get the subtype to insert into the fw_subtype array */
652 if ((rc = siena_nvram_get_subtype(enp, partn, &subtype)) != 0)
653 goto fail5;
654
655 /* Fill out the new version */
656 fwverp = &dcfg->fw_version[partn];
657 EFX_POPULATE_DWORD_1(fwverp->fw_subtype, EFX_DWORD_0, subtype);
658 EFX_POPULATE_WORD_1(fwverp->version_w, EFX_WORD_0, version[0]);
659 EFX_POPULATE_WORD_1(fwverp->version_x, EFX_WORD_0, version[1]);
660 EFX_POPULATE_WORD_1(fwverp->version_y, EFX_WORD_0, version[2]);
661 EFX_POPULATE_WORD_1(fwverp->version_z, EFX_WORD_0, version[3]);
662
663 /* Update the version count */
664 if (nitems < partn + 1) {
665 nitems = partn + 1;
666 EFX_POPULATE_DWORD_1(dcfg->num_fw_version_items,
667 EFX_DWORD_0, nitems);
668 }
669
670 /* Update the checksum */
671 cksum = 0;
672 for (pos = 0; pos < hdr_length; pos++)
673 cksum += ((uint8_t *)dcfg)[pos];
674 dcfg->csum.eb_u8[0] -= cksum;
675
676 /* Erase and write the new partition */
677 if ((rc = siena_nvram_partn_erase(enp, dcfg_partn, 0, dcfg_size)) != 0)
678 goto fail6;
679
680 /* Write out the new structure to nvram */
681 if ((rc = siena_nvram_partn_write(enp, dcfg_partn, 0,
682 (caddr_t)dcfg, vpd_offset + vpd_length)) != 0)
683 goto fail7;
684
685 EFSYS_KMEM_FREE(enp->en_esip, length, dcfg);
686
687 siena_nvram_partn_unlock(enp, dcfg_partn, NULL);
688
689 return (0);
690
691fail7:
692 EFSYS_PROBE(fail7);
693fail6:
694 EFSYS_PROBE(fail6);
695fail5:
696 EFSYS_PROBE(fail5);
697fail4:
698 EFSYS_PROBE(fail4);
699
700 EFSYS_KMEM_FREE(enp->en_esip, length, dcfg);
701fail3:
702 EFSYS_PROBE(fail3);
703fail2:
704 EFSYS_PROBE(fail2);
705fail1:
706 EFSYS_PROBE1(fail1, efx_rc_t, rc);
707
708 return (rc);
709}
