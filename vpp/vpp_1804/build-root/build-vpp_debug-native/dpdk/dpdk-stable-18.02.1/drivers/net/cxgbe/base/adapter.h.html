187/*
188 * Currently there are two types of coalesce WR. Type 0 needs 48 bytes per
189 * packet (if one sgl is present) and type 1 needs 32 bytes. This means
190 * that type 0 can fit a maximum of 10 packets per WR and type 1 can fit
191 * 15 packets. We need to keep track of the mbuf pointers in a coalesce WR
192 * to be able to free those mbufs when we get completions back from the FW.
193 * Allocating the maximum number of pointers in every tx desc is a waste
194 * of memory resources so we only store 2 pointers per tx desc which should
195 * be enough since a tx desc can only fit 2 packets in the best case
196 * scenario where a packet needs 32 bytes.
197 */
198#define ETH_COALESCE_PKT_NUM 15
199#define ETH_COALESCE_PKT_PER_DESC 2

330#define CXGBE_PCI_REG(reg) rte_read32(reg)

332static inline uint64_t cxgbe_read_addr64(volatile void *addr)
333{
334 uint64_t val = CXGBE_PCI_REG(addr);
335 uint64_t val2 = CXGBE_PCI_REG(((volatile uint8_t *)(addr) + 4));
336
337 val2 = (uint64_t)(val2 << 32);
338 val += val2;
339 return val;
340}
341
342static inline uint32_t cxgbe_read_addr(volatile void *addr)
343{
344 return CXGBE_PCI_REG(addr);
345}
346
347#define CXGBE_PCI_REG_ADDR(adap, reg) \
348 ((volatile uint32_t *)((char *)(adap)->regs + (reg)))
349
350#define CXGBE_READ_REG(adap, reg) \
351 cxgbe_read_addr(CXGBE_PCI_REG_ADDR((adap), (reg)))
352
353#define CXGBE_READ_REG64(adap, reg) \
354 cxgbe_read_addr64(CXGBE_PCI_REG_ADDR((adap), (reg)))
355
356#define CXGBE_PCI_REG_WRITE(reg, value) rte_write32((value), (reg))
357
358#define CXGBE_PCI_REG_WRITE_RELAXED(reg, value) \
359 rte_write32_relaxed((value), (reg))
360
361#define CXGBE_WRITE_REG(adap, reg, value) \
362 CXGBE_PCI_REG_WRITE(CXGBE_PCI_REG_ADDR((adap), (reg)), (value))
363
364#define CXGBE_WRITE_REG_RELAXED(adap, reg, value) \
365 CXGBE_PCI_REG_WRITE_RELAXED(CXGBE_PCI_REG_ADDR((adap), (reg)), (value))
366
367static inline uint64_t cxgbe_write_addr64(volatile void *addr, uint64_t val)
368{
369 CXGBE_PCI_REG_WRITE(addr, val);
370 CXGBE_PCI_REG_WRITE(((volatile uint8_t *)(addr) + 4), (val >> 32));
371 return val;
372}
373
374#define CXGBE_WRITE_REG64(adap, reg, value) \
375 cxgbe_write_addr64(CXGBE_PCI_REG_ADDR((adap), (reg)), (value))

557/**
558 * t4_os_find_pci_capability - lookup a capability in the PCI capability list
559 * @adapter: the adapter
560 * @cap: the capability
561 *
562 * Return the address of the given capability within the PCI capability list.
563 */
564static inline int t4_os_find_pci_capability(struct adapter *adapter, int cap)
565{
566 u16 status;
567 int ttl = 48;
568 u8 pos = 0;
569 u8 id = 0;
570
571 t4_os_pci_read_cfg2(adapter, PCI_STATUS, &status);
572 if (!(status & PCI_STATUS_CAP_LIST)) {
573 dev_err(adapter, "PCIe capability reading failed\n");
574 return -1;
575 }
576
577 t4_os_pci_read_cfg(adapter, PCI_CAPABILITY_LIST, &pos);
578 while (ttl-- && pos >= 0x40) {
579 pos &= ~3;
580 t4_os_pci_read_cfg(adapter, (pos + PCI_CAP_LIST_ID), &id);
581
582 if (id == 0xff)
583 break;
584
585 if (id == cap)
586 return (int)pos;
587
588 t4_os_pci_read_cfg(adapter, (pos + PCI_CAP_LIST_NEXT), &pos);
589 }
590 return 0;
591}
