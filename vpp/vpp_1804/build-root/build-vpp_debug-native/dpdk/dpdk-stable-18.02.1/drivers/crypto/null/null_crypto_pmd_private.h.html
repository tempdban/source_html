32/** private data structure for each NULL crypto device */
33struct null_crypto_private {
34 unsigned max_nb_qpairs; /**< Max number of queue pairs */
35 unsigned max_nb_sessions; /**< Max number of sessions */
36};
37
38/** NULL crypto queue pair */
39struct null_crypto_qp {
40 uint16_t id;
41 /**< Queue Pair Identifier */
42 char name[RTE_CRYPTODEV_NAME_MAX_LEN];
43 /**< Unique Queue Pair Name */
44 struct rte_ring *processed_pkts;
45 /**< Ring for placing process packets */
46 struct rte_mempool *sess_mp;
47 /**< Session Mempool */
48 struct rte_cryptodev_stats qp_stats;
49 /**< Queue pair statistics */
50} __rte_cache_aligned;
51
52
53/** NULL crypto private session structure */
54struct null_crypto_session {
55 uint32_t reserved;
56} __rte_cache_aligned;
