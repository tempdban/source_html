34/*
35 * strlist.c - simple linked list of strings
36 */
37
38#include "compiler.h"
39
40#include <string.h>
41
42#include "strlist.h"
43
44static inline StrList *nasm_str_to_strlist(const char *str)
45{
46 size_t l = strlen(str) + 1;
47 StrList *sl = nasm_malloc(l + sizeof sl->next);
48
49 memcpy(sl->str, str, l);
50 sl->next = NULL;
51
52 return sl;
53}
54
55/*
56 * Append a string list entry to a string list if and only if it isn't
57 * already there. Return true if it was added.
58 */
59bool nasm_add_to_strlist(StrList **head, StrList *entry)
60{
61 StrList *list;
62
63 if (!head)
64 return false;
65
66 list = *head;
67 while (list) {
68 if (!strcmp(list->str, entry->str))
69 return false;
70 head = &list->next;
71 list = list->next;
72 }
73
74 *head = entry;
75 entry->next = NULL;
76 return true;
77}
78
79/*
80 * Append a string to a string list if and only if it isn't
81 * already there. Return true if it was added.
82 */
83bool nasm_add_string_to_strlist(StrList **head, const char *str)
84{
85 StrList *list;
86
87 if (!head)
88 return false;
89
90 list = *head;
91 while (list) {
92 if (!strcmp(list->str, str))
93 return false;
94 head = &list->next;
95 list = list->next;
96 }
97
98 *head = nasm_str_to_strlist(str);
99 return true;
100}
