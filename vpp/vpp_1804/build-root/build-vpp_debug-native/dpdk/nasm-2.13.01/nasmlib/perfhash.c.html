34#include "perfhash.h"
35#include "hashtbl.h" /* For crc64i() */
36
37int perfhash_find(const struct perfect_hash *hash, const char *str)
38{
39 uint32_t k1, k2;
40 uint64_t crc;
41 uint16_t ix;
42
43 crc = crc64i(hash->crcinit, str);
44 k1 = (uint32_t)crc & hash->hashmask;
45 k2 = (uint32_t)(crc >> 32) & hash->hashmask;
46
47 ix = hash->hashvals[k1] + hash->hashvals[k2 + hash->hashmask + 1];
48
49 if (ix >= hash->tbllen ||
50 !hash->strings[ix] ||
51 nasm_stricmp(str, hash->strings[ix]))
52 return hash->errval;
53
54 return hash->tbloffs + ix;
55}
