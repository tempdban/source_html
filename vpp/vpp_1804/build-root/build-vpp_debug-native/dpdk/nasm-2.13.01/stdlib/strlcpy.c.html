19/*
20 * Copy src to string dst of size siz. At most siz-1 characters
21 * will be copied. Always NUL terminates (unless siz == 0).
22 * Returns strlen(src); if retval >= siz, truncation occurred.
23 */
24#ifndef HAVE_STRLCPY
25
26size_t strlcpy(char *dst, const char *src, size_t siz)
27{
28 char *d = dst;
29 const char *s = src;
30 size_t n = siz;
31
32 /* Copy as many bytes as will fit */
33 if (n != 0) {
34 while (--n != 0) {
35 if ((*d++ = *s++) == '\0')
36 break;
37 }
38 }
39
40 /* Not enough room in dst, add NUL and traverse rest of src */
41 if (n == 0) {
42 if (siz != 0)
43 *d = '\0'; /* NUL-terminate dst */
44 while (*s++)
45 ;
46 }
47
48 return(s - src - 1); /* count does not include NUL */
49}

51#endif
