34/*
35 * compiler.h
36 *
37 * Compiler-specific macros for NASM. Feel free to add support for
38 * other compilers in here.
39 *
40 * This header file should be included before any other header.
41 */

43#ifndef NASM_COMPILER_H
44#define NASM_COMPILER_H 1

46/*
47 * At least DJGPP and Cygwin have broken header files if __STRICT_ANSI__
48 * is defined.
49 */

111#ifndef __cplusplus /* C++ has false, true, bool as keywords */
112# ifdef HAVE_STDBOOL_H
113# include <stdbool.h>
114# elif defined(HAVE__BOOL)
115# typedef _Bool bool
116# define false 0
117# define true 1
118# else
119/* This is sort of dangerous, since casts will behave different than
120 casting to the standard boolean type. Always use !!, not (bool). */
121typedef enum bool { false, true } bool;
122# endif

125/* Provide a substitute for offsetof() if we don't have one. This
126 variant works on most (but not *all*) systems... */
127#ifndef offsetof
128# define offsetof(t,m) ((size_t)&(((t *)0)->m))
129#endif
130
131/* The container_of construct: if p is a pointer to member m of
132 container class c, then return a pointer to the container of which
133 *p is a member. */
134#ifndef container_of
135# define container_of(p, c, m) ((c *)((char *)(p) - offsetof(c,m)))
136#endif

163/*
164 * Define this to 1 for faster performance if this is a littleendian
165 * platform which can do unaligned memory references. It is safe
166 * to leave it defined to 0 even if that is true.
167 */
168#if defined(__386__) || defined(__i386__) || defined(__x86_64__)
169# define X86_MEMORY 1
170# ifndef WORDS_LITTLEENDIAN
171# define WORDS_LITTLEENDIAN 1
172# endif
173#else
174# define X86_MEMORY 0
175#endif

177/*
178 * Hints to the compiler that a particular branch of code is more or
179 * less likely to be taken.
180 */
181#if HAVE___BUILTIN_EXPECT
182# define likely(x) __builtin_expect(!!(x), 1)
183# define unlikely(x) __builtin_expect(!!(x), 0)
184#else
185# define likely(x) (!!(x))
186# define unlikely(x) (!!(x))
187#endif

189/*
190 * Hints about malloc-like functions that never return NULL
191 */
192#ifdef HAVE_FUNC_ATTRIBUTE_RETURNS_NONNULL
193# define never_null __attribute__((returns_nonnull))
194#else
195# define never_null
196#endif
197
198#ifdef HAVE_FUNC_ATTRIBUTE_MALLOC
199# define safe_alloc never_null __attribute__((malloc))
200#else
201# define safe_alloc
202#endif
203
204#ifdef HAVE_FUNC_ATTRIBUTE_ALLOC_SIZE
205# define safe_malloc(s) safe_alloc __attribute__((alloc_size(s)))
206# define safe_malloc2(s1,s2) safe_alloc __attribute__((alloc_size(s1,s2)))
207# define safe_realloc(s) never_null __attribute__((alloc_size(s)))
208#else
209# define safe_malloc(s) safe_alloc
210# define safe_malloc2(s1,s2) safe_alloc
211# define safe_realloc(s) never_null
212#endif

214/*
215 * How to tell the compiler that a function doesn't return
216 */
217#ifdef HAVE_STDNORETURN_H
218# include <stdnoreturn.h>
219# define no_return noreturn void
220#elif defined(HAVE_FUNC_ATTRIBUTE_NORETURN)
221# define no_return void __attribute__((noreturn))
222#elif defined(_MSC_VER)
223# define no_return __declspec(noreturn) void
224#else
225# define no_return void
226#endif

228/*
229 * How to tell the compiler that a function takes a printf-like string
230 */
231#ifdef HAVE_FUNC_ATTRIBUTE_FORMAT
232# define printf_func(fmt, list) __attribute__((format(printf, fmt, list)))
233#else
234# define printf_func(fmt, list)
235#endif

237/*
238 * How to tell the compiler that a function is pure arithmetic
239 */
240#ifdef HAVE_FUNC_ATTRIBUTE_CONST
241# define const_func __attribute__((const))
242#else
243# define const_func
244#endif

246/*
247 * This function has no side effects, but depends on its arguments,
248 * memory pointed to by its arguments, or global variables.
249 * NOTE: functions that return a value by modifying memory pointed to
250 * by a pointer argument are *NOT* considered pure.
251 */
252#ifdef HAVE_FUNC_ATTRIBUTE_PURE
253# define pure_func __attribute__((pure))
254#else
255# define pure_func
256#endif

258/* Watcom doesn't handle switch statements with 64-bit types, hack around it */
259#ifdef __WATCOMC__
260# define BOGUS_CASE 0x76543210
261
262static inline unsigned int watcom_switch_hack(uint64_t x)
263{
264 if (x > (uint64_t)UINT_MAX)
265 return BOGUS_CASE;
266 else
267 return (unsigned int)x;
268}
269
270# define switch(x) switch(sizeof(x) > sizeof(unsigned int) \
271 ? watcom_switch_hack(x) : (unsigned int)(x))
272
273/* This is to make sure BOGUS_CASE doesn't conflict with anything real... */
274# define default case BOGUS_CASE: default
275#endif
