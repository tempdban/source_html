44/* symbol binding */

48#define GLOBAL_TEMP_BASE 0x40000000 /* bigger than any sane symbol index */

53/* this stuff is needed for the dwarf/stabs debugging format */

56/* Known sections with nonstandard defaults */
57struct elf_known_section {
58 const char *name; /* Name of section */
59 int type; /* Section type (SHT_) */
60 uint32_t flags; /* Section flags (SHF_) */
61 uint32_t align; /* Section alignment */
62};
63extern const struct elf_known_section elf_known_sections[];

65/*
66 * Special ELF sections (after the real sections but before debugging ones)
67 */
68#define sec_shstrtab (nsects + 1)
69#define sec_symtab (nsects + 2)
70#define sec_strtab (nsects + 3)
71#define sec_numspecial 3

77/* stabs */
78#define sec_stab (nsections-3)
79#define sec_stabstr (nsections-2)

82/* stabs symbol table format */
83struct stabentry {
84 uint32_t n_strx;
85 uint8_t n_type;
86 uint8_t n_other;
87 uint16_t n_desc;
88 uint32_t n_value;
89};

91/* dwarf */
92#define sec_debug_aranges (nsections-10)

96#define sec_rela_debug_info (nsections-6)

98#define sec_debug_line (nsections-4)

106void elf_section_attrib(char *name, char *attr, int pass,
107 uint32_t *flags_and, uint32_t *flags_or,
108 uint64_t *align, int *type);
109
110#define WRITE_STAB(p,n_strx,n_type,n_other,n_desc,n_value) \
111 do { \
112 WRITELONG(p, n_strx); \
113 WRITECHAR(p, n_type); \
114 WRITECHAR(p, n_other); \
115 WRITESHORT(p, n_desc); \
116 WRITELONG(p, n_value); \
117 } while (0)
118
119struct elf_reloc {
120 struct elf_reloc *next;
121 int64_t address; /* relative to _start_ of section */
122 int64_t symbol; /* symbol index */
123 int64_t offset; /* symbol addend */
124 int type; /* type of relocation */
125};
126
127struct elf_symbol {
128 struct rbtree symv; /* symbol value and symbol rbtree */
129 int32_t strpos; /* string table position of name */
130 int32_t section; /* section ID of the symbol */
131 int type; /* symbol type */
132 int other; /* symbol visibility */
133 int32_t size; /* size of symbol */
134 int32_t globnum; /* symbol table offset if global */
135 struct elf_symbol *nextfwd; /* list of unresolved-size symbols */
136 char *name; /* used temporarily if in above list */
137};
138
139struct elf_section {
140 struct SAA *data;
141 uint64_t len;
142 uint64_t size;
143 uint64_t nrelocs;
144 int32_t index;
145 int type; /* SHT_PROGBITS or SHT_NOBITS */
146 uint64_t align; /* alignment: power of two */
147 uint64_t flags; /* section flags */
148 char *name;
149 struct SAA *rel;
150 uint64_t rellen;
151 struct elf_reloc *head;
152 struct elf_reloc **tail;
153 struct rbtree *gsyms; /* global symbols in section */
154};
