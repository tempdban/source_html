34/*
35 * outform.c manages a list of output formats, and associates
36 * them with their relevant drivers. Also has a
37 * routine to find the correct driver given a name
38 * for it
39 */
40
41#include "compiler.h"
42
43#include <stdio.h>
44#include <string.h>
45
46#define BUILD_DRIVERS_ARRAY
47#include "outform.h"
48
49const struct ofmt *ofmt_find(const char *name,
50 const struct ofmt_alias **ofmt_alias)
51{
52 const struct ofmt * const *ofp;
53 const struct ofmt *of;
54 unsigned int i;
55
56 *ofmt_alias = NULL;
57
58 /* primary targets first */
59 for (ofp = drivers; (of = *ofp); ofp++) {
60 if (!nasm_stricmp(name, of->shortname))
61 return of;
62 }
63
64 /* lets walk thru aliases then */
65 for (i = 0; i < ARRAY_SIZE(ofmt_aliases); i++) {
66 if (ofmt_aliases[i].shortname &&
67 !nasm_stricmp(name, ofmt_aliases[i].shortname)) {
68 *ofmt_alias = &ofmt_aliases[i];
69 return ofmt_aliases[i].ofmt;
70 }
71 }
72
73 return NULL;
74}
75
76const struct dfmt *dfmt_find(const struct ofmt *ofmt, const char *name)
77{
78 const struct dfmt * const *dfp;
79 const struct dfmt *df;
80
81 for (dfp = ofmt->debug_formats; (df = *dfp); dfp++) {
82 if (!nasm_stricmp(name, df->shortname))
83 return df;
84 }
85 return NULL;
86}
87
88void ofmt_list(const struct ofmt *deffmt, FILE * fp)
89{
90 const struct ofmt * const *ofp, *of;
91 unsigned int i;
92
93 /* primary targets first */
94 for (ofp = drivers; (of = *ofp); ofp++) {
95 fprintf(fp, " %c %-10s%s\n",
96 of == deffmt ? '*' : ' ',
97 of->shortname, of->fullname);
98 }
99
100 /* lets walk through aliases then */
101 for (i = 0; i < ARRAY_SIZE(ofmt_aliases); i++) {
102 if (!ofmt_aliases[i].shortname)
103 continue;
104 fprintf(fp, " %-10s%s\n",
105 ofmt_aliases[i].shortname,
106 ofmt_aliases[i].fullname);
107 }
108}
109
110void dfmt_list(const struct ofmt *ofmt, FILE *fp)
111{
112 const struct dfmt * const *dfp;
113 const struct dfmt *df;
114
115 for (dfp = ofmt->debug_formats; (df = *dfp); dfp++) {
116 fprintf(fp, " %c %-10s%s\n",
117 df == dfmt ? '*' : ' ',
118 df->shortname, df->fullname);
119 }
120}
