34/*
35 * exprdump.c
36 *
37 * Debugging code to dump the contents of an expression vector to stdout
38 */
39
40#include "nasm.h"
41
42static const char *expr_type(int32_t type)
43{
44 static char seg_str[64];
45
46 switch (type) {
47 case 0:
48 return "null";
49 case EXPR_UNKNOWN:
50 return "unknown";
51 case EXPR_SIMPLE:
52 return "simple";
53 case EXPR_WRT:
54 return "wrt";
55 case EXPR_RDSAE:
56 return "sae";
57 default:
58 break;
59 }
60
61 if (type >= EXPR_REG_START && type <= EXPR_REG_END) {
62 return nasm_reg_names[type - EXPR_REG_START];
63 } else if (type >= EXPR_SEGBASE) {
64 snprintf(seg_str, sizeof seg_str, "%sseg %d",
65 (type - EXPR_SEGBASE) == location.segment ? "this " : "",
66 type - EXPR_SEGBASE);
67 return seg_str;
68 } else {
69 return "ERR";
70 }
71}
72
73void dump_expr(const expr *e)
74{
75 printf("[");
76 for (; e->type; e++)
77 printf("<%s(%d),%"PRId64">", expr_type(e->type), e->type, e->value);
78 printf("]\n");
79}
