572const unsigned char *nasm_stdmac_find_package(const char *package)
573{
574 static const struct {
575 const char *package;
576 const unsigned char *macros;
577 } packages[4] = {
578 { "altreg", nasm_usemac_altreg },
579 { "fp", nasm_usemac_fp },
580 { "ifunc", nasm_usemac_ifunc },
581 { "smartalign", nasm_usemac_smartalign },
582 };
583#define UNUSED (65535/3)
584 static const int16_t hash1[4] = {
585 0,
586 0,
587 2,
588 -1,
589 };
590 static const int16_t hash2[4] = {
591 1,
592 2,
593 UNUSED,
594 UNUSED,
595 };
596 uint32_t k1, k2;
597 uint64_t crc;
598 uint16_t ix;
599
600 crc = crc64i(UINT64_C(0x076259c3e291c26c), package);
601 k1 = (uint32_t)crc;
602 k2 = (uint32_t)(crc >> 32);
603
604 ix = hash1[k1 & 0x3] + hash2[k2 & 0x3];
605 if (ix >= 4)
606 return NULL;
607
608 if (nasm_stricmp(packages[ix].package, package))
609 return NULL;
610
611 return packages[ix].macros;
612}
