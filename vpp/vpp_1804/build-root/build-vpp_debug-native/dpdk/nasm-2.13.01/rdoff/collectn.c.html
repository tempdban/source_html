10void collection_init(Collection * c)
11{
12 int i;
13
14 for (i = 0; i < 32; i++)
15 c->p[i] = NULL;
16 c->next = NULL;
17}
18
19void **colln(Collection * c, int index)
20{
21 while (index >= 32) {
22 index -= 32;
23 if (c->next == NULL) {
24 c->next = nasm_malloc(sizeof(Collection));
25 collection_init(c->next);
26 }
27 c = c->next;
28 }
29 return &(c->p[index]);
30}
31
32void collection_reset(Collection * c)
33{
34 int i;
35
36 if (c->next) {
37 collection_reset(c->next);
38 nasm_free(c->next);
39 }
40
41 c->next = NULL;
42 for (i = 0; i < 32; i++)
43 c->p[i] = NULL;
44}
