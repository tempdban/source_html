37#define MBEDTLS_ERR_CTR_DRBG_ENTROPY_SOURCE_FAILED -0x0034 /**< The entropy source failed. */
38#define MBEDTLS_ERR_CTR_DRBG_REQUEST_TOO_BIG -0x0036 /**< The requested random buffer length is too big. */
39#define MBEDTLS_ERR_CTR_DRBG_INPUT_TOO_BIG -0x0038 /**< The input (entropy + additional data) is too large. */
40#define MBEDTLS_ERR_CTR_DRBG_FILE_IO_ERROR -0x003A /**< Read or write error in file. */
41
42#define MBEDTLS_CTR_DRBG_BLOCKSIZE 16 /**< The block size used by the cipher. */
43#define MBEDTLS_CTR_DRBG_KEYSIZE 32 /**< The key size used by the cipher. */
44#define MBEDTLS_CTR_DRBG_KEYBITS ( MBEDTLS_CTR_DRBG_KEYSIZE * 8 ) /**< The key size for the DRBG operation, in bits. */
45#define MBEDTLS_CTR_DRBG_SEEDLEN ( MBEDTLS_CTR_DRBG_KEYSIZE + MBEDTLS_CTR_DRBG_BLOCKSIZE ) /**< The seed length, calculated as (counter + AES key). */

103/**
104 * \brief The CTR_DRBG context structure.
105 */
106typedef struct
107{
108 unsigned char counter[16]; /*!< The counter (V). */
109 int reseed_counter; /*!< The reseed counter. */
110 int prediction_resistance; /*!< This determines whether prediction
111 resistance is enabled, that is
112 whether to systematically reseed before
113 each random generation. */
114 size_t entropy_len; /*!< The amount of entropy grabbed on each
115 seed or reseed operation. */
116 int reseed_interval; /*!< The reseed interval. */
117
118 mbedtls_aes_context aes_ctx; /*!< The AES context. */
119
120 /*
121 * Callbacks (Entropy)
122 */
123 int (*f_entropy)(void *, unsigned char *, size_t);
124 /*!< The entropy callback function. */
125
126 void *p_entropy; /*!< The context for the entropy function. */
127
128#if defined(MBEDTLS_THREADING_C)
129 mbedtls_threading_mutex_t mutex;
130#endif
131}
132mbedtls_ctr_drbg_context;
133
134/**
135 * \brief This function initializes the CTR_DRBG context,
136 * and prepares it for mbedtls_ctr_drbg_seed()
137 * or mbedtls_ctr_drbg_free().
138 *
139 * \param ctx The CTR_DRBG context to initialize.
140 */
141void mbedtls_ctr_drbg_init( mbedtls_ctr_drbg_context *ctx );
142
143/**
144 * \brief This function seeds and sets up the CTR_DRBG
145 * entropy source for future reseeds.
146 *
147 * \note Personalization data can be provided in addition to the more generic
148 * entropy source, to make this instantiation as unique as possible.
149 *
150 * \param ctx The CTR_DRBG context to seed.
151 * \param f_entropy The entropy callback, taking as arguments the
152 * \p p_entropy context, the buffer to fill, and the
153 length of the buffer.
154 * \param p_entropy The entropy context.
155 * \param custom Personalization data, that is device-specific
156 identifiers. Can be NULL.
157 * \param len The length of the personalization data.
158 *
159 * \return \c 0 on success, or
160 * #MBEDTLS_ERR_CTR_DRBG_ENTROPY_SOURCE_FAILED on failure.
161 */
162int mbedtls_ctr_drbg_seed( mbedtls_ctr_drbg_context *ctx,
163 int (*f_entropy)(void *, unsigned char *, size_t),
164 void *p_entropy,
165 const unsigned char *custom,
166 size_t len );
167
168/**
169 * \brief This function clears CTR_CRBG context data.
170 *
171 * \param ctx The CTR_DRBG context to clear.
172 */
173void mbedtls_ctr_drbg_free( mbedtls_ctr_drbg_context *ctx );
174
175/**
176 * \brief This function turns prediction resistance on or off.
177 * The default value is off.
178 *
179 * \note If enabled, entropy is gathered at the beginning of
180 * every call to mbedtls_ctr_drbg_random_with_add().
181 * Only use this if your entropy source has sufficient
182 * throughput.
183 *
184 * \param ctx The CTR_DRBG context.
185 * \param resistance #MBEDTLS_CTR_DRBG_PR_ON or #MBEDTLS_CTR_DRBG_PR_OFF.
186 */
187void mbedtls_ctr_drbg_set_prediction_resistance( mbedtls_ctr_drbg_context *ctx,
188 int resistance );
189
190/**
191 * \brief This function sets the amount of entropy grabbed on each
192 * seed or reseed. The default value is
193 * #MBEDTLS_CTR_DRBG_ENTROPY_LEN.
194 *
195 * \param ctx The CTR_DRBG context.
196 * \param len The amount of entropy to grab.
197 */
198void mbedtls_ctr_drbg_set_entropy_len( mbedtls_ctr_drbg_context *ctx,
199 size_t len );
200
201/**
202 * \brief This function sets the reseed interval.
203 * The default value is #MBEDTLS_CTR_DRBG_RESEED_INTERVAL.
204 *
205 * \param ctx The CTR_DRBG context.
206 * \param interval The reseed interval.
207 */
208void mbedtls_ctr_drbg_set_reseed_interval( mbedtls_ctr_drbg_context *ctx,
209 int interval );
210
211/**
212 * \brief This function reseeds the CTR_DRBG context, that is
213 * extracts data from the entropy source.
214 *
215 * \param ctx The CTR_DRBG context.
216 * \param additional Additional data to add to the state. Can be NULL.
217 * \param len The length of the additional data.
218 *
219 * \return \c 0 on success, or
220 * #MBEDTLS_ERR_CTR_DRBG_ENTROPY_SOURCE_FAILED on failure.
221 */
222int mbedtls_ctr_drbg_reseed( mbedtls_ctr_drbg_context *ctx,
223 const unsigned char *additional, size_t len );
224
225/**
226 * \brief This function updates the state of the CTR_DRBG context.
227 *
228 * \param ctx The CTR_DRBG context.
229 * \param additional The data to update the state with.
230 * \param add_len Length of \p additional data.
231 *
232 * \note If \p add_len is greater than #MBEDTLS_CTR_DRBG_MAX_SEED_INPUT,
233 * only the first #MBEDTLS_CTR_DRBG_MAX_SEED_INPUT Bytes are used.
234 * The remaining Bytes are silently discarded.
235 */
236void mbedtls_ctr_drbg_update( mbedtls_ctr_drbg_context *ctx,
237 const unsigned char *additional, size_t add_len );
238
239/**
240 * \brief This function updates a CTR_DRBG instance with additional
241 * data and uses it to generate random data.
242 *
243 * \note The function automatically reseeds if the reseed counter is exceeded.
244 *
245 * \param p_rng The CTR_DRBG context. This must be a pointer to a
246 * #mbedtls_ctr_drbg_context structure.
247 * \param output The buffer to fill.
248 * \param output_len The length of the buffer.
249 * \param additional Additional data to update. Can be NULL.
250 * \param add_len The length of the additional data.
251 *
252 * \return \c 0 on success, or
253 * #MBEDTLS_ERR_CTR_DRBG_ENTROPY_SOURCE_FAILED or
254 * #MBEDTLS_ERR_CTR_DRBG_REQUEST_TOO_BIG on failure.
255 */
256int mbedtls_ctr_drbg_random_with_add( void *p_rng,
257 unsigned char *output, size_t output_len,
258 const unsigned char *additional, size_t add_len );
259
260/**
261 * \brief This function uses CTR_DRBG to generate random data.
262 *
263 * \note The function automatically reseeds if the reseed counter is exceeded.
264 *
265 * \param p_rng The CTR_DRBG context. This must be a pointer to a
266 * #mbedtls_ctr_drbg_context structure.
267 * \param output The buffer to fill.
268 * \param output_len The length of the buffer.
269 *
270 * \return \c 0 on success, or
271 * #MBEDTLS_ERR_CTR_DRBG_ENTROPY_SOURCE_FAILED or
272 * #MBEDTLS_ERR_CTR_DRBG_REQUEST_TOO_BIG on failure.
273 */
274int mbedtls_ctr_drbg_random( void *p_rng,
275 unsigned char *output, size_t output_len );
