50/**
51 * Container for an X.509 certificate. The certificate may be chained.
52 */
53typedef struct mbedtls_x509_crt
54{
55 mbedtls_x509_buf raw; /**< The raw certificate data (DER). */
56 mbedtls_x509_buf tbs; /**< The raw certificate body (DER). The part that is To Be Signed. */
57
58 int version; /**< The X.509 version. (1=v1, 2=v2, 3=v3) */
59 mbedtls_x509_buf serial; /**< Unique id for certificate issued by a specific CA. */
60 mbedtls_x509_buf sig_oid; /**< Signature algorithm, e.g. sha1RSA */
61
62 mbedtls_x509_buf issuer_raw; /**< The raw issuer data (DER). Used for quick comparison. */
63 mbedtls_x509_buf subject_raw; /**< The raw subject data (DER). Used for quick comparison. */
64
65 mbedtls_x509_name issuer; /**< The parsed issuer data (named information object). */
66 mbedtls_x509_name subject; /**< The parsed subject data (named information object). */
67
68 mbedtls_x509_time valid_from; /**< Start time of certificate validity. */
69 mbedtls_x509_time valid_to; /**< End time of certificate validity. */
70
71 mbedtls_pk_context pk; /**< Container for the public key context. */
72
73 mbedtls_x509_buf issuer_id; /**< Optional X.509 v2/v3 issuer unique identifier. */
74 mbedtls_x509_buf subject_id; /**< Optional X.509 v2/v3 subject unique identifier. */
75 mbedtls_x509_buf v3_ext; /**< Optional X.509 v3 extensions. */
76 mbedtls_x509_sequence subject_alt_names; /**< Optional list of Subject Alternative Names (Only dNSName supported). */
77
78 int ext_types; /**< Bit string containing detected and parsed extensions */
79 int ca_istrue; /**< Optional Basic Constraint extension value: 1 if this certificate belongs to a CA, 0 otherwise. */
80 int max_pathlen; /**< Optional Basic Constraint extension value: The maximum path length to the root certificate. Path length is 1 higher than RFC 5280 'meaning', so 1+ */
81
82 unsigned int key_usage; /**< Optional key usage extension value: See the values in x509.h */
83
84 mbedtls_x509_sequence ext_key_usage; /**< Optional list of extended key usage OIDs. */
85
86 unsigned char ns_cert_type; /**< Optional Netscape certificate type extension value: See the values in x509.h */
87
88 mbedtls_x509_buf sig; /**< Signature: hash of the tbs part signed with the private key. */
89 mbedtls_md_type_t sig_md; /**< Internal representation of the MD algorithm of the signature algorithm, e.g. MBEDTLS_MD_SHA256 */
90 mbedtls_pk_type_t sig_pk; /**< Internal representation of the Public Key algorithm of the signature algorithm, e.g. MBEDTLS_PK_RSA */
91 void *sig_opts; /**< Signature options to be passed to mbedtls_pk_verify_ext(), e.g. for RSASSA-PSS */
92
93 struct mbedtls_x509_crt *next; /**< Next certificate in the CA-chain. */
94}
95mbedtls_x509_crt;

97/**
98 * Build flag from an algorithm/curve identifier (pk, md, ecp)
99 * Since 0 is always XXX_NONE, ignore it.
100 */
101#define MBEDTLS_X509_ID_FLAG( id ) ( 1 << ( id - 1 ) )
102
103/**
104 * Security profile for certificate verification.
105 *
106 * All lists are bitfields, built by ORing flags from MBEDTLS_X509_ID_FLAG().
107 */
108typedef struct
109{
110 uint32_t allowed_mds; /**< MDs for signatures */
111 uint32_t allowed_pks; /**< PK algs for signatures */
112 uint32_t allowed_curves; /**< Elliptic curves for ECDSA */
113 uint32_t rsa_min_bitlen; /**< Minimum size for RSA keys */
114}
115mbedtls_x509_crt_profile;

256/**
257 * \brief Verify the certificate signature
258 *
259 * The verify callback is a user-supplied callback that
260 * can clear / modify / add flags for a certificate. If set,
261 * the verification callback is called for each
262 * certificate in the chain (from the trust-ca down to the
263 * presented crt). The parameters for the callback are:
264 * (void *parameter, mbedtls_x509_crt *crt, int certificate_depth,
265 * int *flags). With the flags representing current flags for
266 * that specific certificate and the certificate depth from
267 * the bottom (Peer cert depth = 0).
268 *
269 * All flags left after returning from the callback
270 * are also returned to the application. The function should
271 * return 0 for anything (including invalid certificates)
272 * other than fatal error, as a non-zero return code
273 * immediately aborts the verification process. For fatal
274 * errors, a specific error code should be used (different
275 * from MBEDTLS_ERR_X509_CERT_VERIFY_FAILED which should not
276 * be returned at this point), or MBEDTLS_ERR_X509_FATAL_ERROR
277 * can be used if no better code is available.
278 *
279 * \note In case verification failed, the results can be displayed
280 * using \c mbedtls_x509_crt_verify_info()
281 *
282 * \note Same as \c mbedtls_x509_crt_verify_with_profile() with the
283 * default security profile.
284 *
285 * \note It is your responsibility to provide up-to-date CRLs for
286 * all trusted CAs. If no CRL is provided for the CA that was
287 * used to sign the certificate, CRL verification is skipped
288 * silently, that is *without* setting any flag.
289 *
290 * \note The \c trust_ca list can contain two types of certificates:
291 * (1) those of trusted root CAs, so that certificates
292 * chaining up to those CAs will be trusted, and (2)
293 * self-signed end-entity certificates to be trusted (for
294 * specific peers you know) - in that case, the self-signed
295 * certificate doesn't need to have the CA bit set.
296 *
297 * \param crt a certificate (chain) to be verified
298 * \param trust_ca the list of trusted CAs (see note above)
299 * \param ca_crl the list of CRLs for trusted CAs (see note above)
300 * \param cn expected Common Name (can be set to
301 * NULL if the CN must not be verified)
302 * \param flags result of the verification
303 * \param f_vrfy verification function
304 * \param p_vrfy verification parameter
305 *
306 * \return 0 (and flags set to 0) if the chain was verified and valid,
307 * MBEDTLS_ERR_X509_CERT_VERIFY_FAILED if the chain was verified
308 * but found to be invalid, in which case *flags will have one
309 * or more MBEDTLS_X509_BADCERT_XXX or MBEDTLS_X509_BADCRL_XXX
310 * flags set, or another error (and flags set to 0xffffffff)
311 * in case of a fatal error encountered during the
312 * verification process.
313 */
314int mbedtls_x509_crt_verify( mbedtls_x509_crt *crt,
315 mbedtls_x509_crt *trust_ca,
316 mbedtls_x509_crl *ca_crl,
317 const char *cn, uint32_t *flags,
318 int (*f_vrfy)(void *, mbedtls_x509_crt *, int, uint32_t *),
319 void *p_vrfy );
320
321/**
322 * \brief Verify the certificate signature according to profile
323 *
324 * \note Same as \c mbedtls_x509_crt_verify(), but with explicit
325 * security profile.
326 *
327 * \note The restrictions on keys (RSA minimum size, allowed curves
328 * for ECDSA) apply to all certificates: trusted root,
329 * intermediate CAs if any, and end entity certificate.
330 *
331 * \param crt a certificate (chain) to be verified
332 * \param trust_ca the list of trusted CAs
333 * \param ca_crl the list of CRLs for trusted CAs
334 * \param profile security profile for verification
335 * \param cn expected Common Name (can be set to
336 * NULL if the CN must not be verified)
337 * \param flags result of the verification
338 * \param f_vrfy verification function
339 * \param p_vrfy verification parameter
340 *
341 * \return 0 if successful or MBEDTLS_ERR_X509_CERT_VERIFY_FAILED
342 * in which case *flags will have one or more
343 * MBEDTLS_X509_BADCERT_XXX or MBEDTLS_X509_BADCRL_XXX flags
344 * set,
345 * or another error in case of a fatal error encountered
346 * during the verification process.
347 */
348int mbedtls_x509_crt_verify_with_profile( mbedtls_x509_crt *crt,
349 mbedtls_x509_crt *trust_ca,
350 mbedtls_x509_crl *ca_crl,
351 const mbedtls_x509_crt_profile *profile,
352 const char *cn, uint32_t *flags,
353 int (*f_vrfy)(void *, mbedtls_x509_crt *, int, uint32_t *),
354 void *p_vrfy );

619/**
620 * \brief Write a built up certificate to a X509 DER structure
621 * Note: data is written at the end of the buffer! Use the
622 * return value to determine where you should start
623 * using the buffer
624 *
625 * \param ctx certificate to write away
626 * \param buf buffer to write to
