99/**
100 * \brief ECP point structure (jacobian coordinates)
101 *
102 * \note All functions expect and return points satisfying
103 * the following condition: Z == 0 or Z == 1. (Other
104 * values of Z are used by internal functions only.)
105 * The point is zero, or "at infinity", if Z == 0.
106 * Otherwise, X and Y are its standard (affine) coordinates.
107 */
108typedef struct
109{
110 mbedtls_mpi X; /*!< the point's X coordinate */
111 mbedtls_mpi Y; /*!< the point's Y coordinate */
112 mbedtls_mpi Z; /*!< the point's Z coordinate */
113}
114mbedtls_ecp_point;

116/**
117 * \brief ECP group structure
118 *
119 * We consider two types of curves equations:
120 * 1. Short Weierstrass y^2 = x^3 + A x + B mod P (SEC1 + RFC 4492)
121 * 2. Montgomery, y^2 = x^3 + A x^2 + x mod P (Curve25519 + draft)
122 * In both cases, a generator G for a prime-order subgroup is fixed. In the
123 * short weierstrass, this subgroup is actually the whole curve, and its
124 * cardinal is denoted by N.
125 *
126 * In the case of Short Weierstrass curves, our code requires that N is an odd
127 * prime. (Use odd in mbedtls_ecp_mul() and prime in mbedtls_ecdsa_sign() for blinding.)
128 *
129 * In the case of Montgomery curves, we don't store A but (A + 2) / 4 which is
130 * the quantity actually used in the formulas. Also, nbits is not the size of N
131 * but the required size for private keys.
132 *
133 * If modp is NULL, reduction modulo P is done using a generic algorithm.
134 * Otherwise, it must point to a function that takes an mbedtls_mpi in the range
135 * 0..2^(2*pbits)-1 and transforms it in-place in an integer of little more
136 * than pbits, so that the integer may be efficiently brought in the 0..P-1
137 * range by a few additions or substractions. It must return 0 on success and
138 * non-zero on failure.
139 */
140typedef struct
141{
142 mbedtls_ecp_group_id id; /*!< internal group identifier */
143 mbedtls_mpi P; /*!< prime modulus of the base field */
144 mbedtls_mpi A; /*!< 1. A in the equation, or 2. (A + 2) / 4 */
145 mbedtls_mpi B; /*!< 1. B in the equation, or 2. unused */
146 mbedtls_ecp_point G; /*!< generator of the (sub)group used */
147 mbedtls_mpi N; /*!< 1. the order of G, or 2. unused */
148 size_t pbits; /*!< number of bits in P */
149 size_t nbits; /*!< number of bits in 1. P, or 2. private keys */
150 unsigned int h; /*!< internal: 1 if the constants are static */
151 int (*modp)(mbedtls_mpi *); /*!< function for fast reduction mod P */
152 int (*t_pre)(mbedtls_ecp_point *, void *); /*!< unused */
153 int (*t_post)(mbedtls_ecp_point *, void *); /*!< unused */
154 void *t_data; /*!< unused */
155 mbedtls_ecp_point *T; /*!< pre-computed points for ecp_mul_comb() */
156 size_t T_size; /*!< number for pre-computed points */
157}
158mbedtls_ecp_group;

193#if !defined(MBEDTLS_ECP_WINDOW_SIZE)
194/*
195 * Maximum "window" size used for point multiplication.
196 * Default: 6.
197 * Minimum value: 2. Maximum value: 7.
198 *
199 * Result is an array of at most ( 1 << ( MBEDTLS_ECP_WINDOW_SIZE - 1 ) )
200 * points used for point multiplication. This value is directly tied to EC
201 * peak memory usage, so decreasing it by one should roughly cut memory usage
202 * by two (if large curves are in use).
203 *
204 * Reduction in size may reduce speed, but larger curves are impacted first.
205 * Sample performances (in ECDHE handshakes/s, with FIXED_POINT_OPTIM = 1):
206 * w-size: 6 5 4 3 2
207 * 521 145 141 135 120 97
208 * 384 214 209 198 177 146
209 * 256 320 320 303 262 226
210
211 * 224 475 475 453 398 342
212 * 192 640 640 633 587 476
213 */
214#define MBEDTLS_ECP_WINDOW_SIZE 6 /**< Maximum window size used */
215#endif /* MBEDTLS_ECP_WINDOW_SIZE */
216
217#if !defined(MBEDTLS_ECP_FIXED_POINT_OPTIM)
218/*
219 * Trade memory for speed on fixed-point multiplication.
220 *
221 * This speeds up repeated multiplication of the generator (that is, the
222 * multiplication in ECDSA signatures, and half of the multiplications in
223 * ECDSA verification and ECDHE) by a factor roughly 3 to 4.
224 *
225 * The cost is increasing EC peak memory usage by a factor roughly 2.
226 *
227 * Change this value to 0 to reduce peak memory usage.
228 */
229#define MBEDTLS_ECP_FIXED_POINT_OPTIM 1 /**< Enable fixed-point speed-up */
230#endif /* MBEDTLS_ECP_FIXED_POINT_OPTIM */

506/**
507 * \brief Multiplication by an integer: R = m * P
508 * (Not thread-safe to use same group in multiple threads)
509 *
510 * \note In order to prevent timing attacks, this function
511 * executes the exact same sequence of (base field)
512 * operations for any valid m. It avoids any if-branch or
513 * array index depending on the value of m.
514 *
515 * \note If f_rng is not NULL, it is used to randomize intermediate
516 * results in order to prevent potential timing attacks
517 * targeting these results. It is recommended to always
518 * provide a non-NULL f_rng (the overhead is negligible).
519 *
520 * \param grp ECP group
521 * \param R Destination point
522 * \param m Integer by which to multiply
523 * \param P Point to multiply
524 * \param f_rng RNG function (see notes)
525 * \param p_rng RNG parameter
526 *
527 * \return 0 if successful,
528 * MBEDTLS_ERR_ECP_INVALID_KEY if m is not a valid privkey
529 * or P is not a valid pubkey,
530 * MBEDTLS_ERR_MPI_ALLOC_FAILED if memory allocation failed
531 */
532int mbedtls_ecp_mul( mbedtls_ecp_group *grp, mbedtls_ecp_point *R,
533 const mbedtls_mpi *m, const mbedtls_ecp_point *P,
534 int (*f_rng)(void *, unsigned char *, size_t), void *p_rng );
535
536/**
537 * \brief Multiplication and addition of two points by integers:
538 * R = m * P + n * Q
539 * (Not thread-safe to use same group in multiple threads)
540 *
541 * \note In contrast to mbedtls_ecp_mul(), this function does not guarantee
542 * a constant execution flow and timing.
543 *
544 * \param grp ECP group
545 * \param R Destination point
546 * \param m Integer by which to multiply P
547 * \param P Point to multiply by m
548 * \param n Integer by which to multiply Q
549 * \param Q Point to be multiplied by n
550 *
551 * \return 0 if successful,
552 * MBEDTLS_ERR_ECP_INVALID_KEY if m or n is not a valid privkey
553 * or P or Q is not a valid pubkey,
554 * MBEDTLS_ERR_MPI_ALLOC_FAILED if memory allocation failed
555 */
556int mbedtls_ecp_muladd( mbedtls_ecp_group *grp, mbedtls_ecp_point *R,
557 const mbedtls_mpi *m, const mbedtls_ecp_point *P,
558 const mbedtls_mpi *n, const mbedtls_ecp_point *Q );
