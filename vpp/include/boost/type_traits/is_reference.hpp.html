12#ifndef BOOST_TT_IS_REFERENCE_HPP_INCLUDED
13#define BOOST_TT_IS_REFERENCE_HPP_INCLUDED
14
15#include <boost/type_traits/config.hpp>
16#include <boost/type_traits/is_lvalue_reference.hpp>
17#include <boost/type_traits/is_rvalue_reference.hpp>
18#include <boost/type_traits/ice.hpp>
19
20// should be the last #include
21#include <boost/type_traits/detail/bool_trait_def.hpp>
22
23namespace boost {
24
25namespace detail {
26
27template <typename T>
28struct is_reference_impl
29{
30 BOOST_STATIC_CONSTANT(bool, value =
31 (::boost::type_traits::ice_or<
32 ::boost::is_lvalue_reference<T>::value, ::boost::is_rvalue_reference<T>::value
33 >::value));
34};
35
36} // namespace detail
37
38BOOST_TT_AUX_BOOL_TRAIT_DEF1(is_reference,T,::boost::detail::is_reference_impl<T>::value)
39
40} // namespace boost
41
42#include <boost/type_traits/detail/bool_trait_undef.hpp>
43
44#endif // BOOST_TT_IS_REFERENCE_HPP_INCLUDED
