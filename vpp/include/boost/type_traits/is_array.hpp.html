50namespace detail {
51
52using ::boost::type_traits::yes_type;
53using ::boost::type_traits::no_type;
54using ::boost::type_traits::wrap;
55
56template< typename T > T(* is_array_tester1(wrap<T>) )(wrap<T>);
57char BOOST_TT_DECL is_array_tester1(...);
58
59template< typename T> no_type is_array_tester2(T(*)(wrap<T>));
60yes_type BOOST_TT_DECL is_array_tester2(...);
61
62template< typename T >
63struct is_array_impl
64{
65 BOOST_STATIC_CONSTANT(bool, value =
66 sizeof(::boost::detail::is_array_tester2(
67 ::boost::detail::is_array_tester1(
68 ::boost::type_traits::wrap<T>()
69 )
70 )) == 1
71 );
72};
73
74#ifndef BOOST_NO_CV_VOID_SPECIALIZATIONS
75BOOST_TT_AUX_BOOL_TRAIT_IMPL_SPEC1(is_array,void,false)
76BOOST_TT_AUX_BOOL_TRAIT_IMPL_SPEC1(is_array,void const,false)
77BOOST_TT_AUX_BOOL_TRAIT_IMPL_SPEC1(is_array,void volatile,false)
78BOOST_TT_AUX_BOOL_TRAIT_IMPL_SPEC1(is_array,void const volatile,false)
79#endif

81} // namespace detail
