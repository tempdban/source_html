59enum basic_errors
60{
61 /// Permission denied.
62 access_denied = BOOST_ASIO_SOCKET_ERROR(EACCES),
63
64 /// Address family not supported by protocol.
65 address_family_not_supported = BOOST_ASIO_SOCKET_ERROR(EAFNOSUPPORT),
66
67 /// Address already in use.
68 address_in_use = BOOST_ASIO_SOCKET_ERROR(EADDRINUSE),
69
70 /// Transport endpoint is already connected.
71 already_connected = BOOST_ASIO_SOCKET_ERROR(EISCONN),
72
73 /// Operation already in progress.
74 already_started = BOOST_ASIO_SOCKET_ERROR(EALREADY),
75
76 /// Broken pipe.
77 broken_pipe = BOOST_ASIO_WIN_OR_POSIX(
78 BOOST_ASIO_NATIVE_ERROR(ERROR_BROKEN_PIPE),
79 BOOST_ASIO_NATIVE_ERROR(EPIPE)),
80
81 /// A connection has been aborted.
82 connection_aborted = BOOST_ASIO_SOCKET_ERROR(ECONNABORTED),
83
84 /// Connection refused.
85 connection_refused = BOOST_ASIO_SOCKET_ERROR(ECONNREFUSED),
86
87 /// Connection reset by peer.
88 connection_reset = BOOST_ASIO_SOCKET_ERROR(ECONNRESET),
89
90 /// Bad file descriptor.
91 bad_descriptor = BOOST_ASIO_SOCKET_ERROR(EBADF),
92
93 /// Bad address.
94 fault = BOOST_ASIO_SOCKET_ERROR(EFAULT),
95
96 /// No route to host.
97 host_unreachable = BOOST_ASIO_SOCKET_ERROR(EHOSTUNREACH),
98
99 /// Operation now in progress.
100 in_progress = BOOST_ASIO_SOCKET_ERROR(EINPROGRESS),
101
102 /// Interrupted system call.
103 interrupted = BOOST_ASIO_SOCKET_ERROR(EINTR),
104
105 /// Invalid argument.
106 invalid_argument = BOOST_ASIO_SOCKET_ERROR(EINVAL),
107
108 /// Message too long.
109 message_size = BOOST_ASIO_SOCKET_ERROR(EMSGSIZE),
110
111 /// The name was too long.
112 name_too_long = BOOST_ASIO_SOCKET_ERROR(ENAMETOOLONG),
113
114 /// Network is down.
115 network_down = BOOST_ASIO_SOCKET_ERROR(ENETDOWN),
116
117 /// Network dropped connection on reset.
118 network_reset = BOOST_ASIO_SOCKET_ERROR(ENETRESET),
119
120 /// Network is unreachable.
121 network_unreachable = BOOST_ASIO_SOCKET_ERROR(ENETUNREACH),
122
123 /// Too many open files.
124 no_descriptors = BOOST_ASIO_SOCKET_ERROR(EMFILE),
125
126 /// No buffer space available.
127 no_buffer_space = BOOST_ASIO_SOCKET_ERROR(ENOBUFS),
128
129 /// Cannot allocate memory.
130 no_memory = BOOST_ASIO_WIN_OR_POSIX(
131 BOOST_ASIO_NATIVE_ERROR(ERROR_OUTOFMEMORY),
132 BOOST_ASIO_NATIVE_ERROR(ENOMEM)),
133
134 /// Operation not permitted.
135 no_permission = BOOST_ASIO_WIN_OR_POSIX(
136 BOOST_ASIO_NATIVE_ERROR(ERROR_ACCESS_DENIED),
137 BOOST_ASIO_NATIVE_ERROR(EPERM)),
138
139 /// Protocol not available.
140 no_protocol_option = BOOST_ASIO_SOCKET_ERROR(ENOPROTOOPT),
141
142 /// Transport endpoint is not connected.
143 not_connected = BOOST_ASIO_SOCKET_ERROR(ENOTCONN),
144
145 /// Socket operation on non-socket.
146 not_socket = BOOST_ASIO_SOCKET_ERROR(ENOTSOCK),
147
148 /// Operation cancelled.
149 operation_aborted = BOOST_ASIO_WIN_OR_POSIX(
150 BOOST_ASIO_NATIVE_ERROR(ERROR_OPERATION_ABORTED),
151 BOOST_ASIO_NATIVE_ERROR(ECANCELED)),
152
153 /// Operation not supported.
154 operation_not_supported = BOOST_ASIO_SOCKET_ERROR(EOPNOTSUPP),
155
156 /// Cannot send after transport endpoint shutdown.
157 shut_down = BOOST_ASIO_SOCKET_ERROR(ESHUTDOWN),
158
159 /// Connection timed out.
160 timed_out = BOOST_ASIO_SOCKET_ERROR(ETIMEDOUT),
161
162 /// Resource temporarily unavailable.
163 try_again = BOOST_ASIO_WIN_OR_POSIX(
164 BOOST_ASIO_NATIVE_ERROR(ERROR_RETRY),
165 BOOST_ASIO_NATIVE_ERROR(EAGAIN)),
166
167 /// The socket is marked non-blocking and the requested operation would block.
168 would_block = BOOST_ASIO_SOCKET_ERROR(EWOULDBLOCK)
169};
170
171enum netdb_errors
172{
173 /// Host not found (authoritative).
174 host_not_found = BOOST_ASIO_NETDB_ERROR(HOST_NOT_FOUND),
175
176 /// Host not found (non-authoritative).
177 host_not_found_try_again = BOOST_ASIO_NETDB_ERROR(TRY_AGAIN),
178
179 /// The query is valid but does not have associated address data.
180 no_data = BOOST_ASIO_NETDB_ERROR(NO_DATA),
181
182 /// A non-recoverable error occurred.
183 no_recovery = BOOST_ASIO_NETDB_ERROR(NO_RECOVERY)
184};
185
186enum addrinfo_errors
187{
188 /// The service is not supported for the given socket type.
189 service_not_found = BOOST_ASIO_WIN_OR_POSIX(
190 BOOST_ASIO_NATIVE_ERROR(WSATYPE_NOT_FOUND),
191 BOOST_ASIO_GETADDRINFO_ERROR(EAI_SERVICE)),
192
193 /// The socket type is not supported.
194 socket_type_not_supported = BOOST_ASIO_WIN_OR_POSIX(
195 BOOST_ASIO_NATIVE_ERROR(WSAESOCKTNOSUPPORT),
196 BOOST_ASIO_GETADDRINFO_ERROR(EAI_SOCKTYPE))
197};
198
199enum misc_errors
200{
201 /// Already open.
202 already_open = 1,
203
204 /// End of file or stream.
205 eof,
206
207 /// Element not found.
208 not_found,
209
210 /// The descriptor cannot fit into the select system call's fd_set.
211 fd_set_failure
212};

214inline const boost::system::error_category& get_system_category()
215{

217}
