24namespace boost {
25namespace asio {
26namespace detail {
27
28class winsock_init_base
29{
30protected:
31 // Structure to track result of initialisation and number of uses. POD is used
32 // to ensure that the values are zero-initialised prior to any code being run.
33 struct data
34 {
35 long init_count_;
36 long result_;
37 };
38
39 BOOST_ASIO_DECL static void startup(data& d,
40 unsigned char major, unsigned char minor);
41
42 BOOST_ASIO_DECL static void cleanup(data& d);
43
44 BOOST_ASIO_DECL static void throw_on_error(data& d);
45};
46
47template <int Major = 2, int Minor = 0>
48class winsock_init : private winsock_init_base
49{
50public:
51 winsock_init(bool allow_throw = true)
52 {
53 startup(data_, Major, Minor);
54 if (allow_throw)
55 throw_on_error(data_);
56 }
57
58 winsock_init(const winsock_init&)
59 {
60 startup(data_, Major, Minor);
61 throw_on_error(data_);
62 }
63
64 ~winsock_init()
65 {
66 cleanup(data_);
67 }
68
69private:
70 static data data_;
71};
72
73template <int Major, int Minor>
74winsock_init_base::data winsock_init<Major, Minor>::data_;
75
76// Static variable to ensure that winsock is initialised before main, and
77// therefore before any other threads can get started.
78static const winsock_init<>& winsock_init_instance = winsock_init<>(false);
