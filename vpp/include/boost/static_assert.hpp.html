71//
72// Implicit instantiation requires that all member declarations be
73// instantiated, but that the definitions are *not* instantiated.
74//
75// It's not particularly clear how this applies to enum's or typedefs;
76// both are described as declarations [7.1.3] and [7.2] in the standard,
77// however some compilers use "delayed evaluation" of one or more of
78// these when implicitly instantiating templates. We use typedef declarations
79// by default, but try defining BOOST_USE_ENUM_STATIC_ASSERT if the enum
80// version gets better results from your compiler...
81//
82// Implementation:
83// Both of these versions rely on sizeof(incomplete_type) generating an error
84// message containing the name of the incomplete type. We use
85// "STATIC_ASSERTION_FAILURE" as the type name here to generate
86// an eye catching error message. The result of the sizeof expression is either
87// used as an enum initialiser, or as a template argument depending which version
88// is in use...
89// Note that the argument to the assert is explicitly cast to bool using old-
90// style casts: too many compilers currently have problems with static_cast
91// when used inside integral constant expressions.
92//
93#if !defined(BOOST_BUGGY_INTEGRAL_CONSTANT_EXPRESSIONS)
94
95#if defined(BOOST_MSVC) && (BOOST_MSVC < 1300)
96// __LINE__ macro broken when -ZI is used see Q199057
97// fortunately MSVC ignores duplicate typedef's.
98#define BOOST_STATIC_ASSERT( B ) \
99 typedef ::boost::static_assert_test<\
100 sizeof(::boost::STATIC_ASSERTION_FAILURE< (bool)( B ) >)\
101 > boost_static_assert_typedef_
102#elif defined(BOOST_MSVC)
103#define BOOST_STATIC_ASSERT( B ) \
104 typedef ::boost::static_assert_test<\
105 sizeof(::boost::STATIC_ASSERTION_FAILURE< BOOST_STATIC_ASSERT_BOOL_CAST ( B ) >)>\
106 BOOST_JOIN(boost_static_assert_typedef_, __COUNTER__)
107#elif defined(BOOST_INTEL_CXX_VERSION) || defined(BOOST_SA_GCC_WORKAROUND)
108// agurt 15/sep/02: a special care is needed to force Intel C++ issue an error
109// instead of warning in case of failure
110# define BOOST_STATIC_ASSERT( B ) \
111 typedef char BOOST_JOIN(boost_static_assert_typedef_, __LINE__) \
112 [ ::boost::STATIC_ASSERTION_FAILURE< BOOST_STATIC_ASSERT_BOOL_CAST( B ) >::value ]
113#elif defined(__sgi)
114// special version for SGI MIPSpro compiler
115#define BOOST_STATIC_ASSERT( B ) \
116 BOOST_STATIC_CONSTANT(bool, \
117 BOOST_JOIN(boost_static_assert_test_, __LINE__) = ( B )); \
118 typedef ::boost::static_assert_test<\
119 sizeof(::boost::STATIC_ASSERTION_FAILURE< \
120 BOOST_JOIN(boost_static_assert_test_, __LINE__) >)>\
121 BOOST_JOIN(boost_static_assert_typedef_, __LINE__)
122#elif BOOST_WORKAROUND(__MWERKS__, <= 0x3003)
123// special version for CodeWarrior <= 8.x
124#define BOOST_STATIC_ASSERT( B ) \
125 BOOST_STATIC_CONSTANT(int, \
126 BOOST_JOIN(boost_static_assert_test_, __LINE__) = \
127 sizeof(::boost::STATIC_ASSERTION_FAILURE< BOOST_STATIC_ASSERT_BOOL_CAST( B ) >) )
128#else
129// generic version
130#define BOOST_STATIC_ASSERT( B ) \
131 typedef ::boost::static_assert_test<\
132 sizeof(::boost::STATIC_ASSERTION_FAILURE< BOOST_STATIC_ASSERT_BOOL_CAST( B ) >)>\
133 BOOST_JOIN(boost_static_assert_typedef_, __LINE__) \
134 BOOST_STATIC_ASSERT_UNUSED_ATTRIBUTE
135#endif
136
137#else
138// alternative enum based implementation:
139#define BOOST_STATIC_ASSERT( B ) \
140 enum { BOOST_JOIN(boost_static_assert_enum_, __LINE__) \
141 = sizeof(::boost::STATIC_ASSERTION_FAILURE< (bool)( B ) >) }
142#endif
143#endif // defined(BOOST_NO_CXX11_STATIC_ASSERT)
