45/*
46 * The IOCTL interface is designed for extensibility by embedding the
47 * structure length (argsz) and flags into structures passed between
48 * kernel and userspace. We therefore use the _IO() macro for these
49 * defines to avoid implicitly embedding a size into the ioctl request.
50 * As structure fields are added, argsz will increase to match and flag
51 * bits will be defined to indicate additional fields with valid data.
52 * It's *always* the caller's responsibility to indicate the size of
53 * the structure passed by setting argsz appropriately.
54 */
55
56#define VFIO_TYPE (';')
57#define VFIO_BASE 100

59/*
60 * For extension of INFO ioctls, VFIO makes use of a capability chain
61 * designed after PCI/e capabilities. A flag bit indicates whether
62 * this capability chain is supported and a field defined in the fixed
63 * structure defines the offset of the first capability in the chain.
64 * This field is only valid when the corresponding bit in the flags
65 * bitmap is set. This offset field is relative to the start of the
66 * INFO buffer, as is the next field within each capability header.
67 * The id within the header is a shared address space per INFO ioctl,
68 * while the version field is specific to the capability id. The
69 * contents following the header are specific to the capability id.
70 */
71struct vfio_info_cap_header {
72 __u16 id; /* Identifies capability */
73 __u16 version; /* Version specific to the capability ID */
74 __u32 next; /* Offset of next capability */
75};
76
77/*
78 * Callers of INFO ioctls passing insufficiently sized buffers will see
79 * the capability chain flag bit set, a zero value for the first capability
80 * offset (if available within the provided argsz), and argsz will be
81 * updated to report the necessary buffer size. For compatibility, the
82 * INFO ioctl will not report error in this case, but the capability chain
83 * will not be available.
84 */

260/*
261 * The device specific type capability allows regions unique to a specific
262 * device or class of devices to be exposed. This helps solve the problem for
263 * vfio bus drivers of defining which region indexes correspond to which region
264 * on the device, without needing to resort to static indexes, as done by
265 * vfio-pci. For instance, if we were to go back in time, we might remove
266 * VFIO_PCI_VGA_REGION_INDEX and let vfio-pci simply define that all indexes
267 * greater than or equal to VFIO_PCI_NUM_REGIONS are device specific and we'd
268 * make a "VGA" device specific type to describe the VGA access space. This
269 * means that non-VGA devices wouldn't need to waste this index, and thus the
270 * address space associated with it due to implementation of device file
271 * descriptor offsets in vfio-pci.
272 *
273 * The current implementation is now part of the user ABI, so we can't use this
274 * for VGA, but there are other upcoming use cases, such as opregions for Intel
275 * IGD devices and framebuffers for vGPU devices. We missed VGA, but we'll
276 * use this for future additions.
277 *
278 * The structure below defines version 1 of this capability.
279 */
280#define VFIO_REGION_INFO_CAP_TYPE 2
281
282struct vfio_region_info_cap_type {
283 struct vfio_info_cap_header header;
284 __u32 type; /* global per bus driver */
285 __u32 subtype; /* type specific */
286};

288/**
289 * VFIO_DEVICE_GET_IRQ_INFO - _IOWR(VFIO_TYPE, VFIO_BASE + 9,
290 * struct vfio_irq_info)
291 *
292 * Retrieve information about a device IRQ. Caller provides
293 * struct vfio_irq_info with index value set. Caller sets argsz.
294 * Implementation of IRQ mapping is bus driver specific. Indexes
295 * using multiple IRQs are primarily intended to support MSI-like
296 * interrupt blocks. Zero count irq blocks may be used to describe
297 * unimplemented interrupt types.
298 *
299 * The EVENTFD flag indicates the interrupt index supports eventfd based
300 * signaling.
301 *
302 * The MASKABLE flags indicates the index supports MASK and UNMASK
303 * actions described below.
304 *
305 * AUTOMASKED indicates that after signaling, the interrupt line is
306 * automatically masked by VFIO and the user needs to unmask the line
307 * to receive new interrupts. This is primarily intended to distinguish
308 * level triggered interrupts.
309 *
310 * The NORESIZE flag indicates that the interrupt lines within the index
311 * are setup as a set and new subindexes cannot be enabled without first
312 * disabling the entire index. This is used for interrupts like PCI MSI
313 * and MSI-X where the driver may only use a subset of the available
314 * indexes, but VFIO needs to enable a specific number of vectors
315 * upfront. In the case of MSI-X, where the user can enable MSI-X and
316 * then add and unmask vectors, it's up to userspace to make the decision
317 * whether to allocate the maximum supported number of vectors or tear
318 * down setup and incrementally increase the vectors as each is enabled.
319 */
320struct vfio_irq_info {
321 __u32 argsz;
322 __u32 flags;
323#define VFIO_IRQ_INFO_EVENTFD (1 << 0)
324#define VFIO_IRQ_INFO_MASKABLE (1 << 1)
325#define VFIO_IRQ_INFO_AUTOMASKED (1 << 2)
326#define VFIO_IRQ_INFO_NORESIZE (1 << 3)
327 __u32 index; /* IRQ index */
328 __u32 count; /* Number of IRQs within this index */
329};
330#define VFIO_DEVICE_GET_IRQ_INFO _IO(VFIO_TYPE, VFIO_BASE + 9)
331
332/**
333 * VFIO_DEVICE_SET_IRQS - _IOW(VFIO_TYPE, VFIO_BASE + 10, struct vfio_irq_set)
334 *
335 * Set signaling, masking, and unmasking of interrupts. Caller provides
336 * struct vfio_irq_set with all fields set. 'start' and 'count' indicate
337 * the range of subindexes being specified.
338 *
339 * The DATA flags specify the type of data provided. If DATA_NONE, the
340 * operation performs the specified action immediately on the specified
341 * interrupt(s). For example, to unmask AUTOMASKED interrupt [0,0]:
342 * flags = (DATA_NONE|ACTION_UNMASK), index = 0, start = 0, count = 1.
343 *
344 * DATA_BOOL allows sparse support for the same on arrays of interrupts.
345 * For example, to mask interrupts [0,1] and [0,3] (but not [0,2]):
346 * flags = (DATA_BOOL|ACTION_MASK), index = 0, start = 1, count = 3,
347 * data = {1,0,1}
348 *
349 * DATA_EVENTFD binds the specified ACTION to the provided __s32 eventfd.
350 * A value of -1 can be used to either de-assign interrupts if already
351 * assigned or skip un-assigned interrupts. For example, to set an eventfd
352 * to be trigger for interrupts [0,0] and [0,2]:
353 * flags = (DATA_EVENTFD|ACTION_TRIGGER), index = 0, start = 0, count = 3,
354 * data = {fd1, -1, fd2}
355 * If index [0,1] is previously set, two count = 1 ioctls calls would be
356 * required to set [0,0] and [0,2] without changing [0,1].
357 *
358 * Once a signaling mechanism is set, DATA_BOOL or DATA_NONE can be used
359 * with ACTION_TRIGGER to perform kernel level interrupt loopback testing
360 * from userspace (ie. simulate hardware triggering).
361 *
362 * Setting of an event triggering mechanism to userspace for ACTION_TRIGGER
363 * enables the interrupt index for the device. Individual subindex interrupts
364 * can be disabled using the -1 value for DATA_EVENTFD or the index can be
365 * disabled as a whole with: flags = (DATA_NONE|ACTION_TRIGGER), count = 0.
366 *
367 * Note that ACTION_[UN]MASK specify user->kernel signaling (irqfds) while
368 * ACTION_TRIGGER specifies kernel->user signaling.
369 */
370struct vfio_irq_set {
371 __u32 argsz;
372 __u32 flags;
373#define VFIO_IRQ_SET_DATA_NONE (1 << 0) /* Data not present */
374#define VFIO_IRQ_SET_DATA_BOOL (1 << 1) /* Data is bool (u8) */
375#define VFIO_IRQ_SET_DATA_EVENTFD (1 << 2) /* Data is eventfd (s32) */
376#define VFIO_IRQ_SET_ACTION_MASK (1 << 3) /* Mask interrupt */
377#define VFIO_IRQ_SET_ACTION_UNMASK (1 << 4) /* Unmask interrupt */
378#define VFIO_IRQ_SET_ACTION_TRIGGER (1 << 5) /* Trigger interrupt */
379 __u32 index;
380 __u32 start;
381 __u32 count;
382 __u8 data[];
383};
384#define VFIO_DEVICE_SET_IRQS _IO(VFIO_TYPE, VFIO_BASE + 10)
385
386#define VFIO_IRQ_SET_DATA_TYPE_MASK (VFIO_IRQ_SET_DATA_NONE | \
387 VFIO_IRQ_SET_DATA_BOOL | \
388 VFIO_IRQ_SET_DATA_EVENTFD)
389#define VFIO_IRQ_SET_ACTION_TYPE_MASK (VFIO_IRQ_SET_ACTION_MASK | \
390 VFIO_IRQ_SET_ACTION_UNMASK | \
391 VFIO_IRQ_SET_ACTION_TRIGGER)
392/**

395 * Reset a device.

539/*
540 * The SPAPR TCE DDW info struct provides the information about
541 * the details of Dynamic DMA window capability.
542 *
543 * @pgsizes contains a page size bitmask, 4K/64K/16M are supported.
544 * @max_dynamic_windows_supported tells the maximum number of windows
545 * which the platform can create.
546 * @levels tells the maximum number of levels in multi-level IOMMU tables;
547 * this allows splitting a table into smaller chunks which reduces
548 * the amount of physically contiguous memory required for the table.
549 */
550struct vfio_iommu_spapr_tce_ddw_info {
551 __u64 pgsizes; /* Bitmap of supported page sizes */
552 __u32 max_dynamic_windows_supported;
553 __u32 levels;
554};
555
556/*
557 * The SPAPR TCE info struct provides the information about the PCI bus
558 * address ranges available for DMA, these values are programmed into
559 * the hardware so the guest has to know that information.
560 *
561 * The DMA 32 bit window start is an absolute PCI bus address.
562 * The IOVA address passed via map/unmap ioctls are absolute PCI bus
563 * addresses too so the window works as a filter rather than an offset
564 * for IOVA addresses.
565 *
566 * Flags supported:
567 * - VFIO_IOMMU_SPAPR_INFO_DDW: informs the userspace that dynamic DMA windows
568 * (DDW) support is present. @ddw is only supported when DDW is present.
569 */
570struct vfio_iommu_spapr_tce_info {
571 __u32 argsz;
572 __u32 flags;
573#define VFIO_IOMMU_SPAPR_INFO_DDW (1 << 0) /* DDW supported */
574 __u32 dma32_window_start; /* 32 bit window start (bytes) */
575 __u32 dma32_window_size; /* 32 bit window size (bytes) */
576 struct vfio_iommu_spapr_tce_ddw_info ddw;
577};

581/*
582 * EEH PE operation struct provides ways to:
583 * - enable/disable EEH functionality;
584 * - unfreeze IO/DMA for frozen PE;
585 * - read PE state;
