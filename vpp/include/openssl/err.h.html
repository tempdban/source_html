237/*
238 * Borland C seems too stupid to be able to shift and do longs in the
239 * pre-processor :-(
240 */
241# define ERR_PACK(l,f,r) (((((unsigned long)l)&0xffL)*0x1000000)| \
242 ((((unsigned long)f)&0xfffL)*0x1000)| \
243 ((((unsigned long)r)&0xfffL)))
244# define ERR_GET_LIB(l) (int)((((unsigned long)l)>>24L)&0xffL)
245# define ERR_GET_FUNC(l) (int)((((unsigned long)l)>>12L)&0xfffL)
246# define ERR_GET_REASON(l) (int)((l)&0xfffL)
247# define ERR_FATAL_ERROR(l) (int)((l)&ERR_R_FATAL)
