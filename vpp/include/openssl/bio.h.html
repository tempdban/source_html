347 /*-
348 * Buffers are setup like this:
349 *
350 * <---------------------- size ----------------------->
351 * +---------------------------------------------------+
352 * | consumed | remaining | free space |
353 * +---------------------------------------------------+
354 * <-- off --><------- len ------->

356 /*- BIO *bio; *//*

358 */

363 int ibuf_off; /* write/read offset */

365 int obuf_len; /* how many bytes are in it */
