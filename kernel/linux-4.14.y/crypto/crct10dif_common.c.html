70__u16 crc_t10dif_generic(__u16 crc, const unsigned char *buffer, size_t len)
71{
72 unsigned int i;
73
74 for (i = 0 ; i < len ; i++)
75 crc = (crc << 8) ^ t10_dif_crc_table[((crc >> 8) ^ buffer[i]) & 0xff];
76
77 return crc;
78}
