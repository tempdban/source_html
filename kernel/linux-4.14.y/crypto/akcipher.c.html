28#ifdef CONFIG_NET
29static int crypto_akcipher_report(struct sk_buff *skb, struct crypto_alg *alg)
30{
31 struct crypto_report_akcipher rakcipher;
32
33 strncpy(rakcipher.type, "akcipher", sizeof(rakcipher.type));
34
35 if (nla_put(skb, CRYPTOCFGA_REPORT_AKCIPHER,
36 sizeof(struct crypto_report_akcipher), &rakcipher))
37 goto nla_put_failure;
38 return 0;
39
40nla_put_failure:
41 return -EMSGSIZE;
42}
43#else
44static int crypto_akcipher_report(struct sk_buff *skb, struct crypto_alg *alg)
45{
46 return -ENOSYS;
47}
48#endif

58static void crypto_akcipher_exit_tfm(struct crypto_tfm *tfm)
59{
60 struct crypto_akcipher *akcipher = __crypto_akcipher_tfm(tfm);
61 struct akcipher_alg *alg = crypto_akcipher_alg(akcipher);
62
63 alg->exit(akcipher);
64}
65
66static int crypto_akcipher_init_tfm(struct crypto_tfm *tfm)
67{
68 struct crypto_akcipher *akcipher = __crypto_akcipher_tfm(tfm);
69 struct akcipher_alg *alg = crypto_akcipher_alg(akcipher);
70
71 if (alg->exit)
72 akcipher->base.exit = crypto_akcipher_exit_tfm;
73
74 if (alg->init)
75 return alg->init(akcipher);
76
77 return 0;
78}

80static void crypto_akcipher_free_instance(struct crypto_instance *inst)

87static const struct crypto_type crypto_akcipher_type = {
88 .extsize = crypto_alg_extsize,
89 .init_tfm = crypto_akcipher_init_tfm,
90 .free = crypto_akcipher_free_instance,
91#ifdef CONFIG_PROC_FS
92 .show = crypto_akcipher_show,
93#endif
94 .report = crypto_akcipher_report,
95 .maskclear = ~CRYPTO_ALG_TYPE_MASK,
96 .maskset = CRYPTO_ALG_TYPE_MASK,
97 .type = CRYPTO_ALG_TYPE_AKCIPHER,
98 .tfmsize = offsetof(struct crypto_akcipher, base),
99};

101int crypto_grab_akcipher(struct crypto_akcipher_spawn *spawn, const char *name,
102 u32 type, u32 mask)
103{
104 spawn->base.frontend = &crypto_akcipher_type;
105 return crypto_grab_spawn(&spawn->base, name, type, mask);
106}
107EXPORT_SYMBOL_GPL(crypto_grab_akcipher);

109struct crypto_akcipher *crypto_alloc_akcipher(const char *alg_name, u32 type,
110 u32 mask)
111{
112 return crypto_alloc_tfm(alg_name, &crypto_akcipher_type, type, mask);
113}
114EXPORT_SYMBOL_GPL(crypto_alloc_akcipher);

116static void akcipher_prepare_alg(struct akcipher_alg *alg)
117{
118 struct crypto_alg *base = &alg->base;
119
120 base->cra_type = &crypto_akcipher_type;
121 base->cra_flags &= ~CRYPTO_ALG_TYPE_MASK;
122 base->cra_flags |= CRYPTO_ALG_TYPE_AKCIPHER;
123}
124
125int crypto_register_akcipher(struct akcipher_alg *alg)
126{
127 struct crypto_alg *base = &alg->base;
128
129 akcipher_prepare_alg(alg);
130 return crypto_register_alg(base);
131}
132EXPORT_SYMBOL_GPL(crypto_register_akcipher);
133
134void crypto_unregister_akcipher(struct akcipher_alg *alg)
135{
136 crypto_unregister_alg(&alg->base);
137}
138EXPORT_SYMBOL_GPL(crypto_unregister_akcipher);

140int akcipher_register_instance(struct crypto_template *tmpl,
141 struct akcipher_instance *inst)
142{
143 akcipher_prepare_alg(&inst->alg);
144 return crypto_register_instance(tmpl, akcipher_crypto_instance(inst));
