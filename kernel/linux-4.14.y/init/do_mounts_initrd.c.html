37static int init_linuxrc(struct subprocess_info *info, struct cred *new)
38{
39 sys_unshare(CLONE_FS | CLONE_FILES);
40 /* stdin/stdout/stderr for /linuxrc */
41 sys_open("/dev/console", O_RDWR, 0);
42 sys_dup(0);
43 sys_dup(0);
44 /* move initrd over / and chdir/chroot in initrd root */
45 sys_chdir("/root");
46 sys_mount(".", "/", NULL, MS_MOVE, NULL);
47 sys_chroot(".");
48 sys_setsid();
49 return 0;
50}

52static void __init handle_initrd(void)
53{
54 struct subprocess_info *info;
55 static char *argv[] = { "linuxrc", NULL, };
56 extern char *envp_init[];
57 int error;
58
59 real_root_dev = new_encode_dev(ROOT_DEV);
60 create_dev("/dev/root.old", Root_RAM0);
61 /* mount initrd on rootfs' /root */
62 mount_block_root("/dev/root.old", root_mountflags & ~MS_RDONLY);
63 sys_mkdir("/old", 0700);
64 sys_chdir("/old");
65
66 /* try loading default modules from initrd */
67 load_default_modules();
68
69 /*
70 * In case that a resume from disk is carried out by linuxrc or one of
71 * its children, we need to tell the freezer not to wait for us.
72 */
73 current->flags |= PF_FREEZER_SKIP;
74
75 info = call_usermodehelper_setup("/linuxrc", argv, envp_init,
76 GFP_KERNEL, init_linuxrc, NULL, NULL);
77 if (!info)
78 return;
79 call_usermodehelper_exec(info, UMH_WAIT_PROC);
80
81 current->flags &= ~PF_FREEZER_SKIP;
82
83 /* move initrd to rootfs' /old */
84 sys_mount("..", ".", NULL, MS_MOVE, NULL);
85 /* switch root and cwd back to / of rootfs */
86 sys_chroot("..");
87
88 if (new_decode_dev(real_root_dev) == Root_RAM0) {
89 sys_chdir("/old");
90 return;
91 }
92
93 sys_chdir("/");
94 ROOT_DEV = new_decode_dev(real_root_dev);
95 mount_root();
96
97 printk(KERN_NOTICE "Trying to move old root to /initrd ... ");
98 error = sys_mount("/old", "/root/initrd", NULL, MS_MOVE, NULL);
99 if (!error)
100 printk("okay\n");
101 else {
102 int fd = sys_open("/dev/root.old", O_RDWR, 0);
103 if (error == -ENOENT)
104 printk("/initrd does not exist. Ignored.\n");
105 else
106 printk("failed\n");
107 printk(KERN_NOTICE "Unmounting old root\n");
108 sys_umount("/old", MNT_DETACH);
109 printk(KERN_NOTICE "Trying to free ramdisk memory ... ");
110 if (fd < 0) {
111 error = fd;
112 } else {
113 error = sys_ioctl(fd, BLKFLSBUF, 0);
114 sys_close(fd);
115 }
116 printk(!error ? "okay\n" : "failed\n");
117 }
118}

120bool __init initrd_load(void)
121{
122 if (mount_initrd) {
123 create_dev("/dev/ram", Root_RAM0);
124 /*
125 * Load the initrd data into /dev/ram0. Execute it as initrd
126 * unless /dev/ram0 is supposed to be our actual root device,
127 * in that case the ram disk is just set up here, and gets
128 * mounted in the normal path.
129 */
130 if (rd_load_image("/initrd.image") && ROOT_DEV != Root_RAM0) {
131 sys_unlink("/initrd.image");
132 handle_initrd();
133 return true;
134 }
135 }
136 sys_unlink("/initrd.image");
137 return false;
138}
