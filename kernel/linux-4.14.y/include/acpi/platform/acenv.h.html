320/*
321 * Debugger threading model
322 * Use single threaded if the entire subsystem is contained in an application
323 * Use multiple threaded when the subsystem is running in the kernel.
324 *
325 * By default the model is single threaded if ACPI_APPLICATION is set,
326 * multi-threaded if ACPI_APPLICATION is not set.
327 */
328#ifndef DEBUGGER_THREADING
329#if !defined (ACPI_APPLICATION) || defined (ACPI_EXEC_APP)
330#define DEBUGGER_THREADING DEBUGGER_MULTI_THREADED
331
332#else
333#define DEBUGGER_THREADING DEBUGGER_SINGLE_THREADED
334#endif
335#endif /* !DEBUGGER_THREADING */

337/******************************************************************************

339 * C library configuration
340 *
341 *****************************************************************************/
342
343/*
344 * ACPI_USE_SYSTEM_CLIBRARY - Define this if linking to an actual C library.
345 * Otherwise, local versions of string/memory functions will be used.
346 * ACPI_USE_STANDARD_HEADERS - Define this if linking to a C library and
347 * the standard header files may be used. Defining this implies that
348 * ACPI_USE_SYSTEM_CLIBRARY has been defined.
349 *
350 * The ACPICA subsystem only uses low level C library functions that do not
351 * call operating system services and may therefore be inlined in the code.
352 *
353 * It may be necessary to tailor these include files to the target
354 * generation environment.
355 */
356
357/* Use the standard C library headers. We want to keep these to a minimum. */
358
359#ifdef ACPI_USE_STANDARD_HEADERS
360
361/* Use the standard headers from the standard locations */
362
363#include <stdlib.h>
364#include <string.h>
365#include <ctype.h>
366#if defined (ACPI_APPLICATION) || defined(ACPI_LIBRARY)
367#include <stdio.h>
368#include <fcntl.h>
369#include <errno.h>
370#include <time.h>
371#include <signal.h>
372#endif
373
374#endif /* ACPI_USE_STANDARD_HEADERS */
