47 * _HID definitions

49 * Linux specific HIDs do not apply to this and begin with LNX:
50 */

54#define ACPI_SYSTEM_HID "LNXSYSTM"

59#define ACPI_BAY_HID "LNXIOBAY"
60#define ACPI_DOCK_HID "LNXDOCK"

62/* Quirk for broken IBM BIOSes */
63#define ACPI_SMBUS_IBM_HID "SMBUSIBM"
64
65/*
66 * For fixed hardware buttons, we fabricate acpi_devices with HID
67 * ACPI_BUTTON_HID_POWERF or ACPI_BUTTON_HID_SLEEPF. Fixed hardware
68 * signals only an event; it doesn't supply a notification value.
69 * To allow drivers to treat notifications from fixed hardware the
70 * same as those from real devices, we turn the events into this
71 * notification value.
72 */
73#define ACPI_FIXED_HARDWARE_EVENT 0x100
