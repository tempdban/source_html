31#ifndef _TTM_PLACEMENT_H_
32#define _TTM_PLACEMENT_H_
33
34#include <linux/types.h>
35
36/*
37 * Memory regions for data placement.
38 */
39
40#define TTM_PL_SYSTEM 0
41#define TTM_PL_TT 1
42#define TTM_PL_VRAM 2
43#define TTM_PL_PRIV 3
44
45#define TTM_PL_FLAG_SYSTEM (1 << TTM_PL_SYSTEM)
46#define TTM_PL_FLAG_TT (1 << TTM_PL_TT)
47#define TTM_PL_FLAG_VRAM (1 << TTM_PL_VRAM)
48#define TTM_PL_FLAG_PRIV (1 << TTM_PL_PRIV)
49#define TTM_PL_MASK_MEM 0x0000FFFF
50
51/*
52 * Other flags that affects data placement.
53 * TTM_PL_FLAG_CACHED indicates cache-coherent mappings
54 * if available.
55 * TTM_PL_FLAG_SHARED means that another application may
56 * reference the buffer.
57 * TTM_PL_FLAG_NO_EVICT means that the buffer may never
58 * be evicted to make room for other buffers.
59 * TTM_PL_FLAG_TOPDOWN requests to be placed from the
60 * top of the memory area, instead of the bottom.
61 */
62
63#define TTM_PL_FLAG_CACHED (1 << 16)
64#define TTM_PL_FLAG_UNCACHED (1 << 17)
65#define TTM_PL_FLAG_WC (1 << 18)
66#define TTM_PL_FLAG_CONTIGUOUS (1 << 19)
67#define TTM_PL_FLAG_NO_EVICT (1 << 21)
68#define TTM_PL_FLAG_TOPDOWN (1 << 22)
69
70#define TTM_PL_MASK_CACHING (TTM_PL_FLAG_CACHED | \
71 TTM_PL_FLAG_UNCACHED | \
72 TTM_PL_FLAG_WC)
73
74#define TTM_PL_MASK_MEMTYPE (TTM_PL_MASK_MEM | TTM_PL_MASK_CACHING)
75
76/**
77 * struct ttm_place
78 *
79 * @fpfn: first valid page frame number to put the object
80 * @lpfn: last valid page frame number to put the object
81 * @flags: memory domain and caching flags for the object
82 *
83 * Structure indicating a possible place to put an object.
84 */
85struct ttm_place {
86 unsigned fpfn;
87 unsigned lpfn;
88 uint32_t flags;
89};
90
91/**
92 * struct ttm_placement
93 *
94 * @num_placement: number of preferred placements
95 * @placement: preferred placements
96 * @num_busy_placement: number of preferred placements when need to evict buffer
97 * @busy_placement: preferred placements when need to evict buffer
98 *
99 * Structure indicating the placement you request for an object.
100 */
101struct ttm_placement {
102 unsigned num_placement;
103 const struct ttm_place *placement;
104 unsigned num_busy_placement;
105 const struct ttm_place *busy_placement;
106};
107
108#endif
