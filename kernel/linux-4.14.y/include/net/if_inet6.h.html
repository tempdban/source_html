213static inline void ipv6_eth_mc_map(const struct in6_addr *addr, char *buf)
214{
215 /*
216 * +-------+-------+-------+-------+-------+-------+
217 * | 33 | 33 | DST13 | DST14 | DST15 | DST16 |
218 * +-------+-------+-------+-------+-------+-------+
219 */
220
221 buf[0]= 0x33;
222 buf[1]= 0x33;
223
224 memcpy(buf + 2, &addr->s6_addr32[3], sizeof(__u32));
225}

227static inline void ipv6_arcnet_mc_map(const struct in6_addr *addr, char *buf)
228{
229 buf[0] = 0x00;
230}

232static inline void ipv6_ib_mc_map(const struct in6_addr *addr,
233 const unsigned char *broadcast, char *buf)
234{
235 unsigned char scope = broadcast[5] & 0xF;
236
237 buf[0] = 0; /* Reserved */
238 buf[1] = 0xff; /* Multicast QPN */
239 buf[2] = 0xff;
240 buf[3] = 0xff;
241 buf[4] = 0xff;
242 buf[5] = 0x10 | scope; /* scope from broadcast address */
243 buf[6] = 0x60; /* IPv6 signature */
244 buf[7] = 0x1b;
245 buf[8] = broadcast[8]; /* P_Key */
246 buf[9] = broadcast[9];
247 memcpy(buf + 10, addr->s6_addr + 6, 10);
248}
249
250static inline int ipv6_ipgre_mc_map(const struct in6_addr *addr,
251 const unsigned char *broadcast, char *buf)
252{
253 if ((broadcast[0] | broadcast[1] | broadcast[2] | broadcast[3]) != 0) {
254 memcpy(buf, broadcast, 4);
255 } else {
256 /* v4mapped? */
257 if ((addr->s6_addr32[0] | addr->s6_addr32[1] |
258 (addr->s6_addr32[2] ^ htonl(0x0000ffff))) != 0)
259 return -EINVAL;
260 memcpy(buf, &addr->s6_addr32[3], 4);
261 }
262 return 0;
263}
