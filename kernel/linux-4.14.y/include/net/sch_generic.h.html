117static inline bool qdisc_run_begin(struct Qdisc *qdisc)
118{
119 if (qdisc_is_running(qdisc))

121 /* Variant of write_seqcount_begin() telling lockdep a trylock

313/* The qdisc root lock is a mechanism by which to top level
314 * of a qdisc tree can be locked from any qdisc node in the
315 * forest. This allows changing the configuration of some
316 * aspect of the qdisc tree while blocking out asynchronous
317 * qdisc access in the packet processing paths.
318 *
319 * It is only legal to do this when the root will not change
320 * on us. Otherwise we'll potentially lock the wrong qdisc
321 * root. This is enforced by holding the RTNL semaphore, which
322 * all users of this lock accessor must do.
323 */
324static inline spinlock_t *qdisc_root_lock(const struct Qdisc *qdisc)
325{
326 struct Qdisc *root = qdisc_root(qdisc);
327
328 ASSERT_RTNL();
329 return qdisc_lock(root);
330}
331
332static inline spinlock_t *qdisc_root_sleeping_lock(const struct Qdisc *qdisc)
333{
334 struct Qdisc *root = qdisc_root_sleeping(qdisc);
335
336 ASSERT_RTNL();
337 return qdisc_lock(root);
338}
339
340static inline seqcount_t *qdisc_root_sleeping_running(const struct Qdisc *qdisc)
341{
342 struct Qdisc *root = qdisc_root_sleeping(qdisc);
343
344 ASSERT_RTNL();
345 return &root->running;
346}

391static inline unsigned int qdisc_class_hash(u32 id, u32 mask)
392{
393 id ^= id >> 8;
394 id ^= id >> 4;
395 return id & mask;
396}
397
398static inline struct Qdisc_class_common *
399qdisc_class_find(const struct Qdisc_class_hash *hash, u32 id)
400{
401 struct Qdisc_class_common *cl;
402 unsigned int h;
403
404 if (!id)
405 return NULL;
406
407 h = qdisc_class_hash(id, hash->hashmask);
408 hlist_for_each_entry(cl, &hash->hash[h], hnode) {
409 if (cl->classid == id)
410 return cl;
411 }
412 return NULL;
413}

469/* Reset all TX qdiscs greater then index of a device. */
470static inline void qdisc_reset_all_tx_gt(struct net_device *dev, unsigned int i)
471{
472 struct Qdisc *qdisc;
473
474 for (; i < dev->num_tx_queues; i++) {
475 qdisc = rtnl_dereference(netdev_get_tx_queue(dev, i)->qdisc);
476 if (qdisc) {
477 spin_lock_bh(qdisc_lock(qdisc));
478 qdisc_reset(qdisc);
479 spin_unlock_bh(qdisc_lock(qdisc));
480 }
481 }
482}

489/* Are all TX queues of the device empty? */
490static inline bool qdisc_all_tx_empty(const struct net_device *dev)
491{
492 unsigned int i;
493
494 rcu_read_lock();
495 for (i = 0; i < dev->num_tx_queues; i++) {
496 struct netdev_queue *txq = netdev_get_tx_queue(dev, i);
497 const struct Qdisc *q = rcu_dereference(txq->qdisc);
498
499 if (q->q.qlen) {
500 rcu_read_unlock();
501 return false;
502 }
503 }
504 rcu_read_unlock();
505 return true;
506}

666static inline int __qdisc_enqueue_tail(struct sk_buff *skb, struct Qdisc *sch,
667 struct qdisc_skb_head *qh)
668{
669 struct sk_buff *last = qh->tail;
670
671 if (last) {
672 skb->next = NULL;
673 last->next = skb;
674 qh->tail = skb;
675 } else {
676 qh->tail = skb;
677 qh->head = skb;
678 }
679 qh->qlen++;
680 qdisc_qstats_backlog_inc(sch, skb);
681
682 return NET_XMIT_SUCCESS;
683}

690static inline struct sk_buff *__qdisc_dequeue_head(struct qdisc_skb_head *qh)
691{
692 struct sk_buff *skb = qh->head;
693
694 if (likely(skb != NULL)) {
695 qh->head = skb->next;
696 qh->qlen--;
697 if (qh->head == NULL)
698 qh->tail = NULL;
699 skb->next = NULL;
700 }
701
702 return skb;
703}
704
705static inline struct sk_buff *qdisc_dequeue_head(struct Qdisc *sch)
706{
707 struct sk_buff *skb = __qdisc_dequeue_head(&sch->q);
708
709 if (likely(skb != NULL)) {
710 qdisc_qstats_backlog_dec(sch, skb);
711 qdisc_bstats_update(sch, skb);
712 }
713
714 return skb;
715}

717/* Instead of calling kfree_skb() while root qdisc lock is held,
718 * queue the skb for future freeing at end of __dev_xmit_skb()
719 */
720static inline void __qdisc_drop(struct sk_buff *skb, struct sk_buff **to_free)
721{
722 skb->next = *to_free;
723 *to_free = skb;
724}

726static inline void __qdisc_drop_all(struct sk_buff *skb,
727 struct sk_buff **to_free)
728{
729 if (skb->prev)
730 skb->prev->next = *to_free;
731 else
732 skb->next = *to_free;
733 *to_free = skb;
734}

736static inline unsigned int __qdisc_queue_drop_head(struct Qdisc *sch,
737 struct qdisc_skb_head *qh,
738 struct sk_buff **to_free)
739{
740 struct sk_buff *skb = __qdisc_dequeue_head(qh);
741
742 if (likely(skb != NULL)) {
743 unsigned int len = qdisc_pkt_len(skb);
744
745 qdisc_qstats_backlog_dec(sch, skb);
746 __qdisc_drop(skb, to_free);
747 return len;

824
825 sch_tree_lock(sch);

865/* Length to Time (L2T) lookup in a qdisc_rate_table, to determine how
866 long it will take to send a packet given its size.
867 */
868static inline u32 qdisc_l2t(struct qdisc_rate_table* rtab, unsigned int pktlen)
869{
870 int slot = pktlen + rtab->rate.cell_align + rtab->rate.overhead;
871 if (slot < 0)
872 slot = 0;
873 slot >>= rtab->rate.cell_log;
874 if (slot > 255)
875 return rtab->data[255]*(slot >> 8) + rtab->data[slot & 0xFF];
876 return rtab->data[slot];
877}

887static inline u64 psched_l2t_ns(const struct psched_ratecfg *r,
888 unsigned int len)
889{
890 len += r->overhead;
891
892 if (unlikely(r->linklayer == TC_LINKLAYER_ATM))
893 return ((u64)(DIV_ROUND_UP(len,48)*53) * r->mult) >> r->shift;
894
895 return ((u64)len * r->mult) >> r->shift;
896}
