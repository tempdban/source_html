262static inline struct ipv6_txoptions *txopt_get(const struct ipv6_pinfo *np)
263{
264 struct ipv6_txoptions *opt;
265
266 rcu_read_lock();
267 opt = rcu_dereference(np->opt);
268 if (opt) {
269 if (!refcount_inc_not_zero(&opt->refcnt))
270 opt = NULL;
271 else
272 opt = rcu_pointer_handoff(opt);
273 }
274 rcu_read_unlock();
275 return opt;
276}
277
278static inline void txopt_put(struct ipv6_txoptions *opt)
279{
280 if (opt && refcount_dec_and_test(&opt->refcnt))
281 kfree_rcu(opt, rcu);
282}

390static inline bool
391ipv6_masked_addr_cmp(const struct in6_addr *a1, const struct in6_addr *m,
392 const struct in6_addr *a2)
393{
394#if defined(CONFIG_HAVE_EFFICIENT_UNALIGNED_ACCESS) && BITS_PER_LONG == 64
395 const unsigned long *ul1 = (const unsigned long *)a1;
396 const unsigned long *ulm = (const unsigned long *)m;
397 const unsigned long *ul2 = (const unsigned long *)a2;
398
399 return !!(((ul1[0] ^ ul2[0]) & ulm[0]) |
400 ((ul1[1] ^ ul2[1]) & ulm[1]));
401#else
402 return !!(((a1->s6_addr32[0] ^ a2->s6_addr32[0]) & m->s6_addr32[0]) |
403 ((a1->s6_addr32[1] ^ a2->s6_addr32[1]) & m->s6_addr32[1]) |
404 ((a1->s6_addr32[2] ^ a2->s6_addr32[2]) & m->s6_addr32[2]) |
405 ((a1->s6_addr32[3] ^ a2->s6_addr32[3]) & m->s6_addr32[3]));
406#endif
407}

409static inline void ipv6_addr_prefix(struct in6_addr *pfx,
410 const struct in6_addr *addr,
411 int plen)
412{
413 /* caller must guarantee 0 <= plen <= 128 */
414 int o = plen >> 3,
415 b = plen & 0x7;
416
417 memset(pfx->s6_addr, 0, sizeof(pfx->s6_addr));
418 memcpy(pfx->s6_addr, addr, o);
419 if (b != 0)
420 pfx->s6_addr[o] = addr->s6_addr[o] & (0xff00 >> b);
421}
422
423static inline void ipv6_addr_prefix_copy(struct in6_addr *addr,
424 const struct in6_addr *pfx,
425 int plen)
426{
427 /* caller must guarantee 0 <= plen <= 128 */
428 int o = plen >> 3,
429 b = plen & 0x7;
430
431 memcpy(addr->s6_addr, pfx, o);
432 if (b != 0) {
433 addr->s6_addr[o] &= ~(0xff00 >> b);
434 addr->s6_addr[o] |= (pfx->s6_addr[o] & (0xff00 >> b));
435 }
436}

438static inline void __ipv6_addr_set_half(__be32 *addr,
439 __be32 wh, __be32 wl)
440{
441#if defined(CONFIG_HAVE_EFFICIENT_UNALIGNED_ACCESS) && BITS_PER_LONG == 64
442#if defined(__BIG_ENDIAN)
443 if (__builtin_constant_p(wh) && __builtin_constant_p(wl)) {
444 *(__force u64 *)addr = ((__force u64)(wh) << 32 | (__force u64)(wl));
445 return;
446 }
447#elif defined(__LITTLE_ENDIAN)
448 if (__builtin_constant_p(wl) && __builtin_constant_p(wh)) {
449 *(__force u64 *)addr = ((__force u64)(wl) << 32 | (__force u64)(wh));
450 return;
451 }
452#endif
453#endif
454 addr[0] = wh;
455 addr[1] = wl;
456}

466static inline bool ipv6_addr_equal(const struct in6_addr *a1,
467 const struct in6_addr *a2)
468{
469#if defined(CONFIG_HAVE_EFFICIENT_UNALIGNED_ACCESS) && BITS_PER_LONG == 64

473 return ((ul1[0] ^ ul2[0]) | (ul1[1] ^ ul2[1])) == 0UL;
474#else
475 return ((a1->s6_addr32[0] ^ a2->s6_addr32[0]) |
476 (a1->s6_addr32[1] ^ a2->s6_addr32[1]) |
477 (a1->s6_addr32[2] ^ a2->s6_addr32[2]) |
478 (a1->s6_addr32[3] ^ a2->s6_addr32[3])) == 0;
479#endif
480}

482#if defined(CONFIG_HAVE_EFFICIENT_UNALIGNED_ACCESS) && BITS_PER_LONG == 64
483static inline bool __ipv6_prefix_equal64_half(const __be64 *a1,
484 const __be64 *a2,
485 unsigned int len)
486{
487 if (len && ((*a1 ^ *a2) & cpu_to_be64((~0UL) << (64 - len))))
488 return false;
489 return true;
490}

492static inline bool ipv6_prefix_equal(const struct in6_addr *addr1,
493 const struct in6_addr *addr2,
494 unsigned int prefixlen)
495{
496 const __be64 *a1 = (const __be64 *)addr1;
497 const __be64 *a2 = (const __be64 *)addr2;
498
499 if (prefixlen >= 64) {
500 if (a1[0] ^ a2[0])
501 return false;
502 return __ipv6_prefix_equal64_half(a1 + 1, a2 + 1, prefixlen - 64);
503 }
504 return __ipv6_prefix_equal64_half(a1, a2, prefixlen);
505}
506#else
507static inline bool ipv6_prefix_equal(const struct in6_addr *addr1,
508 const struct in6_addr *addr2,
509 unsigned int prefixlen)
510{
511 const __be32 *a1 = addr1->s6_addr32;
512 const __be32 *a2 = addr2->s6_addr32;
513 unsigned int pdw, pbi;
514
515 /* check complete u32 in prefix */
516 pdw = prefixlen >> 5;
517 if (pdw && memcmp(a1, a2, pdw << 2))
518 return false;
519
520 /* check incomplete u32 in prefix */
521 pbi = prefixlen & 0x1f;
522 if (pbi && ((a1[pdw] ^ a2[pdw]) & htonl((0xffffffff) << (32 - pbi))))
523 return false;
524
525 return true;
526}
527#endif

585static inline u32 ipv6_addr_hash(const struct in6_addr *a)
586{
587#if defined(CONFIG_HAVE_EFFICIENT_UNALIGNED_ACCESS) && BITS_PER_LONG == 64
588 const unsigned long *ul = (const unsigned long *)a;
589 unsigned long x = ul[0] ^ ul[1];
590
591 return (u32)(x ^ (x >> 32));
592#else
593 return (__force u32)(a->s6_addr32[0] ^ a->s6_addr32[1] ^
594 a->s6_addr32[2] ^ a->s6_addr32[3]);
595#endif
596}

598/* more secured version of ipv6_addr_hash() */
599static inline u32 __ipv6_addr_jhash(const struct in6_addr *a, const u32 initval)
600{
601 u32 v = (__force u32)a->s6_addr32[0] ^ (__force u32)a->s6_addr32[1];
602
603 return jhash_3words(v,
604 (__force u32)a->s6_addr32[2],
605 (__force u32)a->s6_addr32[3],
606 initval);
607}

609static inline bool ipv6_addr_loopback(const struct in6_addr *a)
610{
611#if defined(CONFIG_HAVE_EFFICIENT_UNALIGNED_ACCESS) && BITS_PER_LONG == 64
612 const __be64 *be = (const __be64 *)a;
613
614 return (be[0] | (be[1] ^ cpu_to_be64(1))) == 0UL;
615#else
616 return (a->s6_addr32[0] | a->s6_addr32[1] |

660/*
661 * find the first different bit between two addresses
662 * length of address must be a multiple of 32bits
663 */
664static inline int __ipv6_addr_diff32(const void *token1, const void *token2, int addrlen)
665{
666 const __be32 *a1 = token1, *a2 = token2;
667 int i;
668
669 addrlen >>= 2;
670
671 for (i = 0; i < addrlen; i++) {
672 __be32 xb = a1[i] ^ a2[i];
673 if (xb)
674 return i * 32 + 31 - __fls(ntohl(xb));
675 }
676
677 /*
678 * we should *never* get to this point since that
679 * would mean the addrs are equal
680 *
681 * However, we do get to it 8) And exacly, when
682 * addresses are equal 8)
683 *
684 * ip route add 1111::/128 via ...
685 * ip route add 1111::/64 via ...
686 * and we are here.
687 *
688 * Ideally, this function should stop comparison
689 * at prefix length. It does not, but it is still OK,
690 * if returned value is greater than prefix length.
691 * --ANK (980803)
692 */
693 return addrlen << 5;
694}
695
696#if defined(CONFIG_HAVE_EFFICIENT_UNALIGNED_ACCESS) && BITS_PER_LONG == 64
697static inline int __ipv6_addr_diff64(const void *token1, const void *token2, int addrlen)
698{
699 const __be64 *a1 = token1, *a2 = token2;
700 int i;
701
702 addrlen >>= 3;
703
704 for (i = 0; i < addrlen; i++) {
705 __be64 xb = a1[i] ^ a2[i];
706 if (xb)
707 return i * 64 + 63 - __fls(be64_to_cpu(xb));
708 }
709
710 return addrlen << 6;
711}
712#endif
713
714static inline int __ipv6_addr_diff(const void *token1, const void *token2, int addrlen)
715{
716#if defined(CONFIG_HAVE_EFFICIENT_UNALIGNED_ACCESS) && BITS_PER_LONG == 64
717 if (__builtin_constant_p(addrlen) && !(addrlen & 7))
718 return __ipv6_addr_diff64(token1, token2, addrlen);
719#endif
720 return __ipv6_addr_diff32(token1, token2, addrlen);
721}

724{
725 return __ipv6_addr_diff(a1, a2, sizeof(struct in6_addr));
726}

775static inline __be32 ip6_make_flowlabel(struct net *net, struct sk_buff *skb,
776 __be32 flowlabel, bool autolabel,
777 struct flowi6 *fl6)
778{
779 u32 hash;
780
781 /* @flowlabel may include more than a flow label, eg, the traffic class.
782 * Here we want only the flow label value.
783 */
784 flowlabel &= IPV6_FLOWLABEL_MASK;
785
786 if (flowlabel ||
787 net->ipv6.sysctl.auto_flowlabels == IP6_AUTO_FLOW_LABEL_OFF ||
788 (!autolabel &&
789 net->ipv6.sysctl.auto_flowlabels != IP6_AUTO_FLOW_LABEL_FORCED))
790 return flowlabel;
791
792 hash = skb_get_hash_flowi6(skb, fl6);
793
794 /* Since this is being sent on the wire obfuscate hash a bit
795 * to minimize possbility that any useful information to an
796 * attacker is leaked. Only lower 20 bits are relevant.
797 */
798 hash = rol32(hash, 16);
799
800 flowlabel = (__force __be32)hash & IPV6_FLOWLABEL_MASK;
801
802 if (net->ipv6.sysctl.flowlabel_state_ranges)
803 flowlabel |= IPV6_FLOWLABEL_STATELESS_FLAG;
804
805 return flowlabel;
806}
