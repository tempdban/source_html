31static inline int INET_ECN_is_capable(__u8 dsfield)
32{
33 return dsfield & INET_ECN_ECT_0;

36/*
37 * RFC 3168 9.1.1
38 * The full-functionality option for ECN encapsulation is to copy the
39 * ECN codepoint of the inside header to the outside header on
40 * encapsulation if the inside header is not-ECT or ECT, and to set the
41 * ECN codepoint of the outside header to ECT(0) if the ECN codepoint of
42 * the inside header is CE.
43 */
44static inline __u8 INET_ECN_encapsulate(__u8 outer, __u8 inner)
45{
46 outer &= ~INET_ECN_MASK;
47 outer |= !INET_ECN_is_ce(inner) ? (inner & INET_ECN_MASK) :
48 INET_ECN_ECT_0;
49 return outer;
50}

75static inline int IP_ECN_set_ce(struct iphdr *iph)
76{
77 u32 check = (__force u32)iph->check;
78 u32 ecn = (iph->tos + 1) & INET_ECN_MASK;
79
80 /*
81 * After the last operation we have (in binary):
82 * INET_ECN_NOT_ECT => 01
83 * INET_ECN_ECT_1 => 10
84 * INET_ECN_ECT_0 => 11
85 * INET_ECN_CE => 00
86 */
87 if (!(ecn & 2))
88 return !ecn;
89
90 /*
91 * The following gives us:
92 * INET_ECN_ECT_1 => check += htons(0xFFFD)
93 * INET_ECN_ECT_0 => check += htons(0xFFFE)
94 */
95 check += (__force u16)htons(0xFFFB) + (__force u16)htons(ecn);
96
97 iph->check = (__force __sum16)(check + (check>=0xFFFF));
98 iph->tos |= INET_ECN_CE;
99 return 1;
100}

115/* Note:
116 * IP_ECN_set_ce() has to tweak IPV4 checksum when setting CE,
117 * meaning both changes have no effect on skb->csum if/when CHECKSUM_COMPLETE
118 * In IPv6 case, no checksum compensates the change in IPv6 header,
119 * so we have to update skb->csum.
120 */
121static inline int IP6_ECN_set_ce(struct sk_buff *skb, struct ipv6hdr *iph)
122{
123 __be32 from, to;
124
125 if (INET_ECN_is_not_ect(ipv6_get_dsfield(iph)))
126 return 0;
127
128 from = *(__be32 *)iph;
129 to = from | htonl(INET_ECN_CE << 20);
130 *(__be32 *)iph = to;
131 if (skb->ip_summed == CHECKSUM_COMPLETE)
132 skb->csum = csum_add(csum_sub(skb->csum, (__force __wsum)from),
133 (__force __wsum)to);
134 return 1;
135}

148static inline int INET_ECN_set_ce(struct sk_buff *skb)
149{
150 switch (skb->protocol) {
151 case cpu_to_be16(ETH_P_IP):
152 if (skb_network_header(skb) + sizeof(struct iphdr) <=
153 skb_tail_pointer(skb))
154 return IP_ECN_set_ce(ip_hdr(skb));
155 break;
156
157 case cpu_to_be16(ETH_P_IPV6):
158 if (skb_network_header(skb) + sizeof(struct ipv6hdr) <=
159 skb_tail_pointer(skb))
160 return IP6_ECN_set_ce(skb, ipv6_hdr(skb));
161 break;
162 }
163
164 return 0;
165}

167/*
168 * RFC 6040 4.2
169 * To decapsulate the inner header at the tunnel egress, a compliant
170 * tunnel egress MUST set the outgoing ECN field to the codepoint at the
171 * intersection of the appropriate arriving inner header (row) and outer
172 * header (column) in Figure 4
173 *
174 * +---------+------------------------------------------------+
175 * |Arriving | Arriving Outer Header |
176 * | Inner +---------+------------+------------+------------+
177 * | Header | Not-ECT | ECT(0) | ECT(1) | CE |
178 * +---------+---------+------------+------------+------------+
179 * | Not-ECT | Not-ECT |Not-ECT(!!!)|Not-ECT(!!!)| <drop>(!!!)|
180 * | ECT(0) | ECT(0) | ECT(0) | ECT(1) | CE |
181 * | ECT(1) | ECT(1) | ECT(1) (!) | ECT(1) | CE |
182 * | CE | CE | CE | CE(!!!)| CE |
183 * +---------+---------+------------+------------+------------+
184 *
185 * Figure 4: New IP in IP Decapsulation Behaviour
186 *
187 * returns 0 on success
188 * 1 if something is broken and should be logged (!!! above)
189 * 2 if packet should be dropped
190 */
191static inline int INET_ECN_decapsulate(struct sk_buff *skb,
192 __u8 outer, __u8 inner)
193{
194 if (INET_ECN_is_not_ect(inner)) {
195 switch (outer & INET_ECN_MASK) {
196 case INET_ECN_NOT_ECT:
197 return 0;
198 case INET_ECN_ECT_0:
199 case INET_ECN_ECT_1:
200 return 1;
201 case INET_ECN_CE:
202 return 2;
203 }
204 }
205
206 if (INET_ECN_is_ce(outer))
207 INET_ECN_set_ce(skb);
208
209 return 0;
210}

212static inline int IP_ECN_decapsulate(const struct iphdr *oiph,
213 struct sk_buff *skb)
214{
215 __u8 inner;
216
217 if (skb->protocol == htons(ETH_P_IP))
218 inner = ip_hdr(skb)->tos;
219 else if (skb->protocol == htons(ETH_P_IPV6))
220 inner = ipv6_get_dsfield(ipv6_hdr(skb));
221 else
222 return 0;
223
224 return INET_ECN_decapsulate(skb, oiph->tos, inner);
225}
