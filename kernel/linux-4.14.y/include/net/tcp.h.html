319static inline bool tcp_too_many_orphans(struct sock *sk, int shift)
320{
321 struct percpu_counter *ocp = sk->sk_prot->orphan_count;
322 int orphans = percpu_counter_read_positive(ocp);
323
324 if (orphans << shift > sysctl_tcp_max_orphans) {
325 orphans = percpu_counter_sum_positive(ocp);
326 if (orphans << shift > sysctl_tcp_max_orphans)
327 return true;
328 }
329 return false;
330}

484/* Syncookies use a monotonic timer which increments every 60 seconds.
485 * This counter is used both as a hash input and partially encoded into
486 * the cookie value. A cookie is only validated further if the delta
487 * between the current counter value and the encoded one is less than this,
488 * i.e. a sent cookie is valid only at most for 2*60 seconds (or less if
489 * the counter advances immediately after a cookie is generated).
490 */
491#define MAX_SYNCOOKIE_AGE 2
492#define TCP_SYNCOOKIE_PERIOD (60 * HZ)
493#define TCP_SYNCOOKIE_VALID (MAX_SYNCOOKIE_AGE * TCP_SYNCOOKIE_PERIOD)
494
495/* syncookies: remember time of last synqueue overflow
496 * But do not dirty this field too often (once per second is enough)
497 * It is racy as we do not hold a lock, but race is very minor.
498 */
499static inline void tcp_synq_overflow(const struct sock *sk)
500{
501 unsigned long last_overflow = tcp_sk(sk)->rx_opt.ts_recent_stamp;
502 unsigned long now = jiffies;
503
504 if (time_after(now, last_overflow + HZ))
505 tcp_sk(sk)->rx_opt.ts_recent_stamp = now;
506}
507
508/* syncookies: no recent synqueue overflow on this listening socket? */
509static inline bool tcp_synq_no_recent_overflow(const struct sock *sk)
510{
511 unsigned long last_overflow = tcp_sk(sk)->rx_opt.ts_recent_stamp;
512
513 return time_after(jiffies, last_overflow + TCP_SYNCOOKIE_VALID);
514}

590/* Bound MSS / TSO packet size with the half of the window */
591static inline int tcp_bound_to_half_wnd(struct tcp_sock *tp, int pktsize)
592{
593 int cutoff;
594
595 /* When peer uses tiny windows, there is no use in packetizing
596 * to sub-MSS pieces for the sake of SWS or making sure there
597 * are enough packets in the pipe for fast recovery.
598 *
599 * On the other hand, for extremely large MSS devices, handling
600 * smaller than MSS windows in this way does make sense.
601 */
602 if (tp->max_window > TCP_MSS_DEFAULT)
603 cutoff = (tp->max_window >> 1);
604 else
605 cutoff = tp->max_window;
606
607 if (cutoff && pktsize > cutoff)
608 return max_t(int, cutoff, 68U - tp->tcp_header_len);
609 else
610 return pktsize;
611}

1204/* We follow the spirit of RFC2861 to validate cwnd but implement a more
1205 * flexible approach. The RFC suggests cwnd should not be raised unless
1206 * it was fully used previously. And that's exactly what we do in
1207 * congestion avoidance mode. But in slow start we allow cwnd to grow
1208 * as long as the application has used half the cwnd.
1209 * Example :
1210 * cwnd is 10 (IW10), but application sends 9 frames.
1211 * We allow cwnd to reach 18 when all frames are ACKed.
1212 * This check is safe because it's as aggressive as slow start which already
1213 * risks 100% overshoot. The advantage is that we discourage application to
1214 * either send more filler packets or data to artificially blow up the cwnd
1215 * usage, and allow application-limited process to probe bw more aggressively.
1216 */
1217static inline bool tcp_is_cwnd_limited(const struct sock *sk)
1218{
1219 const struct tcp_sock *tp = tcp_sk(sk);
1220
1221 /* If in slow start, ensure cwnd grows to twice what was ACKed. */
1222 if (tcp_in_slow_start(tp))
1223 return tp->snd_cwnd < 2 * tp->max_packets_out;
1224
1225 return tp->is_cwnd_limited;
1226}

1229 * because qdisc is full or receiver sent a 0 window.
1230 * We do not want to add fuel to the fire, or abort too early,
1231 * so make sure the timer we arm now is at least 200ms in the future,
1232 * regardless of current icsk_rto value (as it could be ~2ms)
1233 */

1399static inline bool tcp_paws_check(const struct tcp_options_received *rx_opt,
1400 int paws_win)
1401{
1402 if ((s32)(rx_opt->ts_recent - rx_opt->rcv_tsval) <= paws_win)
1403 return true;
1404 if (unlikely(get_seconds() >= rx_opt->ts_recent_stamp + TCP_PAWS_24DAYS))
1405 return true;
1406 /*
1407 * Some OSes send SYN and SYNACK messages with tsval=0 tsecr=0,
1408 * then following tcp messages have valid values. Ignore 0 value,
1409 * or else 'negative' tsval might forbid us to accept their packets.
1410 */
1411 if (!rx_opt->ts_recent)
1412 return true;
1413 return false;
1414}

1416static inline bool tcp_paws_reject(const struct tcp_options_received *rx_opt,
1417 int rst)
1418{
1419 if (tcp_paws_check(rx_opt, 0))
1420 return false;
1421
1422 /* RST segments are not recommended to carry timestamp,
1423 and, if they do, it is recommended to ignore PAWS because
1424 "their cleanup function should take precedence over timestamps."
1425 Certainly, it is mistake. It is necessary to understand the reasons
1426 of this constraint to relax it: if peer reboots, clock may go
1427 out-of-sync and half-open connections will not be reset.
1428 Actually, the problem would be not existing if all
1429 the implementations followed draft about maintaining clock
1430 via reboots. Linux-2.2 DOES NOT!
1431
1432 However, we can relax time bounds for RST segments to MSL.
1433 */
1434 if (rst && get_seconds() >= rx_opt->ts_recent_stamp + TCP_PAWS_MSL)
1435 return false;
1436 return true;
1437}

1992static inline int tcp_inq(struct sock *sk)
1993{
1994 struct tcp_sock *tp = tcp_sk(sk);
1995 int answ;
1996
1997 if ((1 << sk->sk_state) & (TCPF_SYN_SENT | TCPF_SYN_RECV)) {
1998 answ = 0;
1999 } else if (sock_flag(sk, SOCK_URGINLINE) ||
2000 !tp->urg_data ||
2001 before(tp->urg_seq, tp->copied_seq) ||
2002 !before(tp->urg_seq, tp->rcv_nxt)) {
2003
2004 answ = tp->rcv_nxt - tp->copied_seq;
2005
2006 /* Subtract 1, if FIN was received */
2007 if (answ && sock_flag(sk, SOCK_DONE))
2008 answ--;
2009 } else {
2010 answ = tp->urg_seq - tp->copied_seq;
2011 }
2012
2013 return answ;
2014}
