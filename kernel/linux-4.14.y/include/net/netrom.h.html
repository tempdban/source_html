118/*********************************************************************
119 * nr_node & nr_neigh lists, refcounting and locking
120 *********************************************************************/
121
122#define nr_node_hold(__nr_node) \
123 refcount_inc(&((__nr_node)->refcount))
124
125static __inline__ void nr_node_put(struct nr_node *nr_node)
126{
127 if (refcount_dec_and_test(&nr_node->refcount)) {
128 kfree(nr_node);
129 }
130}
131
132#define nr_neigh_hold(__nr_neigh) \
133 refcount_inc(&((__nr_neigh)->refcount))
134
135static __inline__ void nr_neigh_put(struct nr_neigh *nr_neigh)
136{
137 if (refcount_dec_and_test(&nr_neigh->refcount)) {
138 if (nr_neigh->ax25)
139 ax25_cb_put(nr_neigh->ax25);
140 kfree(nr_neigh->digipeat);
141 kfree(nr_neigh);
142 }
143}
144
145/* nr_node_lock and nr_node_unlock also hold/put the node's refcounter.
146 */
147static __inline__ void nr_node_lock(struct nr_node *nr_node)
148{
149 nr_node_hold(nr_node);
150 spin_lock_bh(&nr_node->node_lock);
151}
152
153static __inline__ void nr_node_unlock(struct nr_node *nr_node)
154{
155 spin_unlock_bh(&nr_node->node_lock);
156 nr_node_put(nr_node);
157}

233/*
234 * This routine is called when a Connect Acknowledge with the Choke Flag
235 * set is needed to refuse a connection.
236 */
237#define nr_transmit_refusal(skb, mine) \
238do { \
239 __nr_transmit_reply((skb), (mine), NR_CONNACK | NR_CHOKE_FLAG); \
240} while (0)
241
242/*
243 * This routine is called when we don't have a circuit matching an incoming
244 * NET/ROM packet. This is an G8PZT Xrouter extension.
245 */
246#define nr_transmit_reset(skb, mine) \
247do { \
248 __nr_transmit_reply((skb), (mine), NR_RESET); \
249} while (0)
