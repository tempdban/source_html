246static inline struct fib_table *fib_get_table(struct net *net, u32 id)
247{
248 struct hlist_node *tb_hlist;
249 struct hlist_head *ptr;
250
251 ptr = id == RT_TABLE_LOCAL ?
252 &net->ipv4.fib_table_hash[TABLE_LOCAL_INDEX] :
253 &net->ipv4.fib_table_hash[TABLE_MAIN_INDEX];
254
255 tb_hlist = rcu_dereference_rtnl(hlist_first_rcu(ptr));
256
257 return hlist_entry(tb_hlist, struct fib_table, tb_hlist);
258}
259
260static inline struct fib_table *fib_new_table(struct net *net, u32 id)
261{
262 return fib_get_table(net, id);
263}

265static inline int fib_lookup(struct net *net, const struct flowi4 *flp,
266 struct fib_result *res, unsigned int flags)
267{
268 struct fib_table *tb;
269 int err = -ENETUNREACH;
270
271 rcu_read_lock();
272
273 tb = fib_get_table(net, RT_TABLE_MAIN);
274 if (tb)
275 err = fib_table_lookup(tb, flp, res, flags | FIB_LOOKUP_NOREF);
276
277 if (err == -EAGAIN)
278 err = -ENETUNREACH;
279
280 rcu_read_unlock();
281
282 return err;
283}

310static inline int fib_lookup(struct net *net, struct flowi4 *flp,
311 struct fib_result *res, unsigned int flags)
312{
313 struct fib_table *tb;
314 int err = -ENETUNREACH;
315
316 flags |= FIB_LOOKUP_NOREF;
317 if (net->ipv4.fib_has_custom_rules)
318 return __fib_lookup(net, flp, res, flags);
319
320 rcu_read_lock();
321
322 res->tclassid = 0;
323
324 tb = rcu_dereference_rtnl(net->ipv4.fib_main);
325 if (tb)
326 err = fib_table_lookup(tb, flp, res, flags);
327
328 if (!err)
329 goto out;
330
331 tb = rcu_dereference_rtnl(net->ipv4.fib_default);
332 if (tb)
333 err = fib_table_lookup(tb, flp, res, flags);
334
335out:
336 if (err == -EAGAIN)
337 err = -ENETUNREACH;
338
339 rcu_read_unlock();
340
341 return err;
342}

388static inline void fib_combine_itag(u32 *itag, const struct fib_result *res)
389{
390#ifdef CONFIG_IP_ROUTE_CLASSID
391#ifdef CONFIG_IP_MULTIPLE_TABLES
392 u32 rtag;
393#endif
394 *itag = FIB_RES_NH(*res).nh_tclassid<<16;
395#ifdef CONFIG_IP_MULTIPLE_TABLES
396 rtag = res->tclassid;
397 if (*itag == 0)
398 *itag = (rtag<<16);
399 *itag |= (rtag>>16);
400#endif
401#endif
402}
