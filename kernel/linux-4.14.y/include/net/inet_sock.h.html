32/** struct ip_options - IP Options
33 *
34 * @faddr - Saved first hop address
35 * @nexthop - Saved nexthop address in LSRR and SSRR
36 * @is_strictroute - Strict source route
37 * @srr_is_hit - Packet destination addr was our one
38 * @is_changed - IP checksum more not valid
39 * @rr_needaddr - Need to record addr of outgoing dev
40 * @ts_needtime - Need to record timestamp
41 * @ts_needaddr - Need to record addr of outgoing dev
42 */
43struct ip_options {
44 __be32 faddr;
45 __be32 nexthop;
46 unsigned char optlen;
47 unsigned char srr;
48 unsigned char rr;
49 unsigned char ts;
50 unsigned char is_strictroute:1,
51 srr_is_hit:1,
52 is_changed:1,
53 rr_needaddr:1,
54 ts_needtime:1,
55 ts_needaddr:1;
56 unsigned char router_alert;
57 unsigned char cipso;
58 unsigned char __pad2;
59 unsigned char __data[0];
60};

160/** struct inet_sock - representation of INET sockets
161 *
162 * @sk - ancestor class
163 * @pinet6 - pointer to IPv6 control block
164 * @inet_daddr - Foreign IPv4 addr
165 * @inet_rcv_saddr - Bound local IPv4 addr
166 * @inet_dport - Destination port
167 * @inet_num - Local port
168 * @inet_saddr - Sending source
169 * @uc_ttl - Unicast TTL
170 * @inet_sport - Source port
171 * @inet_id - ID counter for DF pkts
172 * @tos - TOS
173 * @mc_ttl - Multicasting TTL
174 * @is_icsk - is this an inet_connection_sock?
175 * @uc_index - Unicast outgoing device index
176 * @mc_index - Multicast device index
177 * @mc_list - Group array
178 * @cork - info to build ip hdr on each ip frag while socket is corked
179 */
180struct inet_sock {
181 /* sk and pinet6 has to be the first two members of inet_sock */
182 struct sock sk;
183#if IS_ENABLED(CONFIG_IPV6)
184 struct ipv6_pinfo *pinet6;
185#endif
186 /* Socket demultiplex comparisons on incoming packets. */
187#define inet_daddr sk.__sk_common.skc_daddr
188#define inet_rcv_saddr sk.__sk_common.skc_rcv_saddr
189#define inet_dport sk.__sk_common.skc_dport
190#define inet_num sk.__sk_common.skc_num
191
192 __be32 inet_saddr;
193 __s16 uc_ttl;
194 __u16 cmsg_flags;
195 __be16 inet_sport;
196 __u16 inet_id;
197
198 struct ip_options_rcu __rcu *inet_opt;
199 int rx_dst_ifindex;
200 __u8 tos;
201 __u8 min_ttl;
202 __u8 mc_ttl;
203 __u8 pmtudisc;
204 __u8 recverr:1,
205 is_icsk:1,
206 freebind:1,
207 hdrincl:1,
208 mc_loop:1,
209 transparent:1,
210 mc_all:1,
211 nodefrag:1;
212 __u8 bind_address_no_port:1,
213 defer_connect:1; /* Indicates that fastopen_connect is set
214 * and cookie exists so we defer connect
215 * until first data frame is written
216 */
217 __u8 rcv_tos;
218 __u8 convert_csum;
219 int uc_index;
220 int mc_index;
221 __be32 mc_addr;
222 struct ip_mc_socklist __rcu *mc_list;
223 struct inet_cork_full cork;
224};

240/**
241 * sk_to_full_sk - Access to a full socket
242 * @sk: pointer to a socket
243 *
244 * SYNACK messages might be attached to request sockets.
245 * Some places want to reach the listener in this case.
246 */
247static inline struct sock *sk_to_full_sk(struct sock *sk)
248{
249#ifdef CONFIG_INET
250 if (sk && sk->sk_state == TCP_NEW_SYN_RECV)
251 sk = inet_reqsk(sk)->rsk_listener;
252#endif
253 return sk;
254}
255
256/* sk_to_full_sk() variant with a const argument */
257static inline const struct sock *sk_const_to_full_sk(const struct sock *sk)
258{
259#ifdef CONFIG_INET
260 if (sk && sk->sk_state == TCP_NEW_SYN_RECV)
261 sk = ((const struct request_sock *)sk)->rsk_listener;
262#endif
263 return sk;
264}
265
266static inline struct sock *skb_to_full_sk(const struct sk_buff *skb)
267{
268 return sk_to_full_sk(skb->sk);
269}

293static inline unsigned int __inet_ehashfn(const __be32 laddr,
294 const __u16 lport,
295 const __be32 faddr,
296 const __be16 fport,
297 u32 initval)
298{
299 return jhash_3words((__force __u32) laddr,
300 (__force __u32) faddr,
301 ((__u32) lport) << 16 | (__force __u32)fport,
302 initval);
303}

309static inline __u8 inet_sk_flowi_flags(const struct sock *sk)
310{
311 __u8 flags = 0;
312

314 flags |= FLOWI_FLAG_ANYSRC;
315 return flags;
316}
