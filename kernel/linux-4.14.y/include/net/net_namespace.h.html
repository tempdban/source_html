202static inline struct net *maybe_get_net(struct net *net)
203{
204 /* Used when we know struct net exists but we
205 * aren't guaranteed a previous reference count
206 * exists. If the reference count is zero this
207 * function fails and returns NULL.
208 */
209 if (!atomic_inc_not_zero(&net->count))
210 net = NULL;
211 return net;
212}

318/*
319 * Use these carefully. If you implement a network device and it
320 * needs per network namespace operations use device pernet operations,
321 * otherwise use pernet subsys operations.
322 *
323 * Network interfaces need to be removed from a dying netns _before_
324 * subsys notifiers can be called, as most of the network code cleanup
325 * (which is done from subsys notifiers) runs with the assumption that
326 * dev_remove_pack has been called so no new packets will arrive during
327 * and after the cleanup functions have been called. dev_remove_pack
328 * is not per namespace so instead the guarantee of no more packets
329 * arriving in a network namespace is provided by ensuring that all
330 * network devices and all sockets have left the network namespace
331 * before the cleanup methods are called.
332 *
333 * For the longest time the ipv4 icmp code was registered as a pernet
334 * device which caused kernel oops, and panics during network
335 * namespace cleanup. So please don't get this wrong.
336 */
337int register_pernet_subsys(struct pernet_operations *);
338void unregister_pernet_subsys(struct pernet_operations *);
339int register_pernet_device(struct pernet_operations *);
340void unregister_pernet_device(struct pernet_operations *);
