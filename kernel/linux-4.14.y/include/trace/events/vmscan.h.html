14#define RECLAIM_WB_ANON 0x0001u
15#define RECLAIM_WB_FILE 0x0002u
16#define RECLAIM_WB_MIXED 0x0010u
17#define RECLAIM_WB_SYNC 0x0004u /* Unused, all reclaim async */
18#define RECLAIM_WB_ASYNC 0x0008u
19#define RECLAIM_WB_LRU (RECLAIM_WB_ANON|RECLAIM_WB_FILE)
20
21#define show_reclaim_flags(flags) \
22 (flags) ? __print_flags(flags, "|", \
23 {RECLAIM_WB_ANON, "RECLAIM_WB_ANON"}, \
24 {RECLAIM_WB_FILE, "RECLAIM_WB_FILE"}, \
25 {RECLAIM_WB_MIXED, "RECLAIM_WB_MIXED"}, \
26 {RECLAIM_WB_SYNC, "RECLAIM_WB_SYNC"}, \
27 {RECLAIM_WB_ASYNC, "RECLAIM_WB_ASYNC"} \
28 ) : "RECLAIM_WB_NONE"
29
30#define trace_reclaim_flags(page) ( \
31 (page_is_file_cache(page) ? RECLAIM_WB_FILE : RECLAIM_WB_ANON) | \
32 (RECLAIM_WB_ASYNC) \
33 )
34
35#define trace_shrink_flags(file) \
36 ( \
37 (file ? RECLAIM_WB_FILE : RECLAIM_WB_ANON) | \
38 (RECLAIM_WB_ASYNC) \
39 )

190 TP_PROTO(struct shrinker *shr, struct shrink_control *sc,
191 long nr_objects_to_shrink, unsigned long pgs_scanned,
192 unsigned long lru_pgs, unsigned long cache_items,
193 unsigned long long delta, unsigned long total_scan),
194
195 TP_ARGS(shr, sc, nr_objects_to_shrink, pgs_scanned, lru_pgs,
196 cache_items, delta, total_scan),
197
198 TP_STRUCT__entry(
199 __field(struct shrinker *, shr)
200 __field(void *, shrink)
201 __field(int, nid)
202 __field(long, nr_objects_to_shrink)
203 __field(gfp_t, gfp_flags)
204 __field(unsigned long, pgs_scanned)
205 __field(unsigned long, lru_pgs)
206 __field(unsigned long, cache_items)
207 __field(unsigned long long, delta)
208 __field(unsigned long, total_scan)
209 ),
210
211 TP_fast_assign(
212 __entry->shr = shr;
213 __entry->shrink = shr->scan_objects;
214 __entry->nid = sc->nid;
215 __entry->nr_objects_to_shrink = nr_objects_to_shrink;
216 __entry->gfp_flags = sc->gfp_mask;
217 __entry->pgs_scanned = pgs_scanned;
218 __entry->lru_pgs = lru_pgs;
219 __entry->cache_items = cache_items;
220 __entry->delta = delta;
221 __entry->total_scan = total_scan;
222 ),
223
224 TP_printk("%pF %p: nid: %d objects to shrink %ld gfp_flags %s pgs_scanned %ld lru_pgs %ld cache items %ld delta %lld total_scan %ld",
225 __entry->shrink,
226 __entry->shr,
227 __entry->nid,
228 __entry->nr_objects_to_shrink,
229 show_gfp_flags(__entry->gfp_flags),
