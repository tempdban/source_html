56DECLARE_EVENT_CLASS(bpf_prog_event,
57
58 TP_PROTO(const struct bpf_prog *prg),
59
60 TP_ARGS(prg),
61
62 TP_STRUCT__entry(
63 __array(u8, prog_tag, 8)
64 __field(u32, type)
65 ),
66
67 TP_fast_assign(
68 BUILD_BUG_ON(sizeof(__entry->prog_tag) != sizeof(prg->tag));
69 memcpy(__entry->prog_tag, prg->tag, sizeof(prg->tag));
70 __entry->type = prg->type;
71 ),
72
73 TP_printk("prog=%s type=%s",
74 __print_hex_str(__entry->prog_tag, 8),
75 __print_symbolic(__entry->type, __PROG_TYPE_SYM_TAB))
76);

228DECLARE_EVENT_CLASS(bpf_map_keyval,
229
230 TP_PROTO(const struct bpf_map *map, int ufd,
231 const void *key, const void *val),
232
233 TP_ARGS(map, ufd, key, val),
234
235 TP_STRUCT__entry(
236 __field(u32, type)
237 __field(u32, key_len)
238 __dynamic_array(u8, key, map->key_size)
239 __field(bool, key_trunc)
240 __field(u32, val_len)
241 __dynamic_array(u8, val, map->value_size)
242 __field(bool, val_trunc)
243 __field(int, ufd)
244 ),
245
246 TP_fast_assign(
247 memcpy(__get_dynamic_array(key), key, map->key_size);
248 memcpy(__get_dynamic_array(val), val, map->value_size);
249 __entry->type = map->map_type;
250 __entry->key_len = min(map->key_size, 16U);
251 __entry->key_trunc = map->key_size != __entry->key_len;
252 __entry->val_len = min(map->value_size, 16U);
253 __entry->val_trunc = map->value_size != __entry->val_len;
254 __entry->ufd = ufd;
255 ),
256
257 TP_printk("map type=%s ufd=%d key=[%s%s] val=[%s%s]",
258 __print_symbolic(__entry->type, __MAP_TYPE_SYM_TAB),
259 __entry->ufd,
260 __print_hex(__get_dynamic_array(key), __entry->key_len),
261 __entry->key_trunc ? " ..." : "",
262 __print_hex(__get_dynamic_array(val), __entry->val_len),
263 __entry->val_trunc ? " ..." : "")
264);

312TRACE_EVENT(bpf_map_next_key,
313
314 TP_PROTO(const struct bpf_map *map, int ufd,
315 const void *key, const void *key_next),
316
317 TP_ARGS(map, ufd, key, key_next),
318
319 TP_STRUCT__entry(
320 __field(u32, type)
321 __field(u32, key_len)
322 __dynamic_array(u8, key, map->key_size)
323 __dynamic_array(u8, nxt, map->key_size)
324 __field(bool, key_trunc)
325 __field(bool, key_null)
326 __field(int, ufd)
327 ),
328
329 TP_fast_assign(
330 if (key)
331 memcpy(__get_dynamic_array(key), key, map->key_size);
332 __entry->key_null = !key;
333 memcpy(__get_dynamic_array(nxt), key_next, map->key_size);
334 __entry->type = map->map_type;
335 __entry->key_len = min(map->key_size, 16U);
336 __entry->key_trunc = map->key_size != __entry->key_len;
337 __entry->ufd = ufd;
338 ),
339
340 TP_printk("map type=%s ufd=%d key=[%s%s] next=[%s%s]",
341 __print_symbolic(__entry->type, __MAP_TYPE_SYM_TAB),
342 __entry->ufd,
343 __entry->key_null ? "NULL" : __print_hex(__get_dynamic_array(key),
344 __entry->key_len),
345 __entry->key_trunc && !__entry->key_null ? " ..." : "",
346 __print_hex(__get_dynamic_array(nxt), __entry->key_len),
347 __entry->key_trunc ? " ..." : "")
348);
