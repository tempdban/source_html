45/**
46 * struct iscsi_transport - iSCSI Transport template
47 *
48 * @name: transport name
49 * @caps: iSCSI Data-Path capabilities
50 * @create_session: create new iSCSI session object
51 * @destroy_session: destroy existing iSCSI session object
52 * @create_conn: create new iSCSI connection
53 * @bind_conn: associate this connection with existing iSCSI session
54 * and specified transport descriptor
55 * @destroy_conn: destroy inactive iSCSI connection
56 * @set_param: set iSCSI parameter. Return 0 on success, -ENODATA
57 * when param is not supported, and a -Exx value on other
58 * error.
59 * @get_param get iSCSI parameter. Must return number of bytes
60 * copied to buffer on success, -ENODATA when param
61 * is not supported, and a -Exx value on other error
62 * @start_conn: set connection to be operational
63 * @stop_conn: suspend/recover/terminate connection
64 * @send_pdu: send iSCSI PDU, Login, Logout, NOP-Out, Reject, Text.
65 * @session_recovery_timedout: notify LLD a block during recovery timed out
66 * @init_task: Initialize a iscsi_task and any internal structs.
67 * When offloading the data path, this is called from
68 * queuecommand with the session lock, or from the
69 * iscsi_conn_send_pdu context with the session lock.
70 * When not offloading the data path, this is called
71 * from the scsi work queue without the session lock.
72 * @xmit_task Requests LLD to transfer cmd task. Returns 0 or the
73 * the number of bytes transferred on success, and -Exyz
74 * value on error. When offloading the data path, this
75 * is called from queuecommand with the session lock, or
76 * from the iscsi_conn_send_pdu context with the session
77 * lock. When not offloading the data path, this is called
78 * from the scsi work queue without the session lock.
79 * @cleanup_task: requests LLD to fail task. Called with session lock
80 * and after the connection has been suspended and
81 * terminated during recovery. If called
82 * from abort task then connection is not suspended
83 * or terminated but sk_callback_lock is held
84 *
85 * Template API provided by iSCSI Transport
86 */
