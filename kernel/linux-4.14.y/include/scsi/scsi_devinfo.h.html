2#ifndef _SCSI_SCSI_DEVINFO_H
3#define _SCSI_SCSI_DEVINFO_H
4/*
5 * Flags for SCSI devices that need special treatment
6 */
7#define BLIST_NOLUN 0x001 /* Only scan LUN 0 */
8#define BLIST_FORCELUN 0x002 /* Known to have LUNs, force scanning,
9 deprecated: Use max_luns=N */
10#define BLIST_BORKEN 0x004 /* Flag for broken handshaking */
11#define BLIST_KEY 0x008 /* unlock by special command */
12#define BLIST_SINGLELUN 0x010 /* Do not use LUNs in parallel */
13#define BLIST_NOTQ 0x020 /* Buggy Tagged Command Queuing */
14#define BLIST_SPARSELUN 0x040 /* Non consecutive LUN numbering */
15#define BLIST_MAX5LUN 0x080 /* Avoid LUNS >= 5 */
16#define BLIST_ISROM 0x100 /* Treat as (removable) CD-ROM */
17#define BLIST_LARGELUN 0x200 /* LUNs past 7 on a SCSI-2 device */
18#define BLIST_INQUIRY_36 0x400 /* override additional length field */
19#define BLIST_NOSTARTONADD 0x1000 /* do not do automatic start on add */
20#define BLIST_REPORTLUN2 0x20000 /* try REPORT_LUNS even for SCSI-2 devs
21 (if HBA supports more than 8 LUNs) */
22#define BLIST_NOREPORTLUN 0x40000 /* don't try REPORT_LUNS scan (SCSI-3 devs) */
23#define BLIST_NOT_LOCKABLE 0x80000 /* don't use PREVENT-ALLOW commands */
24#define BLIST_NO_ULD_ATTACH 0x100000 /* device is actually for RAID config */
25#define BLIST_SELECT_NO_ATN 0x200000 /* select without ATN */
26#define BLIST_RETRY_HWERROR 0x400000 /* retry HARDWARE_ERROR */
27#define BLIST_MAX_512 0x800000 /* maximum 512 sector cdb length */
28#define BLIST_NO_DIF 0x2000000 /* Disable T10 PI (DIF) */
29#define BLIST_SKIP_VPD_PAGES 0x4000000 /* Ignore SBC-3 VPD pages */
30#define BLIST_TRY_VPD_PAGES 0x10000000 /* Attempt to read VPD pages */
31#define BLIST_NO_RSOC 0x20000000 /* don't try to issue RSOC */
32#define BLIST_MAX_1024 0x40000000 /* maximum 1024 sector cdb length */
33#define BLIST_UNMAP_LIMIT_WS 0x80000000 /* Use UNMAP limit for WRITE SAME */
34
35#endif
