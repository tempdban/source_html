23#ifndef _PTP_CLASSIFY_H_
24#define _PTP_CLASSIFY_H_
25
26#include <linux/ip.h>
27#include <linux/skbuff.h>
28
29#define PTP_CLASS_NONE 0x00 /* not a PTP event message */
30#define PTP_CLASS_V1 0x01 /* protocol version 1 */
31#define PTP_CLASS_V2 0x02 /* protocol version 2 */
32#define PTP_CLASS_VMASK 0x0f /* max protocol version is 15 */
33#define PTP_CLASS_IPV4 0x10 /* event in an IPV4 UDP packet */
34#define PTP_CLASS_IPV6 0x20 /* event in an IPV6 UDP packet */
35#define PTP_CLASS_L2 0x40 /* event in a L2 packet */
36#define PTP_CLASS_PMASK 0x70 /* mask for the packet type field */
37#define PTP_CLASS_VLAN 0x80 /* event in a VLAN tagged packet */
38
39#define PTP_CLASS_V1_IPV4 (PTP_CLASS_V1 | PTP_CLASS_IPV4)
40#define PTP_CLASS_V1_IPV6 (PTP_CLASS_V1 | PTP_CLASS_IPV6) /* probably DNE */
41#define PTP_CLASS_V2_IPV4 (PTP_CLASS_V2 | PTP_CLASS_IPV4)
42#define PTP_CLASS_V2_IPV6 (PTP_CLASS_V2 | PTP_CLASS_IPV6)
43#define PTP_CLASS_V2_L2 (PTP_CLASS_V2 | PTP_CLASS_L2)
44#define PTP_CLASS_V2_VLAN (PTP_CLASS_V2 | PTP_CLASS_VLAN)
45#define PTP_CLASS_L4 (PTP_CLASS_IPV4 | PTP_CLASS_IPV6)
46
47#define PTP_EV_PORT 319
48#define PTP_GEN_BIT 0x08 /* indicates general message, if set in message type */
49
50#define OFF_PTP_SOURCE_UUID 22 /* PTPv1 only */
51#define OFF_PTP_SEQUENCE_ID 30
52#define OFF_PTP_CONTROL 32 /* PTPv1 only */
53
54/* Below defines should actually be removed at some point in time. */
55#define IP6_HLEN 40
56#define UDP_HLEN 8
57#define OFF_IHL 14
58#define IPV4_HLEN(data) (((struct iphdr *)(data + OFF_IHL))->ihl << 2)
59
60#if defined(CONFIG_NET_PTP_CLASSIFY)
61/**
62 * ptp_classify_raw - classify a PTP packet
63 * @skb: buffer
64 *
65 * Runs a minimal BPF dissector to classify a network packet to
66 * determine the PTP class. In case the skb does not contain any
67 * PTP protocol data, PTP_CLASS_NONE will be returned, otherwise
68 * PTP_CLASS_V1_IPV{4,6}, PTP_CLASS_V2_IPV{4,6} or
69 * PTP_CLASS_V2_{L2,VLAN}, depending on the packet content.
70 */
71unsigned int ptp_classify_raw(const struct sk_buff *skb);
72
73void __init ptp_classifier_init(void);
74#else
75static inline void ptp_classifier_init(void)
76{
77}
78#endif
79#endif /* _PTP_CLASSIFY_H_ */
