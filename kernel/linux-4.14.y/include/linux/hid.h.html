41/*
42 * We parse each description item into this structure. Short items data
43 * values are expanded to 32-bit signed int, long items contain a pointer
44 * into the data area.
45 */
46
47struct hid_item {
48 unsigned format;
49 __u8 size;
50 __u8 type;
51 __u8 tag;
52 union {
53 __u8 u8;
54 __s8 s8;
55 __u16 u16;
56 __s16 s16;
57 __u32 u32;
58 __s32 s32;
59 __u8 *longdata;
60 } data;
61};

408/*
409 * This is the collection stack. We climb up the stack to determine
410 * application and function of each field.
411 */
412
413struct hid_collection {
414 unsigned type;
415 unsigned usage;
416 unsigned level;
417};
418
419struct hid_usage {
420 unsigned hid; /* hid usage code */
421 unsigned collection_index; /* index into collection array */
422 unsigned usage_index; /* index into usage array */
423 /* hidinput data */
424 __u16 code; /* input driver code */
425 __u8 type; /* input driver type */
426 __s8 hat_min; /* hat switch fun */
427 __s8 hat_max; /* ditto */
428 __s8 hat_dir; /* ditto */
429};
430
431struct hid_input;
432
433struct hid_field {
434 unsigned physical; /* physical usage for this field */
435 unsigned logical; /* logical usage for this field */
436 unsigned application; /* application usage for this field */
437 struct hid_usage *usage; /* usage table for this function */
438 unsigned maxusage; /* maximum usage index */
439 unsigned flags; /* main-item flags (i.e. volatile,array,constant) */
440 unsigned report_offset; /* bit offset in the report */
441 unsigned report_size; /* size of this field in the report */
442 unsigned report_count; /* number of this field in the report */
443 unsigned report_type; /* (input,output,feature) */
444 __s32 *value; /* last known value(s) */
445 __s32 logical_minimum;
446 __s32 logical_maximum;
447 __s32 physical_minimum;
448 __s32 physical_maximum;
449 __s32 unit_exponent;
450 unsigned unit;
451 struct hid_report *report; /* associated report */
452 unsigned index; /* index into report->field[] */
453 /* hidinput data */
454 struct hid_input *hidinput; /* associated input structure */
455 __u16 dpad; /* dpad input code */
456};

673/**
674 * struct hid_driver
675 * @name: driver name (e.g. "Footech_bar-wheel")
676 * @id_table: which devices is this driver for (must be non-NULL for probe
677 * to be called)
678 * @dyn_list: list of dynamically added device ids
679 * @dyn_lock: lock protecting @dyn_list
680 * @probe: new device inserted
681 * @remove: device removed (NULL if not a hot-plug capable driver)
682 * @report_table: on which reports to call raw_event (NULL means all)
683 * @raw_event: if report in report_table, this hook is called (NULL means nop)
684 * @usage_table: on which events to call event (NULL means all)
685 * @event: if usage in usage_table, this hook is called (NULL means nop)
686 * @report: this hook is called after parsing a report (NULL means nop)
687 * @report_fixup: called before report descriptor parsing (NULL means nop)
688 * @input_mapping: invoked on input registering before mapping an usage
689 * @input_mapped: invoked on input registering after mapping an usage
690 * @input_configured: invoked just before the device is registered
691 * @feature_mapping: invoked on feature registering
692 * @suspend: invoked on suspend (NULL means nop)
693 * @resume: invoked on resume if device was not reset (NULL means nop)
694 * @reset_resume: invoked on resume if device was reset (NULL means nop)
695 *
696 * probe should return -errno on error, or 0 on success. During probe,
697 * input will not be passed to raw_event unless hid_device_io_start is
698 * called.
699 *
700 * raw_event and event should return 0 on no action performed, 1 when no
701 * further processing should be done and negative on error
702 *
703 * input_mapping shall return a negative value to completely ignore this usage
704 * (e.g. doubled or invalid usage), zero to continue with parsing of this
705 * usage by generic code (no special handling needed) or positive to skip
706 * generic parsing (needed special handling which was done in the hook already)
707 * input_mapped shall return negative to inform the layer that this usage
708 * should not be considered for further processing or zero to notify that
709 * no processing was performed and should be done in a generic manner
710 * Both these functions may be NULL which means the same behavior as returning
711 * zero from them.
712 */

714 char *name;
715 const struct hid_device_id *id_table;
716
717 struct list_head dyn_list;
718 spinlock_t dyn_lock;
719
720 int (*probe)(struct hid_device *dev, const struct hid_device_id *id);
721 void (*remove)(struct hid_device *dev);
722
723 const struct hid_report_id *report_table;
724 int (*raw_event)(struct hid_device *hdev, struct hid_report *report,
725 u8 *data, int size);
726 const struct hid_usage_id *usage_table;
727 int (*event)(struct hid_device *hdev, struct hid_field *field,
728 struct hid_usage *usage, __s32 value);
729 void (*report)(struct hid_device *hdev, struct hid_report *report);
730
731 __u8 *(*report_fixup)(struct hid_device *hdev, __u8 *buf,
732 unsigned int *size);
733
734 int (*input_mapping)(struct hid_device *hdev,
735 struct hid_input *hidinput, struct hid_field *field,
736 struct hid_usage *usage, unsigned long **bit, int *max);
737 int (*input_mapped)(struct hid_device *hdev,
738 struct hid_input *hidinput, struct hid_field *field,
739 struct hid_usage *usage, unsigned long **bit, int *max);
740 int (*input_configured)(struct hid_device *hdev,
741 struct hid_input *hidinput);
742 void (*feature_mapping)(struct hid_device *hdev,
743 struct hid_field *field,
744 struct hid_usage *usage);
745#ifdef CONFIG_PM
746 int (*suspend)(struct hid_device *hdev, pm_message_t message);
747 int (*resume)(struct hid_device *hdev);

749#endif

751 struct device_driver driver;
752};

874/**
875 * hid_device_io_start - enable HID input during probe, remove
876 *
877 * @hid - the device
878 *
879 * This should only be called during probe or remove and only be
880 * called by the thread calling probe or remove. It will allow
881 * incoming packets to be delivered to the driver.

892/**
893 * hid_device_io_stop - disable HID input during probe, remove
894 *
895 * @hid - the device
896 *
897 * Should only be called after hid_device_io_start. It will prevent
898 * incoming packets from going to the driver for the duration of
899 * probe, remove. If called during probe, packets will still go to the
900 * driver after probe is complete. This function should only be called
901 * by the thread calling probe or remove.
902 */
903static inline void hid_device_io_stop(struct hid_device *hid) {
904 if (!hid->io_started) {
905 dev_warn(&hid->dev, "io already stopped\n");
906 return;
907 }
908 hid->io_started = false;
909 down(&hid->driver_input_lock);
910}

912/**
913 * hid_map_usage - map usage input bits
914 *
915 * @hidinput: hidinput which we are interested in
916 * @usage: usage to fill in
917 * @bit: pointer to input->{}bit (out parameter)
918 * @max: maximal valid usage->code to consider later (out parameter)
919 * @type: input event type (EV_KEY, EV_REL, ...)
920 * @c: code which corresponds to this usage and type
921 */
922static inline void hid_map_usage(struct hid_input *hidinput,
923 struct hid_usage *usage, unsigned long **bit, int *max,
924 __u8 type, __u16 c)
925{
926 struct input_dev *input = hidinput->input;
927
928 usage->type = type;
929 usage->code = c;
930
931 switch (type) {
932 case EV_ABS:
933 *bit = input->absbit;
934 *max = ABS_MAX;
935 break;
936 case EV_REL:
937 *bit = input->relbit;
938 *max = REL_MAX;
939 break;
940 case EV_KEY:
941 *bit = input->keybit;
942 *max = KEY_MAX;
943 break;
944 case EV_LED:
945 *bit = input->ledbit;
946 *max = LED_MAX;
947 break;
948 }
949}

952 * hid_map_usage_clear - map usage input bits and clear the input bit
953 *
954 * The same as hid_map_usage, except the @c bit is also cleared in supported
955 * bits (@bit).

957static inline void hid_map_usage_clear(struct hid_input *hidinput,
958 struct hid_usage *usage, unsigned long **bit, int *max,
959 __u8 type, __u16 c)
960{
961 hid_map_usage(hidinput, usage, bit, max, type, c);
