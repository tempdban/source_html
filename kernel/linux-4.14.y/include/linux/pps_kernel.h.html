29/*
30 * Global defines

32
33struct pps_device;

35/* The specific PPS source info */
36struct pps_source_info {
37 char name[PPS_MAX_NAME_LEN]; /* symbolic name */
38 char path[PPS_MAX_NAME_LEN]; /* path of connected device */
39 int mode; /* PPS allowed mode */
40
41 void (*echo)(struct pps_device *pps,
42 int event, void *data); /* PPS echo function */
43
44 struct module *owner;
45 struct device *dev; /* Parent device for device_create */
46};
47
48struct pps_event_time {
49#ifdef CONFIG_NTP_PPS
50 struct timespec64 ts_raw;
51#endif /* CONFIG_NTP_PPS */
52 struct timespec64 ts_real;
53};
54
55/* The main struct */
56struct pps_device {
57 struct pps_source_info info; /* PSS source info */
58
59 struct pps_kparams params; /* PPS current params */
60
61 __u32 assert_sequence; /* PPS assert event seq # */
62 __u32 clear_sequence; /* PPS clear event seq # */
63 struct pps_ktime assert_tu;
64 struct pps_ktime clear_tu;
65 int current_mode; /* PPS mode at event time */
66
67 unsigned int last_ev; /* last PPS event id */
68 wait_queue_head_t queue; /* PPS event queue */
69
70 unsigned int id; /* PPS source unique ID */
71 void const *lookup_cookie; /* For pps_lookup_dev() only */
72 struct cdev cdev;
73 struct device *dev;
74 struct fasync_struct *async_queue; /* fasync method */
75 spinlock_t lock;
76};

106static inline void timespec_to_pps_ktime(struct pps_ktime *kt,
107 struct timespec64 ts)
108{
109 kt->sec = ts.tv_sec;
110 kt->nsec = ts.tv_nsec;
111}
112
113static inline void pps_get_ts(struct pps_event_time *ts)
114{
115 struct system_time_snapshot snap;
116
117 ktime_get_snapshot(&snap);
118 ts->ts_real = ktime_to_timespec64(snap.real);
119#ifdef CONFIG_NTP_PPS
120 ts->ts_raw = ktime_to_timespec64(snap.raw);
121#endif
122}
123
124/* Subtract known time delay from PPS event time(s) */
125static inline void pps_sub_ts(struct pps_event_time *ts, struct timespec64 delta)
126{
127 ts->ts_real = timespec64_sub(ts->ts_real, delta);
128#ifdef CONFIG_NTP_PPS
129 ts->ts_raw = timespec64_sub(ts->ts_raw, delta);
130#endif
131}
