7static inline bool virtio_legacy_is_little_endian(void)
8{
9#ifdef __LITTLE_ENDIAN
10 return true;
11#else
12 return false;
13#endif

16static inline u16 __virtio16_to_cpu(bool little_endian, __virtio16 val)
17{
18 if (little_endian)
19 return le16_to_cpu((__force __le16)val);
20 else
21 return be16_to_cpu((__force __be16)val);
22}

29 return (__force __virtio16)cpu_to_be16(val);
