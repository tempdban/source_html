36struct rb_node {
37 unsigned long __rb_parent_color;
38 struct rb_node *rb_right;
39 struct rb_node *rb_left;
40} __attribute__((aligned(sizeof(long))));
41 /* The alignment might seem pointless, but allegedly CRIS needs it */
42
43struct rb_root {
44 struct rb_node *rb_node;
45};
46
47/*
48 * Leftmost-cached rbtrees.
49 *
50 * We do not cache the rightmost node based on footprint
51 * size vs number of potential users that could benefit
52 * from O(1) rb_last(). Just not worth it, users that want
53 * this feature can always implement the logic explicitly.
54 * Furthermore, users that want to cache both pointers may
55 * find it a bit asymmetric, but that's ok.
56 */
57struct rb_root_cached {
58 struct rb_root rb_root;
59 struct rb_node *rb_leftmost;
60};

103static inline void rb_link_node(struct rb_node *node, struct rb_node *parent,
104 struct rb_node **rb_link)
105{
106 node->__rb_parent_color = (unsigned long)parent;
107 node->rb_left = node->rb_right = NULL;
108
109 *rb_link = node;
110}
111
112static inline void rb_link_node_rcu(struct rb_node *node, struct rb_node *parent,
113 struct rb_node **rb_link)
114{
115 node->__rb_parent_color = (unsigned long)parent;
116 node->rb_left = node->rb_right = NULL;
117
118 rcu_assign_pointer(*rb_link, node);
119}

121#define rb_entry_safe(ptr, type, member) \
122 ({ typeof(ptr) ____ptr = (ptr); \
123 ____ptr ? rb_entry(____ptr, type, member) : NULL; \
124 })

126/**
127 * rbtree_postorder_for_each_entry_safe - iterate in post-order over rb_root of
128 * given type allowing the backing memory of @pos to be invalidated
129 *
130 * @pos: the 'type *' to use as a loop cursor.
131 * @n: another 'type *' to use as temporary storage
132 * @root: 'rb_root *' of the rbtree.
133 * @field: the name of the rb_node field within 'type'.
134 *
135 * rbtree_postorder_for_each_entry_safe() provides a similar guarantee as
136 * list_for_each_entry_safe() and allows the iteration to continue independent
137 * of changes to @pos by the body of the loop.
138 *
139 * Note, however, that it cannot handle other modifications that re-order the
140 * rbtree it is iterating over. This includes calling rb_erase() on @pos, as
141 * rb_erase() may rebalance the tree, causing us to miss some nodes.
142 */
143#define rbtree_postorder_for_each_entry_safe(pos, n, root, field) \
144 for (pos = rb_entry_safe(rb_first_postorder(root), typeof(*pos), field); \
145 pos && ({ n = rb_entry_safe(rb_next_postorder(&pos->field), \
146 typeof(*pos), field); 1; }); \
147 pos = n)
