84/*
85 * __MAP - apply a macro to syscall arguments
86 * __MAP(n, m, t1, a1, t2, a2, ..., tn, an) will expand to
87 * m(t1, a1), m(t2, a2), ..., m(tn, an)
88 * The first argument must be equal to the amount of type/name
89 * pairs given. Note that this list of pairs (i.e. the arguments
90 * of __MAP starting at the third one) is in the same format as
91 * for SYSCALL_DEFINE<n>/COMPAT_SYSCALL_DEFINE<n>
92 */
93#define __MAP0(m,...)
94#define __MAP1(m,t,a) m(t,a)
95#define __MAP2(m,t,a,...) m(t,a), __MAP1(m,__VA_ARGS__)
96#define __MAP3(m,t,a,...) m(t,a), __MAP2(m,__VA_ARGS__)
97#define __MAP4(m,t,a,...) m(t,a), __MAP3(m,__VA_ARGS__)
98#define __MAP5(m,t,a,...) m(t,a), __MAP4(m,__VA_ARGS__)
99#define __MAP6(m,t,a,...) m(t,a), __MAP5(m,__VA_ARGS__)
100#define __MAP(n,...) __MAP##n(__VA_ARGS__)
101
102#define __SC_DECL(t, a) t a
103#define __TYPE_AS(t, v) __same_type((__force t)0, v)
104#define __TYPE_IS_L(t) (__TYPE_AS(t, 0L))
105#define __TYPE_IS_UL(t) (__TYPE_AS(t, 0UL))
106#define __TYPE_IS_LL(t) (__TYPE_AS(t, 0LL) || __TYPE_AS(t, 0ULL))
107#define __SC_LONG(t, a) __typeof(__builtin_choose_expr(__TYPE_IS_LL(t), 0LL, 0L)) a
108#define __SC_CAST(t, a) (__force t) a
109#define __SC_ARGS(t, a) a
110#define __SC_TEST(t, a) (void)BUILD_BUG_ON_ZERO(!__TYPE_IS_LL(t) && sizeof(t) > sizeof(long))

121#define SYSCALL_TRACE_ENTER_EVENT(sname) \
122 static struct syscall_metadata __syscall_meta_##sname; \
123 static struct trace_event_call __used \
124 event_enter_##sname = { \
125 .class = &event_class_syscall_enter, \
126 { \
127 .name = "sys_enter"#sname, \
128 }, \
129 .event.funcs = &enter_syscall_print_funcs, \
130 .data = (void *)&__syscall_meta_##sname,\
131 .flags = TRACE_EVENT_FL_CAP_ANY, \
132 }; \
133 static struct trace_event_call __used \
134 __attribute__((section("_ftrace_events"))) \
135 *__event_enter_##sname = &event_enter_##sname;
136
137#define SYSCALL_TRACE_EXIT_EVENT(sname) \
138 static struct syscall_metadata __syscall_meta_##sname; \
139 static struct trace_event_call __used \
140 event_exit_##sname = { \
141 .class = &event_class_syscall_exit, \
142 { \
143 .name = "sys_exit"#sname, \
144 }, \
145 .event.funcs = &exit_syscall_print_funcs, \
146 .data = (void *)&__syscall_meta_##sname,\
147 .flags = TRACE_EVENT_FL_CAP_ANY, \
148 }; \
149 static struct trace_event_call __used \
150 __attribute__((section("_ftrace_events"))) \
151 *__event_exit_##sname = &event_exit_##sname;
152
153#define SYSCALL_METADATA(sname, nb, ...) \
154 static const char *types_##sname[] = { \
155 __MAP(nb,__SC_STR_TDECL,__VA_ARGS__) \
156 }; \
157 static const char *args_##sname[] = { \
158 __MAP(nb,__SC_STR_ADECL,__VA_ARGS__) \
159 }; \
160 SYSCALL_TRACE_ENTER_EVENT(sname); \
161 SYSCALL_TRACE_EXIT_EVENT(sname); \
162 static struct syscall_metadata __used \
163 __syscall_meta_##sname = { \
164 .name = "sys"#sname, \
165 .syscall_nr = -1, /* Filled in at boot */ \
166 .nb_args = nb, \
167 .types = nb ? types_##sname : NULL, \
168 .args = nb ? args_##sname : NULL, \
169 .enter_event = &event_enter_##sname, \
170 .exit_event = &event_exit_##sname, \
171 .enter_fields = LIST_HEAD_INIT(__syscall_meta_##sname.enter_fields), \
172 }; \
173 static struct syscall_metadata __used \
174 __attribute__((section("__syscalls_metadata"))) \
175 *__p_syscall_meta_##sname = &__syscall_meta_##sname;

177static inline int is_syscall_trace_event(struct trace_event_call *tp_event)
178{
179 return tp_event->class == &event_class_syscall_enter ||
180 tp_event->class == &event_class_syscall_exit;
181}

199#define SYSCALL_DEFINE4(name, ...) SYSCALL_DEFINEx(4, _##name, __VA_ARGS__)
200#define SYSCALL_DEFINE5(name, ...) SYSCALL_DEFINEx(5, _##name, __VA_ARGS__)

205#define SYSCALL_DEFINEx(x, sname, ...) \
206 SYSCALL_METADATA(sname, x, __VA_ARGS__) \
207 __SYSCALL_DEFINEx(x, sname, __VA_ARGS__)

209#define __PROTECT(...) asmlinkage_protect(__VA_ARGS__)
210#define __SYSCALL_DEFINEx(x, name, ...) \
211 asmlinkage long sys##name(__MAP(x,__SC_DECL,__VA_ARGS__)) \
212 __attribute__((alias(__stringify(SyS##name)))); \
213 static inline long SYSC##name(__MAP(x,__SC_DECL,__VA_ARGS__)); \
214 asmlinkage long SyS##name(__MAP(x,__SC_LONG,__VA_ARGS__)); \
215 asmlinkage long SyS##name(__MAP(x,__SC_LONG,__VA_ARGS__)) \
216 { \
217 long ret = SYSC##name(__MAP(x,__SC_CAST,__VA_ARGS__)); \
218 __MAP(x,__SC_TEST,__VA_ARGS__); \
219 __PROTECT(x, ret,__MAP(x,__SC_ARGS,__VA_ARGS__)); \
220 return ret; \
221 } \
222 static inline long SYSC##name(__MAP(x,__SC_DECL,__VA_ARGS__))

224/*
225 * Called before coming back to user-mode. Returning to user-mode with an
226 * address limit different than USER_DS can allow to overwrite kernel memory.
227 */
228static inline void addr_limit_user_check(void)
229{
230#ifdef TIF_FSCHECK
231 if (!test_thread_flag(TIF_FSCHECK))
232 return;
233#endif
234
235 if (CHECK_DATA_CORRUPTION(!segment_eq(get_fs(), USER_DS),
236 "Invalid address limit on user-mode return"))
237 force_sig(SIGKILL, current);
238
239#ifdef TIF_FSCHECK
240 clear_thread_flag(TIF_FSCHECK);
241#endif
242}
