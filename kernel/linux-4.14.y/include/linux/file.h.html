30struct fd {
31 struct file *file;
32 unsigned int flags;
33};
34#define FDPUT_FPUT 1
35#define FDPUT_POS_UNLOCK 2
36
37static inline void fdput(struct fd fd)
38{
39 if (fd.flags & FDPUT_FPUT)
40 fput(fd.file);
41}

50static inline struct fd __to_fd(unsigned long v)
51{
52 return (struct fd){(struct file *)(v & ~3),v & 3};
53}
54
55static inline struct fd fdget(unsigned int fd)
56{
57 return __to_fd(__fdget(fd));
58}
59
60static inline struct fd fdget_raw(unsigned int fd)
61{
62 return __to_fd(__fdget_raw(fd));
63}
64
65static inline struct fd fdget_pos(int fd)
66{
67 return __to_fd(__fdget_pos(fd));
68}
69
70static inline void fdput_pos(struct fd f)
71{
72 if (f.flags & FDPUT_POS_UNLOCK)
73 __f_unlock_pos(f.file);
74 fdput(f);
75}
