44struct bpf_map {
45 /* 1st cacheline with read-mostly members of which some
46 * are also accessed in fast-path (e.g. ops, max_entries).
47 */
48 const struct bpf_map_ops *ops ____cacheline_aligned;
49 struct bpf_map *inner_map_meta;
50#ifdef CONFIG_SECURITY
51 void *security;
52#endif
53 enum bpf_map_type map_type;
54 u32 key_size;
55 u32 value_size;
56 u32 max_entries;
57 u32 map_flags;
58 u32 pages;
59 u32 id;
60 int numa_node;
61 bool unpriv_array;
62 /* 7 bytes hole */
63
64 /* 2nd cacheline with misc members to avoid false sharing
65 * particularly with refcounting.
66 */
67 struct user_struct *user ____cacheline_aligned;
68 atomic_t refcnt;
69 atomic_t usercnt;
70 struct work_struct work;
71};

73/* function argument constraints */
74enum bpf_arg_type {
75 ARG_DONTCARE = 0, /* unused argument in helper function */
76
77 /* the following constraints used to prototype
78 * bpf_map_lookup/update/delete_elem() functions
79 */
80 ARG_CONST_MAP_PTR, /* const argument used as pointer to bpf_map */
81 ARG_PTR_TO_MAP_KEY, /* pointer to stack used as map key */
82 ARG_PTR_TO_MAP_VALUE, /* pointer to stack used as map value */
83
84 /* the following constraints used to prototype bpf_memcmp() and other
85 * functions that access data on eBPF program stack
86 */
87 ARG_PTR_TO_MEM, /* pointer to valid memory (stack, packet, map value) */
88 ARG_PTR_TO_UNINIT_MEM, /* pointer to memory does not need to be initialized,
89 * helper function must fill all bytes or clear
90 * them in error case.
91 */
92
93 ARG_CONST_SIZE, /* number of bytes accessed from memory */
94 ARG_CONST_SIZE_OR_ZERO, /* number of bytes accessed from memory or 0 */
95
96 ARG_PTR_TO_CTX, /* pointer to context */
97 ARG_ANYTHING, /* any (initialized) argument is ok */
98};

129enum bpf_access_type {
130 BPF_READ = 1,
131 BPF_WRITE = 2
132};

134/* types of values stored in eBPF registers */
135/* Pointer types represent:
136 * pointer
137 * pointer + imm
138 * pointer + (u16) var
139 * pointer + (u16) var + imm
140 * if (range > 0) then [ptr, ptr + range - off) is safe to access
141 * if (id > 0) means that some 'var' was added
142 * if (off > 0) means that 'imm' was added
143 */
144enum bpf_reg_type {
145 NOT_INIT = 0, /* nothing was written into register */
146 SCALAR_VALUE, /* reg doesn't contain a valid pointer */
147 PTR_TO_CTX, /* reg points to bpf_context */
148 CONST_PTR_TO_MAP, /* reg points to struct bpf_map */
149 PTR_TO_MAP_VALUE, /* reg points to map element value */
150 PTR_TO_MAP_VALUE_OR_NULL,/* points to map elem value or NULL */
151 PTR_TO_STACK, /* reg == frame_pointer + offset */
152 PTR_TO_PACKET, /* reg points to skb->data */
153 PTR_TO_PACKET_END, /* skb->data + headlen */
154};

308/* memcpy that is used with 8-byte aligned pointers, power-of-8 size and
309 * forced to use 'long' read/writes to try to atomically copy long counters.
310 * Best-effort only. No barriers here, since it _will_ race with concurrent
311 * updates from BPF programs. Called from bpf syscall and mostly used with
312 * size 8 or 16 bytes, so ask compiler to inline it.
313 */
314static inline void bpf_long_memcpy(void *dst, const void *src, u32 size)
315{
316 const long *lsrc = src;
317 long *ldst = dst;
318
319 size /= sizeof(long);
320 while (size--)
321 *ldst++ = *lsrc++;
322}
