111struct inet6_skb_parm {
112 int iif;
113 __be16 ra;
114 __u16 dst0;
115 __u16 srcrt;
116 __u16 dst1;
117 __u16 lastopt;
118 __u16 nhoff;
119 __u16 flags;
120#if defined(CONFIG_IPV6_MIP6) || defined(CONFIG_IPV6_MIP6_MODULE)
121 __u16 dsthao;
122#endif
123 __u16 frag_max_size;
124
125#define IP6SKB_XFRM_TRANSFORMED 1
126#define IP6SKB_FORWARDED 2
127#define IP6SKB_REROUTED 4
128#define IP6SKB_ROUTERALERT 8
129#define IP6SKB_FRAGMENTED 16
130#define IP6SKB_HOPBYHOP 32
131#define IP6SKB_L3SLAVE 64
132#define IP6SKB_JUMBOGRAM 128
133};

135#if defined(CONFIG_NET_L3_MASTER_DEV)

147#define IP6CB(skb) ((struct inet6_skb_parm*)((skb)->cb))
148#define IP6CBMTU(skb) ((struct ip6_mtuinfo *)((skb)->cb))
149
150static inline int inet6_iif(const struct sk_buff *skb)
151{
152 bool l3_slave = ipv6_l3mdev_skb(IP6CB(skb)->flags);
153
154 return l3_slave ? skb->skb_iif : IP6CB(skb)->iif;
155}
156
157static inline bool inet6_is_jumbogram(const struct sk_buff *skb)
158{
159 return !!(IP6CB(skb)->flags & IP6SKB_JUMBOGRAM);
160}

162/* can not be used in TCP layer after tcp_v6_fill_cb */
163static inline int inet6_sdif(const struct sk_buff *skb)
164{
165#if IS_ENABLED(CONFIG_NET_L3_MASTER_DEV)
166 if (skb && ipv6_l3mdev_skb(IP6CB(skb)->flags))
167 return IP6CB(skb)->iif;
168#endif
169 return 0;
170}
171
172/* can not be used in TCP layer after tcp_v6_fill_cb */
173static inline bool inet6_exact_dif_match(struct net *net, struct sk_buff *skb)
174{
175#if defined(CONFIG_NET_L3_MASTER_DEV)
176 if (!net->ipv4.sysctl_tcp_l3mdev_accept &&
177 skb && ipv6_l3mdev_skb(IP6CB(skb)->flags))
178 return true;
179#endif
180 return false;
181}

197/**
198 * struct ipv6_pinfo - ipv6 private area
199 *
200 * In the struct sock hierarchy (tcp6_sock, upd6_sock, etc)
201 * this _must_ be the last member, so that inet6_sk_generic
202 * is able to calculate its offset from the base struct sock
203 * by using the struct proto->slab_obj_size member. -acme
204 */
205struct ipv6_pinfo {
206 struct in6_addr saddr;
207 struct in6_pktinfo sticky_pktinfo;
208 const struct in6_addr *daddr_cache;
209#ifdef CONFIG_IPV6_SUBTREES
210 const struct in6_addr *saddr_cache;
211#endif
212
213 __be32 flow_label;
214 __u32 frag_size;
215
216 /*
217 * Packed in 16bits.
218 * Omit one shift by by putting the signed field at MSB.
219 */
220#if defined(__BIG_ENDIAN_BITFIELD)
221 __s16 hop_limit:9;
222 __u16 __unused_1:7;
223#else
224 __u16 __unused_1:7;
225 __s16 hop_limit:9;
226#endif
227
228#if defined(__BIG_ENDIAN_BITFIELD)
229 /* Packed in 16bits. */
230 __s16 mcast_hops:9;
231 __u16 __unused_2:6,
232 mc_loop:1;
233#else
234 __u16 mc_loop:1,
235 __unused_2:6;
236 __s16 mcast_hops:9;
237#endif
238 int ucast_oif;
239 int mcast_oif;
240
241 /* pktoption flags */
242 union {
243 struct {
244 __u16 srcrt:1,
245 osrcrt:1,
246 rxinfo:1,
247 rxoinfo:1,
248 rxhlim:1,
249 rxohlim:1,
250 hopopts:1,
251 ohopopts:1,
252 dstopts:1,
253 odstopts:1,
254 rxflow:1,
255 rxtclass:1,
256 rxpmtu:1,
257 rxorigdstaddr:1,
258 recvfragsize:1;
259 /* 1 bits hole */
260 } bits;
261 __u16 all;
262 } rxopt;
263
264 /* sockopt flags */
265 __u16 recverr:1,
266 sndflow:1,
267 repflow:1,
268 pmtudisc:3,
269 padding:1, /* 1 bit hole */
270 srcprefs:3, /* 001: prefer temporary address
271 * 010: prefer public address
272 * 100: prefer care-of address
273 */
274 dontfrag:1,
275 autoflowlabel:1,
276 autoflowlabel_set:1;
277 __u8 min_hopcount;
278 __u8 tclass;
279 __be32 rcv_flowinfo;
280
281 __u32 dst_cookie;
282 __u32 rx_dst_cookie;
283
284 struct ipv6_mc_socklist __rcu *ipv6_mc_list;
285 struct ipv6_ac_socklist *ipv6_ac_list;
286 struct ipv6_fl_socklist __rcu *ipv6_fl_list;
287
288 struct ipv6_txoptions __rcu *opt;
289 struct sk_buff *pktoptions;
290 struct sk_buff *rxpmtu;
291 struct inet6_cork cork;
292};

337static inline void inet_sk_copy_descendant(struct sock *sk_to,
338 const struct sock *sk_from)
339{
340 int ancestor_size = sizeof(struct inet_sock);
341
342 if (sk_from->sk_family == PF_INET6)
343 ancestor_size += sizeof(struct ipv6_pinfo);
344
345 __inet_sk_copy_descendant(sk_to, sk_from, ancestor_size);
346}
