46#ifndef arch_validate_prot
47/*
48 * This is called from mprotect(). PROT_GROWSDOWN and PROT_GROWSUP have
49 * already been masked out.
50 *
51 * Returns true if the prot flags are valid
52 */
53static inline bool arch_validate_prot(unsigned long prot)
54{
55 return (prot & ~(PROT_READ | PROT_WRITE | PROT_EXEC | PROT_SEM)) == 0;
56}
57#define arch_validate_prot arch_validate_prot
58#endif

60/*
61 * Optimisation macro. It is equivalent to:
62 * (x & bit1) ? bit2 : 0
63 * but this version is faster.
64 * ("bit1" and "bit2" must be single bits)
65 */
66#define _calc_vm_trans(x, bit1, bit2) \
67 ((!(bit1) || !(bit2)) ? 0 : \
68 ((bit1) <= (bit2) ? ((x) & (bit1)) * ((bit2) / (bit1)) \
69 : ((x) & (bit1)) / ((bit1) / (bit2))))
70
71/*
72 * Combine the mmap "prot" argument into "vm_flags" used internally.
73 */
74static inline unsigned long
75calc_vm_prot_bits(unsigned long prot, unsigned long pkey)
76{
77 return _calc_vm_trans(prot, PROT_READ, VM_READ ) |
78 _calc_vm_trans(prot, PROT_WRITE, VM_WRITE) |
79 _calc_vm_trans(prot, PROT_EXEC, VM_EXEC) |
80 arch_calc_vm_prot_bits(prot, pkey);
81}
82
83/*
84 * Combine the mmap "flags" argument into "vm_flags" used internally.
85 */
86static inline unsigned long
87calc_vm_flag_bits(unsigned long flags)
88{
89 return _calc_vm_trans(flags, MAP_GROWSDOWN, VM_GROWSDOWN ) |
90 _calc_vm_trans(flags, MAP_DENYWRITE, VM_DENYWRITE ) |
91 _calc_vm_trans(flags, MAP_LOCKED, VM_LOCKED );
92}
