45/**

47 * @ETHTOOL_ID_INACTIVE: Physical ID indicator should be deactivated
48 * @ETHTOOL_ID_ACTIVE: Physical ID indicator should be activated
49 * @ETHTOOL_ID_ON: LED should be turned on (used iff %ETHTOOL_ID_ACTIVE

89/**
90 * ethtool_rxfh_indir_default - get default value for RX flow hash indirection
91 * @index: Index in RX flow hash indirection table
92 * @n_rx_rings: Number of RX rings to use
93 *
94 * This function provides the default policy for RX flow hash indirection.
95 */
96static inline u32 ethtool_rxfh_indir_default(u32 index, u32 n_rx_rings)
97{
98 return index % n_rx_rings;
99}

101/* number of link mode bits/ulongs handled internally by kernel */
102#define __ETHTOOL_LINK_MODE_MASK_NBITS \
103 (__ETHTOOL_LINK_MODE_LAST + 1)

105/* declare a link mode bitmap */
106#define __ETHTOOL_DECLARE_LINK_MODE_MASK(name) \
107 DECLARE_BITMAP(name, __ETHTOOL_LINK_MODE_MASK_NBITS)

109/* drivers must ignore base.cmd and base.link_mode_masks_nwords

112struct ethtool_link_ksettings {
113 struct ethtool_link_settings base;
114 struct {
115 __ETHTOOL_DECLARE_LINK_MODE_MASK(supported);
116 __ETHTOOL_DECLARE_LINK_MODE_MASK(advertising);
117 __ETHTOOL_DECLARE_LINK_MODE_MASK(lp_advertising);
118 } link_modes;
119};

121/**
122 * ethtool_link_ksettings_zero_link_mode - clear link_ksettings link mode mask
123 * @ptr : pointer to struct ethtool_link_ksettings
124 * @name : one of supported/advertising/lp_advertising
125 */
126#define ethtool_link_ksettings_zero_link_mode(ptr, name) \
127 bitmap_zero((ptr)->link_modes.name, __ETHTOOL_LINK_MODE_MASK_NBITS)
128
129/**
130 * ethtool_link_ksettings_add_link_mode - set bit in link_ksettings
131 * link mode mask
132 * @ptr : pointer to struct ethtool_link_ksettings
133 * @name : one of supported/advertising/lp_advertising
134 * @mode : one of the ETHTOOL_LINK_MODE_*_BIT
135 * (not atomic, no bound checking)
136 */
137#define ethtool_link_ksettings_add_link_mode(ptr, name, mode) \
138 __set_bit(ETHTOOL_LINK_MODE_ ## mode ## _BIT, (ptr)->link_modes.name)

140/**
141 * ethtool_link_ksettings_del_link_mode - clear bit in link_ksettings

145 * @mode : one of the ETHTOOL_LINK_MODE_*_BIT

148#define ethtool_link_ksettings_del_link_mode(ptr, name, mode) \

174/**
175 * struct ethtool_ops - optional netdev operations
176 * @get_settings: DEPRECATED, use %get_link_ksettings/%set_link_ksettings
177 * API. Get various device settings including Ethernet link
178 * settings. The @cmd parameter is expected to have been cleared
179 * before get_settings is called. Returns a negative error code
180 * or zero.
181 * @set_settings: DEPRECATED, use %get_link_ksettings/%set_link_ksettings
182 * API. Set various device settings including Ethernet link
183 * settings. Returns a negative error code or zero.
184 * @get_drvinfo: Report driver/device information. Should only set the
185 * @driver, @version, @fw_version and @bus_info fields. If not
186 * implemented, the @driver and @bus_info fields will be filled in
187 * according to the netdev's parent device.
188 * @get_regs_len: Get buffer length required for @get_regs
189 * @get_regs: Get device registers
190 * @get_wol: Report whether Wake-on-Lan is enabled
191 * @set_wol: Turn Wake-on-Lan on or off. Returns a negative error code
192 * or zero.
193 * @get_msglevel: Report driver message level. This should be the value
194 * of the @msg_enable field used by netif logging functions.
195 * @set_msglevel: Set driver message level
196 * @nway_reset: Restart autonegotiation. Returns a negative error code
197 * or zero.
198 * @get_link: Report whether physical link is up. Will only be called if
199 * the netdev is up. Should usually be set to ethtool_op_get_link(),
200 * which uses netif_carrier_ok().
201 * @get_eeprom: Read data from the device EEPROM.
202 * Should fill in the magic field. Don't need to check len for zero
203 * or wraparound. Fill in the data argument with the eeprom values
204 * from offset to offset + len. Update len to the amount read.
205 * Returns an error or zero.
206 * @set_eeprom: Write data to the device EEPROM.
207 * Should validate the magic field. Don't need to check len for zero
208 * or wraparound. Update len to the amount written. Returns an error
209 * or zero.
210 * @get_coalesce: Get interrupt coalescing parameters. Returns a negative
211 * error code or zero.
212 * @set_coalesce: Set interrupt coalescing parameters. Returns a negative
213 * error code or zero.
214 * @get_ringparam: Report ring sizes
215 * @set_ringparam: Set ring sizes. Returns a negative error code or zero.
216 * @get_pauseparam: Report pause parameters
217 * @set_pauseparam: Set pause parameters. Returns a negative error code
218 * or zero.
219 * @self_test: Run specified self-tests
220 * @get_strings: Return a set of strings that describe the requested objects
221 * @set_phys_id: Identify the physical devices, e.g. by flashing an LED
222 * attached to it. The implementation may update the indicator
223 * asynchronously or synchronously, but in either case it must return
224 * quickly. It is initially called with the argument %ETHTOOL_ID_ACTIVE,
225 * and must either activate asynchronous updates and return zero, return
226 * a negative error or return a positive frequency for synchronous
227 * indication (e.g. 1 for one on/off cycle per second). If it returns
228 * a frequency then it will be called again at intervals with the
229 * argument %ETHTOOL_ID_ON or %ETHTOOL_ID_OFF and should set the state of
230 * the indicator accordingly. Finally, it is called with the argument
231 * %ETHTOOL_ID_INACTIVE and must deactivate the indicator. Returns a
232 * negative error code or zero.
233 * @get_ethtool_stats: Return extended statistics about the device.
234 * This is only useful if the device maintains statistics not
235 * included in &struct rtnl_link_stats64.
236 * @begin: Function to be called before any other operation. Returns a
237 * negative error code or zero.
238 * @complete: Function to be called after any other operation except
239 * @begin. Will be called even if the other operation failed.
240 * @get_priv_flags: Report driver-specific feature flags.
241 * @set_priv_flags: Set driver-specific feature flags. Returns a negative
242 * error code or zero.
243 * @get_sset_count: Get number of strings that @get_strings will write.
244 * @get_rxnfc: Get RX flow classification rules. Returns a negative
245 * error code or zero.
246 * @set_rxnfc: Set RX flow classification rules. Returns a negative
247 * error code or zero.
248 * @flash_device: Write a firmware image to device's flash memory.
249 * Returns a negative error code or zero.
250 * @reset: Reset (part of) the device, as specified by a bitmask of
251 * flags from &enum ethtool_reset_flags. Returns a negative
252 * error code or zero.
253 * @get_rxfh_key_size: Get the size of the RX flow hash key.
254 * Returns zero if not supported for this specific device.
255 * @get_rxfh_indir_size: Get the size of the RX flow hash indirection table.
256 * Returns zero if not supported for this specific device.
257 * @get_rxfh: Get the contents of the RX flow hash indirection table, hash key
258 * and/or hash function.
259 * Returns a negative error code or zero.
260 * @set_rxfh: Set the contents of the RX flow hash indirection table, hash
261 * key, and/or hash function. Arguments which are set to %NULL or zero
262 * will remain unchanged.
263 * Returns a negative error code or zero. An error code must be returned
264 * if at least one unsupported change was requested.
265 * @get_channels: Get number of channels.
266 * @set_channels: Set number of channels. Returns a negative error code or
267 * zero.
268 * @get_dump_flag: Get dump flag indicating current dump length, version,
269 * and flag of the device.
270 * @get_dump_data: Get dump data.
271 * @set_dump: Set dump specific flags to the device.
272 * @get_ts_info: Get the time stamping and PTP hardware clock capabilities.
273 * Drivers supporting transmit time stamps in software should set this to
274 * ethtool_op_get_ts_info().
275 * @get_module_info: Get the size and type of the eeprom contained within
276 * a plug-in module.
277 * @get_module_eeprom: Get the eeprom information from the plug-in module
278 * @get_eee: Get Energy-Efficient (EEE) supported and status.
279 * @set_eee: Set EEE status (enable/disable) as well as LPI timers.
280 * @get_per_queue_coalesce: Get interrupt coalescing parameters per queue.
281 * It must check that the given queue number is valid. If neither a RX nor
282 * a TX queue has this number, return -EINVAL. If only a RX queue or a TX
283 * queue has this number, set the inapplicable fields to ~0 and return 0.
284 * Returns a negative error code or zero.
285 * @set_per_queue_coalesce: Set interrupt coalescing parameters per queue.
286 * It must check that the given queue number is valid. If neither a RX nor
287 * a TX queue has this number, return -EINVAL. If only a RX queue or a TX
288 * queue has this number, ignore the inapplicable fields.
289 * Returns a negative error code or zero.
290 * @get_link_ksettings: When defined, takes precedence over the
291 * %get_settings method. Get various device settings
292 * including Ethernet link settings. The %cmd and
293 * %link_mode_masks_nwords fields should be ignored (use
294 * %__ETHTOOL_LINK_MODE_MASK_NBITS instead of the latter), any
295 * change to them will be overwritten by kernel. Returns a
296 * negative error code or zero.
297 * @set_link_ksettings: When defined, takes precedence over the
298 * %set_settings method. Set various device settings including
299 * Ethernet link settings. The %cmd and %link_mode_masks_nwords
300 * fields should be ignored (use %__ETHTOOL_LINK_MODE_MASK_NBITS
301 * instead of the latter), any change to them will be overwritten
302 * by kernel. Returns a negative error code or zero.
303 * @get_fecparam: Get the network device Forward Error Correction parameters.
304 * @set_fecparam: Set the network device Forward Error Correction parameters.
305 *
306 * All operations are optional (i.e. the function pointer may be set
307 * to %NULL) and callers must take this into account. Callers must
308 * hold the RTNL lock.
309 *
310 * See the structures used by these operations for further documentation.
311 * Note that for all operations using a structure ending with a zero-
312 * length array, the array is allocated separately in the kernel and
313 * is passed to the driver as an additional parameter.
314 *
315 * See &struct net_device and &struct net_device_ops for documentation
316 * of the generic netdev features interface.
317 */
