15typedef u64 netdev_features_t;

17enum {
18 NETIF_F_SG_BIT, /* Scatter/gather IO. */
19 NETIF_F_IP_CSUM_BIT, /* Can checksum TCP/UDP over IPv4. */
20 __UNUSED_NETIF_F_1,
21 NETIF_F_HW_CSUM_BIT, /* Can checksum all the packets. */
22 NETIF_F_IPV6_CSUM_BIT, /* Can checksum TCP/UDP over IPV6 */
23 NETIF_F_HIGHDMA_BIT, /* Can DMA to high memory. */
24 NETIF_F_FRAGLIST_BIT, /* Scatter/gather IO. */
25 NETIF_F_HW_VLAN_CTAG_TX_BIT, /* Transmit VLAN CTAG HW acceleration */
26 NETIF_F_HW_VLAN_CTAG_RX_BIT, /* Receive VLAN CTAG HW acceleration */
27 NETIF_F_HW_VLAN_CTAG_FILTER_BIT,/* Receive filtering on VLAN CTAGs */
28 NETIF_F_VLAN_CHALLENGED_BIT, /* Device cannot handle VLAN packets */
29 NETIF_F_GSO_BIT, /* Enable software GSO. */
30 NETIF_F_LLTX_BIT, /* LockLess TX - deprecated. Please */
31 /* do not use LLTX in new drivers */
32 NETIF_F_NETNS_LOCAL_BIT, /* Does not change network namespaces */
33 NETIF_F_GRO_BIT, /* Generic receive offload */
34 NETIF_F_LRO_BIT, /* large receive offload */
35
36 /**/NETIF_F_GSO_SHIFT, /* keep the order of SKB_GSO_* bits */
37 NETIF_F_TSO_BIT /* ... TCPv4 segmentation */
38 = NETIF_F_GSO_SHIFT,
39 NETIF_F_GSO_ROBUST_BIT, /* ... ->SKB_GSO_DODGY */
40 NETIF_F_TSO_ECN_BIT, /* ... TCP ECN support */
41 NETIF_F_TSO_MANGLEID_BIT, /* ... IPV4 ID mangling allowed */
42 NETIF_F_TSO6_BIT, /* ... TCPv6 segmentation */
43 NETIF_F_FSO_BIT, /* ... FCoE segmentation */
44 NETIF_F_GSO_GRE_BIT, /* ... GRE with TSO */
45 NETIF_F_GSO_GRE_CSUM_BIT, /* ... GRE with csum with TSO */
46 NETIF_F_GSO_IPXIP4_BIT, /* ... IP4 or IP6 over IP4 with TSO */
47 NETIF_F_GSO_IPXIP6_BIT, /* ... IP4 or IP6 over IP6 with TSO */
48 NETIF_F_GSO_UDP_TUNNEL_BIT, /* ... UDP TUNNEL with TSO */
49 NETIF_F_GSO_UDP_TUNNEL_CSUM_BIT,/* ... UDP TUNNEL with TSO & CSUM */
50 NETIF_F_GSO_PARTIAL_BIT, /* ... Only segment inner-most L4
51 * in hardware and all other
52 * headers in software.
53 */
54 NETIF_F_GSO_TUNNEL_REMCSUM_BIT, /* ... TUNNEL with TSO & REMCSUM */
55 NETIF_F_GSO_SCTP_BIT, /* ... SCTP fragmentation */
56 NETIF_F_GSO_ESP_BIT, /* ... ESP with TSO */
57 NETIF_F_GSO_UDP_BIT, /* ... UFO, deprecated except tuntap */
58 /**/NETIF_F_GSO_LAST = /* last bit, see GSO_MASK */
59 NETIF_F_GSO_UDP_BIT,
60
61 NETIF_F_FCOE_CRC_BIT, /* FCoE CRC32 */
62 NETIF_F_SCTP_CRC_BIT, /* SCTP checksum offload */
63 NETIF_F_FCOE_MTU_BIT, /* Supports max FCoE MTU, 2158 bytes*/
64 NETIF_F_NTUPLE_BIT, /* N-tuple filters supported */
65 NETIF_F_RXHASH_BIT, /* Receive hashing offload */
66 NETIF_F_RXCSUM_BIT, /* Receive checksumming offload */
67 NETIF_F_NOCACHE_COPY_BIT, /* Use no-cache copyfromuser */
68 NETIF_F_LOOPBACK_BIT, /* Enable loopback */
69 NETIF_F_RXFCS_BIT, /* Append FCS to skb pkt data */
70 NETIF_F_RXALL_BIT, /* Receive errored frames too */
71 NETIF_F_HW_VLAN_STAG_TX_BIT, /* Transmit VLAN STAG HW acceleration */
72 NETIF_F_HW_VLAN_STAG_RX_BIT, /* Receive VLAN STAG HW acceleration */
73 NETIF_F_HW_VLAN_STAG_FILTER_BIT,/* Receive filtering on VLAN STAGs */
74 NETIF_F_HW_L2FW_DOFFLOAD_BIT, /* Allow L2 Forwarding in Hardware */
75
76 NETIF_F_HW_TC_BIT, /* Offload TC infrastructure */
77 NETIF_F_HW_ESP_BIT, /* Hardware ESP transformation offload */
78 NETIF_F_HW_ESP_TX_CSUM_BIT, /* ESP with TX checksum offload */
79 NETIF_F_RX_UDP_TUNNEL_PORT_BIT, /* Offload of RX port for UDP tunnels */

146#define for_each_netdev_feature(mask_addr, bit) \

149/* Features valid for ethtool to change */
150/* = all defined minus driver/device-class-related */
151#define NETIF_F_NEVER_CHANGE (NETIF_F_VLAN_CHALLENGED | \
152 NETIF_F_LLTX | NETIF_F_NETNS_LOCAL)
153
154/* remember that ((t)1 << t_BITS) is undefined in C99 */
155#define NETIF_F_ETHTOOL_BITS ((__NETIF_F_BIT(NETDEV_FEATURE_COUNT - 1) | \
156 (__NETIF_F_BIT(NETDEV_FEATURE_COUNT - 1) - 1)) & \
157 ~NETIF_F_NEVER_CHANGE)
158
159/* Segmentation offload feature mask */
160#define NETIF_F_GSO_MASK (__NETIF_F_BIT(NETIF_F_GSO_LAST + 1) - \
161 __NETIF_F_BIT(NETIF_F_GSO_SHIFT))
162
163/* List of IP checksum features. Note that NETIF_F_ HW_CSUM should not be
164 * set in features when NETIF_F_IP_CSUM or NETIF_F_IPV6_CSUM are set--
165 * this would be contradictory
166 */
167#define NETIF_F_CSUM_MASK (NETIF_F_IP_CSUM | NETIF_F_IPV6_CSUM | \
168 NETIF_F_HW_CSUM)
169
170#define NETIF_F_ALL_TSO (NETIF_F_TSO | NETIF_F_TSO6 | \
171 NETIF_F_TSO_ECN | NETIF_F_TSO_MANGLEID)
172
173#define NETIF_F_ALL_FCOE (NETIF_F_FCOE_CRC | NETIF_F_FCOE_MTU | \
174 NETIF_F_FSO)
175
176/* List of features with software fallbacks. */
177#define NETIF_F_GSO_SOFTWARE (NETIF_F_ALL_TSO | \
178 NETIF_F_GSO_SCTP)
179
180/*
181 * If one device supports one of these features, then enable them
182 * for all in netdev_increment_features.
183 */
184#define NETIF_F_ONE_FOR_ALL (NETIF_F_GSO_SOFTWARE | NETIF_F_GSO_ROBUST | \
185 NETIF_F_SG | NETIF_F_HIGHDMA | \
186 NETIF_F_FRAGLIST | NETIF_F_VLAN_CHALLENGED)
187
188/*
189 * If one device doesn't support one of these features, then disable it
190 * for all in netdev_increment_features.
191 */
192#define NETIF_F_ALL_FOR_ALL (NETIF_F_NOCACHE_COPY | NETIF_F_FSO)
193
194/*
195 * If upper/master device has these features disabled, they must be disabled
196 * on all lower/slave devices as well.
197 */
198#define NETIF_F_UPPER_DISABLES NETIF_F_LRO
199
200/* changeable features with no special hardware requirements */
201#define NETIF_F_SOFT_FEATURES (NETIF_F_GSO | NETIF_F_GRO)
202
203#define NETIF_F_VLAN_FEATURES (NETIF_F_HW_VLAN_CTAG_FILTER | \
204 NETIF_F_HW_VLAN_CTAG_RX | \
205 NETIF_F_HW_VLAN_CTAG_TX | \
206 NETIF_F_HW_VLAN_STAG_FILTER | \
207 NETIF_F_HW_VLAN_STAG_RX | \
208 NETIF_F_HW_VLAN_STAG_TX)

210#define NETIF_F_GSO_ENCAP_ALL (NETIF_F_GSO_GRE | \
211 NETIF_F_GSO_GRE_CSUM | \
212 NETIF_F_GSO_IPXIP4 | \
213 NETIF_F_GSO_IPXIP6 | \
