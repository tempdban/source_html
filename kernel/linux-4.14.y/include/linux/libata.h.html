111static inline u32 ata_msg_init(int dval, int default_msg_enable_bits)
112{
113 if (dval < 0 || dval >= (sizeof(u32) * 8))
114 return default_msg_enable_bits; /* should be 0x1 - only driver info msgs */
115 if (!dval)
116 return 0;
117 return (1 << dval) - 1;
118}

1584/*
1585 * Shortcut notation for iterations
1586 *
1587 * ata_for_each_link() iterates over each link of @ap according to
1588 * @mode. @link points to the current link in the loop. @link is
1589 * NULL after loop termination. ata_for_each_dev() works the same way
1590 * except that it iterates over each device of @link.
1591 *
1592 * Note that the mode prefixes ATA_{L|D}ITER_ shouldn't need to be
1593 * specified when using the following shorthand notations. Only the
1594 * mode itself (EDGE, HOST_FIRST, ENABLED, etc...) should be
1595 * specified. This not only increases brevity but also makes it
1596 * impossible to use ATA_LITER_* for device iteration or vice-versa.
1597 */
1598#define ata_for_each_link(link, ap, mode) \
1599 for ((link) = ata_link_next(NULL, (ap), ATA_LITER_##mode); (link); \
1600 (link) = ata_link_next((link), (ap), ATA_LITER_##mode))
1601
1602#define ata_for_each_dev(dev, link, mode) \
1603 for ((dev) = ata_dev_next(NULL, (link), ATA_DITER_##mode); (dev); \
1604 (dev) = ata_dev_next((dev), (link), ATA_DITER_##mode))

1916/**
1917 * ata_sff_busy_wait - Wait for a port status register
1918 * @ap: Port to wait for.
1919 * @bits: bits that must be clear
1920 * @max: number of 10uS waits to perform
1921 *
1922 * Waits up to max*10 microseconds for the selected bits in the port's
1923 * status register to be cleared.
1924 * Returns final value of status register.
1925 *
1926 * LOCKING:
1927 * Inherited from caller.
1928 */
1929static inline u8 ata_sff_busy_wait(struct ata_port *ap, unsigned int bits,
1930 unsigned int max)
1931{
1932 u8 status;
1933
1934 do {
1935 udelay(10);
1936 status = ap->ops->sff_check_status(ap);
1937 max--;
1938 } while (status != 0xff && (status & bits) && (max > 0));
1939
1940 return status;
1941}

1943/**
1944 * ata_wait_idle - Wait for a port to be idle.
1945 * @ap: Port to wait for.
1946 *
1947 * Waits up to 10ms for port's BUSY and DRQ signals to clear.
1948 * Returns final value of status register.
1949 *
1950 * LOCKING:
1951 * Inherited from caller.
1952 */
1953static inline u8 ata_wait_idle(struct ata_port *ap)
1954{

1959 ata_port_printk(ap, KERN_DEBUG, "abnormal Status 0x%X\n",
1960 status);
1961#endif

1963 return status;
