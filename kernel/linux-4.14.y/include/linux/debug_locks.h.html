25#define DEBUG_LOCKS_WARN_ON(c) \
26({ \
27 int __ret = 0; \
28 \
29 if (!oops_in_progress && unlikely(c)) { \
30 if (debug_locks_off() && !debug_locks_silent) \
31 WARN(1, "DEBUG_LOCKS_WARN_ON(%s)", #c); \
32 __ret = 1; \
33 } \
34 __ret; \
