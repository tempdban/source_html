685/* Offset of logical sectors relative to physical sectors.
686 *
687 * If device has more than one logical sector per physical sector
688 * (aka 512 byte emulation), vendors might offset the "sector 0" address
689 * so sector 63 is "naturally aligned" - e.g. FAT partition table.
690 * This avoids Read/Mod/Write penalties when using FAT partition table
691 * and updating "well aligned" (FS perspective) physical sectors on every
692 * transaction.
693 */
694static inline u16 ata_id_logical_sector_offset(const u16 *id,
695 u8 log2_per_phys)
696{
697 u16 word_209 = id[209];
698
699 if ((log2_per_phys > 1) && (word_209 & 0xc000) == 0x4000) {
700 u16 first = word_209 & 0x3fff;
701 if (first > 0)
702 return (1 << log2_per_phys) - first;
703 }
704 return 0;
705}

836/**
837 * ata_id_major_version - get ATA level of drive
838 * @id: Identify data
839 *
840 * Caveats:
841 * ATA-1 considers identify optional
842 * ATA-2 introduces mandatory identify
843 * ATA-3 introduces word 80 and accurate reporting
844 *
845 * The practical impact of this is that ata_id_major_version cannot
846 * reliably report on drives below ATA3.
847 */
848
849static inline unsigned int ata_id_major_version(const u16 *id)
850{
851 unsigned int mver;
852
853 if (id[ATA_ID_MAJOR_VER] == 0xFFFF)
854 return 0;
855
856 for (mver = 14; mver >= 1; mver--)
857 if (id[ATA_ID_MAJOR_VER] & (1 << mver))
858 break;
859 return mver;
860}

1061/*
1062 * ata_id_is_lba_capacity_ok() performs a sanity check on
1063 * the claimed LBA capacity value for the device.
1064 *
1065 * Returns 1 if LBA capacity looks sensible, 0 otherwise.
1066 *
1067 * It is called only once for each device.
1068 */
1069static inline bool ata_id_is_lba_capacity_ok(u16 *id)
1070{
1071 unsigned long lba_sects, chs_sects, head, tail;
1072
1073 /* No non-LBA info .. so valid! */
1074 if (id[ATA_ID_CYLS] == 0)
1075 return true;
1076
1077 lba_sects = ata_id_u32(id, ATA_ID_LBA_CAPACITY);
1078
1079 /*
1080 * The ATA spec tells large drives to return
1081 * C/H/S = 16383/16/63 independent of their size.
1082 * Some drives can be jumpered to use 15 heads instead of 16.
1083 * Some drives can be jumpered to use 4092 cyls instead of 16383.
1084 */
1085 if ((id[ATA_ID_CYLS] == 16383 ||
1086 (id[ATA_ID_CYLS] == 4092 && id[ATA_ID_CUR_CYLS] == 16383)) &&
1087 id[ATA_ID_SECTORS] == 63 &&
1088 (id[ATA_ID_HEADS] == 15 || id[ATA_ID_HEADS] == 16) &&
1089 (lba_sects >= 16383 * 63 * id[ATA_ID_HEADS]))
1090 return true;
1091
1092 chs_sects = id[ATA_ID_CYLS] * id[ATA_ID_HEADS] * id[ATA_ID_SECTORS];
1093
1094 /* perform a rough sanity check on lba_sects: within 10% is OK */
1095 if (lba_sects - chs_sects < chs_sects/10)
1096 return true;
1097
1098 /* some drives have the word order reversed */
1099 head = (lba_sects >> 16) & 0xffff;
1100 tail = lba_sects & 0xffff;
1101 lba_sects = head | (tail << 16);
1102
1103 if (lba_sects - chs_sects < chs_sects/10) {
1104 *(__le32 *)&id[ATA_ID_LBA_CAPACITY] = __cpu_to_le32(lba_sects);
1105 return true; /* LBA capacity is (now) good */
1106 }
1107
1108 return false; /* LBA capacity value may be bad */
1109}
