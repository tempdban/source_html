20#ifndef __LINUX_DMA_FENCE_ARRAY_H
21#define __LINUX_DMA_FENCE_ARRAY_H
22
23#include <linux/dma-fence.h>
24#include <linux/irq_work.h>
25
26/**
27 * struct dma_fence_array_cb - callback helper for fence array
28 * @cb: fence callback structure for signaling
29 * @array: reference to the parent fence array object
30 */
31struct dma_fence_array_cb {
32 struct dma_fence_cb cb;
33 struct dma_fence_array *array;
34};
35
36/**
37 * struct dma_fence_array - fence to represent an array of fences
38 * @base: fence base class
39 * @lock: spinlock for fence handling
40 * @num_fences: number of fences in the array
41 * @num_pending: fences in the array still pending
42 * @fences: array of the fences
43 */
44struct dma_fence_array {
45 struct dma_fence base;
46
47 spinlock_t lock;
48 unsigned num_fences;
49 atomic_t num_pending;
50 struct dma_fence **fences;
51
52 struct irq_work work;
53};
54
55extern const struct dma_fence_ops dma_fence_array_ops;
56
57/**
58 * dma_fence_is_array - check if a fence is from the array subsclass
59 * @fence: fence to test
60 *
61 * Return true if it is a dma_fence_array and false otherwise.
62 */
63static inline bool dma_fence_is_array(struct dma_fence *fence)
64{
65 return fence->ops == &dma_fence_array_ops;
66}
67
68/**
69 * to_dma_fence_array - cast a fence to a dma_fence_array
70 * @fence: fence to cast to a dma_fence_array
71 *
72 * Returns NULL if the fence is not a dma_fence_array,
73 * or the dma_fence_array otherwise.
74 */
75static inline struct dma_fence_array *
76to_dma_fence_array(struct dma_fence *fence)
77{
78 if (fence->ops != &dma_fence_array_ops)
79 return NULL;
80
81 return container_of(fence, struct dma_fence_array, base);
82}
83
84struct dma_fence_array *dma_fence_array_create(int num_fences,
85 struct dma_fence **fences,
86 u64 context, unsigned seqno,
87 bool signal_on_any);
88
89bool dma_fence_match_context(struct dma_fence *fence, u64 context);

91#endif /* __LINUX_DMA_FENCE_ARRAY_H */
