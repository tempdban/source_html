83/**
84 * make_kqid - Map a user-namespace, type, qid tuple into a kqid.
85 * @from: User namespace that the qid is in
86 * @type: The type of quota
87 * @qid: Quota identifier
88 *
89 * Maps a user-namespace, type qid tuple into a kernel internal
90 * kqid, and returns that kqid.
91 *
92 * When there is no mapping defined for the user-namespace, type,
93 * qid tuple an invalid kqid is returned. Callers are expected to
94 * test for and handle handle invalid kqids being returned.
95 * Invalid kqids may be tested for using qid_valid().
96 */
97static inline struct kqid make_kqid(struct user_namespace *from,
98 enum quota_type type, qid_t qid)
99{
100 struct kqid kqid;
101
102 kqid.type = type;
103 switch (type) {
104 case USRQUOTA:
105 kqid.uid = make_kuid(from, qid);
106 break;
107 case GRPQUOTA:
108 kqid.gid = make_kgid(from, qid);
109 break;
110 case PRJQUOTA:
111 kqid.projid = make_kprojid(from, qid);
112 break;
113 default:
114 BUG();
115 }
116 return kqid;
117}

119/**
120 * make_kqid_invalid - Explicitly make an invalid kqid
121 * @type: The type of quota identifier
122 *
123 * Returns an invalid kqid with the specified type.
124 */
125static inline struct kqid make_kqid_invalid(enum quota_type type)
126{
127 struct kqid kqid;
128
129 kqid.type = type;
130 switch (type) {
131 case USRQUOTA:
132 kqid.uid = INVALID_UID;
133 break;
134 case GRPQUOTA:
135 kqid.gid = INVALID_GID;
136 break;
137 case PRJQUOTA:
138 kqid.projid = INVALID_PROJID;
139 break;
140 default:
141 BUG();
142 }
143 return kqid;
144}

457/**
458 * Quota state flags - they actually come in two flavors - for users and groups.
459 *
460 * Actual typed flags layout:
461 * USRQUOTA GRPQUOTA
462 * DQUOT_USAGE_ENABLED 0x0001 0x0002
463 * DQUOT_LIMITS_ENABLED 0x0004 0x0008
464 * DQUOT_SUSPENDED 0x0010 0x0020
465 *
466 * Following bits are used for non-typed flags:
467 * DQUOT_QUOTA_SYS_FILE 0x0040
468 * DQUOT_NEGATIVE_USAGE 0x0080
469 */
470enum {
471 _DQUOT_USAGE_ENABLED = 0, /* Track disk usage for users */
472 _DQUOT_LIMITS_ENABLED, /* Enforce quota limits for users */
473 _DQUOT_SUSPENDED, /* User diskquotas are off, but
474 * we have necessary info in
475 * memory to turn them on */
476 _DQUOT_STATE_FLAGS
477};
478#define DQUOT_USAGE_ENABLED (1 << _DQUOT_USAGE_ENABLED * MAXQUOTAS)
479#define DQUOT_LIMITS_ENABLED (1 << _DQUOT_LIMITS_ENABLED * MAXQUOTAS)
480#define DQUOT_SUSPENDED (1 << _DQUOT_SUSPENDED * MAXQUOTAS)
481#define DQUOT_STATE_FLAGS (DQUOT_USAGE_ENABLED | DQUOT_LIMITS_ENABLED | \
482 DQUOT_SUSPENDED)
483/* Other quota flags */
484#define DQUOT_STATE_LAST (_DQUOT_STATE_FLAGS * MAXQUOTAS)
485#define DQUOT_QUOTA_SYS_FILE (1 << DQUOT_STATE_LAST)
486 /* Quota file is a special
487 * system file and user cannot
488 * touch it. Filesystem is
489 * responsible for setting
490 * S_NOQUOTA, S_NOATIME flags
491 */
492#define DQUOT_NEGATIVE_USAGE (1 << (DQUOT_STATE_LAST + 1))
493 /* Allow negative quota usage */
494/* Do not track dirty dquots in a list */
495#define DQUOT_NOLIST_DIRTY (1 << (DQUOT_STATE_LAST + 2))
496
497static inline unsigned int dquot_state_flag(unsigned int flags, int type)
498{
499 return flags << type;
500}
501
502static inline unsigned int dquot_generic_flag(unsigned int flags, int type)
503{
504 return (flags >> type) & DQUOT_STATE_FLAGS;
505}
506
507/* Bitmap of quota types where flag is set in flags */
508static __always_inline unsigned dquot_state_types(unsigned flags, unsigned flag)
509{
510 BUILD_BUG_ON_NOT_POWER_OF_2(flag);
511 return (flags / flag) & ((1 << MAXQUOTAS) - 1);
512}
