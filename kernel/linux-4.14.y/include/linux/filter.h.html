322#define bytes_to_bpf_size(bytes) \
323({ \
324 int bpf_size = -EINVAL; \
325 \
326 if (bytes == sizeof(u8)) \
327 bpf_size = BPF_B; \
328 else if (bytes == sizeof(u16)) \
329 bpf_size = BPF_H; \
330 else if (bytes == sizeof(u32)) \
331 bpf_size = BPF_W; \
332 else if (bytes == sizeof(u64)) \
333 bpf_size = BPF_DW; \
334 \
335 bpf_size; \
336})
337
338#define bpf_size_to_bytes(bpf_size) \
339({ \
340 int bytes = -EINVAL; \
341 \
342 if (bpf_size == BPF_B) \
343 bytes = sizeof(u8); \
344 else if (bpf_size == BPF_H) \
345 bytes = sizeof(u16); \
346 else if (bpf_size == BPF_W) \
347 bytes = sizeof(u32); \
348 else if (bpf_size == BPF_DW) \
349 bytes = sizeof(u64); \
350 \
351 bytes; \
352})

354#define BPF_SIZEOF(type) \
355 ({ \
356 const int __size = bytes_to_bpf_size(sizeof(type)); \
357 BUILD_BUG_ON(__size < 0); \
358 __size; \
359 })

361#define BPF_FIELD_SIZEOF(type, field) \

498/* compute the linear packet data range [data, data_end) which
499 * will be accessed by cls_bpf, act_bpf and lwt programs
500 */
501static inline void bpf_compute_data_end(struct sk_buff *skb)
502{
503 struct bpf_skb_data_end *cb = (struct bpf_skb_data_end *)skb->cb;
504
505 BUILD_BUG_ON(sizeof(*cb) > FIELD_SIZEOF(struct sk_buff, cb));
506 cb->data_end = skb->data + skb_headlen(skb);
507}
508
509static inline u8 *bpf_skb_cb(struct sk_buff *skb)
510{
511 /* eBPF programs may read/write skb->cb[] area to transfer meta
512 * data between tail calls. Since this also needs to work with
513 * tc, that scratch memory is mapped to qdisc_skb_cb's data area.
514 *
515 * In some socket filter cases, the cb unfortunately needs to be
516 * saved/restored so that protocol specific skb->cb[] data won't
517 * be lost. In any case, due to unpriviledged eBPF programs
518 * attached to sockets, we need to clear the bpf_skb_cb() area
519 * to not leak previous contents to user space.
520 */
521 BUILD_BUG_ON(FIELD_SIZEOF(struct __sk_buff, cb) != BPF_SKB_CB_LEN);
522 BUILD_BUG_ON(FIELD_SIZEOF(struct __sk_buff, cb) !=
523 FIELD_SIZEOF(struct qdisc_skb_cb, data));
524
525 return qdisc_skb_cb(skb)->data;
526}
527
528static inline u32 bpf_prog_run_save_cb(const struct bpf_prog *prog,
529 struct sk_buff *skb)
530{
531 u8 *cb_data = bpf_skb_cb(skb);
532 u8 cb_saved[BPF_SKB_CB_LEN];
533 u32 res;
534
535 if (unlikely(prog->cb_access)) {
536 memcpy(cb_saved, cb_data, sizeof(cb_saved));
537 memset(cb_data, 0, sizeof(cb_saved));
538 }
539
540 res = BPF_PROG_RUN(prog, skb);
541
542 if (unlikely(prog->cb_access))
543 memcpy(cb_data, cb_saved, sizeof(cb_saved));
544
545 return res;
546}
547
548static inline u32 bpf_prog_run_clear_cb(const struct bpf_prog *prog,
549 struct sk_buff *skb)
550{
551 u8 *cb_data = bpf_skb_cb(skb);
552
553 if (unlikely(prog->cb_access))
554 memset(cb_data, 0, BPF_SKB_CB_LEN);
555
556 return BPF_PROG_RUN(prog, skb);
557}

559static __always_inline u32 bpf_prog_run_xdp(const struct bpf_prog *prog,
560 struct xdp_buff *xdp)
561{
562 /* Caller needs to hold rcu_read_lock() (!), otherwise program
563 * can be released while still running, or map elements could be
564 * freed early while still having concurrent users. XDP fastpath
565 * already takes rcu_read_lock() when fetching the program, so
566 * it's not necessary here anymore.
567 */
568 return BPF_PROG_RUN(prog, xdp);
569}

588static inline bool bpf_prog_was_classic(const struct bpf_prog *prog)
589{
590 /* When classic BPF programs have been loaded and the arch
591 * does not have a classic BPF JIT (anymore), they have been

598static inline bool
599bpf_ctx_narrow_access_ok(u32 off, u32 size, const u32 size_default)
600{
601 bool off_ok;
602#ifdef __LITTLE_ENDIAN
603 off_ok = (off & (size_default - 1)) == 0;
604#else
605 off_ok = (off & (size_default - 1)) + size == size_default;
606#endif
607 return off_ok && size <= size_default && (size & (size - 1)) == 0;
608}

619static inline void bpf_prog_unlock_ro(struct bpf_prog *fp)

889#define BPF_ANC BIT(15)

891static inline bool bpf_needs_clear_a(const struct sock_filter *first)
892{
893 switch (first->code) {
894 case BPF_RET | BPF_K:
895 case BPF_LD | BPF_W | BPF_LEN:
896 return false;
897
898 case BPF_LD | BPF_W | BPF_ABS:
899 case BPF_LD | BPF_H | BPF_ABS:
900 case BPF_LD | BPF_B | BPF_ABS:
901 if (first->k == SKF_AD_OFF + SKF_AD_ALU_XOR_X)
902 return true;
903 return false;
904
905 default:
906 return true;
907 }
908}

910static inline u16 bpf_anc_helper(const struct sock_filter *ftest)
911{
912 BUG_ON(ftest->code & BPF_ANC);
913
914 switch (ftest->code) {
915 case BPF_LD | BPF_W | BPF_ABS:
916 case BPF_LD | BPF_H | BPF_ABS:
917 case BPF_LD | BPF_B | BPF_ABS:
918#define BPF_ANCILLARY(CODE) case SKF_AD_OFF + SKF_AD_##CODE: \
919 return BPF_ANC | SKF_AD_##CODE
920 switch (ftest->k) {
921 BPF_ANCILLARY(PROTOCOL);
922 BPF_ANCILLARY(PKTTYPE);
923 BPF_ANCILLARY(IFINDEX);
924 BPF_ANCILLARY(NLATTR);
925 BPF_ANCILLARY(NLATTR_NEST);
926 BPF_ANCILLARY(MARK);
927 BPF_ANCILLARY(QUEUE);
928 BPF_ANCILLARY(HATYPE);
929 BPF_ANCILLARY(RXHASH);
930 BPF_ANCILLARY(CPU);
931 BPF_ANCILLARY(ALU_XOR_X);
932 BPF_ANCILLARY(VLAN_TAG);
933 BPF_ANCILLARY(VLAN_TAG_PRESENT);
934 BPF_ANCILLARY(PAY_OFFSET);
935 BPF_ANCILLARY(RANDOM);
936 BPF_ANCILLARY(VLAN_TPID);
937 }
938 /* Fallthrough. */
939 default:
940 return ftest->code;
941 }
942}
