21#ifndef DCA_H
22#define DCA_H
23
24#include <linux/pci.h>
25
26/* DCA Provider API */
27
28/* DCA Notifier Interface */
29void dca_register_notify(struct notifier_block *nb);
30void dca_unregister_notify(struct notifier_block *nb);
31
32#define DCA_PROVIDER_ADD 0x0001
33#define DCA_PROVIDER_REMOVE 0x0002
34
35struct dca_provider {
36 struct list_head node;
37 const struct dca_ops *ops;
38 struct device *cd;
39 int id;
40};
41
42struct dca_domain {
43 struct list_head node;
44 struct list_head dca_providers;
45 struct pci_bus *pci_rc;
46};
47
48struct dca_ops {
49 int (*add_requester) (struct dca_provider *, struct device *);
50 int (*remove_requester) (struct dca_provider *, struct device *);
51 u8 (*get_tag) (struct dca_provider *, struct device *,
52 int cpu);
53 int (*dev_managed) (struct dca_provider *, struct device *);
54};
55
56struct dca_provider *alloc_dca_provider(const struct dca_ops *ops,
57 int priv_size);
58void free_dca_provider(struct dca_provider *dca);
59int register_dca_provider(struct dca_provider *dca, struct device *dev);
60void unregister_dca_provider(struct dca_provider *dca, struct device *dev);
61
62static inline void *dca_priv(struct dca_provider *dca)
63{
64 return (void *)dca + sizeof(struct dca_provider);
65}
66
67/* Requester API */
68#define DCA_GET_TAG_TWO_ARGS
69int dca_add_requester(struct device *dev);
70int dca_remove_requester(struct device *dev);
71u8 dca_get_tag(int cpu);
72u8 dca3_get_tag(struct device *dev, int cpu);
73
74/* internal stuff */
75int __init dca_sysfs_init(void);
76void __exit dca_sysfs_exit(void);
77int dca_sysfs_add_provider(struct dca_provider *dca, struct device *dev);
78void dca_sysfs_remove_provider(struct dca_provider *dca);
79int dca_sysfs_add_req(struct dca_provider *dca, struct device *dev, int slot);
80void dca_sysfs_remove_req(struct dca_provider *dca, int slot);
81
82#endif /* DCA_H */
