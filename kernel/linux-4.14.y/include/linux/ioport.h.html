15/*
16 * Resources are tree-like, allowing
17 * nesting etc..
18 */
19struct resource {
20 resource_size_t start;
21 resource_size_t end;
22 const char *name;
23 unsigned long flags;
24 unsigned long desc;
25 struct resource *parent, *sibling, *child;
26};

28/*
29 * IO resources have these defined flags.
30 *
31 * PCI devices expose these flags to userspace in the "resource" sysfs file,
32 * so don't move them.
33 */
34#define IORESOURCE_BITS 0x000000ff /* Bus-specific bits */
35
36#define IORESOURCE_TYPE_BITS 0x00001f00 /* Resource type */
37#define IORESOURCE_IO 0x00000100 /* PCI/ISA I/O ports */
38#define IORESOURCE_MEM 0x00000200
39#define IORESOURCE_REG 0x00000300 /* Register offsets */
40#define IORESOURCE_IRQ 0x00000400
41#define IORESOURCE_DMA 0x00000800
42#define IORESOURCE_BUS 0x00001000
43
44#define IORESOURCE_PREFETCH 0x00002000 /* No side effects */
45#define IORESOURCE_READONLY 0x00004000
46#define IORESOURCE_CACHEABLE 0x00008000
47#define IORESOURCE_RANGELENGTH 0x00010000
48#define IORESOURCE_SHADOWABLE 0x00020000
49
50#define IORESOURCE_SIZEALIGN 0x00040000 /* size indicates alignment */
51#define IORESOURCE_STARTALIGN 0x00080000 /* start field is alignment */
52
53#define IORESOURCE_MEM_64 0x00100000
54#define IORESOURCE_WINDOW 0x00200000 /* forwarded by bridge */
55#define IORESOURCE_MUXED 0x00400000 /* Resource is software muxed */
56
57#define IORESOURCE_EXT_TYPE_BITS 0x01000000 /* Resource extended types */
58#define IORESOURCE_SYSRAM 0x01000000 /* System RAM (modifier) */
59
60#define IORESOURCE_EXCLUSIVE 0x08000000 /* Userland may not map this resource */
61
62#define IORESOURCE_DISABLED 0x10000000
63#define IORESOURCE_UNSET 0x20000000 /* No address assigned yet */
64#define IORESOURCE_AUTO 0x40000000
65#define IORESOURCE_BUSY 0x80000000 /* Driver has marked this resource busy */
66
67/* I/O resource extended types */
68#define IORESOURCE_SYSTEM_RAM (IORESOURCE_MEM|IORESOURCE_SYSRAM)
69
70/* PnP IRQ specific bits (IORESOURCE_BITS) */
71#define IORESOURCE_IRQ_HIGHEDGE (1<<0)
72#define IORESOURCE_IRQ_LOWEDGE (1<<1)
73#define IORESOURCE_IRQ_HIGHLEVEL (1<<2)
74#define IORESOURCE_IRQ_LOWLEVEL (1<<3)
75#define IORESOURCE_IRQ_SHAREABLE (1<<4)
76#define IORESOURCE_IRQ_OPTIONAL (1<<5)
77
78/* PnP DMA specific bits (IORESOURCE_BITS) */
79#define IORESOURCE_DMA_TYPE_MASK (3<<0)
80#define IORESOURCE_DMA_8BIT (0<<0)
81#define IORESOURCE_DMA_8AND16BIT (1<<0)
82#define IORESOURCE_DMA_16BIT (2<<0)
83
84#define IORESOURCE_DMA_MASTER (1<<2)
85#define IORESOURCE_DMA_BYTE (1<<3)
86#define IORESOURCE_DMA_WORD (1<<4)
87
88#define IORESOURCE_DMA_SPEED_MASK (3<<6)
89#define IORESOURCE_DMA_COMPATIBLE (0<<6)
90#define IORESOURCE_DMA_TYPEA (1<<6)
91#define IORESOURCE_DMA_TYPEB (2<<6)
92#define IORESOURCE_DMA_TYPEF (3<<6)
93
94/* PnP memory I/O specific bits (IORESOURCE_BITS) */
95#define IORESOURCE_MEM_WRITEABLE (1<<0) /* dup: IORESOURCE_READONLY */
96#define IORESOURCE_MEM_CACHEABLE (1<<1) /* dup: IORESOURCE_CACHEABLE */
97#define IORESOURCE_MEM_RANGELENGTH (1<<2) /* dup: IORESOURCE_RANGELENGTH */
98#define IORESOURCE_MEM_TYPE_MASK (3<<3)
99#define IORESOURCE_MEM_8BIT (0<<3)
100#define IORESOURCE_MEM_16BIT (1<<3)
101#define IORESOURCE_MEM_8AND16BIT (2<<3)
102#define IORESOURCE_MEM_32BIT (3<<3)
103#define IORESOURCE_MEM_SHADOWABLE (1<<5) /* dup: IORESOURCE_SHADOWABLE */
104#define IORESOURCE_MEM_EXPANSIONROM (1<<6)
105
106/* PnP I/O specific bits (IORESOURCE_BITS) */
107#define IORESOURCE_IO_16BIT_ADDR (1<<0)
108#define IORESOURCE_IO_FIXED (1<<1)
109#define IORESOURCE_IO_SPARSE (1<<2)
110
111/* PCI ROM control bits (IORESOURCE_BITS) */
112#define IORESOURCE_ROM_ENABLE (1<<0) /* ROM is enabled, same as PCI_ROM_ADDRESS_ENABLE */
113#define IORESOURCE_ROM_SHADOW (1<<1) /* Use RAM image, not ROM BAR */
114
115/* PCI control bits. Shares IORESOURCE_BITS with above PCI ROM. */
116#define IORESOURCE_PCI_FIXED (1<<4) /* Do not move resource */
117#define IORESOURCE_PCI_EA_BEI (1<<5) /* BAR Equivalent Indicator */

119/*
120 * I/O Resource Descriptors
121 *
122 * Descriptors are used by walk_iomem_res_desc() and region_intersects()
123 * for searching a specific resource range in the iomem table. Assign
124 * a new descriptor when a resource range supports the search interfaces.
125 * Otherwise, resource.desc must be set to IORES_DESC_NONE (0).
126 */
127enum {
128 IORES_DESC_NONE = 0,
129 IORES_DESC_CRASH_KERNEL = 1,
130 IORES_DESC_ACPI_TABLES = 2,
131 IORES_DESC_ACPI_NV_STORAGE = 3,
132 IORES_DESC_PERSISTENT_MEMORY = 4,
133 IORES_DESC_PERSISTENT_MEMORY_LEGACY = 5,
134 IORES_DESC_DEVICE_PRIVATE_MEMORY = 6,
135 IORES_DESC_DEVICE_PUBLIC_MEMORY = 7,
136};

138/* helpers to define resources */
139#define DEFINE_RES_NAMED(_start, _size, _name, _flags) \
140 { \
141 .start = (_start), \
142 .end = (_start) + (_size) - 1, \
143 .name = (_name), \
144 .flags = (_flags), \
145 .desc = IORES_DESC_NONE, \
146 }
147
148#define DEFINE_RES_IO_NAMED(_start, _size, _name) \
149 DEFINE_RES_NAMED((_start), (_size), (_name), IORESOURCE_IO)
150#define DEFINE_RES_IO(_start, _size) \
151 DEFINE_RES_IO_NAMED((_start), (_size), NULL)
152
153#define DEFINE_RES_MEM_NAMED(_start, _size, _name) \
154 DEFINE_RES_NAMED((_start), (_size), (_name), IORESOURCE_MEM)
155#define DEFINE_RES_MEM(_start, _size) \
156 DEFINE_RES_MEM_NAMED((_start), (_size), NULL)
157
158#define DEFINE_RES_IRQ_NAMED(_irq, _name) \
159 DEFINE_RES_NAMED((_irq), 1, (_name), IORESOURCE_IRQ)
160#define DEFINE_RES_IRQ(_irq) \
161 DEFINE_RES_IRQ_NAMED((_irq), NULL)
162
163#define DEFINE_RES_DMA_NAMED(_dma, _name) \
164 DEFINE_RES_NAMED((_dma), 1, (_name), IORESOURCE_DMA)
165#define DEFINE_RES_DMA(_dma) \
166 DEFINE_RES_DMA_NAMED((_dma), NULL)

196static inline resource_size_t resource_size(const struct resource *res)
197{
198 return res->end - res->start + 1;
199}
200static inline unsigned long resource_type(const struct resource *res)
201{
202 return res->flags & IORESOURCE_TYPE_BITS;
203}
204static inline unsigned long resource_ext_type(const struct resource *res)
205{
206 return res->flags & IORESOURCE_EXT_TYPE_BITS;
207}
208/* True iff r1 completely contains r2 */
209static inline bool resource_contains(struct resource *r1, struct resource *r2)
210{
211 if (resource_type(r1) != resource_type(r2))
212 return false;
213 if (r1->flags & IORESOURCE_UNSET || r2->flags & IORESOURCE_UNSET)
214 return false;
215 return r1->start <= r2->start && r1->end >= r2->end;
216}

280/* True if any part of r1 overlaps r2 */
281static inline bool resource_overlaps(struct resource *r1, struct resource *r2)
282{
283 return (r1->start <= r2->end && r1->end >= r2->start);
