23static inline void br_drop_fake_rtable(struct sk_buff *skb)
24{
25 struct dst_entry *dst = skb_dst(skb);
26
27 if (dst && (dst->flags & DST_FAKE_RTABLE))
28 skb_dst_drop(skb);
29}
