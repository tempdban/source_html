101enum tick_dep_bits {
102 TICK_DEP_BIT_POSIX_TIMER = 0,
103 TICK_DEP_BIT_PERF_EVENTS = 1,
104 TICK_DEP_BIT_SCHED = 2,
105 TICK_DEP_BIT_CLOCK_UNSTABLE = 3
106};
107
108#define TICK_DEP_MASK_NONE 0
109#define TICK_DEP_MASK_POSIX_TIMER (1 << TICK_DEP_BIT_POSIX_TIMER)
110#define TICK_DEP_MASK_PERF_EVENTS (1 << TICK_DEP_BIT_PERF_EVENTS)
111#define TICK_DEP_MASK_SCHED (1 << TICK_DEP_BIT_SCHED)
112#define TICK_DEP_MASK_CLOCK_UNSTABLE (1 << TICK_DEP_BIT_CLOCK_UNSTABLE)

139#ifdef CONFIG_NO_HZ_FULL
140extern bool tick_nohz_full_running;
141extern cpumask_var_t tick_nohz_full_mask;
142extern cpumask_var_t housekeeping_mask;
143
144static inline bool tick_nohz_full_enabled(void)
145{
146 if (!context_tracking_is_enabled())
147 return false;
148
149 return tick_nohz_full_running;
150}
151
152static inline bool tick_nohz_full_cpu(int cpu)
153{
154 if (!tick_nohz_full_enabled())
155 return false;
156
157 return cpumask_test_cpu(cpu, tick_nohz_full_mask);
158}
159
160static inline void tick_nohz_full_add_cpus_to(struct cpumask *mask)
161{
162 if (tick_nohz_full_enabled())
163 cpumask_or(mask, mask, tick_nohz_full_mask);
164}
165
166static inline int housekeeping_any_cpu(void)
167{
168 return cpumask_any_and(housekeeping_mask, cpu_online_mask);
169}

274static inline bool is_housekeeping_cpu(int cpu)
275{
276#ifdef CONFIG_NO_HZ_FULL
