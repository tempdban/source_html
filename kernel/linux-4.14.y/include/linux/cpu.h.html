2/*
3 * include/linux/cpu.h - generic cpu definition
4 *
5 * This is mainly for topological representation. We define the
6 * basic 'struct cpu' here, which can be embedded in per-arch
7 * definitions of processors.
8 *
9 * Basic handling of the devices is done in drivers/base/cpu.c
10 *
11 * CPUs are exported via sysfs in the devices/system/cpu
12 * directory.
13 */

70#define CPU_ONLINE 0x0002 /* CPU (unsigned)v is up */
71#define CPU_UP_PREPARE 0x0003 /* CPU (unsigned)v coming up */
72#define CPU_DEAD 0x0007 /* CPU (unsigned)v dead */
73#define CPU_POST_DEAD 0x0009 /* CPU (unsigned)v dead, cpu_hotplug
74 * lock is dropped */
75#define CPU_BROKEN 0x000B /* CPU (unsigned)v did not die properly,
76 * perhaps due to preemption. */
77
78/* Used for CPU hotplug events occurring while tasks are frozen due to a suspend
79 * operation in progress
80 */
81#define CPU_TASKS_FROZEN 0x0010
82
83#define CPU_ONLINE_FROZEN (CPU_ONLINE | CPU_TASKS_FROZEN)
84#define CPU_UP_PREPARE_FROZEN (CPU_UP_PREPARE | CPU_TASKS_FROZEN)
85#define CPU_UP_CANCELED_FROZEN (CPU_UP_CANCELED | CPU_TASKS_FROZEN)
86#define CPU_DOWN_PREPARE_FROZEN (CPU_DOWN_PREPARE | CPU_TASKS_FROZEN)
87#define CPU_DOWN_FAILED_FROZEN (CPU_DOWN_FAILED | CPU_TASKS_FROZEN)
88#define CPU_DEAD_FROZEN (CPU_DEAD | CPU_TASKS_FROZEN)
