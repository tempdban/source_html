21#include <uapi/linux/ip.h>

23static inline struct iphdr *ip_hdr(const struct sk_buff *skb)
24{
25 return (struct iphdr *)skb_network_header(skb);
26}
27
28static inline struct iphdr *inner_ip_hdr(const struct sk_buff *skb)
29{
30 return (struct iphdr *)skb_inner_network_header(skb);
31}
32
33static inline struct iphdr *ipip_hdr(const struct sk_buff *skb)
34{
35 return (struct iphdr *)skb_transport_header(skb);
36}
