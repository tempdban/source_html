16#ifndef _LINUX_UUID_H_
17#define _LINUX_UUID_H_
18
19#include <uapi/linux/uuid.h>
20
21#define UUID_SIZE 16
22
23typedef struct {
24 __u8 b[UUID_SIZE];
25} uuid_t;
26
27#define UUID_INIT(a, b, c, d0, d1, d2, d3, d4, d5, d6, d7) \
28((uuid_t) \
29{{ ((a) >> 24) & 0xff, ((a) >> 16) & 0xff, ((a) >> 8) & 0xff, (a) & 0xff, \
30 ((b) >> 8) & 0xff, (b) & 0xff, \
31 ((c) >> 8) & 0xff, (c) & 0xff, \
32 (d0), (d1), (d2), (d3), (d4), (d5), (d6), (d7) }})
33
34/*
35 * The length of a UUID string ("aaaaaaaa-bbbb-cccc-dddd-eeeeeeeeeeee")
36 * not including trailing NUL.
37 */
38#define UUID_STRING_LEN 36
39
40extern const guid_t guid_null;
41extern const uuid_t uuid_null;
42
43static inline bool guid_equal(const guid_t *u1, const guid_t *u2)
44{
45 return memcmp(u1, u2, sizeof(guid_t)) == 0;
46}
47
48static inline void guid_copy(guid_t *dst, const guid_t *src)
49{
50 memcpy(dst, src, sizeof(guid_t));
51}
52
53static inline bool guid_is_null(const guid_t *guid)
54{
55 return guid_equal(guid, &guid_null);
56}
57
58static inline bool uuid_equal(const uuid_t *u1, const uuid_t *u2)
59{
60 return memcmp(u1, u2, sizeof(uuid_t)) == 0;
61}
62
63static inline void uuid_copy(uuid_t *dst, const uuid_t *src)
64{
65 memcpy(dst, src, sizeof(uuid_t));
66}
67
68static inline bool uuid_is_null(const uuid_t *uuid)
69{
70 return uuid_equal(uuid, &uuid_null);
71}
72
73void generate_random_uuid(unsigned char uuid[16]);
74
75extern void guid_gen(guid_t *u);
76extern void uuid_gen(uuid_t *u);
77
78bool __must_check uuid_is_valid(const char *uuid);
79
80extern const u8 guid_index[16];
81extern const u8 uuid_index[16];

83int guid_parse(const char *uuid, guid_t *u);
84int uuid_parse(const char *uuid, uuid_t *u);

86/* backwards compatibility, don't use in new code */
87#define uuid_le_gen(u) guid_gen(u)
88#define uuid_le_to_bin(guid, u) guid_parse(guid, u)
89
90static inline int uuid_le_cmp(const guid_t u1, const guid_t u2)
91{
92 return memcmp(&u1, &u2, sizeof(guid_t));
93}
