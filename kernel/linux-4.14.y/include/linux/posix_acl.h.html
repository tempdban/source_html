17struct posix_acl_entry {
18 short e_tag;
19 unsigned short e_perm;
20 union {
21 kuid_t e_uid;
22 kgid_t e_gid;
23 };
24};
25
26struct posix_acl {
27 atomic_t a_refcount;
28 struct rcu_head a_rcu;
29 unsigned int a_count;
30 struct posix_acl_entry a_entries[0];
31};
32
33#define FOREACH_ACL_ENTRY(pa, acl, pe) \
34 for(pa=(acl)->a_entries, pe=pa+(acl)->a_count; pa<pe; pa++)
35
36
37/*
38 * Duplicate an ACL handle.
39 */
40static inline struct posix_acl *
41posix_acl_dup(struct posix_acl *acl)
42{
43 if (acl)
44 atomic_inc(&acl->a_refcount);
45 return acl;
46}
47
48/*
49 * Free an ACL handle.
50 */
51static inline void
52posix_acl_release(struct posix_acl *acl)
53{
54 if (acl && atomic_dec_and_test(&acl->a_refcount))
55 kfree_rcu(acl, a_rcu);
56}
