192static __inline__ bool bad_mask(__be32 mask, __be32 addr)
193{
194 __u32 hmask;
195 if (addr & (mask = ~mask))
196 return true;
197 hmask = ntohl(mask);
198 if (hmask & (hmask+1))
199 return true;
200 return false;
201}

212static inline struct in_device *__in_dev_get_rcu(const struct net_device *dev)

217static inline struct in_device *in_dev_get(const struct net_device *dev)
218{
219 struct in_device *in_dev;
220
221 rcu_read_lock();
222 in_dev = __in_dev_get_rcu(dev);
223 if (in_dev)
224 refcount_inc(&in_dev->refcnt);
225 rcu_read_unlock();
226 return in_dev;
227}

254static __inline__ __be32 inet_make_mask(int logmask)
255{
256 if (logmask)
257 return htonl(~((1U<<(32-logmask))-1));
258 return 0;
259}

261static __inline__ int inet_mask_len(__be32 mask)
262{
263 __u32 hmask = ntohl(mask);
264 if (!hmask)
265 return 0;
266 return 32 - ffz(~hmask);
267}
