313/*
314 * F2FS uses 4 bytes to represent block address. As a result, supported size of

319/*
320 * Note that f2fs_sit_entry->vblocks has the following bit-field information.
321 * [15:10] : allocation type such as CURSEG_XXXX_TYPE
322 * [9:0] : valid block count
323 */
324#define SIT_VBLOCKS_SHIFT 10
325#define SIT_VBLOCKS_MASK ((1 << SIT_VBLOCKS_SHIFT) - 1)
326#define GET_SIT_VBLOCKS(raw_sit) \
327 (le16_to_cpu((raw_sit)->vblocks) & SIT_VBLOCKS_MASK)
328#define GET_SIT_TYPE(raw_sit) \
329 ((le16_to_cpu((raw_sit)->vblocks) & ~SIT_VBLOCKS_MASK) \
330 >> SIT_VBLOCKS_SHIFT)

342/*
343 * For segment summary
344 *
345 * One summary block contains exactly 512 summary entries, which represents
346 * exactly 2MB segment by default. Not allow to change the basic units.
347 *
348 * NOTE: For initializing fields, you must use set_summary
349 *
350 * - If data page, nid represents dnode's nid
351 * - If node page, nid represents the node page's nid.
352 *
353 * The ofs_in_node is used by only data page. It represents offset
354 * from node's page's beginning to get a data block address.
355 * ex) data_blkaddr = (block_t)(nodepage_start_address + ofs_in_node)
356 */
357#define ENTRIES_IN_SUM 512
358#define SUMMARY_SIZE (7) /* sizeof(struct summary) */
359#define SUM_FOOTER_SIZE (5) /* sizeof(struct summary_footer) */
360#define SUM_ENTRY_SIZE (SUMMARY_SIZE * ENTRIES_IN_SUM)

475/*
476 * space utilization of regular dentry and inline dentry (w/o extra reservation)
477 * regular dentry inline dentry
478 * bitmap 1 * 27 = 27 1 * 23 = 23
479 * reserved 1 * 3 = 3 1 * 7 = 7
480 * dentry 11 * 214 = 2354 11 * 182 = 2002
481 * filename 8 * 214 = 1712 8 * 182 = 1456
482 * total 4096 3488
483 *
484 * Note: there are more reserved space in inline dentry than in regular
485 * dentry, when converting inline dentry we should handle this carefully.
486 */
487#define NR_DENTRY_IN_BLOCK 214 /* the number of dentry in a block */
488#define SIZE_OF_DIR_ENTRY 11 /* by byte */
489#define SIZE_OF_DENTRY_BITMAP ((NR_DENTRY_IN_BLOCK + BITS_PER_BYTE - 1) / \
490 BITS_PER_BYTE)
491#define SIZE_OF_RESERVED (PAGE_SIZE - ((SIZE_OF_DIR_ENTRY + \
492 F2FS_SLOT_LEN) * \
493 NR_DENTRY_IN_BLOCK + SIZE_OF_DENTRY_BITMAP))
494
