21struct plat_serial8250_port {
22 unsigned long iobase; /* io base address */
23 void __iomem *membase; /* ioremap cookie or NULL */
24 resource_size_t mapbase; /* resource base */
25 unsigned int irq; /* interrupt number */
26 unsigned long irqflags; /* request_irq flags */
27 unsigned int uartclk; /* UART clock rate */
28 void *private_data;
29 unsigned char regshift; /* register shift */
30 unsigned char iotype; /* UPIO_* */
31 unsigned char hub6;
32 upf_t flags; /* UPF_* flags */
33 unsigned int type; /* If UPF_FIXED_TYPE */
34 unsigned int (*serial_in)(struct uart_port *, int);
35 void (*serial_out)(struct uart_port *, int, int);
36 void (*set_termios)(struct uart_port *,
37 struct ktermios *new,

39 void (*set_ldisc)(struct uart_port *,
40 struct ktermios *);

42 int (*handle_irq)(struct uart_port *);
43 void (*pm)(struct uart_port *, unsigned int state,
44 unsigned old);
45 void (*handle_break)(struct uart_port *);
46};

69/**
70 * 8250 core driver operations
71 *
72 * @setup_irq() Setup irq handling. The universal 8250 driver links this
73 * port to the irq chain. Other drivers may @request_irq().
74 * @release_irq() Undo irq handling. The universal 8250 driver unlinks
75 * the port from the irq chain.
76 */
77struct uart_8250_ops {
78 int (*setup_irq)(struct uart_8250_port *);
79 void (*release_irq)(struct uart_8250_port *);
80};

89/*
90 * This should be used by drivers which want to register
91 * their own 8250 ports without registering their own
92 * platform device. Using these will make your driver
93 * dependent on the 8250 driver.
94 */
95
96struct uart_8250_port {
97 struct uart_port port;
98 struct timer_list timer; /* "no irq" timer */
99 struct list_head list; /* ports on this IRQ */
100 u32 capabilities; /* port capabilities */
101 unsigned short bugs; /* port bugs */
102 bool fifo_bug; /* min RX trigger if enabled */
103 unsigned int tx_loadsz; /* transmit fifo load size */
104 unsigned char acr;
105 unsigned char fcr;
106 unsigned char ier;
107 unsigned char lcr;
108 unsigned char mcr;
109 unsigned char mcr_mask; /* mask of user bits */
110 unsigned char mcr_force; /* mask of forced bits */
111 unsigned char cur_iotype; /* Running I/O type */
112 unsigned int rpm_tx_active;
113 unsigned char canary; /* non-zero during system sleep
114 * if no_console_suspend
115 */
116 unsigned char probe;
117#define UART_PROBE_RSA (1 << 0)
118
119 /*
120 * Some bits in registers are cleared on a read, so they must
121 * be saved whenever the register is read but the bits will not
122 * be immediately processed.
123 */
124#define LSR_SAVE_FLAGS UART_LSR_BRK_ERROR_BITS
125 unsigned char lsr_saved_flags;
126#define MSR_SAVE_FLAGS UART_MSR_ANY_DELTA
127 unsigned char msr_saved_flags;
128
129 struct uart_8250_dma *dma;
130 const struct uart_8250_ops *ops;
131
132 /* 8250 specific callbacks */
133 int (*dl_read)(struct uart_8250_port *);
134 void (*dl_write)(struct uart_8250_port *, int);
135
136 struct uart_8250_em485 *em485;
137};
