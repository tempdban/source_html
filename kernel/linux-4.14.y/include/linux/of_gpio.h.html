25/*
26 * This is Linux-specific flags. By default controllers' and Linux' mapping
27 * match, but GPIO controllers are free to translate their own flags to
28 * Linux-specific in their .xlate callback. Though, 1:1 mapping is recommended.
29 */
30enum of_gpio_flags {
31 OF_GPIO_ACTIVE_LOW = 0x1,
32 OF_GPIO_SINGLE_ENDED = 0x2,
33 OF_GPIO_OPEN_DRAIN = 0x4,
34 OF_GPIO_SLEEP_MAY_LOOSE_VALUE = 0x8,
35};

91/**
92 * of_gpio_named_count() - Count GPIOs for a device
93 * @np: device node to count GPIOs for
94 * @propname: property name containing gpio specifier(s)
95 *
96 * The function returns the count of GPIOs specified for a node.
97 * Note that the empty GPIO specifiers count too. Returns either
98 * Number of gpios defined in property,
99 * -EINVAL for an incorrectly formed gpios property, or
100 * -ENOENT for a missing gpios property
101 *
102 * Example:
103 * gpios = <0
104 * &gpio1 1 2
105 * 0
106 * &gpio2 3 4>;
107 *
108 * The above example defines four GPIOs, two of which are not specified.
109 * This function will return '4'
110 */
111static inline int of_gpio_named_count(struct device_node *np, const char* propname)
112{
113 return of_count_phandle_with_args(np, propname, "#gpio-cells");
114}
115
116/**
117 * of_gpio_count() - Count GPIOs for a device
118 * @np: device node to count GPIOs for
119 *
120 * Same as of_gpio_named_count, but hard coded to use the 'gpios' property
121 */
122static inline int of_gpio_count(struct device_node *np)
123{
124 return of_gpio_named_count(np, "gpios");
125}
126
127static inline int of_get_gpio_flags(struct device_node *np, int index,
128 enum of_gpio_flags *flags)
129{
130 return of_get_named_gpio_flags(np, "gpios", index, flags);
131}
132
133/**
134 * of_get_named_gpio() - Get a GPIO number to use with GPIO API
135 * @np: device node to get GPIO from
136 * @propname: Name of property containing gpio specifier(s)
137 * @index: index of the GPIO
138 *
139 * Returns GPIO number to use with Linux generic GPIO API, or one of the errno
140 * value on the error condition.
141 */
142static inline int of_get_named_gpio(struct device_node *np,
143 const char *propname, int index)
144{
145 return of_get_named_gpio_flags(np, propname, index, NULL);
146}
147
148/**
149 * of_get_gpio() - Get a GPIO number to use with GPIO API
150 * @np: device node to get GPIO from
151 * @index: index of the GPIO
152 *
153 * Returns GPIO number to use with Linux generic GPIO API, or one of the errno
154 * value on the error condition.
155 */
156static inline int of_get_gpio(struct device_node *np, int index)
157{
158 return of_get_gpio_flags(np, index, NULL);
159}
