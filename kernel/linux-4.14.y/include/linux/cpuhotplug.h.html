10 * BP AP BP AP
11 *
12 * OFFLINE OFFLINE
13 * | ^
14 * v |
15 * BRINGUP_CPU->AP_OFFLINE BRINGUP_CPU <- AP_IDLE_DEAD (idle thread/play_dead)
16 * | AP_OFFLINE
17 * v (IRQ-off) ,---------------^
18 * AP_ONLNE | (stop_machine)
19 * | TEARDOWN_CPU <- AP_ONLINE_IDLE

21 * v |
22 * AP_ACTIVE AP_ACTIVE
23 */

182/**
183 * cpuhp_setup_state - Setup hotplug state callbacks with calling the callbacks
184 * @state: The state for which the calls are installed
185 * @name: Name of the callback (will be used in debug output)
186 * @startup: startup callback function
187 * @teardown: teardown callback function
188 *
189 * Installs the callback functions and invokes the startup callback on
190 * the present cpus which have already reached the @state.
191 */
192static inline int cpuhp_setup_state(enum cpuhp_state state,
193 const char *name,
194 int (*startup)(unsigned int cpu),
195 int (*teardown)(unsigned int cpu))
196{
197 return __cpuhp_setup_state(state, name, true, startup, teardown, false);
198}
199
200static inline int cpuhp_setup_state_cpuslocked(enum cpuhp_state state,
201 const char *name,
202 int (*startup)(unsigned int cpu),
203 int (*teardown)(unsigned int cpu))
204{
205 return __cpuhp_setup_state_cpuslocked(state, name, true, startup,
206 teardown, false);
207}
208
209/**
210 * cpuhp_setup_state_nocalls - Setup hotplug state callbacks without calling the
211 * callbacks
212 * @state: The state for which the calls are installed
213 * @name: Name of the callback.
214 * @startup: startup callback function
215 * @teardown: teardown callback function
216 *
217 * Same as @cpuhp_setup_state except that no calls are executed are invoked
218 * during installation of this callback. NOP if SMP=n or HOTPLUG_CPU=n.
219 */
220static inline int cpuhp_setup_state_nocalls(enum cpuhp_state state,
221 const char *name,
222 int (*startup)(unsigned int cpu),
223 int (*teardown)(unsigned int cpu))
224{
225 return __cpuhp_setup_state(state, name, false, startup, teardown,
226 false);
227}
228
229static inline int cpuhp_setup_state_nocalls_cpuslocked(enum cpuhp_state state,
230 const char *name,
231 int (*startup)(unsigned int cpu),
232 int (*teardown)(unsigned int cpu))
233{
234 return __cpuhp_setup_state_cpuslocked(state, name, false, startup,
235 teardown, false);
236}

238/**
239 * cpuhp_setup_state_multi - Add callbacks for multi state
240 * @state: The state for which the calls are installed
241 * @name: Name of the callback.
242 * @startup: startup callback function
243 * @teardown: teardown callback function
244 *
245 * Sets the internal multi_instance flag and prepares a state to work as a multi
246 * instance callback. No callbacks are invoked at this point. The callbacks are
247 * invoked once an instance for this state are registered via
248 * @cpuhp_state_add_instance or @cpuhp_state_add_instance_nocalls.
249 */
250static inline int cpuhp_setup_state_multi(enum cpuhp_state state,
251 const char *name,
252 int (*startup)(unsigned int cpu,
253 struct hlist_node *node),
254 int (*teardown)(unsigned int cpu,
255 struct hlist_node *node))
256{
257 return __cpuhp_setup_state(state, name, false,
258 (void *) startup,
259 (void *) teardown, true);
260}

267/**
268 * cpuhp_state_add_instance - Add an instance for a state and invoke startup
269 * callback.
270 * @state: The state for which the instance is installed
271 * @node: The node for this individual state.
272 *
273 * Installs the instance for the @state and invokes the startup callback on
274 * the present cpus which have already reached the @state. The @state must have
275 * been earlier marked as multi-instance by @cpuhp_setup_state_multi.
276 */
277static inline int cpuhp_state_add_instance(enum cpuhp_state state,
278 struct hlist_node *node)
279{
280 return __cpuhp_state_add_instance(state, node, true);
281}
282
283/**
284 * cpuhp_state_add_instance_nocalls - Add an instance for a state without
285 * invoking the startup callback.
286 * @state: The state for which the instance is installed
287 * @node: The node for this individual state.
288 *
289 * Installs the instance for the @state The @state must have been earlier
290 * marked as multi-instance by @cpuhp_setup_state_multi.
291 */
292static inline int cpuhp_state_add_instance_nocalls(enum cpuhp_state state,
293 struct hlist_node *node)
294{
295 return __cpuhp_state_add_instance(state, node, false);
296}

335/**
336 * cpuhp_remove_multi_state - Remove hotplug multi state callback
337 * @state: The state for which the calls are removed
338 *
339 * Removes the callback functions from a multi state. This is the reverse of
340 * cpuhp_setup_state_multi(). All instances should have been removed before
341 * invoking this function.
