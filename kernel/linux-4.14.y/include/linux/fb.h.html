242/*
243 * Frame buffer operations
244 *
245 * LOCKING NOTE: those functions must _ALL_ be called with the console
246 * semaphore held, this is the only suitable locking mechanism we have
247 * in 2.6. Some may be called at interrupt time at this point though.
248 *
249 * The exception to this is the debug related hooks. Putting the fb
250 * into a debug state (e.g. flipping to the kernel console) and restoring
251 * it must be done in a lock-free manner, so low level drivers should
252 * keep track of the initial console (if applicable) and may need to
253 * perform direct, unlocked hardware writes in these hooks.
254 */

403/* FBINFO_* = fb_info.flags bit flags */
404#define FBINFO_DEFAULT 0
405#define FBINFO_HWACCEL_DISABLED 0x0002
406 /* When FBINFO_HWACCEL_DISABLED is set:
407 * Hardware acceleration is turned off. Software implementations
408 * of required functions (copyarea(), fillrect(), and imageblit())
409 * takes over; acceleration engine should be in a quiescent state */
410
411/* hints */
412#define FBINFO_VIRTFB 0x0004 /* FB is System RAM, not device. */
413#define FBINFO_PARTIAL_PAN_OK 0x0040 /* otw use pan only for double-buffering */
414#define FBINFO_READS_FAST 0x0080 /* soft-copy faster than rendering */
415
416/* hardware supported ops */
417/* semantics: when a bit is set, it indicates that the operation is
418 * accelerated by hardware.
419 * required functions will still work even if the bit is not set.
420 * optional functions may not even exist if the flag bit is not set.
421 */
422#define FBINFO_HWACCEL_NONE 0x0000
423#define FBINFO_HWACCEL_COPYAREA 0x0100 /* required */
424#define FBINFO_HWACCEL_FILLRECT 0x0200 /* required */
425#define FBINFO_HWACCEL_IMAGEBLIT 0x0400 /* required */
426#define FBINFO_HWACCEL_ROTATE 0x0800 /* optional */
427#define FBINFO_HWACCEL_XPAN 0x1000 /* optional */
428#define FBINFO_HWACCEL_YPAN 0x2000 /* optional */
429#define FBINFO_HWACCEL_YWRAP 0x4000 /* optional */
430
431#define FBINFO_MISC_USEREVENT 0x10000 /* event request
432 from userspace */
433#define FBINFO_MISC_TILEBLITTING 0x20000 /* use tile blitting */
434
435/* A driver may set this flag to indicate that it does want a set_par to be
436 * called every time when fbcon_switch is executed. The advantage is that with
437 * this flag set you can really be sure that set_par is always called before
438 * any of the functions dependent on the correct hardware state or altering
439 * that state, even if you are using some broken X releases. The disadvantage
440 * is that it introduces unwanted delays to every console switch if set_par
441 * is slow. It is a good idea to try this flag in the drivers initialization
442 * code whenever there is a bug report related to switching between X and the
443 * framebuffer console.
444 */
445#define FBINFO_MISC_ALWAYS_SETPAR 0x40000
446
447/* where the fb is a firmware driver, and can be replaced with a proper one */
448#define FBINFO_MISC_FIRMWARE 0x80000
449/*
450 * Host and GPU endianness differ.
451 */
452#define FBINFO_FOREIGN_ENDIAN 0x100000
453/*
454 * Big endian math. This is the same flags as above, but with different
455 * meaning, it is set by the fb subsystem depending FOREIGN_ENDIAN flag
456 * and host endianness. Drivers should not use this flag.
457 */
458#define FBINFO_BE_MATH 0x100000

653static inline void __fb_pad_aligned_buffer(u8 *dst, u32 d_pitch,
654 u8 *src, u32 s_pitch, u32 height)
655{
656 u32 i, j;
657
658 d_pitch -= s_pitch;
659
660 for (i = height; i--; ) {
661 /* s_pitch is a few bytes at the most, memcpy is suboptimal */
662 for (j = 0; j < s_pitch; j++)
663 *dst++ = *src++;
664 dst += d_pitch;
665 }
666}
