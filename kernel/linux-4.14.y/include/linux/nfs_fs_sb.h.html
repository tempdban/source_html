24/*
25 * The nfs_client identifies our client state to the server.
26 */
27struct nfs_client {
28 atomic_t cl_count;
29 atomic_t cl_mds_count;
30 int cl_cons_state; /* current construction state (-ve: init error) */
31#define NFS_CS_READY 0 /* ready to be used */
32#define NFS_CS_INITING 1 /* busy initialising */
33#define NFS_CS_SESSION_INITING 2 /* busy initialising session */
34 unsigned long cl_res_state; /* NFS resources state */
35#define NFS_CS_CALLBACK 1 /* - callback started */
36#define NFS_CS_IDMAP 2 /* - idmap started */
37#define NFS_CS_RENEWD 3 /* - renewd started */
38#define NFS_CS_STOP_RENEW 4 /* no more state to renew */
39#define NFS_CS_CHECK_LEASE_TIME 5 /* need to check lease time */
40 unsigned long cl_flags; /* behavior switches */
41#define NFS_CS_NORESVPORT 0 /* - use ephemeral src port */
42#define NFS_CS_DISCRTRY 1 /* - disconnect on RPC retry */
43#define NFS_CS_MIGRATION 2 /* - transparent state migr */
44#define NFS_CS_INFINITE_SLOTS 3 /* - don't limit TCP slots */
45#define NFS_CS_NO_RETRANS_TIMEOUT 4 /* - Disable retransmit timeouts */
46#define NFS_CS_TSM_POSSIBLE 5 /* - Maybe state migration */
47 struct sockaddr_storage cl_addr; /* server identifier */
48 size_t cl_addrlen;
49 char * cl_hostname; /* hostname of server */
50 char * cl_acceptor; /* GSSAPI acceptor name */
51 struct list_head cl_share_link; /* link in global client list */
52 struct list_head cl_superblocks; /* List of nfs_server structs */
53
54 struct rpc_clnt * cl_rpcclient;
55 const struct nfs_rpc_ops *rpc_ops; /* NFS protocol vector */
56 int cl_proto; /* Network transport protocol */
57 struct nfs_subversion * cl_nfs_mod; /* pointer to nfs version module */
58
59 u32 cl_minorversion;/* NFSv4 minorversion */
60 struct rpc_cred *cl_machine_cred;
61
62#if IS_ENABLED(CONFIG_NFS_V4)
63 struct list_head cl_ds_clients; /* auth flavor data servers */
64 u64 cl_clientid; /* constant */
65 nfs4_verifier cl_confirm; /* Clientid verifier */
66 unsigned long cl_state;
67
68 spinlock_t cl_lock;
69
70 unsigned long cl_lease_time;
71 unsigned long cl_last_renewal;
72 struct delayed_work cl_renewd;
73
74 struct rpc_wait_queue cl_rpcwaitq;
75
76 /* idmapper */
77 struct idmap * cl_idmap;
78
79 /* Client owner identifier */
80 const char * cl_owner_id;
81
82 u32 cl_cb_ident; /* v4.0 callback identifier */
83 const struct nfs4_minor_version_ops *cl_mvops;
84 unsigned long cl_mig_gen;
85
86 /* NFSv4.0 transport blocking */
87 struct nfs4_slot_table *cl_slot_tbl;
88
89 /* The sequence id to use for the next CREATE_SESSION */
90 u32 cl_seqid;
91 /* The flags used for obtaining the clientid during EXCHANGE_ID */
92 u32 cl_exchange_flags;
93 struct nfs4_session *cl_session; /* shared session */
94 bool cl_preserve_clid;
95 struct nfs41_server_owner *cl_serverowner;
96 struct nfs41_server_scope *cl_serverscope;
97 struct nfs41_impl_id *cl_implid;
98 /* nfs 4.1+ state protection modes: */
99 unsigned long cl_sp4_flags;
100#define NFS_SP4_MACH_CRED_MINIMAL 1 /* Minimal sp4_mach_cred - state ops
101 * must use machine cred */
102#define NFS_SP4_MACH_CRED_CLEANUP 2 /* CLOSE and LOCKU */
103#define NFS_SP4_MACH_CRED_SECINFO 3 /* SECINFO and SECINFO_NO_NAME */
104#define NFS_SP4_MACH_CRED_STATEID 4 /* TEST_STATEID and FREE_STATEID */
105#define NFS_SP4_MACH_CRED_WRITE 5 /* WRITE */
106#define NFS_SP4_MACH_CRED_COMMIT 6 /* COMMIT */
107#define NFS_SP4_MACH_CRED_PNFS_CLEANUP 7 /* LAYOUTRETURN */
108#if IS_ENABLED(CONFIG_NFS_V4_1)
109 wait_queue_head_t cl_lock_waitq;
110#endif /* CONFIG_NFS_V4_1 */
111#endif /* CONFIG_NFS_V4 */
112
113 /* Our own IP address, as a null-terminated string.
114 * This is used to generate the mv0 callback address.
115 */
116 char cl_ipaddr[48];
117
118#ifdef CONFIG_NFS_FSCACHE
119 struct fscache_cookie *fscache; /* client index cache cookie */
120#endif
121
122 struct net *cl_net;
123};
