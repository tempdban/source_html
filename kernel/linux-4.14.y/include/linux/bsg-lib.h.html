23#ifndef _BLK_BSG_
24#define _BLK_BSG_
25
26#include <linux/blkdev.h>
27#include <scsi/scsi_request.h>
28
29struct request;
30struct device;
31struct scatterlist;
32struct request_queue;
33
34struct bsg_buffer {
35 unsigned int payload_len;
36 int sg_cnt;
37 struct scatterlist *sg_list;
38};
39
40struct bsg_job {
41 struct scsi_request sreq;
42 struct device *dev;
43 struct request *req;
44
45 struct kref kref;
46
47 /* Transport/driver specific request/reply structs */
48 void *request;
49 void *reply;
50
51 unsigned int request_len;
52 unsigned int reply_len;
53 /*
54 * On entry : reply_len indicates the buffer size allocated for
55 * the reply.
56 *
57 * Upon completion : the message handler must set reply_len
58 * to indicates the size of the reply to be returned to the
59 * caller.
60 */
61
62 /* DMA payloads for the request/response */
63 struct bsg_buffer request_payload;
64 struct bsg_buffer reply_payload;
65
66 void *dd_data; /* Used for driver-specific storage */
67};
68
69void bsg_job_done(struct bsg_job *job, int result,
70 unsigned int reply_payload_rcv_len);
71struct request_queue *bsg_setup_queue(struct device *dev, const char *name,
72 bsg_job_fn *job_fn, int dd_job_size,
73 void (*release)(struct device *));
74void bsg_job_put(struct bsg_job *job);
75int __must_check bsg_job_get(struct bsg_job *job);
76
77#endif
