2/*
3 * Generic RTC interface.
4 * This version contains the part of the user interface to the Real Time Clock
5 * service. It is used with both the legacy mc146818 and also EFI
6 * Struct rtc_time and first 12 ioctl by Paul Gortmaker, 1996 - separated out
7 * from <linux/mc146818rtc.h> to this file for 2.4 kernels.
8 *
9 * Copyright (C) 1999 Hewlett-Packard Co.
10 * Copyright (C) 1999 Stephane Eranian <eranian@hpl.hp.com>
11 */
12#ifndef _UAPI_LINUX_RTC_H_
13#define _UAPI_LINUX_RTC_H_
14
15/*
16 * The struct used to pass data via the following ioctl. Similar to the
17 * struct tm in <time.h>, but it needs to be here so that the kernel
18 * source is self contained, allowing cross-compiles, etc. etc.
19 */
20
21struct rtc_time {
22 int tm_sec;
23 int tm_min;
24 int tm_hour;
25 int tm_mday;
26 int tm_mon;
27 int tm_year;
28 int tm_wday;
29 int tm_yday;
30 int tm_isdst;
31};
32
33/*
34 * This data structure is inspired by the EFI (v0.92) wakeup
35 * alarm API.
36 */
37struct rtc_wkalrm {
38 unsigned char enabled; /* 0 = alarm disabled, 1 = alarm enabled */
39 unsigned char pending; /* 0 = alarm not pending, 1 = alarm pending */
40 struct rtc_time time; /* time the alarm is set to */
41};
42
43/*
44 * Data structure to control PLL correction some better RTC feature
45 * pll_value is used to get or set current value of correction,
46 * the rest of the struct is used to query HW capabilities.
47 * This is modeled after the RTC used in Q40/Q60 computers but
48 * should be sufficiently flexible for other devices
49 *
50 * +ve pll_value means clock will run faster by
51 * pll_value*pll_posmult/pll_clock
52 * -ve pll_value means clock will run slower by
53 * pll_value*pll_negmult/pll_clock
54 */
55
56struct rtc_pll_info {
57 int pll_ctrl; /* placeholder for fancier control */
58 int pll_value; /* get/set correction value */
59 int pll_max; /* max +ve (faster) adjustment value */
60 int pll_min; /* max -ve (slower) adjustment value */
61 int pll_posmult; /* factor for +ve correction */
62 int pll_negmult; /* factor for -ve correction */
63 long pll_clock; /* base PLL frequency */
64};
65
66/*
67 * ioctl calls that are permitted to the /dev/rtc interface, if
68 * any of the RTC drivers are enabled.
69 */
70
71#define RTC_AIE_ON _IO('p', 0x01) /* Alarm int. enable on */
72#define RTC_AIE_OFF _IO('p', 0x02) /* ... off */
73#define RTC_UIE_ON _IO('p', 0x03) /* Update int. enable on */
74#define RTC_UIE_OFF _IO('p', 0x04) /* ... off */
75#define RTC_PIE_ON _IO('p', 0x05) /* Periodic int. enable on */
76#define RTC_PIE_OFF _IO('p', 0x06) /* ... off */
77#define RTC_WIE_ON _IO('p', 0x0f) /* Watchdog int. enable on */
78#define RTC_WIE_OFF _IO('p', 0x10) /* ... off */
79
80#define RTC_ALM_SET _IOW('p', 0x07, struct rtc_time) /* Set alarm time */
81#define RTC_ALM_READ _IOR('p', 0x08, struct rtc_time) /* Read alarm time */
82#define RTC_RD_TIME _IOR('p', 0x09, struct rtc_time) /* Read RTC time */
83#define RTC_SET_TIME _IOW('p', 0x0a, struct rtc_time) /* Set RTC time */
84#define RTC_IRQP_READ _IOR('p', 0x0b, unsigned long) /* Read IRQ rate */
85#define RTC_IRQP_SET _IOW('p', 0x0c, unsigned long) /* Set IRQ rate */
86#define RTC_EPOCH_READ _IOR('p', 0x0d, unsigned long) /* Read epoch */
87#define RTC_EPOCH_SET _IOW('p', 0x0e, unsigned long) /* Set epoch */
88
89#define RTC_WKALM_SET _IOW('p', 0x0f, struct rtc_wkalrm)/* Set wakeup alarm*/
90#define RTC_WKALM_RD _IOR('p', 0x10, struct rtc_wkalrm)/* Get wakeup alarm*/
91
92#define RTC_PLL_GET _IOR('p', 0x11, struct rtc_pll_info) /* Get PLL correction */
93#define RTC_PLL_SET _IOW('p', 0x12, struct rtc_pll_info) /* Set PLL correction */
94
95#define RTC_VL_READ _IOR('p', 0x13, int) /* Voltage low detector */
96#define RTC_VL_CLR _IO('p', 0x14) /* Clear voltage low information */
97
98/* interrupt flags */
99#define RTC_IRQF 0x80 /* Any of the following is active */
100#define RTC_PF 0x40 /* Periodic interrupt */
101#define RTC_AF 0x20 /* Alarm interrupt */
102#define RTC_UF 0x10 /* Update interrupt for 1Hz RTC */
103
104
105#define RTC_MAX_FREQ 8192
106

108#endif /* _UAPI_LINUX_RTC_H_ */
