12#ifndef _LINUX_FIEMAP_H
13#define _LINUX_FIEMAP_H
14
15#include <linux/types.h>
16
17struct fiemap_extent {
18 __u64 fe_logical; /* logical offset in bytes for the start of
19 * the extent from the beginning of the file */
20 __u64 fe_physical; /* physical offset in bytes for the start
21 * of the extent from the beginning of the disk */
22 __u64 fe_length; /* length in bytes for this extent */
23 __u64 fe_reserved64[2];
24 __u32 fe_flags; /* FIEMAP_EXTENT_* flags for this extent */
25 __u32 fe_reserved[3];
26};
27
28struct fiemap {
29 __u64 fm_start; /* logical offset (inclusive) at
30 * which to start mapping (in) */
31 __u64 fm_length; /* logical length of mapping which
32 * userspace wants (in) */
33 __u32 fm_flags; /* FIEMAP_FLAG_* flags for request (in/out) */
34 __u32 fm_mapped_extents;/* number of extents that were mapped (out) */
35 __u32 fm_extent_count; /* size of fm_extents array (in) */
36 __u32 fm_reserved;
37 struct fiemap_extent fm_extents[0]; /* array of mapped extents (out) */
38};
39
40#define FIEMAP_MAX_OFFSET (~0ULL)
41
42#define FIEMAP_FLAG_SYNC 0x00000001 /* sync file data before map */
43#define FIEMAP_FLAG_XATTR 0x00000002 /* map extended attribute tree */
44#define FIEMAP_FLAG_CACHE 0x00000004 /* request caching of the extents */
45
46#define FIEMAP_FLAGS_COMPAT (FIEMAP_FLAG_SYNC | FIEMAP_FLAG_XATTR)
47
48#define FIEMAP_EXTENT_LAST 0x00000001 /* Last extent in file. */
49#define FIEMAP_EXTENT_UNKNOWN 0x00000002 /* Data location unknown. */
50#define FIEMAP_EXTENT_DELALLOC 0x00000004 /* Location still pending.
51 * Sets EXTENT_UNKNOWN. */
52#define FIEMAP_EXTENT_ENCODED 0x00000008 /* Data can not be read
53 * while fs is unmounted */
54#define FIEMAP_EXTENT_DATA_ENCRYPTED 0x00000080 /* Data is encrypted by fs.
55 * Sets EXTENT_NO_BYPASS. */
56#define FIEMAP_EXTENT_NOT_ALIGNED 0x00000100 /* Extent offsets may not be
57 * block aligned. */
58#define FIEMAP_EXTENT_DATA_INLINE 0x00000200 /* Data mixed with metadata.
59 * Sets EXTENT_NOT_ALIGNED.*/
60#define FIEMAP_EXTENT_DATA_TAIL 0x00000400 /* Multiple files in block.
61 * Sets EXTENT_NOT_ALIGNED.*/
62#define FIEMAP_EXTENT_UNWRITTEN 0x00000800 /* Space allocated, but
63 * no data (i.e. zero). */
64#define FIEMAP_EXTENT_MERGED 0x00001000 /* File does not natively
65 * support extents. Result
66 * merged for efficiency. */
67#define FIEMAP_EXTENT_SHARED 0x00002000 /* Space shared with other
68 * files. */
69
70#endif /* _LINUX_FIEMAP_H */
