18#ifndef _UAPI_LINUX_UDP_H
19#define _UAPI_LINUX_UDP_H
20
21#include <linux/types.h>
22
23struct udphdr {
24 __be16 source;
25 __be16 dest;
26 __be16 len;
27 __sum16 check;
28};
29
30/* UDP socket options */
31#define UDP_CORK 1 /* Never send partially complete segments */
32#define UDP_ENCAP 100 /* Set the socket to accept encapsulated packets */
33#define UDP_NO_CHECK6_TX 101 /* Disable sending checksum for UDP6X */
34#define UDP_NO_CHECK6_RX 102 /* Disable accpeting checksum for UDP6 */
35
36/* UDP encapsulation types */
37#define UDP_ENCAP_ESPINUDP_NON_IKE 1 /* draft-ietf-ipsec-nat-t-ike-00/01 */
38#define UDP_ENCAP_ESPINUDP 2 /* draft-ietf-ipsec-udp-encaps-06 */
39#define UDP_ENCAP_L2TPINUDP 3 /* rfc2661 */
40#define UDP_ENCAP_GTP0 4 /* GSM TS 09.60 */
41#define UDP_ENCAP_GTP1U 5 /* 3GPP TS 29.060 */
42
43#endif /* _UAPI_LINUX_UDP_H */
