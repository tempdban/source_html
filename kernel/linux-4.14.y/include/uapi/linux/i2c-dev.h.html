24#ifndef _UAPI_LINUX_I2C_DEV_H
25#define _UAPI_LINUX_I2C_DEV_H
26
27#include <linux/types.h>
28#include <linux/compiler.h>
29
30/* /dev/i2c-X ioctl commands. The ioctl's parameter is always an
31 * unsigned long, except for:
32 * - I2C_FUNCS, takes pointer to an unsigned long
33 * - I2C_RDWR, takes pointer to struct i2c_rdwr_ioctl_data
34 * - I2C_SMBUS, takes pointer to struct i2c_smbus_ioctl_data
35 */
36#define I2C_RETRIES 0x0701 /* number of times a device address should
37 be polled when not acknowledging */
38#define I2C_TIMEOUT 0x0702 /* set timeout in units of 10 ms */
39
40/* NOTE: Slave address is 7 or 10 bits, but 10-bit addresses
41 * are NOT supported! (due to code brokenness)
42 */
43#define I2C_SLAVE 0x0703 /* Use this slave address */
44#define I2C_SLAVE_FORCE 0x0706 /* Use this slave address, even if it
45 is already in use by a driver! */
46#define I2C_TENBIT 0x0704 /* 0 for 7 bit addrs, != 0 for 10 bit */
47
48#define I2C_FUNCS 0x0705 /* Get the adapter functionality mask */
49
50#define I2C_RDWR 0x0707 /* Combined R/W transfer (one STOP only) */
51
52#define I2C_PEC 0x0708 /* != 0 to use PEC with SMBus */
53#define I2C_SMBUS 0x0720 /* SMBus transfer */
54
55
56/* This is the structure as used in the I2C_SMBUS ioctl call */
57struct i2c_smbus_ioctl_data {
58 __u8 read_write;
59 __u8 command;
60 __u32 size;
61 union i2c_smbus_data __user *data;
62};
63
64/* This is the structure as used in the I2C_RDWR ioctl call */
65struct i2c_rdwr_ioctl_data {
66 struct i2c_msg __user *msgs; /* pointers to i2c_msgs */
67 __u32 nmsgs; /* number of i2c_msgs */
68};
69
70#define I2C_RDWR_IOCTL_MAX_MSGS 42
71/* Originally defined with a typo, keep it for compatibility */
72#define I2C_RDRW_IOCTL_MAX_MSGS I2C_RDWR_IOCTL_MAX_MSGS
73
74
75#endif /* _UAPI_LINUX_I2C_DEV_H */
