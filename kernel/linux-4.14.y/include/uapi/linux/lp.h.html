10
11
12/*
13 * Per POSIX guidelines, this module reserves the LP and lp prefixes
14 * These are the lp_table[minor].flags flags...
15 */
16#define LP_EXIST 0x0001
17#define LP_SELEC 0x0002
18#define LP_BUSY 0x0004
19#define LP_BUSY_BIT_POS 2
20#define LP_OFFL 0x0008
21#define LP_NOPA 0x0010
22#define LP_ERR 0x0020
23#define LP_ABORT 0x0040
24#define LP_CAREFUL 0x0080 /* obsoleted -arca */
25#define LP_ABORTOPEN 0x0100
26
27#define LP_TRUST_IRQ_ 0x0200 /* obsolete */
28#define LP_NO_REVERSE 0x0400 /* No reverse mode available. */
29#define LP_DATA_AVAIL 0x0800 /* Data is available. */
30
31/*
32 * bit defines for 8255 status port
33 * base + 1
34 * accessed with LP_S(minor), which gets the byte...
35 */
36#define LP_PBUSY 0x80 /* inverted input, active high */
37#define LP_PACK 0x40 /* unchanged input, active low */
38#define LP_POUTPA 0x20 /* unchanged input, active high */
39#define LP_PSELECD 0x10 /* unchanged input, active high */
40#define LP_PERRORP 0x08 /* unchanged input, active low */
41
42/* timeout for each character. This is relative to bus cycles -- it
43 * is the count in a busy loop. THIS IS THE VALUE TO CHANGE if you
44 * have extremely slow printing, or if the machine seems to slow down
45 * a lot when you print. If you have slow printing, increase this
46 * number and recompile, and if your system gets bogged down, decrease
47 * this number. This can be changed with the tunelp(8) command as well.
48 */
49
50#define LP_INIT_CHAR 1000
51
52/* The parallel port specs apparently say that there needs to be
53 * a .5usec wait before and after the strobe.
54 */
55
56#define LP_INIT_WAIT 1
57
58/* This is the amount of time that the driver waits for the printer to
59 * catch up when the printer's buffer appears to be filled. If you
60 * want to tune this and have a fast printer (i.e. HPIIIP), decrease
61 * this number, and if you have a slow printer, increase this number.
62 * This is in hundredths of a second, the default 2 being .05 second.
63 * Or use the tunelp(8) command, which is especially nice if you want
64 * change back and forth between character and graphics printing, which
65 * are wildly different...
66 */
67
68#define LP_INIT_TIME 2
69
70/* IOCTL numbers */
71#define LPCHAR 0x0601 /* corresponds to LP_INIT_CHAR */
72#define LPTIME 0x0602 /* corresponds to LP_INIT_TIME */
73#define LPABORT 0x0604 /* call with TRUE arg to abort on error,
74 FALSE to retry. Default is retry. */
75#define LPSETIRQ 0x0605 /* call with new IRQ number,
76 or 0 for polling (no IRQ) */
77#define LPGETIRQ 0x0606 /* get the current IRQ number */
78#define LPWAIT 0x0608 /* corresponds to LP_INIT_WAIT */
79/* NOTE: LPCAREFUL is obsoleted and it' s always the default right now -arca */
80#define LPCAREFUL 0x0609 /* call with TRUE arg to require out-of-paper, off-
81 line, and error indicators good on all writes,
82 FALSE to ignore them. Default is ignore. */
83#define LPABORTOPEN 0x060a /* call with TRUE arg to abort open() on error,
84 FALSE to ignore error. Default is ignore. */
85#define LPGETSTATUS 0x060b /* return LP_S(minor) */
86#define LPRESET 0x060c /* reset printer */
87#ifdef LP_STATS
88#define LPGETSTATS 0x060d /* get statistics (struct lp_stats) */
89#endif
90#define LPGETFLAGS 0x060e /* get status flags */
91#define LPSETTIMEOUT 0x060f /* set parport timeout */
92
93/* timeout for printk'ing a timeout, in jiffies (100ths of a second).
94 This is also used for re-checking error conditions if LP_ABORT is
95 not set. This is the default behavior. */
96
97#define LP_TIMEOUT_INTERRUPT (60 * HZ)
98#define LP_TIMEOUT_POLLED (10 * HZ)
99
100
101#endif /* _UAPI_LINUX_LP_H */
