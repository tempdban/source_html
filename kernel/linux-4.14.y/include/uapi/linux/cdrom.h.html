18/*******************************************************
19 * As of Linux 2.1.x, all Linux CD-ROM application programs will use this
20 * (and only this) include file. It is my hope to provide Linux with
21 * a uniform interface between software accessing CD-ROMs and the various
22 * device drivers that actually talk to the drives. There may still be
23 * 23 different kinds of strange CD-ROM drives, but at least there will
24 * now be one, and only one, Linux CD-ROM interface.
25 *
26 * Additionally, as of Linux 2.1.x, all Linux application programs
27 * should use the O_NONBLOCK option when opening a CD-ROM device
28 * for subsequent ioctl commands. This allows for neat system errors
29 * like "No medium found" or "Wrong medium type" upon attempting to
30 * mount or play an empty slot, mount an audio disc, or play a data disc.
31 * Generally, changing an application program to support O_NONBLOCK
32 * is as easy as the following:
33 * - drive = open("/dev/cdrom", O_RDONLY);
34 * + drive = open("/dev/cdrom", O_RDONLY | O_NONBLOCK);
35 * It is worth the small change.
36 *
37 * Patches for many common CD programs (provided by David A. van Leeuwen)
38 * can be found at: ftp://ftp.gwdg.de/pub/linux/cdrom/drivers/cm206/
39 *
40 *******************************************************/
41

44 * borrow the "Operation not supported" error from the network folks to

295/*
296 * A CD-ROM physical sector size is 2048, 2052, 2056, 2324, 2332, 2336,
297 * 2340, or 2352 bytes long.
298
299* Sector types of the standard CD-ROM data formats:
300 *
301 * format sector type user data size (bytes)
302 * -----------------------------------------------------------------------------
303 * 1 (Red Book) CD-DA 2352 (CD_FRAMESIZE_RAW)
304 * 2 (Yellow Book) Mode1 Form1 2048 (CD_FRAMESIZE)
305 * 3 (Yellow Book) Mode1 Form2 2336 (CD_FRAMESIZE_RAW0)
306 * 4 (Green Book) Mode2 Form1 2048 (CD_FRAMESIZE)
307 * 5 (Green Book) Mode2 Form2 2328 (2324+4 spare bytes)
308 *
309 *
310 * The layout of the standard CD-ROM data formats:
311 * -----------------------------------------------------------------------------
312 * - audio (red): | audio_sample_bytes |
313 * | 2352 |
314 *
315 * - data (yellow, mode1): | sync - head - data - EDC - zero - ECC |
316 * | 12 - 4 - 2048 - 4 - 8 - 276 |
317 *
318 * - data (yellow, mode2): | sync - head - data |
319 * | 12 - 4 - 2336 |
320 *
321 * - XA data (green, mode2 form1): | sync - head - sub - data - EDC - ECC |
322 * | 12 - 4 - 8 - 2048 - 4 - 276 |
323 *
324 * - XA data (green, mode2 form2): | sync - head - sub - data - Spare |
325 * | 12 - 4 - 8 - 2324 - 4 |
326 *
327 */
328
329/* Some generally useful CD-ROM information -- mostly based on the above */
330#define CD_MINS 74 /* max. minutes per CD, not really a limit */
331#define CD_SECS 60 /* seconds per minute */
332#define CD_FRAMES 75 /* frames per second */
333#define CD_SYNC_SIZE 12 /* 12 sync bytes per raw data frame */
334#define CD_MSF_OFFSET 150 /* MSF numbering offset of first frame */
335#define CD_CHUNK_SIZE 24 /* lowest-level "data bytes piece" */
336#define CD_NUM_OF_CHUNKS 98 /* chunks per frame */
337#define CD_FRAMESIZE_SUB 96 /* subchannel data "frame" size */
338#define CD_HEAD_SIZE 4 /* header (address) bytes per raw data frame */
339#define CD_SUBHEAD_SIZE 8 /* subheader bytes per raw XA data frame */
340#define CD_EDC_SIZE 4 /* bytes EDC per most raw data frame types */
341#define CD_ZERO_SIZE 8 /* bytes zero per yellow book mode 1 frame */
342#define CD_ECC_SIZE 276 /* bytes ECC per most raw data frame types */
343#define CD_FRAMESIZE 2048 /* bytes per frame, "cooked" mode */
344#define CD_FRAMESIZE_RAW 2352 /* bytes per frame, "raw" mode */
345#define CD_FRAMESIZE_RAWER 2646 /* The maximum possible returned bytes */
346/* most drives don't deliver everything: */
347#define CD_FRAMESIZE_RAW1 (CD_FRAMESIZE_RAW-CD_SYNC_SIZE) /*2340*/
348#define CD_FRAMESIZE_RAW0 (CD_FRAMESIZE_RAW-CD_SYNC_SIZE-CD_HEAD_SIZE) /*2336*/
349
350#define CD_XA_HEAD (CD_HEAD_SIZE+CD_SUBHEAD_SIZE) /* "before data" part of raw XA frame */
351#define CD_XA_TAIL (CD_EDC_SIZE+CD_ECC_SIZE) /* "after data" part of raw XA frame */
352#define CD_XA_SYNC_HEAD (CD_SYNC_SIZE+CD_XA_HEAD) /* sync bytes + header of XA frame */
