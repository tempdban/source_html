18#ifndef _UAPI_LINUX_UUID_H_
19#define _UAPI_LINUX_UUID_H_
20
21#include <linux/types.h>
22#include <linux/string.h>
23
24typedef struct {
25 __u8 b[16];
26} guid_t;
27
28#define GUID_INIT(a, b, c, d0, d1, d2, d3, d4, d5, d6, d7) \
29((guid_t) \
30{{ (a) & 0xff, ((a) >> 8) & 0xff, ((a) >> 16) & 0xff, ((a) >> 24) & 0xff, \
31 (b) & 0xff, ((b) >> 8) & 0xff, \
32 (c) & 0xff, ((c) >> 8) & 0xff, \
33 (d0), (d1), (d2), (d3), (d4), (d5), (d6), (d7) }})
34
35/* backwards compatibility, don't use in new code */
36typedef guid_t uuid_le;
37#define UUID_LE(a, b, c, d0, d1, d2, d3, d4, d5, d6, d7) \
38 GUID_INIT(a, b, c, d0, d1, d2, d3, d4, d5, d6, d7)
39#define NULL_UUID_LE \
40 UUID_LE(0x00000000, 0x0000, 0x0000, 0x00, 0x00, 0x00, 0x00, \
41 0x00, 0x00, 0x00, 0x00)
42
43#endif /* _UAPI_LINUX_UUID_H_ */
