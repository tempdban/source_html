18#ifndef __UAPI_POSIX_ACL_H
19#define __UAPI_POSIX_ACL_H
20
21#define ACL_UNDEFINED_ID (-1)
22
23/* a_type field in acl_user_posix_entry_t */
24#define ACL_TYPE_ACCESS (0x8000)
25#define ACL_TYPE_DEFAULT (0x4000)
26
27/* e_tag entry in struct posix_acl_entry */
28#define ACL_USER_OBJ (0x01)
29#define ACL_USER (0x02)
30#define ACL_GROUP_OBJ (0x04)
31#define ACL_GROUP (0x08)
32#define ACL_MASK (0x10)
33#define ACL_OTHER (0x20)
34
35/* permissions in the e_perm field */
36#define ACL_READ (0x04)
37#define ACL_WRITE (0x02)
38#define ACL_EXECUTE (0x01)
39
40#endif /* __UAPI_POSIX_ACL_H */
