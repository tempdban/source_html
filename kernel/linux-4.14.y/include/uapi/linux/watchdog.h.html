10#ifndef _UAPI_LINUX_WATCHDOG_H
11#define _UAPI_LINUX_WATCHDOG_H
12
13#include <linux/ioctl.h>
14#include <linux/types.h>
15
16#define WATCHDOG_IOCTL_BASE 'W'
17
18struct watchdog_info {
19 __u32 options; /* Options the card/driver supports */
20 __u32 firmware_version; /* Firmware version of the card */
21 __u8 identity[32]; /* Identity of the board */
22};
23
24#define WDIOC_GETSUPPORT _IOR(WATCHDOG_IOCTL_BASE, 0, struct watchdog_info)
25#define WDIOC_GETSTATUS _IOR(WATCHDOG_IOCTL_BASE, 1, int)
26#define WDIOC_GETBOOTSTATUS _IOR(WATCHDOG_IOCTL_BASE, 2, int)
27#define WDIOC_GETTEMP _IOR(WATCHDOG_IOCTL_BASE, 3, int)
28#define WDIOC_SETOPTIONS _IOR(WATCHDOG_IOCTL_BASE, 4, int)
29#define WDIOC_KEEPALIVE _IOR(WATCHDOG_IOCTL_BASE, 5, int)
30#define WDIOC_SETTIMEOUT _IOWR(WATCHDOG_IOCTL_BASE, 6, int)
31#define WDIOC_GETTIMEOUT _IOR(WATCHDOG_IOCTL_BASE, 7, int)
32#define WDIOC_SETPRETIMEOUT _IOWR(WATCHDOG_IOCTL_BASE, 8, int)
33#define WDIOC_GETPRETIMEOUT _IOR(WATCHDOG_IOCTL_BASE, 9, int)
34#define WDIOC_GETTIMELEFT _IOR(WATCHDOG_IOCTL_BASE, 10, int)
35
36#define WDIOF_UNKNOWN -1 /* Unknown flag error */
37#define WDIOS_UNKNOWN -1 /* Unknown status error */
38
39#define WDIOF_OVERHEAT 0x0001 /* Reset due to CPU overheat */
40#define WDIOF_FANFAULT 0x0002 /* Fan failed */
41#define WDIOF_EXTERN1 0x0004 /* External relay 1 */
42#define WDIOF_EXTERN2 0x0008 /* External relay 2 */
43#define WDIOF_POWERUNDER 0x0010 /* Power bad/power fault */
44#define WDIOF_CARDRESET 0x0020 /* Card previously reset the CPU */
45#define WDIOF_POWEROVER 0x0040 /* Power over voltage */
46#define WDIOF_SETTIMEOUT 0x0080 /* Set timeout (in seconds) */
47#define WDIOF_MAGICCLOSE 0x0100 /* Supports magic close char */
48#define WDIOF_PRETIMEOUT 0x0200 /* Pretimeout (in seconds), get/set */
49#define WDIOF_ALARMONLY 0x0400 /* Watchdog triggers a management or
50 other external alarm not a reboot */
51#define WDIOF_KEEPALIVEPING 0x8000 /* Keep alive ping reply */
52
53#define WDIOS_DISABLECARD 0x0001 /* Turn off the watchdog timer */
54#define WDIOS_ENABLECARD 0x0002 /* Turn on the watchdog timer */
55#define WDIOS_TEMPPANIC 0x0004 /* Kernel panic on temperature trip */
56
57
58#endif /* _UAPI_LINUX_WATCHDOG_H */
