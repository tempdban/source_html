33/* Feature bits */
34#define VIRTIO_BLK_F_SIZE_MAX 1 /* Indicates maximum segment size */
35#define VIRTIO_BLK_F_SEG_MAX 2 /* Indicates maximum # of segments */
36#define VIRTIO_BLK_F_GEOMETRY 4 /* Legacy geometry available */
37#define VIRTIO_BLK_F_RO 5 /* Disk is read-only */
38#define VIRTIO_BLK_F_BLK_SIZE 6 /* Block size of disk is available*/
39#define VIRTIO_BLK_F_TOPOLOGY 10 /* Topology information is available */
40#define VIRTIO_BLK_F_MQ 12 /* support more than one vq */
41
42/* Legacy feature bits */
43#ifndef VIRTIO_BLK_NO_LEGACY
44#define VIRTIO_BLK_F_BARRIER 0 /* Does host support barriers? */
45#define VIRTIO_BLK_F_SCSI 7 /* Supports scsi command passthru */
46#define VIRTIO_BLK_F_FLUSH 9 /* Flush command supported */
47#define VIRTIO_BLK_F_CONFIG_WCE 11 /* Writeback mode available in config */
48#ifndef __KERNEL__
49/* Old (deprecated) name for VIRTIO_BLK_F_FLUSH. */
50#define VIRTIO_BLK_F_WCE VIRTIO_BLK_F_FLUSH
51#endif
52#endif /* !VIRTIO_BLK_NO_LEGACY */
53
54#define VIRTIO_BLK_ID_BYTES 20 /* ID string length */
55
56struct virtio_blk_config {
57 /* The capacity (in 512-byte sectors). */
58 __u64 capacity;
59 /* The maximum segment size (if VIRTIO_BLK_F_SIZE_MAX) */
60 __u32 size_max;
61 /* The maximum number of segments (if VIRTIO_BLK_F_SEG_MAX) */
62 __u32 seg_max;
63 /* geometry of the device (if VIRTIO_BLK_F_GEOMETRY) */
64 struct virtio_blk_geometry {
65 __u16 cylinders;
66 __u8 heads;
67 __u8 sectors;
68 } geometry;
69
70 /* block size of device (if VIRTIO_BLK_F_BLK_SIZE) */
71 __u32 blk_size;
72
73 /* the next 4 entries are guarded by VIRTIO_BLK_F_TOPOLOGY */
74 /* exponent for physical block per logical block. */
75 __u8 physical_block_exp;
76 /* alignment offset in logical blocks. */
77 __u8 alignment_offset;
78 /* minimum I/O size without performance penalty in logical blocks. */
79 __u16 min_io_size;
80 /* optimal sustained I/O size in logical blocks. */
81 __u32 opt_io_size;
82
83 /* writeback mode (if VIRTIO_BLK_F_CONFIG_WCE) */
84 __u8 wce;
85 __u8 unused;
86
87 /* number of vqs, only available when VIRTIO_BLK_F_MQ is set */
88 __u16 num_queues;
89} __attribute__((packed));

91/*
92 * Command types
93 *
94 * Usage is a bit tricky as some bits are used as flags and some are not.
95 *
96 * Rules:
97 * VIRTIO_BLK_T_OUT may be combined with VIRTIO_BLK_T_SCSI_CMD or
98 * VIRTIO_BLK_T_BARRIER. VIRTIO_BLK_T_FLUSH is a command of its own
99 * and may not be combined with any of the other flags.
100 */
101
102/* These two define direction. */
103#define VIRTIO_BLK_T_IN 0
104#define VIRTIO_BLK_T_OUT 1
105
106#ifndef VIRTIO_BLK_NO_LEGACY
107/* This bit says it's a scsi command, not an actual read or write. */
108#define VIRTIO_BLK_T_SCSI_CMD 2
109#endif /* VIRTIO_BLK_NO_LEGACY */
110
111/* Cache flush command */
112#define VIRTIO_BLK_T_FLUSH 4
113
114/* Get device ID command */
115#define VIRTIO_BLK_T_GET_ID 8
116
117#ifndef VIRTIO_BLK_NO_LEGACY
118/* Barrier before this op. */
119#define VIRTIO_BLK_T_BARRIER 0x80000000
120#endif /* !VIRTIO_BLK_NO_LEGACY */
121
122/*
123 * This comes first in the read scatter-gather list.
124 * For legacy virtio, if VIRTIO_F_ANY_LAYOUT is not negotiated,
125 * this is the first element of the read scatter-gather list.
126 */
127struct virtio_blk_outhdr {
128 /* VIRTIO_BLK_T* */
129 __virtio32 type;
130 /* io priority. */
131 __virtio32 ioprio;
132 /* Sector (ie. 512 byte offset) */
133 __virtio64 sector;
134};
135
136#ifndef VIRTIO_BLK_NO_LEGACY
137struct virtio_scsi_inhdr {
138 __virtio32 errors;
139 __virtio32 data_len;
140 __virtio32 sense_len;
141 __virtio32 residual;
142};
143#endif /* !VIRTIO_BLK_NO_LEGACY */
144
145/* And this is the final byte of the write scatter-gather list. */
146#define VIRTIO_BLK_S_OK 0
147#define VIRTIO_BLK_S_IOERR 1
148#define VIRTIO_BLK_S_UNSUPP 2
149#endif /* _LINUX_VIRTIO_BLK_H */
