26#ifndef _UAPI__HID_H
27#define _UAPI__HID_H
28
29
30
31/*
32 * USB HID (Human Interface Device) interface class code
33 */
34
35#define USB_INTERFACE_CLASS_HID 3
36
37/*
38 * USB HID interface subclass and protocol codes
39 */
40
41#define USB_INTERFACE_SUBCLASS_BOOT 1
42#define USB_INTERFACE_PROTOCOL_KEYBOARD 1
43#define USB_INTERFACE_PROTOCOL_MOUSE 2
44
45/*
46 * HID class requests
47 */
48
49#define HID_REQ_GET_REPORT 0x01
50#define HID_REQ_GET_IDLE 0x02
51#define HID_REQ_GET_PROTOCOL 0x03
52#define HID_REQ_SET_REPORT 0x09
53#define HID_REQ_SET_IDLE 0x0A
54#define HID_REQ_SET_PROTOCOL 0x0B
55
56/*
57 * HID class descriptor types
58 */
59
60#define HID_DT_HID (USB_TYPE_CLASS | 0x01)
61#define HID_DT_REPORT (USB_TYPE_CLASS | 0x02)
62#define HID_DT_PHYSICAL (USB_TYPE_CLASS | 0x03)
63
64#define HID_MAX_DESCRIPTOR_SIZE 4096
65

67#endif /* _UAPI__HID_H */
