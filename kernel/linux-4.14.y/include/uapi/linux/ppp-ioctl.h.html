18/*
19 * Bit definitions for flags argument to PPPIOCGFLAGS/PPPIOCSFLAGS.
20 */
21#define SC_COMP_PROT 0x00000001 /* protocol compression (output) */
22#define SC_COMP_AC 0x00000002 /* header compression (output) */
23#define SC_COMP_TCP 0x00000004 /* TCP (VJ) compression (output) */
24#define SC_NO_TCP_CCID 0x00000008 /* disable VJ connection-id comp. */
25#define SC_REJ_COMP_AC 0x00000010 /* reject adrs/ctrl comp. on input */
26#define SC_REJ_COMP_TCP 0x00000020 /* reject TCP (VJ) comp. on input */
27#define SC_CCP_OPEN 0x00000040 /* Look at CCP packets */
28#define SC_CCP_UP 0x00000080 /* May send/recv compressed packets */
29#define SC_ENABLE_IP 0x00000100 /* IP packets may be exchanged */
30#define SC_LOOP_TRAFFIC 0x00000200 /* send traffic to pppd */
31#define SC_MULTILINK 0x00000400 /* do multilink encapsulation */
32#define SC_MP_SHORTSEQ 0x00000800 /* use short MP sequence numbers */
33#define SC_COMP_RUN 0x00001000 /* compressor has been inited */
34#define SC_DECOMP_RUN 0x00002000 /* decompressor has been inited */
35#define SC_MP_XSHORTSEQ 0x00004000 /* transmit short MP seq numbers */
36#define SC_DEBUG 0x00010000 /* enable debug messages */
37#define SC_LOG_INPKT 0x00020000 /* log contents of good pkts recvd */
38#define SC_LOG_OUTPKT 0x00040000 /* log contents of pkts sent */
39#define SC_LOG_RAWIN 0x00080000 /* log all chars received */
40#define SC_LOG_FLUSH 0x00100000 /* log all chars flushed */
41#define SC_SYNC 0x00200000 /* synchronous serial mode */
42#define SC_MUST_COMP 0x00400000 /* no uncompressed packets may be sent or received */
43#define SC_MASK 0x0f600fff /* bits that user can change */
44
45/* state bits */
46#define SC_XMIT_BUSY 0x10000000 /* (used by isdn_ppp?) */
47#define SC_RCV_ODDP 0x08000000 /* have rcvd char with odd parity */
48#define SC_RCV_EVNP 0x04000000 /* have rcvd char with even parity */
49#define SC_RCV_B7_1 0x02000000 /* have rcvd char with bit 7 = 1 */
50#define SC_RCV_B7_0 0x01000000 /* have rcvd char with bit 7 = 0 */
51#define SC_DC_FERROR 0x00800000 /* fatal decomp error detected */
52#define SC_DC_ERROR 0x00400000 /* non-fatal decomp error detected */
53
54/* Used with PPPIOCGNPMODE/PPPIOCSNPMODE */
55struct npioctl {
56 int protocol; /* PPP protocol, e.g. PPP_IP */
57 enum NPmode mode;
58};
59
60/* Structure describing a CCP configuration option, for PPPIOCSCOMPRESS */
61struct ppp_option_data {
62 __u8 __user *ptr;
63 __u32 length;
64 int transmit;
65};
66
67/* For PPPIOCGL2TPSTATS */
68struct pppol2tp_ioc_stats {
69 __u16 tunnel_id; /* redundant */
70 __u16 session_id; /* if zero, get tunnel stats */
71 __u32 using_ipsec:1; /* valid only for session_id == 0 */
72 __aligned_u64 tx_packets;
73 __aligned_u64 tx_bytes;
74 __aligned_u64 tx_errors;
75 __aligned_u64 rx_packets;
76 __aligned_u64 rx_bytes;
77 __aligned_u64 rx_seq_discards;
78 __aligned_u64 rx_oos_packets;
79 __aligned_u64 rx_errors;
80};

82/*
83 * Ioctl definitions.
84 */
85
86#define PPPIOCGFLAGS _IOR('t', 90, int) /* get configuration flags */
87#define PPPIOCSFLAGS _IOW('t', 89, int) /* set configuration flags */
88#define PPPIOCGASYNCMAP _IOR('t', 88, int) /* get async map */
89#define PPPIOCSASYNCMAP _IOW('t', 87, int) /* set async map */
90#define PPPIOCGUNIT _IOR('t', 86, int) /* get ppp unit number */
91#define PPPIOCGRASYNCMAP _IOR('t', 85, int) /* get receive async map */
92#define PPPIOCSRASYNCMAP _IOW('t', 84, int) /* set receive async map */
93#define PPPIOCGMRU _IOR('t', 83, int) /* get max receive unit */
94#define PPPIOCSMRU _IOW('t', 82, int) /* set max receive unit */
95#define PPPIOCSMAXCID _IOW('t', 81, int) /* set VJ max slot ID */
96#define PPPIOCGXASYNCMAP _IOR('t', 80, ext_accm) /* get extended ACCM */
97#define PPPIOCSXASYNCMAP _IOW('t', 79, ext_accm) /* set extended ACCM */
98#define PPPIOCXFERUNIT _IO('t', 78) /* transfer PPP unit */
99#define PPPIOCSCOMPRESS _IOW('t', 77, struct ppp_option_data)
100#define PPPIOCGNPMODE _IOWR('t', 76, struct npioctl) /* get NP mode */
101#define PPPIOCSNPMODE _IOW('t', 75, struct npioctl) /* set NP mode */
102#define PPPIOCSPASS _IOW('t', 71, struct sock_fprog) /* set pass filter */
103#define PPPIOCSACTIVE _IOW('t', 70, struct sock_fprog) /* set active filt */
104#define PPPIOCGDEBUG _IOR('t', 65, int) /* Read debug level */
105#define PPPIOCSDEBUG _IOW('t', 64, int) /* Set debug level */
106#define PPPIOCGIDLE _IOR('t', 63, struct ppp_idle) /* get idle time */
107#define PPPIOCNEWUNIT _IOWR('t', 62, int) /* create new ppp unit */
108#define PPPIOCATTACH _IOW('t', 61, int) /* attach to ppp unit */
109#define PPPIOCDETACH _IOW('t', 60, int) /* detach from ppp unit/chan */
110#define PPPIOCSMRRU _IOW('t', 59, int) /* set multilink MRU */
111#define PPPIOCCONNECT _IOW('t', 58, int) /* connect channel to unit */
112#define PPPIOCDISCONN _IO('t', 57) /* disconnect channel */
113#define PPPIOCATTCHAN _IOW('t', 56, int) /* attach to ppp channel */
114#define PPPIOCGCHAN _IOR('t', 55, int) /* get ppp channel number */
115#define PPPIOCGL2TPSTATS _IOR('t', 54, struct pppol2tp_ioc_stats)
116
117#define SIOCGPPPSTATS (SIOCDEVPRIVATE + 0)
118#define SIOCGPPPVER (SIOCDEVPRIVATE + 1) /* NEVER change this!! */
119#define SIOCGPPPCSTATS (SIOCDEVPRIVATE + 2)

121#endif /* _PPP_IOCTL_H */
