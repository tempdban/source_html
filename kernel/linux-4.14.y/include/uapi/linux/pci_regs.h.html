26/*

34/*
35 * Under PCI, each device has 256 bytes of configuration address space,
36 * of which the first 64 bytes are standardized as follows:
37 */
38#define PCI_STD_HEADER_SIZEOF 64
39#define PCI_VENDOR_ID 0x00 /* 16 bits */

71#define PCI_CLASS_REVISION 0x08 /* High 24 bits are class, low 8 revision */

91 * 0xffffffff to the register, and reading it back. Only
92 * 1 bits are decoded.
