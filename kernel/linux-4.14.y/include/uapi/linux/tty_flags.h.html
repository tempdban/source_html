5/*
6 * Definitions for async_struct (and serial_struct) flags field also
7 * shared by the tty_port flags structures.
8 *
9 * Define ASYNCB_* for convenient use with {test,set,clear}_bit.
10 *
11 * Bits [0..ASYNCB_LAST_USER] are userspace defined/visible/changeable
12 * [x] in the bit comments indicates the flag is defunct and no longer used.
13 */
14#define ASYNCB_HUP_NOTIFY 0 /* Notify getty on hangups and closes
15 * on the callout port */
16#define ASYNCB_FOURPORT 1 /* Set OU1, OUT2 per AST Fourport settings */
17#define ASYNCB_SAK 2 /* Secure Attention Key (Orange book) */
18#define ASYNCB_SPLIT_TERMIOS 3 /* [x] Separate termios for dialin/callout */
19#define ASYNCB_SPD_HI 4 /* Use 57600 instead of 38400 bps */
20#define ASYNCB_SPD_VHI 5 /* Use 115200 instead of 38400 bps */
21#define ASYNCB_SKIP_TEST 6 /* Skip UART test during autoconfiguration */
22#define ASYNCB_AUTO_IRQ 7 /* Do automatic IRQ during
23 * autoconfiguration */
24#define ASYNCB_SESSION_LOCKOUT 8 /* [x] Lock out cua opens based on session */
25#define ASYNCB_PGRP_LOCKOUT 9 /* [x] Lock out cua opens based on pgrp */
26#define ASYNCB_CALLOUT_NOHUP 10 /* [x] Don't do hangups for cua device */
27#define ASYNCB_HARDPPS_CD 11 /* Call hardpps when CD goes high */
28#define ASYNCB_SPD_SHI 12 /* Use 230400 instead of 38400 bps */
29#define ASYNCB_LOW_LATENCY 13 /* Request low latency behaviour */
30#define ASYNCB_BUGGY_UART 14 /* This is a buggy UART, skip some safety
31 * checks. Note: can be dangerous! */
32#define ASYNCB_AUTOPROBE 15 /* [x] Port was autoprobed by PCI/PNP code */
33#define ASYNCB_MAGIC_MULTIPLIER 16 /* Use special CLK or divisor */
34#define ASYNCB_LAST_USER 16
35
36/*
37 * Internal flags used only by kernel (read-only)
38 *
39 * WARNING: These flags are no longer used and have been superceded by the
40 * TTY_PORT_ flags in the iflags field (and not userspace-visible)
41 */
42#ifndef _KERNEL_
43#define ASYNCB_INITIALIZED 31 /* Serial port was initialized */
44#define ASYNCB_SUSPENDED 30 /* Serial port is suspended */
45#define ASYNCB_NORMAL_ACTIVE 29 /* Normal device is active */
46#define ASYNCB_BOOT_AUTOCONF 28 /* Autoconfigure port on bootup */
47#define ASYNCB_CLOSING 27 /* Serial port is closing */
48#define ASYNCB_CTS_FLOW 26 /* Do CTS flow control */
49#define ASYNCB_CHECK_CD 25 /* i.e., CLOCAL */
50#define ASYNCB_SHARE_IRQ 24 /* for multifunction cards, no longer used */
51#define ASYNCB_CONS_FLOW 23 /* flow control for console */
52#define ASYNCB_FIRST_KERNEL 22
53#endif
54
55/* Masks */
56#define ASYNC_HUP_NOTIFY (1U << ASYNCB_HUP_NOTIFY)
57#define ASYNC_SUSPENDED (1U << ASYNCB_SUSPENDED)
58#define ASYNC_FOURPORT (1U << ASYNCB_FOURPORT)
59#define ASYNC_SAK (1U << ASYNCB_SAK)
60#define ASYNC_SPLIT_TERMIOS (1U << ASYNCB_SPLIT_TERMIOS)
61#define ASYNC_SPD_HI (1U << ASYNCB_SPD_HI)
62#define ASYNC_SPD_VHI (1U << ASYNCB_SPD_VHI)
63#define ASYNC_SKIP_TEST (1U << ASYNCB_SKIP_TEST)
64#define ASYNC_AUTO_IRQ (1U << ASYNCB_AUTO_IRQ)
65#define ASYNC_SESSION_LOCKOUT (1U << ASYNCB_SESSION_LOCKOUT)
66#define ASYNC_PGRP_LOCKOUT (1U << ASYNCB_PGRP_LOCKOUT)
67#define ASYNC_CALLOUT_NOHUP (1U << ASYNCB_CALLOUT_NOHUP)
68#define ASYNC_HARDPPS_CD (1U << ASYNCB_HARDPPS_CD)
69#define ASYNC_SPD_SHI (1U << ASYNCB_SPD_SHI)
70#define ASYNC_LOW_LATENCY (1U << ASYNCB_LOW_LATENCY)
71#define ASYNC_BUGGY_UART (1U << ASYNCB_BUGGY_UART)
72#define ASYNC_AUTOPROBE (1U << ASYNCB_AUTOPROBE)
73#define ASYNC_MAGIC_MULTIPLIER (1U << ASYNCB_MAGIC_MULTIPLIER)
74
75#define ASYNC_FLAGS ((1U << (ASYNCB_LAST_USER + 1)) - 1)
76#define ASYNC_DEPRECATED (ASYNC_SESSION_LOCKOUT | ASYNC_PGRP_LOCKOUT | \
77 ASYNC_CALLOUT_NOHUP | ASYNC_AUTOPROBE)
78#define ASYNC_USR_MASK (ASYNC_SPD_MASK|ASYNC_CALLOUT_NOHUP| \
79 ASYNC_LOW_LATENCY)
80#define ASYNC_SPD_CUST (ASYNC_SPD_HI|ASYNC_SPD_VHI)
81#define ASYNC_SPD_WARP (ASYNC_SPD_HI|ASYNC_SPD_SHI)
82#define ASYNC_SPD_MASK (ASYNC_SPD_HI|ASYNC_SPD_VHI|ASYNC_SPD_SHI)

85/* These flags are no longer used (and were always masked from userspace) */
86#define ASYNC_INITIALIZED (1U << ASYNCB_INITIALIZED)

93#define ASYNC_CONS_FLOW (1U << ASYNCB_CONS_FLOW)
94#define ASYNC_INTERNAL_FLAGS (~((1U << ASYNCB_FIRST_KERNEL) - 1))
95#endif
