5#include <linux/types.h>

7enum xt_l2tp_type {
8 XT_L2TP_TYPE_CONTROL,
9 XT_L2TP_TYPE_DATA,

12/* L2TP matching stuff */
13struct xt_l2tp_info {
14 __u32 tid; /* tunnel id */
15 __u32 sid; /* session id */
16 __u8 version; /* L2TP protocol version */
17 __u8 type; /* L2TP packet type */
18 __u8 flags; /* which fields to match */
19};
20
21enum {
22 XT_L2TP_TID = (1 << 0), /* match L2TP tunnel id */
23 XT_L2TP_SID = (1 << 1), /* match L2TP session id */
24 XT_L2TP_VERSION = (1 << 2), /* match L2TP protocol version */
25 XT_L2TP_TYPE = (1 << 3), /* match L2TP packet type */
26};
