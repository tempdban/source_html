23#define ICMP_ECHOREPLY 0 /* Echo Reply */
24#define ICMP_DEST_UNREACH 3 /* Destination Unreachable */
25#define ICMP_SOURCE_QUENCH 4 /* Source Quench */
26#define ICMP_REDIRECT 5 /* Redirect (change route) */
27#define ICMP_ECHO 8 /* Echo Request */
28#define ICMP_TIME_EXCEEDED 11 /* Time Exceeded */
29#define ICMP_PARAMETERPROB 12 /* Parameter Problem */
30#define ICMP_TIMESTAMP 13 /* Timestamp Request */
31#define ICMP_TIMESTAMPREPLY 14 /* Timestamp Reply */
32#define ICMP_INFO_REQUEST 15 /* Information Request */
33#define ICMP_INFO_REPLY 16 /* Information Reply */
34#define ICMP_ADDRESS 17 /* Address Mask Request */
35#define ICMP_ADDRESSREPLY 18 /* Address Mask Reply */
36#define NR_ICMP_TYPES 18

39/* Codes for UNREACH. */
40#define ICMP_NET_UNREACH 0 /* Network Unreachable */
41#define ICMP_HOST_UNREACH 1 /* Host Unreachable */
42#define ICMP_PROT_UNREACH 2 /* Protocol Unreachable */
43#define ICMP_PORT_UNREACH 3 /* Port Unreachable */
44#define ICMP_FRAG_NEEDED 4 /* Fragmentation Needed/DF set */
45#define ICMP_SR_FAILED 5 /* Source Route failed */
46#define ICMP_NET_UNKNOWN 6
47#define ICMP_HOST_UNKNOWN 7
48#define ICMP_HOST_ISOLATED 8
49#define ICMP_NET_ANO 9
50#define ICMP_HOST_ANO 10
51#define ICMP_NET_UNR_TOS 11
52#define ICMP_HOST_UNR_TOS 12
53#define ICMP_PKT_FILTERED 13 /* Packet filtered */
54#define ICMP_PREC_VIOLATION 14 /* Precedence violation */
55#define ICMP_PREC_CUTOFF 15 /* Precedence cut off */
56#define NR_ICMP_UNREACH 15 /* instead of hardcoding immediate value */
57
58/* Codes for REDIRECT. */
59#define ICMP_REDIR_NET 0 /* Redirect Net */
60#define ICMP_REDIR_HOST 1 /* Redirect Host */
61#define ICMP_REDIR_NETTOS 2 /* Redirect Net for TOS */
62#define ICMP_REDIR_HOSTTOS 3 /* Redirect Host for TOS */
63
64/* Codes for TIME_EXCEEDED. */
65#define ICMP_EXC_TTL 0 /* TTL count exceeded */
66#define ICMP_EXC_FRAGTIME 1 /* Fragment Reass time exceeded */
67
68
69struct icmphdr {
70 __u8 type;
71 __u8 code;
72 __sum16 checksum;
73 union {
74 struct {
75 __be16 id;
76 __be16 sequence;
77 } echo;
78 __be32 gateway;
79 struct {
80 __be16 __unused;
81 __be16 mtu;
82 } frag;
83 __u8 reserved[4];
84 } un;
85};
