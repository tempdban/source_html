16#ifndef _LINUX_CGROUPSTATS_H
17#define _LINUX_CGROUPSTATS_H
18
19#include <linux/types.h>
20#include <linux/taskstats.h>
21
22/*
23 * Data shared between user space and kernel space on a per cgroup
24 * basis. This data is shared using taskstats.
25 *
26 * Most of these states are derived by looking at the task->state value
27 * For the nr_io_wait state, a flag in the delay accounting structure
28 * indicates that the task is waiting on IO
29 *
30 * Each member is aligned to a 8 byte boundary.
31 */
32struct cgroupstats {
33 __u64 nr_sleeping; /* Number of tasks sleeping */
34 __u64 nr_running; /* Number of tasks running */
35 __u64 nr_stopped; /* Number of tasks in stopped state */
36 __u64 nr_uninterruptible; /* Number of tasks in uninterruptible */
37 /* state */
38 __u64 nr_io_wait; /* Number of tasks waiting on IO */
39};
40
41/*
42 * Commands sent from userspace
43 * Not versioned. New commands should only be inserted at the enum's end
44 * prior to __CGROUPSTATS_CMD_MAX
45 */
46
47enum {
48 CGROUPSTATS_CMD_UNSPEC = __TASKSTATS_CMD_MAX, /* Reserved */
49 CGROUPSTATS_CMD_GET, /* user->kernel request/get-response */
50 CGROUPSTATS_CMD_NEW, /* kernel->user event */
51 __CGROUPSTATS_CMD_MAX,
52};
53
54#define CGROUPSTATS_CMD_MAX (__CGROUPSTATS_CMD_MAX - 1)
55
56enum {
57 CGROUPSTATS_TYPE_UNSPEC = 0, /* Reserved */
58 CGROUPSTATS_TYPE_CGROUP_STATS, /* contains name + stats */
59 __CGROUPSTATS_TYPE_MAX,
60};
61
62#define CGROUPSTATS_TYPE_MAX (__CGROUPSTATS_TYPE_MAX - 1)
63
64enum {
65 CGROUPSTATS_CMD_ATTR_UNSPEC = 0,
66 CGROUPSTATS_CMD_ATTR_FD,
67 __CGROUPSTATS_CMD_ATTR_MAX,
68};
69
70#define CGROUPSTATS_CMD_ATTR_MAX (__CGROUPSTATS_CMD_ATTR_MAX - 1)
71
72#endif /* _LINUX_CGROUPSTATS_H */
