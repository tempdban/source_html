33/*
34 * Generic ptrace interface that exports the architecture specific regsets
35 * using the corresponding NT_* types (which are also used in the core dump).
36 * Please note that the NT_PRSTATUS note type in a core dump contains a full
37 * 'struct elf_prstatus'. But the user_regset for NT_PRSTATUS contains just the
38 * elf_gregset_t that is the pr_reg field of 'struct elf_prstatus'. For all the
39 * other user_regset flavors, the user_regset layout and the ELF core dump note
40 * payload are exactly the same layout.
41 *
42 * This interface usage is as follows:
43 * struct iovec iov = { buf, len};
44 *
45 * ret = ptrace(PTRACE_GETREGSET/PTRACE_SETREGSET, pid, NT_XXX_TYPE, &iov);
46 *
47 * On the successful completion, iov.len will be updated by the kernel,
48 * specifying how much the kernel has written/read to/from the user's iov.buf.
49 */
50#define PTRACE_GETREGSET 0x4204
51#define PTRACE_SETREGSET 0x4205
52
53#define PTRACE_SEIZE 0x4206
54#define PTRACE_INTERRUPT 0x4207
55#define PTRACE_LISTEN 0x4208

73/* Wait extended result codes for the above trace options. */
74#define PTRACE_EVENT_FORK 1
75#define PTRACE_EVENT_VFORK 2
76#define PTRACE_EVENT_CLONE 3
77#define PTRACE_EVENT_EXEC 4
78#define PTRACE_EVENT_VFORK_DONE 5
79#define PTRACE_EVENT_EXIT 6
80#define PTRACE_EVENT_SECCOMP 7
81/* Extended result codes which enabled by means other than options. */
82#define PTRACE_EVENT_STOP 128
83
84/* Options set using PTRACE_SETOPTIONS or using PTRACE_SEIZE @data param */
85#define PTRACE_O_TRACESYSGOOD 1
86#define PTRACE_O_TRACEFORK (1 << PTRACE_EVENT_FORK)
87#define PTRACE_O_TRACEVFORK (1 << PTRACE_EVENT_VFORK)
88#define PTRACE_O_TRACECLONE (1 << PTRACE_EVENT_CLONE)
89#define PTRACE_O_TRACEEXEC (1 << PTRACE_EVENT_EXEC)
90#define PTRACE_O_TRACEVFORKDONE (1 << PTRACE_EVENT_VFORK_DONE)
91#define PTRACE_O_TRACEEXIT (1 << PTRACE_EVENT_EXIT)
92#define PTRACE_O_TRACESECCOMP (1 << PTRACE_EVENT_SECCOMP)
93
94/* eventless options */
95#define PTRACE_O_EXITKILL (1 << 20)
96#define PTRACE_O_SUSPEND_SECCOMP (1 << 21)
97
98#define PTRACE_O_MASK (\
99 0x000000ff | PTRACE_O_EXITKILL | PTRACE_O_SUSPEND_SECCOMP)
