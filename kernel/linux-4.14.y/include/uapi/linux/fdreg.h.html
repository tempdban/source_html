2#ifndef _LINUX_FDREG_H
3#define _LINUX_FDREG_H
4/*
5 * This file contains some defines for the floppy disk controller.
6 * Various sources. Mostly "IBM Microcomputers: A Programmers
7 * Handbook", Sanches and Canton.
8 */
9
10#ifdef FDPATCHES
11#define FD_IOPORT fdc_state[fdc].address
12#else
13/* It would be a lot saner just to force fdc_state[fdc].address to always
14 be set ! FIXME */
15#define FD_IOPORT 0x3f0
16#endif
17
18/* Fd controller regs. S&C, about page 340 */
19#define FD_STATUS (4 + FD_IOPORT )
20#define FD_DATA (5 + FD_IOPORT )
21
22/* Digital Output Register */
23#define FD_DOR (2 + FD_IOPORT )
24
25/* Digital Input Register (read) */
26#define FD_DIR (7 + FD_IOPORT )
27
28/* Diskette Control Register (write)*/
29#define FD_DCR (7 + FD_IOPORT )
30
31/* Bits of main status register */
32#define STATUS_BUSYMASK 0x0F /* drive busy mask */
33#define STATUS_BUSY 0x10 /* FDC busy */
34#define STATUS_DMA 0x20 /* 0- DMA mode */
35#define STATUS_DIR 0x40 /* 0- cpu->fdc */
36#define STATUS_READY 0x80 /* Data reg ready */
37
38/* Bits of FD_ST0 */
39#define ST0_DS 0x03 /* drive select mask */
40#define ST0_HA 0x04 /* Head (Address) */
41#define ST0_NR 0x08 /* Not Ready */
42#define ST0_ECE 0x10 /* Equipment check error */
43#define ST0_SE 0x20 /* Seek end */
44#define ST0_INTR 0xC0 /* Interrupt code mask */
45
46/* Bits of FD_ST1 */
47#define ST1_MAM 0x01 /* Missing Address Mark */
48#define ST1_WP 0x02 /* Write Protect */
49#define ST1_ND 0x04 /* No Data - unreadable */
50#define ST1_OR 0x10 /* OverRun */
51#define ST1_CRC 0x20 /* CRC error in data or addr */
52#define ST1_EOC 0x80 /* End Of Cylinder */
53
54/* Bits of FD_ST2 */
55#define ST2_MAM 0x01 /* Missing Address Mark (again) */
56#define ST2_BC 0x02 /* Bad Cylinder */
57#define ST2_SNS 0x04 /* Scan Not Satisfied */
58#define ST2_SEH 0x08 /* Scan Equal Hit */
59#define ST2_WC 0x10 /* Wrong Cylinder */
60#define ST2_CRC 0x20 /* CRC error in data field */
61#define ST2_CM 0x40 /* Control Mark = deleted */
62
63/* Bits of FD_ST3 */
64#define ST3_HA 0x04 /* Head (Address) */
65#define ST3_DS 0x08 /* drive is double-sided */
66#define ST3_TZ 0x10 /* Track Zero signal (1=track 0) */
67#define ST3_RY 0x20 /* drive is ready */
68#define ST3_WP 0x40 /* Write Protect */
69#define ST3_FT 0x80 /* Drive Fault */
70
71/* Values for FD_COMMAND */
72#define FD_RECALIBRATE 0x07 /* move to track 0 */
73#define FD_SEEK 0x0F /* seek track */
74#define FD_READ 0xE6 /* read with MT, MFM, SKip deleted */
75#define FD_WRITE 0xC5 /* write with MT, MFM */
76#define FD_SENSEI 0x08 /* Sense Interrupt Status */
77#define FD_SPECIFY 0x03 /* specify HUT etc */
78#define FD_FORMAT 0x4D /* format one track */
79#define FD_VERSION 0x10 /* get version code */
80#define FD_CONFIGURE 0x13 /* configure FIFO operation */
81#define FD_PERPENDICULAR 0x12 /* perpendicular r/w mode */
82#define FD_GETSTATUS 0x04 /* read ST3 */
83#define FD_DUMPREGS 0x0E /* dump the contents of the fdc regs */
84#define FD_READID 0xEA /* prints the header of a sector */
85#define FD_UNLOCK 0x14 /* Fifo config unlock */
86#define FD_LOCK 0x94 /* Fifo config lock */
87#define FD_RSEEK_OUT 0x8f /* seek out (i.e. to lower tracks) */
88#define FD_RSEEK_IN 0xcf /* seek in (i.e. to higher tracks) */
89
90/* the following commands are new in the 82078. They are not used in the
91 * floppy driver, except the first three. These commands may be useful for apps
92 * which use the FDRAWCMD interface. For doc, get the 82078 spec sheets at
93 * http://www.intel.com/design/archives/periphrl/docs/29046803.htm */
94
95#define FD_PARTID 0x18 /* part id ("extended" version cmd) */
96#define FD_SAVE 0x2e /* save fdc regs for later restore */
97#define FD_DRIVESPEC 0x8e /* drive specification: Access to the
98 * 2 Mbps data transfer rate for tape
99 * drives */
100
101#define FD_RESTORE 0x4e /* later restore */
102#define FD_POWERDOWN 0x27 /* configure FDC's powersave features */
103#define FD_FORMAT_N_WRITE 0xef /* format and write in one go. */
104#define FD_OPTION 0x33 /* ISO format (which is a clean way to
105 * pack more sectors on a track) */
106
107/* DMA commands */
108#define DMA_READ 0x46
109#define DMA_WRITE 0x4A
110
111/* FDC version return types */
112#define FDC_NONE 0x00
113#define FDC_UNKNOWN 0x10 /* DO NOT USE THIS TYPE EXCEPT IF IDENTIFICATION
114 FAILS EARLY */
115#define FDC_8272A 0x20 /* Intel 8272a, NEC 765 */
116#define FDC_765ED 0x30 /* Non-Intel 1MB-compatible FDC, can't detect */
117#define FDC_82072 0x40 /* Intel 82072; 8272a + FIFO + DUMPREGS */
118#define FDC_82072A 0x45 /* 82072A (on Sparcs) */
119#define FDC_82077_ORIG 0x51 /* Original version of 82077AA, sans LOCK */
120#define FDC_82077 0x52 /* 82077AA-1 */
121#define FDC_82078_UNKN 0x5f /* Unknown 82078 variant */
122#define FDC_82078 0x60 /* 44pin 82078 or 64pin 82078SL */
123#define FDC_82078_1 0x61 /* 82078-1 (2Mbps fdc) */
124#define FDC_S82078B 0x62 /* S82078B (first seen on Adaptec AVA-2825 VLB
125 * SCSI/EIDE/Floppy controller) */
126#define FDC_87306 0x63 /* National Semiconductor PC 87306 */
127
128/*
129 * Beware: the fdc type list is roughly sorted by increasing features.
130 * Presence of features is tested by comparing the FDC version id with the
131 * "oldest" version that has the needed feature.
132 * If during FDC detection, an obscure test fails late in the sequence, don't
133 * assign FDC_UNKNOWN. Else the FDC will be treated as a dumb 8272a, or worse.
134 * This is especially true if the tests are unneeded.
135 */
136
137#define FD_RESET_DELAY 20
138#endif
