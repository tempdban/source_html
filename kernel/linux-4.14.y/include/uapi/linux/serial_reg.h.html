24#define UART_IER 1 /* Out: Interrupt Enable Register */
25#define UART_IER_MSI 0x08 /* Enable Modem status interrupt */
26#define UART_IER_RLSI 0x04 /* Enable receiver line status interrupt */
27#define UART_IER_THRI 0x02 /* Enable Transmitter holding register int. */
28#define UART_IER_RDI 0x01 /* Enable receiver data interrupt */

30 * Sleep mode for ST16650 and TI16750. For the ST16650, EFR[4]=1

32#define UART_IERX_SLEEP 0x10 /* Enable sleep mode */

34#define UART_IIR 2 /* In: Interrupt ID Register */
35#define UART_IIR_NO_INT 0x01 /* No interrupts pending */
36#define UART_IIR_ID 0x0e /* Mask for the interrupt ID */
37#define UART_IIR_MSI 0x00 /* Modem status interrupt */
38#define UART_IIR_THRI 0x02 /* Transmitter holding register empty */
39#define UART_IIR_RDI 0x04 /* Receiver data interrupt */
40#define UART_IIR_RLSI 0x06 /* Receiver line status interrupt */
41
42#define UART_IIR_BUSY 0x07 /* DesignWare APB Busy Detect */
43
44#define UART_IIR_RX_TIMEOUT 0x0c /* OMAP RX Timeout interrupt */
45#define UART_IIR_XOFF 0x10 /* OMAP XOFF/Special Character */
46#define UART_IIR_CTS_RTS_DSR 0x20 /* OMAP CTS/RTS/DSR Change */

48#define UART_FCR 2 /* Out: FIFO Control Register */
49#define UART_FCR_ENABLE_FIFO 0x01 /* Enable the FIFO */
50#define UART_FCR_CLEAR_RCVR 0x02 /* Clear the RCVR FIFO */
51#define UART_FCR_CLEAR_XMIT 0x04 /* Clear the XMIT FIFO */
52#define UART_FCR_DMA_SELECT 0x08 /* For DMA applications */
53/*
54 * Note: The FIFO trigger levels are chip specific:
55 * RX:76 = 00 01 10 11 TX:54 = 00 01 10 11
56 * PC16550D: 1 4 8 14 xx xx xx xx
57 * TI16C550A: 1 4 8 14 xx xx xx xx
58 * TI16C550C: 1 4 8 14 xx xx xx xx
59 * ST16C550: 1 4 8 14 xx xx xx xx
60 * ST16C650: 8 16 24 28 16 8 24 30 PORT_16650V2
61 * NS16C552: 1 4 8 14 xx xx xx xx
62 * ST16C654: 8 16 56 60 8 16 32 56 PORT_16654
63 * TI16C750: 1 16 32 56 xx xx xx xx PORT_16750
64 * TI16C752: 8 16 56 60 8 16 32 56
65 * Tegra: 1 4 8 14 16 8 4 1 PORT_TEGRA
66 */
67#define UART_FCR_R_TRIG_00 0x00
68#define UART_FCR_R_TRIG_01 0x40
69#define UART_FCR_R_TRIG_10 0x80
70#define UART_FCR_R_TRIG_11 0xc0
71#define UART_FCR_T_TRIG_00 0x00
72#define UART_FCR_T_TRIG_01 0x10
73#define UART_FCR_T_TRIG_10 0x20
74#define UART_FCR_T_TRIG_11 0x30
75
76#define UART_FCR_TRIGGER_MASK 0xC0 /* Mask for the FIFO trigger range */
77#define UART_FCR_TRIGGER_1 0x00 /* Mask for trigger set at 1 */
78#define UART_FCR_TRIGGER_4 0x40 /* Mask for trigger set at 4 */
79#define UART_FCR_TRIGGER_8 0x80 /* Mask for trigger set at 8 */
80#define UART_FCR_TRIGGER_14 0xC0 /* Mask for trigger set at 14 */
81/* 16650 definitions */
82#define UART_FCR6_R_TRIGGER_8 0x00 /* Mask for receive trigger set at 1 */
83#define UART_FCR6_R_TRIGGER_16 0x40 /* Mask for receive trigger set at 4 */
84#define UART_FCR6_R_TRIGGER_24 0x80 /* Mask for receive trigger set at 8 */
85#define UART_FCR6_R_TRIGGER_28 0xC0 /* Mask for receive trigger set at 14 */
86#define UART_FCR6_T_TRIGGER_16 0x00 /* Mask for transmit trigger set at 16 */
87#define UART_FCR6_T_TRIGGER_8 0x10 /* Mask for transmit trigger set at 8 */
88#define UART_FCR6_T_TRIGGER_24 0x20 /* Mask for transmit trigger set at 24 */
89#define UART_FCR6_T_TRIGGER_30 0x30 /* Mask for transmit trigger set at 30 */
90#define UART_FCR7_64BYTE 0x20 /* Go into 64 byte mode (TI16C750 and
91 some Freescale UARTs) */
92
93#define UART_FCR_R_TRIG_SHIFT 6
94#define UART_FCR_R_TRIG_BITS(x) \
95 (((x) & UART_FCR_TRIGGER_MASK) >> UART_FCR_R_TRIG_SHIFT)
96#define UART_FCR_R_TRIG_MAX_STATE 4

98#define UART_LCR 3 /* Out: Line Control Register */
99/*
100 * Note: if the word length is 5 bits (UART_LCR_WLEN5), then setting
101 * UART_LCR_STOP will select 1.5 stop bits, not 2 stop bits.
102 */
103#define UART_LCR_DLAB 0x80 /* Divisor latch access bit */
104#define UART_LCR_SBC 0x40 /* Set break control */
105#define UART_LCR_SPAR 0x20 /* Stick parity (?) */
106#define UART_LCR_EPAR 0x10 /* Even parity select */
107#define UART_LCR_PARITY 0x08 /* Parity Enable */
108#define UART_LCR_STOP 0x04 /* Stop bits: 0=1 bit, 1=2 bits */
109#define UART_LCR_WLEN5 0x00 /* Wordlength: 5 bits */
110#define UART_LCR_WLEN6 0x01 /* Wordlength: 6 bits */
111#define UART_LCR_WLEN7 0x02 /* Wordlength: 7 bits */
112#define UART_LCR_WLEN8 0x03 /* Wordlength: 8 bits */

121#define UART_MCR 4 /* Out: Modem Control Register */
122#define UART_MCR_CLKSEL 0x80 /* Divide clock by 4 (TI16C752, EFR[4]=1) */
123#define UART_MCR_TCRTLR 0x40 /* Access TCR/TLR (TI16C752, EFR[4]=1) */
124#define UART_MCR_XONANY 0x20 /* Enable Xon Any (TI16C752, EFR[4]=1) */
125#define UART_MCR_AFE 0x20 /* Enable auto-RTS/CTS (TI16C550C/TI16C750) */
126#define UART_MCR_LOOP 0x10 /* Enable loopback test mode */
127#define UART_MCR_OUT2 0x08 /* Out2 complement */
128#define UART_MCR_OUT1 0x04 /* Out1 complement */
129#define UART_MCR_RTS 0x02 /* RTS complement */

132#define UART_LSR 5 /* In: Line Status Register */
133#define UART_LSR_FIFOE 0x80 /* Fifo error */
134#define UART_LSR_TEMT 0x40 /* Transmitter empty */
135#define UART_LSR_THRE 0x20 /* Transmit-hold-register empty */
136#define UART_LSR_BI 0x10 /* Break interrupt indicator */
137#define UART_LSR_FE 0x08 /* Frame error indicator */
138#define UART_LSR_PE 0x04 /* Parity error indicator */
139#define UART_LSR_OE 0x02 /* Overrun error indicator */
140#define UART_LSR_DR 0x01 /* Receiver data ready */
141#define UART_LSR_BRK_ERROR_BITS 0x1E /* BI, FE, PE, OE bits */

143#define UART_MSR 6 /* In: Modem Status Register */
144#define UART_MSR_DCD 0x80 /* Data Carrier Detect */
145#define UART_MSR_RI 0x40 /* Ring Indicator */
146#define UART_MSR_DSR 0x20 /* Data Set Ready */
147#define UART_MSR_CTS 0x10 /* Clear to Send */
148#define UART_MSR_DDCD 0x08 /* Delta DCD */
149#define UART_MSR_TERI 0x04 /* Trailing edge ring indicator */
150#define UART_MSR_DDSR 0x02 /* Delta DSR */
151#define UART_MSR_DCTS 0x01 /* Delta CTS */
152#define UART_MSR_ANY_DELTA 0x0F /* Any of the delta bits! */
153
154#define UART_SCR 7 /* I/O: Scratch Register */

282/*
283 * These definitions are for the RSA-DV II/S card, from
284 *
285 * Kiyokazu SUTO <suto@ks-and-ks.ne.jp>

287
288#define UART_RSA_BASE (-8)
289
290#define UART_RSA_MSR ((UART_RSA_BASE) + 0) /* I/O: Mode Select Register */
291
292#define UART_RSA_MSR_SWAP (1 << 0) /* Swap low/high 8 bytes in I/O port addr */
293#define UART_RSA_MSR_FIFO (1 << 2) /* Enable the external FIFO */
294#define UART_RSA_MSR_FLOW (1 << 3) /* Enable the auto RTS/CTS flow control */
295#define UART_RSA_MSR_ITYP (1 << 4) /* Level (1) / Edge triger (0) */
296
297#define UART_RSA_IER ((UART_RSA_BASE) + 1) /* I/O: Interrupt Enable Register */
298
299#define UART_RSA_IER_Rx_FIFO_H (1 << 0) /* Enable Rx FIFO half full int. */
300#define UART_RSA_IER_Tx_FIFO_H (1 << 1) /* Enable Tx FIFO half full int. */
301#define UART_RSA_IER_Tx_FIFO_E (1 << 2) /* Enable Tx FIFO empty int. */
302#define UART_RSA_IER_Rx_TOUT (1 << 3) /* Enable char receive timeout int */
303#define UART_RSA_IER_TIMER (1 << 4) /* Enable timer interrupt */
304
305#define UART_RSA_SRR ((UART_RSA_BASE) + 2) /* IN: Status Read Register */
306
307#define UART_RSA_SRR_Tx_FIFO_NEMP (1 << 0) /* Tx FIFO is not empty (1) */
308#define UART_RSA_SRR_Tx_FIFO_NHFL (1 << 1) /* Tx FIFO is not half full (1) */
309#define UART_RSA_SRR_Tx_FIFO_NFUL (1 << 2) /* Tx FIFO is not full (1) */
310#define UART_RSA_SRR_Rx_FIFO_NEMP (1 << 3) /* Rx FIFO is not empty (1) */
311#define UART_RSA_SRR_Rx_FIFO_NHFL (1 << 4) /* Rx FIFO is not half full (1) */
312#define UART_RSA_SRR_Rx_FIFO_NFUL (1 << 5) /* Rx FIFO is not full (1) */
313#define UART_RSA_SRR_Rx_TOUT (1 << 6) /* Character reception timeout occurred (1) */
314#define UART_RSA_SRR_TIMER (1 << 7) /* Timer interrupt occurred */
315
316#define UART_RSA_FRR ((UART_RSA_BASE) + 2) /* OUT: FIFO Reset Register */
317
318#define UART_RSA_TIVSR ((UART_RSA_BASE) + 3) /* I/O: Timer Interval Value Set Register */
319
320#define UART_RSA_TCR ((UART_RSA_BASE) + 4) /* OUT: Timer Control Register */

322#define UART_RSA_TCR_SWITCH (1 << 0) /* Timer on */
