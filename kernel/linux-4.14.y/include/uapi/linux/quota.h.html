55/*
56 * Command definitions for the 'quotactl' system call.
57 * The commands are broken into a main command defined below
58 * and a subcommand that is used to convey the type of
59 * quota that is being manipulated (see above).
60 */
61#define SUBCMDMASK 0x00ff
62#define SUBCMDSHIFT 8
63#define QCMD(cmd, type) (((cmd) << SUBCMDSHIFT) | ((type) & SUBCMDMASK))
64
65#define Q_SYNC 0x800001 /* sync disk copy of a filesystems quotas */
66#define Q_QUOTAON 0x800002 /* turn quotas on */
67#define Q_QUOTAOFF 0x800003 /* turn quotas off */
68#define Q_GETFMT 0x800004 /* get quota format used on given filesystem */
69#define Q_GETINFO 0x800005 /* get information about quota files */
70#define Q_SETINFO 0x800006 /* set information about quota files */
71#define Q_GETQUOTA 0x800007 /* get user quota structure */
72#define Q_SETQUOTA 0x800008 /* set user quota structure */
73#define Q_GETNEXTQUOTA 0x800009 /* get disk limits and usage >= ID */
74
75/* Quota format type IDs */
76#define QFMT_VFS_OLD 1
77#define QFMT_VFS_V0 2
78#define QFMT_OCFS2 3
79#define QFMT_VFS_V1 4

81/* Size of block in which space limits are passed through the quota
82 * interface */
83#define QIF_DQBLKSIZE_BITS 10
84#define QIF_DQBLKSIZE (1 << QIF_DQBLKSIZE_BITS)
85
86/*
87 * Quota structure used for communication with userspace via quotactl
88 * Following flags are used to specify which fields are valid
89 */
90enum {
91 QIF_BLIMITS_B = 0,
92 QIF_SPACE_B,
93 QIF_ILIMITS_B,
94 QIF_INODES_B,
95 QIF_BTIME_B,
96 QIF_ITIME_B,
97};
98
99#define QIF_BLIMITS (1 << QIF_BLIMITS_B)
100#define QIF_SPACE (1 << QIF_SPACE_B)
101#define QIF_ILIMITS (1 << QIF_ILIMITS_B)
102#define QIF_INODES (1 << QIF_INODES_B)
103#define QIF_BTIME (1 << QIF_BTIME_B)
104#define QIF_ITIME (1 << QIF_ITIME_B)
105#define QIF_LIMITS (QIF_BLIMITS | QIF_ILIMITS)
106#define QIF_USAGE (QIF_SPACE | QIF_INODES)
107#define QIF_TIMES (QIF_BTIME | QIF_ITIME)
108#define QIF_ALL (QIF_LIMITS | QIF_USAGE | QIF_TIMES)
109
110struct if_dqblk {
111 __u64 dqb_bhardlimit;
112 __u64 dqb_bsoftlimit;
113 __u64 dqb_curspace;
114 __u64 dqb_ihardlimit;
115 __u64 dqb_isoftlimit;
116 __u64 dqb_curinodes;
117 __u64 dqb_btime;
118 __u64 dqb_itime;
119 __u32 dqb_valid;
120};

122struct if_nextdqblk {
123 __u64 dqb_bhardlimit;
124 __u64 dqb_bsoftlimit;

127 __u64 dqb_isoftlimit;
128 __u64 dqb_curinodes;
129 __u64 dqb_btime;
130 __u64 dqb_itime;
131 __u32 dqb_valid;
132 __u32 dqb_id;
133};

135/*
136 * Structure used for setting quota information about file via quotactl
137 * Following flags are used to specify which fields are valid
138 */
139#define IIF_BGRACE 1
140#define IIF_IGRACE 2
141#define IIF_FLAGS 4
142#define IIF_ALL (IIF_BGRACE | IIF_IGRACE | IIF_FLAGS)
143
144enum {
145 DQF_ROOT_SQUASH_B = 0,
146 DQF_SYS_FILE_B = 16,
147 /* Kernel internal flags invisible to userspace */
148 DQF_PRIVATE
149};
150
151/* Root squash enabled (for v1 quota format) */
152#define DQF_ROOT_SQUASH (1 << DQF_ROOT_SQUASH_B)
153/* Quota stored in a system file */
154#define DQF_SYS_FILE (1 << DQF_SYS_FILE_B)
155
156struct if_dqinfo {
157 __u64 dqi_bgrace;
158 __u64 dqi_igrace;
159 __u32 dqi_flags; /* DFQ_* */
160 __u32 dqi_valid;
161};

163/*
164 * Definitions for quota netlink interface

166#define QUOTA_NL_NOWARN 0
167#define QUOTA_NL_IHARDWARN 1 /* Inode hardlimit reached */
168#define QUOTA_NL_ISOFTLONGWARN 2 /* Inode grace time expired */
169#define QUOTA_NL_ISOFTWARN 3 /* Inode softlimit reached */
170#define QUOTA_NL_BHARDWARN 4 /* Block hardlimit reached */
171#define QUOTA_NL_BSOFTLONGWARN 5 /* Block grace time expired */
172#define QUOTA_NL_BSOFTWARN 6 /* Block softlimit reached */
173#define QUOTA_NL_IHARDBELOW 7 /* Usage got below inode hardlimit */
174#define QUOTA_NL_ISOFTBELOW 8 /* Usage got below inode softlimit */
175#define QUOTA_NL_BHARDBELOW 9 /* Usage got below block hardlimit */
176#define QUOTA_NL_BSOFTBELOW 10 /* Usage got below block softlimit */
