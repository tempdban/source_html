285/* Mapping between MDIO PRTAD/DEVAD and mii_ioctl_data::phy_id */
286
287#define MDIO_PHY_ID_C45 0x8000
288#define MDIO_PHY_ID_PRTAD 0x03e0
289#define MDIO_PHY_ID_DEVAD 0x001f
290#define MDIO_PHY_ID_C45_MASK \
291 (MDIO_PHY_ID_C45 | MDIO_PHY_ID_PRTAD | MDIO_PHY_ID_DEVAD)
292
293static inline __u16 mdio_phy_id_c45(int prtad, int devad)
294{
295 return MDIO_PHY_ID_C45 | (prtad << 5) | devad;
296}
