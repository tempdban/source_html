50/*
51 * socket option name coding functions
52 *
53 * Note that __SO_ENCODE and __SO_LEVEL are somewhat a hack since the
54 * << 22 only reserves 9 bits for the level. On some architectures
55 * SOL_SOCKET is 0xFFFF, so that's a bit of a problem
56 */
57
58#define __SO_ENCODE(l,n,t) ((((l) & 0x1FF) << 22) | ((n) << 16) | \
59 sizeof(t))
60#define __SO_LEVEL_MATCH(c,m) (((c) >> 22) == ((m) & 0x1FF))
61#define __SO_NUMBER(c) (((c) >> 16) & 0x3f)
62#define __SO_SIZE(c) ((c) & 0x3fff)

136struct atm_trafprm {
137 unsigned char traffic_class; /* traffic class (ATM_UBR, ...) */
138 int max_pcr; /* maximum PCR in cells per second */
139 int pcr; /* desired PCR in cells per second */
140 int min_pcr; /* minimum PCR in cells per second */
141 int max_cdv; /* maximum CDV in microseconds */
142 int max_sdu; /* maximum SDU in bytes */
143 /* extra params for ABR */
144 unsigned int icr; /* Initial Cell Rate (24-bit) */
145 unsigned int tbe; /* Transient Buffer Exposure (24-bit) */
146 unsigned int frtt : 24; /* Fixed Round Trip Time (24-bit) */
147 unsigned int rif : 4; /* Rate Increment Factor (4-bit) */
148 unsigned int rdf : 4; /* Rate Decrease Factor (4-bit) */
149 unsigned int nrm_pres :1; /* nrm present bit */
150 unsigned int trm_pres :1; /* rm present bit */
151 unsigned int adtf_pres :1; /* adtf present bit */
152 unsigned int cdf_pres :1; /* cdf present bit*/
153 unsigned int nrm :3; /* Max # of Cells for each forward RM cell (3-bit) */
154 unsigned int trm :3; /* Time between forward RM cells (3-bit) */
155 unsigned int adtf :10; /* ACR Decrease Time Factor (10-bit) */
156 unsigned int cdf :3; /* Cutoff Decrease Factor (3-bit) */
157 unsigned int spare :9; /* spare bits */
158};
159
160struct atm_qos {
161 struct atm_trafprm txtp; /* parameters in TX direction */
162 struct atm_trafprm rxtp __ATM_API_ALIGN;
163 /* parameters in RX direction */
164 unsigned char aal __ATM_API_ALIGN;
165};

176struct sockaddr_atmpvc {
177 unsigned short sap_family; /* address family, AF_ATMPVC */
178 struct { /* PVC address */
179 short itf; /* ATM interface */
180 short vpi; /* VPI (only 8 bits at UNI) */
181 int vci; /* VCI (only 16 bits at UNI) */
182 } sap_addr __ATM_API_ALIGN; /* PVC address */
183};

206struct sockaddr_atmsvc {
207 unsigned short sas_family; /* address family, AF_ATMSVC */
208 struct { /* SVC address */
209 unsigned char prv[ATM_ESA_LEN];/* private ATM address */
210 char pub[ATM_E164_LEN+1]; /* public address (E.164) */
211 /* unused addresses must be bzero'ed */
212 char lij_type; /* role in LIJ call; one of ATM_LIJ* */
213 __u32 lij_id; /* LIJ call identifier */
214 } sas_addr __ATM_API_ALIGN; /* SVC address */
215};
216
217
218static __inline__ int atmsvc_addr_in_use(struct sockaddr_atmsvc addr)
219{
220 return *addr.sas_addr.prv || *addr.sas_addr.pub;
221}
222
223
224static __inline__ int atmpvc_addr_in_use(struct sockaddr_atmpvc addr)
225{
226 return addr.sap_addr.itf || addr.sap_addr.vpi || addr.sap_addr.vci;
227}
