25#include <linux/cache.h>
26#include <linux/module.h>

31struct net_protocol __rcu *inet_protos[MAX_INET_PROTOS] __read_mostly;
32const struct net_offload __rcu *inet_offloads[MAX_INET_PROTOS] __read_mostly;
33EXPORT_SYMBOL(inet_offloads);
34
35int inet_add_protocol(const struct net_protocol *prot, unsigned char protocol)
36{
37 if (!prot->netns_ok) {
38 pr_err("Protocol %u is not namespace aware, cannot register.\n",
39 protocol);
40 return -EINVAL;
41 }
42
43 return !cmpxchg((const struct net_protocol **)&inet_protos[protocol],
44 NULL, prot) ? 0 : -1;
45}
46EXPORT_SYMBOL(inet_add_protocol);
47
48int inet_add_offload(const struct net_offload *prot, unsigned char protocol)
49{
50 return !cmpxchg((const struct net_offload **)&inet_offloads[protocol],
51 NULL, prot) ? 0 : -1;
52}
53EXPORT_SYMBOL(inet_add_offload);
54
55int inet_del_protocol(const struct net_protocol *prot, unsigned char protocol)
56{
57 int ret;
58
59 ret = (cmpxchg((const struct net_protocol **)&inet_protos[protocol],
60 prot, NULL) == prot) ? 0 : -1;
61
62 synchronize_net();
63
64 return ret;
65}
66EXPORT_SYMBOL(inet_del_protocol);
67
68int inet_del_offload(const struct net_offload *prot, unsigned char protocol)
69{
70 int ret;
71
72 ret = (cmpxchg((const struct net_offload **)&inet_offloads[protocol],
73 prot, NULL) == prot) ? 0 : -1;
74
75 synchronize_net();
76
77 return ret;
78}
79EXPORT_SYMBOL(inet_del_offload);
