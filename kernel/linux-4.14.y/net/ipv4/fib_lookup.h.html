10 struct hlist_node fa_list;

12 u8 fa_tos;
13 u8 fa_type;
14 u8 fa_state;
15 u8 fa_slen;

18 struct rcu_head rcu;
19};
20
21#define FA_S_ACCESSED 0x01
22
23/* Dont write on fa_state unless needed, to keep it shared on all cpus */
24static inline void fib_alias_accessed(struct fib_alias *fa)
25{
26 if (!(fa->fa_state & FA_S_ACCESSED))
27 fa->fa_state |= FA_S_ACCESSED;
28}
