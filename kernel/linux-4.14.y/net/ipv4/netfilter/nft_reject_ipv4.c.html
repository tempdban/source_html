22static void nft_reject_ipv4_eval(const struct nft_expr *expr,
23 struct nft_regs *regs,
24 const struct nft_pktinfo *pkt)
25{
26 struct nft_reject *priv = nft_expr_priv(expr);
27
28 switch (priv->type) {
29 case NFT_REJECT_ICMP_UNREACH:
30 nf_send_unreach(pkt->skb, priv->icmp_code, nft_hook(pkt));
31 break;
32 case NFT_REJECT_TCP_RST:
33 nf_send_reset(nft_net(pkt), pkt->skb, nft_hook(pkt));
34 break;
35 default:
36 break;
37 }
38
39 regs->verdict.code = NF_DROP;
40}
41
42static struct nft_expr_type nft_reject_ipv4_type;
43static const struct nft_expr_ops nft_reject_ipv4_ops = {
44 .type = &nft_reject_ipv4_type,
45 .size = NFT_EXPR_SIZE(sizeof(struct nft_reject)),
46 .eval = nft_reject_ipv4_eval,
47 .init = nft_reject_init,
48 .dump = nft_reject_dump,
49 .validate = nft_reject_validate,
50};
51
52static struct nft_expr_type nft_reject_ipv4_type __read_mostly = {
53 .family = NFPROTO_IPV4,
54 .name = "reject",
55 .ops = &nft_reject_ipv4_ops,
56 .policy = nft_reject_policy,
57 .maxattr = NFTA_REJECT_MAX,
58 .owner = THIS_MODULE,
59};
60
61static int __init nft_reject_ipv4_module_init(void)
62{
63 return nft_register_expr(&nft_reject_ipv4_type);
64}
65
66static void __exit nft_reject_ipv4_module_exit(void)
67{
68 nft_unregister_expr(&nft_reject_ipv4_type);
69}
70
71module_init(nft_reject_ipv4_module_init);
72module_exit(nft_reject_ipv4_module_exit);
