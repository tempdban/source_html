23#define FILTER_VALID_HOOKS ((1 << NF_INET_LOCAL_IN) | \
24 (1 << NF_INET_FORWARD) | \
25 (1 << NF_INET_LOCAL_OUT))
26static int __net_init iptable_filter_table_init(struct net *net);
27
28static const struct xt_table packet_filter = {
29 .name = "filter",
30 .valid_hooks = FILTER_VALID_HOOKS,
31 .me = THIS_MODULE,
32 .af = NFPROTO_IPV4,
33 .priority = NF_IP_PRI_FILTER,
34 .table_init = iptable_filter_table_init,
35};
36
37static unsigned int
38iptable_filter_hook(void *priv, struct sk_buff *skb,
39 const struct nf_hook_state *state)
40{
41 if (state->hook == NF_INET_LOCAL_OUT &&
42 (skb->len < sizeof(struct iphdr) ||
43 ip_hdrlen(skb) < sizeof(struct iphdr)))
44 /* root is playing with raw sockets. */
45 return NF_ACCEPT;
46
47 return ipt_do_table(skb, state, state->net->ipv4.iptable_filter);
48}
49
50static struct nf_hook_ops *filter_ops __read_mostly;
51
52/* Default to forward because I got too much mail already. */
53static bool forward __read_mostly = true;
54module_param(forward, bool, 0000);
55
56static int __net_init iptable_filter_table_init(struct net *net)
57{
58 struct ipt_replace *repl;
59 int err;
60
61 if (net->ipv4.iptable_filter)
62 return 0;
63
64 repl = ipt_alloc_initial_table(&packet_filter);
65 if (repl == NULL)
66 return -ENOMEM;
67 /* Entry 1 is the FORWARD hook */
68 ((struct ipt_standard *)repl->entries)[1].target.verdict =
69 forward ? -NF_ACCEPT - 1 : -NF_DROP - 1;
70
71 err = ipt_register_table(net, &packet_filter, repl, filter_ops,
72 &net->ipv4.iptable_filter);
73 kfree(repl);
74 return err;
75}
76
77static int __net_init iptable_filter_net_init(struct net *net)
78{
79 if (net == &init_net || !forward)
80 return iptable_filter_table_init(net);
81
82 return 0;
83}
84
85static void __net_exit iptable_filter_net_exit(struct net *net)
86{
87 if (!net->ipv4.iptable_filter)
88 return;
89 ipt_unregister_table(net, net->ipv4.iptable_filter, filter_ops);
90 net->ipv4.iptable_filter = NULL;
91}
92
93static struct pernet_operations iptable_filter_net_ops = {
94 .init = iptable_filter_net_init,
95 .exit = iptable_filter_net_exit,
96};
97
98static int __init iptable_filter_init(void)
99{
100 int ret;
101
102 filter_ops = xt_hook_ops_alloc(&packet_filter, iptable_filter_hook);
103 if (IS_ERR(filter_ops))
104 return PTR_ERR(filter_ops);
105
106 ret = register_pernet_subsys(&iptable_filter_net_ops);
107 if (ret < 0)
108 kfree(filter_ops);
109
110 return ret;
111}
112
113static void __exit iptable_filter_fini(void)
114{
115 unregister_pernet_subsys(&iptable_filter_net_ops);
116 kfree(filter_ops);
117}
118
119module_init(iptable_filter_init);
120module_exit(iptable_filter_fini);
