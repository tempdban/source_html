24static unsigned int nf_route_table_hook(void *priv,
25 struct sk_buff *skb,
26 const struct nf_hook_state *state)
27{
28 unsigned int ret;
29 struct nft_pktinfo pkt;
30 u32 mark;
31 __be32 saddr, daddr;
32 u_int8_t tos;
33 const struct iphdr *iph;
34 int err;
35
36 /* root is playing with raw sockets. */
37 if (skb->len < sizeof(struct iphdr) ||
38 ip_hdrlen(skb) < sizeof(struct iphdr))
39 return NF_ACCEPT;
40
41 nft_set_pktinfo_ipv4(&pkt, skb, state);
42
43 mark = skb->mark;
44 iph = ip_hdr(skb);
45 saddr = iph->saddr;
46 daddr = iph->daddr;
47 tos = iph->tos;
48
49 ret = nft_do_chain(&pkt, priv);
50 if (ret != NF_DROP && ret != NF_STOLEN) {
51 iph = ip_hdr(skb);
52
53 if (iph->saddr != saddr ||
54 iph->daddr != daddr ||
55 skb->mark != mark ||
56 iph->tos != tos) {
57 err = ip_route_me_harder(state->net, skb, RTN_UNSPEC);
58 if (err < 0)
59 ret = NF_DROP_ERR(err);
60 }
61 }
62 return ret;
63}

65static const struct nf_chain_type nft_chain_route_ipv4 = {
66 .name = "route",
67 .type = NFT_CHAIN_T_ROUTE,
68 .family = NFPROTO_IPV4,
69 .owner = THIS_MODULE,
70 .hook_mask = (1 << NF_INET_LOCAL_OUT),
71 .hooks = {
72 [NF_INET_LOCAL_OUT] = nf_route_table_hook,
73 },
74};
75
76static int __init nft_chain_route_init(void)
77{
78 return nft_register_chain_type(&nft_chain_route_ipv4);
79}
80
81static void __exit nft_chain_route_exit(void)
82{
83 nft_unregister_chain_type(&nft_chain_route_ipv4);
84}
85
86module_init(nft_chain_route_init);
87module_exit(nft_chain_route_exit);
