27static inline int
28gnet_stats_copy(struct gnet_dump *d, int type, void *buf, int size, int padattr)
29{
30 if (nla_put_64bit(d->skb, type, size, buf, padattr))
31 goto nla_put_failure;
32 return 0;
33
34nla_put_failure:
35 if (d->lock)
36 spin_unlock_bh(d->lock);
37 kfree(d->xstats);
38 d->xstats = NULL;
39 d->xstats_len = 0;
40 return -1;
41}

43/**
44 * gnet_stats_start_copy_compat - start dumping procedure in compatibility mode
45 * @skb: socket buffer to put statistics TLVs into
46 * @type: TLV type for top level statistic TLV
47 * @tc_stats_type: TLV type for backward compatibility struct tc_stats TLV
48 * @xstats_type: TLV type for backward compatibility xstats TLV
49 * @lock: statistics lock
50 * @d: dumping handle
51 * @padattr: padding attribute
52 *
53 * Initializes the dumping handle, grabs the statistic lock and appends
54 * an empty TLV header to the socket buffer for use a container for all
55 * other statistic TLVS.
56 *
57 * The dumping handle is marked to be in backward compatibility mode telling
58 * all gnet_stats_copy_XXX() functions to fill a local copy of struct tc_stats.
59 *
60 * Returns 0 on success or -1 if the room in the socket buffer was not sufficient.
61 */
62int
63gnet_stats_start_copy_compat(struct sk_buff *skb, int type, int tc_stats_type,
64 int xstats_type, spinlock_t *lock,
65 struct gnet_dump *d, int padattr)
66 __acquires(lock)
67{
68 memset(d, 0, sizeof(*d));
69
70 if (type)
71 d->tail = (struct nlattr *)skb_tail_pointer(skb);
72 d->skb = skb;
73 d->compat_tc_stats = tc_stats_type;
74 d->compat_xstats = xstats_type;
75 d->padattr = padattr;
76 if (lock) {
77 d->lock = lock;
78 spin_lock_bh(lock);
79 }
80 if (d->tail) {
81 int ret = gnet_stats_copy(d, type, NULL, 0, padattr);
82
83 /* The initial attribute added in gnet_stats_copy() may be
84 * preceded by a padding attribute, in which case d->tail will
85 * end up pointing at the padding instead of the real attribute.
86 * Fix this so gnet_stats_finish_copy() adjusts the length of
87 * the right attribute.
88 */
89 if (ret == 0 && d->tail->nla_type == padattr)
90 d->tail = (struct nlattr *)((char *)d->tail +
91 NLA_ALIGN(d->tail->nla_len));
92 return ret;
93 }
94
95 return 0;
96}
97EXPORT_SYMBOL(gnet_stats_start_copy_compat);

99/**
100 * gnet_stats_start_copy - start dumping procedure in compatibility mode
101 * @skb: socket buffer to put statistics TLVs into
102 * @type: TLV type for top level statistic TLV
103 * @lock: statistics lock
104 * @d: dumping handle
105 * @padattr: padding attribute
106 *
107 * Initializes the dumping handle, grabs the statistic lock and appends
108 * an empty TLV header to the socket buffer for use a container for all
109 * other statistic TLVS.
110 *
111 * Returns 0 on success or -1 if the room in the socket buffer was not sufficient.
112 */
113int
114gnet_stats_start_copy(struct sk_buff *skb, int type, spinlock_t *lock,
115 struct gnet_dump *d, int padattr)
116{
117 return gnet_stats_start_copy_compat(skb, type, 0, 0, lock, d, padattr);
118}
119EXPORT_SYMBOL(gnet_stats_start_copy);

121static void
122__gnet_stats_copy_basic_cpu(struct gnet_stats_basic_packed *bstats,
123 struct gnet_stats_basic_cpu __percpu *cpu)
124{
125 int i;
126
127 for_each_possible_cpu(i) {
128 struct gnet_stats_basic_cpu *bcpu = per_cpu_ptr(cpu, i);
129 unsigned int start;
130 u64 bytes;
131 u32 packets;
132
133 do {
134 start = u64_stats_fetch_begin_irq(&bcpu->syncp);
135 bytes = bcpu->bstats.bytes;
136 packets = bcpu->bstats.packets;
137 } while (u64_stats_fetch_retry_irq(&bcpu->syncp, start));
138
139 bstats->bytes += bytes;
140 bstats->packets += packets;
141 }
142}

144void
145__gnet_stats_copy_basic(const seqcount_t *running,
146 struct gnet_stats_basic_packed *bstats,
147 struct gnet_stats_basic_cpu __percpu *cpu,
148 struct gnet_stats_basic_packed *b)
149{
150 unsigned int seq;
151
152 if (cpu) {
153 __gnet_stats_copy_basic_cpu(bstats, cpu);
154 return;
155 }
156 do {
157 if (running)
158 seq = read_seqcount_begin(running);
159 bstats->bytes = b->bytes;
160 bstats->packets = b->packets;
161 } while (running && read_seqcount_retry(running, seq));
162}
163EXPORT_SYMBOL(__gnet_stats_copy_basic);

165/**
166 * gnet_stats_copy_basic - copy basic statistics into statistic TLV
167 * @running: seqcount_t pointer
168 * @d: dumping handle
169 * @cpu: copy statistic per cpu
170 * @b: basic statistics
171 *
172 * Appends the basic statistics to the top level TLV created by
173 * gnet_stats_start_copy().
174 *
175 * Returns 0 on success or -1 with the statistic lock released
176 * if the room in the socket buffer was not sufficient.
177 */
178int
179gnet_stats_copy_basic(const seqcount_t *running,
180 struct gnet_dump *d,
181 struct gnet_stats_basic_cpu __percpu *cpu,
182 struct gnet_stats_basic_packed *b)
183{
184 struct gnet_stats_basic_packed bstats = {0};
185
186 __gnet_stats_copy_basic(running, &bstats, cpu, b);
187
188 if (d->compat_tc_stats) {
189 d->tc_stats.bytes = bstats.bytes;
190 d->tc_stats.packets = bstats.packets;
191 }
192
193 if (d->tail) {
194 struct gnet_stats_basic sb;
195
196 memset(&sb, 0, sizeof(sb));
197 sb.bytes = bstats.bytes;
198 sb.packets = bstats.packets;
199 return gnet_stats_copy(d, TCA_STATS_BASIC, &sb, sizeof(sb),
200 TCA_STATS_PAD);
201 }
202 return 0;
203}
204EXPORT_SYMBOL(gnet_stats_copy_basic);

206/**
207 * gnet_stats_copy_rate_est - copy rate estimator statistics into statistics TLV
208 * @d: dumping handle
209 * @rate_est: rate estimator
210 *
211 * Appends the rate estimator statistics to the top level TLV created by
212 * gnet_stats_start_copy().
213 *
214 * Returns 0 on success or -1 with the statistic lock released
215 * if the room in the socket buffer was not sufficient.
216 */
217int
218gnet_stats_copy_rate_est(struct gnet_dump *d,
219 struct net_rate_estimator __rcu **rate_est)
220{
221 struct gnet_stats_rate_est64 sample;
222 struct gnet_stats_rate_est est;
223 int res;
224
225 if (!gen_estimator_read(rate_est, &sample))
226 return 0;
227 est.bps = min_t(u64, UINT_MAX, sample.bps);
228 /* we have some time before reaching 2^32 packets per second */
229 est.pps = sample.pps;
230
231 if (d->compat_tc_stats) {
232 d->tc_stats.bps = est.bps;
233 d->tc_stats.pps = est.pps;
234 }
235
236 if (d->tail) {
237 res = gnet_stats_copy(d, TCA_STATS_RATE_EST, &est, sizeof(est),
238 TCA_STATS_PAD);
239 if (res < 0 || est.bps == sample.bps)
240 return res;
241 /* emit 64bit stats only if needed */
242 return gnet_stats_copy(d, TCA_STATS_RATE_EST64, &sample,
243 sizeof(sample), TCA_STATS_PAD);
244 }
245
246 return 0;
247}
248EXPORT_SYMBOL(gnet_stats_copy_rate_est);

250static void
251__gnet_stats_copy_queue_cpu(struct gnet_stats_queue *qstats,
252 const struct gnet_stats_queue __percpu *q)
253{
254 int i;
255
256 for_each_possible_cpu(i) {
257 const struct gnet_stats_queue *qcpu = per_cpu_ptr(q, i);
258
259 qstats->qlen = 0;
260 qstats->backlog += qcpu->backlog;
261 qstats->drops += qcpu->drops;
262 qstats->requeues += qcpu->requeues;
263 qstats->overlimits += qcpu->overlimits;
264 }
265}

267static void __gnet_stats_copy_queue(struct gnet_stats_queue *qstats,
268 const struct gnet_stats_queue __percpu *cpu,
269 const struct gnet_stats_queue *q,
270 __u32 qlen)
271{
272 if (cpu) {
273 __gnet_stats_copy_queue_cpu(qstats, cpu);
274 } else {
275 qstats->qlen = q->qlen;
276 qstats->backlog = q->backlog;
277 qstats->drops = q->drops;
278 qstats->requeues = q->requeues;
279 qstats->overlimits = q->overlimits;
280 }
281
282 qstats->qlen = qlen;
283}

285/**
286 * gnet_stats_copy_queue - copy queue statistics into statistics TLV
287 * @d: dumping handle
288 * @cpu_q: per cpu queue statistics
289 * @q: queue statistics
290 * @qlen: queue length statistics
291 *
292 * Appends the queue statistics to the top level TLV created by
293 * gnet_stats_start_copy(). Using per cpu queue statistics if
294 * they are available.
295 *
296 * Returns 0 on success or -1 with the statistic lock released
297 * if the room in the socket buffer was not sufficient.
298 */
299int
300gnet_stats_copy_queue(struct gnet_dump *d,
301 struct gnet_stats_queue __percpu *cpu_q,
302 struct gnet_stats_queue *q, __u32 qlen)
303{
304 struct gnet_stats_queue qstats = {0};
305
306 __gnet_stats_copy_queue(&qstats, cpu_q, q, qlen);
307
308 if (d->compat_tc_stats) {
309 d->tc_stats.drops = qstats.drops;
310 d->tc_stats.qlen = qstats.qlen;
311 d->tc_stats.backlog = qstats.backlog;
312 d->tc_stats.overlimits = qstats.overlimits;
313 }
314
315 if (d->tail)
316 return gnet_stats_copy(d, TCA_STATS_QUEUE,
317 &qstats, sizeof(qstats),
318 TCA_STATS_PAD);
319
320 return 0;
321}
322EXPORT_SYMBOL(gnet_stats_copy_queue);

324/**
325 * gnet_stats_copy_app - copy application specific statistics into statistics TLV
326 * @d: dumping handle
327 * @st: application specific statistics data
328 * @len: length of data
329 *
330 * Appends the application specific statistics to the top level TLV created by
331 * gnet_stats_start_copy() and remembers the data for XSTATS if the dumping
332 * handle is in backward compatibility mode.
333 *
334 * Returns 0 on success or -1 with the statistic lock released
335 * if the room in the socket buffer was not sufficient.
336 */
337int
338gnet_stats_copy_app(struct gnet_dump *d, void *st, int len)
339{
340 if (d->compat_xstats) {
341 d->xstats = kmemdup(st, len, GFP_ATOMIC);
342 if (!d->xstats)
343 goto err_out;
344 d->xstats_len = len;
345 }
346
347 if (d->tail)
348 return gnet_stats_copy(d, TCA_STATS_APP, st, len,
349 TCA_STATS_PAD);
350
351 return 0;
352
353err_out:
354 if (d->lock)
355 spin_unlock_bh(d->lock);
356 d->xstats_len = 0;
357 return -1;
358}
359EXPORT_SYMBOL(gnet_stats_copy_app);

361/**
362 * gnet_stats_finish_copy - finish dumping procedure
363 * @d: dumping handle
364 *
365 * Corrects the length of the top level TLV to include all TLVs added
366 * by gnet_stats_copy_XXX() calls. Adds the backward compatibility TLVs
367 * if gnet_stats_start_copy_compat() was used and releases the statistics
368 * lock.
369 *
370 * Returns 0 on success or -1 with the statistic lock released
371 * if the room in the socket buffer was not sufficient.
372 */
373int
374gnet_stats_finish_copy(struct gnet_dump *d)
375{
376 if (d->tail)
377 d->tail->nla_len = skb_tail_pointer(d->skb) - (u8 *)d->tail;
378
379 if (d->compat_tc_stats)
380 if (gnet_stats_copy(d, d->compat_tc_stats, &d->tc_stats,
381 sizeof(d->tc_stats), d->padattr) < 0)
382 return -1;
383
384 if (d->compat_xstats && d->xstats) {
385 if (gnet_stats_copy(d, d->compat_xstats, d->xstats,
386 d->xstats_len, d->padattr) < 0)
387 return -1;
388 }
389
390 if (d->lock)
391 spin_unlock_bh(d->lock);
392 kfree(d->xstats);
393 d->xstats = NULL;
394 d->xstats_len = 0;
395 return 0;
396}
397EXPORT_SYMBOL(gnet_stats_finish_copy);
