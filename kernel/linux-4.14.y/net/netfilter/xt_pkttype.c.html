19MODULE_LICENSE("GPL");
20MODULE_AUTHOR("Michal Ludvig <michal@logix.cz>");
21MODULE_DESCRIPTION("Xtables: link layer packet type match");
22MODULE_ALIAS("ipt_pkttype");
23MODULE_ALIAS("ip6t_pkttype");
24
25static bool
26pkttype_mt(const struct sk_buff *skb, struct xt_action_param *par)
27{
28 const struct xt_pkttype_info *info = par->matchinfo;
29 u_int8_t type;
30
31 if (skb->pkt_type != PACKET_LOOPBACK)
32 type = skb->pkt_type;
33 else if (xt_family(par) == NFPROTO_IPV4 &&
34 ipv4_is_multicast(ip_hdr(skb)->daddr))
35 type = PACKET_MULTICAST;
36 else if (xt_family(par) == NFPROTO_IPV6)
37 type = PACKET_MULTICAST;
38 else
39 type = PACKET_BROADCAST;
40
41 return (type == info->pkttype) ^ info->invert;
42}
43
44static struct xt_match pkttype_mt_reg __read_mostly = {
45 .name = "pkttype",
46 .revision = 0,
47 .family = NFPROTO_UNSPEC,
48 .match = pkttype_mt,
49 .matchsize = sizeof(struct xt_pkttype_info),
50 .me = THIS_MODULE,
51};
52
53static int __init pkttype_mt_init(void)
54{
55 return xt_register_match(&pkttype_mt_reg);
56}
57
58static void __exit pkttype_mt_exit(void)
59{
60 xt_unregister_match(&pkttype_mt_reg);
61}
62
63module_init(pkttype_mt_init);
64module_exit(pkttype_mt_exit);
