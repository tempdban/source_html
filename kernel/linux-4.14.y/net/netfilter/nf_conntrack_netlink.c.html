63static int ctnetlink_dump_tuples_proto(struct sk_buff *skb,
64 const struct nf_conntrack_tuple *tuple,
65 const struct nf_conntrack_l4proto *l4proto)
66{
67 int ret = 0;
68 struct nlattr *nest_parms;
69
70 nest_parms = nla_nest_start(skb, CTA_TUPLE_PROTO | NLA_F_NESTED);
71 if (!nest_parms)
72 goto nla_put_failure;
73 if (nla_put_u8(skb, CTA_PROTO_NUM, tuple->dst.protonum))
74 goto nla_put_failure;
75
76 if (likely(l4proto->tuple_to_nlattr))
77 ret = l4proto->tuple_to_nlattr(skb, tuple);
78
79 nla_nest_end(skb, nest_parms);
80
81 return ret;
82
83nla_put_failure:
84 return -1;
85}

109static int ctnetlink_dump_tuples(struct sk_buff *skb,
110 const struct nf_conntrack_tuple *tuple)
111{
112 const struct nf_conntrack_l3proto *l3proto;
113 const struct nf_conntrack_l4proto *l4proto;
114 int ret;
115
116 rcu_read_lock();
117 l3proto = __nf_ct_l3proto_find(tuple->src.l3num);
118 ret = ctnetlink_dump_tuples_ip(skb, tuple, l3proto);
119
120 if (ret >= 0) {
121 l4proto = __nf_ct_l4proto_find(tuple->src.l3num,
122 tuple->dst.protonum);
123 ret = ctnetlink_dump_tuples_proto(skb, tuple, l4proto);
124 }
125 rcu_read_unlock();
126 return ret;
127}

188static int ctnetlink_dump_helpinfo(struct sk_buff *skb,
189 const struct nf_conn *ct)
190{
191 struct nlattr *nest_helper;
192 const struct nf_conn_help *help = nfct_help(ct);
193 struct nf_conntrack_helper *helper;
194
195 if (!help)
196 return 0;
197
198 helper = rcu_dereference(help->helper);
199 if (!helper)
200 goto out;
201
202 nest_helper = nla_nest_start(skb, CTA_HELP | NLA_F_NESTED);
203 if (!nest_helper)
204 goto nla_put_failure;
205 if (nla_put_string(skb, CTA_HELP_NAME, helper->name))
206 goto nla_put_failure;
207
208 if (helper->to_nlattr)
209 helper->to_nlattr(skb, ct);
210
211 nla_nest_end(skb, nest_helper);
212out:
213 return 0;
214
215nla_put_failure:
216 return -1;
217}

219static int
220dump_counters(struct sk_buff *skb, struct nf_conn_acct *acct,
221 enum ip_conntrack_dir dir, int type)
222{
223 enum ctattr_type attr = dir ? CTA_COUNTERS_REPLY: CTA_COUNTERS_ORIG;
224 struct nf_conn_counter *counter = acct->counter;
225 struct nlattr *nest_count;
226 u64 pkts, bytes;
227
228 if (type == IPCTNL_MSG_CT_GET_CTRZERO) {
229 pkts = atomic64_xchg(&counter[dir].packets, 0);
230 bytes = atomic64_xchg(&counter[dir].bytes, 0);
231 } else {
232 pkts = atomic64_read(&counter[dir].packets);
233 bytes = atomic64_read(&counter[dir].bytes);
234 }
235
236 nest_count = nla_nest_start(skb, attr | NLA_F_NESTED);
237 if (!nest_count)
238 goto nla_put_failure;
239
240 if (nla_put_be64(skb, CTA_COUNTERS_PACKETS, cpu_to_be64(pkts),
241 CTA_COUNTERS_PAD) ||
242 nla_put_be64(skb, CTA_COUNTERS_BYTES, cpu_to_be64(bytes),
243 CTA_COUNTERS_PAD))
244 goto nla_put_failure;
245
246 nla_nest_end(skb, nest_count);
247
248 return 0;
249
250nla_put_failure:
251 return -1;
252}

289 goto nla_put_failure;
290 nla_nest_end(skb, nest_count);
291
292 return 0;
293
294nla_put_failure:
295 return -1;

422static int ctnetlink_dump_ct_seq_adj(struct sk_buff *skb, struct nf_conn *ct)
423{
424 struct nf_conn_seqadj *seqadj = nfct_seqadj(ct);
425 struct nf_ct_seqadj *seq;
426
427 if (!(ct->status & IPS_SEQ_ADJUST) || !seqadj)
428 return 0;
429
430 spin_lock_bh(&ct->lock);
431 seq = &seqadj->seq[IP_CT_DIR_ORIGINAL];
432 if (dump_ct_seq_adj(skb, seq, CTA_SEQ_ADJ_ORIG) == -1)
433 goto err;
434
435 seq = &seqadj->seq[IP_CT_DIR_REPLY];
436 if (dump_ct_seq_adj(skb, seq, CTA_SEQ_ADJ_REPLY) == -1)
437 goto err;
438
439 spin_unlock_bh(&ct->lock);
440 return 0;
441err:
442 spin_unlock_bh(&ct->lock);
443 return -1;
444}

466static int
467ctnetlink_fill_info(struct sk_buff *skb, u32 portid, u32 seq, u32 type,
468 struct nf_conn *ct)
469{
470 const struct nf_conntrack_zone *zone;
471 struct nlmsghdr *nlh;
472 struct nfgenmsg *nfmsg;
473 struct nlattr *nest_parms;
474 unsigned int flags = portid ? NLM_F_MULTI : 0, event;
475
476 event = nfnl_msg_type(NFNL_SUBSYS_CTNETLINK, IPCTNL_MSG_CT_NEW);
477 nlh = nlmsg_put(skb, portid, seq, event, sizeof(*nfmsg), flags);
478 if (nlh == NULL)
479 goto nlmsg_failure;
480
481 nfmsg = nlmsg_data(nlh);
482 nfmsg->nfgen_family = nf_ct_l3num(ct);
483 nfmsg->version = NFNETLINK_V0;
484 nfmsg->res_id = 0;
485
486 zone = nf_ct_zone(ct);
487
488 nest_parms = nla_nest_start(skb, CTA_TUPLE_ORIG | NLA_F_NESTED);
489 if (!nest_parms)
490 goto nla_put_failure;
491 if (ctnetlink_dump_tuples(skb, nf_ct_tuple(ct, IP_CT_DIR_ORIGINAL)) < 0)
492 goto nla_put_failure;
493 if (ctnetlink_dump_zone_id(skb, CTA_TUPLE_ZONE, zone,
494 NF_CT_ZONE_DIR_ORIG) < 0)
495 goto nla_put_failure;
496 nla_nest_end(skb, nest_parms);
497
498 nest_parms = nla_nest_start(skb, CTA_TUPLE_REPLY | NLA_F_NESTED);
499 if (!nest_parms)
500 goto nla_put_failure;
501 if (ctnetlink_dump_tuples(skb, nf_ct_tuple(ct, IP_CT_DIR_REPLY)) < 0)
502 goto nla_put_failure;
503 if (ctnetlink_dump_zone_id(skb, CTA_TUPLE_ZONE, zone,
504 NF_CT_ZONE_DIR_REPL) < 0)
505 goto nla_put_failure;
506 nla_nest_end(skb, nest_parms);
507
508 if (ctnetlink_dump_zone_id(skb, CTA_ZONE, zone,
509 NF_CT_DEFAULT_ZONE_DIR) < 0)
510 goto nla_put_failure;
511
512 if (ctnetlink_dump_status(skb, ct) < 0 ||
513 ctnetlink_dump_timeout(skb, ct) < 0 ||
514 ctnetlink_dump_acct(skb, ct, type) < 0 ||
515 ctnetlink_dump_timestamp(skb, ct) < 0 ||
516 ctnetlink_dump_protoinfo(skb, ct) < 0 ||
517 ctnetlink_dump_helpinfo(skb, ct) < 0 ||
518 ctnetlink_dump_mark(skb, ct) < 0 ||
519 ctnetlink_dump_secctx(skb, ct) < 0 ||
520 ctnetlink_dump_labels(skb, ct) < 0 ||
521 ctnetlink_dump_id(skb, ct) < 0 ||
522 ctnetlink_dump_use(skb, ct) < 0 ||
523 ctnetlink_dump_master(skb, ct) < 0 ||
524 ctnetlink_dump_ct_seq_adj(skb, ct) < 0)
525 goto nla_put_failure;
526
527 nlmsg_end(skb, nlh);
528 return skb->len;
529
530nlmsg_failure:
531nla_put_failure:
532 nlmsg_cancel(skb, nlh);
533 return -1;
534}

536static inline size_t ctnetlink_proto_size(const struct nf_conn *ct)
537{
538 const struct nf_conntrack_l3proto *l3proto;
539 const struct nf_conntrack_l4proto *l4proto;
540 size_t len;
541
542 l3proto = __nf_ct_l3proto_find(nf_ct_l3num(ct));
543 len = l3proto->nla_size;
544 len *= 3u; /* ORIG, REPLY, MASTER */
545
546 l4proto = __nf_ct_l4proto_find(nf_ct_l3num(ct), nf_ct_protonum(ct));
547 len += l4proto->nla_size;
548
549 return len;
550}

621static int
622ctnetlink_conntrack_event(unsigned int events, struct nf_ct_event *item)
623{
624 const struct nf_conntrack_zone *zone;
625 struct net *net;
626 struct nlmsghdr *nlh;
627 struct nfgenmsg *nfmsg;
628 struct nlattr *nest_parms;
629 struct nf_conn *ct = item->ct;
630 struct sk_buff *skb;
631 unsigned int type;
632 unsigned int flags = 0, group;
633 int err;
634
635 if (events & (1 << IPCT_DESTROY)) {
636 type = IPCTNL_MSG_CT_DELETE;
637 group = NFNLGRP_CONNTRACK_DESTROY;
638 } else if (events & ((1 << IPCT_NEW) | (1 << IPCT_RELATED))) {
639 type = IPCTNL_MSG_CT_NEW;
640 flags = NLM_F_CREATE|NLM_F_EXCL;
641 group = NFNLGRP_CONNTRACK_NEW;
642 } else if (events) {
643 type = IPCTNL_MSG_CT_NEW;
644 group = NFNLGRP_CONNTRACK_UPDATE;
645 } else
646 return 0;
647
648 net = nf_ct_net(ct);
649 if (!item->report && !nfnetlink_has_listeners(net, group))
650 return 0;
651
652 skb = nlmsg_new(ctnetlink_nlmsg_size(ct), GFP_ATOMIC);
653 if (skb == NULL)
654 goto errout;
655
656 type = nfnl_msg_type(NFNL_SUBSYS_CTNETLINK, type);
657 nlh = nlmsg_put(skb, item->portid, 0, type, sizeof(*nfmsg), flags);
658 if (nlh == NULL)
659 goto nlmsg_failure;
660
661 nfmsg = nlmsg_data(nlh);
662 nfmsg->nfgen_family = nf_ct_l3num(ct);
663 nfmsg->version = NFNETLINK_V0;
664 nfmsg->res_id = 0;
665
666 zone = nf_ct_zone(ct);
667
668 nest_parms = nla_nest_start(skb, CTA_TUPLE_ORIG | NLA_F_NESTED);
669 if (!nest_parms)
670 goto nla_put_failure;
671 if (ctnetlink_dump_tuples(skb, nf_ct_tuple(ct, IP_CT_DIR_ORIGINAL)) < 0)
672 goto nla_put_failure;
673 if (ctnetlink_dump_zone_id(skb, CTA_TUPLE_ZONE, zone,
674 NF_CT_ZONE_DIR_ORIG) < 0)
675 goto nla_put_failure;
676 nla_nest_end(skb, nest_parms);
677
678 nest_parms = nla_nest_start(skb, CTA_TUPLE_REPLY | NLA_F_NESTED);
679 if (!nest_parms)
680 goto nla_put_failure;
681 if (ctnetlink_dump_tuples(skb, nf_ct_tuple(ct, IP_CT_DIR_REPLY)) < 0)
682 goto nla_put_failure;
683 if (ctnetlink_dump_zone_id(skb, CTA_TUPLE_ZONE, zone,
684 NF_CT_ZONE_DIR_REPL) < 0)
685 goto nla_put_failure;
686 nla_nest_end(skb, nest_parms);
687
688 if (ctnetlink_dump_zone_id(skb, CTA_ZONE, zone,
689 NF_CT_DEFAULT_ZONE_DIR) < 0)
690 goto nla_put_failure;
691
692 if (ctnetlink_dump_id(skb, ct) < 0)
693 goto nla_put_failure;
694
695 if (ctnetlink_dump_status(skb, ct) < 0)
696 goto nla_put_failure;
697
698 if (events & (1 << IPCT_DESTROY)) {
699 if (ctnetlink_dump_acct(skb, ct, type) < 0 ||
700 ctnetlink_dump_timestamp(skb, ct) < 0)
701 goto nla_put_failure;
702 } else {
703 if (ctnetlink_dump_timeout(skb, ct) < 0)
704 goto nla_put_failure;
705
706 if (events & (1 << IPCT_PROTOINFO)
707 && ctnetlink_dump_protoinfo(skb, ct) < 0)
708 goto nla_put_failure;
709
710 if ((events & (1 << IPCT_HELPER) || nfct_help(ct))
711 && ctnetlink_dump_helpinfo(skb, ct) < 0)
712 goto nla_put_failure;
713
714#ifdef CONFIG_NF_CONNTRACK_SECMARK
715 if ((events & (1 << IPCT_SECMARK) || ct->secmark)
716 && ctnetlink_dump_secctx(skb, ct) < 0)
717 goto nla_put_failure;
718#endif
719 if (events & (1 << IPCT_LABEL) &&
720 ctnetlink_dump_labels(skb, ct) < 0)
721 goto nla_put_failure;
722
723 if (events & (1 << IPCT_RELATED) &&
724 ctnetlink_dump_master(skb, ct) < 0)
725 goto nla_put_failure;
726
727 if (events & (1 << IPCT_SEQADJ) &&
728 ctnetlink_dump_ct_seq_adj(skb, ct) < 0)
729 goto nla_put_failure;
730 }
731
732#ifdef CONFIG_NF_CONNTRACK_MARK
733 if ((events & (1 << IPCT_MARK) || ct->mark)
734 && ctnetlink_dump_mark(skb, ct) < 0)
735 goto nla_put_failure;
736#endif
737 nlmsg_end(skb, nlh);
738 err = nfnetlink_send(skb, net, item->portid, group, item->report,
739 GFP_ATOMIC);
740 if (err == -ENOBUFS || err == -EAGAIN)
741 return -ENOBUFS;
742
743 return 0;
744
745nla_put_failure:
746 nlmsg_cancel(skb, nlh);
747nlmsg_failure:
748 kfree_skb(skb);
749errout:
750 if (nfnetlink_set_err(net, 0, group, -ENOBUFS) > 0)
751 return -ENOBUFS;
752
753 return 0;
754}
755#endif /* CONFIG_NF_CONNTRACK_EVENTS */

806static int
807ctnetlink_dump_table(struct sk_buff *skb, struct netlink_callback *cb)
808{
809 struct net *net = sock_net(skb->sk);
810 struct nf_conn *ct, *last;
811 struct nf_conntrack_tuple_hash *h;
812 struct hlist_nulls_node *n;
813 struct nfgenmsg *nfmsg = nlmsg_data(cb->nlh);
814 u_int8_t l3proto = nfmsg->nfgen_family;
815 struct nf_conn *nf_ct_evict[8];
816 int res, i;
817 spinlock_t *lockp;
818
819 last = (struct nf_conn *)cb->args[1];
820 i = 0;
821
822 local_bh_disable();
823 for (; cb->args[0] < nf_conntrack_htable_size; cb->args[0]++) {
824restart:
825 while (i) {
826 i--;
827 if (nf_ct_should_gc(nf_ct_evict[i]))
828 nf_ct_kill(nf_ct_evict[i]);
829 nf_ct_put(nf_ct_evict[i]);
830 }
831
832 lockp = &nf_conntrack_locks[cb->args[0] % CONNTRACK_LOCKS];
833 nf_conntrack_lock(lockp);
834 if (cb->args[0] >= nf_conntrack_htable_size) {
835 spin_unlock(lockp);
836 goto out;
837 }
838 hlist_nulls_for_each_entry(h, n, &nf_conntrack_hash[cb->args[0]],
839 hnnode) {
840 if (NF_CT_DIRECTION(h) != IP_CT_DIR_ORIGINAL)
841 continue;
842 ct = nf_ct_tuplehash_to_ctrack(h);
843 if (nf_ct_is_expired(ct)) {
844 if (i < ARRAY_SIZE(nf_ct_evict) &&
845 atomic_inc_not_zero(&ct->ct_general.use))
846 nf_ct_evict[i++] = ct;
847 continue;
848 }
849
850 if (!net_eq(net, nf_ct_net(ct)))
851 continue;
852
853 /* Dump entries of a given L3 protocol number.
854 * If it is not specified, ie. l3proto == 0,
855 * then dump everything. */
856 if (l3proto && nf_ct_l3num(ct) != l3proto)
857 continue;
858 if (cb->args[1]) {
859 if (ct != last)
860 continue;
861 cb->args[1] = 0;
862 }
863 if (!ctnetlink_filter_match(ct, cb->data))
864 continue;
865
866 rcu_read_lock();
867 res =
868 ctnetlink_fill_info(skb, NETLINK_CB(cb->skb).portid,
869 cb->nlh->nlmsg_seq,
870 NFNL_MSG_TYPE(cb->nlh->nlmsg_type),
871 ct);
872 rcu_read_unlock();
873 if (res < 0) {
874 nf_conntrack_get(&ct->ct_general);
875 cb->args[1] = (unsigned long)ct;
876 spin_unlock(lockp);
877 goto out;
878 }
879 }
880 spin_unlock(lockp);
881 if (cb->args[1]) {
882 cb->args[1] = 0;
883 goto restart;
884 }
885 }
886out:
887 local_bh_enable();
888 if (last) {
889 /* nf ct hash resize happened, now clear the leftover. */
890 if ((struct nf_conn *)cb->args[1] == last)
891 cb->args[1] = 0;
892
893 nf_ct_put(last);
894 }
895
896 while (i) {
897 i--;
898 if (nf_ct_should_gc(nf_ct_evict[i]))
899 nf_ct_kill(nf_ct_evict[i]);
900 nf_ct_put(nf_ct_evict[i]);
901 }
902
903 return skb->len;
904}

1180static int ctnetlink_get_conntrack(struct net *net, struct sock *ctnl,
1181 struct sk_buff *skb,
1182 const struct nlmsghdr *nlh,
1183 const struct nlattr * const cda[],
1184 struct netlink_ext_ack *extack)
1185{
1186 struct nf_conntrack_tuple_hash *h;
1187 struct nf_conntrack_tuple tuple;
1188 struct nf_conn *ct;
1189 struct sk_buff *skb2 = NULL;
1190 struct nfgenmsg *nfmsg = nlmsg_data(nlh);
1191 u_int8_t u3 = nfmsg->nfgen_family;
1192 struct nf_conntrack_zone zone;
1193 int err;
1194
1195 if (nlh->nlmsg_flags & NLM_F_DUMP) {
1196 struct netlink_dump_control c = {
1197 .dump = ctnetlink_dump_table,
1198 .done = ctnetlink_done,
1199 };
1200
1201 if (cda[CTA_MARK] && cda[CTA_MARK_MASK]) {
1202 struct ctnetlink_filter *filter;
1203
1204 filter = ctnetlink_alloc_filter(cda);
1205 if (IS_ERR(filter))
1206 return PTR_ERR(filter);
1207
1208 c.data = filter;
1209 }
1210 return netlink_dump_start(ctnl, skb, nlh, &c);
1211 }
1212
1213 err = ctnetlink_parse_zone(cda[CTA_ZONE], &zone);
1214 if (err < 0)
1215 return err;
1216
1217 if (cda[CTA_TUPLE_ORIG])
1218 err = ctnetlink_parse_tuple(cda, &tuple, CTA_TUPLE_ORIG,
1219 u3, &zone);
1220 else if (cda[CTA_TUPLE_REPLY])
1221 err = ctnetlink_parse_tuple(cda, &tuple, CTA_TUPLE_REPLY,
1222 u3, &zone);
1223 else
1224 return -EINVAL;
1225
1226 if (err < 0)
1227 return err;
1228
1229 h = nf_conntrack_find_get(net, &zone, &tuple);
1230 if (!h)
1231 return -ENOENT;
1232
1233 ct = nf_ct_tuplehash_to_ctrack(h);
1234
1235 err = -ENOMEM;
1236 skb2 = nlmsg_new(NLMSG_DEFAULT_SIZE, GFP_KERNEL);
1237 if (skb2 == NULL) {
1238 nf_ct_put(ct);
1239 return -ENOMEM;
1240 }
1241
1242 rcu_read_lock();
1243 err = ctnetlink_fill_info(skb2, NETLINK_CB(skb).portid, nlh->nlmsg_seq,
1244 NFNL_MSG_TYPE(nlh->nlmsg_type), ct);
1245 rcu_read_unlock();
1246 nf_ct_put(ct);
1247 if (err <= 0)
1248 goto free;
1249
1250 err = netlink_unicast(ctnl, skb2, NETLINK_CB(skb).portid, MSG_DONTWAIT);
1251 if (err < 0)
1252 goto out;
1253
1254 return 0;
1255
1256free:
1257 kfree_skb(skb2);
1258out:
1259 /* this avoids a loop in nfnetlink. */
1260 return err == -EAGAIN ? -ENOBUFS : err;
1261}

1270static int
1271ctnetlink_dump_list(struct sk_buff *skb, struct netlink_callback *cb, bool dying)
1272{

1275 struct hlist_nulls_node *n;

1277 u_int8_t l3proto = nfmsg->nfgen_family;
1278 int res;
1279 int cpu;
1280 struct hlist_nulls_head *list;
1281 struct net *net = sock_net(skb->sk);
1282
1283 if (cb->args[2])
1284 return 0;
1285
1286 last = (struct nf_conn *)cb->args[1];
1287
1288 for (cpu = cb->args[0]; cpu < nr_cpu_ids; cpu++) {
1289 struct ct_pcpu *pcpu;
1290

1292 continue;

1294 pcpu = per_cpu_ptr(net->ct.pcpu_lists, cpu);

1296 list = dying ? &pcpu->dying : &pcpu->unconfirmed;
1297restart:
1298 hlist_nulls_for_each_entry(h, n, list, hnnode) {
1299 ct = nf_ct_tuplehash_to_ctrack(h);
1300 if (l3proto && nf_ct_l3num(ct) != l3proto)
1301 continue;
1302 if (cb->args[1]) {
1303 if (ct != last)
1304 continue;
1305 cb->args[1] = 0;
1306 }
1307 rcu_read_lock();
1308 res = ctnetlink_fill_info(skb, NETLINK_CB(cb->skb).portid,
1309 cb->nlh->nlmsg_seq,
1310 NFNL_MSG_TYPE(cb->nlh->nlmsg_type),
1311 ct);
1312 rcu_read_unlock();
1313 if (res < 0) {
1314 if (!atomic_inc_not_zero(&ct->ct_general.use))
1315 continue;
1316 cb->args[0] = cpu;
1317 cb->args[1] = (unsigned long)ct;
1318 spin_unlock_bh(&pcpu->lock);
1319 goto out;
1320 }
1321 }
1322 if (cb->args[1]) {
1323 cb->args[1] = 0;
1324 goto restart;
1325 }
1326 spin_unlock_bh(&pcpu->lock);
1327 }
1328 cb->args[2] = 1;
1329out:
1330 if (last)
1331 nf_ct_put(last);
1332
1333 return skb->len;
1334}

1429static void
1430__ctnetlink_change_status(struct nf_conn *ct, unsigned long on,
1431 unsigned long off)
1432{
1433 unsigned int bit;
1434
1435 /* Ignore these unchangable bits */
1436 on &= ~IPS_UNCHANGEABLE_MASK;
1437 off &= ~IPS_UNCHANGEABLE_MASK;
1438
1439 for (bit = 0; bit < __IPS_MAX_BIT; bit++) {
1440 if (on & (1 << bit))
1441 set_bit(bit, &ct->status);
1442 else if (off & (1 << bit))
1443 clear_bit(bit, &ct->status);
1444 }
1445}

1447static int
1448ctnetlink_change_status(struct nf_conn *ct, const struct nlattr * const cda[])
1449{
1450 unsigned long d;
1451 unsigned int status = ntohl(nla_get_be32(cda[CTA_STATUS]));
1452 d = ct->status ^ status;
1453
1454 if (d & (IPS_EXPECTED|IPS_CONFIRMED|IPS_DYING))
1455 /* unchangeable */
1456 return -EBUSY;
1457
1458 if (d & IPS_SEEN_REPLY && !(status & IPS_SEEN_REPLY))
1459 /* SEEN_REPLY bit can only be set */
1460 return -EBUSY;
1461
1462 if (d & IPS_ASSURED && !(status & IPS_ASSURED))
1463 /* ASSURED bit can only be set */
1464 return -EBUSY;
1465
1466 __ctnetlink_change_status(ct, status, 0);
1467 return 0;
1468}

1494static int ctnetlink_change_helper(struct nf_conn *ct,
1495 const struct nlattr * const cda[])
1496{
1497 struct nf_conntrack_helper *helper;
1498 struct nf_conn_help *help = nfct_help(ct);
1499 char *helpname = NULL;
1500 struct nlattr *helpinfo = NULL;
1501 int err;
1502
1503 err = ctnetlink_parse_help(cda[CTA_HELP], &helpname, &helpinfo);
1504 if (err < 0)
1505 return err;
1506
1507 /* don't change helper of sibling connections */
1508 if (ct->master) {
1509 /* If we try to change the helper to the same thing twice,
1510 * treat the second attempt as a no-op instead of returning
1511 * an error.
1512 */
1513 err = -EBUSY;
1514 if (help) {
1515 rcu_read_lock();
1516 helper = rcu_dereference(help->helper);
1517 if (helper && !strcmp(helper->name, helpname))
1518 err = 0;
1519 rcu_read_unlock();
1520 }
1521
1522 return err;
1523 }
1524
1525 if (!strcmp(helpname, "")) {
1526 if (help && help->helper) {
1527 /* we had a helper before ... */
1528 nf_ct_remove_expectations(ct);
1529 RCU_INIT_POINTER(help->helper, NULL);
1530 }
1531
1532 return 0;
1533 }
1534
1535 rcu_read_lock();
1536 helper = __nf_conntrack_helper_find(helpname, nf_ct_l3num(ct),
1537 nf_ct_protonum(ct));
1538 if (helper == NULL) {
1539 rcu_read_unlock();
1540 return -EOPNOTSUPP;
1541 }
1542
1543 if (help) {
1544 if (help->helper == helper) {
1545 /* update private helper data if allowed. */
1546 if (helper->from_nlattr)
1547 helper->from_nlattr(helpinfo, ct);
1548 err = 0;
1549 } else
1550 err = -EBUSY;
1551 } else {
1552 /* we cannot set a helper for an existing conntrack */
1553 err = -EOPNOTSUPP;
1554 }
1555
1556 rcu_read_unlock();
1557 return err;
1558}

1753static struct nf_conn *
1754ctnetlink_create_conntrack(struct net *net,
1755 const struct nf_conntrack_zone *zone,
1756 const struct nlattr * const cda[],
1757 struct nf_conntrack_tuple *otuple,
1758 struct nf_conntrack_tuple *rtuple,
1759 u8 u3)
1760{
1761 struct nf_conn *ct;
1762 int err = -EINVAL;
1763 struct nf_conntrack_helper *helper;
1764 struct nf_conn_tstamp *tstamp;
1765
1766 ct = nf_conntrack_alloc(net, zone, otuple, rtuple, GFP_ATOMIC);
1767 if (IS_ERR(ct))
1768 return ERR_PTR(-ENOMEM);
1769
1770 if (!cda[CTA_TIMEOUT])
1771 goto err1;
1772
1773 ct->timeout = nfct_time_stamp + ntohl(nla_get_be32(cda[CTA_TIMEOUT])) * HZ;
1774
1775 rcu_read_lock();
1776 if (cda[CTA_HELP]) {
1777 char *helpname = NULL;
1778 struct nlattr *helpinfo = NULL;
1779
1780 err = ctnetlink_parse_help(cda[CTA_HELP], &helpname, &helpinfo);
1781 if (err < 0)
1782 goto err2;
1783
1784 helper = __nf_conntrack_helper_find(helpname, nf_ct_l3num(ct),
1785 nf_ct_protonum(ct));
1786 if (helper == NULL) {
1787 rcu_read_unlock();
1788#ifdef CONFIG_MODULES
1789 if (request_module("nfct-helper-%s", helpname) < 0) {
1790 err = -EOPNOTSUPP;
1791 goto err1;
1792 }
1793
1794 rcu_read_lock();
1795 helper = __nf_conntrack_helper_find(helpname,
1796 nf_ct_l3num(ct),
1797 nf_ct_protonum(ct));
1798 if (helper) {
1799 err = -EAGAIN;
1800 goto err2;
1801 }
1802 rcu_read_unlock();
1803#endif
1804 err = -EOPNOTSUPP;
1805 goto err1;
1806 } else {
1807 struct nf_conn_help *help;
1808
1809 help = nf_ct_helper_ext_add(ct, helper, GFP_ATOMIC);
1810 if (help == NULL) {
1811 err = -ENOMEM;
1812 goto err2;
1813 }
1814 /* set private helper data if allowed. */
1815 if (helper->from_nlattr)
1816 helper->from_nlattr(helpinfo, ct);
1817
1818 /* not in hash table yet so not strictly necessary */
1819 RCU_INIT_POINTER(help->helper, helper);
1820 }
1821 } else {
1822 /* try an implicit helper assignation */
1823 err = __nf_ct_try_assign_helper(ct, NULL, GFP_ATOMIC);
1824 if (err < 0)
1825 goto err2;
1826 }
1827
1828 err = ctnetlink_setup_nat(ct, cda);
1829 if (err < 0)
1830 goto err2;
1831
1832 nf_ct_acct_ext_add(ct, GFP_ATOMIC);
1833 nf_ct_tstamp_ext_add(ct, GFP_ATOMIC);
1834 nf_ct_ecache_ext_add(ct, 0, 0, GFP_ATOMIC);
1835 nf_ct_labels_ext_add(ct);
1836 nfct_seqadj_ext_add(ct);
1837 nfct_synproxy_ext_add(ct);
1838
1839 /* we must add conntrack extensions before confirmation. */
1840 ct->status |= IPS_CONFIRMED;
1841
1842 if (cda[CTA_STATUS]) {
1843 err = ctnetlink_change_status(ct, cda);
1844 if (err < 0)
1845 goto err2;
1846 }
1847
1848 if (cda[CTA_SEQ_ADJ_ORIG] || cda[CTA_SEQ_ADJ_REPLY]) {
1849 err = ctnetlink_change_seq_adj(ct, cda);
1850 if (err < 0)
1851 goto err2;
1852 }
1853
1854 memset(&ct->proto, 0, sizeof(ct->proto));
1855 if (cda[CTA_PROTOINFO]) {
1856 err = ctnetlink_change_protoinfo(ct, cda);
1857 if (err < 0)
1858 goto err2;
1859 }
1860
1861#if defined(CONFIG_NF_CONNTRACK_MARK)
1862 if (cda[CTA_MARK])
1863 ct->mark = ntohl(nla_get_be32(cda[CTA_MARK]));
1864#endif
1865
1866 /* setup master conntrack: this is a confirmed expectation */
1867 if (cda[CTA_TUPLE_MASTER]) {
1868 struct nf_conntrack_tuple master;
1869 struct nf_conntrack_tuple_hash *master_h;
1870 struct nf_conn *master_ct;
1871
1872 err = ctnetlink_parse_tuple(cda, &master, CTA_TUPLE_MASTER,
1873 u3, NULL);
1874 if (err < 0)
1875 goto err2;
1876
1877 master_h = nf_conntrack_find_get(net, zone, &master);
1878 if (master_h == NULL) {
1879 err = -ENOENT;
1880 goto err2;
1881 }
1882 master_ct = nf_ct_tuplehash_to_ctrack(master_h);
1883 __set_bit(IPS_EXPECTED_BIT, &ct->status);
1884 ct->master = master_ct;
1885 }
1886 tstamp = nf_conn_tstamp_find(ct);
1887 if (tstamp)
1888 tstamp->start = ktime_get_real_ns();
1889
1890 err = nf_conntrack_hash_check_insert(ct);
1891 if (err < 0)
1892 goto err2;
1893
1894 rcu_read_unlock();
1895
1896 return ct;
1897
1898err2:
1899 rcu_read_unlock();
1900err1:
1901 nf_conntrack_free(ct);
1902 return ERR_PTR(err);
1903}

1905static int ctnetlink_new_conntrack(struct net *net, struct sock *ctnl,
1906 struct sk_buff *skb,
1907 const struct nlmsghdr *nlh,
1908 const struct nlattr * const cda[],
1909 struct netlink_ext_ack *extack)
1910{
1911 struct nf_conntrack_tuple otuple, rtuple;
1912 struct nf_conntrack_tuple_hash *h = NULL;
1913 struct nfgenmsg *nfmsg = nlmsg_data(nlh);
1914 struct nf_conn *ct;
1915 u_int8_t u3 = nfmsg->nfgen_family;
1916 struct nf_conntrack_zone zone;
1917 int err;
1918
1919 err = ctnetlink_parse_zone(cda[CTA_ZONE], &zone);
1920 if (err < 0)
1921 return err;
1922
1923 if (cda[CTA_TUPLE_ORIG]) {
1924 err = ctnetlink_parse_tuple(cda, &otuple, CTA_TUPLE_ORIG,
1925 u3, &zone);
1926 if (err < 0)
1927 return err;
1928 }
1929
1930 if (cda[CTA_TUPLE_REPLY]) {
1931 err = ctnetlink_parse_tuple(cda, &rtuple, CTA_TUPLE_REPLY,
1932 u3, &zone);
1933 if (err < 0)
1934 return err;
1935 }
1936
1937 if (cda[CTA_TUPLE_ORIG])
1938 h = nf_conntrack_find_get(net, &zone, &otuple);
1939 else if (cda[CTA_TUPLE_REPLY])
1940 h = nf_conntrack_find_get(net, &zone, &rtuple);
1941
1942 if (h == NULL) {
1943 err = -ENOENT;
1944 if (nlh->nlmsg_flags & NLM_F_CREATE) {
1945 enum ip_conntrack_events events;
1946
1947 if (!cda[CTA_TUPLE_ORIG] || !cda[CTA_TUPLE_REPLY])
1948 return -EINVAL;
1949 if (otuple.dst.protonum != rtuple.dst.protonum)
1950 return -EINVAL;
1951
1952 ct = ctnetlink_create_conntrack(net, &zone, cda, &otuple,
1953 &rtuple, u3);
1954 if (IS_ERR(ct))
1955 return PTR_ERR(ct);
1956
1957 err = 0;
1958 if (test_bit(IPS_EXPECTED_BIT, &ct->status))
1959 events = 1 << IPCT_RELATED;
1960 else
1961 events = 1 << IPCT_NEW;
1962
1963 if (cda[CTA_LABELS] &&
1964 ctnetlink_attach_labels(ct, cda) == 0)
1965 events |= (1 << IPCT_LABEL);

2300static int
2301ctnetlink_update_status(struct nf_conn *ct, const struct nlattr * const cda[])
2302{
2303 unsigned int status = ntohl(nla_get_be32(cda[CTA_STATUS]));
2304 unsigned long d = ct->status ^ status;
2305
2306 if (d & IPS_SEEN_REPLY && !(status & IPS_SEEN_REPLY))
2307 /* SEEN_REPLY bit can only be set */
2308 return -EBUSY;
2309
2310 if (d & IPS_ASSURED && !(status & IPS_ASSURED))
2311 /* ASSURED bit can only be set */
2312 return -EBUSY;
2313
2314 /* This check is less strict than ctnetlink_change_status()
2315 * because callers often flip IPS_EXPECTED bits when sending
2316 * an NFQA_CT attribute to the kernel. So ignore the
2317 * unchangeable bits but do not error out. Also user programs
2318 * are allowed to clear the bits that they are allowed to change.
2319 */
2320 __ctnetlink_change_status(ct, status, ~status);
2321 return 0;
2322}
