29static unsigned int help(struct sk_buff *skb,
30 enum ip_conntrack_info ctinfo,
31 unsigned int protoff,
32 unsigned int matchoff,
33 unsigned int matchlen,
34 struct nf_conntrack_expect *exp)
35{
36 char buffer[sizeof("4294967296 65635")];
37 struct nf_conn *ct = exp->master;
38 union nf_inet_addr newaddr;
39 u_int16_t port;
40
41 /* Reply comes from server. */
42 newaddr = ct->tuplehash[IP_CT_DIR_REPLY].tuple.dst.u3;
43
44 exp->saved_proto.tcp.port = exp->tuple.dst.u.tcp.port;
45 exp->dir = IP_CT_DIR_REPLY;
46 exp->expectfn = nf_nat_follow_master;
47
48 /* Try to get same port: if not, try to change it. */
49 for (port = ntohs(exp->saved_proto.tcp.port); port != 0; port++) {
50 int ret;
51
52 exp->tuple.dst.u.tcp.port = htons(port);
53 ret = nf_ct_expect_related(exp);
54 if (ret == 0)
55 break;
56 else if (ret != -EBUSY) {
57 port = 0;
58 break;
59 }
60 }
61
62 if (port == 0) {
63 nf_ct_helper_log(skb, ct, "all ports in use");
64 return NF_DROP;
65 }
66
67 /* strlen("\1DCC CHAT chat AAAAAAAA P\1\n")=27
68 * strlen("\1DCC SCHAT chat AAAAAAAA P\1\n")=28
69 * strlen("\1DCC SEND F AAAAAAAA P S\1\n")=26
70 * strlen("\1DCC MOVE F AAAAAAAA P S\1\n")=26
71 * strlen("\1DCC TSEND F AAAAAAAA P S\1\n")=27
72 *
73 * AAAAAAAAA: bound addr (1.0.0.0==16777216, min 8 digits,
74 * 255.255.255.255==4294967296, 10 digits)
75 * P: bound port (min 1 d, max 5d (65635))
76 * F: filename (min 1 d )
77 * S: size (min 1 d )
78 * 0x01, \n: terminators
79 */
80 /* AAA = "us", ie. where server normally talks to. */
81 snprintf(buffer, sizeof(buffer), "%u %u", ntohl(newaddr.ip), port);
82 pr_debug("nf_nat_irc: inserting '%s' == %pI4, port %u\n",
83 buffer, &newaddr.ip, port);
84
85 if (!nf_nat_mangle_tcp_packet(skb, ct, ctinfo, protoff, matchoff,
86 matchlen, buffer, strlen(buffer))) {
87 nf_ct_helper_log(skb, ct, "cannot mangle packet");
88 nf_ct_unexpect_related(exp);
89 return NF_DROP;
90 }
91
92 return NF_ACCEPT;
93}

95static void __exit nf_nat_irc_fini(void)
96{
97 RCU_INIT_POINTER(nf_nat_irc_hook, NULL);
98 synchronize_rcu();
99}
100
101static int __init nf_nat_irc_init(void)
102{
103 BUG_ON(nf_nat_irc_hook != NULL);
104 RCU_INIT_POINTER(nf_nat_irc_hook, help);
105 return 0;
106}

108/* Prior to 2.6.11, we had a ports param. No longer, but don't break users. */
109static int warn_set(const char *val, struct kernel_param *kp)
110{
111 printk(KERN_INFO KBUILD_MODNAME
112 ": kernel >= 2.6.10 only uses 'ports' for conntrack modules\n");
113 return 0;
114}
115module_param_call(ports, warn_set, NULL, NULL, 0);
