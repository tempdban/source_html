18MODULE_AUTHOR("Patrick McHardy <kaber@trash.net>");
19MODULE_DESCRIPTION("Xtables: packet logging to netlink using NFLOG");
20MODULE_LICENSE("GPL");
21MODULE_ALIAS("ipt_NFLOG");
22MODULE_ALIAS("ip6t_NFLOG");
23
24static unsigned int
25nflog_tg(struct sk_buff *skb, const struct xt_action_param *par)
26{
27 const struct xt_nflog_info *info = par->targinfo;
28 struct net *net = xt_net(par);
29 struct nf_loginfo li;
30
31 li.type = NF_LOG_TYPE_ULOG;
32 li.u.ulog.copy_len = info->len;
33 li.u.ulog.group = info->group;
34 li.u.ulog.qthreshold = info->threshold;
35 li.u.ulog.flags = 0;
36
37 if (info->flags & XT_NFLOG_F_COPY_LEN)
38 li.u.ulog.flags |= NF_LOG_F_COPY_LEN;
39
40 nfulnl_log_packet(net, xt_family(par), xt_hooknum(par), skb,
41 xt_in(par), xt_out(par), &li, info->prefix);
42 return XT_CONTINUE;
43}
44
45static int nflog_tg_check(const struct xt_tgchk_param *par)
46{
47 const struct xt_nflog_info *info = par->targinfo;
48
49 if (info->flags & ~XT_NFLOG_MASK)
50 return -EINVAL;
51 if (info->prefix[sizeof(info->prefix) - 1] != '\0')
52 return -EINVAL;
53 return 0;
54}
55
56static struct xt_target nflog_tg_reg __read_mostly = {
57 .name = "NFLOG",
58 .revision = 0,
59 .family = NFPROTO_UNSPEC,
60 .checkentry = nflog_tg_check,
61 .target = nflog_tg,
62 .targetsize = sizeof(struct xt_nflog_info),
63 .me = THIS_MODULE,
64};
65
66static int __init nflog_tg_init(void)
67{
68 return xt_register_target(&nflog_tg_reg);
69}
70
71static void __exit nflog_tg_exit(void)
72{
73 xt_unregister_target(&nflog_tg_reg);
74}
75
76module_init(nflog_tg_init);
77module_exit(nflog_tg_exit);
