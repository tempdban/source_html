44struct idletimer_tg_attr {
45 struct attribute attr;
46 ssize_t (*show)(struct kobject *kobj,
47 struct attribute *attr, char *buf);
48};
49
50struct idletimer_tg {
51 struct list_head entry;
52 struct timer_list timer;
53 struct work_struct work;
54
55 struct kobject *kobj;
56 struct idletimer_tg_attr attr;
57
58 unsigned int refcnt;
59};
60
61static LIST_HEAD(idletimer_tg_list);
62static DEFINE_MUTEX(list_mutex);
63
64static struct kobject *idletimer_tg_kobj;
65
66static
67struct idletimer_tg *__idletimer_tg_find_by_label(const char *label)
68{
69 struct idletimer_tg *entry;
70
71 BUG_ON(!label);
72
73 list_for_each_entry(entry, &idletimer_tg_list, entry) {
74 if (!strcmp(label, entry->attr.attr.name))
75 return entry;
76 }
77
78 return NULL;
79}
80
81static ssize_t idletimer_tg_show(struct kobject *kobj, struct attribute *attr,
82 char *buf)
83{
84 struct idletimer_tg *timer;
85 unsigned long expires = 0;
86
87 mutex_lock(&list_mutex);
88
89 timer = __idletimer_tg_find_by_label(attr->name);
90 if (timer)
91 expires = timer->timer.expires;
92
93 mutex_unlock(&list_mutex);
94
95 if (time_after(expires, jiffies))
96 return sprintf(buf, "%u\n",
97 jiffies_to_msecs(expires - jiffies) / 1000);
98
99 return sprintf(buf, "0\n");
100}
101
102static void idletimer_tg_work(struct work_struct *work)
103{
104 struct idletimer_tg *timer = container_of(work, struct idletimer_tg,
105 work);
106
107 sysfs_notify(idletimer_tg_kobj, NULL, timer->attr.attr.name);
108}
109
110static void idletimer_tg_expired(unsigned long data)
111{
112 struct idletimer_tg *timer = (struct idletimer_tg *) data;
113
114 pr_debug("timer %s expired\n", timer->attr.attr.name);
115
116 schedule_work(&timer->work);
117}

119static int idletimer_tg_create(struct idletimer_tg_info *info)
120{
121 int ret;
122
123 info->timer = kmalloc(sizeof(*info->timer), GFP_KERNEL);
124 if (!info->timer) {
125 ret = -ENOMEM;
126 goto out;
127 }
128
129 sysfs_attr_init(&info->timer->attr.attr);
130 info->timer->attr.attr.name = kstrdup(info->label, GFP_KERNEL);
131 if (!info->timer->attr.attr.name) {
132 ret = -ENOMEM;
133 goto out_free_timer;
134 }
135 info->timer->attr.attr.mode = S_IRUGO;
136 info->timer->attr.show = idletimer_tg_show;
137
138 ret = sysfs_create_file(idletimer_tg_kobj, &info->timer->attr.attr);
139 if (ret < 0) {
140 pr_debug("couldn't add file to sysfs");
141 goto out_free_attr;
142 }
143
144 list_add(&info->timer->entry, &idletimer_tg_list);
145
146 setup_timer(&info->timer->timer, idletimer_tg_expired,
147 (unsigned long) info->timer);
148 info->timer->refcnt = 1;
149
150 INIT_WORK(&info->timer->work, idletimer_tg_work);
151
152 mod_timer(&info->timer->timer,
153 msecs_to_jiffies(info->timeout * 1000) + jiffies);
154
155 return 0;
156
157out_free_attr:
158 kfree(info->timer->attr.attr.name);
159out_free_timer:
160 kfree(info->timer);
161out:
162 return ret;
163}

168static unsigned int idletimer_tg_target(struct sk_buff *skb,
169 const struct xt_action_param *par)
170{
171 const struct idletimer_tg_info *info = par->targinfo;
172
173 pr_debug("resetting timer %s, timeout period %u\n",
174 info->label, info->timeout);
175
176 BUG_ON(!info->timer);
177
178 mod_timer(&info->timer->timer,
179 msecs_to_jiffies(info->timeout * 1000) + jiffies);
180
181 return XT_CONTINUE;
182}

184static int idletimer_tg_checkentry(const struct xt_tgchk_param *par)
185{
186 struct idletimer_tg_info *info = par->targinfo;
187 int ret;
188
189 pr_debug("checkentry targinfo%s\n", info->label);
190
191 if (info->timeout == 0) {
192 pr_debug("timeout value is zero\n");
193 return -EINVAL;
194 }
195 if (info->timeout >= INT_MAX / 1000) {
196 pr_debug("timeout value is too big\n");
197 return -EINVAL;
198 }
199 if (info->label[0] == '\0' ||
200 strnlen(info->label,
201 MAX_IDLETIMER_LABEL_SIZE) == MAX_IDLETIMER_LABEL_SIZE) {
202 pr_debug("label is empty or not nul-terminated\n");
203 return -EINVAL;
204 }
205
206 mutex_lock(&list_mutex);
207
208 info->timer = __idletimer_tg_find_by_label(info->label);
209 if (info->timer) {
210 info->timer->refcnt++;
211 mod_timer(&info->timer->timer,
212 msecs_to_jiffies(info->timeout * 1000) + jiffies);
213
214 pr_debug("increased refcnt of timer %s to %u\n",
215 info->label, info->timer->refcnt);
216 } else {
217 ret = idletimer_tg_create(info);
218 if (ret < 0) {
219 pr_debug("failed to create timer\n");
220 mutex_unlock(&list_mutex);
221 return ret;
222 }
223 }
224
225 mutex_unlock(&list_mutex);
226 return 0;
227}
228
229static void idletimer_tg_destroy(const struct xt_tgdtor_param *par)
230{
231 const struct idletimer_tg_info *info = par->targinfo;
232
233 pr_debug("destroy targinfo %s\n", info->label);
234
235 mutex_lock(&list_mutex);
236
237 if (--info->timer->refcnt == 0) {
238 pr_debug("deleting timer %s\n", info->label);
239
240 list_del(&info->timer->entry);
241 del_timer_sync(&info->timer->timer);
242 cancel_work_sync(&info->timer->work);
243 sysfs_remove_file(idletimer_tg_kobj, &info->timer->attr.attr);
244 kfree(info->timer->attr.attr.name);
245 kfree(info->timer);
246 } else {
247 pr_debug("decreased refcnt of timer %s to %u\n",
248 info->label, info->timer->refcnt);
249 }
250
251 mutex_unlock(&list_mutex);
252}

254static struct xt_target idletimer_tg __read_mostly = {
255 .name = "IDLETIMER",
256 .family = NFPROTO_UNSPEC,
257 .target = idletimer_tg_target,
258 .targetsize = sizeof(struct idletimer_tg_info),
259 .usersize = offsetof(struct idletimer_tg_info, timer),
260 .checkentry = idletimer_tg_checkentry,
261 .destroy = idletimer_tg_destroy,
262 .me = THIS_MODULE,
263};

265static struct class *idletimer_tg_class;
266
267static struct device *idletimer_tg_device;

269static int __init idletimer_tg_init(void)
270{
271 int err;
272
273 idletimer_tg_class = class_create(THIS_MODULE, "xt_idletimer");
274 err = PTR_ERR(idletimer_tg_class);
275 if (IS_ERR(idletimer_tg_class)) {
276 pr_debug("couldn't register device class\n");
277 goto out;
278 }
279
280 idletimer_tg_device = device_create(idletimer_tg_class, NULL,
281 MKDEV(0, 0), NULL, "timers");
282 err = PTR_ERR(idletimer_tg_device);
283 if (IS_ERR(idletimer_tg_device)) {
284 pr_debug("couldn't register system device\n");
285 goto out_class;
286 }
287
288 idletimer_tg_kobj = &idletimer_tg_device->kobj;
289
290 err = xt_register_target(&idletimer_tg);
291 if (err < 0) {
292 pr_debug("couldn't register xt target\n");
293 goto out_dev;
294 }
295
296 return 0;
297out_dev:
298 device_destroy(idletimer_tg_class, MKDEV(0, 0));
299out_class:
300 class_destroy(idletimer_tg_class);
301out:
302 return err;
303}
304
305static void __exit idletimer_tg_exit(void)
306{
307 xt_unregister_target(&idletimer_tg);
308
309 device_destroy(idletimer_tg_class, MKDEV(0, 0));
310 class_destroy(idletimer_tg_class);
311}

313module_init(idletimer_tg_init);
