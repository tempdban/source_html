27/* Returns 1 if the spi is matched by the range, 0 otherwise */
28static inline bool
29spi_match(u_int32_t min, u_int32_t max, u_int32_t spi, bool invert)
30{
31 bool r;
32 pr_debug("spi_match:%c 0x%x <= 0x%x <= 0x%x\n",
33 invert ? '!' : ' ', min, spi, max);
34 r = (spi >= min && spi <= max) ^ invert;
35 pr_debug(" result %s\n", r ? "PASS" : "FAILED");
36 return r;
37}

39static bool esp_mt(const struct sk_buff *skb, struct xt_action_param *par)
40{
41 const struct ip_esp_hdr *eh;
42 struct ip_esp_hdr _esp;
43 const struct xt_esp *espinfo = par->matchinfo;
44
45 /* Must not be a fragment. */
46 if (par->fragoff != 0)
47 return false;
48
49 eh = skb_header_pointer(skb, par->thoff, sizeof(_esp), &_esp);
50 if (eh == NULL) {
51 /* We've been asked to examine this packet, and we
52 * can't. Hence, no choice but to drop.
53 */
54 pr_debug("Dropping evil ESP tinygram.\n");
55 par->hotdrop = true;
56 return false;
57 }
58
59 return spi_match(espinfo->spis[0], espinfo->spis[1], ntohl(eh->spi),
60 !!(espinfo->invflags & XT_ESP_INV_SPI));
61}
62
63static int esp_mt_check(const struct xt_mtchk_param *par)
64{
65 const struct xt_esp *espinfo = par->matchinfo;
66
67 if (espinfo->invflags & ~XT_ESP_INV_MASK) {
68 pr_debug("unknown flags %X\n", espinfo->invflags);
69 return -EINVAL;
70 }
71
72 return 0;
73}

75static struct xt_match esp_mt_reg[] __read_mostly = {
76 {
77 .name = "esp",
78 .family = NFPROTO_IPV4,
79 .checkentry = esp_mt_check,
80 .match = esp_mt,
81 .matchsize = sizeof(struct xt_esp),
82 .proto = IPPROTO_ESP,
83 .me = THIS_MODULE,
84 },
85 {
86 .name = "esp",
87 .family = NFPROTO_IPV6,
88 .checkentry = esp_mt_check,
89 .match = esp_mt,
90 .matchsize = sizeof(struct xt_esp),
91 .proto = IPPROTO_ESP,
92 .me = THIS_MODULE,
93 },
94};
95
96static int __init esp_mt_init(void)
97{
98 return xt_register_matches(esp_mt_reg, ARRAY_SIZE(esp_mt_reg));
99}
100
101static void __exit esp_mt_exit(void)
102{
103 xt_unregister_matches(esp_mt_reg, ARRAY_SIZE(esp_mt_reg));
104}
105
106module_init(esp_mt_init);
107module_exit(esp_mt_exit);
