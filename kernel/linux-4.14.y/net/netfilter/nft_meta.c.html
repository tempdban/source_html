33void nft_meta_get_eval(const struct nft_expr *expr,
34 struct nft_regs *regs,
35 const struct nft_pktinfo *pkt)
36{
37 const struct nft_meta *priv = nft_expr_priv(expr);
38 const struct sk_buff *skb = pkt->skb;
39 const struct net_device *in = nft_in(pkt), *out = nft_out(pkt);
40 struct sock *sk;
41 u32 *dest = &regs->data[priv->dreg];
42
43 switch (priv->key) {
44 case NFT_META_LEN:
45 *dest = skb->len;
46 break;
47 case NFT_META_PROTOCOL:
48 nft_reg_store16(dest, (__force u16)skb->protocol);
49 break;
50 case NFT_META_NFPROTO:
51 nft_reg_store8(dest, nft_pf(pkt));
52 break;
53 case NFT_META_L4PROTO:
54 if (!pkt->tprot_set)
55 goto err;
56 nft_reg_store8(dest, pkt->tprot);
57 break;
58 case NFT_META_PRIORITY:
59 *dest = skb->priority;
60 break;
61 case NFT_META_MARK:
62 *dest = skb->mark;
63 break;
64 case NFT_META_IIF:
65 if (in == NULL)
66 goto err;
67 *dest = in->ifindex;
68 break;
69 case NFT_META_OIF:
70 if (out == NULL)
71 goto err;
72 *dest = out->ifindex;
73 break;
74 case NFT_META_IIFNAME:
75 if (in == NULL)
76 goto err;
77 strncpy((char *)dest, in->name, IFNAMSIZ);
78 break;
79 case NFT_META_OIFNAME:
80 if (out == NULL)
81 goto err;
82 strncpy((char *)dest, out->name, IFNAMSIZ);
83 break;
84 case NFT_META_IIFTYPE:
85 if (in == NULL)
86 goto err;
87 nft_reg_store16(dest, in->type);
88 break;
89 case NFT_META_OIFTYPE:
90 if (out == NULL)
91 goto err;
92 nft_reg_store16(dest, out->type);
93 break;
94 case NFT_META_SKUID:
95 sk = skb_to_full_sk(skb);
96 if (!sk || !sk_fullsock(sk))
97 goto err;
98
99 read_lock_bh(&sk->sk_callback_lock);
100 if (sk->sk_socket == NULL ||
101 sk->sk_socket->file == NULL) {
102 read_unlock_bh(&sk->sk_callback_lock);
103 goto err;
104 }
105
106 *dest = from_kuid_munged(&init_user_ns,
107 sk->sk_socket->file->f_cred->fsuid);
108 read_unlock_bh(&sk->sk_callback_lock);
109 break;
110 case NFT_META_SKGID:
111 sk = skb_to_full_sk(skb);
112 if (!sk || !sk_fullsock(sk))
113 goto err;
114
115 read_lock_bh(&sk->sk_callback_lock);
116 if (sk->sk_socket == NULL ||
117 sk->sk_socket->file == NULL) {
118 read_unlock_bh(&sk->sk_callback_lock);
119 goto err;
120 }
121 *dest = from_kgid_munged(&init_user_ns,
122 sk->sk_socket->file->f_cred->fsgid);
123 read_unlock_bh(&sk->sk_callback_lock);
124 break;
125#ifdef CONFIG_IP_ROUTE_CLASSID
126 case NFT_META_RTCLASSID: {
127 const struct dst_entry *dst = skb_dst(skb);
128
129 if (dst == NULL)
130 goto err;
131 *dest = dst->tclassid;
132 break;
133 }
134#endif
135#ifdef CONFIG_NETWORK_SECMARK
136 case NFT_META_SECMARK:
137 *dest = skb->secmark;
138 break;
139#endif
140 case NFT_META_PKTTYPE:
141 if (skb->pkt_type != PACKET_LOOPBACK) {
142 nft_reg_store8(dest, skb->pkt_type);
143 break;
144 }
145
146 switch (nft_pf(pkt)) {
147 case NFPROTO_IPV4:
148 if (ipv4_is_multicast(ip_hdr(skb)->daddr))
149 nft_reg_store8(dest, PACKET_MULTICAST);
150 else
151 nft_reg_store8(dest, PACKET_BROADCAST);
152 break;
153 case NFPROTO_IPV6:
154 nft_reg_store8(dest, PACKET_MULTICAST);
155 break;
156 case NFPROTO_NETDEV:
157 switch (skb->protocol) {
158 case htons(ETH_P_IP): {
159 int noff = skb_network_offset(skb);
160 struct iphdr *iph, _iph;
161
162 iph = skb_header_pointer(skb, noff,
163 sizeof(_iph), &_iph);
164 if (!iph)
165 goto err;
166
167 if (ipv4_is_multicast(iph->daddr))
168 nft_reg_store8(dest, PACKET_MULTICAST);
169 else
170 nft_reg_store8(dest, PACKET_BROADCAST);
171
172 break;
173 }
174 case htons(ETH_P_IPV6):
175 nft_reg_store8(dest, PACKET_MULTICAST);
176 break;
177 default:
178 WARN_ON_ONCE(1);
179 goto err;
180 }
181 break;
182 default:
183 WARN_ON_ONCE(1);
184 goto err;
185 }
186 break;
187 case NFT_META_CPU:
188 *dest = raw_smp_processor_id();
189 break;
190 case NFT_META_IIFGROUP:
191 if (in == NULL)
192 goto err;
193 *dest = in->group;
194 break;
195 case NFT_META_OIFGROUP:
196 if (out == NULL)
197 goto err;
198 *dest = out->group;
199 break;
200#ifdef CONFIG_CGROUP_NET_CLASSID
201 case NFT_META_CGROUP:
202 sk = skb_to_full_sk(skb);
203 if (!sk || !sk_fullsock(sk))
204 goto err;
205 *dest = sock_cgroup_classid(&sk->sk_cgrp_data);
206 break;
207#endif
208 case NFT_META_PRANDOM: {
209 struct rnd_state *state = this_cpu_ptr(&nft_prandom_state);
210 *dest = prandom_u32_state(state);
211 break;
212 }
213 default:
214 WARN_ON(1);
215 goto err;
216 }
217 return;
218
219err:
220 regs->verdict.code = NFT_BREAK;
221}
222EXPORT_SYMBOL_GPL(nft_meta_get_eval);

224void nft_meta_set_eval(const struct nft_expr *expr,
225 struct nft_regs *regs,
226 const struct nft_pktinfo *pkt)
227{
228 const struct nft_meta *meta = nft_expr_priv(expr);
229 struct sk_buff *skb = pkt->skb;
230 u32 *sreg = &regs->data[meta->sreg];
231 u32 value = *sreg;
232 u8 value8;
233
234 switch (meta->key) {
235 case NFT_META_MARK:
236 skb->mark = value;
237 break;
238 case NFT_META_PRIORITY:
239 skb->priority = value;
240 break;
241 case NFT_META_PKTTYPE:
242 value8 = nft_reg_load8(sreg);
243
244 if (skb->pkt_type != value8 &&
245 skb_pkt_type_ok(value8) &&
246 skb_pkt_type_ok(skb->pkt_type))
247 skb->pkt_type = value8;
248 break;
249 case NFT_META_NFTRACE:
250 value8 = nft_reg_load8(sreg);
251
252 skb->nf_trace = !!value8;
253 break;
254 default:
255 WARN_ON(1);
256 }
257}
258EXPORT_SYMBOL_GPL(nft_meta_set_eval);

267int nft_meta_get_init(const struct nft_ctx *ctx,
268 const struct nft_expr *expr,
269 const struct nlattr * const tb[])
270{
271 struct nft_meta *priv = nft_expr_priv(expr);
272 unsigned int len;
273
274 priv->key = ntohl(nla_get_be32(tb[NFTA_META_KEY]));
275 switch (priv->key) {
276 case NFT_META_PROTOCOL:
277 case NFT_META_IIFTYPE:
278 case NFT_META_OIFTYPE:
279 len = sizeof(u16);
280 break;
281 case NFT_META_NFPROTO:
282 case NFT_META_L4PROTO:
283 case NFT_META_LEN:
284 case NFT_META_PRIORITY:
285 case NFT_META_MARK:
286 case NFT_META_IIF:
287 case NFT_META_OIF:
288 case NFT_META_SKUID:
289 case NFT_META_SKGID:
290#ifdef CONFIG_IP_ROUTE_CLASSID
291 case NFT_META_RTCLASSID:
292#endif
293#ifdef CONFIG_NETWORK_SECMARK
294 case NFT_META_SECMARK:
295#endif
296 case NFT_META_PKTTYPE:
297 case NFT_META_CPU:
298 case NFT_META_IIFGROUP:
299 case NFT_META_OIFGROUP:
300#ifdef CONFIG_CGROUP_NET_CLASSID
301 case NFT_META_CGROUP:
302#endif
303 len = sizeof(u32);
304 break;
305 case NFT_META_IIFNAME:
306 case NFT_META_OIFNAME:
307 len = IFNAMSIZ;
308 break;
309 case NFT_META_PRANDOM:
310 prandom_init_once(&nft_prandom_state);
311 len = sizeof(u32);
312 break;
313 default:
314 return -EOPNOTSUPP;
315 }
316
317 priv->dreg = nft_parse_register(tb[NFTA_META_DREG]);
318 return nft_validate_register_store(ctx, priv->dreg, NULL,
319 NFT_DATA_VALUE, len);
320}
321EXPORT_SYMBOL_GPL(nft_meta_get_init);

323int nft_meta_set_validate(const struct nft_ctx *ctx,
324 const struct nft_expr *expr,
325 const struct nft_data **data)
326{
327 struct nft_meta *priv = nft_expr_priv(expr);
328 unsigned int hooks;
329
330 if (priv->key != NFT_META_PKTTYPE)
331 return 0;
332
333 switch (ctx->afi->family) {
334 case NFPROTO_BRIDGE:
335 hooks = 1 << NF_BR_PRE_ROUTING;
336 break;
337 case NFPROTO_NETDEV:
338 hooks = 1 << NF_NETDEV_INGRESS;
339 break;
340 case NFPROTO_IPV4:
341 case NFPROTO_IPV6:
342 case NFPROTO_INET:
343 hooks = 1 << NF_INET_PRE_ROUTING;
344 break;
345 default:
346 return -EOPNOTSUPP;
347 }
348
349 return nft_chain_validate_hooks(ctx->chain, hooks);
350}
351EXPORT_SYMBOL_GPL(nft_meta_set_validate);
352
353int nft_meta_set_init(const struct nft_ctx *ctx,
354 const struct nft_expr *expr,
355 const struct nlattr * const tb[])
356{
357 struct nft_meta *priv = nft_expr_priv(expr);
358 unsigned int len;
359 int err;
360
361 priv->key = ntohl(nla_get_be32(tb[NFTA_META_KEY]));
362 switch (priv->key) {
363 case NFT_META_MARK:
364 case NFT_META_PRIORITY:
365 len = sizeof(u32);
366 break;
367 case NFT_META_NFTRACE:
368 len = sizeof(u8);
369 break;
370 case NFT_META_PKTTYPE:
371 len = sizeof(u8);
372 break;
373 default:
374 return -EOPNOTSUPP;
375 }
376
377 priv->sreg = nft_parse_register(tb[NFTA_META_SREG]);
378 err = nft_validate_register_load(priv->sreg, len);
379 if (err < 0)
380 return err;
381
382 if (priv->key == NFT_META_NFTRACE)
383 static_branch_inc(&nft_trace_enabled);
384
385 return 0;
386}
387EXPORT_SYMBOL_GPL(nft_meta_set_init);

389int nft_meta_get_dump(struct sk_buff *skb,
390 const struct nft_expr *expr)
391{
392 const struct nft_meta *priv = nft_expr_priv(expr);
393
394 if (nla_put_be32(skb, NFTA_META_KEY, htonl(priv->key)))
395 goto nla_put_failure;
396 if (nft_dump_register(skb, NFTA_META_DREG, priv->dreg))
397 goto nla_put_failure;
398 return 0;
399
400nla_put_failure:
401 return -1;
402}
403EXPORT_SYMBOL_GPL(nft_meta_get_dump);

435 .size = NFT_EXPR_SIZE(sizeof(struct nft_meta)),

441static const struct nft_expr_ops nft_meta_set_ops = {
442 .type = &nft_meta_type,
443 .size = NFT_EXPR_SIZE(sizeof(struct nft_meta)),
444 .eval = nft_meta_set_eval,
445 .init = nft_meta_set_init,
446 .destroy = nft_meta_set_destroy,
447 .dump = nft_meta_set_dump,
448 .validate = nft_meta_set_validate,
449};
450
451static const struct nft_expr_ops *
452nft_meta_select_ops(const struct nft_ctx *ctx,
453 const struct nlattr * const tb[])
454{
455 if (tb[NFTA_META_KEY] == NULL)
456 return ERR_PTR(-EINVAL);
457
458 if (tb[NFTA_META_DREG] && tb[NFTA_META_SREG])
459 return ERR_PTR(-EINVAL);
460
461 if (tb[NFTA_META_DREG])
462 return &nft_meta_get_ops;
463
464 if (tb[NFTA_META_SREG])
465 return &nft_meta_set_ops;
466
467 return ERR_PTR(-EINVAL);
468}

470static struct nft_expr_type nft_meta_type __read_mostly = {
471 .name = "meta",
472 .select_ops = nft_meta_select_ops,
473 .policy = nft_meta_policy,
474 .maxattr = NFTA_META_MAX,
475 .owner = THIS_MODULE,
476};
