133/* nl_table locking explained:
134 * Lookup and traversal are protected with an RCU read-side lock. Insertion
135 * and removal are protected with per bucket lock while using RCU list
136 * modification primitives and may run in parallel to RCU protected lookups.
137 * Destruction of the Netlink socket may only occur *after* nl_table_lock has
138 * been acquired * either during or after the socket has been removed from
139 * the list and after an RCU grace period.
140 */
141DEFINE_RWLOCK(nl_table_lock);
142EXPORT_SYMBOL_GPL(nl_table_lock);
143static atomic_t nl_table_users = ATOMIC_INIT(0);

145#define nl_deref_protected(X) rcu_dereference_protected(X, lockdep_is_held(&nl_table_lock));

159static struct sk_buff *netlink_to_full_skb(const struct sk_buff *skb,
160 gfp_t gfp_mask)
161{
162 unsigned int len = skb_end_offset(skb);
163 struct sk_buff *new;
164
165 new = alloc_skb(len, gfp_mask);
166 if (new == NULL)
167 return NULL;
168
169 NETLINK_CB(new).portid = NETLINK_CB(skb).portid;
170 NETLINK_CB(new).dst_group = NETLINK_CB(skb).dst_group;
171 NETLINK_CB(new).creds = NETLINK_CB(skb).creds;
172
173 skb_put_data(new, skb->data, len);
174 return new;
175}

177int netlink_add_tap(struct netlink_tap *nt)
178{
179 if (unlikely(nt->dev->type != ARPHRD_NETLINK))
180 return -EINVAL;
181
182 spin_lock(&netlink_tap_lock);
183 list_add_rcu(&nt->list, &netlink_tap_all);
184 spin_unlock(&netlink_tap_lock);
185
186 __module_get(nt->module);
187
188 return 0;
189}
190EXPORT_SYMBOL_GPL(netlink_add_tap);

192static int __netlink_remove_tap(struct netlink_tap *nt)
193{
194 bool found = false;
195 struct netlink_tap *tmp;
196
197 spin_lock(&netlink_tap_lock);
198
199 list_for_each_entry(tmp, &netlink_tap_all, list) {
200 if (nt == tmp) {
201 list_del_rcu(&nt->list);
202 found = true;
203 goto out;
204 }
205 }
206
207 pr_warn("__netlink_remove_tap: %p not found\n", nt);
208out:
209 spin_unlock(&netlink_tap_lock);
210
211 if (found)
212 module_put(nt->module);
213
214 return found ? 0 : -ENODEV;
215}

228static bool netlink_filter_tap(const struct sk_buff *skb)
229{
230 struct sock *sk = skb->sk;
231
232 /* We take the more conservative approach and
233 * whitelist socket protocols that may pass.
234 */
235 switch (sk->sk_protocol) {
236 case NETLINK_ROUTE:
237 case NETLINK_USERSOCK:
238 case NETLINK_SOCK_DIAG:
239 case NETLINK_NFLOG:
240 case NETLINK_XFRM:
241 case NETLINK_FIB_LOOKUP:
242 case NETLINK_NETFILTER:
243 case NETLINK_GENERIC:
244 return true;
245 }
246
247 return false;
248}

250static int __netlink_deliver_tap_skb(struct sk_buff *skb,
251 struct net_device *dev)
252{
253 struct sk_buff *nskb;
254 struct sock *sk = skb->sk;
255 int ret = -ENOMEM;
256
257 if (!net_eq(dev_net(dev), sock_net(sk)))
258 return 0;
259
260 dev_hold(dev);
261
262 if (is_vmalloc_addr(skb->head))
263 nskb = netlink_to_full_skb(skb, GFP_ATOMIC);
264 else
265 nskb = skb_clone(skb, GFP_ATOMIC);
266 if (nskb) {
267 nskb->dev = dev;
268 nskb->protocol = htons((u16) sk->sk_protocol);
269 nskb->pkt_type = netlink_is_kernel(sk) ?
270 PACKET_KERNEL : PACKET_USER;
271 skb_reset_network_header(nskb);
272 ret = dev_queue_xmit(nskb);
273 if (unlikely(ret > 0))
274 ret = net_xmit_errno(ret);
275 }
276
277 dev_put(dev);
278 return ret;
279}

281static void __netlink_deliver_tap(struct sk_buff *skb)
282{
283 int ret;
284 struct netlink_tap *tmp;
285
286 if (!netlink_filter_tap(skb))
287 return;
288
289 list_for_each_entry_rcu(tmp, &netlink_tap_all, list) {
290 ret = __netlink_deliver_tap_skb(skb, tmp->dev);
291 if (unlikely(ret))
292 break;
293 }
294}

296static void netlink_deliver_tap(struct sk_buff *skb)
297{
298 rcu_read_lock();
299
300 if (unlikely(!list_empty(&netlink_tap_all)))
301 __netlink_deliver_tap(skb);
302
303 rcu_read_unlock();
304}

313static void netlink_overrun(struct sock *sk)
314{
315 struct netlink_sock *nlk = nlk_sk(sk);
316
317 if (!(nlk->flags & NETLINK_F_RECV_NO_ENOBUFS)) {
318 if (!test_and_set_bit(NETLINK_S_CONGESTED,
319 &nlk_sk(sk)->state)) {
320 sk->sk_err = ENOBUFS;
321 sk->sk_error_report(sk);
322 }
323 }
324 atomic_inc(&sk->sk_drops);
325}

328{
329 struct netlink_sock *nlk = nlk_sk(sk);

331 if (skb_queue_empty(&sk->sk_receive_queue))
332 clear_bit(NETLINK_S_CONGESTED, &nlk->state);
333 if (!test_bit(NETLINK_S_CONGESTED, &nlk->state))
334 wake_up_interruptible(&nlk->wait);

359static void netlink_sock_destruct(struct sock *sk)
360{
361 struct netlink_sock *nlk = nlk_sk(sk);
362
363 if (nlk->cb_running) {
364 if (nlk->cb.done)
365 nlk->cb.done(&nlk->cb);
366 module_put(nlk->cb.module);
367 kfree_skb(nlk->cb.skb);
368 }
369
370 skb_queue_purge(&sk->sk_receive_queue);
371
372 if (!sock_flag(sk, SOCK_DEAD)) {
373 printk(KERN_ERR "Freeing alive netlink socket %p\n", sk);
374 return;
375 }
376
377 WARN_ON(atomic_read(&sk->sk_rmem_alloc));
378 WARN_ON(refcount_read(&sk->sk_wmem_alloc));
379 WARN_ON(nlk_sk(sk)->groups);
380}

390/* This lock without WQ_FLAG_EXCLUSIVE is good on UP and it is _very_ bad on
391 * SMP. Look, when several writers sleep and reader wakes them up, all but one
392 * immediately hit write lock and grab all the cpus. Exclusive sleep solves
393 * this, _but_ remember, it adds useless work on UP machines.
394 */
395
396void netlink_table_grab(void)
397 __acquires(nl_table_lock)
398{
399 might_sleep();
400
401 write_lock_irq(&nl_table_lock);
402
403 if (atomic_read(&nl_table_users)) {
404 DECLARE_WAITQUEUE(wait, current);
405
406 add_wait_queue_exclusive(&nl_table_wait, &wait);
407 for (;;) {
408 set_current_state(TASK_UNINTERRUPTIBLE);
409 if (atomic_read(&nl_table_users) == 0)
410 break;
411 write_unlock_irq(&nl_table_lock);
412 schedule();
413 write_lock_irq(&nl_table_lock);
414 }
415
416 __set_current_state(TASK_RUNNING);
417 remove_wait_queue(&nl_table_wait, &wait);
418 }
419}
420
421void netlink_table_ungrab(void)
422 __releases(nl_table_lock)
423{
424 write_unlock_irq(&nl_table_lock);
425 wake_up(&nl_table_wait);
426}

428static inline void
429netlink_lock_table(void)
430{
431 /* read_lock() synchronizes us to netlink_table_grab */
432
433 read_lock(&nl_table_lock);
434 atomic_inc(&nl_table_users);
435 read_unlock(&nl_table_lock);
436}

438static inline void
439netlink_unlock_table(void)
440{
441 if (atomic_dec_and_test(&nl_table_users))
442 wake_up(&nl_table_wait);
443}

509static void
510netlink_update_listeners(struct sock *sk)
511{
512 struct netlink_table *tbl = &nl_table[sk->sk_protocol];
513 unsigned long mask;
514 unsigned int i;
515 struct listeners *listeners;
516
517 listeners = nl_deref_protected(tbl->listeners);
518 if (!listeners)
519 return;
520
521 for (i = 0; i < NLGRPLONGS(tbl->groups); i++) {
522 mask = 0;
523 sk_for_each_bound(sk, &tbl->mc_list) {
524 if (i < NLGRPLONGS(nlk_sk(sk)->ngroups))
525 mask |= nlk_sk(sk)->groups[i];
526 }
527 listeners->masks[i] = mask;
528 }
529 /* this function is only called with the netlink table "grabbed", which
530 * makes sure updates are visible before bind or setsockopt return. */
531}

533static int netlink_insert(struct sock *sk, u32 portid)
534{
535 struct netlink_table *table = &nl_table[sk->sk_protocol];
536 int err;
537
538 lock_sock(sk);
539
540 err = nlk_sk(sk)->portid == portid ? 0 : -EBUSY;
541 if (nlk_sk(sk)->bound)
542 goto err;
543
544 err = -ENOMEM;
545 if (BITS_PER_LONG > 32 &&
546 unlikely(atomic_read(&table->hash.nelems) >= UINT_MAX))
547 goto err;
548
549 nlk_sk(sk)->portid = portid;
550 sock_hold(sk);
551
552 err = __netlink_insert(table, sk);
553 if (err) {
554 /* In case the hashtable backend returns with -EBUSY
555 * from here, it must not escape to the caller.
556 */
557 if (unlikely(err == -EBUSY))
558 err = -EOVERFLOW;
559 if (err == -EEXIST)
560 err = -EADDRINUSE;
561 sock_put(sk);
562 goto err;
563 }
564
565 /* We need to ensure that the socket is hashed and visible. */
566 smp_wmb();
567 nlk_sk(sk)->bound = portid;
568
569err:
570 release_sock(sk);
571 return err;
572}

574static void netlink_remove(struct sock *sk)
575{
576 struct netlink_table *table;
577
578 table = &nl_table[sk->sk_protocol];
579 if (!rhashtable_remove_fast(&table->hash, &nlk_sk(sk)->node,
580 netlink_rhashtable_params)) {
581 WARN_ON(refcount_read(&sk->sk_refcnt) == 1);
582 __sock_put(sk);
583 }
584
585 netlink_table_grab();
586 if (nlk_sk(sk)->subscriptions) {
587 __sk_del_bind_node(sk);
588 netlink_update_listeners(sk);
589 }
590 if (sk->sk_protocol == NETLINK_GENERIC)
591 atomic_inc(&genl_sk_destructing_cnt);
592 netlink_table_ungrab();
593}

601static int __netlink_create(struct net *net, struct socket *sock,
602 struct mutex *cb_mutex, int protocol,
603 int kern)
604{
605 struct sock *sk;
606 struct netlink_sock *nlk;
607
608 sock->ops = &netlink_ops;
609
610 sk = sk_alloc(net, PF_NETLINK, GFP_KERNEL, &netlink_proto, kern);
611 if (!sk)
612 return -ENOMEM;
613
614 sock_init_data(sock, sk);
615
616 nlk = nlk_sk(sk);
617 if (cb_mutex) {
618 nlk->cb_mutex = cb_mutex;
619 } else {
620 nlk->cb_mutex = &nlk->cb_def_mutex;
621 mutex_init(nlk->cb_mutex);
622 lockdep_set_class_and_name(nlk->cb_mutex,
623 nlk_cb_mutex_keys + protocol,
624 nlk_cb_mutex_key_strings[protocol]);
625 }
626 init_waitqueue_head(&nlk->wait);
627
628 sk->sk_destruct = netlink_sock_destruct;
629 sk->sk_protocol = protocol;
630 return 0;
631}

633static int netlink_create(struct net *net, struct socket *sock, int protocol,
634 int kern)
635{
636 struct module *module = NULL;
637 struct mutex *cb_mutex;
638 struct netlink_sock *nlk;
639 int (*bind)(struct net *net, int group);
640 void (*unbind)(struct net *net, int group);
641 int err = 0;
642
643 sock->state = SS_UNCONNECTED;
644
645 if (sock->type != SOCK_RAW && sock->type != SOCK_DGRAM)
646 return -ESOCKTNOSUPPORT;
647
648 if (protocol < 0 || protocol >= MAX_LINKS)
649 return -EPROTONOSUPPORT;
650
651 netlink_lock_table();
652#ifdef CONFIG_MODULES
653 if (!nl_table[protocol].registered) {
654 netlink_unlock_table();
655 request_module("net-pf-%d-proto-%d", PF_NETLINK, protocol);
656 netlink_lock_table();
657 }
658#endif
659 if (nl_table[protocol].registered &&
660 try_module_get(nl_table[protocol].module))
661 module = nl_table[protocol].module;
662 else
663 err = -EPROTONOSUPPORT;
664 cb_mutex = nl_table[protocol].cb_mutex;
665 bind = nl_table[protocol].bind;
666 unbind = nl_table[protocol].unbind;
667 netlink_unlock_table();
668
669 if (err < 0)
670 goto out;
671
672 err = __netlink_create(net, sock, cb_mutex, protocol, kern);
673 if (err < 0)
674 goto out_module;
675
676 local_bh_disable();
677 sock_prot_inuse_add(net, &netlink_proto, 1);
678 local_bh_enable();
679
680 nlk = nlk_sk(sock->sk);
681 nlk->module = module;
682 nlk->netlink_bind = bind;
683 nlk->netlink_unbind = unbind;
684out:
685 return err;
686
687out_module:
688 module_put(module);
689 goto out;
690}

692static void deferred_put_nlk_sk(struct rcu_head *head)
693{
694 struct netlink_sock *nlk = container_of(head, struct netlink_sock, rcu);
695 struct sock *sk = &nlk->sk;
696
697 kfree(nlk->groups);
698 nlk->groups = NULL;
699
700 if (!refcount_dec_and_test(&sk->sk_refcnt))
701 return;
702
703 if (nlk->cb_running && nlk->cb.done) {
704 INIT_WORK(&nlk->work, netlink_sock_destruct_work);
705 schedule_work(&nlk->work);
706 return;
707 }
708
709 sk_free(sk);
710}

712static int netlink_release(struct socket *sock)
713{
714 struct sock *sk = sock->sk;
715 struct netlink_sock *nlk;
716
717 if (!sk)
718 return 0;
719
720 netlink_remove(sk);
721 sock_orphan(sk);
722 nlk = nlk_sk(sk);
723
724 /*
725 * OK. Socket is unlinked, any packets that arrive now
726 * will be purged.
727 */
728
729 /* must not acquire netlink_table_lock in any way again before unbind
730 * and notifying genetlink is done as otherwise it might deadlock
731 */
732 if (nlk->netlink_unbind) {
733 int i;
734
735 for (i = 0; i < nlk->ngroups; i++)
736 if (test_bit(i, nlk->groups))
737 nlk->netlink_unbind(sock_net(sk), i + 1);
738 }
739 if (sk->sk_protocol == NETLINK_GENERIC &&
740 atomic_dec_return(&genl_sk_destructing_cnt) == 0)
741 wake_up(&genl_sk_destructing_waitq);
742
743 sock->sk = NULL;
744 wake_up_interruptible_all(&nlk->wait);
745
746 skb_queue_purge(&sk->sk_write_queue);
747
748 if (nlk->portid && nlk->bound) {
749 struct netlink_notify n = {
750 .net = sock_net(sk),
751 .protocol = sk->sk_protocol,
752 .portid = nlk->portid,
753 };
754 blocking_notifier_call_chain(&netlink_chain,
755 NETLINK_URELEASE, &n);
756 }
757
758 module_put(nlk->module);
759
760 if (netlink_is_kernel(sk)) {
761 netlink_table_grab();
762 BUG_ON(nl_table[sk->sk_protocol].registered == 0);
763 if (--nl_table[sk->sk_protocol].registered == 0) {
764 struct listeners *old;
765
766 old = nl_deref_protected(nl_table[sk->sk_protocol].listeners);
767 RCU_INIT_POINTER(nl_table[sk->sk_protocol].listeners, NULL);
768 kfree_rcu(old, rcu);
769 nl_table[sk->sk_protocol].module = NULL;
770 nl_table[sk->sk_protocol].bind = NULL;
771 nl_table[sk->sk_protocol].unbind = NULL;
772 nl_table[sk->sk_protocol].flags = 0;
773 nl_table[sk->sk_protocol].registered = 0;
774 }
775 netlink_table_ungrab();
776 }
777
778 local_bh_disable();
779 sock_prot_inuse_add(sock_net(sk), &netlink_proto, -1);
780 local_bh_enable();
781 call_rcu(&nlk->rcu, deferred_put_nlk_sk);
782 return 0;
783}

785static int netlink_autobind(struct socket *sock)
786{
787 struct sock *sk = sock->sk;
788 struct net *net = sock_net(sk);
789 struct netlink_table *table = &nl_table[sk->sk_protocol];
790 s32 portid = task_tgid_vnr(current);
791 int err;
792 s32 rover = -4096;
793 bool ok;
794
795retry:
796 cond_resched();
797 rcu_read_lock();
798 ok = !__netlink_lookup(table, portid, net);
799 rcu_read_unlock();
800 if (!ok) {
801 /* Bind collision, search negative portid values. */
802 if (rover == -4096)
803 /* rover will be in range [S32_MIN, -4097] */
804 rover = S32_MIN + prandom_u32_max(-4096 - S32_MIN);
805 else if (rover >= -4096)
806 rover = -4097;
807 portid = rover--;
808 goto retry;
809 }
810
811 err = netlink_insert(sk, portid);
812 if (err == -EADDRINUSE)
813 goto retry;
814
815 /* If 2 threads race to autobind, that is fine. */
816 if (err == -EBUSY)
817 err = 0;
818
819 return err;
820}

954static int netlink_bind(struct socket *sock, struct sockaddr *addr,
955 int addr_len)
956{
957 struct sock *sk = sock->sk;
958 struct net *net = sock_net(sk);
959 struct netlink_sock *nlk = nlk_sk(sk);
960 struct sockaddr_nl *nladdr = (struct sockaddr_nl *)addr;
961 int err = 0;
962 long unsigned int groups = nladdr->nl_groups;
963 bool bound;
964
965 if (addr_len < sizeof(struct sockaddr_nl))
966 return -EINVAL;
967
968 if (nladdr->nl_family != AF_NETLINK)
969 return -EINVAL;
970
971 /* Only superuser is allowed to listen multicasts */
972 if (groups) {
973 if (!netlink_allowed(sock, NL_CFG_F_NONROOT_RECV))
974 return -EPERM;
975 err = netlink_realloc_groups(sk);
976 if (err)
977 return err;
978 }
979
980 bound = nlk->bound;
981 if (bound) {
982 /* Ensure nlk->portid is up-to-date. */
983 smp_rmb();
984
985 if (nladdr->nl_pid != nlk->portid)
986 return -EINVAL;
987 }
988
989 netlink_lock_table();
990 if (nlk->netlink_bind && groups) {
991 int group;
992
993 for (group = 0; group < nlk->ngroups; group++) {
994 if (!test_bit(group, &groups))
995 continue;
996 err = nlk->netlink_bind(net, group + 1);
997 if (!err)
998 continue;
999 netlink_undo_bind(group, groups, sk);
1000 goto unlock;
1001 }
1002 }
1003
1004 /* No need for barriers here as we return to user-space without
1005 * using any of the bound attributes.
1006 */
1007 if (!bound) {
1008 err = nladdr->nl_pid ?
1009 netlink_insert(sk, nladdr->nl_pid) :
1010 netlink_autobind(sock);
1011 if (err) {
1012 netlink_undo_bind(nlk->ngroups, groups, sk);
1013 goto unlock;
1014 }
1015 }
1016
1017 if (!groups && (nlk->groups == NULL || !(u32)nlk->groups[0]))
1018 goto unlock;
1019 netlink_unlock_table();
1020
1021 netlink_table_grab();
1022 netlink_update_subscriptions(sk, nlk->subscriptions +
1023 hweight32(groups) -
1024 hweight32(nlk->groups[0]));
1025 nlk->groups[0] = (nlk->groups[0] & ~0xffffffffUL) | groups;
1026 netlink_update_listeners(sk);
1027 netlink_table_ungrab();
1028
1029 return 0;
1030
1031unlock:
1032 netlink_unlock_table();
1033 return err;
1034}

1036static int netlink_connect(struct socket *sock, struct sockaddr *addr,
1037 int alen, int flags)
1038{
1039 int err = 0;
1040 struct sock *sk = sock->sk;
1041 struct netlink_sock *nlk = nlk_sk(sk);
1042 struct sockaddr_nl *nladdr = (struct sockaddr_nl *)addr;
1043
1044 if (alen < sizeof(addr->sa_family))
1045 return -EINVAL;
1046
1047 if (addr->sa_family == AF_UNSPEC) {
1048 sk->sk_state = NETLINK_UNCONNECTED;
1049 nlk->dst_portid = 0;
1050 nlk->dst_group = 0;
1051 return 0;
1052 }
1053 if (addr->sa_family != AF_NETLINK)
1054 return -EINVAL;
1055
1056 if (alen < sizeof(struct sockaddr_nl))
1057 return -EINVAL;
1058
1059 if ((nladdr->nl_groups || nladdr->nl_pid) &&
1060 !netlink_allowed(sock, NL_CFG_F_NONROOT_SEND))
1061 return -EPERM;
1062
1063 /* No need for barriers here as we return to user-space without
1064 * using any of the bound attributes.
1065 */
1066 if (!nlk->bound)
1067 err = netlink_autobind(sock);
1068
1069 if (err == 0) {
1070 sk->sk_state = NETLINK_CONNECTED;
1071 nlk->dst_portid = nladdr->nl_pid;
1072 nlk->dst_group = ffs(nladdr->nl_groups);
1073 }
1074
1075 return err;
1076}

1169/*
1170 * Attach a skb to a netlink socket.
1171 * The caller must hold a reference to the destination socket. On error, the
1172 * reference is dropped. The skb is not send to the destination, just all
1173 * all error checks are performed and memory in the queue is reserved.
1174 * Return values:
1175 * < 0: error. skb freed, reference to sock dropped.
1176 * 0: continue
1177 * 1: repeat lookup - reference dropped while waiting for socket memory.
1178 */
1179int netlink_attachskb(struct sock *sk, struct sk_buff *skb,
1180 long *timeo, struct sock *ssk)
1181{
1182 struct netlink_sock *nlk;
1183
1184 nlk = nlk_sk(sk);
1185
1186 if ((atomic_read(&sk->sk_rmem_alloc) > sk->sk_rcvbuf ||
1187 test_bit(NETLINK_S_CONGESTED, &nlk->state))) {
1188 DECLARE_WAITQUEUE(wait, current);
1189 if (!*timeo) {
1190 if (!ssk || netlink_is_kernel(ssk))
1191 netlink_overrun(sk);
1192 sock_put(sk);
1193 kfree_skb(skb);
1194 return -EAGAIN;
1195 }
1196
1197 __set_current_state(TASK_INTERRUPTIBLE);
1198 add_wait_queue(&nlk->wait, &wait);
1199
1200 if ((atomic_read(&sk->sk_rmem_alloc) > sk->sk_rcvbuf ||
1201 test_bit(NETLINK_S_CONGESTED, &nlk->state)) &&
1202 !sock_flag(sk, SOCK_DEAD))
1203 *timeo = schedule_timeout(*timeo);
1204
1205 __set_current_state(TASK_RUNNING);
1206 remove_wait_queue(&nlk->wait, &wait);
1207 sock_put(sk);
1208
1209 if (signal_pending(current)) {
1210 kfree_skb(skb);
1211 return sock_intr_errno(*timeo);
1212 }
1213 return 1;
1214 }
1215 netlink_skb_set_owner_r(skb, sk);
1216 return 0;
1217}

1244static struct sk_buff *netlink_trim(struct sk_buff *skb, gfp_t allocation)
1245{
1246 int delta;
1247
1248 WARN_ON(skb->sk != NULL);
1249 delta = skb->end - skb->tail;
1250 if (is_vmalloc_addr(skb->head) || delta * 2 < skb->truesize)
1251 return skb;
1252
1253 if (skb_shared(skb)) {
1254 struct sk_buff *nskb = skb_clone(skb, allocation);
1255 if (!nskb)
1256 return skb;
1257 consume_skb(skb);
1258 skb = nskb;
1259 }
1260
1261 pskb_expand_head(skb, 0, -delta,
1262 (allocation & ~__GFP_DIRECT_RECLAIM) |
1263 __GFP_NOWARN | __GFP_NORETRY);
1264 return skb;
1265}

1371static void do_one_broadcast(struct sock *sk,
1372 struct netlink_broadcast_data *p)
1373{
1374 struct netlink_sock *nlk = nlk_sk(sk);
1375 int val;
1376
1377 if (p->exclude_sk == sk)
1378 return;
1379
1380 if (nlk->portid == p->portid || p->group - 1 >= nlk->ngroups ||
1381 !test_bit(p->group - 1, nlk->groups))
1382 return;
1383
1384 if (!net_eq(sock_net(sk), p->net)) {
1385 if (!(nlk->flags & NETLINK_F_LISTEN_ALL_NSID))
1386 return;
1387
1388 if (!peernet_has_id(sock_net(sk), p->net))
1389 return;
1390
1391 if (!file_ns_capable(sk->sk_socket->file, p->net->user_ns,
1392 CAP_NET_BROADCAST))
1393 return;
1394 }
1395
1396 if (p->failure) {
1397 netlink_overrun(sk);
1398 return;
1399 }
1400
1401 sock_hold(sk);
1402 if (p->skb2 == NULL) {
1403 if (skb_shared(p->skb)) {
1404 p->skb2 = skb_clone(p->skb, p->allocation);
1405 } else {
1406 p->skb2 = skb_get(p->skb);
1407 /*
1408 * skb ownership may have been set when
1409 * delivered to a previous socket.
1410 */
1411 skb_orphan(p->skb2);
1412 }
1413 }
1414 if (p->skb2 == NULL) {
1415 netlink_overrun(sk);
1416 /* Clone failed. Notify ALL listeners. */
1417 p->failure = 1;
1418 if (nlk->flags & NETLINK_F_BROADCAST_SEND_ERROR)
1419 p->delivery_failure = 1;
1420 goto out;
1421 }
1422 if (p->tx_filter && p->tx_filter(sk, p->skb2, p->tx_data)) {
1423 kfree_skb(p->skb2);
1424 p->skb2 = NULL;
1425 goto out;
1426 }
1427 if (sk_filter(sk, p->skb2)) {
1428 kfree_skb(p->skb2);
1429 p->skb2 = NULL;
1430 goto out;
1431 }
1432 NETLINK_CB(p->skb2).nsid = peernet2id(sock_net(sk), p->net);
1433 if (NETLINK_CB(p->skb2).nsid != NETNSA_NSID_NOT_ASSIGNED)
1434 NETLINK_CB(p->skb2).nsid_is_set = true;
1435 val = netlink_broadcast_deliver(sk, p->skb2);
1436 if (val < 0) {
1437 netlink_overrun(sk);
1438 if (nlk->flags & NETLINK_F_BROADCAST_SEND_ERROR)
1439 p->delivery_failure = 1;
1440 } else {
1441 p->congested |= val;
1442 p->delivered = 1;
1443 p->skb2 = NULL;
1444 }
1445out:
1446 sock_put(sk);

1461 info.net = net;
1462 info.portid = portid;

1495 }

1590static int netlink_setsockopt(struct socket *sock, int level, int optname,
1591 char __user *optval, unsigned int optlen)
1592{
1593 struct sock *sk = sock->sk;
1594 struct netlink_sock *nlk = nlk_sk(sk);
1595 unsigned int val = 0;
1596 int err;
1597
1598 if (level != SOL_NETLINK)
1599 return -ENOPROTOOPT;
1600
1601 if (optlen >= sizeof(int) &&
1602 get_user(val, (unsigned int __user *)optval))
1603 return -EFAULT;
1604
1605 switch (optname) {
1606 case NETLINK_PKTINFO:
1607 if (val)
1608 nlk->flags |= NETLINK_F_RECV_PKTINFO;
1609 else
1610 nlk->flags &= ~NETLINK_F_RECV_PKTINFO;
1611 err = 0;
1612 break;
1613 case NETLINK_ADD_MEMBERSHIP:
1614 case NETLINK_DROP_MEMBERSHIP: {
1615 if (!netlink_allowed(sock, NL_CFG_F_NONROOT_RECV))
1616 return -EPERM;
1617 err = netlink_realloc_groups(sk);
1618 if (err)
1619 return err;
1620 if (!val || val - 1 >= nlk->ngroups)
1621 return -EINVAL;
1622 if (optname == NETLINK_ADD_MEMBERSHIP && nlk->netlink_bind) {
1623 err = nlk->netlink_bind(sock_net(sk), val);
1624 if (err)
1625 return err;
1626 }
1627 netlink_table_grab();
1628 netlink_update_socket_mc(nlk, val,
1629 optname == NETLINK_ADD_MEMBERSHIP);
1630 netlink_table_ungrab();
1631 if (optname == NETLINK_DROP_MEMBERSHIP && nlk->netlink_unbind)
1632 nlk->netlink_unbind(sock_net(sk), val);
1633
1634 err = 0;
1635 break;
1636 }
1637 case NETLINK_BROADCAST_ERROR:
1638 if (val)
1639 nlk->flags |= NETLINK_F_BROADCAST_SEND_ERROR;
1640 else
1641 nlk->flags &= ~NETLINK_F_BROADCAST_SEND_ERROR;
1642 err = 0;
1643 break;
1644 case NETLINK_NO_ENOBUFS:
1645 if (val) {
1646 nlk->flags |= NETLINK_F_RECV_NO_ENOBUFS;
1647 clear_bit(NETLINK_S_CONGESTED, &nlk->state);
1648 wake_up_interruptible(&nlk->wait);
1649 } else {
1650 nlk->flags &= ~NETLINK_F_RECV_NO_ENOBUFS;
1651 }
1652 err = 0;
1653 break;
1654 case NETLINK_LISTEN_ALL_NSID:
1655 if (!ns_capable(sock_net(sk)->user_ns, CAP_NET_BROADCAST))
1656 return -EPERM;
1657
1658 if (val)
1659 nlk->flags |= NETLINK_F_LISTEN_ALL_NSID;
1660 else
1661 nlk->flags &= ~NETLINK_F_LISTEN_ALL_NSID;
1662 err = 0;
1663 break;
1664 case NETLINK_CAP_ACK:
1665 if (val)
1666 nlk->flags |= NETLINK_F_CAP_ACK;
1667 else
1668 nlk->flags &= ~NETLINK_F_CAP_ACK;
1669 err = 0;
1670 break;
1671 case NETLINK_EXT_ACK:
1672 if (val)
1673 nlk->flags |= NETLINK_F_EXT_ACK;
1674 else
1675 nlk->flags &= ~NETLINK_F_EXT_ACK;
1676 err = 0;
1677 break;
1678 default:
1679 err = -ENOPROTOOPT;
1680 }
1681 return err;
1682}

1684static int netlink_getsockopt(struct socket *sock, int level, int optname,
1685 char __user *optval, int __user *optlen)
1686{
1687 struct sock *sk = sock->sk;
1688 struct netlink_sock *nlk = nlk_sk(sk);
1689 int len, val, err;
1690
1691 if (level != SOL_NETLINK)
1692 return -ENOPROTOOPT;
1693
1694 if (get_user(len, optlen))
1695 return -EFAULT;
1696 if (len < 0)
1697 return -EINVAL;
1698
1699 switch (optname) {
1700 case NETLINK_PKTINFO:
1701 if (len < sizeof(int))
1702 return -EINVAL;
1703 len = sizeof(int);
1704 val = nlk->flags & NETLINK_F_RECV_PKTINFO ? 1 : 0;
1705 if (put_user(len, optlen) ||
1706 put_user(val, optval))
1707 return -EFAULT;
1708 err = 0;
1709 break;
1710 case NETLINK_BROADCAST_ERROR:
1711 if (len < sizeof(int))
1712 return -EINVAL;
1713 len = sizeof(int);
1714 val = nlk->flags & NETLINK_F_BROADCAST_SEND_ERROR ? 1 : 0;
1715 if (put_user(len, optlen) ||
1716 put_user(val, optval))
1717 return -EFAULT;
1718 err = 0;
1719 break;
1720 case NETLINK_NO_ENOBUFS:
1721 if (len < sizeof(int))
1722 return -EINVAL;
1723 len = sizeof(int);
1724 val = nlk->flags & NETLINK_F_RECV_NO_ENOBUFS ? 1 : 0;
1725 if (put_user(len, optlen) ||
1726 put_user(val, optval))
1727 return -EFAULT;
1728 err = 0;
1729 break;
1730 case NETLINK_LIST_MEMBERSHIPS: {
1731 int pos, idx, shift;
1732
1733 err = 0;
1734 netlink_lock_table();
1735 for (pos = 0; pos * 8 < nlk->ngroups; pos += sizeof(u32)) {
1736 if (len - pos < sizeof(u32))
1737 break;
1738
1739 idx = pos / sizeof(unsigned long);
1740 shift = (pos % sizeof(unsigned long)) * 8;
1741 if (put_user((u32)(nlk->groups[idx] >> shift),
1742 (u32 __user *)(optval + pos))) {
1743 err = -EFAULT;
1744 break;
1745 }
1746 }
1747 if (put_user(ALIGN(nlk->ngroups / 8, sizeof(u32)), optlen))
1748 err = -EFAULT;
1749 netlink_unlock_table();
1750 break;
1751 }
1752 case NETLINK_CAP_ACK:
1753 if (len < sizeof(int))
1754 return -EINVAL;
1755 len = sizeof(int);
1756 val = nlk->flags & NETLINK_F_CAP_ACK ? 1 : 0;
1757 if (put_user(len, optlen) ||
1758 put_user(val, optval))
1759 return -EFAULT;
1760 err = 0;
1761 break;
1762 case NETLINK_EXT_ACK:
1763 if (len < sizeof(int))
1764 return -EINVAL;
1765 len = sizeof(int);
1766 val = nlk->flags & NETLINK_F_EXT_ACK ? 1 : 0;
1767 if (put_user(len, optlen) || put_user(val, optval))
1768 return -EFAULT;
1769 err = 0;
1770 break;
1771 default:
1772 err = -ENOPROTOOPT;
1773 }
1774 return err;
1775}

1795static int netlink_sendmsg(struct socket *sock, struct msghdr *msg, size_t len)
1796{
1797 struct sock *sk = sock->sk;
1798 struct netlink_sock *nlk = nlk_sk(sk);
1799 DECLARE_SOCKADDR(struct sockaddr_nl *, addr, msg->msg_name);
1800 u32 dst_portid;
1801 u32 dst_group;
1802 struct sk_buff *skb;
1803 int err;
1804 struct scm_cookie scm;
1805 u32 netlink_skb_flags = 0;
1806
1807 if (msg->msg_flags&MSG_OOB)
1808 return -EOPNOTSUPP;
1809
1810 err = scm_send(sock, msg, &scm, true);
1811 if (err < 0)
1812 return err;
1813
1814 if (msg->msg_namelen) {
1815 err = -EINVAL;
1816 if (msg->msg_namelen < sizeof(struct sockaddr_nl))
1817 goto out;
1818 if (addr->nl_family != AF_NETLINK)
1819 goto out;
1820 dst_portid = addr->nl_pid;
1821 dst_group = ffs(addr->nl_groups);
1822 err = -EPERM;
1823 if ((dst_group || dst_portid) &&
1824 !netlink_allowed(sock, NL_CFG_F_NONROOT_SEND))
1825 goto out;
1826 netlink_skb_flags |= NETLINK_SKB_DST;
1827 } else {
1828 dst_portid = nlk->dst_portid;
1829 dst_group = nlk->dst_group;
1830 }
1831
1832 if (!nlk->bound) {
1833 err = netlink_autobind(sock);
1834 if (err)
1835 goto out;
1836 } else {
1837 /* Ensure nlk is hashed and visible. */
1838 smp_rmb();
1839 }
1840
1841 err = -EMSGSIZE;
1842 if (len > sk->sk_sndbuf - 32)
1843 goto out;
1844 err = -ENOBUFS;
1845 skb = netlink_alloc_large_skb(len, dst_group);
1846 if (skb == NULL)
1847 goto out;
1848
1849 NETLINK_CB(skb).portid = nlk->portid;
1850 NETLINK_CB(skb).dst_group = dst_group;
1851 NETLINK_CB(skb).creds = scm.creds;
1852 NETLINK_CB(skb).flags = netlink_skb_flags;
1853
1854 err = -EFAULT;
1855 if (memcpy_from_msg(skb_put(skb, len), msg, len)) {
1856 kfree_skb(skb);
1857 goto out;
1858 }
1859
1860 err = security_netlink_send(sk, skb);
1861 if (err) {
1862 kfree_skb(skb);
1863 goto out;
1864 }
1865
1866 if (dst_group) {
1867 refcount_inc(&skb->users);
1868 netlink_broadcast(sk, skb, dst_portid, dst_group, GFP_KERNEL);
1869 }
1870 err = netlink_unicast(sk, skb, dst_portid, msg->msg_flags&MSG_DONTWAIT);
1871
1872out:
1873 scm_destroy(&scm);
1874 return err;
1875}

1877static int netlink_recvmsg(struct socket *sock, struct msghdr *msg, size_t len,
1878 int flags)
1879{
1880 struct scm_cookie scm;
1881 struct sock *sk = sock->sk;
1882 struct netlink_sock *nlk = nlk_sk(sk);
1883 int noblock = flags&MSG_DONTWAIT;
1884 size_t copied;
1885 struct sk_buff *skb, *data_skb;
1886 int err, ret;
1887
1888 if (flags&MSG_OOB)
1889 return -EOPNOTSUPP;
1890
1891 copied = 0;
1892
1893 skb = skb_recv_datagram(sk, flags, noblock, &err);
1894 if (skb == NULL)
1895 goto out;
1896
1897 data_skb = skb;
1898
1899#ifdef CONFIG_COMPAT_NETLINK_MESSAGES
1900 if (unlikely(skb_shinfo(skb)->frag_list)) {
1901 /*
1902 * If this skb has a frag_list, then here that means that we
1903 * will have to use the frag_list skb's data for compat tasks
1904 * and the regular skb's data for normal (non-compat) tasks.
1905 *
1906 * If we need to send the compat skb, assign it to the
1907 * 'data_skb' variable so that it will be used below for data
1908 * copying. We keep 'skb' for everything else, including
1909 * freeing both later.
1910 */
1911 if (flags & MSG_CMSG_COMPAT)
1912 data_skb = skb_shinfo(skb)->frag_list;
1913 }
1914#endif
1915
1916 /* Record the max length of recvmsg() calls for future allocations */
1917 nlk->max_recvmsg_len = max(nlk->max_recvmsg_len, len);
1918 nlk->max_recvmsg_len = min_t(size_t, nlk->max_recvmsg_len,
1919 SKB_WITH_OVERHEAD(32768));
1920
1921 copied = data_skb->len;
1922 if (len < copied) {
1923 msg->msg_flags |= MSG_TRUNC;
1924 copied = len;
1925 }
1926
1927 skb_reset_transport_header(data_skb);
1928 err = skb_copy_datagram_msg(data_skb, 0, msg, copied);
1929
1930 if (msg->msg_name) {
1931 DECLARE_SOCKADDR(struct sockaddr_nl *, addr, msg->msg_name);
1932 addr->nl_family = AF_NETLINK;
1933 addr->nl_pad = 0;
1934 addr->nl_pid = NETLINK_CB(skb).portid;
1935 addr->nl_groups = netlink_group_mask(NETLINK_CB(skb).dst_group);
1936 msg->msg_namelen = sizeof(*addr);
1937 }
1938
1939 if (nlk->flags & NETLINK_F_RECV_PKTINFO)
1940 netlink_cmsg_recv_pktinfo(msg, skb);
1941 if (nlk->flags & NETLINK_F_LISTEN_ALL_NSID)
1942 netlink_cmsg_listen_all_nsid(sk, msg, skb);
1943
1944 memset(&scm, 0, sizeof(scm));
1945 scm.creds = *NETLINK_CREDS(skb);
1946 if (flags & MSG_TRUNC)
1947 copied = data_skb->len;
1948
1949 skb_free_datagram(sk, skb);
1950
1951 if (nlk->cb_running &&
1952 atomic_read(&sk->sk_rmem_alloc) <= sk->sk_rcvbuf / 2) {
1953 ret = netlink_dump(sk);
1954 if (ret) {
1955 sk->sk_err = -ret;
1956 sk->sk_error_report(sk);
1957 }
1958 }
1959
1960 scm_recv(sock, msg, &scm, flags);
1961out:
1962 netlink_rcv_wake(sk);
1963 return err ? : copied;
1964}

1978__netlink_kernel_create(struct net *net, int unit, struct module *module,
1979 struct netlink_kernel_cfg *cfg)
1980{
1981 struct socket *sock;
1982 struct sock *sk;
1983 struct netlink_sock *nlk;
1984 struct listeners *listeners = NULL;
1985 struct mutex *cb_mutex = cfg ? cfg->cb_mutex : NULL;
1986 unsigned int groups;
1987
1988 BUG_ON(!nl_table);
1989
1990 if (unit < 0 || unit >= MAX_LINKS)
1991 return NULL;
1992
1993 if (sock_create_lite(PF_NETLINK, SOCK_DGRAM, unit, &sock))
1994 return NULL;
1995
1996 if (__netlink_create(net, sock, cb_mutex, unit, 1) < 0)
1997 goto out_sock_release_nosk;
1998
1999 sk = sock->sk;
2000
2001 if (!cfg || cfg->groups < 32)
2002 groups = 32;
2003 else
2004 groups = cfg->groups;
2005
2006 listeners = kzalloc(sizeof(*listeners) + NLGRPSZ(groups), GFP_KERNEL);
2007 if (!listeners)
2008 goto out_sock_release;
2009
2010 sk->sk_data_ready = netlink_data_ready;
2011 if (cfg && cfg->input)
2012 nlk_sk(sk)->netlink_rcv = cfg->input;
2013
2014 if (netlink_insert(sk, 0))
2015 goto out_sock_release;
2016
2017 nlk = nlk_sk(sk);
2018 nlk->flags |= NETLINK_F_KERNEL_SOCKET;
2019
2020 netlink_table_grab();
2021 if (!nl_table[unit].registered) {
2022 nl_table[unit].groups = groups;
2023 rcu_assign_pointer(nl_table[unit].listeners, listeners);
2024 nl_table[unit].cb_mutex = cb_mutex;
2025 nl_table[unit].module = module;
2026 if (cfg) {
2027 nl_table[unit].bind = cfg->bind;
2028 nl_table[unit].unbind = cfg->unbind;
2029 nl_table[unit].flags = cfg->flags;
2030 if (cfg->compare)
2031 nl_table[unit].compare = cfg->compare;
2032 }
2033 nl_table[unit].registered = 1;
2034 } else {
2035 kfree(listeners);
2036 nl_table[unit].registered++;
2037 }
2038 netlink_table_ungrab();
2039 return sk;
2040
2041out_sock_release:
2042 kfree(listeners);
2043 netlink_kernel_release(sk);
2044 return NULL;
2045
2046out_sock_release_nosk:
2047 sock_release(sock);
2048 return NULL;
2049}

2147 int err = -ENOBUFS;
2148 int alloc_min_size;
2149 int alloc_size;
2150
2151 mutex_lock(nlk->cb_mutex);
2152 if (!nlk->cb_running) {
2153 err = -EINVAL;
2154 goto errout_skb;
2155 }
2156
2157 if (atomic_read(&sk->sk_rmem_alloc) >= sk->sk_rcvbuf)
2158 goto errout_skb;
2159
2160 /* NLMSG_GOODSIZE is small to avoid high order allocations being
2161 * required, but it makes sense to _attempt_ a 16K bytes allocation
2162 * to reduce number of system calls on dump operations, if user
2163 * ever provided a big enough buffer.
2164 */
2165 cb = &nlk->cb;
2166 alloc_min_size = max_t(int, cb->min_dump_alloc, NLMSG_GOODSIZE);
2167
2168 if (alloc_min_size < nlk->max_recvmsg_len) {
2169 alloc_size = nlk->max_recvmsg_len;
2170 skb = alloc_skb(alloc_size,
2171 (GFP_KERNEL & ~__GFP_DIRECT_RECLAIM) |
2172 __GFP_NOWARN | __GFP_NORETRY);
2173 }
2174 if (!skb) {
2175 alloc_size = alloc_min_size;
2176 skb = alloc_skb(alloc_size, GFP_KERNEL);
2177 }
2178 if (!skb)

2181 /* Trim skb to allocated size. User is expected to provide buffer as
2182 * large as max(min_dump_alloc, 16KiB (mac_recvmsg_len capped at
2183 * netlink_recvmsg())). dump will pack as many smaller messages as
2184 * could fit within the allocated skb. skb is typically allocated
2185 * with larger space than required (could be as much as near 2x the
2186 * requested size with align to next power of 2 approach). Allowing
2187 * dump to use the excess space makes it difficult for a user to have a
2188 * reasonable static buffer based on the expected largest dump of a
2189 * single netdev. The outcome is MSG_TRUNC error.
2190 */
2191 skb_reserve(skb, skb_tailroom(skb) - alloc_size);
2192 netlink_skb_set_owner_r(skb, sk);
2193
2194 if (nlk->dump_done_errno > 0)
2195 nlk->dump_done_errno = cb->dump(skb, cb);
2196
2197 if (nlk->dump_done_errno > 0 ||
2198 skb_tailroom(skb) < nlmsg_total_size(sizeof(nlk->dump_done_errno))) {
2199 mutex_unlock(nlk->cb_mutex);
2200
2201 if (sk_filter(sk, skb))
2202 kfree_skb(skb);
2203 else
2204 __netlink_sendskb(sk, skb);
2205 return 0;
2206 }
2207
2208 nlh = nlmsg_put_answer(skb, cb, NLMSG_DONE,
2209 sizeof(nlk->dump_done_errno), NLM_F_MULTI);
2210 if (WARN_ON(!nlh))
2211 goto errout_skb;
2212
2213 nl_dump_check_consistent(cb, nlh);
2214
2215 memcpy(nlmsg_data(nlh), &nlk->dump_done_errno,
2216 sizeof(nlk->dump_done_errno));
2217
2218 if (sk_filter(sk, skb))
2219 kfree_skb(skb);
2220 else
2221 __netlink_sendskb(sk, skb);
2222
2223 if (cb->done)
2224 cb->done(cb);
2225
2226 nlk->cb_running = false;
2227 module = cb->module;
2228 skb = cb->skb;
2229 mutex_unlock(nlk->cb_mutex);
2230 module_put(module);
2231 consume_skb(skb);
2232 return 0;
2233
2234errout_skb:
2235 mutex_unlock(nlk->cb_mutex);
2236 kfree_skb(skb);
2237 return err;
2238}

2315void netlink_ack(struct sk_buff *in_skb, struct nlmsghdr *nlh, int err,
2316 const struct netlink_ext_ack *extack)
2317{

2328 * requests to cap the error message, and get extra error data if
2329 * requested.

2338 tlvlen += nla_total_size(strlen(extack->_msg) + 1);
2339 if (extack->bad_attr)
2340 tlvlen += nla_total_size(sizeof(u32));
2341 }
2342 } else {
2343 flags |= NLM_F_CAPPED;
2344
2345 if (nlk_has_extack && extack && extack->cookie_len)
2346 tlvlen += nla_total_size(extack->cookie_len);
2347 }
2348
2349 if (tlvlen)
2350 flags |= NLM_F_ACK_TLVS;

2352 skb = nlmsg_new(payload + tlvlen, GFP_KERNEL);
2353 if (!skb) {
2354 struct sock *sk;
2355
2356 sk = netlink_lookup(sock_net(in_skb->sk),
2357 in_skb->sk->sk_protocol,
2358 NETLINK_CB(in_skb).portid);

2361 sk->sk_error_report(sk);
2362 sock_put(sk);

2368 NLMSG_ERROR, payload, flags);

2370 errmsg->error = err;
2371 memcpy(&errmsg->msg, nlh, payload > sizeof(*errmsg) ? nlh->nlmsg_len : sizeof(*nlh));

2377 extack->_msg));

2379 !WARN_ON((u8 *)extack->bad_attr < in_skb->data ||
2380 (u8 *)extack->bad_attr >= in_skb->data +
2381 in_skb->len))
2382 WARN_ON(nla_put_u32(skb, NLMSGERR_ATTR_OFFS,
2383 (u8 *)extack->bad_attr -
2384 in_skb->data));
2385 } else {

2387 WARN_ON(nla_put(skb, NLMSGERR_ATTR_COOKIE,
2388 extack->cookie_len,
2389 extack->cookie));
2390 }
2391 }
2392
2393 nlmsg_end(skb, rep);
2394
2395 netlink_unicast(in_skb->sk, skb, NETLINK_CB(in_skb).portid, MSG_DONTWAIT);
2396}

2400 struct nlmsghdr *,
2401 struct netlink_ext_ack *))
2402{
2403 struct netlink_ext_ack extack;

2408 int msglen;

2432

2435 if (msglen > skb->len)
2436 msglen = skb->len;
2437 skb_pull(skb, msglen);
