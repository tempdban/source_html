26/*
27 * Handle changes in state of network devices enslaved to a bridge.
28 *
29 * Note: don't care about up/down if bridge itself is down, because
30 * port state is checked when bridge is brought up.
31 */
32static int br_device_event(struct notifier_block *unused, unsigned long event, void *ptr)
33{
34 struct net_device *dev = netdev_notifier_info_to_dev(ptr);
35 struct net_bridge_port *p;
36 struct net_bridge *br;
37 bool changed_addr;
38 int err;
39
40 /* register of bridge completed, add sysfs entries */
41 if ((dev->priv_flags & IFF_EBRIDGE) && event == NETDEV_REGISTER) {
42 br_sysfs_addbr(dev);
43 return NOTIFY_DONE;
44 }
45
46 /* not a port of a bridge */
47 p = br_port_get_rtnl(dev);
48 if (!p)
49 return NOTIFY_DONE;
50
51 br = p->br;
52
53 switch (event) {
54 case NETDEV_CHANGEMTU:
55 dev_set_mtu(br->dev, br_min_mtu(br));
56 break;
57
58 case NETDEV_CHANGEADDR:
59 spin_lock_bh(&br->lock);
60 br_fdb_changeaddr(p, dev->dev_addr);
61 changed_addr = br_stp_recalculate_bridge_id(br);
62 spin_unlock_bh(&br->lock);
63
64 if (changed_addr)
65 call_netdevice_notifiers(NETDEV_CHANGEADDR, br->dev);
66
67 break;
68
69 case NETDEV_CHANGE:
70 br_port_carrier_check(p);
71 break;
72
73 case NETDEV_FEAT_CHANGE:
74 netdev_update_features(br->dev);
75 break;
76
77 case NETDEV_DOWN:
78 spin_lock_bh(&br->lock);
79 if (br->dev->flags & IFF_UP)
80 br_stp_disable_port(p);
81 spin_unlock_bh(&br->lock);
82 break;
83
84 case NETDEV_UP:
85 if (netif_running(br->dev) && netif_oper_up(dev)) {
86 spin_lock_bh(&br->lock);
87 br_stp_enable_port(p);
88 spin_unlock_bh(&br->lock);
89 }
90 break;
91
92 case NETDEV_UNREGISTER:
93 br_del_if(br, dev);
94 break;
95
96 case NETDEV_CHANGENAME:
97 err = br_sysfs_renameif(p);
98 if (err)
99 return notifier_from_errno(err);
100 break;
101
102 case NETDEV_PRE_TYPE_CHANGE:
103 /* Forbid underlaying device to change its type. */
104 return NOTIFY_BAD;
105
106 case NETDEV_RESEND_IGMP:
107 /* Propagate to master device */
108 call_netdevice_notifiers(event, br->dev);
109 break;
110 }
111
112 /* Events that may cause spanning tree to refresh */
113 if (event == NETDEV_CHANGEADDR || event == NETDEV_UP ||
114 event == NETDEV_CHANGE || event == NETDEV_DOWN)
115 br_ifinfo_notify(RTM_NEWLINK, p);
116
117 return NOTIFY_DONE;
118}

120static struct notifier_block br_device_notifier = {
121 .notifier_call = br_device_event
122};

124/* called with RTNL or RCU */
125static int br_switchdev_event(struct notifier_block *unused,
126 unsigned long event, void *ptr)
127{
128 struct net_device *dev = switchdev_notifier_info_to_dev(ptr);
129 struct net_bridge_port *p;
130 struct net_bridge *br;
131 struct switchdev_notifier_fdb_info *fdb_info;
132 int err = NOTIFY_DONE;
133
134 p = br_port_get_rtnl_rcu(dev);
135 if (!p)
136 goto out;
137
138 br = p->br;
139
140 switch (event) {
141 case SWITCHDEV_FDB_ADD_TO_BRIDGE:
142 fdb_info = ptr;
143 err = br_fdb_external_learn_add(br, p, fdb_info->addr,
144 fdb_info->vid);
145 if (err) {
146 err = notifier_from_errno(err);
147 break;
148 }
149 br_fdb_offloaded_set(br, p, fdb_info->addr,
150 fdb_info->vid);
151 break;
152 case SWITCHDEV_FDB_DEL_TO_BRIDGE:
153 fdb_info = ptr;
154 err = br_fdb_external_learn_del(br, p, fdb_info->addr,
155 fdb_info->vid);
156 if (err)
157 err = notifier_from_errno(err);
158 break;
159 case SWITCHDEV_FDB_OFFLOADED:
160 fdb_info = ptr;
161 br_fdb_offloaded_set(br, p, fdb_info->addr,
162 fdb_info->vid);
163 break;
164 }
165
166out:
167 return err;
168}

170static struct notifier_block br_switchdev_notifier = {
171 .notifier_call = br_switchdev_event,
172};

174static void __net_exit br_net_exit(struct net *net)
175{
176 struct net_device *dev;
177 LIST_HEAD(list);
178
179 rtnl_lock();
180 for_each_netdev(net, dev)
181 if (dev->priv_flags & IFF_EBRIDGE)
182 br_dev_delete(dev, &list);
183
184 unregister_netdevice_many(&list);
185 rtnl_unlock();
186
187}

197static int __init br_init(void)
198{
199 int err;
200
201 BUILD_BUG_ON(sizeof(struct br_input_skb_cb) > FIELD_SIZEOF(struct sk_buff, cb));
202
203 err = stp_proto_register(&br_stp_proto);
204 if (err < 0) {
205 pr_err("bridge: can't register sap for STP\n");
206 return err;
207 }
208
209 err = br_fdb_init();
210 if (err)
211 goto err_out;
212
213 err = register_pernet_subsys(&br_net_ops);
214 if (err)
215 goto err_out1;
216
217 err = br_nf_core_init();
218 if (err)
219 goto err_out2;
220
221 err = register_netdevice_notifier(&br_device_notifier);
222 if (err)
223 goto err_out3;
224
225 err = register_switchdev_notifier(&br_switchdev_notifier);
226 if (err)
227 goto err_out4;
228
229 err = br_netlink_init();
230 if (err)
231 goto err_out5;
232
233 brioctl_set(br_ioctl_deviceless_stub);
234
235#if IS_ENABLED(CONFIG_ATM_LANE)
236 br_fdb_test_addr_hook = br_fdb_test_addr;
237#endif
238
239#if IS_MODULE(CONFIG_BRIDGE_NETFILTER)
240 pr_info("bridge: filtering via arp/ip/ip6tables is no longer available "
241 "by default. Update your scripts to load br_netfilter if you "
242 "need this.\n");
243#endif
244
245 return 0;
246
247err_out5:
248 unregister_switchdev_notifier(&br_switchdev_notifier);
249err_out4:
250 unregister_netdevice_notifier(&br_device_notifier);
251err_out3:
252 br_nf_core_fini();
253err_out2:
254 unregister_pernet_subsys(&br_net_ops);
255err_out1:
256 br_fdb_fini();
257err_out:
258 stp_proto_unregister(&br_stp_proto);
259 return err;
260}
261
262static void __exit br_deinit(void)
263{
264 stp_proto_unregister(&br_stp_proto);
265 br_netlink_fini();
266 unregister_switchdev_notifier(&br_switchdev_notifier);
267 unregister_netdevice_notifier(&br_device_notifier);
268 brioctl_set(NULL);
269 unregister_pernet_subsys(&br_net_ops);
270
271 rcu_barrier(); /* Wait for completion of call_rcu()'s */
272
273 br_nf_core_fini();
274#if IS_ENABLED(CONFIG_ATM_LANE)
275 br_fdb_test_addr_hook = NULL;
276#endif
277 br_fdb_fini();
278}
