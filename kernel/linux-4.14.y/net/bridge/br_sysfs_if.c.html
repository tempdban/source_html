24struct brport_attribute {
25 struct attribute attr;
26 ssize_t (*show)(struct net_bridge_port *, char *);
27 int (*store)(struct net_bridge_port *, unsigned long);
28};
29
30#define BRPORT_ATTR(_name, _mode, _show, _store) \
31const struct brport_attribute brport_attr_##_name = { \
32 .attr = {.name = __stringify(_name), \
33 .mode = _mode }, \
34 .show = _show, \
35 .store = _store, \
36};
37
38#define BRPORT_ATTR_FLAG(_name, _mask) \
39static ssize_t show_##_name(struct net_bridge_port *p, char *buf) \
40{ \
41 return sprintf(buf, "%d\n", !!(p->flags & _mask)); \
42} \
43static int store_##_name(struct net_bridge_port *p, unsigned long v) \
44{ \
45 return store_flag(p, v, _mask); \
46} \
47static BRPORT_ATTR(_name, S_IRUGO | S_IWUSR, \
48 show_##_name, store_##_name)
49
50static int store_flag(struct net_bridge_port *p, unsigned long v,
51 unsigned long mask)
52{
53 unsigned long flags;
54
55 flags = p->flags;
56
57 if (v)
58 flags |= mask;
59 else
60 flags &= ~mask;
61
62 if (flags != p->flags) {
63 p->flags = flags;
64 br_port_flags_change(p, mask);
65 }
66 return 0;
67}

229#define to_brport_attr(_at) container_of(_at, struct brport_attribute, attr)
230#define to_brport(obj) container_of(obj, struct net_bridge_port, kobj)
231
232static ssize_t brport_show(struct kobject *kobj,
233 struct attribute *attr, char *buf)
234{
235 struct brport_attribute *brport_attr = to_brport_attr(attr);
236 struct net_bridge_port *p = to_brport(kobj);
237
238 if (!brport_attr->show)
239 return -EINVAL;
240
241 return brport_attr->show(p, buf);
242}
243
244static ssize_t brport_store(struct kobject *kobj,
245 struct attribute *attr,
246 const char *buf, size_t count)
247{
248 struct brport_attribute *brport_attr = to_brport_attr(attr);
249 struct net_bridge_port *p = to_brport(kobj);
250 ssize_t ret = -EINVAL;
251 char *endp;
252 unsigned long val;
253
254 if (!ns_capable(dev_net(p->dev)->user_ns, CAP_NET_ADMIN))
255 return -EPERM;
256
257 val = simple_strtoul(buf, &endp, 0);
258 if (endp != buf) {
259 if (!rtnl_trylock())
260 return restart_syscall();
261 if (p->dev && p->br && brport_attr->store) {
262 spin_lock_bh(&p->br->lock);
263 ret = brport_attr->store(p, val);
264 spin_unlock_bh(&p->br->lock);
265 if (!ret) {
266 br_ifinfo_notify(RTM_NEWLINK, p);
267 ret = count;
268 }
269 }
270 rtnl_unlock();
271 }
272 return ret;
273}

280/*
281 * Add sysfs entries to ethernet device added to a bridge.
282 * Creates a brport subdirectory with bridge attributes.
283 * Puts symlink in bridge's brif subdirectory
284 */
285int br_sysfs_addif(struct net_bridge_port *p)
286{
287 struct net_bridge *br = p->br;
288 const struct brport_attribute **a;
289 int err;
290
291 err = sysfs_create_link(&p->kobj, &br->dev->dev.kobj,
292 SYSFS_BRIDGE_PORT_LINK);
293 if (err)
294 return err;
295
296 for (a = brport_attrs; *a; ++a) {
297 err = sysfs_create_file(&p->kobj, &((*a)->attr));
298 if (err)
299 return err;
300 }
301
302 strlcpy(p->sysfs_name, p->dev->name, IFNAMSIZ);
303 return sysfs_create_link(br->ifobj, &p->kobj, p->sysfs_name);
304}
305
306/* Rename bridge's brif symlink */
307int br_sysfs_renameif(struct net_bridge_port *p)
308{
309 struct net_bridge *br = p->br;
310 int err;
311
312 /* If a rename fails, the rollback will cause another
313 * rename call with the existing name.
314 */
315 if (!strncmp(p->sysfs_name, p->dev->name, IFNAMSIZ))
316 return 0;
317
318 err = sysfs_rename_link(br->ifobj, &p->kobj,
319 p->sysfs_name, p->dev->name);
320 if (err)
321 netdev_notice(br->dev, "unable to rename link %s to %s",
322 p->sysfs_name, p->dev->name);
323 else
324 strlcpy(p->sysfs_name, p->dev->name, IFNAMSIZ);
325
326 return err;
327}
