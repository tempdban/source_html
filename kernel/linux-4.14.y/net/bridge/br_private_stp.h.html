32struct br_config_bpdu {
33 unsigned int topology_change:1;
34 unsigned int topology_change_ack:1;
35 bridge_id root;
36 int root_path_cost;
37 bridge_id bridge_id;
38 port_id port_id;
39 int message_age;
40 int max_age;
41 int hello_time;
42 int forward_delay;
43};
44
45/* called under bridge lock */
46static inline int br_is_designated_port(const struct net_bridge_port *p)
47{
48 return !memcmp(&p->designated_bridge, &p->br->bridge_id, 8) &&
49 (p->designated_port == p->port_id);
50}
