92struct udf_part_map {
93 union {
94 struct udf_bitmap *s_bitmap;
95 struct inode *s_table;
96 } s_uspace;
97 union {
98 struct udf_bitmap *s_bitmap;
99 struct inode *s_table;
100 } s_fspace;
101 __u32 s_partition_root;
102 __u32 s_partition_len;
103 __u16 s_partition_type;
104 __u16 s_partition_num;
105 union {
106 struct udf_sparing_data s_sparing;
107 struct udf_virtual_data s_virtual;
108 struct udf_meta_data s_metadata;
109 } s_type_specific;
110 __u32 (*s_partition_func)(struct super_block *, __u32, __u16, __u32);
111 __u16 s_volumeseqnum;
112 __u16 s_partition_flags;
113};

117struct udf_sb_info {
118 struct udf_part_map *s_partmaps;
119 __u8 s_volume_ident[32];
120
121 /* Overall info */
122 __u16 s_partitions;
123 __u16 s_partition;
124
125 /* Sector headers */
126 __s32 s_session;
127 __u32 s_anchor;
128 __u32 s_last_block;
129
130 struct buffer_head *s_lvid_bh;
131
132 /* Default permissions */
133 umode_t s_umask;
134 kgid_t s_gid;

137 umode_t s_dmode;

139 rwlock_t s_cred_lock;

141 /* Root Info */
142 struct timespec s_record_time;
143
144 /* Fileset Info */
145 __u16 s_serial_number;
146
147 /* highest UDF revision we have recorded to this media */

162};
