26static int gc_thread_func(void *data)
27{
28 struct f2fs_sb_info *sbi = data;
29 struct f2fs_gc_kthread *gc_th = sbi->gc_thread;
30 wait_queue_head_t *wq = &sbi->gc_thread->gc_wait_queue_head;
31 unsigned int wait_ms;
32
33 wait_ms = gc_th->min_sleep_time;
34
35 set_freezable();
36 do {
37 wait_event_interruptible_timeout(*wq,
38 kthread_should_stop() || freezing(current) ||
39 gc_th->gc_wake,
40 msecs_to_jiffies(wait_ms));
41
42 /* give it a try one time */
43 if (gc_th->gc_wake)
44 gc_th->gc_wake = 0;
45
46 if (try_to_freeze())
47 continue;
48 if (kthread_should_stop())
49 break;
50
51 if (sbi->sb->s_writers.frozen >= SB_FREEZE_WRITE) {
52 increase_sleep_time(gc_th, &wait_ms);
53 continue;
54 }
55
56#ifdef CONFIG_F2FS_FAULT_INJECTION
57 if (time_to_inject(sbi, FAULT_CHECKPOINT)) {
58 f2fs_show_injection_info(FAULT_CHECKPOINT);
59 f2fs_stop_checkpoint(sbi, false);
60 }
61#endif
62
63 if (!sb_start_write_trylock(sbi->sb))
64 continue;
65
66 /*
67 * [GC triggering condition]
68 * 0. GC is not conducted currently.
69 * 1. There are enough dirty segments.
70 * 2. IO subsystem is idle by checking the # of writeback pages.
71 * 3. IO subsystem is idle by checking the # of requests in
72 * bdev's request list.
73 *
74 * Note) We have to avoid triggering GCs frequently.
75 * Because it is possible that some segments can be
76 * invalidated soon after by user update or deletion.
77 * So, I'd like to wait some time to collect dirty segments.
78 */
79 if (!mutex_trylock(&sbi->gc_mutex))
80 goto next;
81
82 if (gc_th->gc_urgent) {
83 wait_ms = gc_th->urgent_sleep_time;
84 goto do_gc;
85 }
86
87 if (!is_idle(sbi)) {
88 increase_sleep_time(gc_th, &wait_ms);
89 mutex_unlock(&sbi->gc_mutex);
90 goto next;
91 }
92
93 if (has_enough_invalid_blocks(sbi))
94 decrease_sleep_time(gc_th, &wait_ms);
95 else
96 increase_sleep_time(gc_th, &wait_ms);
97do_gc:
98 stat_inc_bggc_count(sbi);
99
100 /* if return value is not zero, no victim was selected */
101 if (f2fs_gc(sbi, test_opt(sbi, FORCE_FG_GC), true, NULL_SEGNO))
102 wait_ms = gc_th->no_gc_sleep_time;
103
104 trace_f2fs_background_gc(sbi->sb, wait_ms,
105 prefree_segments(sbi), free_segments(sbi));
106
107 /* balancing f2fs's metadata periodically */
108 f2fs_balance_fs_bg(sbi);
109next:
110 sb_end_write(sbi->sb);
111
112 } while (!kthread_should_stop());
113 return 0;
114}

116int start_gc_thread(struct f2fs_sb_info *sbi)
117{
118 struct f2fs_gc_kthread *gc_th;
119 dev_t dev = sbi->sb->s_bdev->bd_dev;
120 int err = 0;
121
122 gc_th = f2fs_kmalloc(sbi, sizeof(struct f2fs_gc_kthread), GFP_KERNEL);
123 if (!gc_th) {
124 err = -ENOMEM;
125 goto out;
126 }
127
128 gc_th->urgent_sleep_time = DEF_GC_THREAD_URGENT_SLEEP_TIME;
129 gc_th->min_sleep_time = DEF_GC_THREAD_MIN_SLEEP_TIME;
130 gc_th->max_sleep_time = DEF_GC_THREAD_MAX_SLEEP_TIME;
131 gc_th->no_gc_sleep_time = DEF_GC_THREAD_NOGC_SLEEP_TIME;
132
133 gc_th->gc_idle = 0;
134 gc_th->gc_urgent = 0;
135 gc_th->gc_wake= 0;
136
137 sbi->gc_thread = gc_th;
138 init_waitqueue_head(&sbi->gc_thread->gc_wait_queue_head);
139 sbi->gc_thread->f2fs_gc_task = kthread_run(gc_thread_func, sbi,
140 "f2fs_gc-%u:%u", MAJOR(dev), MINOR(dev));
141 if (IS_ERR(gc_th->f2fs_gc_task)) {
142 err = PTR_ERR(gc_th->f2fs_gc_task);
143 kfree(gc_th);
144 sbi->gc_thread = NULL;
145 }
146out:
147 return err;
148}

150void stop_gc_thread(struct f2fs_sb_info *sbi)
151{
152 struct f2fs_gc_kthread *gc_th = sbi->gc_thread;
153 if (!gc_th)
154 return;
155 kthread_stop(gc_th->f2fs_gc_task);
156 kfree(gc_th);
157 sbi->gc_thread = NULL;
158}

160static int select_gc_type(struct f2fs_gc_kthread *gc_th, int gc_type)
161{
162 int gc_mode = (gc_type == BG_GC) ? GC_CB : GC_GREEDY;
163
164 if (gc_th && gc_th->gc_idle) {
165 if (gc_th->gc_idle == 1)
166 gc_mode = GC_CB;
167 else if (gc_th->gc_idle == 2)
168 gc_mode = GC_GREEDY;
169 }
170 return gc_mode;
171}

173static void select_policy(struct f2fs_sb_info *sbi, int gc_type,
174 int type, struct victim_sel_policy *p)
175{
176 struct dirty_seglist_info *dirty_i = DIRTY_I(sbi);
177
178 if (p->alloc_mode == SSR) {
179 p->gc_mode = GC_GREEDY;
180 p->dirty_segmap = dirty_i->dirty_segmap[type];
181 p->max_search = dirty_i->nr_dirty[type];
182 p->ofs_unit = 1;
183 } else {
184 p->gc_mode = select_gc_type(sbi->gc_thread, gc_type);
185 p->dirty_segmap = dirty_i->dirty_segmap[DIRTY];
186 p->max_search = dirty_i->nr_dirty[DIRTY];
187 p->ofs_unit = sbi->segs_per_sec;
188 }
189
190 /* we need to check every dirty segments in the FG_GC case */
191 if (gc_type != FG_GC && p->max_search > sbi->max_victim_search)
192 p->max_search = sbi->max_victim_search;
193
194 /* let's select beginning hot/small space first in no_heap mode*/
195 if (test_opt(sbi, NOHEAP) &&
196 (type == CURSEG_HOT_DATA || IS_NODESEG(type)))
197 p->offset = 0;
198 else
199 p->offset = SIT_I(sbi)->last_victim[p->gc_mode];
200}

202static unsigned int get_max_cost(struct f2fs_sb_info *sbi,
203 struct victim_sel_policy *p)
204{
205 /* SSR allocates in a segment unit */
206 if (p->alloc_mode == SSR)
207 return sbi->blocks_per_seg;
208 if (p->gc_mode == GC_GREEDY)
209 return 2 * sbi->blocks_per_seg * p->ofs_unit;
210 else if (p->gc_mode == GC_CB)
211 return UINT_MAX;
212 else /* No other gc_mode */
213 return 0;
214}

216static unsigned int check_bg_victims(struct f2fs_sb_info *sbi)
217{
218 struct dirty_seglist_info *dirty_i = DIRTY_I(sbi);
219 unsigned int secno;
220
221 /*
222 * If the gc_type is FG_GC, we can select victim segments
223 * selected by background GC before.
224 * Those segments guarantee they have small valid blocks.
225 */
226 for_each_set_bit(secno, dirty_i->victim_secmap, MAIN_SECS(sbi)) {
227 if (sec_usage_check(sbi, secno))
228 continue;
229
230 if (no_fggc_candidate(sbi, secno))
231 continue;
232
233 clear_bit(secno, dirty_i->victim_secmap);
234 return GET_SEG_FROM_SEC(sbi, secno);
235 }
236 return NULL_SEGNO;
237}

239static unsigned int get_cb_cost(struct f2fs_sb_info *sbi, unsigned int segno)
240{
241 struct sit_info *sit_i = SIT_I(sbi);
242 unsigned int secno = GET_SEC_FROM_SEG(sbi, segno);
243 unsigned int start = GET_SEG_FROM_SEC(sbi, secno);
244 unsigned long long mtime = 0;
245 unsigned int vblocks;
246 unsigned char age = 0;
247 unsigned char u;
248 unsigned int i;
249
250 for (i = 0; i < sbi->segs_per_sec; i++)
251 mtime += get_seg_entry(sbi, start + i)->mtime;
252 vblocks = get_valid_blocks(sbi, segno, true);
253
254 mtime = div_u64(mtime, sbi->segs_per_sec);
255 vblocks = div_u64(vblocks, sbi->segs_per_sec);
256
257 u = (vblocks * 100) >> sbi->log_blocks_per_seg;
258
259 /* Handle if the system time has changed by the user */
260 if (mtime < sit_i->min_mtime)
261 sit_i->min_mtime = mtime;
262 if (mtime > sit_i->max_mtime)
263 sit_i->max_mtime = mtime;
264 if (sit_i->max_mtime != sit_i->min_mtime)
265 age = 100 - div64_u64(100 * (mtime - sit_i->min_mtime),
266 sit_i->max_mtime - sit_i->min_mtime);
267
268 return UINT_MAX - ((100 * (100 - u) * age) / (100 + u));
269}

271static unsigned int get_greedy_cost(struct f2fs_sb_info *sbi,
272 unsigned int segno)
273{
274 unsigned int valid_blocks =
275 get_valid_blocks(sbi, segno, true);
276
277 return IS_DATASEG(get_seg_entry(sbi, segno)->type) ?
278 valid_blocks * 2 : valid_blocks;
279}

281static inline unsigned int get_gc_cost(struct f2fs_sb_info *sbi,
282 unsigned int segno, struct victim_sel_policy *p)
283{
284 if (p->alloc_mode == SSR)
285 return get_seg_entry(sbi, segno)->ckpt_valid_blocks;
286
287 /* alloc_mode == LFS */
288 if (p->gc_mode == GC_GREEDY)
289 return get_greedy_cost(sbi, segno);
290 else
291 return get_cb_cost(sbi, segno);
292}

294static unsigned int count_bits(const unsigned long *addr,
295 unsigned int offset, unsigned int len)
296{
297 unsigned int end = offset + len, sum = 0;
298
299 while (offset < end) {
300 if (test_bit(offset++, addr))
301 ++sum;
302 }
303 return sum;
304}

306/*
307 * This function is called from two paths.
308 * One is garbage collection and the other is SSR segment selection.
309 * When it is called during GC, it just gets a victim segment
310 * and it does not remove it from dirty seglist.
311 * When it is called from SSR segment selection, it finds a segment
312 * which has minimum valid blocks and removes it from dirty seglist.
313 */
314static int get_victim_by_default(struct f2fs_sb_info *sbi,
315 unsigned int *result, int gc_type, int type, char alloc_mode)
316{
317 struct dirty_seglist_info *dirty_i = DIRTY_I(sbi);
318 struct sit_info *sm = SIT_I(sbi);
319 struct victim_sel_policy p;
320 unsigned int secno, last_victim;
321 unsigned int last_segment = MAIN_SEGS(sbi);
322 unsigned int nsearched = 0;
323
324 mutex_lock(&dirty_i->seglist_lock);
325
326 p.alloc_mode = alloc_mode;
327 select_policy(sbi, gc_type, type, &p);
328
329 p.min_segno = NULL_SEGNO;
330 p.min_cost = get_max_cost(sbi, &p);
331
332 if (*result != NULL_SEGNO) {
333 if (IS_DATASEG(get_seg_entry(sbi, *result)->type) &&
334 get_valid_blocks(sbi, *result, false) &&
335 !sec_usage_check(sbi, GET_SEC_FROM_SEG(sbi, *result)))
336 p.min_segno = *result;
337 goto out;
338 }
339
340 if (p.max_search == 0)
341 goto out;
342
343 last_victim = sm->last_victim[p.gc_mode];
344 if (p.alloc_mode == LFS && gc_type == FG_GC) {
345 p.min_segno = check_bg_victims(sbi);
346 if (p.min_segno != NULL_SEGNO)
347 goto got_it;
348 }
349
350 while (1) {
351 unsigned long cost;
352 unsigned int segno;
353
354 segno = find_next_bit(p.dirty_segmap, last_segment, p.offset);
355 if (segno >= last_segment) {
356 if (sm->last_victim[p.gc_mode]) {
357 last_segment =
358 sm->last_victim[p.gc_mode];
359 sm->last_victim[p.gc_mode] = 0;
360 p.offset = 0;
361 continue;
362 }
363 break;
364 }
365
366 p.offset = segno + p.ofs_unit;
367 if (p.ofs_unit > 1) {
368 p.offset -= segno % p.ofs_unit;
369 nsearched += count_bits(p.dirty_segmap,
370 p.offset - p.ofs_unit,
371 p.ofs_unit);
372 } else {
373 nsearched++;
374 }
375
376 secno = GET_SEC_FROM_SEG(sbi, segno);
377
378 if (sec_usage_check(sbi, secno))
379 goto next;
380 if (gc_type == BG_GC && test_bit(secno, dirty_i->victim_secmap))
381 goto next;
382 if (gc_type == FG_GC && p.alloc_mode == LFS &&
383 no_fggc_candidate(sbi, secno))
384 goto next;
385
386 cost = get_gc_cost(sbi, segno, &p);
387
388 if (p.min_cost > cost) {
389 p.min_segno = segno;
390 p.min_cost = cost;
391 }
392next:
393 if (nsearched >= p.max_search) {
394 if (!sm->last_victim[p.gc_mode] && segno <= last_victim)
395 sm->last_victim[p.gc_mode] = last_victim + 1;
396 else
397 sm->last_victim[p.gc_mode] = segno + 1;
398 sm->last_victim[p.gc_mode] %= MAIN_SEGS(sbi);
399 break;
400 }
401 }
402 if (p.min_segno != NULL_SEGNO) {
403got_it:
404 if (p.alloc_mode == LFS) {
405 secno = GET_SEC_FROM_SEG(sbi, p.min_segno);
406 if (gc_type == FG_GC)
407 sbi->cur_victim_sec = secno;
408 else
409 set_bit(secno, dirty_i->victim_secmap);
410 }
411 *result = (p.min_segno / p.ofs_unit) * p.ofs_unit;
412
413 trace_f2fs_get_victim(sbi->sb, type, gc_type, &p,
414 sbi->cur_victim_sec,
415 prefree_segments(sbi), free_segments(sbi));
416 }
417out:
418 mutex_unlock(&dirty_i->seglist_lock);
419
420 return (p.min_segno == NULL_SEGNO) ? 0 : 1;
421}

427static struct inode *find_gc_inode(struct gc_inode_list *gc_list, nid_t ino)
428{
429 struct inode_entry *ie;
430
431 ie = radix_tree_lookup(&gc_list->iroot, ino);
432 if (ie)
433 return ie->inode;
434 return NULL;
435}
436
437static void add_gc_inode(struct gc_inode_list *gc_list, struct inode *inode)
438{
439 struct inode_entry *new_ie;
440
441 if (inode == find_gc_inode(gc_list, inode->i_ino)) {
442 iput(inode);
443 return;
444 }
445 new_ie = f2fs_kmem_cache_alloc(inode_entry_slab, GFP_NOFS);
446 new_ie->inode = inode;
447
448 f2fs_radix_tree_insert(&gc_list->iroot, inode->i_ino, new_ie);
449 list_add_tail(&new_ie->list, &gc_list->ilist);
450}
451
452static void put_gc_inode(struct gc_inode_list *gc_list)
453{
454 struct inode_entry *ie, *next_ie;
455 list_for_each_entry_safe(ie, next_ie, &gc_list->ilist, list) {
456 radix_tree_delete(&gc_list->iroot, ie->inode->i_ino);
457 iput(ie->inode);
458 list_del(&ie->list);
459 kmem_cache_free(inode_entry_slab, ie);
460 }
461}

463static int check_valid_map(struct f2fs_sb_info *sbi,
464 unsigned int segno, int offset)
465{
466 struct sit_info *sit_i = SIT_I(sbi);
467 struct seg_entry *sentry;
468 int ret;
469
470 mutex_lock(&sit_i->sentry_lock);
471 sentry = get_seg_entry(sbi, segno);
472 ret = f2fs_test_bit(offset, sentry->cur_valid_map);
473 mutex_unlock(&sit_i->sentry_lock);
474 return ret;
475}

477/*
478 * This function compares node address got in summary with that in NAT.
479 * On validity, copy that node with cold status, otherwise (invalid node)
480 * ignore that.
481 */
482static void gc_node_segment(struct f2fs_sb_info *sbi,
483 struct f2fs_summary *sum, unsigned int segno, int gc_type)
484{
485 struct f2fs_summary *entry;
486 block_t start_addr;
487 int off;
488 int phase = 0;
489
490 start_addr = START_BLOCK(sbi, segno);
491
492next_step:
493 entry = sum;
494
495 for (off = 0; off < sbi->blocks_per_seg; off++, entry++) {
496 nid_t nid = le32_to_cpu(entry->nid);
497 struct page *node_page;
498 struct node_info ni;
499
500 /* stop BG_GC if there is not enough free sections. */
501 if (gc_type == BG_GC && has_not_enough_free_secs(sbi, 0, 0))
502 return;
503
504 if (check_valid_map(sbi, segno, off) == 0)
505 continue;
506
507 if (phase == 0) {
508 ra_meta_pages(sbi, NAT_BLOCK_OFFSET(nid), 1,
509 META_NAT, true);
510 continue;
511 }
512
513 if (phase == 1) {
514 ra_node_page(sbi, nid);
515 continue;
516 }
517
518 /* phase == 2 */
519 node_page = get_node_page(sbi, nid);
520 if (IS_ERR(node_page))
521 continue;
522
523 /* block may become invalid during get_node_page */
524 if (check_valid_map(sbi, segno, off) == 0) {
525 f2fs_put_page(node_page, 1);
526 continue;
527 }
528
529 get_node_info(sbi, nid, &ni);
530 if (ni.blk_addr != start_addr + off) {
531 f2fs_put_page(node_page, 1);
532 continue;
533 }
534
535 move_node_page(node_page, gc_type);
536 stat_inc_node_blk_count(sbi, 1, gc_type);
537 }
538
539 if (++phase < 3)
540 goto next_step;
541}

543/*
544 * Calculate start block index indicating the given node offset.
545 * Be careful, caller should give this node offset only indicating direct node
546 * blocks. If any node offsets, which point the other types of node blocks such
547 * as indirect or double indirect node blocks, are given, it must be a caller's
548 * bug.
549 */
550block_t start_bidx_of_node(unsigned int node_ofs, struct inode *inode)
551{
552 unsigned int indirect_blks = 2 * NIDS_PER_BLOCK + 4;
553 unsigned int bidx;
554
555 if (node_ofs == 0)
556 return 0;
557
558 if (node_ofs <= 2) {
559 bidx = node_ofs - 1;
560 } else if (node_ofs <= indirect_blks) {
561 int dec = (node_ofs - 4) / (NIDS_PER_BLOCK + 1);
562 bidx = node_ofs - 2 - dec;
563 } else {
564 int dec = (node_ofs - indirect_blks - 3) / (NIDS_PER_BLOCK + 1);
565 bidx = node_ofs - 5 - dec;
566 }
567 return bidx * ADDRS_PER_BLOCK + ADDRS_PER_INODE(inode);
568}

570static bool is_alive(struct f2fs_sb_info *sbi, struct f2fs_summary *sum,
571 struct node_info *dni, block_t blkaddr, unsigned int *nofs)
572{
573 struct page *node_page;
574 nid_t nid;
575 unsigned int ofs_in_node;
576 block_t source_blkaddr;
577
578 nid = le32_to_cpu(sum->nid);
579 ofs_in_node = le16_to_cpu(sum->ofs_in_node);
580
581 node_page = get_node_page(sbi, nid);
582 if (IS_ERR(node_page))
583 return false;
584
585 get_node_info(sbi, nid, dni);
586
587 if (sum->version != dni->version) {
588 f2fs_msg(sbi->sb, KERN_WARNING,
589 "%s: valid data with mismatched node version.",
590 __func__);
591 set_sbi_flag(sbi, SBI_NEED_FSCK);
592 }
593
594 *nofs = ofs_of_node(node_page);
595 source_blkaddr = datablock_addr(NULL, node_page, ofs_in_node);
596 f2fs_put_page(node_page, 1);
597
598 if (source_blkaddr != blkaddr)
599 return false;
600 return true;
601}

603/*
604 * Move data block via META_MAPPING while keeping locked data page.
605 * This can be used to move blocks, aka LBAs, directly on disk.
606 */
607static void move_data_block(struct inode *inode, block_t bidx,
608 unsigned int segno, int off)
609{
610 struct f2fs_io_info fio = {
611 .sbi = F2FS_I_SB(inode),
612 .type = DATA,
613 .temp = COLD,
614 .op = REQ_OP_READ,
615 .op_flags = 0,
616 .encrypted_page = NULL,
617 .in_list = false,
618 };
619 struct dnode_of_data dn;
620 struct f2fs_summary sum;
621 struct node_info ni;
622 struct page *page;
623 block_t newaddr;
624 int err;
625
626 /* do not read out */
627 page = f2fs_grab_cache_page(inode->i_mapping, bidx, false);
628 if (!page)
629 return;
630
631 if (!check_valid_map(F2FS_I_SB(inode), segno, off))
632 goto out;
633
634 if (f2fs_is_atomic_file(inode))
635 goto out;
636
637 set_new_dnode(&dn, inode, NULL, NULL, 0);
638 err = get_dnode_of_data(&dn, bidx, LOOKUP_NODE);
639 if (err)
640 goto out;
641
642 if (unlikely(dn.data_blkaddr == NULL_ADDR)) {
643 ClearPageUptodate(page);
644 goto put_out;
645 }
646
647 /*
648 * don't cache encrypted data into meta inode until previous dirty
649 * data were writebacked to avoid racing between GC and flush.
650 */
651 f2fs_wait_on_page_writeback(page, DATA, true);
652
653 get_node_info(fio.sbi, dn.nid, &ni);
654 set_summary(&sum, dn.nid, dn.ofs_in_node, ni.version);
655
656 /* read page */
657 fio.page = page;
658 fio.new_blkaddr = fio.old_blkaddr = dn.data_blkaddr;
659
660 allocate_data_block(fio.sbi, NULL, fio.old_blkaddr, &newaddr,
661 &sum, CURSEG_COLD_DATA, NULL, false);
662
663 fio.encrypted_page = pagecache_get_page(META_MAPPING(fio.sbi), newaddr,
664 FGP_LOCK | FGP_CREAT, GFP_NOFS);
665 if (!fio.encrypted_page) {
666 err = -ENOMEM;
667 goto recover_block;
668 }
669
670 err = f2fs_submit_page_bio(&fio);
671 if (err)
672 goto put_page_out;
673
674 /* write page */
675 lock_page(fio.encrypted_page);
676
677 if (unlikely(fio.encrypted_page->mapping != META_MAPPING(fio.sbi))) {
678 err = -EIO;
679 goto put_page_out;
680 }
681 if (unlikely(!PageUptodate(fio.encrypted_page))) {
682 err = -EIO;
683 goto put_page_out;
684 }
685
686 set_page_dirty(fio.encrypted_page);
687 f2fs_wait_on_page_writeback(fio.encrypted_page, DATA, true);
688 if (clear_page_dirty_for_io(fio.encrypted_page))
689 dec_page_count(fio.sbi, F2FS_DIRTY_META);
690
691 set_page_writeback(fio.encrypted_page);
692
693 /* allocate block address */
694 f2fs_wait_on_page_writeback(dn.node_page, NODE, true);
695
696 fio.op = REQ_OP_WRITE;
697 fio.op_flags = REQ_SYNC;
698 fio.new_blkaddr = newaddr;
699 err = f2fs_submit_page_write(&fio);
700 if (err) {
701 if (PageWriteback(fio.encrypted_page))
702 end_page_writeback(fio.encrypted_page);
703 goto put_page_out;
704 }
705
706 f2fs_update_iostat(fio.sbi, FS_GC_DATA_IO, F2FS_BLKSIZE);
707
708 f2fs_update_data_blkaddr(&dn, newaddr);
709 set_inode_flag(inode, FI_APPEND_WRITE);
710 if (page->index == 0)
711 set_inode_flag(inode, FI_FIRST_BLOCK_WRITTEN);
712put_page_out:
713 f2fs_put_page(fio.encrypted_page, 1);
714recover_block:
715 if (err)
716 __f2fs_replace_block(fio.sbi, &sum, newaddr, fio.old_blkaddr,
717 true, true);
718put_out:
719 f2fs_put_dnode(&dn);
720out:
721 f2fs_put_page(page, 1);
722}

724static void move_data_page(struct inode *inode, block_t bidx, int gc_type,
725 unsigned int segno, int off)
726{
727 struct page *page;
728
729 page = get_lock_data_page(inode, bidx, true);
730 if (IS_ERR(page))
731 return;
732
733 if (!check_valid_map(F2FS_I_SB(inode), segno, off))
734 goto out;
735
736 if (f2fs_is_atomic_file(inode))
737 goto out;
738
739 if (gc_type == BG_GC) {
740 if (PageWriteback(page))
741 goto out;
742 set_page_dirty(page);
743 set_cold_data(page);
744 } else {
745 struct f2fs_io_info fio = {
746 .sbi = F2FS_I_SB(inode),
747 .type = DATA,
748 .temp = COLD,
749 .op = REQ_OP_WRITE,
750 .op_flags = REQ_SYNC,
751 .old_blkaddr = NULL_ADDR,
752 .page = page,
753 .encrypted_page = NULL,
754 .need_lock = LOCK_REQ,
755 .io_type = FS_GC_DATA_IO,
756 };
757 bool is_dirty = PageDirty(page);
758 int err;
759
760retry:
761 set_page_dirty(page);
762 f2fs_wait_on_page_writeback(page, DATA, true);
763 if (clear_page_dirty_for_io(page)) {
764 inode_dec_dirty_pages(inode);
765 remove_dirty_inode(inode);
766 }
767
768 set_cold_data(page);
769
770 err = do_write_data_page(&fio);
771 if (err == -ENOMEM && is_dirty) {
772 congestion_wait(BLK_RW_ASYNC, HZ/50);
773 goto retry;
774 }
775 }
776out:
777 f2fs_put_page(page, 1);
778}

780/*
781 * This function tries to get parent node of victim data block, and identifies
782 * data block validity. If the block is valid, copy that with cold status and
783 * modify parent node.
784 * If the parent node is not valid or the data block address is different,
785 * the victim data block is ignored.
786 */
787static void gc_data_segment(struct f2fs_sb_info *sbi, struct f2fs_summary *sum,
788 struct gc_inode_list *gc_list, unsigned int segno, int gc_type)
789{
790 struct super_block *sb = sbi->sb;
791 struct f2fs_summary *entry;
792 block_t start_addr;
793 int off;
794 int phase = 0;
795
796 start_addr = START_BLOCK(sbi, segno);
797
798next_step:
799 entry = sum;
800
801 for (off = 0; off < sbi->blocks_per_seg; off++, entry++) {
802 struct page *data_page;
803 struct inode *inode;
804 struct node_info dni; /* dnode info for the data */
805 unsigned int ofs_in_node, nofs;
806 block_t start_bidx;
807 nid_t nid = le32_to_cpu(entry->nid);
808
809 /* stop BG_GC if there is not enough free sections. */
810 if (gc_type == BG_GC && has_not_enough_free_secs(sbi, 0, 0))
811 return;
812
813 if (check_valid_map(sbi, segno, off) == 0)
814 continue;
815
816 if (phase == 0) {
817 ra_meta_pages(sbi, NAT_BLOCK_OFFSET(nid), 1,
818 META_NAT, true);
819 continue;
820 }
821
822 if (phase == 1) {
823 ra_node_page(sbi, nid);
824 continue;
825 }
826
827 /* Get an inode by ino with checking validity */
828 if (!is_alive(sbi, entry, &dni, start_addr + off, &nofs))
829 continue;
830
831 if (phase == 2) {
832 ra_node_page(sbi, dni.ino);
833 continue;
834 }
835
836 ofs_in_node = le16_to_cpu(entry->ofs_in_node);
837
838 if (phase == 3) {
839 inode = f2fs_iget(sb, dni.ino);
840 if (IS_ERR(inode) || is_bad_inode(inode))
841 continue;
842
843 /* if encrypted inode, let's go phase 3 */
844 if (f2fs_encrypted_file(inode)) {
845 add_gc_inode(gc_list, inode);
846 continue;
847 }
848
849 start_bidx = start_bidx_of_node(nofs, inode);
850 data_page = get_read_data_page(inode,
851 start_bidx + ofs_in_node, REQ_RAHEAD,
852 true);
853 if (IS_ERR(data_page)) {
854 iput(inode);
855 continue;
856 }
857
858 f2fs_put_page(data_page, 0);
859 add_gc_inode(gc_list, inode);
860 continue;
861 }
862
863 /* phase 4 */
864 inode = find_gc_inode(gc_list, dni.ino);
865 if (inode) {
866 struct f2fs_inode_info *fi = F2FS_I(inode);
867 bool locked = false;
868
869 if (S_ISREG(inode->i_mode)) {
870 if (!down_write_trylock(&fi->dio_rwsem[READ]))
871 continue;
872 if (!down_write_trylock(
873 &fi->dio_rwsem[WRITE])) {
874 up_write(&fi->dio_rwsem[READ]);
875 continue;
876 }
877 locked = true;
878
879 /* wait for all inflight aio data */
880 inode_dio_wait(inode);
881 }
882
883 start_bidx = start_bidx_of_node(nofs, inode)
884 + ofs_in_node;
885 if (f2fs_encrypted_file(inode))
886 move_data_block(inode, start_bidx, segno, off);
887 else
888 move_data_page(inode, start_bidx, gc_type,
889 segno, off);
890
891 if (locked) {
892 up_write(&fi->dio_rwsem[WRITE]);
893 up_write(&fi->dio_rwsem[READ]);
894 }
895
896 stat_inc_data_blk_count(sbi, 1, gc_type);
897 }
898 }
899
900 if (++phase < 5)
901 goto next_step;
902}

904static int __get_victim(struct f2fs_sb_info *sbi, unsigned int *victim,
905 int gc_type)
906{
907 struct sit_info *sit_i = SIT_I(sbi);
908 int ret;
909
910 mutex_lock(&sit_i->sentry_lock);
911 ret = DIRTY_I(sbi)->v_ops->get_victim(sbi, victim, gc_type,
912 NO_CHECK_TYPE, LFS);
913 mutex_unlock(&sit_i->sentry_lock);
914 return ret;
915}

917static int do_garbage_collect(struct f2fs_sb_info *sbi,
918 unsigned int start_segno,
919 struct gc_inode_list *gc_list, int gc_type)
920{
921 struct page *sum_page;
922 struct f2fs_summary_block *sum;
923 struct blk_plug plug;
924 unsigned int segno = start_segno;
925 unsigned int end_segno = start_segno + sbi->segs_per_sec;
926 int seg_freed = 0;
927 unsigned char type = IS_DATASEG(get_seg_entry(sbi, segno)->type) ?
928 SUM_TYPE_DATA : SUM_TYPE_NODE;
929
930 /* readahead multi ssa blocks those have contiguous address */
931 if (sbi->segs_per_sec > 1)
932 ra_meta_pages(sbi, GET_SUM_BLOCK(sbi, segno),
933 sbi->segs_per_sec, META_SSA, true);
934
935 /* reference all summary page */
936 while (segno < end_segno) {
937 sum_page = get_sum_page(sbi, segno++);
938 unlock_page(sum_page);
939 }
940
941 blk_start_plug(&plug);
942
943 for (segno = start_segno; segno < end_segno; segno++) {
944
945 /* find segment summary of victim */
946 sum_page = find_get_page(META_MAPPING(sbi),
947 GET_SUM_BLOCK(sbi, segno));
948 f2fs_put_page(sum_page, 0);
949
950 if (get_valid_blocks(sbi, segno, false) == 0 ||
951 !PageUptodate(sum_page) ||
952 unlikely(f2fs_cp_error(sbi)))
953 goto next;
954
955 sum = page_address(sum_page);
956 f2fs_bug_on(sbi, type != GET_SUM_TYPE((&sum->footer)));
957
958 /*
959 * this is to avoid deadlock:
960 * - lock_page(sum_page) - f2fs_replace_block
961 * - check_valid_map() - mutex_lock(sentry_lock)
962 * - mutex_lock(sentry_lock) - change_curseg()
963 * - lock_page(sum_page)
964 */
965 if (type == SUM_TYPE_NODE)
966 gc_node_segment(sbi, sum->entries, segno, gc_type);
967 else
968 gc_data_segment(sbi, sum->entries, gc_list, segno,
969 gc_type);
970
971 stat_inc_seg_count(sbi, type, gc_type);
972
973 if (gc_type == FG_GC &&
974 get_valid_blocks(sbi, segno, false) == 0)
975 seg_freed++;
976next:
977 f2fs_put_page(sum_page, 0);
978 }
979
980 if (gc_type == FG_GC)
981 f2fs_submit_merged_write(sbi,
982 (type == SUM_TYPE_NODE) ? NODE : DATA);
983
984 blk_finish_plug(&plug);
985
986 stat_inc_call_count(sbi->stat_info);
987
988 return seg_freed;
989}

991int f2fs_gc(struct f2fs_sb_info *sbi, bool sync,
992 bool background, unsigned int segno)
993{
994 int gc_type = sync ? FG_GC : BG_GC;
995 int sec_freed = 0, seg_freed = 0, total_freed = 0;
996 int ret = 0;
997 struct cp_control cpc;
998 unsigned int init_segno = segno;
999 struct gc_inode_list gc_list = {
1000 .ilist = LIST_HEAD_INIT(gc_list.ilist),
1001 .iroot = RADIX_TREE_INIT(GFP_NOFS),
1002 };
1003
1004 trace_f2fs_gc_begin(sbi->sb, sync, background,
1005 get_pages(sbi, F2FS_DIRTY_NODES),
1006 get_pages(sbi, F2FS_DIRTY_DENTS),
1007 get_pages(sbi, F2FS_DIRTY_IMETA),
1008 free_sections(sbi),
1009 free_segments(sbi),
1010 reserved_segments(sbi),
1011 prefree_segments(sbi));
1012
1013 cpc.reason = __get_cp_reason(sbi);
1014gc_more:
1015 if (unlikely(!(sbi->sb->s_flags & MS_ACTIVE))) {
1016 ret = -EINVAL;
1017 goto stop;
1018 }
1019 if (unlikely(f2fs_cp_error(sbi))) {
1020 ret = -EIO;
1021 goto stop;
1022 }
1023
1024 if (gc_type == BG_GC && has_not_enough_free_secs(sbi, 0, 0)) {
1025 /*
1026 * For example, if there are many prefree_segments below given
1027 * threshold, we can make them free by checkpoint. Then, we
1028 * secure free segments which doesn't need fggc any more.
1029 */
1030 if (prefree_segments(sbi)) {
1031 ret = write_checkpoint(sbi, &cpc);
1032 if (ret)
1033 goto stop;
1034 }
1035 if (has_not_enough_free_secs(sbi, 0, 0))
1036 gc_type = FG_GC;
1037 }
1038
1039 /* f2fs_balance_fs doesn't need to do BG_GC in critical path. */
1040 if (gc_type == BG_GC && !background) {
1041 ret = -EINVAL;
1042 goto stop;
1043 }
1044 if (!__get_victim(sbi, &segno, gc_type)) {
1045 ret = -ENODATA;
1046 goto stop;
1047 }
1048
1049 seg_freed = do_garbage_collect(sbi, segno, &gc_list, gc_type);
1050 if (gc_type == FG_GC && seg_freed == sbi->segs_per_sec)
1051 sec_freed++;
1052 total_freed += seg_freed;
1053
1054 if (gc_type == FG_GC)
1055 sbi->cur_victim_sec = NULL_SEGNO;
1056
1057 if (!sync) {
1058 if (has_not_enough_free_secs(sbi, sec_freed, 0)) {
1059 segno = NULL_SEGNO;
1060 goto gc_more;
1061 }
1062
1063 if (gc_type == FG_GC)
1064 ret = write_checkpoint(sbi, &cpc);
1065 }
1066stop:
1067 SIT_I(sbi)->last_victim[ALLOC_NEXT] = 0;
1068 SIT_I(sbi)->last_victim[FLUSH_DEVICE] = init_segno;
1069
1070 trace_f2fs_gc_end(sbi->sb, ret, total_freed, sec_freed,
1071 get_pages(sbi, F2FS_DIRTY_NODES),
1072 get_pages(sbi, F2FS_DIRTY_DENTS),
1073 get_pages(sbi, F2FS_DIRTY_IMETA),
1074 free_sections(sbi),
1075 free_segments(sbi),
1076 reserved_segments(sbi),
1077 prefree_segments(sbi));
1078
1079 mutex_unlock(&sbi->gc_mutex);
1080
1081 put_gc_inode(&gc_list);
1082
1083 if (sync)
1084 ret = sec_freed ? 0 : -EAGAIN;
1085 return ret;
1086}

1088void build_gc_manager(struct f2fs_sb_info *sbi)
1089{
1090 u64 main_count, resv_count, ovp_count;
1091
1092 DIRTY_I(sbi)->v_ops = &default_v_ops;
1093
1094 /* threshold of # of valid blocks in a section for victims of FG_GC */
1095 main_count = SM_I(sbi)->main_segments << sbi->log_blocks_per_seg;
1096 resv_count = SM_I(sbi)->reserved_segments << sbi->log_blocks_per_seg;
1097 ovp_count = SM_I(sbi)->ovp_segments << sbi->log_blocks_per_seg;
1098
1099 sbi->fggc_threshold = div64_u64((main_count - ovp_count) *
1100 BLKS_PER_SEC(sbi), (main_count - resv_count));
1101
1102 /* give warm/cold data area from slower device */
1103 if (sbi->s_ndevs && sbi->segs_per_sec == 1)
1104 SIT_I(sbi)->last_victim[ALLOC_NEXT] =
1105 GET_SEGNO(sbi, FDEV(0).end_blk) + 1;
1106}
