31void f2fs_set_inode_flags(struct inode *inode)
32{
33 unsigned int flags = F2FS_I(inode)->i_flags;
34 unsigned int new_fl = 0;
35
36 if (flags & FS_SYNC_FL)
37 new_fl |= S_SYNC;
38 if (flags & FS_APPEND_FL)
39 new_fl |= S_APPEND;
40 if (flags & FS_IMMUTABLE_FL)
41 new_fl |= S_IMMUTABLE;
42 if (flags & FS_NOATIME_FL)
43 new_fl |= S_NOATIME;
44 if (flags & FS_DIRSYNC_FL)
45 new_fl |= S_DIRSYNC;
46 inode_set_flags(inode, new_fl,
47 S_SYNC|S_APPEND|S_IMMUTABLE|S_NOATIME|S_DIRSYNC);
48}

50static void __get_inode_rdev(struct inode *inode, struct f2fs_inode *ri)
51{
52 int extra_size = get_extra_isize(inode);
53
54 if (S_ISCHR(inode->i_mode) || S_ISBLK(inode->i_mode) ||
55 S_ISFIFO(inode->i_mode) || S_ISSOCK(inode->i_mode)) {
56 if (ri->i_addr[extra_size])
57 inode->i_rdev = old_decode_dev(
58 le32_to_cpu(ri->i_addr[extra_size]));
59 else
60 inode->i_rdev = new_decode_dev(
61 le32_to_cpu(ri->i_addr[extra_size + 1]));
62 }
63}

65static bool __written_first_block(struct f2fs_inode *ri)
66{
67 block_t addr = le32_to_cpu(ri->i_addr[offset_in_addr(ri)]);
68
69 if (addr != NEW_ADDR && addr != NULL_ADDR)
70 return true;
71 return false;
72}
73
74static void __set_inode_rdev(struct inode *inode, struct f2fs_inode *ri)
75{
76 int extra_size = get_extra_isize(inode);
77
78 if (S_ISCHR(inode->i_mode) || S_ISBLK(inode->i_mode)) {
79 if (old_valid_dev(inode->i_rdev)) {
80 ri->i_addr[extra_size] =
81 cpu_to_le32(old_encode_dev(inode->i_rdev));
82 ri->i_addr[extra_size + 1] = 0;
83 } else {
84 ri->i_addr[extra_size] = 0;
85 ri->i_addr[extra_size + 1] =
86 cpu_to_le32(new_encode_dev(inode->i_rdev));
87 ri->i_addr[extra_size + 2] = 0;
88 }
89 }
90}

92static void __recover_inline_status(struct inode *inode, struct page *ipage)
93{
94 void *inline_data = inline_data_addr(inode, ipage);
95 __le32 *start = inline_data;
96 __le32 *end = start + MAX_INLINE_DATA(inode) / sizeof(__le32);
97
98 while (start < end) {
99 if (*start++) {
100 f2fs_wait_on_page_writeback(ipage, NODE, true);
101
102 set_inode_flag(inode, FI_DATA_EXIST);
103 set_raw_inline(inode, F2FS_INODE(ipage));
104 set_page_dirty(ipage);
105 return;
106 }
107 }
108 return;
109}

111static bool f2fs_enable_inode_chksum(struct f2fs_sb_info *sbi, struct page *page)
112{
113 struct f2fs_inode *ri = &F2FS_NODE(page)->i;
114 int extra_isize = le32_to_cpu(ri->i_extra_isize);
115
116 if (!f2fs_sb_has_inode_chksum(sbi->sb))
117 return false;
118
119 if (!RAW_IS_INODE(F2FS_NODE(page)) || !(ri->i_inline & F2FS_EXTRA_ATTR))
120 return false;
121
122 if (!F2FS_FITS_IN_INODE(ri, extra_isize, i_inode_checksum))
123 return false;
124
125 return true;
126}

128static __u32 f2fs_inode_chksum(struct f2fs_sb_info *sbi, struct page *page)
129{
130 struct f2fs_node *node = F2FS_NODE(page);
131 struct f2fs_inode *ri = &node->i;
132 __le32 ino = node->footer.ino;
133 __le32 gen = ri->i_generation;
134 __u32 chksum, chksum_seed;
135 __u32 dummy_cs = 0;
136 unsigned int offset = offsetof(struct f2fs_inode, i_inode_checksum);
137 unsigned int cs_size = sizeof(dummy_cs);
138
139 chksum = f2fs_chksum(sbi, sbi->s_chksum_seed, (__u8 *)&ino,
140 sizeof(ino));
141 chksum_seed = f2fs_chksum(sbi, chksum, (__u8 *)&gen, sizeof(gen));
142
143 chksum = f2fs_chksum(sbi, chksum_seed, (__u8 *)ri, offset);
144 chksum = f2fs_chksum(sbi, chksum, (__u8 *)&dummy_cs, cs_size);
145 offset += cs_size;
146 chksum = f2fs_chksum(sbi, chksum, (__u8 *)ri + offset,
147 F2FS_BLKSIZE - offset);
148 return chksum;
149}

151bool f2fs_inode_chksum_verify(struct f2fs_sb_info *sbi, struct page *page)
152{
153 struct f2fs_inode *ri;
154 __u32 provided, calculated;
155
156 if (!f2fs_enable_inode_chksum(sbi, page) ||
157 PageDirty(page) || PageWriteback(page))
158 return true;
159
160 ri = &F2FS_NODE(page)->i;
161 provided = le32_to_cpu(ri->i_inode_checksum);
162 calculated = f2fs_inode_chksum(sbi, page);
163
164 if (provided != calculated)
165 f2fs_msg(sbi->sb, KERN_WARNING,
166 "checksum invalid, ino = %x, %x vs. %x",
167 ino_of_node(page), provided, calculated);
168
169 return provided == calculated;
170}

172void f2fs_inode_chksum_set(struct f2fs_sb_info *sbi, struct page *page)
173{
174 struct f2fs_inode *ri = &F2FS_NODE(page)->i;
175
176 if (!f2fs_enable_inode_chksum(sbi, page))
177 return;
178
179 ri->i_inode_checksum = cpu_to_le32(f2fs_inode_chksum(sbi, page));
180}

182static int do_read_inode(struct inode *inode)
183{
184 struct f2fs_sb_info *sbi = F2FS_I_SB(inode);
185 struct f2fs_inode_info *fi = F2FS_I(inode);
186 struct page *node_page;
187 struct f2fs_inode *ri;
188 projid_t i_projid;
189
190 /* Check if ino is within scope */
191 if (check_nid_range(sbi, inode->i_ino))
192 return -EINVAL;
193
194 node_page = get_node_page(sbi, inode->i_ino);
195 if (IS_ERR(node_page))
196 return PTR_ERR(node_page);
197
198 ri = F2FS_INODE(node_page);
199
200 inode->i_mode = le16_to_cpu(ri->i_mode);
201 i_uid_write(inode, le32_to_cpu(ri->i_uid));
202 i_gid_write(inode, le32_to_cpu(ri->i_gid));
203 set_nlink(inode, le32_to_cpu(ri->i_links));
204 inode->i_size = le64_to_cpu(ri->i_size);
205 inode->i_blocks = SECTOR_FROM_BLOCK(le64_to_cpu(ri->i_blocks) - 1);
206
207 inode->i_atime.tv_sec = le64_to_cpu(ri->i_atime);
208 inode->i_ctime.tv_sec = le64_to_cpu(ri->i_ctime);
209 inode->i_mtime.tv_sec = le64_to_cpu(ri->i_mtime);
210 inode->i_atime.tv_nsec = le32_to_cpu(ri->i_atime_nsec);
211 inode->i_ctime.tv_nsec = le32_to_cpu(ri->i_ctime_nsec);
212 inode->i_mtime.tv_nsec = le32_to_cpu(ri->i_mtime_nsec);
213 inode->i_generation = le32_to_cpu(ri->i_generation);
214
215 fi->i_current_depth = le32_to_cpu(ri->i_current_depth);
216 fi->i_xattr_nid = le32_to_cpu(ri->i_xattr_nid);
217 fi->i_flags = le32_to_cpu(ri->i_flags);
218 fi->flags = 0;
219 fi->i_advise = ri->i_advise;
220 fi->i_pino = le32_to_cpu(ri->i_pino);
221 fi->i_dir_level = ri->i_dir_level;
222
223 if (f2fs_init_extent_tree(inode, &ri->i_ext))
224 set_page_dirty(node_page);
225
226 get_inline_info(inode, ri);
227
228 fi->i_extra_isize = f2fs_has_extra_attr(inode) ?
229 le16_to_cpu(ri->i_extra_isize) : 0;
230
231 /* check data exist */
232 if (f2fs_has_inline_data(inode) && !f2fs_exist_data(inode))
233 __recover_inline_status(inode, node_page);
234
235 /* get rdev by using inline_info */
236 __get_inode_rdev(inode, ri);
237
238 if (__written_first_block(ri))
239 set_inode_flag(inode, FI_FIRST_BLOCK_WRITTEN);
240
241 if (!need_inode_block_update(sbi, inode->i_ino))
242 fi->last_disk_size = inode->i_size;
243
244 if (fi->i_flags & FS_PROJINHERIT_FL)
245 set_inode_flag(inode, FI_PROJ_INHERIT);
246
247 if (f2fs_has_extra_attr(inode) && f2fs_sb_has_project_quota(sbi->sb) &&
248 F2FS_FITS_IN_INODE(ri, fi->i_extra_isize, i_projid))
249 i_projid = (projid_t)le32_to_cpu(ri->i_projid);
250 else
251 i_projid = F2FS_DEF_PROJID;
252 fi->i_projid = make_kprojid(&init_user_ns, i_projid);
253
254 f2fs_put_page(node_page, 1);
255
256 stat_inc_inline_xattr(inode);
257 stat_inc_inline_inode(inode);
258 stat_inc_inline_dir(inode);
259
260 return 0;
261}

263struct inode *f2fs_iget(struct super_block *sb, unsigned long ino)
264{
265 struct f2fs_sb_info *sbi = F2FS_SB(sb);
266 struct inode *inode;
267 int ret = 0;
268
269 inode = iget_locked(sb, ino);
270 if (!inode)
271 return ERR_PTR(-ENOMEM);
272
273 if (!(inode->i_state & I_NEW)) {
274 trace_f2fs_iget(inode);
275 return inode;
276 }
277 if (ino == F2FS_NODE_INO(sbi) || ino == F2FS_META_INO(sbi))
278 goto make_now;
279
280 ret = do_read_inode(inode);
281 if (ret)
282 goto bad_inode;
283make_now:
284 if (ino == F2FS_NODE_INO(sbi)) {
285 inode->i_mapping->a_ops = &f2fs_node_aops;
286 mapping_set_gfp_mask(inode->i_mapping, GFP_F2FS_ZERO);
287 } else if (ino == F2FS_META_INO(sbi)) {
288 inode->i_mapping->a_ops = &f2fs_meta_aops;
289 mapping_set_gfp_mask(inode->i_mapping, GFP_F2FS_ZERO);
290 } else if (S_ISREG(inode->i_mode)) {
291 inode->i_op = &f2fs_file_inode_operations;
292 inode->i_fop = &f2fs_file_operations;
293 inode->i_mapping->a_ops = &f2fs_dblock_aops;
294 } else if (S_ISDIR(inode->i_mode)) {
295 inode->i_op = &f2fs_dir_inode_operations;
296 inode->i_fop = &f2fs_dir_operations;
297 inode->i_mapping->a_ops = &f2fs_dblock_aops;
298 mapping_set_gfp_mask(inode->i_mapping, GFP_F2FS_HIGH_ZERO);
299 } else if (S_ISLNK(inode->i_mode)) {
300 if (f2fs_encrypted_inode(inode))
301 inode->i_op = &f2fs_encrypted_symlink_inode_operations;
302 else
303 inode->i_op = &f2fs_symlink_inode_operations;
304 inode_nohighmem(inode);
305 inode->i_mapping->a_ops = &f2fs_dblock_aops;
306 } else if (S_ISCHR(inode->i_mode) || S_ISBLK(inode->i_mode) ||
307 S_ISFIFO(inode->i_mode) || S_ISSOCK(inode->i_mode)) {
308 inode->i_op = &f2fs_special_inode_operations;
309 init_special_inode(inode, inode->i_mode, inode->i_rdev);
310 } else {
311 ret = -EIO;
312 goto bad_inode;
313 }
314 f2fs_set_inode_flags(inode);
315 unlock_new_inode(inode);
316 trace_f2fs_iget(inode);
317 return inode;
318
319bad_inode:
320 iget_failed(inode);
321 trace_f2fs_iget_exit(inode, ret);
322 return ERR_PTR(ret);
323}

325struct inode *f2fs_iget_retry(struct super_block *sb, unsigned long ino)
326{
327 struct inode *inode;
328retry:
329 inode = f2fs_iget(sb, ino);
330 if (IS_ERR(inode)) {
331 if (PTR_ERR(inode) == -ENOMEM) {
332 congestion_wait(BLK_RW_ASYNC, HZ/50);
333 goto retry;
334 }
335 }
336 return inode;
337}

339int update_inode(struct inode *inode, struct page *node_page)
340{
341 struct f2fs_inode *ri;
342 struct extent_tree *et = F2FS_I(inode)->extent_tree;
343
344 f2fs_inode_synced(inode);
345
346 f2fs_wait_on_page_writeback(node_page, NODE, true);
347
348 ri = F2FS_INODE(node_page);
349
350 ri->i_mode = cpu_to_le16(inode->i_mode);
351 ri->i_advise = F2FS_I(inode)->i_advise;
352 ri->i_uid = cpu_to_le32(i_uid_read(inode));
353 ri->i_gid = cpu_to_le32(i_gid_read(inode));
354 ri->i_links = cpu_to_le32(inode->i_nlink);
355 ri->i_size = cpu_to_le64(i_size_read(inode));
356 ri->i_blocks = cpu_to_le64(SECTOR_TO_BLOCK(inode->i_blocks) + 1);
357
358 if (et) {
359 read_lock(&et->lock);
360 set_raw_extent(&et->largest, &ri->i_ext);
361 read_unlock(&et->lock);
362 } else {
363 memset(&ri->i_ext, 0, sizeof(ri->i_ext));
364 }
365 set_raw_inline(inode, ri);
366
367 ri->i_atime = cpu_to_le64(inode->i_atime.tv_sec);
368 ri->i_ctime = cpu_to_le64(inode->i_ctime.tv_sec);
369 ri->i_mtime = cpu_to_le64(inode->i_mtime.tv_sec);
370 ri->i_atime_nsec = cpu_to_le32(inode->i_atime.tv_nsec);
371 ri->i_ctime_nsec = cpu_to_le32(inode->i_ctime.tv_nsec);
372 ri->i_mtime_nsec = cpu_to_le32(inode->i_mtime.tv_nsec);
373 ri->i_current_depth = cpu_to_le32(F2FS_I(inode)->i_current_depth);
374 ri->i_xattr_nid = cpu_to_le32(F2FS_I(inode)->i_xattr_nid);
375 ri->i_flags = cpu_to_le32(F2FS_I(inode)->i_flags);
376 ri->i_pino = cpu_to_le32(F2FS_I(inode)->i_pino);
377 ri->i_generation = cpu_to_le32(inode->i_generation);
378 ri->i_dir_level = F2FS_I(inode)->i_dir_level;
379
380 if (f2fs_has_extra_attr(inode)) {
381 ri->i_extra_isize = cpu_to_le16(F2FS_I(inode)->i_extra_isize);
382
383 if (f2fs_sb_has_project_quota(F2FS_I_SB(inode)->sb) &&
384 F2FS_FITS_IN_INODE(ri, F2FS_I(inode)->i_extra_isize,
385 i_projid)) {
386 projid_t i_projid;
387
388 i_projid = from_kprojid(&init_user_ns,
389 F2FS_I(inode)->i_projid);
390 ri->i_projid = cpu_to_le32(i_projid);
391 }
392 }
393
394 __set_inode_rdev(inode, ri);
395 set_cold_node(inode, node_page);
396
397 /* deleted inode */
398 if (inode->i_nlink == 0)
399 clear_inline_node(node_page);
400
401 return set_page_dirty(node_page);
402}

404int update_inode_page(struct inode *inode)
405{
406 struct f2fs_sb_info *sbi = F2FS_I_SB(inode);
407 struct page *node_page;
408 int ret = 0;
409retry:
410 node_page = get_node_page(sbi, inode->i_ino);
411 if (IS_ERR(node_page)) {
412 int err = PTR_ERR(node_page);
413 if (err == -ENOMEM) {
414 cond_resched();
415 goto retry;
416 } else if (err != -ENOENT) {
417 f2fs_stop_checkpoint(sbi, false);
418 }
419 return 0;
420 }
421 ret = update_inode(inode, node_page);
422 f2fs_put_page(node_page, 1);
423 return ret;
424}

426int f2fs_write_inode(struct inode *inode, struct writeback_control *wbc)
427{
428 struct f2fs_sb_info *sbi = F2FS_I_SB(inode);
429
430 if (inode->i_ino == F2FS_NODE_INO(sbi) ||
431 inode->i_ino == F2FS_META_INO(sbi))
432 return 0;
433
434 if (!is_inode_flag_set(inode, FI_DIRTY_INODE))
435 return 0;
436
437 /*
438 * We need to balance fs here to prevent from producing dirty node pages
439 * during the urgent cleaning time when runing out of free sections.
440 */
441 update_inode_page(inode);
442 if (wbc && wbc->nr_to_write)
443 f2fs_balance_fs(sbi, true);
444 return 0;
445}

447/*
448 * Called at the last iput() if i_nlink is zero
449 */
450void f2fs_evict_inode(struct inode *inode)
451{
452 struct f2fs_sb_info *sbi = F2FS_I_SB(inode);
453 nid_t xnid = F2FS_I(inode)->i_xattr_nid;
454 int err = 0;
455
456 /* some remained atomic pages should discarded */
457 if (f2fs_is_atomic_file(inode))
458 drop_inmem_pages(inode);
459
460 trace_f2fs_evict_inode(inode);
461 truncate_inode_pages_final(&inode->i_data);
462
463 if (inode->i_ino == F2FS_NODE_INO(sbi) ||
464 inode->i_ino == F2FS_META_INO(sbi))
465 goto out_clear;
466
467 f2fs_bug_on(sbi, get_dirty_pages(inode));
468 remove_dirty_inode(inode);
469
470 f2fs_destroy_extent_tree(inode);
471
472 if (inode->i_nlink || is_bad_inode(inode))
473 goto no_delete;
474
475 dquot_initialize(inode);
476
477 remove_ino_entry(sbi, inode->i_ino, APPEND_INO);
478 remove_ino_entry(sbi, inode->i_ino, UPDATE_INO);
479
480 sb_start_intwrite(inode->i_sb);
481 set_inode_flag(inode, FI_NO_ALLOC);
482 i_size_write(inode, 0);
483retry:
484 if (F2FS_HAS_BLOCKS(inode))
485 err = f2fs_truncate(inode);
486
487#ifdef CONFIG_F2FS_FAULT_INJECTION
488 if (time_to_inject(sbi, FAULT_EVICT_INODE)) {
489 f2fs_show_injection_info(FAULT_EVICT_INODE);
490 err = -EIO;
491 }
492#endif
493 if (!err) {
494 f2fs_lock_op(sbi);
495 err = remove_inode_page(inode);
496 f2fs_unlock_op(sbi);
497 if (err == -ENOENT)
498 err = 0;
499 }
500
501 /* give more chances, if ENOMEM case */
502 if (err == -ENOMEM) {
503 err = 0;
504 goto retry;
505 }
506
507 if (err)
508 update_inode_page(inode);
509 dquot_free_inode(inode);
510 sb_end_intwrite(inode->i_sb);
511no_delete:
512 dquot_drop(inode);
513
514 stat_dec_inline_xattr(inode);
515 stat_dec_inline_dir(inode);
516 stat_dec_inline_inode(inode);
517
518 if (!is_set_ckpt_flags(sbi, CP_ERROR_FLAG))
519 f2fs_bug_on(sbi, is_inode_flag_set(inode, FI_DIRTY_INODE));
520
521 /* ino == 0, if f2fs_new_inode() was failed t*/
522 if (inode->i_ino)
523 invalidate_mapping_pages(NODE_MAPPING(sbi), inode->i_ino,
524 inode->i_ino);
525 if (xnid)
526 invalidate_mapping_pages(NODE_MAPPING(sbi), xnid, xnid);
527 if (inode->i_nlink) {
528 if (is_inode_flag_set(inode, FI_APPEND_WRITE))
529 add_ino_entry(sbi, inode->i_ino, APPEND_INO);
530 if (is_inode_flag_set(inode, FI_UPDATE_WRITE))
531 add_ino_entry(sbi, inode->i_ino, UPDATE_INO);
532 }
533 if (is_inode_flag_set(inode, FI_FREE_NID)) {
534 alloc_nid_failed(sbi, inode->i_ino);
535 clear_inode_flag(inode, FI_FREE_NID);
536 } else {
537 /*
538 * If xattr nid is corrupted, we can reach out error condition,
539 * err & !exist_written_data(sbi, inode->i_ino, ORPHAN_INO)).
540 * In that case, check_nid_range() is enough to give a clue.
541 */
542 }
543out_clear:
544 fscrypt_put_encryption_info(inode, NULL);
545 clear_inode(inode);
546}

548/* caller should call f2fs_lock_op() */
549void handle_failed_inode(struct inode *inode)
550{
551 struct f2fs_sb_info *sbi = F2FS_I_SB(inode);
552 struct node_info ni;
553
554 /*
555 * clear nlink of inode in order to release resource of inode
556 * immediately.
557 */
558 clear_nlink(inode);
559
560 /*
561 * we must call this to avoid inode being remained as dirty, resulting
562 * in a panic when flushing dirty inodes in gdirty_list.
563 */
564 update_inode_page(inode);
565 f2fs_inode_synced(inode);
566
567 /* don't make bad inode, since it becomes a regular file. */
568 unlock_new_inode(inode);
569
570 /*
571 * Note: we should add inode to orphan list before f2fs_unlock_op()
572 * so we can prevent losing this orphan when encoutering checkpoint
573 * and following suddenly power-off.
574 */
575 get_node_info(sbi, inode->i_ino, &ni);
576
577 if (ni.blk_addr != NULL_ADDR) {
578 int err = acquire_orphan_inode(sbi);
579 if (err) {
580 set_sbi_flag(sbi, SBI_NEED_FSCK);
581 f2fs_msg(sbi->sb, KERN_WARNING,
582 "Too many orphan inodes, run fsck to fix.");
583 } else {
584 add_orphan_inode(inode);
585 }
586 alloc_nid_done(sbi, inode->i_ino);
587 } else {
588 set_inode_flag(inode, FI_FREE_NID);
589 }
590
591 f2fs_unlock_op(sbi);
592
593 /* iput will drop the inode object */
594 iput(inode);
595}
