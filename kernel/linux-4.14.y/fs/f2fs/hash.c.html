23/*
24 * Hashing code copied from ext3
25 */
26#define DELTA 0x9E3779B9
27
28static void TEA_transform(unsigned int buf[4], unsigned int const in[])
29{
30 __u32 sum = 0;
31 __u32 b0 = buf[0], b1 = buf[1];
32 __u32 a = in[0], b = in[1], c = in[2], d = in[3];
33 int n = 16;
34
35 do {
36 sum += DELTA;
37 b0 += ((b1 << 4)+a) ^ (b1+sum) ^ ((b1 >> 5)+b);
38 b1 += ((b0 << 4)+c) ^ (b0+sum) ^ ((b0 >> 5)+d);
39 } while (--n);
40
41 buf[0] += b0;
42 buf[1] += b1;
43}

45static void str2hashbuf(const unsigned char *msg, size_t len,
46 unsigned int *buf, int num)
47{
48 unsigned pad, val;
49 int i;
50
51 pad = (__u32)len | ((__u32)len << 8);
52 pad |= pad << 16;
53
54 val = pad;
55 if (len > num * 4)
56 len = num * 4;
57 for (i = 0; i < len; i++) {
58 if ((i % 4) == 0)
59 val = pad;
60 val = msg[i] + (val << 8);
61 if ((i % 4) == 3) {
62 *buf++ = val;
63 val = pad;
64 num--;
65 }
66 }
67 if (--num >= 0)
68 *buf++ = val;
69 while (--num >= 0)
70 *buf++ = pad;
71}

73f2fs_hash_t f2fs_dentry_hash(const struct qstr *name_info,
74 struct fscrypt_name *fname)
75{
76 __u32 hash;
77 f2fs_hash_t f2fs_hash;
78 const unsigned char *p;
79 __u32 in[8], buf[4];
80 const unsigned char *name = name_info->name;
81 size_t len = name_info->len;
82
83 /* encrypted bigname case */
84 if (fname && !fname->disk_name.name)
85 return cpu_to_le32(fname->hash);
86
87 if (is_dot_dotdot(name_info))
88 return 0;
89
90 /* Initialize the default seed for the hash checksum functions */
91 buf[0] = 0x67452301;
92 buf[1] = 0xefcdab89;
93 buf[2] = 0x98badcfe;
94 buf[3] = 0x10325476;
95
96 p = name;
97 while (1) {
98 str2hashbuf(p, len, in, 4);
99 TEA_transform(buf, in);
100 p += 16;
101 if (len <= 16)
102 break;
103 len -= 16;
104 }
105 hash = buf[0];
106 f2fs_hash = cpu_to_le32(hash & ~F2FS_HASH_COL_BIT);
107 return f2fs_hash;
108}
