32/*
33 * Check whether the given nid is within node id range.
34 */
35int check_nid_range(struct f2fs_sb_info *sbi, nid_t nid)
36{
37 if (unlikely(nid < F2FS_ROOT_INO(sbi) || nid >= NM_I(sbi)->max_nid)) {
38 set_sbi_flag(sbi, SBI_NEED_FSCK);
39 f2fs_msg(sbi->sb, KERN_WARNING,
40 "%s: out-of-range nid=%x, run fsck to fix.",
41 __func__, nid);
42 return -EINVAL;
43 }
44 return 0;
45}

47bool available_free_memory(struct f2fs_sb_info *sbi, int type)
48{
49 struct f2fs_nm_info *nm_i = NM_I(sbi);
50 struct sysinfo val;
51 unsigned long avail_ram;
52 unsigned long mem_size = 0;
53 bool res = false;
54
55 si_meminfo(&val);
56
57 /* only uses low memory */
58 avail_ram = val.totalram - val.totalhigh;
59
60 /*
61 * give 25%, 25%, 50%, 50%, 50% memory for each components respectively
62 */
63 if (type == FREE_NIDS) {
64 mem_size = (nm_i->nid_cnt[FREE_NID_LIST] *
65 sizeof(struct free_nid)) >> PAGE_SHIFT;
66 res = mem_size < ((avail_ram * nm_i->ram_thresh / 100) >> 2);
67 } else if (type == NAT_ENTRIES) {
68 mem_size = (nm_i->nat_cnt * sizeof(struct nat_entry)) >>
69 PAGE_SHIFT;
70 res = mem_size < ((avail_ram * nm_i->ram_thresh / 100) >> 2);
71 if (excess_cached_nats(sbi))
72 res = false;
73 } else if (type == DIRTY_DENTS) {
74 if (sbi->sb->s_bdi->wb.dirty_exceeded)
75 return false;
76 mem_size = get_pages(sbi, F2FS_DIRTY_DENTS);
77 res = mem_size < ((avail_ram * nm_i->ram_thresh / 100) >> 1);
78 } else if (type == INO_ENTRIES) {
79 int i;
80
81 for (i = 0; i <= UPDATE_INO; i++)
82 mem_size += sbi->im[i].ino_num *
83 sizeof(struct ino_entry);
84 mem_size >>= PAGE_SHIFT;
85 res = mem_size < ((avail_ram * nm_i->ram_thresh / 100) >> 1);
86 } else if (type == EXTENT_CACHE) {
87 mem_size = (atomic_read(&sbi->total_ext_tree) *
88 sizeof(struct extent_tree) +
89 atomic_read(&sbi->total_ext_node) *
90 sizeof(struct extent_node)) >> PAGE_SHIFT;
91 res = mem_size < ((avail_ram * nm_i->ram_thresh / 100) >> 1);
92 } else {
93 if (!sbi->sb->s_bdi->wb.dirty_exceeded)
94 return true;
95 }
96 return res;
97}

99static void clear_node_page_dirty(struct page *page)
100{
101 struct address_space *mapping = page->mapping;
102 unsigned int long flags;
103
104 if (PageDirty(page)) {
105 spin_lock_irqsave(&mapping->tree_lock, flags);
106 radix_tree_tag_clear(&mapping->page_tree,
107 page_index(page),
108 PAGECACHE_TAG_DIRTY);
109 spin_unlock_irqrestore(&mapping->tree_lock, flags);

111 clear_page_dirty_for_io(page);
112 dec_page_count(F2FS_M_SB(mapping), F2FS_DIRTY_NODES);
113 }
114 ClearPageUptodate(page);
115}

123static struct page *get_next_nat_page(struct f2fs_sb_info *sbi, nid_t nid)
124{
125 struct page *src_page;
126 struct page *dst_page;
127 pgoff_t src_off;
128 pgoff_t dst_off;
129 void *src_addr;
130 void *dst_addr;
131 struct f2fs_nm_info *nm_i = NM_I(sbi);
132
133 src_off = current_nat_addr(sbi, nid);
134 dst_off = next_nat_addr(sbi, src_off);
135
136 /* get current nat block page with lock */
137 src_page = get_meta_page(sbi, src_off);
138 dst_page = grab_meta_page(sbi, dst_off);
139 f2fs_bug_on(sbi, PageDirty(src_page));
140
141 src_addr = page_address(src_page);
142 dst_addr = page_address(dst_page);
143 memcpy(dst_addr, src_addr, PAGE_SIZE);
144 set_page_dirty(dst_page);
145 f2fs_put_page(src_page, 1);
146
147 set_to_next_nat(nm_i, nid);
148
149 return dst_page;
150}

171static void __set_nat_cache_dirty(struct f2fs_nm_info *nm_i,
172 struct nat_entry *ne)
173{
174 nid_t set = NAT_BLOCK_OFFSET(ne->ni.nid);
175 struct nat_entry_set *head;
176
177 head = radix_tree_lookup(&nm_i->nat_set_root, set);
178 if (!head) {
179 head = f2fs_kmem_cache_alloc(nat_entry_set_slab, GFP_NOFS);
180
181 INIT_LIST_HEAD(&head->entry_list);
182 INIT_LIST_HEAD(&head->set_list);
183 head->set = set;
184 head->entry_cnt = 0;
185 f2fs_radix_tree_insert(&nm_i->nat_set_root, set, head);
186 }
187
188 if (get_nat_flag(ne, IS_DIRTY))
189 goto refresh_list;
190
191 nm_i->dirty_nat_cnt++;
192 head->entry_cnt++;
193 set_nat_flag(ne, IS_DIRTY, true);
194refresh_list:
195 if (nat_get_blkaddr(ne) == NEW_ADDR)
196 list_del_init(&ne->list);
197 else
198 list_move_tail(&ne->list, &head->entry_list);
199}

264static struct nat_entry *grab_nat_entry(struct f2fs_nm_info *nm_i, nid_t nid,
265 bool no_fail)
266{
267 struct nat_entry *new;
268
269 if (no_fail) {
270 new = f2fs_kmem_cache_alloc(nat_entry_slab, GFP_NOFS);
271 f2fs_radix_tree_insert(&nm_i->nat_root, nid, new);
272 } else {
273 new = kmem_cache_alloc(nat_entry_slab, GFP_NOFS);
274 if (!new)
275 return NULL;
276 if (radix_tree_insert(&nm_i->nat_root, nid, new)) {
277 kmem_cache_free(nat_entry_slab, new);
278 return NULL;
279 }
280 }
281
282 memset(new, 0, sizeof(struct nat_entry));
283 nat_set_nid(new, nid);
284 nat_reset_flag(new);
285 list_add_tail(&new->list, &nm_i->nat_entries);
286 nm_i->nat_cnt++;
287 return new;
288}

290static void cache_nat_entry(struct f2fs_sb_info *sbi, nid_t nid,
291 struct f2fs_nat_entry *ne)
292{
293 struct f2fs_nm_info *nm_i = NM_I(sbi);
294 struct nat_entry *e;
295
296 e = __lookup_nat_cache(nm_i, nid);
297 if (!e) {
298 e = grab_nat_entry(nm_i, nid, false);
299 if (e)
300 node_info_from_raw_nat(&e->ni, ne);
301 } else {
302 f2fs_bug_on(sbi, nat_get_ino(e) != le32_to_cpu(ne->ino) ||
303 nat_get_blkaddr(e) !=
304 le32_to_cpu(ne->block_addr) ||
305 nat_get_version(e) != ne->version);
306 }
307}

309static void set_node_addr(struct f2fs_sb_info *sbi, struct node_info *ni,
310 block_t new_blkaddr, bool fsync_done)
311{
312 struct f2fs_nm_info *nm_i = NM_I(sbi);
313 struct nat_entry *e;
314
315 down_write(&nm_i->nat_tree_lock);
316 e = __lookup_nat_cache(nm_i, ni->nid);
317 if (!e) {
318 e = grab_nat_entry(nm_i, ni->nid, true);
319 copy_node_info(&e->ni, ni);
320 f2fs_bug_on(sbi, ni->blk_addr == NEW_ADDR);
321 } else if (new_blkaddr == NEW_ADDR) {
322 /*
323 * when nid is reallocated,
324 * previous nat entry can be remained in nat cache.
325 * So, reinitialize it with new information.
326 */
327 copy_node_info(&e->ni, ni);
328 f2fs_bug_on(sbi, ni->blk_addr != NULL_ADDR);
329 }
330
331 /* sanity check */
332 f2fs_bug_on(sbi, nat_get_blkaddr(e) != ni->blk_addr);
333 f2fs_bug_on(sbi, nat_get_blkaddr(e) == NULL_ADDR &&
334 new_blkaddr == NULL_ADDR);
335 f2fs_bug_on(sbi, nat_get_blkaddr(e) == NEW_ADDR &&
336 new_blkaddr == NEW_ADDR);
337 f2fs_bug_on(sbi, nat_get_blkaddr(e) != NEW_ADDR &&
338 nat_get_blkaddr(e) != NULL_ADDR &&
339 new_blkaddr == NEW_ADDR);
340
341 /* increment version no as node is removed */
342 if (nat_get_blkaddr(e) != NEW_ADDR && new_blkaddr == NULL_ADDR) {
343 unsigned char version = nat_get_version(e);
344 nat_set_version(e, inc_node_version(version));
345
346 /* in order to reuse the nid */
347 if (nm_i->next_scan_nid > ni->nid)
348 nm_i->next_scan_nid = ni->nid;
349 }
350
351 /* change address */
352 nat_set_blkaddr(e, new_blkaddr);
353 if (new_blkaddr == NEW_ADDR || new_blkaddr == NULL_ADDR)
354 set_nat_flag(e, IS_CHECKPOINTED, false);
355 __set_nat_cache_dirty(nm_i, e);
356
357 /* update fsync_mark if its inode nat entry is still alive */
358 if (ni->nid != ni->ino)
359 e = __lookup_nat_cache(nm_i, ni->ino);
360 if (e) {
361 if (fsync_done && ni->nid == ni->ino)
362 set_nat_flag(e, HAS_FSYNCED_INODE, true);
363 set_nat_flag(e, HAS_LAST_FSYNC, fsync_done);
364 }
365 up_write(&nm_i->nat_tree_lock);
366}

368int try_to_free_nats(struct f2fs_sb_info *sbi, int nr_shrink)
369{
370 struct f2fs_nm_info *nm_i = NM_I(sbi);
371 int nr = nr_shrink;
372
373 if (!down_write_trylock(&nm_i->nat_tree_lock))
374 return 0;
375
376 while (nr_shrink && !list_empty(&nm_i->nat_entries)) {
377 struct nat_entry *ne;
378 ne = list_first_entry(&nm_i->nat_entries,
379 struct nat_entry, list);
380 __del_from_nat_cache(nm_i, ne);
381 nr_shrink--;
382 }
383 up_write(&nm_i->nat_tree_lock);
384 return nr - nr_shrink;
385}

387/*
388 * This function always returns success
389 */
390void get_node_info(struct f2fs_sb_info *sbi, nid_t nid, struct node_info *ni)
391{
392 struct f2fs_nm_info *nm_i = NM_I(sbi);
393 struct curseg_info *curseg = CURSEG_I(sbi, CURSEG_HOT_DATA);
394 struct f2fs_journal *journal = curseg->journal;
395 nid_t start_nid = START_NID(nid);
396 struct f2fs_nat_block *nat_blk;
397 struct page *page = NULL;
398 struct f2fs_nat_entry ne;
399 struct nat_entry *e;
400 pgoff_t index;
401 int i;
402
403 ni->nid = nid;
404
405 /* Check nat cache */
406 down_read(&nm_i->nat_tree_lock);
407 e = __lookup_nat_cache(nm_i, nid);
408 if (e) {
409 ni->ino = nat_get_ino(e);
410 ni->blk_addr = nat_get_blkaddr(e);
411 ni->version = nat_get_version(e);
412 up_read(&nm_i->nat_tree_lock);
413 return;
414 }
415
416 memset(&ne, 0, sizeof(struct f2fs_nat_entry));
417
418 /* Check current segment summary */
419 down_read(&curseg->journal_rwsem);
420 i = lookup_journal_in_cursum(journal, NAT_JOURNAL, nid, 0);
421 if (i >= 0) {
422 ne = nat_in_journal(journal, i);
423 node_info_from_raw_nat(ni, &ne);
424 }
425 up_read(&curseg->journal_rwsem);
426 if (i >= 0) {
427 up_read(&nm_i->nat_tree_lock);
428 goto cache;
429 }
430
431 /* Fill node_info from nat page */
432 index = current_nat_addr(sbi, nid);
433 up_read(&nm_i->nat_tree_lock);
434
435 page = get_meta_page(sbi, index);
436 nat_blk = (struct f2fs_nat_block *)page_address(page);
437 ne = nat_blk->entries[nid - start_nid];
438 node_info_from_raw_nat(ni, &ne);
439 f2fs_put_page(page, 1);
440cache:
441 /* cache nat entry */
442 down_write(&nm_i->nat_tree_lock);
443 cache_nat_entry(sbi, nid, &ne);
444 up_write(&nm_i->nat_tree_lock);
445}

447/*
448 * readahead MAX_RA_NODE number of node pages.
449 */
450static void ra_node_pages(struct page *parent, int start, int n)
451{
452 struct f2fs_sb_info *sbi = F2FS_P_SB(parent);
453 struct blk_plug plug;
454 int i, end;
455 nid_t nid;
456
457 blk_start_plug(&plug);
458
459 /* Then, try readahead for siblings of the desired node */
460 end = start + n;
461 end = min(end, NIDS_PER_BLOCK);
462 for (i = start; i < end; i++) {
463 nid = get_nid(parent, i, false);
464 ra_node_page(sbi, nid);
465 }
466
467 blk_finish_plug(&plug);
468}

470pgoff_t get_next_page_offset(struct dnode_of_data *dn, pgoff_t pgofs)
471{
472 const long direct_index = ADDRS_PER_INODE(dn->inode);
473 const long direct_blks = ADDRS_PER_BLOCK;
474 const long indirect_blks = ADDRS_PER_BLOCK * NIDS_PER_BLOCK;
475 unsigned int skipped_unit = ADDRS_PER_BLOCK;
476 int cur_level = dn->cur_level;
477 int max_level = dn->max_level;
478 pgoff_t base = 0;
479
480 if (!dn->max_level)
481 return pgofs + 1;
482
483 while (max_level-- > cur_level)
484 skipped_unit *= NIDS_PER_BLOCK;
485
486 switch (dn->max_level) {
487 case 3:
488 base += 2 * indirect_blks;
489 case 2:
490 base += 2 * direct_blks;
491 case 1:
492 base += direct_index;
493 break;
494 default:
495 f2fs_bug_on(F2FS_I_SB(dn->inode), 1);
496 }
497
498 return ((pgofs - base) / skipped_unit + 1) * skipped_unit + base;
499}

501/*
502 * The maximum depth is four.
503 * Offset[0] will have raw inode offset.
504 */
505static int get_node_path(struct inode *inode, long block,
506 int offset[4], unsigned int noffset[4])
507{
508 const long direct_index = ADDRS_PER_INODE(inode);
509 const long direct_blks = ADDRS_PER_BLOCK;
510 const long dptrs_per_blk = NIDS_PER_BLOCK;
511 const long indirect_blks = ADDRS_PER_BLOCK * NIDS_PER_BLOCK;
512 const long dindirect_blks = indirect_blks * NIDS_PER_BLOCK;
513 int n = 0;
514 int level = 0;
515
516 noffset[0] = 0;
517
518 if (block < direct_index) {
519 offset[n] = block;
520 goto got;
521 }
522 block -= direct_index;
523 if (block < direct_blks) {
524 offset[n++] = NODE_DIR1_BLOCK;
525 noffset[n] = 1;
526 offset[n] = block;
527 level = 1;
528 goto got;
529 }
530 block -= direct_blks;
531 if (block < direct_blks) {
532 offset[n++] = NODE_DIR2_BLOCK;
533 noffset[n] = 2;
534 offset[n] = block;
535 level = 1;
536 goto got;
537 }
538 block -= direct_blks;
539 if (block < indirect_blks) {
540 offset[n++] = NODE_IND1_BLOCK;
541 noffset[n] = 3;
542 offset[n++] = block / direct_blks;
543 noffset[n] = 4 + offset[n - 1];
544 offset[n] = block % direct_blks;
545 level = 2;
546 goto got;
547 }
548 block -= indirect_blks;
549 if (block < indirect_blks) {
550 offset[n++] = NODE_IND2_BLOCK;
551 noffset[n] = 4 + dptrs_per_blk;
552 offset[n++] = block / direct_blks;
553 noffset[n] = 5 + dptrs_per_blk + offset[n - 1];
554 offset[n] = block % direct_blks;
555 level = 2;
556 goto got;
557 }
558 block -= indirect_blks;
559 if (block < dindirect_blks) {
560 offset[n++] = NODE_DIND_BLOCK;
561 noffset[n] = 5 + (dptrs_per_blk * 2);
562 offset[n++] = block / indirect_blks;
563 noffset[n] = 6 + (dptrs_per_blk * 2) +
564 offset[n - 1] * (dptrs_per_blk + 1);
565 offset[n++] = (block / direct_blks) % dptrs_per_blk;
566 noffset[n] = 7 + (dptrs_per_blk * 2) +
567 offset[n - 2] * (dptrs_per_blk + 1) +
568 offset[n - 1];
569 offset[n] = block % direct_blks;
570 level = 3;
571 goto got;
572 } else {
573 return -E2BIG;
574 }
575got:
576 return level;
577}

579/*
580 * Caller should call f2fs_put_dnode(dn).
581 * Also, it should grab and release a rwsem by calling f2fs_lock_op() and
582 * f2fs_unlock_op() only if ro is not set RDONLY_NODE.
583 * In the case of RDONLY_NODE, we don't need to care about mutex.
584 */
585int get_dnode_of_data(struct dnode_of_data *dn, pgoff_t index, int mode)
586{
587 struct f2fs_sb_info *sbi = F2FS_I_SB(dn->inode);
588 struct page *npage[4];
589 struct page *parent = NULL;
590 int offset[4];
591 unsigned int noffset[4];
592 nid_t nids[4];
593 int level, i = 0;
594 int err = 0;
595
596 level = get_node_path(dn->inode, index, offset, noffset);
597 if (level < 0)
598 return level;
599
600 nids[0] = dn->inode->i_ino;
601 npage[0] = dn->inode_page;
602
603 if (!npage[0]) {
604 npage[0] = get_node_page(sbi, nids[0]);
605 if (IS_ERR(npage[0]))
606 return PTR_ERR(npage[0]);
607 }
608
609 /* if inline_data is set, should not report any block indices */
610 if (f2fs_has_inline_data(dn->inode) && index) {
611 err = -ENOENT;
612 f2fs_put_page(npage[0], 1);
613 goto release_out;
614 }
615
616 parent = npage[0];
617 if (level != 0)
618 nids[1] = get_nid(parent, offset[0], true);
619 dn->inode_page = npage[0];
620 dn->inode_page_locked = true;
621
622 /* get indirect or direct nodes */
623 for (i = 1; i <= level; i++) {
624 bool done = false;
625
626 if (!nids[i] && mode == ALLOC_NODE) {
627 /* alloc new node */
628 if (!alloc_nid(sbi, &(nids[i]))) {
629 err = -ENOSPC;
630 goto release_pages;
631 }
632
633 dn->nid = nids[i];
634 npage[i] = new_node_page(dn, noffset[i]);
635 if (IS_ERR(npage[i])) {
636 alloc_nid_failed(sbi, nids[i]);
637 err = PTR_ERR(npage[i]);
638 goto release_pages;
639 }
640
641 set_nid(parent, offset[i - 1], nids[i], i == 1);
642 alloc_nid_done(sbi, nids[i]);
643 done = true;
644 } else if (mode == LOOKUP_NODE_RA && i == level && level > 1) {
645 npage[i] = get_node_page_ra(parent, offset[i - 1]);
646 if (IS_ERR(npage[i])) {
647 err = PTR_ERR(npage[i]);
648 goto release_pages;
649 }
650 done = true;
651 }
652 if (i == 1) {
653 dn->inode_page_locked = false;
654 unlock_page(parent);
655 } else {
656 f2fs_put_page(parent, 1);
657 }
658
659 if (!done) {
660 npage[i] = get_node_page(sbi, nids[i]);
661 if (IS_ERR(npage[i])) {
662 err = PTR_ERR(npage[i]);
663 f2fs_put_page(npage[0], 0);
664 goto release_out;
665 }
666 }
667 if (i < level) {
668 parent = npage[i];
669 nids[i + 1] = get_nid(parent, offset[i], false);
670 }
671 }
672 dn->nid = nids[level];
673 dn->ofs_in_node = offset[level];
674 dn->node_page = npage[level];
675 dn->data_blkaddr = datablock_addr(dn->inode,
676 dn->node_page, dn->ofs_in_node);
677 return 0;
678
679release_pages:
680 f2fs_put_page(parent, 1);
681 if (i > 1)
682 f2fs_put_page(npage[0], 0);
683release_out:
684 dn->inode_page = NULL;
685 dn->node_page = NULL;
686 if (err == -ENOENT) {
687 dn->cur_level = i;
688 dn->max_level = level;
689 dn->ofs_in_node = offset[level];
690 }
691 return err;
692}

694static void truncate_node(struct dnode_of_data *dn)
695{
696 struct f2fs_sb_info *sbi = F2FS_I_SB(dn->inode);
697 struct node_info ni;
698
699 get_node_info(sbi, dn->nid, &ni);
700 f2fs_bug_on(sbi, ni.blk_addr == NULL_ADDR);

717
718 invalidate_mapping_pages(NODE_MAPPING(sbi),
719 dn->node_page->index, dn->node_page->index);

721 dn->node_page = NULL;

747static int truncate_nodes(struct dnode_of_data *dn, unsigned int nofs,
748 int ofs, int depth)
749{
750 struct dnode_of_data rdn = *dn;
751 struct page *page;
752 struct f2fs_node *rn;
753 nid_t child_nid;
754 unsigned int child_nofs;
755 int freed = 0;
756 int i, ret;
757
758 if (dn->nid == 0)
759 return NIDS_PER_BLOCK + 1;
760
761 trace_f2fs_truncate_nodes_enter(dn->inode, dn->nid, dn->data_blkaddr);
762
763 page = get_node_page(F2FS_I_SB(dn->inode), dn->nid);
764 if (IS_ERR(page)) {
765 trace_f2fs_truncate_nodes_exit(dn->inode, PTR_ERR(page));
766 return PTR_ERR(page);
767 }
768
769 ra_node_pages(page, ofs, NIDS_PER_BLOCK);
770
771 rn = F2FS_NODE(page);
772 if (depth < 3) {
773 for (i = ofs; i < NIDS_PER_BLOCK; i++, freed++) {
774 child_nid = le32_to_cpu(rn->in.nid[i]);
775 if (child_nid == 0)
776 continue;
777 rdn.nid = child_nid;
778 ret = truncate_dnode(&rdn);
779 if (ret < 0)
780 goto out_err;
781 if (set_nid(page, i, 0, false))
782 dn->node_changed = true;
783 }
784 } else {
785 child_nofs = nofs + ofs * (NIDS_PER_BLOCK + 1) + 1;
786 for (i = ofs; i < NIDS_PER_BLOCK; i++) {
787 child_nid = le32_to_cpu(rn->in.nid[i]);
788 if (child_nid == 0) {
789 child_nofs += NIDS_PER_BLOCK + 1;
790 continue;
791 }
792 rdn.nid = child_nid;
793 ret = truncate_nodes(&rdn, child_nofs, 0, depth - 1);
794 if (ret == (NIDS_PER_BLOCK + 1)) {
795 if (set_nid(page, i, 0, false))
796 dn->node_changed = true;
797 child_nofs += ret;
798 } else if (ret < 0 && ret != -ENOENT) {
799 goto out_err;
800 }
801 }
802 freed = child_nofs;
803 }
804
805 if (!ofs) {
806 /* remove current indirect node */
807 dn->node_page = page;
808 truncate_node(dn);
809 freed++;
810 } else {
811 f2fs_put_page(page, 1);
812 }
813 trace_f2fs_truncate_nodes_exit(dn->inode, freed);
814 return freed;
815
816out_err:
817 f2fs_put_page(page, 1);
818 trace_f2fs_truncate_nodes_exit(dn->inode, ret);
819 return ret;
820}

822static int truncate_partial_nodes(struct dnode_of_data *dn,
823 struct f2fs_inode *ri, int *offset, int depth)
824{
825 struct page *pages[2];
826 nid_t nid[3];
827 nid_t child_nid;
828 int err = 0;
829 int i;
830 int idx = depth - 2;
831
832 nid[0] = le32_to_cpu(ri->i_nid[offset[0] - NODE_DIR1_BLOCK]);
833 if (!nid[0])
834 return 0;
835
836 /* get indirect nodes in the path */
837 for (i = 0; i < idx + 1; i++) {
838 /* reference count'll be increased */
839 pages[i] = get_node_page(F2FS_I_SB(dn->inode), nid[i]);
840 if (IS_ERR(pages[i])) {
841 err = PTR_ERR(pages[i]);
842 idx = i - 1;
843 goto fail;
844 }
845 nid[i + 1] = get_nid(pages[i], offset[i + 1], false);
846 }
847
848 ra_node_pages(pages[idx], offset[idx + 1], NIDS_PER_BLOCK);
849
850 /* free direct nodes linked to a partial indirect node */
851 for (i = offset[idx + 1]; i < NIDS_PER_BLOCK; i++) {
852 child_nid = get_nid(pages[idx], i, false);
853 if (!child_nid)
854 continue;
855 dn->nid = child_nid;
856 err = truncate_dnode(dn);
857 if (err < 0)
858 goto fail;
859 if (set_nid(pages[idx], i, 0, false))
860 dn->node_changed = true;
861 }
862
863 if (offset[idx + 1] == 0) {
864 dn->node_page = pages[idx];
865 dn->nid = nid[idx];
866 truncate_node(dn);
867 } else {
868 f2fs_put_page(pages[idx], 1);
869 }
870 offset[idx]++;
871 offset[idx + 1] = 0;
872 idx--;
873fail:
874 for (i = idx; i >= 0; i--)
875 f2fs_put_page(pages[i], 1);
876
877 trace_f2fs_truncate_partial_nodes(dn->inode, nid, depth, err);
878
879 return err;
880}

882/*
883 * All the block addresses of data and nodes should be nullified.
884 */
885int truncate_inode_blocks(struct inode *inode, pgoff_t from)
886{
887 struct f2fs_sb_info *sbi = F2FS_I_SB(inode);
888 int err = 0, cont = 1;
889 int level, offset[4], noffset[4];
890 unsigned int nofs = 0;
891 struct f2fs_inode *ri;
892 struct dnode_of_data dn;
893 struct page *page;
894
895 trace_f2fs_truncate_inode_blocks_enter(inode, from);
896
897 level = get_node_path(inode, from, offset, noffset);
898 if (level < 0)
899 return level;
900
901 page = get_node_page(sbi, inode->i_ino);
902 if (IS_ERR(page)) {
903 trace_f2fs_truncate_inode_blocks_exit(inode, PTR_ERR(page));
904 return PTR_ERR(page);
905 }
906
907 set_new_dnode(&dn, inode, page, NULL, 0);
908 unlock_page(page);
909
910 ri = F2FS_INODE(page);
911 switch (level) {
912 case 0:
913 case 1:
914 nofs = noffset[1];
915 break;
916 case 2:
917 nofs = noffset[1];
918 if (!offset[level - 1])
919 goto skip_partial;
920 err = truncate_partial_nodes(&dn, ri, offset, level);
921 if (err < 0 && err != -ENOENT)
922 goto fail;
923 nofs += 1 + NIDS_PER_BLOCK;
924 break;
925 case 3:
926 nofs = 5 + 2 * NIDS_PER_BLOCK;
927 if (!offset[level - 1])
928 goto skip_partial;
929 err = truncate_partial_nodes(&dn, ri, offset, level);
930 if (err < 0 && err != -ENOENT)
931 goto fail;
932 break;
933 default:
934 BUG();
935 }
936
937skip_partial:
938 while (cont) {
939 dn.nid = le32_to_cpu(ri->i_nid[offset[0] - NODE_DIR1_BLOCK]);
940 switch (offset[0]) {
941 case NODE_DIR1_BLOCK:
942 case NODE_DIR2_BLOCK:
943 err = truncate_dnode(&dn);
944 break;
945
946 case NODE_IND1_BLOCK:
947 case NODE_IND2_BLOCK:
948 err = truncate_nodes(&dn, nofs, offset[1], 2);
949 break;
950
951 case NODE_DIND_BLOCK:
952 err = truncate_nodes(&dn, nofs, offset[1], 3);
953 cont = 0;
954 break;
955
956 default:
957 BUG();
958 }
959 if (err < 0 && err != -ENOENT)
960 goto fail;
961 if (offset[1] == 0 &&
962 ri->i_nid[offset[0] - NODE_DIR1_BLOCK]) {
963 lock_page(page);
964 BUG_ON(page->mapping != NODE_MAPPING(sbi));
965 f2fs_wait_on_page_writeback(page, NODE, true);
966 ri->i_nid[offset[0] - NODE_DIR1_BLOCK] = 0;
967 set_page_dirty(page);
968 unlock_page(page);
969 }
970 offset[1] = 0;
971 offset[0]++;
972 nofs += err;
973 }
974fail:
975 f2fs_put_page(page, 0);
976 trace_f2fs_truncate_inode_blocks_exit(inode, err);
977 return err > 0 ? 0 : err;
978}

1050{

1052 struct node_info new_ni;
1053 struct page *page;

1055
1056 if (unlikely(is_inode_flag_set(dn->inode, FI_NO_ALLOC)))
1057 return ERR_PTR(-EPERM);
1058
1059 page = f2fs_grab_cache_page(NODE_MAPPING(sbi), dn->nid, false);
1060 if (!page)
1061 return ERR_PTR(-ENOMEM);
1062
1063 if (unlikely((err = inc_valid_node_count(sbi, dn->inode, !ofs))))
1064 goto fail;

1087
1088 if (ofs == 0)

1092fail:

1157static struct page *__get_node_page(struct f2fs_sb_info *sbi, pgoff_t nid,
1158 struct page *parent, int start)
1159{
1160 struct page *page;
1161 int err;
1162
1163 if (!nid)
1164 return ERR_PTR(-ENOENT);
1165 if (check_nid_range(sbi, nid))
1166 return ERR_PTR(-EINVAL);
1167repeat:
1168 page = f2fs_grab_cache_page(NODE_MAPPING(sbi), nid, false);
1169 if (!page)
1170 return ERR_PTR(-ENOMEM);
1171
1172 err = read_node_page(page, 0);
1173 if (err < 0) {
1174 f2fs_put_page(page, 1);
1175 return ERR_PTR(err);
1176 } else if (err == LOCKED_PAGE) {
1177 err = 0;
1178 goto page_hit;
1179 }
1180
1181 if (parent)

1187 f2fs_put_page(page, 1);

1189 }

1191 if (unlikely(!PageUptodate(page))) {
1192 err = -EIO;

1204 nid, nid_of_node(page), ino_of_node(page),
1205 ofs_of_node(page), cpver_of_node(page),
1206 next_blkaddr_of_node(page));

1208out_err:

1210 f2fs_put_page(page, 1);

1212 }
1213 return page;
1214}

1295static struct page *last_fsync_dnode(struct f2fs_sb_info *sbi, nid_t ino)
1296{
1297 pgoff_t index, end;
1298 struct pagevec pvec;
1299 struct page *last_page = NULL;
1300
1301 pagevec_init(&pvec, 0);
1302 index = 0;
1303 end = ULONG_MAX;
1304
1305 while (index <= end) {
1306 int i, nr_pages;
1307 nr_pages = pagevec_lookup_tag(&pvec, NODE_MAPPING(sbi), &index,
1308 PAGECACHE_TAG_DIRTY,
1309 min(end - index, (pgoff_t)PAGEVEC_SIZE-1) + 1);
1310 if (nr_pages == 0)
1311 break;
1312
1313 for (i = 0; i < nr_pages; i++) {
1314 struct page *page = pvec.pages[i];
1315
1316 if (unlikely(f2fs_cp_error(sbi))) {
1317 f2fs_put_page(last_page, 0);
1318 pagevec_release(&pvec);
1319 return ERR_PTR(-EIO);
1320 }
1321
1322 if (!IS_DNODE(page) || !is_cold_node(page))
1323 continue;
1324 if (ino_of_node(page) != ino)
1325 continue;
1326
1327 lock_page(page);
1328
1329 if (unlikely(page->mapping != NODE_MAPPING(sbi))) {
1330continue_unlock:
1331 unlock_page(page);
1332 continue;
1333 }
1334 if (ino_of_node(page) != ino)
1335 goto continue_unlock;
1336
1337 if (!PageDirty(page)) {
1338 /* someone wrote it for us */
1339 goto continue_unlock;
1340 }
1341
1342 if (last_page)
1343 f2fs_put_page(last_page, 0);
1344
1345 get_page(page);
1346 last_page = page;
1347 unlock_page(page);
1348 }
1349 pagevec_release(&pvec);
1350 cond_resched();
1351 }
1352 return last_page;
1353}

1355static int __write_node_page(struct page *page, bool atomic, bool *submitted,
1356 struct writeback_control *wbc, bool do_balance,
1357 enum iostat_type io_type)
1358{
1359 struct f2fs_sb_info *sbi = F2FS_P_SB(page);
1360 nid_t nid;
1361 struct node_info ni;
1362 struct f2fs_io_info fio = {
1363 .sbi = sbi,
1364 .type = NODE,
1365 .op = REQ_OP_WRITE,
1366 .op_flags = wbc_to_write_flags(wbc),
1367 .page = page,
1368 .encrypted_page = NULL,
1369 .submitted = false,
1370 .io_type = io_type,
1371 };
1372
1373 trace_f2fs_writepage(page, NODE);
1374
1375 if (unlikely(is_sbi_flag_set(sbi, SBI_POR_DOING)))
1376 goto redirty_out;
1377 if (unlikely(f2fs_cp_error(sbi)))
1378 goto redirty_out;
1379
1380 /* get old block addr of this node page */
1381 nid = nid_of_node(page);
1382 f2fs_bug_on(sbi, page->index != nid);
1383
1384 if (wbc->for_reclaim) {
1385 if (!down_read_trylock(&sbi->node_write))
1386 goto redirty_out;
1387 } else {
1388 down_read(&sbi->node_write);
1389 }
1390
1391 get_node_info(sbi, nid, &ni);
1392
1393 /* This page is already truncated */
1394 if (unlikely(ni.blk_addr == NULL_ADDR)) {
1395 ClearPageUptodate(page);
1396 dec_page_count(sbi, F2FS_DIRTY_NODES);
1397 up_read(&sbi->node_write);
1398 unlock_page(page);
1399 return 0;
1400 }
1401
1402 if (atomic && !test_opt(sbi, NOBARRIER))
1403 fio.op_flags |= REQ_PREFLUSH | REQ_FUA;
1404
1405 set_page_writeback(page);
1406 fio.old_blkaddr = ni.blk_addr;
1407 write_node_page(nid, &fio);
1408 set_node_addr(sbi, &ni, fio.new_blkaddr, is_fsync_dnode(page));
1409 dec_page_count(sbi, F2FS_DIRTY_NODES);
1410 up_read(&sbi->node_write);
1411
1412 if (wbc->for_reclaim) {
1413 f2fs_submit_merged_write_cond(sbi, page->mapping->host, 0,
1414 page->index, NODE);
1415 submitted = NULL;
1416 }
1417
1418 unlock_page(page);
1419
1420 if (unlikely(f2fs_cp_error(sbi))) {
1421 f2fs_submit_merged_write(sbi, NODE);
1422 submitted = NULL;
1423 }
1424 if (submitted)
1425 *submitted = fio.submitted;
1426
1427 if (do_balance)
1428 f2fs_balance_fs(sbi, false);
1429 return 0;
1430
1431redirty_out:
1432 redirty_page_for_writepage(wbc, page);
1433 return AOP_WRITEPAGE_ACTIVATE;
1434}

1442int fsync_node_pages(struct f2fs_sb_info *sbi, struct inode *inode,
1443 struct writeback_control *wbc, bool atomic)
1444{
1445 pgoff_t index, end;
1446 pgoff_t last_idx = ULONG_MAX;
1447 struct pagevec pvec;
1448 int ret = 0;
1449 struct page *last_page = NULL;
1450 bool marked = false;
1451 nid_t ino = inode->i_ino;
1452
1453 if (atomic) {
1454 last_page = last_fsync_dnode(sbi, ino);
1455 if (IS_ERR_OR_NULL(last_page))
1456 return PTR_ERR_OR_ZERO(last_page);
1457 }
1458retry:
1459 pagevec_init(&pvec, 0);
1460 index = 0;
1461 end = ULONG_MAX;
1462
1463 while (index <= end) {
1464 int i, nr_pages;
1465 nr_pages = pagevec_lookup_tag(&pvec, NODE_MAPPING(sbi), &index,
1466 PAGECACHE_TAG_DIRTY,
1467 min(end - index, (pgoff_t)PAGEVEC_SIZE-1) + 1);
1468 if (nr_pages == 0)
1469 break;
1470
1471 for (i = 0; i < nr_pages; i++) {
1472 struct page *page = pvec.pages[i];
1473 bool submitted = false;
1474
1475 if (unlikely(f2fs_cp_error(sbi))) {
1476 f2fs_put_page(last_page, 0);
1477 pagevec_release(&pvec);
1478 ret = -EIO;
1479 goto out;
1480 }
1481
1482 if (!IS_DNODE(page) || !is_cold_node(page))
1483 continue;
1484 if (ino_of_node(page) != ino)
1485 continue;
1486
1487 lock_page(page);
1488
1489 if (unlikely(page->mapping != NODE_MAPPING(sbi))) {
1490continue_unlock:
1491 unlock_page(page);
1492 continue;
1493 }
1494 if (ino_of_node(page) != ino)
1495 goto continue_unlock;
1496
1497 if (!PageDirty(page) && page != last_page) {
1498 /* someone wrote it for us */
1499 goto continue_unlock;
1500 }
1501
1502 f2fs_wait_on_page_writeback(page, NODE, true);
1503 BUG_ON(PageWriteback(page));
1504
1505 set_fsync_mark(page, 0);
1506 set_dentry_mark(page, 0);
1507
1508 if (!atomic || page == last_page) {
1509 set_fsync_mark(page, 1);
1510 if (IS_INODE(page)) {
1511 if (is_inode_flag_set(inode,
1512 FI_DIRTY_INODE))
1513 update_inode(inode, page);
1514 set_dentry_mark(page,
1515 need_dentry_mark(sbi, ino));
1516 }
1517 /* may be written by other thread */
1518 if (!PageDirty(page))
1519 set_page_dirty(page);
1520 }
1521
1522 if (!clear_page_dirty_for_io(page))
1523 goto continue_unlock;
1524
1525 ret = __write_node_page(page, atomic &&
1526 page == last_page,
1527 &submitted, wbc, true,
1528 FS_NODE_IO);
1529 if (ret) {
1530 unlock_page(page);
1531 f2fs_put_page(last_page, 0);
1532 break;
1533 } else if (submitted) {
1534 last_idx = page->index;
1535 }
1536
1537 if (page == last_page) {
1538 f2fs_put_page(page, 0);
1539 marked = true;
1540 break;
1541 }
1542 }
1543 pagevec_release(&pvec);
1544 cond_resched();
1545
1546 if (ret || marked)
1547 break;
1548 }
1549 if (!ret && atomic && !marked) {
1550 f2fs_msg(sbi->sb, KERN_DEBUG,
1551 "Retry to write fsync mark: ino=%u, idx=%lx",
1552 ino, last_page->index);
1553 lock_page(last_page);
1554 f2fs_wait_on_page_writeback(last_page, NODE, true);
1555 set_page_dirty(last_page);
1556 unlock_page(last_page);
1557 goto retry;
1558 }
1559out:
1560 if (last_idx != ULONG_MAX)
1561 f2fs_submit_merged_write_cond(sbi, NULL, ino, last_idx, NODE);
1562 return ret ? -EIO: 0;
1563}

1565int sync_node_pages(struct f2fs_sb_info *sbi, struct writeback_control *wbc,
1566 bool do_balance, enum iostat_type io_type)
1567{
1568 pgoff_t index, end;
1569 struct pagevec pvec;
1570 int step = 0;
1571 int nwritten = 0;
1572 int ret = 0;
1573
1574 pagevec_init(&pvec, 0);
1575
1576next_step:
1577 index = 0;
1578 end = ULONG_MAX;
1579
1580 while (index <= end) {
1581 int i, nr_pages;
1582 nr_pages = pagevec_lookup_tag(&pvec, NODE_MAPPING(sbi), &index,
1583 PAGECACHE_TAG_DIRTY,
1584 min(end - index, (pgoff_t)PAGEVEC_SIZE-1) + 1);
1585 if (nr_pages == 0)
1586 break;
1587
1588 for (i = 0; i < nr_pages; i++) {
1589 struct page *page = pvec.pages[i];
1590 bool submitted = false;
1591
1592 if (unlikely(f2fs_cp_error(sbi))) {
1593 pagevec_release(&pvec);
1594 ret = -EIO;
1595 goto out;
1596 }
1597
1598 /*
1599 * flushing sequence with step:
1600 * 0. indirect nodes
1601 * 1. dentry dnodes
1602 * 2. file dnodes
1603 */
1604 if (step == 0 && IS_DNODE(page))
1605 continue;
1606 if (step == 1 && (!IS_DNODE(page) ||
1607 is_cold_node(page)))
1608 continue;
1609 if (step == 2 && (!IS_DNODE(page) ||
1610 !is_cold_node(page)))
1611 continue;
1612lock_node:
1613 if (!trylock_page(page))
1614 continue;
1615
1616 if (unlikely(page->mapping != NODE_MAPPING(sbi))) {
1617continue_unlock:
1618 unlock_page(page);
1619 continue;
1620 }
1621
1622 if (!PageDirty(page)) {
1623 /* someone wrote it for us */
1624 goto continue_unlock;
1625 }
1626
1627 /* flush inline_data */
1628 if (is_inline_node(page)) {
1629 clear_inline_node(page);
1630 unlock_page(page);
1631 flush_inline_data(sbi, ino_of_node(page));
1632 goto lock_node;
1633 }
1634
1635 f2fs_wait_on_page_writeback(page, NODE, true);
1636
1637 BUG_ON(PageWriteback(page));
1638 if (!clear_page_dirty_for_io(page))
1639 goto continue_unlock;
1640
1641 set_fsync_mark(page, 0);
1642 set_dentry_mark(page, 0);
1643
1644 ret = __write_node_page(page, false, &submitted,
1645 wbc, do_balance, io_type);
1646 if (ret)
1647 unlock_page(page);
1648 else if (submitted)
1649 nwritten++;
1650
1651 if (--wbc->nr_to_write == 0)
1652 break;
1653 }
1654 pagevec_release(&pvec);
1655 cond_resched();
1656
1657 if (wbc->nr_to_write == 0) {
1658 step = 2;
1659 break;
1660 }
1661 }
1662
1663 if (step < 2) {
1664 step++;
1665 goto next_step;
1666 }
1667out:
1668 if (nwritten)
1669 f2fs_submit_merged_write(sbi, NODE);
1670 return ret;
1671}

1712static int f2fs_write_node_pages(struct address_space *mapping,
1713 struct writeback_control *wbc)
1714{
1715 struct f2fs_sb_info *sbi = F2FS_M_SB(mapping);
1716 struct blk_plug plug;
1717 long diff;
1718
1719 if (unlikely(is_sbi_flag_set(sbi, SBI_POR_DOING)))
1720 goto skip_write;
1721
1722 /* balancing f2fs's metadata in background */
1723 f2fs_balance_fs_bg(sbi);
1724
1725 /* collect a number of dirty node pages and write together */
1726 if (get_pages(sbi, F2FS_DIRTY_NODES) < nr_pages_to_skip(sbi, NODE))
1727 goto skip_write;
1728
1729 trace_f2fs_writepages(mapping->host, wbc, NODE);
1730
1731 diff = nr_pages_to_write(sbi, NODE, wbc);
1732 wbc->sync_mode = WB_SYNC_NONE;
1733 blk_start_plug(&plug);
1734 sync_node_pages(sbi, wbc, true, FS_NODE_IO);
1735 blk_finish_plug(&plug);
1736 wbc->nr_to_write = max((long)0, wbc->nr_to_write - diff);
1737 return 0;
1738
1739skip_write:
1740 wbc->pages_skipped += get_pages(sbi, F2FS_DIRTY_NODES);
1741 trace_f2fs_writepages(mapping->host, wbc, NODE);
1742 return 0;
1743}

1951{
1952 struct f2fs_nm_info *nm_i = NM_I(sbi);
1953 struct curseg_info *curseg = CURSEG_I(sbi, CURSEG_HOT_DATA);

1961 continue;

1963 continue;

1965 nid_t nid;
1966

1971 add_free_nid(sbi, nid, true);

1974 goto out;
1975 }
1976 }

1992}

1994static void __build_free_nids(struct f2fs_sb_info *sbi, bool sync, bool mount)

2586 for (i = 0; i < nm_i->nat_bits_blocks; i++) {
2587 struct page *page = get_meta_page(sbi, nat_bits_addr++);
2588
2589 memcpy(nm_i->nat_bits + (i << F2FS_BLKSIZE_BITS),
2590 page_address(page), F2FS_BLKSIZE);
2591 f2fs_put_page(page, 1);
2592 }
2593
2594 cp_ver |= (cur_cp_crc(ckpt) << 32);
2595 if (cpu_to_le64(cp_ver) != *(__le64 *)nm_i->nat_bits) {
2596 disable_nat_bits(sbi, true);
2597 return 0;
2598 }
2599
2600 nm_i->full_nat_bits = nm_i->nat_bits + 8;
2601 nm_i->empty_nat_bits = nm_i->full_nat_bits + nat_bits_bytes;
2602
2603 f2fs_msg(sbi->sb, KERN_NOTICE, "Found nat_bits in checkpoint");
2604 return 0;
2605}

2607static inline void load_free_nid_bitmap(struct f2fs_sb_info *sbi)
2608{

2613 if (!enabled_nat_bits(sbi, NULL))

2746void destroy_node_manager(struct f2fs_sb_info *sbi)
2747{
2748 struct f2fs_nm_info *nm_i = NM_I(sbi);
2749 struct free_nid *i, *next_i;
2750 struct nat_entry *natvec[NATVEC_SIZE];
2751 struct nat_entry_set *setvec[SETVEC_SIZE];
2752 nid_t nid = 0;
2753 unsigned int found;
2754
2755 if (!nm_i)
2756 return;
2757
2758 /* destroy free nid list */
2759 spin_lock(&nm_i->nid_list_lock);
2760 list_for_each_entry_safe(i, next_i, &nm_i->nid_list[FREE_NID_LIST],
2761 list) {
2762 __remove_nid_from_list(sbi, i, FREE_NID_LIST, false);
2763 spin_unlock(&nm_i->nid_list_lock);
2764 kmem_cache_free(free_nid_slab, i);
2765 spin_lock(&nm_i->nid_list_lock);
2766 }
2767 f2fs_bug_on(sbi, nm_i->nid_cnt[FREE_NID_LIST]);
2768 f2fs_bug_on(sbi, nm_i->nid_cnt[ALLOC_NID_LIST]);
2769 f2fs_bug_on(sbi, !list_empty(&nm_i->nid_list[ALLOC_NID_LIST]));
2770 spin_unlock(&nm_i->nid_list_lock);
2771
2772 /* destroy nat cache */
2773 down_write(&nm_i->nat_tree_lock);
2774 while ((found = __gang_lookup_nat_cache(nm_i,
2775 nid, NATVEC_SIZE, natvec))) {
2776 unsigned idx;
2777
2778 nid = nat_get_nid(natvec[found - 1]) + 1;
2779 for (idx = 0; idx < found; idx++)
2780 __del_from_nat_cache(nm_i, natvec[idx]);
2781 }
2782 f2fs_bug_on(sbi, nm_i->nat_cnt);
2783
2784 /* destroy nat set cache */
2785 nid = 0;
2786 while ((found = __gang_lookup_nat_set(nm_i,
2787 nid, SETVEC_SIZE, setvec))) {
2788 unsigned idx;
2789
2790 nid = setvec[found - 1]->set + 1;
2791 for (idx = 0; idx < found; idx++) {
2792 /* entry_cnt is not zero, when cp_error was occurred */
2793 f2fs_bug_on(sbi, !list_empty(&setvec[idx]->entry_list));
2794 radix_tree_delete(&nm_i->nat_set_root, setvec[idx]->set);
2795 kmem_cache_free(nat_entry_set_slab, setvec[idx]);
2796 }
2797 }
2798 up_write(&nm_i->nat_tree_lock);
2799
2800 kvfree(nm_i->nat_block_bitmap);
2801 kvfree(nm_i->free_nid_bitmap);
2802 kvfree(nm_i->free_nid_count);
2803
2804 kfree(nm_i->nat_bitmap);
2805 kfree(nm_i->nat_bits);
2806#ifdef CONFIG_F2FS_CHECK_FS
2807 kfree(nm_i->nat_bitmap_mir);
2808#endif
2809 sbi->nm_info = NULL;
2810 kfree(nm_i);
2811}
