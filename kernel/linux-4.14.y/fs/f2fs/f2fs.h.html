423static inline void make_dentry_ptr_inline(struct inode *inode,
424 struct f2fs_dentry_ptr *d, void *t)
425{
426 int entry_cnt = NR_INLINE_DENTRY(inode);
427 int bitmap_size = INLINE_DENTRY_BITMAP_SIZE(inode);
428 int reserved_size = INLINE_RESERVED_SIZE(inode);
429
430 d->inode = inode;
431 d->max = entry_cnt;
432 d->nr_bitmap = bitmap_size;
433 d->bitmap = t;
434 d->dentry = t + bitmap_size + reserved_size;
435 d->filename = t + bitmap_size + reserved_size +
436 SIZE_OF_DIR_ENTRY * entry_cnt;
437}

1488static inline void f2fs_i_blocks_write(struct inode *, block_t, bool, bool);
1489static inline int inc_valid_block_count(struct f2fs_sb_info *sbi,
1490 struct inode *inode, blkcnt_t *count)
1491{
1492 blkcnt_t diff = 0, release = 0;
1493 block_t avail_user_block_count;
1494 int ret;
1495
1496 ret = dquot_reserve_block(inode, *count);
1497 if (ret)
1498 return ret;
1499
1500#ifdef CONFIG_F2FS_FAULT_INJECTION
1501 if (time_to_inject(sbi, FAULT_BLOCK)) {
1502 f2fs_show_injection_info(FAULT_BLOCK);
1503 release = *count;
1504 goto enospc;
1505 }
1506#endif
1507 /*
1508 * let's increase this in prior to actual block count change in order
1509 * for f2fs_sync_file to avoid data races when deciding checkpoint.
1510 */
1511 percpu_counter_add(&sbi->alloc_valid_block_count, (*count));
1512
1513 spin_lock(&sbi->stat_lock);
1514 sbi->total_valid_block_count += (block_t)(*count);
1515 avail_user_block_count = sbi->user_block_count - sbi->reserved_blocks;
1516 if (unlikely(sbi->total_valid_block_count > avail_user_block_count)) {
1517 diff = sbi->total_valid_block_count - avail_user_block_count;
1518 *count -= diff;
1519 release = diff;
1520 sbi->total_valid_block_count = avail_user_block_count;
1521 if (!*count) {
1522 spin_unlock(&sbi->stat_lock);
1523 percpu_counter_sub(&sbi->alloc_valid_block_count, diff);
1524 goto enospc;
1525 }
1526 }
1527 spin_unlock(&sbi->stat_lock);
1528
1529 if (release)
1530 dquot_release_reservation_block(inode, release);
1531 f2fs_i_blocks_write(inode, *count, true, true);
1532 return 0;
1533
1534enospc:
1535 dquot_release_reservation_block(inode, release);
1536 return -ENOSPC;
1537}

1679static inline int inc_valid_node_count(struct f2fs_sb_info *sbi,
1680 struct inode *inode, bool is_inode)
1681{
1682 block_t valid_block_count;
1683 unsigned int valid_node_count;
1684 bool quota = inode && !is_inode;
1685
1686 if (quota) {
1687 int ret = dquot_reserve_block(inode, 1);
1688 if (ret)
1689 return ret;
1690 }
1691
1692 spin_lock(&sbi->stat_lock);
1693
1694 valid_block_count = sbi->total_valid_block_count + 1;
1695 if (unlikely(valid_block_count + sbi->reserved_blocks >
1696 sbi->user_block_count)) {
1697 spin_unlock(&sbi->stat_lock);
1698 goto enospc;
1699 }
1700
1701 valid_node_count = sbi->total_valid_node_count + 1;
1702 if (unlikely(valid_node_count > sbi->total_node_count)) {
1703 spin_unlock(&sbi->stat_lock);
1704 goto enospc;
1705 }
1706
1707 sbi->total_valid_node_count++;
1708 sbi->total_valid_block_count++;
1709 spin_unlock(&sbi->stat_lock);
1710
1711 if (inode) {
1712 if (is_inode)
1713 f2fs_mark_inode_dirty_sync(inode, true);
1714 else
1715 f2fs_i_blocks_write(inode, 1, true, true);
1716 }
1717
1718 percpu_counter_inc(&sbi->alloc_valid_block_count);
1719 return 0;
1720
1721enospc:
1722 if (quota)
1723 dquot_release_reservation_block(inode, 1);
1724 return -ENOSPC;
1725}
