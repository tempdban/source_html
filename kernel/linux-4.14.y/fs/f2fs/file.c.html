36static int f2fs_filemap_fault(struct vm_fault *vmf)
37{
38 struct inode *inode = file_inode(vmf->vma->vm_file);
39 int err;
40
41 down_read(&F2FS_I(inode)->i_mmap_sem);
42 err = filemap_fault(vmf);
43 up_read(&F2FS_I(inode)->i_mmap_sem);
44
45 return err;
46}

48static int f2fs_vm_page_mkwrite(struct vm_fault *vmf)
49{
50 struct page *page = vmf->page;
51 struct inode *inode = file_inode(vmf->vma->vm_file);
52 struct f2fs_sb_info *sbi = F2FS_I_SB(inode);
53 struct dnode_of_data dn;
54 int err;
55
56 sb_start_pagefault(inode->i_sb);
57
58 f2fs_bug_on(sbi, f2fs_has_inline_data(inode));
59
60 /* block allocation */
61 f2fs_lock_op(sbi);
62 set_new_dnode(&dn, inode, NULL, NULL, 0);
63 err = f2fs_reserve_block(&dn, page->index);
64 if (err) {
65 f2fs_unlock_op(sbi);
66 goto out;
67 }
68 f2fs_put_dnode(&dn);
69 f2fs_unlock_op(sbi);
70
71 f2fs_balance_fs(sbi, dn.node_changed);
72
73 file_update_time(vmf->vma->vm_file);
74 down_read(&F2FS_I(inode)->i_mmap_sem);
75 lock_page(page);
76 if (unlikely(page->mapping != inode->i_mapping ||
77 page_offset(page) > i_size_read(inode) ||
78 !PageUptodate(page))) {
79 unlock_page(page);
80 err = -EFAULT;
81 goto out_sem;
82 }
83
84 /*
85 * check to see if the page is mapped already (no holes)
86 */
87 if (PageMappedToDisk(page))
88 goto mapped;
89
90 /* page is wholly or partially inside EOF */
91 if (((loff_t)(page->index + 1) << PAGE_SHIFT) >
92 i_size_read(inode)) {
93 unsigned offset;
94 offset = i_size_read(inode) & ~PAGE_MASK;
95 zero_user_segment(page, offset, PAGE_SIZE);
96 }
97 set_page_dirty(page);
98 if (!PageUptodate(page))
99 SetPageUptodate(page);
100
101 f2fs_update_iostat(sbi, APP_MAPPED_IO, F2FS_BLKSIZE);
102
103 trace_f2fs_vm_page_mkwrite(page, DATA);
104mapped:
105 /* fill the page */
106 f2fs_wait_on_page_writeback(page, DATA, false);
107
108 /* wait for GCed encrypted page writeback */
109 if (f2fs_encrypted_file(inode))
110 f2fs_wait_on_block_writeback(sbi, dn.data_blkaddr);
111
112out_sem:
113 up_read(&F2FS_I(inode)->i_mmap_sem);
114out:
115 sb_end_pagefault(inode->i_sb);
116 f2fs_update_time(sbi, REQ_TIME);
117 return block_page_mkwrite_return(err);
118}

126static int get_parent_ino(struct inode *inode, nid_t *pino)
127{
128 struct dentry *dentry;
129
130 inode = igrab(inode);
131 dentry = d_find_any_alias(inode);
132 iput(inode);
133 if (!dentry)
134 return 0;
135
136 *pino = parent_ino(dentry);
137 dput(dentry);

141static inline bool need_do_checkpoint(struct inode *inode)
142{
143 struct f2fs_sb_info *sbi = F2FS_I_SB(inode);
144 bool need_cp = false;
145
146 if (!S_ISREG(inode->i_mode) || inode->i_nlink != 1)
147 need_cp = true;
148 else if (is_sbi_flag_set(sbi, SBI_NEED_CP))
149 need_cp = true;
150 else if (file_wrong_pino(inode))
151 need_cp = true;
152 else if (!space_for_roll_forward(sbi))
153 need_cp = true;
154 else if (!is_checkpointed_node(sbi, F2FS_I(inode)->i_pino))
155 need_cp = true;
156 else if (test_opt(sbi, FASTBOOT))
157 need_cp = true;
158 else if (sbi->active_logs == 2)
159 need_cp = true;
160
161 return need_cp;
162}

164static bool need_inode_page_update(struct f2fs_sb_info *sbi, nid_t ino)
165{
166 struct page *i = find_get_page(NODE_MAPPING(sbi), ino);
167 bool ret = false;
168 /* But we need to avoid that there are some inode updates */
169 if ((i && PageDirty(i)) || need_inode_block_update(sbi, ino))
170 ret = true;
171 f2fs_put_page(i, 0);
172 return ret;
173}

175static void try_to_fix_pino(struct inode *inode)
176{
177 struct f2fs_inode_info *fi = F2FS_I(inode);
178 nid_t pino;
179
180 down_write(&fi->i_sem);
181 if (file_wrong_pino(inode) && inode->i_nlink == 1 &&
182 get_parent_ino(inode, &pino)) {
183 f2fs_i_pino_write(inode, pino);
184 file_got_pino(inode);
185 }
186 up_write(&fi->i_sem);
187}

189static int f2fs_do_sync_file(struct file *file, loff_t start, loff_t end,
190 int datasync, bool atomic)
191{
192 struct inode *inode = file->f_mapping->host;
193 struct f2fs_sb_info *sbi = F2FS_I_SB(inode);
194 nid_t ino = inode->i_ino;
195 int ret = 0;
196 bool need_cp = false;
197 struct writeback_control wbc = {
198 .sync_mode = WB_SYNC_ALL,
199 .nr_to_write = LONG_MAX,
200 .for_reclaim = 0,
201 };
202
203 if (unlikely(f2fs_readonly(inode->i_sb)))
204 return 0;
205
206 trace_f2fs_sync_file_enter(inode);
207
208 /* if fdatasync is triggered, let's do in-place-update */
209 if (datasync || get_dirty_pages(inode) <= SM_I(sbi)->min_fsync_blocks)
210 set_inode_flag(inode, FI_NEED_IPU);
211 ret = file_write_and_wait_range(file, start, end);
212 clear_inode_flag(inode, FI_NEED_IPU);
213
214 if (ret) {
215 trace_f2fs_sync_file_exit(inode, need_cp, datasync, ret);
216 return ret;
217 }
218
219 /* if the inode is dirty, let's recover all the time */
220 if (!f2fs_skip_inode_update(inode, datasync)) {
221 f2fs_write_inode(inode, NULL);
222 goto go_write;
223 }
224
225 /*
226 * if there is no written data, don't waste time to write recovery info.
227 */
228 if (!is_inode_flag_set(inode, FI_APPEND_WRITE) &&
229 !exist_written_data(sbi, ino, APPEND_INO)) {
230
231 /* it may call write_inode just prior to fsync */
232 if (need_inode_page_update(sbi, ino))
233 goto go_write;
234
235 if (is_inode_flag_set(inode, FI_UPDATE_WRITE) ||
236 exist_written_data(sbi, ino, UPDATE_INO))
237 goto flush_out;
238 goto out;
239 }
240go_write:
241 /*
242 * Both of fdatasync() and fsync() are able to be recovered from
243 * sudden-power-off.
244 */
245 down_read(&F2FS_I(inode)->i_sem);
246 need_cp = need_do_checkpoint(inode);
247 up_read(&F2FS_I(inode)->i_sem);
248
249 if (need_cp) {
250 /* all the dirty node pages should be flushed for POR */
251 ret = f2fs_sync_fs(inode->i_sb, 1);
252
253 /*
254 * We've secured consistency through sync_fs. Following pino
255 * will be used only for fsynced inodes after checkpoint.
256 */
257 try_to_fix_pino(inode);
258 clear_inode_flag(inode, FI_APPEND_WRITE);
259 clear_inode_flag(inode, FI_UPDATE_WRITE);
260 goto out;
261 }
262sync_nodes:
263 ret = fsync_node_pages(sbi, inode, &wbc, atomic);
264 if (ret)
265 goto out;
266
267 /* if cp_error was enabled, we should avoid infinite loop */
268 if (unlikely(f2fs_cp_error(sbi))) {
269 ret = -EIO;
270 goto out;
271 }
272
273 if (need_inode_block_update(sbi, ino)) {
274 f2fs_mark_inode_dirty_sync(inode, true);
275 f2fs_write_inode(inode, NULL);
276 goto sync_nodes;
277 }
278
279 /*
280 * If it's atomic_write, it's just fine to keep write ordering. So
281 * here we don't need to wait for node write completion, since we use
282 * node chain which serializes node blocks. If one of node writes are
283 * reordered, we can see simply broken chain, resulting in stopping
284 * roll-forward recovery. It means we'll recover all or none node blocks
285 * given fsync mark.
286 */
287 if (!atomic) {
288 ret = wait_on_node_pages_writeback(sbi, ino);
289 if (ret)
290 goto out;
291 }
292
293 /* once recovery info is written, don't need to tack this */
294 remove_ino_entry(sbi, ino, APPEND_INO);
295 clear_inode_flag(inode, FI_APPEND_WRITE);
296flush_out:
297 remove_ino_entry(sbi, ino, UPDATE_INO);
298 clear_inode_flag(inode, FI_UPDATE_WRITE);
299 if (!atomic)
300 ret = f2fs_issue_flush(sbi);
301 f2fs_update_time(sbi, REQ_TIME);
302out:
303 trace_f2fs_sync_file_exit(inode, need_cp, datasync, ret);
304 f2fs_trace_ios(NULL, 1);
305 return ret;
306}

313static pgoff_t __get_first_dirty_index(struct address_space *mapping,
314 pgoff_t pgofs, int whence)
315{
316 struct pagevec pvec;
317 int nr_pages;
318
319 if (whence != SEEK_DATA)
320 return 0;
321
322 /* find first dirty page index */
323 pagevec_init(&pvec, 0);
324 nr_pages = pagevec_lookup_tag(&pvec, mapping, &pgofs,
325 PAGECACHE_TAG_DIRTY, 1);
326 pgofs = nr_pages ? pvec.pages[0]->index : ULONG_MAX;
327 pagevec_release(&pvec);
328 return pgofs;
329}

331static bool __found_offset(block_t blkaddr, pgoff_t dirty, pgoff_t pgofs,
332 int whence)
333{
334 switch (whence) {
335 case SEEK_DATA:
336 if ((blkaddr == NEW_ADDR && dirty == pgofs) ||
337 (blkaddr != NEW_ADDR && blkaddr != NULL_ADDR))
338 return true;
339 break;
340 case SEEK_HOLE:
341 if (blkaddr == NULL_ADDR)
342 return true;
343 break;
344 }
345 return false;
346}

348static loff_t f2fs_seek_block(struct file *file, loff_t offset, int whence)
349{
350 struct inode *inode = file->f_mapping->host;
351 loff_t maxbytes = inode->i_sb->s_maxbytes;
352 struct dnode_of_data dn;
353 pgoff_t pgofs, end_offset, dirty;
354 loff_t data_ofs = offset;
355 loff_t isize;
356 int err = 0;
357
358 inode_lock(inode);
359
360 isize = i_size_read(inode);
361 if (offset >= isize)
362 goto fail;
363
364 /* handle inline data case */
365 if (f2fs_has_inline_data(inode) || f2fs_has_inline_dentry(inode)) {
366 if (whence == SEEK_HOLE)
367 data_ofs = isize;
368 goto found;
369 }
370
371 pgofs = (pgoff_t)(offset >> PAGE_SHIFT);
372
373 dirty = __get_first_dirty_index(inode->i_mapping, pgofs, whence);
374
375 for (; data_ofs < isize; data_ofs = (loff_t)pgofs << PAGE_SHIFT) {
376 set_new_dnode(&dn, inode, NULL, NULL, 0);
377 err = get_dnode_of_data(&dn, pgofs, LOOKUP_NODE);
378 if (err && err != -ENOENT) {
379 goto fail;
380 } else if (err == -ENOENT) {
381 /* direct node does not exists */
382 if (whence == SEEK_DATA) {
383 pgofs = get_next_page_offset(&dn, pgofs);
384 continue;
385 } else {
386 goto found;
387 }
388 }
389
390 end_offset = ADDRS_PER_PAGE(dn.node_page, inode);
391
392 /* find data/hole in dnode block */
393 for (; dn.ofs_in_node < end_offset;
394 dn.ofs_in_node++, pgofs++,
395 data_ofs = (loff_t)pgofs << PAGE_SHIFT) {
396 block_t blkaddr;
397 blkaddr = datablock_addr(dn.inode,
398 dn.node_page, dn.ofs_in_node);
399
400 if (__found_offset(blkaddr, dirty, pgofs, whence)) {
401 f2fs_put_dnode(&dn);
402 goto found;
403 }
404 }
405 f2fs_put_dnode(&dn);
406 }
407
408 if (whence == SEEK_DATA)
409 goto fail;
410found:
411 if (whence == SEEK_HOLE && data_ofs > isize)
412 data_ofs = isize;
413 inode_unlock(inode);
414 return vfs_setpos(file, data_ofs, maxbytes);
415fail:
416 inode_unlock(inode);
417 return -ENXIO;
418}

420static loff_t f2fs_llseek(struct file *file, loff_t offset, int whence)
421{
422 struct inode *inode = file->f_mapping->host;
423 loff_t maxbytes = inode->i_sb->s_maxbytes;
424
425 switch (whence) {
426 case SEEK_SET:
427 case SEEK_CUR:
428 case SEEK_END:
429 return generic_file_llseek_size(file, offset, whence,
430 maxbytes, i_size_read(inode));
431 case SEEK_DATA:
432 case SEEK_HOLE:
433 if (offset < 0)
434 return -ENXIO;
435 return f2fs_seek_block(file, offset, whence);
436 }
437
438 return -EINVAL;
439}

477int truncate_data_blocks_range(struct dnode_of_data *dn, int count)
478{
479 struct f2fs_sb_info *sbi = F2FS_I_SB(dn->inode);
480 struct f2fs_node *raw_node;
481 int nr_free = 0, ofs = dn->ofs_in_node, len = count;
482 __le32 *addr;
483 int base = 0;
484
485 if (IS_INODE(dn->node_page) && f2fs_has_extra_attr(dn->inode))
486 base = get_extra_isize(dn->inode);
487
488 raw_node = F2FS_NODE(dn->node_page);
489 addr = blkaddr_in_node(raw_node) + base + ofs;
490
491 for (; count > 0; count--, addr++, dn->ofs_in_node++) {
492 block_t blkaddr = le32_to_cpu(*addr);
493 if (blkaddr == NULL_ADDR)
494 continue;
495
496 dn->data_blkaddr = NULL_ADDR;
497 set_data_blkaddr(dn);
498 invalidate_blocks(sbi, blkaddr);
499 if (dn->ofs_in_node == 0 && IS_INODE(dn->node_page))
500 clear_inode_flag(dn->inode, FI_FIRST_BLOCK_WRITTEN);
501 nr_free++;
502 }
503
504 if (nr_free) {
505 pgoff_t fofs;
506 /*
507 * once we invalidate valid blkaddr in range [ofs, ofs + count],
508 * we will invalidate all blkaddr in the whole range.
509 */
510 fofs = start_bidx_of_node(ofs_of_node(dn->node_page),
511 dn->inode) + ofs;
512 f2fs_update_extent_cache_range(dn, fofs, 0, len);
513 dec_valid_block_count(sbi, dn->inode, nr_free);
514 }
515 dn->ofs_in_node = ofs;
516
517 f2fs_update_time(sbi, REQ_TIME);
518 trace_f2fs_truncate_data_blocks_range(dn->inode, dn->nid,
519 dn->ofs_in_node, nr_free);
520 return nr_free;
521}

528static int truncate_partial_data_page(struct inode *inode, u64 from,
529 bool cache_only)
530{
531 unsigned offset = from & (PAGE_SIZE - 1);
532 pgoff_t index = from >> PAGE_SHIFT;
533 struct address_space *mapping = inode->i_mapping;
534 struct page *page;
535
536 if (!offset && !cache_only)
537 return 0;
538
539 if (cache_only) {
540 page = find_lock_page(mapping, index);
541 if (page && PageUptodate(page))
542 goto truncate_out;
543 f2fs_put_page(page, 1);
544 return 0;
545 }
546
547 page = get_lock_data_page(inode, index, true);
548 if (IS_ERR(page))
549 return PTR_ERR(page) == -ENOENT ? 0 : PTR_ERR(page);
550truncate_out:
551 f2fs_wait_on_page_writeback(page, DATA, true);
552 zero_user(page, offset, PAGE_SIZE - offset);
553
554 /* An encrypted inode should have a key and truncate the last page. */
555 f2fs_bug_on(F2FS_I_SB(inode), cache_only && f2fs_encrypted_inode(inode));
556 if (!cache_only)
557 set_page_dirty(page);
558 f2fs_put_page(page, 1);
559 return 0;
560}

562int truncate_blocks(struct inode *inode, u64 from, bool lock)
563{
564 struct f2fs_sb_info *sbi = F2FS_I_SB(inode);
565 unsigned int blocksize = inode->i_sb->s_blocksize;
566 struct dnode_of_data dn;
567 pgoff_t free_from;
568 int count = 0, err = 0;
569 struct page *ipage;
570 bool truncate_page = false;
571
572 trace_f2fs_truncate_blocks_enter(inode, from);
573
574 free_from = (pgoff_t)F2FS_BYTES_TO_BLK(from + blocksize - 1);
575
576 if (free_from >= sbi->max_file_blocks)
577 goto free_partial;
578
579 if (lock)
580 f2fs_lock_op(sbi);
581
582 ipage = get_node_page(sbi, inode->i_ino);
583 if (IS_ERR(ipage)) {
584 err = PTR_ERR(ipage);
585 goto out;
586 }
587
588 if (f2fs_has_inline_data(inode)) {
589 truncate_inline_inode(inode, ipage, from);
590 f2fs_put_page(ipage, 1);
591 truncate_page = true;
592 goto out;
593 }
594
595 set_new_dnode(&dn, inode, ipage, NULL, 0);
596 err = get_dnode_of_data(&dn, free_from, LOOKUP_NODE_RA);
597 if (err) {
598 if (err == -ENOENT)
599 goto free_next;
600 goto out;
601 }
602
603 count = ADDRS_PER_PAGE(dn.node_page, inode);
604
605 count -= dn.ofs_in_node;
606 f2fs_bug_on(sbi, count < 0);
607
608 if (dn.ofs_in_node || IS_INODE(dn.node_page)) {
609 truncate_data_blocks_range(&dn, count);
610 free_from += count;
611 }
612
613 f2fs_put_dnode(&dn);
614free_next:
615 err = truncate_inode_blocks(inode, free_from);
616out:
617 if (lock)
618 f2fs_unlock_op(sbi);
619free_partial:
620 /* lastly zero out the first data page */
621 if (!err)
622 err = truncate_partial_data_page(inode, from, truncate_page);
623
624 trace_f2fs_truncate_blocks_exit(inode, err);
625 return err;
626}

628int f2fs_truncate(struct inode *inode)
629{
630 int err;
631
632 if (!(S_ISREG(inode->i_mode) || S_ISDIR(inode->i_mode) ||
633 S_ISLNK(inode->i_mode)))
634 return 0;
635
636 trace_f2fs_truncate(inode);
637
638#ifdef CONFIG_F2FS_FAULT_INJECTION
639 if (time_to_inject(F2FS_I_SB(inode), FAULT_TRUNCATE)) {
640 f2fs_show_injection_info(FAULT_TRUNCATE);
641 return -EIO;
642 }
643#endif
644 /* we should check inline_data size */
645 if (!f2fs_may_inline_data(inode)) {
646 err = f2fs_convert_inline_inode(inode);
647 if (err)
648 return err;
649 }
650
651 err = truncate_blocks(inode, i_size_read(inode), true);
652 if (err)
653 return err;
654
655 inode->i_mtime = inode->i_ctime = current_time(inode);
656 f2fs_mark_inode_dirty_sync(inode, false);
657 return 0;
658}

660int f2fs_getattr(const struct path *path, struct kstat *stat,
661 u32 request_mask, unsigned int query_flags)
662{
663 struct inode *inode = d_inode(path->dentry);
664 struct f2fs_inode_info *fi = F2FS_I(inode);
665 unsigned int flags;
666
667 flags = fi->i_flags & (FS_FL_USER_VISIBLE | FS_PROJINHERIT_FL);
668 if (flags & FS_APPEND_FL)
669 stat->attributes |= STATX_ATTR_APPEND;
670 if (flags & FS_COMPR_FL)
671 stat->attributes |= STATX_ATTR_COMPRESSED;
672 if (f2fs_encrypted_inode(inode))
673 stat->attributes |= STATX_ATTR_ENCRYPTED;
674 if (flags & FS_IMMUTABLE_FL)
675 stat->attributes |= STATX_ATTR_IMMUTABLE;
676 if (flags & FS_NODUMP_FL)
677 stat->attributes |= STATX_ATTR_NODUMP;
678
679 stat->attributes_mask |= (STATX_ATTR_APPEND |
680 STATX_ATTR_COMPRESSED |
681 STATX_ATTR_ENCRYPTED |
682 STATX_ATTR_IMMUTABLE |
683 STATX_ATTR_NODUMP);
684
685 generic_fillattr(inode, stat);
686
687 /* we need to show initial sectors used for inline_data/dentries */
688 if ((S_ISREG(inode->i_mode) && f2fs_has_inline_data(inode)) ||
689 f2fs_has_inline_dentry(inode))
690 stat->blocks += (stat->size + 511) >> 9;
691
692 return 0;
693}

696static void __setattr_copy(struct inode *inode, const struct iattr *attr)
697{
698 unsigned int ia_valid = attr->ia_valid;
699

702 if (ia_valid & ATTR_GID)
703 inode->i_gid = attr->ia_gid;
704 if (ia_valid & ATTR_ATIME)

707 if (ia_valid & ATTR_MTIME)

725int f2fs_setattr(struct dentry *dentry, struct iattr *attr)
726{
727 struct inode *inode = d_inode(dentry);
728 int err;
729 bool size_changed = false;
730
731 err = setattr_prepare(dentry, attr);
732 if (err)
733 return err;
734
735 if (is_quota_modification(inode, attr)) {
736 err = dquot_initialize(inode);
737 if (err)
738 return err;
739 }
740 if ((attr->ia_valid & ATTR_UID &&
741 !uid_eq(attr->ia_uid, inode->i_uid)) ||
742 (attr->ia_valid & ATTR_GID &&
743 !gid_eq(attr->ia_gid, inode->i_gid))) {
744 err = dquot_transfer(inode, attr);
745 if (err)
746 return err;
747 }
748
749 if (attr->ia_valid & ATTR_SIZE) {
750 if (f2fs_encrypted_inode(inode)) {
751 err = fscrypt_get_encryption_info(inode);
752 if (err)
753 return err;
754 if (!fscrypt_has_encryption_key(inode))
755 return -ENOKEY;
756 }
757
758 if (attr->ia_size <= i_size_read(inode)) {
759 down_write(&F2FS_I(inode)->i_mmap_sem);
760 truncate_setsize(inode, attr->ia_size);
761 err = f2fs_truncate(inode);
762 up_write(&F2FS_I(inode)->i_mmap_sem);
763 if (err)
764 return err;
765 } else {
766 /*
767 * do not trim all blocks after i_size if target size is
768 * larger than i_size.
769 */
770 down_write(&F2FS_I(inode)->i_mmap_sem);
771 truncate_setsize(inode, attr->ia_size);
772 up_write(&F2FS_I(inode)->i_mmap_sem);
773
774 /* should convert inline inode here */
775 if (!f2fs_may_inline_data(inode)) {
776 err = f2fs_convert_inline_inode(inode);
777 if (err)
778 return err;
779 }
780 inode->i_mtime = inode->i_ctime = current_time(inode);
781 }
782
783 size_changed = true;
784 }
785
786 __setattr_copy(inode, attr);
787
788 if (attr->ia_valid & ATTR_MODE) {
789 err = posix_acl_chmod(inode, get_inode_mode(inode));
790 if (err || is_inode_flag_set(inode, FI_ACL_MODE)) {
791 inode->i_mode = F2FS_I(inode)->i_acl_mode;
792 clear_inode_flag(inode, FI_ACL_MODE);
793 }
794 }
795
796 /* file size may changed here */
797 f2fs_mark_inode_dirty_sync(inode, size_changed);
798
799 /* inode change will produce dirty node pages flushed by checkpoint */
800 f2fs_balance_fs(F2FS_I_SB(inode), true);
801
802 return err;
803}

816static int fill_zero(struct inode *inode, pgoff_t index,
817 loff_t start, loff_t len)
818{
819 struct f2fs_sb_info *sbi = F2FS_I_SB(inode);
820 struct page *page;
821
822 if (!len)
823 return 0;
824
825 f2fs_balance_fs(sbi, true);
826
827 f2fs_lock_op(sbi);
828 page = get_new_data_page(inode, NULL, index, false);
829 f2fs_unlock_op(sbi);
830
831 if (IS_ERR(page))
832 return PTR_ERR(page);
833
834 f2fs_wait_on_page_writeback(page, DATA, true);
835 zero_user(page, start, len);
836 set_page_dirty(page);
837 f2fs_put_page(page, 1);
838 return 0;
839}

841int truncate_hole(struct inode *inode, pgoff_t pg_start, pgoff_t pg_end)
842{
843 int err;
844
845 while (pg_start < pg_end) {
846 struct dnode_of_data dn;
847 pgoff_t end_offset, count;
848
849 set_new_dnode(&dn, inode, NULL, NULL, 0);
850 err = get_dnode_of_data(&dn, pg_start, LOOKUP_NODE);
851 if (err) {
852 if (err == -ENOENT) {
853 pg_start++;
854 continue;
855 }
856 return err;
857 }
858
859 end_offset = ADDRS_PER_PAGE(dn.node_page, inode);
860 count = min(end_offset - dn.ofs_in_node, pg_end - pg_start);
861
862 f2fs_bug_on(F2FS_I_SB(inode), count == 0 || count > end_offset);
863
864 truncate_data_blocks_range(&dn, count);
865 f2fs_put_dnode(&dn);
866
867 pg_start += count;
868 }
869 return 0;
870}

872static int punch_hole(struct inode *inode, loff_t offset, loff_t len)
873{
874 pgoff_t pg_start, pg_end;
875 loff_t off_start, off_end;
876 int ret;
877
878 ret = f2fs_convert_inline_inode(inode);
879 if (ret)
880 return ret;
881
882 pg_start = ((unsigned long long) offset) >> PAGE_SHIFT;
883 pg_end = ((unsigned long long) offset + len) >> PAGE_SHIFT;
884
885 off_start = offset & (PAGE_SIZE - 1);
886 off_end = (offset + len) & (PAGE_SIZE - 1);
887
888 if (pg_start == pg_end) {
889 ret = fill_zero(inode, pg_start, off_start,
890 off_end - off_start);
891 if (ret)
892 return ret;
893 } else {
894 if (off_start) {
895 ret = fill_zero(inode, pg_start++, off_start,
896 PAGE_SIZE - off_start);
897 if (ret)
898 return ret;
899 }
900 if (off_end) {
901 ret = fill_zero(inode, pg_end, 0, off_end);
902 if (ret)
903 return ret;
904 }
905
906 if (pg_start < pg_end) {
907 struct address_space *mapping = inode->i_mapping;
908 loff_t blk_start, blk_end;
909 struct f2fs_sb_info *sbi = F2FS_I_SB(inode);
910
911 f2fs_balance_fs(sbi, true);
912
913 blk_start = (loff_t)pg_start << PAGE_SHIFT;
914 blk_end = (loff_t)pg_end << PAGE_SHIFT;
915 down_write(&F2FS_I(inode)->i_mmap_sem);
916 truncate_inode_pages_range(mapping, blk_start,
917 blk_end - 1);
918
919 f2fs_lock_op(sbi);
920 ret = truncate_hole(inode, pg_start, pg_end);
921 f2fs_unlock_op(sbi);
922 up_write(&F2FS_I(inode)->i_mmap_sem);
923 }
924 }
925
926 return ret;
927}

929static int __read_out_blkaddrs(struct inode *inode, block_t *blkaddr,
930 int *do_replace, pgoff_t off, pgoff_t len)
931{
932 struct f2fs_sb_info *sbi = F2FS_I_SB(inode);
933 struct dnode_of_data dn;
934 int ret, done, i;
935
936next_dnode:
937 set_new_dnode(&dn, inode, NULL, NULL, 0);
938 ret = get_dnode_of_data(&dn, off, LOOKUP_NODE_RA);
939 if (ret && ret != -ENOENT) {
940 return ret;
941 } else if (ret == -ENOENT) {
942 if (dn.max_level == 0)
943 return -ENOENT;
944 done = min((pgoff_t)ADDRS_PER_BLOCK - dn.ofs_in_node, len);
945 blkaddr += done;
946 do_replace += done;
947 goto next;
948 }
949
950 done = min((pgoff_t)ADDRS_PER_PAGE(dn.node_page, inode) -
951 dn.ofs_in_node, len);
952 for (i = 0; i < done; i++, blkaddr++, do_replace++, dn.ofs_in_node++) {
953 *blkaddr = datablock_addr(dn.inode,
954 dn.node_page, dn.ofs_in_node);
955 if (!is_checkpointed_data(sbi, *blkaddr)) {
956
957 if (test_opt(sbi, LFS)) {
958 f2fs_put_dnode(&dn);
959 return -ENOTSUPP;
960 }
961
962 /* do not invalidate this block address */
963 f2fs_update_data_blkaddr(&dn, NULL_ADDR);
964 *do_replace = 1;
965 }
966 }
967 f2fs_put_dnode(&dn);
968next:
969 len -= done;
970 off += done;
971 if (len)
972 goto next_dnode;
973 return 0;
974}

976static int __roll_back_blkaddrs(struct inode *inode, block_t *blkaddr,
977 int *do_replace, pgoff_t off, int len)
978{
979 struct f2fs_sb_info *sbi = F2FS_I_SB(inode);
980 struct dnode_of_data dn;
981 int ret, i;
982
983 for (i = 0; i < len; i++, do_replace++, blkaddr++) {
984 if (*do_replace == 0)
985 continue;
986
987 set_new_dnode(&dn, inode, NULL, NULL, 0);
988 ret = get_dnode_of_data(&dn, off + i, LOOKUP_NODE_RA);
989 if (ret) {
990 dec_valid_block_count(sbi, inode, 1);
991 invalidate_blocks(sbi, *blkaddr);
992 } else {
993 f2fs_update_data_blkaddr(&dn, *blkaddr);
994 }
995 f2fs_put_dnode(&dn);
996 }
997 return 0;
998}

1000static int __clone_blkaddrs(struct inode *src_inode, struct inode *dst_inode,
1001 block_t *blkaddr, int *do_replace,
1002 pgoff_t src, pgoff_t dst, pgoff_t len, bool full)
1003{
1004 struct f2fs_sb_info *sbi = F2FS_I_SB(src_inode);
1005 pgoff_t i = 0;
1006 int ret;
1007
1008 while (i < len) {
1009 if (blkaddr[i] == NULL_ADDR && !full) {
1010 i++;
1011 continue;
1012 }
1013
1014 if (do_replace[i] || blkaddr[i] == NULL_ADDR) {
1015 struct dnode_of_data dn;
1016 struct node_info ni;
1017 size_t new_size;
1018 pgoff_t ilen;
1019
1020 set_new_dnode(&dn, dst_inode, NULL, NULL, 0);
1021 ret = get_dnode_of_data(&dn, dst + i, ALLOC_NODE);
1022 if (ret)
1023 return ret;
1024
1025 get_node_info(sbi, dn.nid, &ni);
1026 ilen = min((pgoff_t)
1027 ADDRS_PER_PAGE(dn.node_page, dst_inode) -
1028 dn.ofs_in_node, len - i);
1029 do {
1030 dn.data_blkaddr = datablock_addr(dn.inode,
1031 dn.node_page, dn.ofs_in_node);
1032 truncate_data_blocks_range(&dn, 1);
1033
1034 if (do_replace[i]) {
1035 f2fs_i_blocks_write(src_inode,
1036 1, false, false);
1037 f2fs_i_blocks_write(dst_inode,
1038 1, true, false);
1039 f2fs_replace_block(sbi, &dn, dn.data_blkaddr,
1040 blkaddr[i], ni.version, true, false);
1041
1042 do_replace[i] = 0;
1043 }
1044 dn.ofs_in_node++;
1045 i++;
1046 new_size = (dst + i) << PAGE_SHIFT;
1047 if (dst_inode->i_size < new_size)
1048 f2fs_i_size_write(dst_inode, new_size);
1049 } while (--ilen && (do_replace[i] || blkaddr[i] == NULL_ADDR));
1050
1051 f2fs_put_dnode(&dn);
1052 } else {
1053 struct page *psrc, *pdst;
1054
1055 psrc = get_lock_data_page(src_inode, src + i, true);
1056 if (IS_ERR(psrc))
1057 return PTR_ERR(psrc);
1058 pdst = get_new_data_page(dst_inode, NULL, dst + i,
1059 true);
1060 if (IS_ERR(pdst)) {
1061 f2fs_put_page(psrc, 1);
1062 return PTR_ERR(pdst);
1063 }
1064 f2fs_copy_page(psrc, pdst);
1065 set_page_dirty(pdst);
1066 f2fs_put_page(pdst, 1);
1067 f2fs_put_page(psrc, 1);
1068
1069 ret = truncate_hole(src_inode, src + i, src + i + 1);
1070 if (ret)
1071 return ret;
1072 i++;
1073 }
1074 }
1075 return 0;
1076}

1078static int __exchange_data_block(struct inode *src_inode,
1079 struct inode *dst_inode, pgoff_t src, pgoff_t dst,
1080 pgoff_t len, bool full)
1081{
1082 block_t *src_blkaddr;
1083 int *do_replace;
1084 pgoff_t olen;
1085 int ret;
1086
1087 while (len) {
1088 olen = min((pgoff_t)4 * ADDRS_PER_BLOCK, len);
1089
1090 src_blkaddr = kvzalloc(sizeof(block_t) * olen, GFP_KERNEL);
1091 if (!src_blkaddr)
1092 return -ENOMEM;
1093
1094 do_replace = kvzalloc(sizeof(int) * olen, GFP_KERNEL);
1095 if (!do_replace) {
1096 kvfree(src_blkaddr);
1097 return -ENOMEM;
1098 }
1099
1100 ret = __read_out_blkaddrs(src_inode, src_blkaddr,
1101 do_replace, src, olen);
1102 if (ret)
1103 goto roll_back;
1104
1105 ret = __clone_blkaddrs(src_inode, dst_inode, src_blkaddr,
1106 do_replace, src, dst, olen, full);
1107 if (ret)
1108 goto roll_back;
1109
1110 src += olen;
1111 dst += olen;
1112 len -= olen;
1113
1114 kvfree(src_blkaddr);
1115 kvfree(do_replace);
1116 }
1117 return 0;
1118
1119roll_back:
1120 __roll_back_blkaddrs(src_inode, src_blkaddr, do_replace, src, len);
1121 kvfree(src_blkaddr);
1122 kvfree(do_replace);
1123 return ret;
1124}

1142static int f2fs_collapse_range(struct inode *inode, loff_t offset, loff_t len)
1143{
1144 pgoff_t pg_start, pg_end;
1145 loff_t new_size;
1146 int ret;
1147
1148 if (offset + len >= i_size_read(inode))
1149 return -EINVAL;
1150
1151 /* collapse range should be aligned to block size of f2fs. */
1152 if (offset & (F2FS_BLKSIZE - 1) || len & (F2FS_BLKSIZE - 1))
1153 return -EINVAL;
1154
1155 ret = f2fs_convert_inline_inode(inode);
1156 if (ret)
1157 return ret;
1158
1159 pg_start = offset >> PAGE_SHIFT;
1160 pg_end = (offset + len) >> PAGE_SHIFT;
1161
1162 down_write(&F2FS_I(inode)->i_mmap_sem);
1163 /* write out all dirty pages from offset */
1164 ret = filemap_write_and_wait_range(inode->i_mapping, offset, LLONG_MAX);
1165 if (ret)
1166 goto out;
1167
1168 truncate_pagecache(inode, offset);
1169
1170 ret = f2fs_do_collapse(inode, pg_start, pg_end);
1171 if (ret)
1172 goto out;
1173
1174 /* write out all moved pages, if possible */
1175 filemap_write_and_wait_range(inode->i_mapping, offset, LLONG_MAX);
1176 truncate_pagecache(inode, offset);
1177
1178 new_size = i_size_read(inode) - len;
1179 truncate_pagecache(inode, new_size);
1180
1181 ret = truncate_blocks(inode, new_size, true);
1182 if (!ret)
1183 f2fs_i_size_write(inode, new_size);
1184
1185out:
1186 up_write(&F2FS_I(inode)->i_mmap_sem);
1187 return ret;
1188}

1190static int f2fs_do_zero_range(struct dnode_of_data *dn, pgoff_t start,
1191 pgoff_t end)
1192{
1193 struct f2fs_sb_info *sbi = F2FS_I_SB(dn->inode);
1194 pgoff_t index = start;
1195 unsigned int ofs_in_node = dn->ofs_in_node;
1196 blkcnt_t count = 0;
1197 int ret;
1198
1199 for (; index < end; index++, dn->ofs_in_node++) {
1200 if (datablock_addr(dn->inode, dn->node_page,
1201 dn->ofs_in_node) == NULL_ADDR)
1202 count++;
1203 }
1204
1205 dn->ofs_in_node = ofs_in_node;
1206 ret = reserve_new_blocks(dn, count);
1207 if (ret)
1208 return ret;
1209
1210 dn->ofs_in_node = ofs_in_node;
1211 for (index = start; index < end; index++, dn->ofs_in_node++) {
1212 dn->data_blkaddr = datablock_addr(dn->inode,
1213 dn->node_page, dn->ofs_in_node);
1214 /*
1215 * reserve_new_blocks will not guarantee entire block
1216 * allocation.
1217 */
1218 if (dn->data_blkaddr == NULL_ADDR) {
1219 ret = -ENOSPC;
1220 break;
1221 }
1222 if (dn->data_blkaddr != NEW_ADDR) {
1223 invalidate_blocks(sbi, dn->data_blkaddr);
1224 dn->data_blkaddr = NEW_ADDR;
1225 set_data_blkaddr(dn);
1226 }
1227 }
1228
1229 f2fs_update_extent_cache_range(dn, start, 0, index - start);
1230
1231 return ret;
1232}

1234static int f2fs_zero_range(struct inode *inode, loff_t offset, loff_t len,
1235 int mode)
1236{
1237 struct f2fs_sb_info *sbi = F2FS_I_SB(inode);
1238 struct address_space *mapping = inode->i_mapping;
1239 pgoff_t index, pg_start, pg_end;
1240 loff_t new_size = i_size_read(inode);
1241 loff_t off_start, off_end;
1242 int ret = 0;
1243
1244 ret = inode_newsize_ok(inode, (len + offset));
1245 if (ret)
1246 return ret;
1247
1248 ret = f2fs_convert_inline_inode(inode);
1249 if (ret)
1250 return ret;
1251
1252 down_write(&F2FS_I(inode)->i_mmap_sem);
1253 ret = filemap_write_and_wait_range(mapping, offset, offset + len - 1);
1254 if (ret)
1255 goto out_sem;
1256
1257 truncate_pagecache_range(inode, offset, offset + len - 1);
1258
1259 pg_start = ((unsigned long long) offset) >> PAGE_SHIFT;
1260 pg_end = ((unsigned long long) offset + len) >> PAGE_SHIFT;
1261
1262 off_start = offset & (PAGE_SIZE - 1);
1263 off_end = (offset + len) & (PAGE_SIZE - 1);
1264
1265 if (pg_start == pg_end) {
1266 ret = fill_zero(inode, pg_start, off_start,
1267 off_end - off_start);
1268 if (ret)
1269 goto out_sem;
1270
1271 new_size = max_t(loff_t, new_size, offset + len);
1272 } else {
1273 if (off_start) {
1274 ret = fill_zero(inode, pg_start++, off_start,
1275 PAGE_SIZE - off_start);
1276 if (ret)
1277 goto out_sem;
1278
1279 new_size = max_t(loff_t, new_size,
1280 (loff_t)pg_start << PAGE_SHIFT);
1281 }
1282
1283 for (index = pg_start; index < pg_end;) {
1284 struct dnode_of_data dn;
1285 unsigned int end_offset;
1286 pgoff_t end;
1287
1288 f2fs_lock_op(sbi);
1289
1290 set_new_dnode(&dn, inode, NULL, NULL, 0);
1291 ret = get_dnode_of_data(&dn, index, ALLOC_NODE);
1292 if (ret) {
1293 f2fs_unlock_op(sbi);
1294 goto out;
1295 }
1296
1297 end_offset = ADDRS_PER_PAGE(dn.node_page, inode);
1298 end = min(pg_end, end_offset - dn.ofs_in_node + index);
1299
1300 ret = f2fs_do_zero_range(&dn, index, end);
1301 f2fs_put_dnode(&dn);
1302 f2fs_unlock_op(sbi);
1303
1304 f2fs_balance_fs(sbi, dn.node_changed);
1305
1306 if (ret)
1307 goto out;
1308
1309 index = end;
1310 new_size = max_t(loff_t, new_size,
1311 (loff_t)index << PAGE_SHIFT);
1312 }
1313
1314 if (off_end) {
1315 ret = fill_zero(inode, pg_end, 0, off_end);
1316 if (ret)
1317 goto out;
1318
1319 new_size = max_t(loff_t, new_size, offset + len);
1320 }
1321 }
1322
1323out:
1324 if (new_size > i_size_read(inode)) {
1325 if (mode & FALLOC_FL_KEEP_SIZE)
1326 file_set_keep_isize(inode);
1327 else
1328 f2fs_i_size_write(inode, new_size);
1329 }
1330out_sem:
1331 up_write(&F2FS_I(inode)->i_mmap_sem);
1332
1333 return ret;
1334}

1336static int f2fs_insert_range(struct inode *inode, loff_t offset, loff_t len)
1337{
1338 struct f2fs_sb_info *sbi = F2FS_I_SB(inode);
1339 pgoff_t nr, pg_start, pg_end, delta, idx;
1340 loff_t new_size;
1341 int ret = 0;
1342
1343 new_size = i_size_read(inode) + len;
1344 ret = inode_newsize_ok(inode, new_size);
1345 if (ret)
1346 return ret;
1347
1348 if (offset >= i_size_read(inode))
1349 return -EINVAL;
1350
1351 /* insert range should be aligned to block size of f2fs. */
1352 if (offset & (F2FS_BLKSIZE - 1) || len & (F2FS_BLKSIZE - 1))
1353 return -EINVAL;
1354
1355 ret = f2fs_convert_inline_inode(inode);
1356 if (ret)
1357 return ret;
1358
1359 f2fs_balance_fs(sbi, true);
1360
1361 down_write(&F2FS_I(inode)->i_mmap_sem);
1362 ret = truncate_blocks(inode, i_size_read(inode), true);
1363 if (ret)
1364 goto out;
1365
1366 /* write out all dirty pages from offset */
1367 ret = filemap_write_and_wait_range(inode->i_mapping, offset, LLONG_MAX);
1368 if (ret)
1369 goto out;
1370
1371 truncate_pagecache(inode, offset);
1372
1373 pg_start = offset >> PAGE_SHIFT;
1374 pg_end = (offset + len) >> PAGE_SHIFT;
1375 delta = pg_end - pg_start;
1376 idx = (i_size_read(inode) + PAGE_SIZE - 1) / PAGE_SIZE;
1377
1378 while (!ret && idx > pg_start) {
1379 nr = idx - pg_start;
1380 if (nr > delta)
1381 nr = delta;
1382 idx -= nr;
1383
1384 f2fs_lock_op(sbi);
1385 f2fs_drop_extent_tree(inode);
1386
1387 ret = __exchange_data_block(inode, inode, idx,
1388 idx + delta, nr, false);
1389 f2fs_unlock_op(sbi);
1390 }
1391
1392 /* write out all moved pages, if possible */
1393 filemap_write_and_wait_range(inode->i_mapping, offset, LLONG_MAX);
1394 truncate_pagecache(inode, offset);
1395
1396 if (!ret)
1397 f2fs_i_size_write(inode, new_size);
1398out:
1399 up_write(&F2FS_I(inode)->i_mmap_sem);
1400 return ret;
1401}

1403static int expand_inode_data(struct inode *inode, loff_t offset,
1404 loff_t len, int mode)
1405{
1406 struct f2fs_sb_info *sbi = F2FS_I_SB(inode);
1407 struct f2fs_map_blocks map = { .m_next_pgofs = NULL };
1408 pgoff_t pg_end;
1409 loff_t new_size = i_size_read(inode);
1410 loff_t off_end;
1411 int err;
1412
1413 err = inode_newsize_ok(inode, (len + offset));
1414 if (err)
1415 return err;
1416
1417 err = f2fs_convert_inline_inode(inode);
1418 if (err)
1419 return err;
1420
1421 f2fs_balance_fs(sbi, true);
1422
1423 pg_end = ((unsigned long long)offset + len) >> PAGE_SHIFT;
1424 off_end = (offset + len) & (PAGE_SIZE - 1);
1425
1426 map.m_lblk = ((unsigned long long)offset) >> PAGE_SHIFT;
1427 map.m_len = pg_end - map.m_lblk;
1428 if (off_end)
1429 map.m_len++;
1430
1431 err = f2fs_map_blocks(inode, &map, 1, F2FS_GET_BLOCK_PRE_AIO);
1432 if (err) {
1433 pgoff_t last_off;
1434
1435 if (!map.m_len)
1436 return err;
1437
1438 last_off = map.m_lblk + map.m_len - 1;
1439
1440 /* update new size to the failed position */
1441 new_size = (last_off == pg_end) ? offset + len:
1442 (loff_t)(last_off + 1) << PAGE_SHIFT;
1443 } else {
1444 new_size = ((loff_t)pg_end << PAGE_SHIFT) + off_end;
1445 }
1446
1447 if (!(mode & FALLOC_FL_KEEP_SIZE) && i_size_read(inode) < new_size)
1448 f2fs_i_size_write(inode, new_size);
1449
1450 return err;
1451}

1453static long f2fs_fallocate(struct file *file, int mode,
1454 loff_t offset, loff_t len)
1455{
1456 struct inode *inode = file_inode(file);
1457 long ret = 0;
1458
1459 /* f2fs only support ->fallocate for regular file */
1460 if (!S_ISREG(inode->i_mode))
1461 return -EINVAL;
1462
1463 if (f2fs_encrypted_inode(inode) &&
1464 (mode & (FALLOC_FL_COLLAPSE_RANGE | FALLOC_FL_INSERT_RANGE)))
1465 return -EOPNOTSUPP;
1466
1467 if (mode & ~(FALLOC_FL_KEEP_SIZE | FALLOC_FL_PUNCH_HOLE |
1468 FALLOC_FL_COLLAPSE_RANGE | FALLOC_FL_ZERO_RANGE |
1469 FALLOC_FL_INSERT_RANGE))
1470 return -EOPNOTSUPP;
1471
1472 inode_lock(inode);
1473
1474 if (mode & FALLOC_FL_PUNCH_HOLE) {
1475 if (offset >= inode->i_size)
1476 goto out;
1477
1478 ret = punch_hole(inode, offset, len);
1479 } else if (mode & FALLOC_FL_COLLAPSE_RANGE) {
1480 ret = f2fs_collapse_range(inode, offset, len);
1481 } else if (mode & FALLOC_FL_ZERO_RANGE) {
1482 ret = f2fs_zero_range(inode, offset, len, mode);
1483 } else if (mode & FALLOC_FL_INSERT_RANGE) {
1484 ret = f2fs_insert_range(inode, offset, len);
1485 } else {
1486 ret = expand_inode_data(inode, offset, len, mode);
1487 }
1488
1489 if (!ret) {
1490 inode->i_mtime = inode->i_ctime = current_time(inode);
1491 f2fs_mark_inode_dirty_sync(inode, false);
1492 if (mode & FALLOC_FL_KEEP_SIZE)
1493 file_set_keep_isize(inode);
1494 f2fs_update_time(F2FS_I_SB(inode), REQ_TIME);
1495 }
1496
1497out:
1498 inode_unlock(inode);
1499
1500 trace_f2fs_fallocate(inode, mode, offset, len, ret);
1501 return ret;
1502}

1504static int f2fs_release_file(struct inode *inode, struct file *filp)
1505{
1506 /*
1507 * f2fs_relase_file is called at every close calls. So we should
1508 * not drop any inmemory pages by close called by other process.
1509 */
1510 if (!(filp->f_mode & FMODE_WRITE) ||
1511 atomic_read(&inode->i_writecount) != 1)
1512 return 0;
1513
1514 /* some remained atomic pages should discarded */
1515 if (f2fs_is_atomic_file(inode))
1516 drop_inmem_pages(inode);
1517 if (f2fs_is_volatile_file(inode)) {
1518 clear_inode_flag(inode, FI_VOLATILE_FILE);
1519 stat_dec_volatile_write(inode);
1520 set_inode_flag(inode, FI_DROP_CACHE);
1521 filemap_fdatawrite(inode->i_mapping);
1522 clear_inode_flag(inode, FI_DROP_CACHE);
1523 }
1524 return 0;
1525}

1797static int f2fs_ioc_shutdown(struct file *filp, unsigned long arg)
1798{
1799 struct inode *inode = file_inode(filp);
1800 struct f2fs_sb_info *sbi = F2FS_I_SB(inode);
1801 struct super_block *sb = sbi->sb;
1802 __u32 in;
1803 int ret;
1804
1805 if (!capable(CAP_SYS_ADMIN))
1806 return -EPERM;
1807
1808 if (get_user(in, (__u32 __user *)arg))
1809 return -EFAULT;
1810
1811 ret = mnt_want_write_file(filp);
1812 if (ret)
1813 return ret;
1814
1815 switch (in) {
1816 case F2FS_GOING_DOWN_FULLSYNC:
1817 sb = freeze_bdev(sb->s_bdev);
1818 if (sb && !IS_ERR(sb)) {
1819 f2fs_stop_checkpoint(sbi, false);
1820 thaw_bdev(sb->s_bdev, sb);
1821 }
1822 break;
1823 case F2FS_GOING_DOWN_METASYNC:
1824 /* do checkpoint only */
1825 f2fs_sync_fs(sb, 1);
1826 f2fs_stop_checkpoint(sbi, false);
1827 break;
1828 case F2FS_GOING_DOWN_NOSYNC:
1829 f2fs_stop_checkpoint(sbi, false);
1830 break;
1831 case F2FS_GOING_DOWN_METAFLUSH:
1832 sync_meta_pages(sbi, META, LONG_MAX, FS_META_IO);
1833 f2fs_stop_checkpoint(sbi, false);
1834 break;
1835 default:
1836 ret = -EINVAL;
1837 goto out;
1838 }
1839 f2fs_update_time(sbi, REQ_TIME);
1840out:
1841 mnt_drop_write_file(filp);
1842 return ret;
1843}

2041 struct file *filp,
2042 struct f2fs_defragment *range)

2045 struct f2fs_map_blocks map = { .m_next_pgofs = NULL };
2046 struct extent_info ei = {0,0,0};
2047 pgoff_t pg_start, pg_end;
2048 unsigned int blk_per_seg = sbi->blocks_per_seg;
2049 unsigned int total = 0, sec_num;
2050 block_t blk_end = 0;
2051 bool fragmented = false;
2052 int err;

2055 if (need_inplace_update_policy(inode, NULL))
2056 return -EINVAL;

2059 pg_end = (range->start + range->len) >> PAGE_SHIFT;

2065 /* writeback all dirty pages in the range */
2066 err = filemap_write_and_wait_range(inode->i_mapping, range->start,
2067 range->start + range->len - 1);

2070
2071 /*
2072 * lookup mapping info in extent cache, skip defragmenting if physical
2073 * block addresses are continuous.
2074 */
2075 if (f2fs_lookup_extent_cache(inode, pg_start, &ei)) {
2076 if (ei.fofs + ei.len >= pg_end)
2077 goto out;
2078 }
2079
2080 map.m_lblk = pg_start;
2081
2082 /*
2083 * lookup mapping info in dnode page cache, skip defragmenting if all
2084 * physical block addresses are continuous even if there are hole(s)
2085 * in logical blocks.
2086 */
2087 while (map.m_lblk < pg_end) {
2088 map.m_len = pg_end - map.m_lblk;
2089 err = f2fs_map_blocks(inode, &map, 0, F2FS_GET_BLOCK_DEFAULT);
2090 if (err)
2091 goto out;
2092
2093 if (!(map.m_flags & F2FS_MAP_FLAGS)) {
2094 map.m_lblk++;
2095 continue;
2096 }
2097
2098 if (blk_end && blk_end != map.m_pblk) {
2099 fragmented = true;
2100 break;
2101 }
2102 blk_end = map.m_pblk + map.m_len;
2103
2104 map.m_lblk += map.m_len;
2105 }
2106
2107 if (!fragmented)
2108 goto out;
2109
2110 map.m_lblk = pg_start;
2111 map.m_len = pg_end - pg_start;
2112
2113 sec_num = (map.m_len + BLKS_PER_SEC(sbi) - 1) / BLKS_PER_SEC(sbi);
2114
2115 /*
2116 * make sure there are enough free section for LFS allocation, this can
2117 * avoid defragment running in SSR mode when free section are allocated
2118 * intensively
2119 */
2120 if (has_not_enough_free_secs(sbi, 0, sec_num)) {
2121 err = -EAGAIN;
2122 goto out;
2123 }
2124
2125 while (map.m_lblk < pg_end) {
2126 pgoff_t idx;
2127 int cnt = 0;
2128
2129do_map:
2130 map.m_len = pg_end - map.m_lblk;
2131 err = f2fs_map_blocks(inode, &map, 0, F2FS_GET_BLOCK_DEFAULT);
2132 if (err)
2133 goto clear_out;
2134
2135 if (!(map.m_flags & F2FS_MAP_FLAGS)) {
2136 map.m_lblk++;
2137 continue;
2138 }
2139
2140 set_inode_flag(inode, FI_DO_DEFRAG);
2141
2142 idx = map.m_lblk;
2143 while (idx < map.m_lblk + map.m_len && cnt < blk_per_seg) {
2144 struct page *page;
2145
2146 page = get_lock_data_page(inode, idx, true);
2147 if (IS_ERR(page)) {
2148 err = PTR_ERR(page);
2149 goto clear_out;
2150 }
2151
2152 set_page_dirty(page);
2153 f2fs_put_page(page, 1);
2154
2155 idx++;
2156 cnt++;
2157 total++;
2158 }
2159
2160 map.m_lblk = idx;
2161
2162 if (idx < pg_end && cnt < blk_per_seg)
2163 goto do_map;
2164
2165 clear_inode_flag(inode, FI_DO_DEFRAG);
2166
2167 err = filemap_fdatawrite(inode->i_mapping);
2168 if (err)
2169 goto out;
2170 }
2171clear_out:
2172 clear_inode_flag(inode, FI_DO_DEFRAG);
2173out:
2174 inode_unlock(inode);
2175 if (!err)
2176 range->len = (u64)total << PAGE_SHIFT;
2177 return err;
2178}

2316 }
2317 f2fs_unlock_op(sbi);
2318out_unlock:
2319 if (src != dst)
2320 inode_unlock(dst);
2321out:
2322 inode_unlock(src);
2323 return ret;
2324}

2694 inode_lock(inode);
2695 ret = generic_write_checks(iocb, from);
2696 if (ret > 0) {
2697 bool preallocated = false;
2698 size_t target_size = 0;
2699 int err;
2700
2701 if (iov_iter_fault_in_readable(from, iov_iter_count(from)))
2702 set_inode_flag(inode, FI_NO_PREALLOC);
2703
2704 preallocated = true;
2705 target_size = iocb->ki_pos + iov_iter_count(from);
2706
2707 err = f2fs_preallocate_blocks(iocb, from);
2708 if (err) {
2709 clear_inode_flag(inode, FI_NO_PREALLOC);
2710 inode_unlock(inode);
2711 return err;
2712 }
2713 blk_start_plug(&plug);
2714 ret = __generic_file_write_iter(iocb, from);
2715 blk_finish_plug(&plug);
2716 clear_inode_flag(inode, FI_NO_PREALLOC);
2717
2718 /* if we couldn't write data, we should deallocate blocks. */
2719 if (preallocated && i_size_read(inode) < target_size)
2720 f2fs_truncate(inode);
2721
2722 if (ret > 0)
2723 f2fs_update_iostat(F2FS_I_SB(inode), APP_WRITE_IO, ret);
2724 }
2725 inode_unlock(inode);
2726
2727 if (ret > 0)
2728 ret = generic_write_sync(iocb, ret);
2729 return ret;
2730}
