13#define LOAD_INT(x) ((x) >> FSHIFT)
14#define LOAD_FRAC(x) LOAD_INT(((x) & (FIXED_1-1)) * 100)
15
16static int loadavg_proc_show(struct seq_file *m, void *v)
17{
18 unsigned long avnrun[3];
19
20 get_avenrun(avnrun, FIXED_1/200, 0);
21
22 seq_printf(m, "%lu.%02lu %lu.%02lu %lu.%02lu %ld/%d %d\n",
23 LOAD_INT(avnrun[0]), LOAD_FRAC(avnrun[0]),
24 LOAD_INT(avnrun[1]), LOAD_FRAC(avnrun[1]),
25 LOAD_INT(avnrun[2]), LOAD_FRAC(avnrun[2]),
26 nr_running(), nr_threads,
27 task_active_pid_ns(current)->last_pid);
28 return 0;
29}
30
31static int loadavg_proc_open(struct inode *inode, struct file *file)
32{
33 return single_open(file, loadavg_proc_show, NULL);
34}
35
36static const struct file_operations loadavg_proc_fops = {
37 .open = loadavg_proc_open,
38 .read = seq_read,
39 .llseek = seq_lseek,
40 .release = single_release,
41};
42
43static int __init proc_loadavg_init(void)
44{
45 proc_create("loadavg", 0, NULL, &loadavg_proc_fops);
46 return 0;
47}
48fs_initcall(proc_loadavg_init);
