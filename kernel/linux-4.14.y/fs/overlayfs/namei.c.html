29static int ovl_check_redirect(struct dentry *dentry, struct ovl_lookup_data *d,
30 size_t prelen, const char *post)
31{
32 int res;
33 char *s, *next, *buf = NULL;
34
35 res = vfs_getxattr(dentry, OVL_XATTR_REDIRECT, NULL, 0);
36 if (res < 0) {
37 if (res == -ENODATA || res == -EOPNOTSUPP)
38 return 0;
39 goto fail;
40 }
41 buf = kzalloc(prelen + res + strlen(post) + 1, GFP_KERNEL);
42 if (!buf)
43 return -ENOMEM;
44
45 if (res == 0)
46 goto invalid;
47
48 res = vfs_getxattr(dentry, OVL_XATTR_REDIRECT, buf, res);
49 if (res < 0)
50 goto fail;
51 if (res == 0)
52 goto invalid;
53 if (buf[0] == '/') {
54 for (s = buf; *s++ == '/'; s = next) {
55 next = strchrnul(s, '/');
56 if (s == next)
57 goto invalid;
58 }
59 /*
60 * One of the ancestor path elements in an absolute path
61 * lookup in ovl_lookup_layer() could have been opaque and
62 * that will stop further lookup in lower layers (d->stop=true)
63 * But we have found an absolute redirect in decendant path
64 * element and that should force continue lookup in lower
65 * layers (reset d->stop).
66 */
67 d->stop = false;
68 } else {
69 if (strchr(buf, '/') != NULL)
70 goto invalid;
71
72 memmove(buf + prelen, buf, res);
73 memcpy(buf, d->name.name, prelen);
74 }
75
76 strcat(buf, post);
77 kfree(d->redirect);
78 d->redirect = buf;
79 d->name.name = d->redirect;
80 d->name.len = strlen(d->redirect);
81
82 return 0;
83
84err_free:
85 kfree(buf);
86 return 0;
87fail:
88 pr_warn_ratelimited("overlayfs: failed to get redirect (%i)\n", res);
89 goto err_free;
90invalid:
91 pr_warn_ratelimited("overlayfs: invalid redirect (%s)\n", buf);
92 goto err_free;
93}

100static struct ovl_fh *ovl_get_origin_fh(struct dentry *dentry)
101{
102 int res;
103 struct ovl_fh *fh = NULL;
104
105 res = vfs_getxattr(dentry, OVL_XATTR_ORIGIN, NULL, 0);
106 if (res < 0) {
107 if (res == -ENODATA || res == -EOPNOTSUPP)
108 return NULL;
109 goto fail;
110 }
111 /* Zero size value means "copied up but origin unknown" */
112 if (res == 0)
113 return NULL;
114
115 fh = kzalloc(res, GFP_KERNEL);
116 if (!fh)
117 return ERR_PTR(-ENOMEM);
118
119 res = vfs_getxattr(dentry, OVL_XATTR_ORIGIN, fh, res);
120 if (res < 0)
121 goto fail;
122
123 if (res < sizeof(struct ovl_fh) || res < fh->len)
124 goto invalid;
125
126 if (fh->magic != OVL_FH_MAGIC)
127 goto invalid;
128
129 /* Treat larger version and unknown flags as "origin unknown" */
130 if (fh->version > OVL_FH_VERSION || fh->flags & ~OVL_FH_FLAG_ALL)
131 goto out;
132
133 /* Treat endianness mismatch as "origin unknown" */
134 if (!(fh->flags & OVL_FH_FLAG_ANY_ENDIAN) &&
135 (fh->flags & OVL_FH_FLAG_BIG_ENDIAN) != OVL_FH_FLAG_CPU_ENDIAN)
136 goto out;
137
138 return fh;
139
140out:
141 kfree(fh);
142 return NULL;
143
144fail:
145 pr_warn_ratelimited("overlayfs: failed to get origin (%i)\n", res);
146 goto out;
147invalid:
148 pr_warn_ratelimited("overlayfs: invalid origin (%*phN)\n", res, fh);
149 goto out;
150}

152static struct dentry *ovl_get_origin(struct dentry *dentry,
153 struct vfsmount *mnt)
154{
155 struct dentry *origin = NULL;
156 struct ovl_fh *fh = ovl_get_origin_fh(dentry);
157 int bytes;
158
159 if (IS_ERR_OR_NULL(fh))
160 return (struct dentry *)fh;
161
162 /*
163 * Make sure that the stored uuid matches the uuid of the lower
164 * layer where file handle will be decoded.
165 */
166 if (!uuid_equal(&fh->uuid, &mnt->mnt_sb->s_uuid))
167 goto out;
168
169 bytes = (fh->len - offsetof(struct ovl_fh, fid));
170 origin = exportfs_decode_fh(mnt, (struct fid *)fh->fid,
171 bytes >> 2, (int)fh->type,
172 ovl_acceptable, NULL);
173 if (IS_ERR(origin)) {
174 /* Treat stale file handle as "origin unknown" */
175 if (origin == ERR_PTR(-ESTALE))
176 origin = NULL;
177 goto out;
178 }
179
180 if (ovl_dentry_weird(origin) ||
181 ((d_inode(origin)->i_mode ^ d_inode(dentry)->i_mode) & S_IFMT))
182 goto invalid;
183
184out:
185 kfree(fh);
186 return origin;
187
188invalid:
189 pr_warn_ratelimited("overlayfs: invalid origin (%pd2)\n", origin);
190 dput(origin);
191 origin = NULL;
192 goto out;
193}

200static int ovl_lookup_single(struct dentry *base, struct ovl_lookup_data *d,
201 const char *name, unsigned int namelen,
202 size_t prelen, const char *post,
203 struct dentry **ret)
204{
205 struct dentry *this;
206 int err;
207
208 this = lookup_one_len_unlocked(name, base, namelen);
209 if (IS_ERR(this)) {
210 err = PTR_ERR(this);
211 this = NULL;
212 if (err == -ENOENT || err == -ENAMETOOLONG)
213 goto out;
214 goto out_err;
215 }
216 if (!this->d_inode)
217 goto put_and_out;
218
219 if (ovl_dentry_weird(this)) {
220 /* Don't support traversing automounts and other weirdness */
221 err = -EREMOTE;
222 goto out_err;
223 }
224 if (ovl_is_whiteout(this)) {
225 d->stop = d->opaque = true;
226 goto put_and_out;
227 }
228 if (!d_can_lookup(this)) {
229 d->stop = true;
230 if (d->is_dir)
231 goto put_and_out;
232 goto out;
233 }
234 d->is_dir = true;
235 if (!d->last && ovl_is_opaquedir(this)) {
236 d->stop = d->opaque = true;
237 goto out;
238 }
239 err = ovl_check_redirect(this, d, prelen, post);
240 if (err)
241 goto out_err;
242out:
243 *ret = this;
244 return 0;
245
246put_and_out:
247 dput(this);
248 this = NULL;
249 goto out;
250
251out_err:
252 dput(this);
253 return err;
254}

256static int ovl_lookup_layer(struct dentry *base, struct ovl_lookup_data *d,
257 struct dentry **ret)
258{
259 /* Counting down from the end, since the prefix can change */
260 size_t rem = d->name.len - 1;
261 struct dentry *dentry = NULL;
262 int err;
263
264 if (d->name.name[0] != '/')
265 return ovl_lookup_single(base, d, d->name.name, d->name.len,
266 0, "", ret);
267
268 while (!IS_ERR_OR_NULL(base) && d_can_lookup(base)) {
269 const char *s = d->name.name + d->name.len - rem;
270 const char *next = strchrnul(s, '/');
271 size_t thislen = next - s;
272 bool end = !next[0];
273
274 /* Verify we did not go off the rails */
275 if (WARN_ON(s[-1] != '/'))
276 return -EIO;
277
278 err = ovl_lookup_single(base, d, s, thislen,
279 d->name.len - rem, next, &base);
280 dput(dentry);
281 if (err)
282 return err;
283 dentry = base;
284 if (end)
285 break;
286
287 rem -= thislen + 1;
288
289 if (WARN_ON(rem >= d->name.len))
290 return -EIO;
291 }
292 *ret = dentry;
293 return 0;
294}

297static int ovl_check_origin(struct dentry *upperdentry,
298 struct path *lowerstack, unsigned int numlower,
299 struct path **stackp, unsigned int *ctrp)
300{
301 struct vfsmount *mnt;
302 struct dentry *origin = NULL;
303 int i;
304
305
306 for (i = 0; i < numlower; i++) {
307 mnt = lowerstack[i].mnt;
308 origin = ovl_get_origin(upperdentry, mnt);
309 if (IS_ERR(origin))
310 return PTR_ERR(origin);
311
312 if (origin)
313 break;
314 }
315
316 if (!origin)
317 return 0;
318
319 BUG_ON(*ctrp);
320 if (!*stackp)
321 *stackp = kmalloc(sizeof(struct path), GFP_KERNEL);
322 if (!*stackp) {
323 dput(origin);
324 return -ENOMEM;
325 }
326 **stackp = (struct path) { .dentry = origin, .mnt = mnt };
327 *ctrp = 1;
328
329 return 0;
330}

332/*
333 * Verify that @fh matches the origin file handle stored in OVL_XATTR_ORIGIN.
334 * Return 0 on match, -ESTALE on mismatch, < 0 on error.
335 */
336static int ovl_verify_origin_fh(struct dentry *dentry, const struct ovl_fh *fh)
337{
338 struct ovl_fh *ofh = ovl_get_origin_fh(dentry);
339 int err = 0;
340
341 if (!ofh)
342 return -ENODATA;
343
344 if (IS_ERR(ofh))
345 return PTR_ERR(ofh);
346
347 if (fh->len != ofh->len || memcmp(fh, ofh, fh->len))
348 err = -ESTALE;
349
350 kfree(ofh);
351 return err;
352}

354/*
355 * Verify that an inode matches the origin file handle stored in upper inode.
356 *
357 * If @set is true and there is no stored file handle, encode and store origin
358 * file handle in OVL_XATTR_ORIGIN.
359 *
360 * Return 0 on match, -ESTALE on mismatch, < 0 on error.
361 */
362int ovl_verify_origin(struct dentry *dentry, struct vfsmount *mnt,
363 struct dentry *origin, bool is_upper, bool set)
364{
365 struct inode *inode;
366 struct ovl_fh *fh;
367 int err;
368
369 fh = ovl_encode_fh(origin, is_upper);
370 err = PTR_ERR(fh);
371 if (IS_ERR(fh))
372 goto fail;
373
374 err = ovl_verify_origin_fh(dentry, fh);
375 if (set && err == -ENODATA)
376 err = ovl_do_setxattr(dentry, OVL_XATTR_ORIGIN, fh, fh->len, 0);
377 if (err)
378 goto fail;
379
380out:
381 kfree(fh);
382 return err;
383
384fail:
385 inode = d_inode(origin);
386 pr_warn_ratelimited("overlayfs: failed to verify origin (%pd2, ino=%lu, err=%i)\n",
387 origin, inode ? inode->i_ino : 0, err);
388 goto out;
389}

391/*
392 * Verify that an index entry name matches the origin file handle stored in
393 * OVL_XATTR_ORIGIN and that origin file handle can be decoded to lower path.
394 * Return 0 on match, -ESTALE on mismatch or stale origin, < 0 on error.
395 */
396int ovl_verify_index(struct dentry *index, struct path *lowerstack,
397 unsigned int numlower)
398{
399 struct ovl_fh *fh = NULL;
400 size_t len;
401 struct path origin = { };
402 struct path *stack = &origin;
403 unsigned int ctr = 0;
404 int err;
405
406 if (!d_inode(index))
407 return 0;
408
409 /*
410 * Directory index entries are going to be used for looking up
411 * redirected upper dirs by lower dir fh when decoding an overlay
412 * file handle of a merge dir. Whiteout index entries are going to be
413 * used as an indication that an exported overlay file handle should
414 * be treated as stale (i.e. after unlink of the overlay inode).
415 * We don't know the verification rules for directory and whiteout
416 * index entries, because they have not been implemented yet, so return
417 * EINVAL if those entries are found to abort the mount to avoid
418 * corrupting an index that was created by a newer kernel.
419 */
420 err = -EINVAL;
421 if (d_is_dir(index) || ovl_is_whiteout(index))
422 goto fail;
423
424 if (index->d_name.len < sizeof(struct ovl_fh)*2)
425 goto fail;
426
427 err = -ENOMEM;
428 len = index->d_name.len / 2;
429 fh = kzalloc(len, GFP_KERNEL);
430 if (!fh)
431 goto fail;
432
433 err = -EINVAL;
434 if (hex2bin((u8 *)fh, index->d_name.name, len) || len != fh->len)
435 goto fail;
436
437 err = ovl_verify_origin_fh(index, fh);
438 if (err)
439 goto fail;
440
441 err = ovl_check_origin(index, lowerstack, numlower, &stack, &ctr);
442 if (!err && !ctr)
443 err = -ESTALE;
444 if (err)
445 goto fail;
446
447 /* Check if index is orphan and don't warn before cleaning it */
448 if (d_inode(index)->i_nlink == 1 &&
449 ovl_get_nlink(origin.dentry, index, 0) == 0)
450 err = -ENOENT;
451
452 dput(origin.dentry);
453out:
454 kfree(fh);
455 return err;
456
457fail:
458 pr_warn_ratelimited("overlayfs: failed to verify index (%pd2, ftype=%x, err=%i)\n",
459 index, d_inode(index)->i_mode & S_IFMT, err);
460 goto out;
461}

463/*
464 * Lookup in indexdir for the index entry of a lower real inode or a copy up
465 * origin inode. The index entry name is the hex representation of the lower
466 * inode file handle.
467 *
468 * If the index dentry in negative, then either no lower aliases have been
469 * copied up yet, or aliases have been copied up in older kernels and are
470 * not indexed.
471 *
472 * If the index dentry for a copy up origin inode is positive, but points
473 * to an inode different than the upper inode, then either the upper inode
474 * has been copied up and not indexed or it was indexed, but since then
475 * index dir was cleared. Either way, that index cannot be used to indentify
476 * the overlay inode.
477 */
478int ovl_get_index_name(struct dentry *origin, struct qstr *name)
479{
480 int err;
481 struct ovl_fh *fh;
482 char *n, *s;
483
484 fh = ovl_encode_fh(origin, false);
485 if (IS_ERR(fh))
486 return PTR_ERR(fh);
487
488 err = -ENOMEM;
489 n = kzalloc(fh->len * 2, GFP_KERNEL);
490 if (n) {
491 s = bin2hex(n, fh, fh->len);
492 *name = (struct qstr) QSTR_INIT(n, s - n);
493 err = 0;
494 }
495 kfree(fh);
496
497 return err;
498
499}

501static struct dentry *ovl_lookup_index(struct dentry *dentry,
502 struct dentry *upper,
503 struct dentry *origin)
504{
505 struct ovl_fs *ofs = dentry->d_sb->s_fs_info;
506 struct dentry *index;
507 struct inode *inode;
508 struct qstr name;
509 int err;
510
511 err = ovl_get_index_name(origin, &name);
512 if (err)
513 return ERR_PTR(err);
514
515 index = lookup_one_len_unlocked(name.name, ofs->indexdir, name.len);
516 if (IS_ERR(index)) {
517 err = PTR_ERR(index);
518 if (err == -ENOENT) {
519 index = NULL;
520 goto out;
521 }
522 pr_warn_ratelimited("overlayfs: failed inode index lookup (ino=%lu, key=%*s, err=%i);\n"
523 "overlayfs: mount with '-o index=off' to disable inodes index.\n",
524 d_inode(origin)->i_ino, name.len, name.name,
525 err);
526 goto out;
527 }
528
529 inode = d_inode(index);
530 if (d_is_negative(index)) {
531 goto out_dput;
532 } else if (upper && d_inode(upper) != inode) {
533 goto out_dput;
534 } else if (ovl_dentry_weird(index) || ovl_is_whiteout(index) ||
535 ((inode->i_mode ^ d_inode(origin)->i_mode) & S_IFMT)) {
536 /*
537 * Index should always be of the same file type as origin
538 * except for the case of a whiteout index. A whiteout
539 * index should only exist if all lower aliases have been
540 * unlinked, which means that finding a lower origin on lookup
541 * whose index is a whiteout should be treated as an error.
542 */
543 pr_warn_ratelimited("overlayfs: bad index found (index=%pd2, ftype=%x, origin ftype=%x).\n",
544 index, d_inode(index)->i_mode & S_IFMT,
545 d_inode(origin)->i_mode & S_IFMT);
546 goto fail;
547 }
548
549out:
550 kfree(name.name);
551 return index;
552
553out_dput:
554 dput(index);
555 index = NULL;
556 goto out;
557
558fail:
559 dput(index);
560 index = ERR_PTR(-EIO);
561 goto out;
562}

564/*
565 * Returns next layer in stack starting from top.
566 * Returns -1 if this is the last layer.
567 */
568int ovl_path_next(int idx, struct dentry *dentry, struct path *path)
569{
570 struct ovl_entry *oe = dentry->d_fsdata;
571
572 BUG_ON(idx < 0);
573 if (idx == 0) {
574 ovl_path_upper(dentry, path);
575 if (path->dentry)
576 return oe->numlower ? 1 : -1;
577 idx++;
578 }
579 BUG_ON(idx > oe->numlower);
580 *path = oe->lowerstack[idx - 1];
581
582 return (idx < oe->numlower) ? idx + 1 : -1;
583}

585struct dentry *ovl_lookup(struct inode *dir, struct dentry *dentry,
586 unsigned int flags)
587{
588 struct ovl_entry *oe;
589 const struct cred *old_cred;
590 struct ovl_fs *ofs = dentry->d_sb->s_fs_info;
591 struct ovl_entry *poe = dentry->d_parent->d_fsdata;
592 struct ovl_entry *roe = dentry->d_sb->s_root->d_fsdata;
593 struct path *stack = NULL;
594 struct dentry *upperdir, *upperdentry = NULL;
595 struct dentry *index = NULL;
596 unsigned int ctr = 0;
597 struct inode *inode = NULL;
598 bool upperopaque = false;
599 char *upperredirect = NULL;
600 struct dentry *this;
601 unsigned int i;
602 int err;
603 struct ovl_lookup_data d = {
604 .name = dentry->d_name,
605 .is_dir = false,
606 .opaque = false,
607 .stop = false,
608 .last = !poe->numlower,
609 .redirect = NULL,
610 };
611
612 if (dentry->d_name.len > ofs->namelen)
613 return ERR_PTR(-ENAMETOOLONG);
614
615 old_cred = ovl_override_creds(dentry->d_sb);
616 upperdir = ovl_dentry_upper(dentry->d_parent);
617 if (upperdir) {
618 err = ovl_lookup_layer(upperdir, &d, &upperdentry);
619 if (err)
620 goto out;
621
622 if (upperdentry && unlikely(ovl_dentry_remote(upperdentry))) {
623 dput(upperdentry);
624 err = -EREMOTE;
625 goto out;
626 }
627 if (upperdentry && !d.is_dir) {
628 BUG_ON(!d.stop || d.redirect);
629 /*
630 * Lookup copy up origin by decoding origin file handle.
631 * We may get a disconnected dentry, which is fine,
632 * because we only need to hold the origin inode in
633 * cache and use its inode number. We may even get a
634 * connected dentry, that is not under any of the lower
635 * layers root. That is also fine for using it's inode
636 * number - it's the same as if we held a reference
637 * to a dentry in lower layer that was moved under us.
638 */
639 err = ovl_check_origin(upperdentry, roe->lowerstack,
640 roe->numlower, &stack, &ctr);
641 if (err)
642 goto out_put_upper;
643 }
644
645 if (d.redirect) {
646 err = -ENOMEM;
647 upperredirect = kstrdup(d.redirect, GFP_KERNEL);
648 if (!upperredirect)
649 goto out_put_upper;
650 if (d.redirect[0] == '/')
651 poe = roe;
652 }
653 upperopaque = d.opaque;
654 }
655
656 if (!d.stop && poe->numlower) {
657 err = -ENOMEM;
658 stack = kcalloc(ofs->numlower, sizeof(struct path),
659 GFP_KERNEL);
660 if (!stack)
661 goto out_put_upper;
662 }
663
664 for (i = 0; !d.stop && i < poe->numlower; i++) {
665 struct path lowerpath = poe->lowerstack[i];
666
667 d.last = i == poe->numlower - 1;
668 err = ovl_lookup_layer(lowerpath.dentry, &d, &this);
669 if (err)
670 goto out_put;
671
672 if (!this)
673 continue;
674
675 stack[ctr].dentry = this;
676 stack[ctr].mnt = lowerpath.mnt;
677 ctr++;
678
679 if (d.stop)
680 break;
681
682 if (d.redirect && d.redirect[0] == '/' && poe != roe) {
683 poe = roe;
684
685 /* Find the current layer on the root dentry */
686 for (i = 0; i < poe->numlower; i++)
687 if (poe->lowerstack[i].mnt == lowerpath.mnt)
688 break;
689 if (WARN_ON(i == poe->numlower))
690 break;
691 }
692 }
693
694 /* Lookup index by lower inode and verify it matches upper inode */
695 if (ctr && !d.is_dir && ovl_indexdir(dentry->d_sb)) {
696 struct dentry *origin = stack[0].dentry;
697
698 index = ovl_lookup_index(dentry, upperdentry, origin);
699 if (IS_ERR(index)) {
700 err = PTR_ERR(index);
701 index = NULL;
702 goto out_put;
703 }
704 }
705
706 oe = ovl_alloc_entry(ctr);
707 err = -ENOMEM;
708 if (!oe)
709 goto out_put;
710
711 oe->opaque = upperopaque;
712 memcpy(oe->lowerstack, stack, sizeof(struct path) * ctr);
713 dentry->d_fsdata = oe;
714
715 if (upperdentry)
716 ovl_dentry_set_upper_alias(dentry);
717 else if (index)
718 upperdentry = dget(index);
719
720 if (upperdentry || ctr) {
721 inode = ovl_get_inode(dentry, upperdentry, index);
722 err = PTR_ERR(inode);
723 if (IS_ERR(inode))
724 goto out_free_oe;
725
726 OVL_I(inode)->redirect = upperredirect;
727 if (index)
728 ovl_set_flag(OVL_INDEX, inode);
729 }
730
731 revert_creds(old_cred);
732 dput(index);
733 kfree(stack);
734 kfree(d.redirect);
735 d_add(dentry, inode);
736
737 return NULL;
738
739out_free_oe:
740 dentry->d_fsdata = NULL;
741 kfree(oe);
742out_put:
743 dput(index);
744 for (i = 0; i < ctr; i++)
745 dput(stack[i].dentry);
746 kfree(stack);
747out_put_upper:
748 dput(upperdentry);
749 kfree(upperredirect);
750out:
751 kfree(d.redirect);
752 revert_creds(old_cred);
753 return ERR_PTR(err);
754}

756bool ovl_lower_positive(struct dentry *dentry)
757{
758 struct ovl_entry *oe = dentry->d_fsdata;
759 struct ovl_entry *poe = dentry->d_parent->d_fsdata;
760 const struct qstr *name = &dentry->d_name;
761 unsigned int i;
762 bool positive = false;
763 bool done = false;
764
765 /*
766 * If dentry is negative, then lower is positive iff this is a
767 * whiteout.
768 */
769 if (!dentry->d_inode)
770 return oe->opaque;
771
772 /* Negative upper -> positive lower */
773 if (!ovl_dentry_upper(dentry))
774 return true;
775
776 /* Positive upper -> have to look up lower to see whether it exists */
777 for (i = 0; !done && !positive && i < poe->numlower; i++) {
778 struct dentry *this;
779 struct dentry *lowerdir = poe->lowerstack[i].dentry;
780
781 this = lookup_one_len_unlocked(name->name, lowerdir,
782 name->len);
783 if (IS_ERR(this)) {
784 switch (PTR_ERR(this)) {
785 case -ENOENT:
786 case -ENAMETOOLONG:
787 break;
788
789 default:
790 /*
791 * Assume something is there, we just couldn't
792 * access it.
793 */
794 positive = true;
795 break;
796 }
797 } else {
798 if (this->d_inode) {
799 positive = !ovl_is_whiteout(this);
800 done = true;
801 }
802 dput(this);
803 }
804 }
805
806 return positive;
807}
