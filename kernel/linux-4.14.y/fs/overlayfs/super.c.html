42static void ovl_dentry_release(struct dentry *dentry)
43{
44 struct ovl_entry *oe = dentry->d_fsdata;
45
46 if (oe) {
47 unsigned int i;
48
49 for (i = 0; i < oe->numlower; i++)
50 dput(oe->lowerstack[i].dentry);
51 kfree_rcu(oe, rcu);
52 }
53}

55static int ovl_check_append_only(struct inode *inode, int flag)
56{
57 /*
58 * This test was moot in vfs may_open() because overlay inode does
59 * not have the S_APPEND flag, so re-check on real upper inode
60 */
61 if (IS_APPEND(inode)) {
62 if ((flag & O_ACCMODE) != O_RDONLY && !(flag & O_APPEND))
63 return -EPERM;
64 if (flag & O_TRUNC)
65 return -EPERM;
66 }
67
68 return 0;
69}

71static struct dentry *ovl_d_real(struct dentry *dentry,
72 const struct inode *inode,
73 unsigned int open_flags, unsigned int flags)
74{
75 struct dentry *real;
76 int err;
77
78 if (flags & D_REAL_UPPER)
79 return ovl_dentry_upper(dentry);
80
81 if (!d_is_reg(dentry)) {
82 if (!inode || inode == d_inode(dentry))
83 return dentry;
84 goto bug;
85 }
86
87 if (open_flags) {
88 err = ovl_open_maybe_copy_up(dentry, open_flags);
89 if (err)
90 return ERR_PTR(err);
91 }
92
93 real = ovl_dentry_upper(dentry);
94 if (real && (!inode || inode == d_inode(real))) {
95 if (!inode) {
96 err = ovl_check_append_only(d_inode(real), open_flags);
97 if (err)
98 return ERR_PTR(err);
99 }
100 return real;
101 }
102
103 real = ovl_dentry_lower(dentry);
104 if (!real)
105 goto bug;
106
107 /* Handle recursion */
108 real = d_real(real, inode, open_flags, 0);
109
110 if (!inode || inode == d_inode(real))
111 return real;
112bug:
113 WARN(1, "ovl_d_real(%pd4, %s:%lu): real dentry not found\n", dentry,
114 inode ? inode->i_sb->s_id : "NULL", inode ? inode->i_ino : 0);
115 return dentry;
116}

118static int ovl_dentry_revalidate(struct dentry *dentry, unsigned int flags)
119{
120 struct ovl_entry *oe = dentry->d_fsdata;
121 unsigned int i;
122 int ret = 1;
123
124 for (i = 0; i < oe->numlower; i++) {
125 struct dentry *d = oe->lowerstack[i].dentry;
126
127 if (d->d_flags & DCACHE_OP_REVALIDATE) {
128 ret = d->d_op->d_revalidate(d, flags);
129 if (ret < 0)
130 return ret;
131 if (!ret) {
132 if (!(flags & LOOKUP_RCU))
133 d_invalidate(d);
134 return -ESTALE;
135 }
136 }
137 }
138 return 1;
139}

141static int ovl_dentry_weak_revalidate(struct dentry *dentry, unsigned int flags)
142{
143 struct ovl_entry *oe = dentry->d_fsdata;
144 unsigned int i;
145 int ret = 1;
146
147 for (i = 0; i < oe->numlower; i++) {
148 struct dentry *d = oe->lowerstack[i].dentry;
149
150 if (d->d_flags & DCACHE_OP_WEAK_REVALIDATE) {
151 ret = d->d_op->d_weak_revalidate(d, flags);
152 if (ret <= 0)
153 break;
154 }
155 }
156 return ret;
157}

173static struct inode *ovl_alloc_inode(struct super_block *sb)
174{
175 struct ovl_inode *oi = kmem_cache_alloc(ovl_inode_cachep, GFP_KERNEL);
176
177 if (!oi)
178 return NULL;
179
180 oi->cache = NULL;
181 oi->redirect = NULL;
182 oi->version = 0;
183 oi->flags = 0;
184 oi->__upperdentry = NULL;
185 oi->lower = NULL;
186 mutex_init(&oi->lock);
187
188 return &oi->vfs_inode;
189}

198static void ovl_destroy_inode(struct inode *inode)
199{
200 struct ovl_inode *oi = OVL_I(inode);
201
202 dput(oi->__upperdentry);
203 iput(oi->lower);
204 kfree(oi->redirect);
205 ovl_dir_cache_free(inode);
206 mutex_destroy(&oi->lock);
207
208 call_rcu(&inode->i_rcu, ovl_i_callback);
209}

211static void ovl_put_super(struct super_block *sb)
212{
213 struct ovl_fs *ufs = sb->s_fs_info;
214 unsigned i;
215
216 dput(ufs->indexdir);
217 dput(ufs->workdir);
218 if (ufs->workdir_locked)
219 ovl_inuse_unlock(ufs->workbasedir);
220 dput(ufs->workbasedir);
221 if (ufs->upper_mnt && ufs->upperdir_locked)
222 ovl_inuse_unlock(ufs->upper_mnt->mnt_root);
223 mntput(ufs->upper_mnt);
224 for (i = 0; i < ufs->numlower; i++)
225 mntput(ufs->lower_mnt[i]);
226 kfree(ufs->lower_mnt);
227
228 kfree(ufs->config.lowerdir);
229 kfree(ufs->config.upperdir);
230 kfree(ufs->config.workdir);
231 put_cred(ufs->creator_cred);
232 kfree(ufs);
233}

235static int ovl_sync_fs(struct super_block *sb, int wait)
236{
237 struct ovl_fs *ufs = sb->s_fs_info;
238 struct super_block *upper_sb;
239 int ret;
240
241 if (!ufs->upper_mnt)
242 return 0;
243 upper_sb = ufs->upper_mnt->mnt_sb;
244 if (!upper_sb->s_op->sync_fs)
245 return 0;
246
247 /* real inodes have already been synced by sync_filesystem(ovl_sb) */
248 down_read(&upper_sb->s_umount);
249 ret = upper_sb->s_op->sync_fs(upper_sb, wait);
250 up_read(&upper_sb->s_umount);
251 return ret;
252}

254/**
255 * ovl_statfs
256 * @sb: The overlayfs super block
257 * @buf: The struct kstatfs to fill in with stats
258 *
259 * Get the filesystem statistics. As writes always target the upper layer
260 * filesystem pass the statfs to the upper filesystem (if it exists)
261 */
262static int ovl_statfs(struct dentry *dentry, struct kstatfs *buf)
263{
264 struct ovl_fs *ofs = dentry->d_sb->s_fs_info;
265 struct dentry *root_dentry = dentry->d_sb->s_root;
266 struct path path;
267 int err;
268
269 ovl_path_real(root_dentry, &path);
270
271 err = vfs_statfs(&path, buf);
272 if (!err) {
273 buf->f_namelen = ofs->namelen;
274 buf->f_type = OVERLAYFS_SUPER_MAGIC;
275 }
276
277 return err;
278}

287 * ovl_show_options
288 *
289 * Prints the mount options for a given superblock.
290 * Returns zero; does not fail.
291 */
292static int ovl_show_options(struct seq_file *m, struct dentry *dentry)
293{
294 struct super_block *sb = dentry->d_sb;
295 struct ovl_fs *ufs = sb->s_fs_info;
296
297 seq_show_option(m, "lowerdir", ufs->config.lowerdir);
298 if (ufs->config.upperdir) {
299 seq_show_option(m, "upperdir", ufs->config.upperdir);
300 seq_show_option(m, "workdir", ufs->config.workdir);
301 }
302 if (ufs->config.default_permissions)
303 seq_puts(m, ",default_permissions");
304 if (ufs->config.redirect_dir != ovl_redirect_dir_def)
305 seq_printf(m, ",redirect_dir=%s",
306 ufs->config.redirect_dir ? "on" : "off");
307 if (ufs->config.index != ovl_index_def)
308 seq_printf(m, ",index=%s",
309 ufs->config.index ? "on" : "off");
310 return 0;
311}

358static char *ovl_next_opt(char **s)
359{
360 char *sbegin = *s;
361 char *p;
362
363 if (sbegin == NULL)
364 return NULL;
365
366 for (p = sbegin; *p; p++) {
367 if (*p == '\\') {
368 p++;
369 if (!*p)
370 break;
371 } else if (*p == ',') {
372 *p = '\0';
373 *s = p + 1;
374 return sbegin;
375 }
376 }
377 *s = NULL;
378 return sbegin;
379}

381static int ovl_parse_opt(char *opt, struct ovl_config *config)
382{
383 char *p;
384
385 while ((p = ovl_next_opt(&opt)) != NULL) {
386 int token;
387 substring_t args[MAX_OPT_ARGS];
388
389 if (!*p)
390 continue;
391
392 token = match_token(p, ovl_tokens, args);
393 switch (token) {
394 case OPT_UPPERDIR:
395 kfree(config->upperdir);
396 config->upperdir = match_strdup(&args[0]);
397 if (!config->upperdir)
398 return -ENOMEM;
399 break;
400
401 case OPT_LOWERDIR:
402 kfree(config->lowerdir);
403 config->lowerdir = match_strdup(&args[0]);
404 if (!config->lowerdir)
405 return -ENOMEM;
406 break;
407
408 case OPT_WORKDIR:
409 kfree(config->workdir);
410 config->workdir = match_strdup(&args[0]);
411 if (!config->workdir)
412 return -ENOMEM;
413 break;
414
415 case OPT_DEFAULT_PERMISSIONS:
416 config->default_permissions = true;
417 break;
418
419 case OPT_REDIRECT_DIR_ON:
420 config->redirect_dir = true;
421 break;
422
423 case OPT_REDIRECT_DIR_OFF:
424 config->redirect_dir = false;
425 break;
426
427 case OPT_INDEX_ON:
428 config->index = true;
429 break;
430
431 case OPT_INDEX_OFF:
432 config->index = false;
433 break;
434
435 default:
436 pr_err("overlayfs: unrecognized mount option \"%s\" or missing value\n", p);
437 return -EINVAL;
438 }
439 }
440
441 /* Workdir is useless in non-upper mount */
442 if (!config->upperdir && config->workdir) {
443 pr_info("overlayfs: option \"workdir=%s\" is useless in a non-upper mount, ignore\n",
444 config->workdir);
445 kfree(config->workdir);
446 config->workdir = NULL;
447 }
448
449 return 0;
450}

455static struct dentry *ovl_workdir_create(struct super_block *sb,
456 struct ovl_fs *ufs,
457 struct dentry *dentry,
458 const char *name, bool persist)
459{
460 struct inode *dir = dentry->d_inode;
461 struct vfsmount *mnt = ufs->upper_mnt;
462 struct dentry *work;
463 int err;
464 bool retried = false;
465 bool locked = false;
466
467 err = mnt_want_write(mnt);
468 if (err)
469 goto out_err;
470
471 inode_lock_nested(dir, I_MUTEX_PARENT);
472 locked = true;
473
474retry:
475 work = lookup_one_len(name, dentry, strlen(name));
476
477 if (!IS_ERR(work)) {
478 struct iattr attr = {
479 .ia_valid = ATTR_MODE,
480 .ia_mode = S_IFDIR | 0,
481 };
482
483 if (work->d_inode) {
484 err = -EEXIST;
485 if (retried)
486 goto out_dput;
487
488 if (persist)
489 goto out_unlock;
490
491 retried = true;
492 ovl_workdir_cleanup(dir, mnt, work, 0);
493 dput(work);
494 goto retry;
495 }
496
497 err = ovl_create_real(dir, work,
498 &(struct cattr){.mode = S_IFDIR | 0},
499 NULL, true);
500 if (err)
501 goto out_dput;
502
503 /*
504 * Try to remove POSIX ACL xattrs from workdir. We are good if:
505 *
506 * a) success (there was a POSIX ACL xattr and was removed)
507 * b) -ENODATA (there was no POSIX ACL xattr)
508 * c) -EOPNOTSUPP (POSIX ACL xattrs are not supported)
509 *
510 * There are various other error values that could effectively
511 * mean that the xattr doesn't exist (e.g. -ERANGE is returned
512 * if the xattr name is too long), but the set of filesystems
513 * allowed as upper are limited to "normal" ones, where checking
514 * for the above two errors is sufficient.
515 */
516 err = vfs_removexattr(work, XATTR_NAME_POSIX_ACL_DEFAULT);
517 if (err && err != -ENODATA && err != -EOPNOTSUPP)
518 goto out_dput;
519
520 err = vfs_removexattr(work, XATTR_NAME_POSIX_ACL_ACCESS);
521 if (err && err != -ENODATA && err != -EOPNOTSUPP)
522 goto out_dput;
523
524 /* Clear any inherited mode bits */
525 inode_lock(work->d_inode);
526 err = notify_change(work, &attr, NULL);
527 inode_unlock(work->d_inode);
528 if (err)
529 goto out_dput;
530 } else {
531 err = PTR_ERR(work);
532 goto out_err;
533 }
534out_unlock:
535 mnt_drop_write(mnt);
536 if (locked)
537 inode_unlock(dir);
538
539 return work;
540
541out_dput:
542 dput(work);
543out_err:
544 pr_warn("overlayfs: failed to create directory %s/%s (errno: %i); mounting read-only\n",
545 ufs->config.workdir, name, -err);
546 sb->s_flags |= MS_RDONLY;
547 work = NULL;
548 goto out_unlock;
549}

551static void ovl_unescape(char *s)
552{
553 char *d = s;
554
555 for (;; s++, d++) {

557 s++;

559 if (!*s)

561 }

564static int ovl_mount_dir_noesc(const char *name, struct path *path)
565{
566 int err = -EINVAL;
567
568 if (!*name) {
569 pr_err("overlayfs: empty lowerdir\n");
570 goto out;
571 }
572 err = kern_path(name, LOOKUP_FOLLOW, path);
573 if (err) {
574 pr_err("overlayfs: failed to resolve '%s': %i\n", name, err);
575 goto out;
576 }
577 err = -EINVAL;
578 if (ovl_dentry_weird(path->dentry)) {
579 pr_err("overlayfs: filesystem on '%s' not supported\n", name);
580 goto out_put;
581 }
582 if (!d_is_dir(path->dentry)) {
583 pr_err("overlayfs: '%s' not a directory\n", name);
584 goto out_put;
585 }
586 return 0;
587
588out_put:
589 path_put(path);
590out:
591 return err;
592}

594static int ovl_mount_dir(const char *name, struct path *path)
595{
596 int err = -ENOMEM;
597 char *tmp = kstrdup(name, GFP_KERNEL);
598
599 if (tmp) {
600 ovl_unescape(tmp);
601 err = ovl_mount_dir_noesc(tmp, path);
602
603 if (!err)
604 if (ovl_dentry_remote(path->dentry)) {
605 pr_err("overlayfs: filesystem on '%s' not supported as upperdir\n",
606 tmp);
607 path_put(path);
608 err = -EINVAL;
609 }
610 kfree(tmp);
611 }
612 return err;
613}

615static int ovl_check_namelen(struct path *path, struct ovl_fs *ofs,
616 const char *name)
617{
618 struct kstatfs statfs;

620
621 if (err)

623 else
624 ofs->namelen = max(ofs->namelen, statfs.f_namelen);
625
626 return err;

629static int ovl_lower_dir(const char *name, struct path *path,
630 struct ovl_fs *ofs, int *stack_depth, bool *remote)
631{
632 int err;
633
634 err = ovl_mount_dir_noesc(name, path);
635 if (err)
636 goto out;
637
638 err = ovl_check_namelen(path, ofs, name);
639 if (err)
640 goto out_put;
641
642 *stack_depth = max(*stack_depth, path->mnt->mnt_sb->s_stack_depth);
643
644 if (ovl_dentry_remote(path->dentry))
645 *remote = true;
646
647 /*
648 * The inodes index feature needs to encode and decode file
649 * handles, so it requires that all layers support them.
650 */
651 if (ofs->config.index && !ovl_can_decode_fh(path->dentry->d_sb)) {
652 ofs->config.index = false;
653 pr_warn("overlayfs: fs on '%s' does not support file handles, falling back to index=off.\n", name);
654 }
655
656 return 0;
657
658out_put:
659 path_put(path);
660out:
661 return err;
662}

664/* Workdir should not be subdir of upperdir and vice versa */
665static bool ovl_workdir_ok(struct dentry *workdir, struct dentry *upperdir)
666{
667 bool ok = false;
668
669 if (workdir != upperdir) {
670 ok = (lock_rename(workdir, upperdir) == NULL);
671 unlock_rename(workdir, upperdir);
672 }
673 return ok;
674}

676static unsigned int ovl_split_lowerdirs(char *str)
677{
678 unsigned int ctr = 1;
679 char *s, *d;
680
681 for (s = d = str;; s++, d++) {
682 if (*s == '\\') {
683 s++;
684 } else if (*s == ':') {
685 *d = '\0';
686 ctr++;
687 continue;
688 }
689 *d = *s;
690 if (!*s)
691 break;
692 }
693 return ctr;
694}

704static int __maybe_unused
705ovl_posix_acl_xattr_set(const struct xattr_handler *handler,
706 struct dentry *dentry, struct inode *inode,
707 const char *name, const void *value,
708 size_t size, int flags)
709{
710 struct dentry *workdir = ovl_workdir(dentry);
711 struct inode *realinode = ovl_inode_real(inode);
712 struct posix_acl *acl = NULL;
713 int err;
714
715 /* Check that everything is OK before copy-up */
716 if (value) {
717 acl = posix_acl_from_xattr(&init_user_ns, value, size);
718 if (IS_ERR(acl))
719 return PTR_ERR(acl);
720 }
721 err = -EOPNOTSUPP;
722 if (!IS_POSIXACL(d_inode(workdir)))
723 goto out_acl_release;
724 if (!realinode->i_op->set_acl)
725 goto out_acl_release;
726 if (handler->flags == ACL_TYPE_DEFAULT && !S_ISDIR(inode->i_mode)) {
727 err = acl ? -EACCES : 0;
728 goto out_acl_release;
729 }
730 err = -EPERM;
731 if (!inode_owner_or_capable(inode))
732 goto out_acl_release;
733
734 posix_acl_release(acl);
735
736 /*
737 * Check if sgid bit needs to be cleared (actual setacl operation will
738 * be done with mounter's capabilities and so that won't do it for us).
739 */
740 if (unlikely(inode->i_mode & S_ISGID) &&
741 handler->flags == ACL_TYPE_ACCESS &&
742 !in_group_p(inode->i_gid) &&
743 !capable_wrt_inode_uidgid(inode, CAP_FSETID)) {
744 struct iattr iattr = { .ia_valid = ATTR_KILL_SGID };
745
746 err = ovl_setattr(dentry, &iattr);
747 if (err)
748 return err;
749 }
750
751 err = ovl_xattr_set(dentry, inode, handler->name, value, size, flags);
752 if (!err)
753 ovl_copyattr(ovl_inode_real(inode), inode);
754
755 return err;
756
757out_acl_release:
758 posix_acl_release(acl);
759 return err;
760}

769static int ovl_own_xattr_set(const struct xattr_handler *handler,
770 struct dentry *dentry, struct inode *inode,
771 const char *name, const void *value,

830static int ovl_fill_super(struct super_block *sb, void *data, int silent)
831{
832 struct path upperpath = { };
833 struct path workpath = { };
834 struct dentry *root_dentry;
835 struct ovl_entry *oe;
836 struct ovl_fs *ufs;
837 struct path *stack = NULL;
838 char *lowertmp;
839 char *lower;
840 unsigned int numlower;
841 unsigned int stacklen = 0;
842 unsigned int i;
843 bool remote = false;
844 struct cred *cred;
845 int err;
846
847 err = -ENOMEM;
848 ufs = kzalloc(sizeof(struct ovl_fs), GFP_KERNEL);
849 if (!ufs)
850 goto out;
851
852 ufs->config.redirect_dir = ovl_redirect_dir_def;
853 ufs->config.index = ovl_index_def;
854 err = ovl_parse_opt((char *) data, &ufs->config);
855 if (err)
856 goto out_free_config;
857
858 err = -EINVAL;
859 if (!ufs->config.lowerdir) {
860 if (!silent)
861 pr_err("overlayfs: missing 'lowerdir'\n");
862 goto out_free_config;
863 }
864
865 sb->s_stack_depth = 0;
866 sb->s_maxbytes = MAX_LFS_FILESIZE;
867 if (ufs->config.upperdir) {
868 if (!ufs->config.workdir) {
869 pr_err("overlayfs: missing 'workdir'\n");
870 goto out_free_config;
871 }
872
873 err = ovl_mount_dir(ufs->config.upperdir, &upperpath);
874 if (err)
875 goto out_free_config;
876
877 /* Upper fs should not be r/o */
878 if (sb_rdonly(upperpath.mnt->mnt_sb)) {
879 pr_err("overlayfs: upper fs is r/o, try multi-lower layers mount\n");
880 err = -EINVAL;
881 goto out_put_upperpath;
882 }
883
884 err = ovl_check_namelen(&upperpath, ufs, ufs->config.upperdir);
885 if (err)
886 goto out_put_upperpath;
887
888 err = -EBUSY;
889 if (ovl_inuse_trylock(upperpath.dentry)) {
890 ufs->upperdir_locked = true;
891 } else if (ufs->config.index) {
892 pr_err("overlayfs: upperdir is in-use by another mount, mount with '-o index=off' to override exclusive upperdir protection.\n");
893 goto out_put_upperpath;
894 } else {
895 pr_warn("overlayfs: upperdir is in-use by another mount, accessing files from both mounts will result in undefined behavior.\n");
896 }
897
898 err = ovl_mount_dir(ufs->config.workdir, &workpath);
899 if (err)
900 goto out_unlock_upperdentry;
901
902 err = -EINVAL;
903 if (upperpath.mnt != workpath.mnt) {
904 pr_err("overlayfs: workdir and upperdir must reside under the same mount\n");
905 goto out_put_workpath;
906 }
907 if (!ovl_workdir_ok(workpath.dentry, upperpath.dentry)) {
908 pr_err("overlayfs: workdir and upperdir must be separate subtrees\n");
909 goto out_put_workpath;
910 }
911
912 err = -EBUSY;
913 if (ovl_inuse_trylock(workpath.dentry)) {
914 ufs->workdir_locked = true;
915 } else if (ufs->config.index) {
916 pr_err("overlayfs: workdir is in-use by another mount, mount with '-o index=off' to override exclusive workdir protection.\n");
917 goto out_put_workpath;
918 } else {
919 pr_warn("overlayfs: workdir is in-use by another mount, accessing files from both mounts will result in undefined behavior.\n");
920 }
921
922 ufs->workbasedir = workpath.dentry;
923 sb->s_stack_depth = upperpath.mnt->mnt_sb->s_stack_depth;
924 }
925 err = -ENOMEM;
926 lowertmp = kstrdup(ufs->config.lowerdir, GFP_KERNEL);
927 if (!lowertmp)
928 goto out_unlock_workdentry;
929
930 err = -EINVAL;
931 stacklen = ovl_split_lowerdirs(lowertmp);
932 if (stacklen > OVL_MAX_STACK) {
933 pr_err("overlayfs: too many lower directories, limit is %d\n",
934 OVL_MAX_STACK);
935 goto out_free_lowertmp;
936 } else if (!ufs->config.upperdir && stacklen == 1) {
937 pr_err("overlayfs: at least 2 lowerdir are needed while upperdir nonexistent\n");
938 goto out_free_lowertmp;
939 }
940
941 err = -ENOMEM;
942 stack = kcalloc(stacklen, sizeof(struct path), GFP_KERNEL);
943 if (!stack)
944 goto out_free_lowertmp;
945
946 err = -EINVAL;
947 lower = lowertmp;
948 for (numlower = 0; numlower < stacklen; numlower++) {
949 err = ovl_lower_dir(lower, &stack[numlower], ufs,
950 &sb->s_stack_depth, &remote);
951 if (err)
952 goto out_put_lowerpath;
953
954 lower = strchr(lower, '\0') + 1;
955 }
956
957 err = -EINVAL;
958 sb->s_stack_depth++;
959 if (sb->s_stack_depth > FILESYSTEM_MAX_STACK_DEPTH) {
960 pr_err("overlayfs: maximum fs stacking depth exceeded\n");
961 goto out_put_lowerpath;
962 }
963
964 if (ufs->config.upperdir) {
965 ufs->upper_mnt = clone_private_mount(&upperpath);
966 err = PTR_ERR(ufs->upper_mnt);
967 if (IS_ERR(ufs->upper_mnt)) {
968 pr_err("overlayfs: failed to clone upperpath\n");
969 goto out_put_lowerpath;
970 }
971
972 /* Don't inherit atime flags */
973 ufs->upper_mnt->mnt_flags &= ~(MNT_NOATIME | MNT_NODIRATIME | MNT_RELATIME);
974
975 sb->s_time_gran = ufs->upper_mnt->mnt_sb->s_time_gran;
976
977 ufs->workdir = ovl_workdir_create(sb, ufs, workpath.dentry,
978 OVL_WORKDIR_NAME, false);
979 /*
980 * Upper should support d_type, else whiteouts are visible.
981 * Given workdir and upper are on same fs, we can do
982 * iterate_dir() on workdir. This check requires successful
983 * creation of workdir in previous step.
984 */
985 if (ufs->workdir) {
986 struct dentry *temp;
987
988 err = ovl_check_d_type_supported(&workpath);
989 if (err < 0)
990 goto out_put_workdir;
991
992 /*
993 * We allowed this configuration and don't want to
994 * break users over kernel upgrade. So warn instead
995 * of erroring out.
996 */
997 if (!err)
998 pr_warn("overlayfs: upper fs needs to support d_type.\n");
999
1000 /* Check if upper/work fs supports O_TMPFILE */
1001 temp = ovl_do_tmpfile(ufs->workdir, S_IFREG | 0);
1002 ufs->tmpfile = !IS_ERR(temp);
1003 if (ufs->tmpfile)
1004 dput(temp);
1005 else
1006 pr_warn("overlayfs: upper fs does not support tmpfile.\n");
1007
1008 /*
1009 * Check if upper/work fs supports trusted.overlay.*
1010 * xattr
1011 */
1012 err = ovl_do_setxattr(ufs->workdir, OVL_XATTR_OPAQUE,
1013 "0", 1, 0);
1014 if (err) {
1015 ufs->noxattr = true;
1016 pr_warn("overlayfs: upper fs does not support xattr.\n");
1017 } else {
1018 vfs_removexattr(ufs->workdir, OVL_XATTR_OPAQUE);
1019 }
1020
1021 /* Check if upper/work fs supports file handles */
1022 if (ufs->config.index &&
1023 !ovl_can_decode_fh(ufs->workdir->d_sb)) {
1024 ufs->config.index = false;
1025 pr_warn("overlayfs: upper fs does not support file handles, falling back to index=off.\n");
1026 }
1027 }
1028 }
1029
1030 err = -ENOMEM;
1031 ufs->lower_mnt = kcalloc(numlower, sizeof(struct vfsmount *), GFP_KERNEL);
1032 if (ufs->lower_mnt == NULL)
1033 goto out_put_workdir;
1034 for (i = 0; i < numlower; i++) {
1035 struct vfsmount *mnt = clone_private_mount(&stack[i]);
1036
1037 err = PTR_ERR(mnt);
1038 if (IS_ERR(mnt)) {
1039 pr_err("overlayfs: failed to clone lowerpath\n");
1040 goto out_put_lower_mnt;
1041 }
1042 /*
1043 * Make lower_mnt R/O. That way fchmod/fchown on lower file
1044 * will fail instead of modifying lower fs.
1045 */
1046 mnt->mnt_flags |= MNT_READONLY | MNT_NOATIME;
1047
1048 ufs->lower_mnt[ufs->numlower] = mnt;
1049 ufs->numlower++;
1050
1051 /* Check if all lower layers are on same sb */
1052 if (i == 0)
1053 ufs->same_sb = mnt->mnt_sb;
1054 else if (ufs->same_sb != mnt->mnt_sb)
1055 ufs->same_sb = NULL;
1056 }
1057
1058 /* If the upper fs is nonexistent, we mark overlayfs r/o too */
1059 if (!ufs->upper_mnt)
1060 sb->s_flags |= MS_RDONLY;
1061 else if (ufs->upper_mnt->mnt_sb != ufs->same_sb)
1062 ufs->same_sb = NULL;
1063
1064 if (!(ovl_force_readonly(ufs)) && ufs->config.index) {
1065 /* Verify lower root is upper root origin */
1066 err = ovl_verify_origin(upperpath.dentry, ufs->lower_mnt[0],
1067 stack[0].dentry, false, true);
1068 if (err) {
1069 pr_err("overlayfs: failed to verify upper root origin\n");
1070 goto out_put_lower_mnt;
1071 }
1072
1073 ufs->indexdir = ovl_workdir_create(sb, ufs, workpath.dentry,
1074 OVL_INDEXDIR_NAME, true);
1075 if (ufs->indexdir) {
1076 /* Verify upper root is index dir origin */
1077 err = ovl_verify_origin(ufs->indexdir, ufs->upper_mnt,
1078 upperpath.dentry, true, true);
1079 if (err)
1080 pr_err("overlayfs: failed to verify index dir origin\n");
1081
1082 /* Cleanup bad/stale/orphan index entries */
1083 if (!err)
1084 err = ovl_indexdir_cleanup(ufs->indexdir,
1085 ufs->upper_mnt,
1086 stack, numlower);
1087 }
1088 if (err || !ufs->indexdir)
1089 pr_warn("overlayfs: try deleting index dir or mounting with '-o index=off' to disable inodes index.\n");
1090 if (err)
1091 goto out_put_indexdir;
1092 }
1093
1094 /* Show index=off/on in /proc/mounts for any of the reasons above */
1095 if (!ufs->indexdir)
1096 ufs->config.index = false;
1097
1098 if (remote)
1099 sb->s_d_op = &ovl_reval_dentry_operations;
1100 else
1101 sb->s_d_op = &ovl_dentry_operations;
1102
1103 err = -ENOMEM;
1104 ufs->creator_cred = cred = prepare_creds();
1105 if (!cred)
1106 goto out_put_indexdir;
1107
1108 /* Never override disk quota limits or use reserved space */
1109 cap_lower(cred->cap_effective, CAP_SYS_RESOURCE);
1110
1111 err = -ENOMEM;
1112 oe = ovl_alloc_entry(numlower);
1113 if (!oe)
1114 goto out_put_cred;
1115
1116 sb->s_magic = OVERLAYFS_SUPER_MAGIC;
1117 sb->s_op = &ovl_super_operations;
1118 sb->s_xattr = ovl_xattr_handlers;
1119 sb->s_fs_info = ufs;
1120 sb->s_flags |= MS_POSIXACL | MS_NOREMOTELOCK;
1121
1122 root_dentry = d_make_root(ovl_new_inode(sb, S_IFDIR, 0));
1123 if (!root_dentry)
1124 goto out_free_oe;
1125
1126 mntput(upperpath.mnt);
1127 for (i = 0; i < numlower; i++)
1128 mntput(stack[i].mnt);
1129 mntput(workpath.mnt);
1130 kfree(lowertmp);
1131
1132 if (upperpath.dentry) {
1133 oe->has_upper = true;
1134 if (ovl_is_impuredir(upperpath.dentry))
1135 ovl_set_flag(OVL_IMPURE, d_inode(root_dentry));
1136 }
1137 for (i = 0; i < numlower; i++) {
1138 oe->lowerstack[i].dentry = stack[i].dentry;
1139 oe->lowerstack[i].mnt = ufs->lower_mnt[i];
1140 }
1141 kfree(stack);
1142
1143 root_dentry->d_fsdata = oe;
1144
1145 ovl_inode_init(d_inode(root_dentry), upperpath.dentry,
1146 ovl_dentry_lower(root_dentry));
1147
1148 sb->s_root = root_dentry;
1149
1150 return 0;
1151
1152out_free_oe:
1153 kfree(oe);
1154out_put_cred:
1155 put_cred(ufs->creator_cred);
1156out_put_indexdir:
1157 dput(ufs->indexdir);
1158out_put_lower_mnt:
1159 for (i = 0; i < ufs->numlower; i++)
1160 mntput(ufs->lower_mnt[i]);
1161 kfree(ufs->lower_mnt);
1162out_put_workdir:
1163 dput(ufs->workdir);
1164 mntput(ufs->upper_mnt);
1165out_put_lowerpath:
1166 for (i = 0; i < numlower; i++)
1167 path_put(&stack[i]);
1168 kfree(stack);
1169out_free_lowertmp:
1170 kfree(lowertmp);
1171out_unlock_workdentry:
1172 if (ufs->workdir_locked)
1173 ovl_inuse_unlock(workpath.dentry);
1174out_put_workpath:
1175 path_put(&workpath);
1176out_unlock_upperdentry:
1177 if (ufs->upperdir_locked)
1178 ovl_inuse_unlock(upperpath.dentry);
1179out_put_upperpath:
1180 path_put(&upperpath);
1181out_free_config:
1182 kfree(ufs->config.lowerdir);
1183 kfree(ufs->config.upperdir);
1184 kfree(ufs->config.workdir);
1185 kfree(ufs);
1186out:
1187 return err;
1188}
