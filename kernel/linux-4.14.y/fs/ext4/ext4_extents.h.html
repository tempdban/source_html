52/*
53 * ext4_inode has i_block array (60 bytes total).
54 * The first 12 bytes store ext4_extent_header;
55 * the remainder stores an array of ext4_extent.
56 * For non-inode extent blocks, ext4_extent_tail
57 * follows the array.
58 */
59
60/*
61 * This is the extent tail on-disk structure.
62 * All other extent structures are 12 bytes long. It turns out that
63 * block_size % 12 >= 4 for at least all powers of 2 greater than 512, which
64 * covers all valid ext4 block sizes. Therefore, this tail structure can be
65 * crammed into the end of the block without having to rebalance the tree.
66 */
67struct ext4_extent_tail {
68 __le32 et_checksum; /* crc32c(uuid+inum+extent_block) */
69};
70
71/*
72 * This is the extent on-disk structure.
73 * It's used at the bottom of the tree.
74 */
75struct ext4_extent {
76 __le32 ee_block; /* first logical block extent covers */
77 __le16 ee_len; /* number of blocks covered by extent */
78 __le16 ee_start_hi; /* high 16 bits of physical block */
79 __le32 ee_start_lo; /* low 32 bits of physical block */
80};
81
82/*
83 * This is index on-disk structure.
84 * It's used at all the levels except the bottom.
85 */
86struct ext4_extent_idx {
87 __le32 ei_block; /* index covers logical blocks from 'block' */
88 __le32 ei_leaf_lo; /* pointer to the physical block of the next *
89 * level. leaf or next index could be there */
90 __le16 ei_leaf_hi; /* high 16 bits of physical block */
91 __u16 ei_unused;
92};
93
94/*
95 * Each block (leaves and indexes), even inode-stored has header.
96 */
97struct ext4_extent_header {
98 __le16 eh_magic; /* probably will support different formats */
99 __le16 eh_entries; /* number of valid entries */
100 __le16 eh_max; /* capacity of store in entries */
101 __le16 eh_depth; /* has tree real underlying blocks? */
102 __le32 eh_generation; /* generation of the tree */
103};
104
105#define EXT4_EXT_MAGIC cpu_to_le16(0xf30a)
106#define EXT4_MAX_EXTENT_DEPTH 5
107
108#define EXT4_EXTENT_TAIL_OFFSET(hdr) \
109 (sizeof(struct ext4_extent_header) + \
110 (sizeof(struct ext4_extent) * le16_to_cpu((hdr)->eh_max)))
111
112static inline struct ext4_extent_tail *
113find_ext4_extent_tail(struct ext4_extent_header *eh)
114{
115 return (struct ext4_extent_tail *)(((void *)eh) +
116 EXT4_EXTENT_TAIL_OFFSET(eh));
117}

119/*
120 * Array of ext4_ext_path contains path to some extent.
121 * Creation/lookup routines use it for traversal/splitting/etc.
122 * Truncate uses it to simulate recursive walking.
123 */
124struct ext4_ext_path {
125 ext4_fsblk_t p_block;
126 __u16 p_depth;
127 __u16 p_maxdepth;
128 struct ext4_extent *p_ext;
129 struct ext4_extent_idx *p_idx;
130 struct ext4_extent_header *p_hdr;
131 struct buffer_head *p_bh;
132};

138/*
139 * EXT_INIT_MAX_LEN is the maximum number of blocks we can have in an
140 * initialized extent. This is 2^15 and not (2^16 - 1), since we use the
141 * MSB of ee_len field in the extent datastructure to signify if this
142 * particular extent is an initialized extent or an unwritten (i.e.
143 * preallocated).
144 * EXT_UNWRITTEN_MAX_LEN is the maximum number of blocks we can have in an
145 * unwritten extent.
146 * If ee_len is <= 0x8000, it is an initialized extent. Otherwise, it is an
147 * unwritten one. In other words, if MSB of ee_len is set, it is an
148 * unwritten extent with only one special scenario when ee_len = 0x8000.
149 * In this case we can not have an unwritten extent of zero length and
150 * thus we make it as a special case of initialized extent with 0x8000 length.
151 * This way we get better extent-to-group alignment for initialized extents.
152 * Hence, the maximum number of blocks we can have in an *initialized*
153 * extent is 2^15 (32768) and in an *unwritten* extent is 2^15-1 (32767).
154 */
155#define EXT_INIT_MAX_LEN (1UL << 15)
156#define EXT_UNWRITTEN_MAX_LEN (EXT_INIT_MAX_LEN - 1)

159#define EXT_FIRST_EXTENT(__hdr__) \
160 ((struct ext4_extent *) (((char *) (__hdr__)) + \
161 sizeof(struct ext4_extent_header)))
162#define EXT_FIRST_INDEX(__hdr__) \
163 ((struct ext4_extent_idx *) (((char *) (__hdr__)) + \
164 sizeof(struct ext4_extent_header)))
165#define EXT_HAS_FREE_INDEX(__path__) \
166 (le16_to_cpu((__path__)->p_hdr->eh_entries) \
167 < le16_to_cpu((__path__)->p_hdr->eh_max))
168#define EXT_LAST_EXTENT(__hdr__) \
169 (EXT_FIRST_EXTENT((__hdr__)) + le16_to_cpu((__hdr__)->eh_entries) - 1)
170#define EXT_LAST_INDEX(__hdr__) \
171 (EXT_FIRST_INDEX((__hdr__)) + le16_to_cpu((__hdr__)->eh_entries) - 1)
172#define EXT_MAX_EXTENT(__hdr__) \
173 (EXT_FIRST_EXTENT((__hdr__)) + le16_to_cpu((__hdr__)->eh_max) - 1)
174#define EXT_MAX_INDEX(__hdr__) \
175 (EXT_FIRST_INDEX((__hdr__)) + le16_to_cpu((__hdr__)->eh_max) - 1)

177static inline struct ext4_extent_header *ext_inode_hdr(struct inode *inode)
178{
179 return (struct ext4_extent_header *) EXT4_I(inode)->i_data;
180}

182static inline struct ext4_extent_header *ext_block_hdr(struct buffer_head *bh)
183{
184 return (struct ext4_extent_header *) bh->b_data;
185}

187static inline unsigned short ext_depth(struct inode *inode)
188{
189 return le16_to_cpu(ext_inode_hdr(inode)->eh_depth);
190}

192static inline void ext4_ext_mark_unwritten(struct ext4_extent *ext)
193{
194 /* We can not have an unwritten extent of zero length! */
195 BUG_ON((le16_to_cpu(ext->ee_len) & ~EXT_INIT_MAX_LEN) == 0);
196 ext->ee_len |= cpu_to_le16(EXT_INIT_MAX_LEN);
197}
198
199static inline int ext4_ext_is_unwritten(struct ext4_extent *ext)
200{
201 /* Extent with ee_len of 0x8000 is treated as an initialized extent */
202 return (le16_to_cpu(ext->ee_len) > EXT_INIT_MAX_LEN);
203}
204
205static inline int ext4_ext_get_actual_len(struct ext4_extent *ext)
206{
207 return (le16_to_cpu(ext->ee_len) <= EXT_INIT_MAX_LEN ?
208 le16_to_cpu(ext->ee_len) :
209 (le16_to_cpu(ext->ee_len) - EXT_INIT_MAX_LEN));
210}
211
212static inline void ext4_ext_mark_initialized(struct ext4_extent *ext)
213{
214 ext->ee_len = cpu_to_le16(ext4_ext_get_actual_len(ext));
215}

217/*
218 * ext4_ext_pblock:
219 * combine low and high parts of physical block number into ext4_fsblk_t
220 */
221static inline ext4_fsblk_t ext4_ext_pblock(struct ext4_extent *ex)
222{
223 ext4_fsblk_t block;
224
225 block = le32_to_cpu(ex->ee_start_lo);
226 block |= ((ext4_fsblk_t) le16_to_cpu(ex->ee_start_hi) << 31) << 1;
227 return block;
228}
229
230/*
231 * ext4_idx_pblock:
232 * combine low and high parts of a leaf physical block number into ext4_fsblk_t
233 */
234static inline ext4_fsblk_t ext4_idx_pblock(struct ext4_extent_idx *ix)
235{
236 ext4_fsblk_t block;
237
238 block = le32_to_cpu(ix->ei_leaf_lo);
239 block |= ((ext4_fsblk_t) le16_to_cpu(ix->ei_leaf_hi) << 31) << 1;
240 return block;
241}

243/*
244 * ext4_ext_store_pblock:
245 * stores a large physical block number into an extent struct,
246 * breaking it into parts
247 */
248static inline void ext4_ext_store_pblock(struct ext4_extent *ex,
249 ext4_fsblk_t pb)
250{
251 ex->ee_start_lo = cpu_to_le32((unsigned long) (pb & 0xffffffff));
252 ex->ee_start_hi = cpu_to_le16((unsigned long) ((pb >> 31) >> 1) &
253 0xffff);
254}
255
256/*
257 * ext4_idx_store_pblock:
258 * stores a large physical block number into an index struct,
259 * breaking it into parts
260 */
261static inline void ext4_idx_store_pblock(struct ext4_extent_idx *ix,
262 ext4_fsblk_t pb)
263{
264 ix->ei_leaf_lo = cpu_to_le32((unsigned long) (pb & 0xffffffff));
265 ix->ei_leaf_hi = cpu_to_le16((unsigned long) ((pb >> 31) >> 1) &
266 0xffff);
267}
