113static inline struct config_item * to_item(struct dentry * dentry)
114{
115 struct configfs_dirent * sd = dentry->d_fsdata;
116 return ((struct config_item *) sd->s_element);
117}
118
119static inline struct configfs_attribute * to_attr(struct dentry * dentry)
120{
121 struct configfs_dirent * sd = dentry->d_fsdata;
122 return ((struct configfs_attribute *) sd->s_element);
123}
124
125static inline struct configfs_bin_attribute *to_bin_attr(struct dentry *dentry)
126{
127 struct configfs_attribute *attr = to_attr(dentry);
128
129 return container_of(attr, struct configfs_bin_attribute, cb_attr);
130}
131
132static inline struct config_item *configfs_get_config_item(struct dentry *dentry)
133{
134 struct config_item * item = NULL;
135
136 spin_lock(&dentry->d_lock);
137 if (!d_unhashed(dentry)) {
138 struct configfs_dirent * sd = dentry->d_fsdata;
139 if (sd->s_type & CONFIGFS_ITEM_LINK) {
140 struct configfs_symlink * sl = sd->s_element;
141 item = config_item_get(sl->sl_target);
142 } else
143 item = config_item_get(sd->s_element);
144 }
145 spin_unlock(&dentry->d_lock);
146
147 return item;
148}
149
150static inline void release_configfs_dirent(struct configfs_dirent * sd)
151{
152 if (!(sd->s_type & CONFIGFS_ROOT)) {
153 kfree(sd->s_iattr);
154 kmem_cache_free(configfs_dir_cachep, sd);
155 }
156}
157
158static inline struct configfs_dirent * configfs_get(struct configfs_dirent * sd)
159{
160 if (sd) {
161 WARN_ON(!atomic_read(&sd->s_count));
162 atomic_inc(&sd->s_count);
163 }
164 return sd;
165}
166
167static inline void configfs_put(struct configfs_dirent * sd)
168{
169 WARN_ON(!atomic_read(&sd->s_count));
170 if (atomic_dec_and_test(&sd->s_count))
171 release_configfs_dirent(sd);
172}
