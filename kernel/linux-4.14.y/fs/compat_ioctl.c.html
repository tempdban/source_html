122#define convert_in_user(srcptr, dstptr) \
123({ \
124 typeof(*srcptr) val; \
125 \
126 get_user(val, srcptr) || put_user(val, dstptr); \
127})

131 int err;
132
133 err = security_file_ioctl(file, cmd, arg);

182 &up->u.size.aspect_ratio);

184 err = -EFAULT;
185 }
186
187 return err;

276typedef struct sg_iovec32 {
277 compat_uint_t iov_base;
278 compat_uint_t iov_len;
279} sg_iovec32_t;

281static int sg_build_iovec(sg_io_hdr_t __user *sgio, void __user *dxferp, u16 iovec_count)
282{
283 sg_iovec_t __user *iov = (sg_iovec_t __user *) (sgio + 1);
284 sg_iovec32_t __user *iov32 = dxferp;
285 int i;
286
287 for (i = 0; i < iovec_count; i++) {
288 u32 base, len;
289
290 if (get_user(base, &iov32[i].iov_base) ||
291 get_user(len, &iov32[i].iov_len) ||
292 put_user(compat_ptr(base), &iov[i].iov_base) ||
293 put_user(len, &iov[i].iov_len))
294 return -EFAULT;
295 }
296
297 if (put_user(iov, &sgio->dxferp))
298 return -EFAULT;
299 return 0;
300}

302static int sg_ioctl_trans(struct file *file, unsigned int cmd,
303 sg_io_hdr32_t __user *sgio32)
304{
305 sg_io_hdr_t __user *sgio;
306 u16 iovec_count;
307 u32 data;
308 void __user *dxferp;
309 int err;
310 int interface_id;
311
312 if (get_user(interface_id, &sgio32->interface_id))
313 return -EFAULT;
314 if (interface_id != 'S')
315 return do_ioctl(file, cmd, (unsigned long)sgio32);
316
317 if (get_user(iovec_count, &sgio32->iovec_count))
318 return -EFAULT;
319
320 {
321 void __user *top = compat_alloc_user_space(0);
322 void __user *new = compat_alloc_user_space(sizeof(sg_io_hdr_t) +
323 (iovec_count * sizeof(sg_iovec_t)));
324 if (new > top)
325 return -EINVAL;
326
327 sgio = new;
328 }
329
330 /* Ok, now construct. */
331 if (copy_in_user(&sgio->interface_id, &sgio32->interface_id,
332 (2 * sizeof(int)) +
333 (2 * sizeof(unsigned char)) +
334 (1 * sizeof(unsigned short)) +
335 (1 * sizeof(unsigned int))))
336 return -EFAULT;
337
338 if (get_user(data, &sgio32->dxferp))
339 return -EFAULT;
340 dxferp = compat_ptr(data);
341 if (iovec_count) {
342 if (sg_build_iovec(sgio, dxferp, iovec_count))
343 return -EFAULT;
344 } else {
345 if (put_user(dxferp, &sgio->dxferp))
346 return -EFAULT;
347 }
348
349 {
350 unsigned char __user *cmdp;
351 unsigned char __user *sbp;
352
353 if (get_user(data, &sgio32->cmdp))
354 return -EFAULT;
355 cmdp = compat_ptr(data);
356
357 if (get_user(data, &sgio32->sbp))
358 return -EFAULT;
359 sbp = compat_ptr(data);
360
361 if (put_user(cmdp, &sgio->cmdp) ||
362 put_user(sbp, &sgio->sbp))
363 return -EFAULT;
364 }
365
366 if (copy_in_user(&sgio->timeout, &sgio32->timeout,
367 3 * sizeof(int)))
368 return -EFAULT;
369
370 if (get_user(data, &sgio32->usr_ptr))
371 return -EFAULT;
372 if (put_user(compat_ptr(data), &sgio->usr_ptr))
373 return -EFAULT;
374
375 err = do_ioctl(file, cmd, (unsigned long) sgio);
376
377 if (err >= 0) {
378 void __user *datap;
379
380 if (copy_in_user(&sgio32->pack_id, &sgio->pack_id,
381 sizeof(int)) ||
382 get_user(datap, &sgio->usr_ptr) ||
383 put_user((u32)(unsigned long)datap,
384 &sgio32->usr_ptr) ||
385 copy_in_user(&sgio32->status, &sgio->status,
386 (4 * sizeof(unsigned char)) +
387 (2 * sizeof(unsigned short)) +
388 (3 * sizeof(int))))
389 err = -EFAULT;
390 }
391
392 return err;
393}

634static int serial_struct_ioctl(struct file *file,
635 unsigned cmd, struct serial_struct32 __user *ss32)
636{
637 typedef struct serial_struct32 SS32;
638 int err;
639 struct serial_struct __user *ss = compat_alloc_user_space(sizeof(*ss));
640 __u32 udata;
641 unsigned int base;
642 unsigned char *iomem_base;
643
644 if (ss == NULL)
645 return -EFAULT;
646 if (cmd == TIOCSSERIAL) {
647 if (copy_in_user(ss, ss32, offsetof(SS32, iomem_base)) ||
648 get_user(udata, &ss32->iomem_base))
649 return -EFAULT;
650 iomem_base = compat_ptr(udata);
651 if (put_user(iomem_base, &ss->iomem_base) ||
652 convert_in_user(&ss32->iomem_reg_shift,
653 &ss->iomem_reg_shift) ||
654 convert_in_user(&ss32->port_high, &ss->port_high) ||
655 put_user(0UL, &ss->iomap_base))
656 return -EFAULT;
657 }
658 err = do_ioctl(file, cmd, (unsigned long)ss);
659 if (cmd == TIOCGSERIAL && err >= 0) {
660 if (copy_in_user(ss32, ss, offsetof(SS32, iomem_base)) ||
661 get_user(iomem_base, &ss->iomem_base))
662 return -EFAULT;
663 base = (unsigned long)iomem_base >> 32 ?
664 0xffffffff : (unsigned)(unsigned long)iomem_base;
665 if (put_user(base, &ss32->iomem_base) ||
666 convert_in_user(&ss->iomem_reg_shift,
667 &ss32->iomem_reg_shift) ||
668 convert_in_user(&ss->port_high, &ss32->port_high))
669 return -EFAULT;
670 }
671 return err;
672}

699 struct i2c_msg msgs[0];
700};

702static int do_i2c_rdwr_ioctl(struct file *file,
703 unsigned int cmd, struct i2c_rdwr_ioctl_data32 __user *udata)
704{
705 struct i2c_rdwr_aligned __user *tdata;
706 struct i2c_msg __user *tmsgs;
707 struct i2c_msg32 __user *umsgs;
708 compat_caddr_t datap;
709 u32 nmsgs;
710 int i;
711
712 if (get_user(nmsgs, &udata->nmsgs))
713 return -EFAULT;
714 if (nmsgs > I2C_RDWR_IOCTL_MAX_MSGS)
715 return -EINVAL;
716
717 if (get_user(datap, &udata->msgs))
718 return -EFAULT;
719 umsgs = compat_ptr(datap);
720
721 tdata = compat_alloc_user_space(sizeof(*tdata) +
722 nmsgs * sizeof(struct i2c_msg));
723 tmsgs = &tdata->msgs[0];
724
725 if (put_user(nmsgs, &tdata->cmd.nmsgs) ||
726 put_user(tmsgs, &tdata->cmd.msgs))
727 return -EFAULT;
728
729 for (i = 0; i < nmsgs; i++) {
730 if (copy_in_user(&tmsgs[i].addr, &umsgs[i].addr, 3*sizeof(u16)))
731 return -EFAULT;
732 if (get_user(datap, &umsgs[i].buf) ||
733 put_user(compat_ptr(datap), &tmsgs[i].buf))
734 return -EFAULT;
735 }
736 return do_ioctl(file, cmd, (unsigned long)tdata);
737}

739static int do_i2c_smbus_ioctl(struct file *file,
740 unsigned int cmd, struct i2c_smbus_ioctl_data32 __user *udata)
741{
742 struct i2c_smbus_ioctl_data __user *tdata;
743 union {
744 /* beginnings of those have identical layouts */
745 struct i2c_smbus_ioctl_data32 data32;
746 struct i2c_smbus_ioctl_data data;
747 } v;
748
749 tdata = compat_alloc_user_space(sizeof(*tdata));
750 if (tdata == NULL)
751 return -ENOMEM;
752
753 memset(&v, 0, sizeof(v));
754 if (copy_from_user(&v.data32, udata, sizeof(v.data32)))
755 return -EFAULT;
756 v.data.data = compat_ptr(v.data32.data);
757
758 if (copy_to_user(tdata, &v.data, sizeof(v.data)))
759 return -EFAULT;
760
761 return do_ioctl(file, cmd, (unsigned long)tdata);
762}

769static int rtc_ioctl(struct file *file,
770 unsigned cmd, void __user *argp)
771{
772 unsigned long __user *valp = compat_alloc_user_space(sizeof(*valp));
773 int ret;
774
775 if (valp == NULL)
776 return -EFAULT;
777 switch (cmd) {
778 case RTC_IRQP_READ32:
779 case RTC_EPOCH_READ32:
780 ret = do_ioctl(file, (cmd == RTC_IRQP_READ32) ?
781 RTC_IRQP_READ : RTC_EPOCH_READ,
782 (unsigned long)valp);
783 if (ret)
784 return ret;
785 return convert_in_user(valp, (unsigned int __user *)argp);
786 case RTC_IRQP_SET32:
787 return do_ioctl(file, RTC_IRQP_SET, (unsigned long)argp);
788 case RTC_EPOCH_SET32:
789 return do_ioctl(file, RTC_EPOCH_SET, (unsigned long)argp);
790 }
791
792 return -ENOIOCTLCMD;
793}

1519static int compat_ioctl_check_table(unsigned int xcmd)
1520{
1521 int i;
1522 const int max = ARRAY_SIZE(ioctl_pointer) - 1;
1523
1524 BUILD_BUG_ON(max >= (1 << 16));
1525
1526 /* guess initial offset into table, assuming a
1527 normalized distribution */
1528 i = ((xcmd >> 16) * max) >> 16;
1529
1530 /* do linear search up first, until greater or equal */
1531 while (ioctl_pointer[i] < xcmd && i < max)
1532 i++;
1533
1534 /* then do linear search down */
1535 while (ioctl_pointer[i] > xcmd && i > 0)
1536 i--;
1537
1538 return ioctl_pointer[i] == xcmd;
1539}

1541COMPAT_SYSCALL_DEFINE3(ioctl, unsigned int, fd, unsigned int, cmd,
1542 compat_ulong_t, arg32)
1543{
1544 unsigned long arg = arg32;
1545 struct fd f = fdget(fd);
1546 int error = -EBADF;
1547 if (!f.file)
1548 goto out;
1549
1550 /* RED-PEN how should LSM module know it's handling 32bit? */
1551 error = security_file_ioctl(f.file, cmd, arg);
1552 if (error)
1553 goto out_fput;
1554
1555 /*
1556 * To allow the compat_ioctl handlers to be self contained
1557 * we need to check the common ioctls here first.
1558 * Just handle them with the standard handlers below.
1559 */
1560 switch (cmd) {
1561 case FIOCLEX:
1562 case FIONCLEX:
1563 case FIONBIO:
1564 case FIOASYNC:
1565 case FIOQSIZE:
1566 break;
1567
1568#if defined(CONFIG_IA64) || defined(CONFIG_X86_64)
1569 case FS_IOC_RESVSP_32:
1570 case FS_IOC_RESVSP64_32:
1571 error = compat_ioctl_preallocate(f.file, compat_ptr(arg));
1572 goto out_fput;
1573#else
1574 case FS_IOC_RESVSP:
1575 case FS_IOC_RESVSP64:
1576 error = ioctl_preallocate(f.file, compat_ptr(arg));
1577 goto out_fput;
1578#endif
1579
1580 case FICLONE:
1581 case FICLONERANGE:
1582 case FIDEDUPERANGE:
1583 goto do_ioctl;
1584
1585 case FIBMAP:
1586 case FIGETBSZ:
1587 case FIONREAD:
1588 if (S_ISREG(file_inode(f.file)->i_mode))
1589 break;
1590 /*FALL THROUGH*/
1591
1592 default:
1593 if (f.file->f_op->compat_ioctl) {
1594 error = f.file->f_op->compat_ioctl(f.file, cmd, arg);
1595 if (error != -ENOIOCTLCMD)
1596 goto out_fput;
1597 }
1598
1599 if (!f.file->f_op->unlocked_ioctl)
1600 goto do_ioctl;
1601 break;
1602 }
1603
1604 if (compat_ioctl_check_table(XFORM(cmd)))
1605 goto found_handler;
1606
1607 error = do_ioctl_trans(cmd, arg, f.file);
1608 if (error == -ENOIOCTLCMD)
1609 error = -ENOTTY;
1610
1611 goto out_fput;
1612
1613 found_handler:
1614 arg = (unsigned long)compat_ptr(arg);
1615 do_ioctl:
1616 error = do_vfs_ioctl(f.file, fd, cmd, arg);
1617 out_fput:
1618 fdput(f);
1619 out:
1620 return error;
1621}
