12struct fsnotify_iter_info {
13 struct fsnotify_mark *inode_mark;
14 struct fsnotify_mark *vfsmount_mark;
15 int srcu_idx;
16};
17
18/* destroy all events sitting in this groups notification queue */
19extern void fsnotify_flush_notify(struct fsnotify_group *group);
20
21/* protects reads of inode and vfsmount marks list */
22extern struct srcu_struct fsnotify_mark_srcu;
23
24/* compare two groups for sorting of marks lists */
25extern int fsnotify_compare_groups(struct fsnotify_group *a,
26 struct fsnotify_group *b);
27
28/* Destroy all marks connected via given connector */
29extern void fsnotify_destroy_marks(struct fsnotify_mark_connector __rcu **connp);
30/* run the list of all marks associated with inode and destroy them */
31static inline void fsnotify_clear_marks_by_inode(struct inode *inode)
32{
33 fsnotify_destroy_marks(&inode->i_fsnotify_marks);
34}
35/* run the list of all marks associated with vfsmount and destroy them */
36static inline void fsnotify_clear_marks_by_mount(struct vfsmount *mnt)
37{
38 fsnotify_destroy_marks(&real_mount(mnt)->mnt_fsnotify_marks);
39}
40/* Wait until all marks queued for destruction are destroyed */
41extern void fsnotify_wait_marks_destroyed(void);
42
43/*
44 * update the dentry->d_flags of all of inode's children to indicate if inode cares
45 * about events that happen to its children.
46 */
47extern void __fsnotify_update_child_dentry_flags(struct inode *inode);
48
49/* allocate and destroy and event holder to attach events to notification/access queues */
50extern struct fsnotify_event_holder *fsnotify_alloc_event_holder(void);
51extern void fsnotify_destroy_event_holder(struct fsnotify_event_holder *holder);
52
