3 * Debugging printout:
4 */

6#include <linux/kallsyms.h>
7
8#define ___P(f) if (desc->status_use_accessors & f) printk("%14s set\n", #f)
9#define ___PS(f) if (desc->istate & f) printk("%14s set\n", #f)
10/* FIXME */
11#define ___PD(f) do { } while (0)
12
13static inline void print_irq_desc(unsigned int irq, struct irq_desc *desc)
14{
15 static DEFINE_RATELIMIT_STATE(ratelimit, 5 * HZ, 5);
16
17 if (!__ratelimit(&ratelimit))
18 return;
19
20 printk("irq %d, desc: %p, depth: %d, count: %d, unhandled: %d\n",
21 irq, desc, desc->depth, desc->irq_count, desc->irqs_unhandled);
22 printk("->handle_irq(): %p, ", desc->handle_irq);
23 print_symbol("%s\n", (unsigned long)desc->handle_irq);
24 printk("->irq_data.chip(): %p, ", desc->irq_data.chip);
25 print_symbol("%s\n", (unsigned long)desc->irq_data.chip);
26 printk("->action(): %p\n", desc->action);
27 if (desc->action) {
28 printk("->action->handler(): %p, ", desc->action->handler);
29 print_symbol("%s\n", (unsigned long)desc->action->handler);
30 }
31
32 ___P(IRQ_LEVEL);
33 ___P(IRQ_PER_CPU);
34 ___P(IRQ_NOPROBE);
35 ___P(IRQ_NOREQUEST);
36 ___P(IRQ_NOTHREAD);
37 ___P(IRQ_NOAUTOEN);
38
39 ___PS(IRQS_AUTODETECT);
40 ___PS(IRQS_REPLAY);
41 ___PS(IRQS_WAITING);
42 ___PS(IRQS_PENDING);
43
44 ___PD(IRQS_INPROGRESS);
45 ___PD(IRQS_DISABLED);
46 ___PD(IRQS_MASKED);
47}
48
49#undef ___P
50#undef ___PS
51#undef ___PD
