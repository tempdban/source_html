18/*
19 * lock for reading
20 */
21void __sched down_read(struct rw_semaphore *sem)
22{
23 might_sleep();
24 rwsem_acquire_read(&sem->dep_map, 0, 0, _RET_IP_);
25
26 LOCK_CONTENDED(sem, __down_read_trylock, __down_read);
27 rwsem_set_reader_owned(sem);
28}
29
30EXPORT_SYMBOL(down_read);
31
32/*
33 * trylock for reading -- returns 1 if successful, 0 if contention
34 */
35int down_read_trylock(struct rw_semaphore *sem)
36{
37 int ret = __down_read_trylock(sem);
38
39 if (ret == 1) {
40 rwsem_acquire_read(&sem->dep_map, 0, 1, _RET_IP_);
41 rwsem_set_reader_owned(sem);
42 }
43 return ret;
44}
45
46EXPORT_SYMBOL(down_read_trylock);

48/*
49 * lock for writing
50 */
51void __sched down_write(struct rw_semaphore *sem)
52{
53 might_sleep();
54 rwsem_acquire(&sem->dep_map, 0, 0, _RET_IP_);
55
56 LOCK_CONTENDED(sem, __down_write_trylock, __down_write);
57 rwsem_set_owner(sem);
58}
59
60EXPORT_SYMBOL(down_write);
61
62/*
63 * lock for writing
64 */
65int __sched down_write_killable(struct rw_semaphore *sem)
66{
67 might_sleep();
68 rwsem_acquire(&sem->dep_map, 0, 0, _RET_IP_);
69
70 if (LOCK_CONTENDED_RETURN(sem, __down_write_trylock, __down_write_killable)) {
71 rwsem_release(&sem->dep_map, 1, _RET_IP_);
72 return -EINTR;
73 }
74
75 rwsem_set_owner(sem);
76 return 0;
77}
78
79EXPORT_SYMBOL(down_write_killable);
80
81/*
82 * trylock for writing -- returns 1 if successful, 0 if contention
83 */
84int down_write_trylock(struct rw_semaphore *sem)
85{
86 int ret = __down_write_trylock(sem);
87
88 if (ret == 1) {
89 rwsem_acquire(&sem->dep_map, 0, 1, _RET_IP_);
90 rwsem_set_owner(sem);
91 }
92
93 return ret;
94}
95
96EXPORT_SYMBOL(down_write_trylock);
97
98/*
99 * release a read lock
100 */
101void up_read(struct rw_semaphore *sem)
102{
103 rwsem_release(&sem->dep_map, 1, _RET_IP_);
104
105 __up_read(sem);
106}
107
108EXPORT_SYMBOL(up_read);
109
110/*
111 * release a write lock
112 */
113void up_write(struct rw_semaphore *sem)
114{
115 rwsem_release(&sem->dep_map, 1, _RET_IP_);
116
117 rwsem_clear_owner(sem);
118 __up_write(sem);
119}
120
121EXPORT_SYMBOL(up_write);
122
123/*
124 * downgrade write lock to read lock
125 */
126void downgrade_write(struct rw_semaphore *sem)
127{
128 lock_downgrade(&sem->dep_map, _RET_IP_);
129
130 rwsem_set_reader_owned(sem);
131 __downgrade_write(sem);
132}
133
134EXPORT_SYMBOL(downgrade_write);
