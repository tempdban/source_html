74/*
75 * Increase resolution of nice-level calculations for 64-bit architectures.
76 * The extra resolution improves shares distribution and load balancing of
77 * low-weight task groups (eg. nice +19 on an autogroup), deeper taskgroup
78 * hierarchies, especially on larger systems. This is not a user-visible change
79 * and does not change the user-interface for setting shares/weights.
80 *
81 * We increase resolution only if we have enough bits to allow this increased
82 * resolution (i.e. 64bit). The costs for increasing resolution when 32bit are
83 * pretty high and the returns do not justify the increased costs.
84 *
85 * Really only required when CONFIG_FAIR_GROUP_SCHED is also set, but to
86 * increase coverage and consistency always enable it on 64bit platforms.
87 */
88#ifdef CONFIG_64BIT
89# define NICE_0_LOAD_SHIFT (SCHED_FIXEDPOINT_SHIFT + SCHED_FIXEDPOINT_SHIFT)
90# define scale_load(w) ((w) << SCHED_FIXEDPOINT_SHIFT)
91# define scale_load_down(w) ((w) >> SCHED_FIXEDPOINT_SHIFT)
92#else
93# define NICE_0_LOAD_SHIFT (SCHED_FIXEDPOINT_SHIFT)
94# define scale_load(w) (w)
95# define scale_load_down(w) (w)
96#endif

99 * Task weight (visible to users) and its load (invisible to users) have
100 * independent resolution, but they should be well calibrated. We use
101 * scale_load() and scale_load_down(w) to convert between them. The
102 * following must be true:
103 *
104 * scale_load(sched_prio_to_weight[USER_PRIO(NICE_TO_PRIO(0))]) == NICE_0_LOAD
105 *
106 */
107#define NICE_0_LOAD (1L << NICE_0_LOAD_SHIFT)

187/*
188 * To keep the bandwidth of -deadline tasks and groups under control
189 * we need some place where:
190 * - store the maximum -deadline bandwidth of the system (the group);
191 * - cache the fraction of that bandwidth that is currently allocated.
192 *
193 * This is all done in the data structure below. It is similar to the
194 * one used for RT-throttling (rt_bandwidth), with the main difference
195 * that, since here we are only interested in admission control, we
196 * do not decrease any runtime while the group "executes", neither we
197 * need a timer to replenish it.
198 *
199 * With respect to SMP, the bandwidth is given on a per-CPU basis,
200 * meaning that:
201 * - dl_bw (< 100%) is the bandwidth of the system (group) on each CPU;
202 * - dl_total_bw array contains, in the i-eth element, the currently
203 * allocated bandwidth on the i-eth CPU.
204 * Moreover, groups consume bandwidth on each CPU, while tasks only
205 * consume bandwidth on the CPU they're running on.
206 * Finally, dl_total_bw_cpu is used to cache the index of dl_total_bw
207 * that will be shown the next time the proc or cgroup controls will
208 * be red. It on its turn can be changed by writing on its own
209 * control.
210 */
211struct dl_bandwidth {
212 raw_spinlock_t dl_runtime_lock;
213 u64 dl_runtime;
214 u64 dl_period;
215};

231{
232 dl_b->total_bw -= tsk_bw;

236static inline
237void __dl_add(struct dl_bw *dl_b, u64 tsk_bw, int cpus)
238{
239 dl_b->total_bw += tsk_bw;
240 __dl_update(dl_b, -((s32)tsk_bw / cpus));
241}

243static inline
244bool __dl_overflow(struct dl_bw *dl_b, int cpus, u64 old_bw, u64 new_bw)
245{
246 return dl_b->bw != -1 &&
247 dl_b->bw * cpus < dl_b->total_bw - old_bw + new_bw;
248}

854/*
855 * rq::clock_update_flags bits
856 *
857 * %RQCF_REQ_SKIP - will request skipping of clock update on the next
858 * call to __schedule(). This is an optimisation to avoid
859 * neighbouring rq clock updates.
860 *
861 * %RQCF_ACT_SKIP - is set from inside of __schedule() when skipping is
862 * in effect and calls to update_rq_clock() are being ignored.
863 *
864 * %RQCF_UPDATED - is a debug flag that indicates whether a call has been
865 * made to update_rq_clock() since the last time rq::lock was pinned.
866 *
867 * If inside of __schedule(), clock_update_flags will have been
868 * shifted left (a left shift is a cheap operation for the fast path
869 * to promote %RQCF_REQ_SKIP to %RQCF_ACT_SKIP), so you must use,
870 *
871 * if (rq-clock_update_flags >= RQCF_UPDATED)
872 *
873 * to check if %RQCF_UPADTED is set. It'll never be shifted more than
874 * one position though, because the next rq_unpin_lock() will shift it
875 * back.
876 */
877#define RQCF_REQ_SKIP 0x01
878#define RQCF_ACT_SKIP 0x02
879#define RQCF_UPDATED 0x04

888}

1784/*
1785 * fair double_lock_balance: Safely acquires both rq->locks in a fair
1786 * way at the expense of forcing extra atomic operations in all
1787 * invocations. This assures that the double_lock is acquired using the
1788 * same underlying policy as the spinlock_t on this architecture, which
1789 * reduces latency compared to the unfair variant below. However, it
1790 * also adds more overhead and therefore may reduce throughput.
1791 */
1792static inline int _double_lock_balance(struct rq *this_rq, struct rq *busiest)
1793 __releases(this_rq->lock)
1794 __acquires(busiest->lock)
1795 __acquires(this_rq->lock)
1796{
1797 raw_spin_unlock(&this_rq->lock);
1798 double_rq_lock(this_rq, busiest);
1799
1800 return 1;
1801}
1802
1803#else
1804/*
1805 * Unfair double_lock_balance: Optimizes throughput at the expense of
1806 * latency by eliminating extra atomic operations when the locks are
1807 * already in proper order on entry. This favors lower cpu-ids and will
1808 * grant the double lock to lower cpus over higher ids under contention,
1809 * regardless of entry order into the function.
1810 */
1811static inline int _double_lock_balance(struct rq *this_rq, struct rq *busiest)
1812 __releases(this_rq->lock)
1813 __acquires(busiest->lock)
1814 __acquires(this_rq->lock)
1815{
1816 int ret = 0;
1817
1818 if (unlikely(!raw_spin_trylock(&busiest->lock))) {
1819 if (busiest < this_rq) {
1820 raw_spin_unlock(&this_rq->lock);
1821 raw_spin_lock(&busiest->lock);
1822 raw_spin_lock_nested(&this_rq->lock,
1823 SINGLE_DEPTH_NESTING);
1824 ret = 1;
1825 } else
1826 raw_spin_lock_nested(&busiest->lock,
1827 SINGLE_DEPTH_NESTING);
1828 }
1829 return ret;
1830}
