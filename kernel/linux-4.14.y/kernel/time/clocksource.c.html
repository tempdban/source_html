39/**
40 * clocks_calc_mult_shift - calculate mult/shift factors for scaled math of clocks
41 * @mult: pointer to mult variable
42 * @shift: pointer to shift variable
43 * @from: frequency to convert from
44 * @to: frequency to convert to
45 * @maxsec: guaranteed runtime conversion range in seconds
46 *
47 * The function evaluates the shift/mult pair for the scaled math
48 * operations of clocksources and clockevents.
49 *
50 * @to and @from are frequency values in HZ. For clock sources @to is
51 * NSEC_PER_SEC == 1GHz and @from is the counter frequency. For clock
52 * event @to is the counter frequency and @from is NSEC_PER_SEC.
53 *
54 * The @maxsec conversion range argument controls the time frame in
55 * seconds which must be covered by the runtime conversion with the
56 * calculated mult and shift factors. This guarantees that no 64bit
57 * overflow happens when the input value of the conversion is
58 * multiplied with the calculated mult factor. Larger ranges may
59 * reduce the conversion accuracy by chosing smaller mult and shift
60 * factors.
61 */
62void
63clocks_calc_mult_shift(u32 *mult, u32 *shift, u32 from, u32 to, u32 maxsec)
64{
65 u64 tmp;
66 u32 sft, sftacc= 32;
67
68 /*
69 * Calculate the shift factor which is limiting the conversion
70 * range:
71 */
72 tmp = ((u64)maxsec * from) >> 32;
73 while (tmp) {
74 tmp >>=1;
75 sftacc--;
76 }
77
78 /*
79 * Find the conversion shift/mult pair which has the best
80 * accuracy and fits the maxsec conversion range:
81 */
82 for (sft = 32; sft > 0; sft--) {
83 tmp = (u64) to << sft;
84 tmp += from / 2;
85 do_div(tmp, from);
86 if ((tmp >> sftacc) == 0)
87 break;
88 }
89 *mult = tmp;
90 *shift = sft;
91}
92EXPORT_SYMBOL_GPL(clocks_calc_mult_shift);

140static void __clocksource_unstable(struct clocksource *cs)
141{
142 cs->flags &= ~(CLOCK_SOURCE_VALID_FOR_HRES | CLOCK_SOURCE_WATCHDOG);
143 cs->flags |= CLOCK_SOURCE_UNSTABLE;
144
145 if (cs->mark_unstable)
146 cs->mark_unstable(cs);
147
148 if (finished_booting)
149 schedule_work(&watchdog_work);
150}

152/**
153 * clocksource_mark_unstable - mark clocksource unstable via watchdog
154 * @cs: clocksource to be marked unstable
155 *
156 * This function is called instead of clocksource_change_rating from
157 * cpu hotplug code to avoid a deadlock between the clocksource mutex
158 * and the cpu hotplug mutex. It defers the update of the clocksource
159 * to the watchdog thread.
160 */
161void clocksource_mark_unstable(struct clocksource *cs)
162{
163 unsigned long flags;
164
165 spin_lock_irqsave(&watchdog_lock, flags);
166 if (!(cs->flags & CLOCK_SOURCE_UNSTABLE)) {
167 if (list_empty(&cs->wd_list))
168 list_add(&cs->wd_list, &watchdog_list);
169 __clocksource_unstable(cs);
170 }
171 spin_unlock_irqrestore(&watchdog_lock, flags);
172}

174static void clocksource_watchdog(unsigned long data)
175{
176 struct clocksource *cs;
177 u64 csnow, wdnow, cslast, wdlast, delta;
178 int64_t wd_nsec, cs_nsec;
179 int next_cpu, reset_pending;
180
181 spin_lock(&watchdog_lock);
182 if (!watchdog_running)
183 goto out;
184
185 reset_pending = atomic_read(&watchdog_reset_pending);
186
187 list_for_each_entry(cs, &watchdog_list, wd_list) {
188
189 /* Clocksource already marked unstable? */
190 if (cs->flags & CLOCK_SOURCE_UNSTABLE) {
191 if (finished_booting)
192 schedule_work(&watchdog_work);
193 continue;
194 }
195
196 local_irq_disable();
197 csnow = cs->read(cs);
198 wdnow = watchdog->read(watchdog);
199 local_irq_enable();
200
201 /* Clocksource initialized ? */
202 if (!(cs->flags & CLOCK_SOURCE_WATCHDOG) ||
203 atomic_read(&watchdog_reset_pending)) {
204 cs->flags |= CLOCK_SOURCE_WATCHDOG;
205 cs->wd_last = wdnow;
206 cs->cs_last = csnow;
207 continue;
208 }
209
210 delta = clocksource_delta(wdnow, cs->wd_last, watchdog->mask);
211 wd_nsec = clocksource_cyc2ns(delta, watchdog->mult,
212 watchdog->shift);
213
214 delta = clocksource_delta(csnow, cs->cs_last, cs->mask);
215 cs_nsec = clocksource_cyc2ns(delta, cs->mult, cs->shift);
216 wdlast = cs->wd_last; /* save these in case we print them */
217 cslast = cs->cs_last;
218 cs->cs_last = csnow;
219 cs->wd_last = wdnow;
220
221 if (atomic_read(&watchdog_reset_pending))
222 continue;
223
224 /* Check the deviation from the watchdog clocksource. */
225 if (abs(cs_nsec - wd_nsec) > WATCHDOG_THRESHOLD) {
226 pr_warn("timekeeping watchdog on CPU%d: Marking clocksource '%s' as unstable because the skew is too large:\n",
227 smp_processor_id(), cs->name);
228 pr_warn(" '%s' wd_now: %llx wd_last: %llx mask: %llx\n",
229 watchdog->name, wdnow, wdlast, watchdog->mask);
230 pr_warn(" '%s' cs_now: %llx cs_last: %llx mask: %llx\n",
231 cs->name, csnow, cslast, cs->mask);
232 __clocksource_unstable(cs);
233 continue;
234 }
235
236 if (cs == curr_clocksource && cs->tick_stable)
237 cs->tick_stable(cs);
238
239 if (!(cs->flags & CLOCK_SOURCE_VALID_FOR_HRES) &&
240 (cs->flags & CLOCK_SOURCE_IS_CONTINUOUS) &&
241 (watchdog->flags & CLOCK_SOURCE_IS_CONTINUOUS)) {
242 /* Mark it valid for high-res. */
243 cs->flags |= CLOCK_SOURCE_VALID_FOR_HRES;
244
245 /*
246 * clocksource_done_booting() will sort it if
247 * finished_booting is not set yet.
248 */
249 if (!finished_booting)
250 continue;
251
252 /*
253 * If this is not the current clocksource let
254 * the watchdog thread reselect it. Due to the
255 * change to high res this clocksource might
256 * be preferred now. If it is the current
257 * clocksource let the tick code know about
258 * that change.
259 */
260 if (cs != curr_clocksource) {
261 cs->flags |= CLOCK_SOURCE_RESELECT;
262 schedule_work(&watchdog_work);
263 } else {
264 tick_clock_notify();
265 }
266 }
267 }
268
269 /*
270 * We only clear the watchdog_reset_pending, when we did a
271 * full cycle through all clocksources.
272 */
273 if (reset_pending)
274 atomic_dec(&watchdog_reset_pending);
275
276 /*
277 * Cycle through CPUs to check if the CPUs stay synchronized
278 * to each other.
279 */
280 next_cpu = cpumask_next(raw_smp_processor_id(), cpu_online_mask);
281 if (next_cpu >= nr_cpu_ids)
282 next_cpu = cpumask_first(cpu_online_mask);
283 watchdog_timer.expires += WATCHDOG_INTERVAL;
284 add_timer_on(&watchdog_timer, next_cpu);
285out:
286 spin_unlock(&watchdog_lock);
287}

289static inline void clocksource_start_watchdog(void)
290{
291 if (watchdog_running || !watchdog || list_empty(&watchdog_list))
292 return;
293 init_timer(&watchdog_timer);
294 watchdog_timer.function = clocksource_watchdog;
295 watchdog_timer.expires = jiffies + WATCHDOG_INTERVAL;
296 add_timer_on(&watchdog_timer, cpumask_first(cpu_online_mask));
297 watchdog_running = 1;
298}
299
300static inline void clocksource_stop_watchdog(void)
301{
302 if (!watchdog_running || (watchdog && !list_empty(&watchdog_list)))
303 return;
304 del_timer(&watchdog_timer);
305 watchdog_running = 0;
306}

308static inline void clocksource_reset_watchdog(void)
309{
310 struct clocksource *cs;

312 list_for_each_entry(cs, &watchdog_list, wd_list)
313 cs->flags &= ~CLOCK_SOURCE_WATCHDOG;
314}

316static void clocksource_resume_watchdog(void)
317{
318 atomic_inc(&watchdog_reset_pending);

321static void clocksource_enqueue_watchdog(struct clocksource *cs)
322{
323 unsigned long flags;
324
325 INIT_LIST_HEAD(&cs->wd_list);
326
327 spin_lock_irqsave(&watchdog_lock, flags);
328 if (cs->flags & CLOCK_SOURCE_MUST_VERIFY) {
329 /* cs is a clocksource to be watched. */
330 list_add(&cs->wd_list, &watchdog_list);
331 cs->flags &= ~CLOCK_SOURCE_WATCHDOG;
332 } else {
333 /* cs is a watchdog. */
334 if (cs->flags & CLOCK_SOURCE_IS_CONTINUOUS)
335 cs->flags |= CLOCK_SOURCE_VALID_FOR_HRES;
336 }
337 spin_unlock_irqrestore(&watchdog_lock, flags);
338}

340static void clocksource_select_watchdog(bool fallback)
341{
342 struct clocksource *cs, *old_wd;
343 unsigned long flags;
344
345 spin_lock_irqsave(&watchdog_lock, flags);
346 /* save current watchdog */
347 old_wd = watchdog;
348 if (fallback)
349 watchdog = NULL;
350
351 list_for_each_entry(cs, &clocksource_list, list) {
352 /* cs is a clocksource to be watched. */
353 if (cs->flags & CLOCK_SOURCE_MUST_VERIFY)
354 continue;
355
356 /* Skip current if we were requested for a fallback. */
357 if (fallback && cs == old_wd)
358 continue;
359
360 /* Pick the best watchdog. */
361 if (!watchdog || cs->rating > watchdog->rating)
362 watchdog = cs;
363 }
364 /* If we failed to find a fallback restore the old one. */
365 if (!watchdog)
366 watchdog = old_wd;
367
368 /* If we changed the watchdog we need to reset cycles. */
369 if (watchdog != old_wd)
370 clocksource_reset_watchdog();
371
372 /* Check if the watchdog timer needs to be started. */
373 clocksource_start_watchdog();
374 spin_unlock_irqrestore(&watchdog_lock, flags);
375}

377static void clocksource_dequeue_watchdog(struct clocksource *cs)
378{
379 unsigned long flags;
380
381 spin_lock_irqsave(&watchdog_lock, flags);
382 if (cs != watchdog) {
383 if (cs->flags & CLOCK_SOURCE_MUST_VERIFY) {
384 /* cs is a watched clocksource. */
385 list_del_init(&cs->wd_list);
386 /* Check if the watchdog timer needs to be stopped. */
387 clocksource_stop_watchdog();
388 }
389 }
390 spin_unlock_irqrestore(&watchdog_lock, flags);
391}

393static int __clocksource_watchdog_kthread(void)
394{
395 struct clocksource *cs, *tmp;
396 unsigned long flags;
397 LIST_HEAD(unstable);
398 int select = 0;
399
400 spin_lock_irqsave(&watchdog_lock, flags);
401 list_for_each_entry_safe(cs, tmp, &watchdog_list, wd_list) {
402 if (cs->flags & CLOCK_SOURCE_UNSTABLE) {
403 list_del_init(&cs->wd_list);
404 list_add(&cs->wd_list, &unstable);
405 select = 1;
406 }
407 if (cs->flags & CLOCK_SOURCE_RESELECT) {
408 cs->flags &= ~CLOCK_SOURCE_RESELECT;
409 select = 1;
410 }
411 }
412 /* Check if the watchdog timer needs to be stopped. */
413 clocksource_stop_watchdog();
414 spin_unlock_irqrestore(&watchdog_lock, flags);
415
416 /* Needs to be done outside of watchdog lock */
417 list_for_each_entry_safe(cs, tmp, &unstable, wd_list) {
418 list_del_init(&cs->wd_list);
419 __clocksource_change_rating(cs, 0);
420 }
421 return select;
422}

424static int clocksource_watchdog_kthread(void *data)
425{
426 mutex_lock(&clocksource_mutex);
427 if (__clocksource_watchdog_kthread())
428 clocksource_select();
429 mutex_unlock(&clocksource_mutex);
430 return 0;
431}

493/**
494 * clocksource_max_adjustment- Returns max adjustment amount
495 * @cs: Pointer to clocksource
496 *
497 */
498static u32 clocksource_max_adjustment(struct clocksource *cs)
499{
500 u64 ret;
501 /*
502 * We won't try to correct for more than 11% adjustments (110,000 ppm),
503 */
504 ret = (u64)cs->mult * 11;
505 do_div(ret,100);
506 return (u32)ret;
507}

509/**
510 * clocks_calc_max_nsecs - Returns maximum nanoseconds that can be converted
511 * @mult: cycle to nanosecond multiplier
512 * @shift: cycle to nanosecond divisor (power of two)
513 * @maxadj: maximum adjustment value to mult (~11%)
514 * @mask: bitmask for two's complement subtraction of non 64 bit counters
515 * @max_cyc: maximum cycle value before potential overflow (does not include
516 * any safety margin)
517 *
518 * NOTE: This function includes a safety margin of 50%, in other words, we
519 * return half the number of nanoseconds the hardware counter can technically
520 * cover. This is done so that we can potentially detect problems caused by
521 * delayed timers or bad hardware, which might result in time intervals that
522 * are larger than what the math used can handle without overflows.
523 */
524u64 clocks_calc_max_nsecs(u32 mult, u32 shift, u32 maxadj, u64 mask, u64 *max_cyc)
525{
526 u64 max_nsecs, max_cycles;
527
528 /*
529 * Calculate the maximum number of cycles that we can pass to the
530 * cyc2ns() function without overflowing a 64-bit result.
531 */
532 max_cycles = ULLONG_MAX;
533 do_div(max_cycles, mult+maxadj);
534
535 /*
536 * The actual maximum number of cycles we can defer the clocksource is
537 * determined by the minimum of max_cycles and mask.
538 * Note: Here we subtract the maxadj to make sure we don't sleep for
539 * too long if there's a large negative adjustment.
540 */
541 max_cycles = min(max_cycles, mask);
542 max_nsecs = clocksource_cyc2ns(max_cycles, mult - maxadj, shift);
543
544 /* return the max_cycles value as well if requested */
545 if (max_cyc)
546 *max_cyc = max_cycles;
547
548 /* Return 50% of the actual maximum, so we can detect bad values */
549 max_nsecs >>= 1;
550
551 return max_nsecs;
552}

555 * clocksource_update_max_deferment - Updates the clocksource max_idle_ns & max_cycles

568static struct clocksource *clocksource_find_best(bool oneshot, bool skipcur)
569{
570 struct clocksource *cs;
571
572 if (!finished_booting || list_empty(&clocksource_list))
573 return NULL;
574
575 /*
576 * We pick the clocksource with the highest rating. If oneshot
577 * mode is active, we pick the highres valid clocksource with
578 * the best rating.
579 */
580 list_for_each_entry(cs, &clocksource_list, list) {
581 if (skipcur && cs == curr_clocksource)
582 continue;
583 if (oneshot && !(cs->flags & CLOCK_SOURCE_VALID_FOR_HRES))
584 continue;
585 return cs;
586 }
587 return NULL;
588}

590static void __clocksource_select(bool skipcur)
591{
592 bool oneshot = tick_oneshot_mode_active();
593 struct clocksource *best, *cs;
594
595 /* Find the best suitable clocksource */
596 best = clocksource_find_best(oneshot, skipcur);
597 if (!best)
598 return;
599
600 /* Check for the override clocksource. */
601 list_for_each_entry(cs, &clocksource_list, list) {
602 if (skipcur && cs == curr_clocksource)
603 continue;
604 if (strcmp(cs->name, override_name) != 0)
605 continue;
606 /*
607 * Check to make sure we don't switch to a non-highres
608 * capable clocksource if the tick code is in oneshot
609 * mode (highres or nohz)
610 */
611 if (!(cs->flags & CLOCK_SOURCE_VALID_FOR_HRES) && oneshot) {
612 /* Override clocksource cannot be used. */
613 if (cs->flags & CLOCK_SOURCE_UNSTABLE) {
614 pr_warn("Override clocksource %s is unstable and not HRT compatible - cannot switch while in HRT/NOHZ mode\n",
615 cs->name);
616 override_name[0] = 0;
617 } else {
618 /*
619 * The override cannot be currently verified.
620 * Deferring to let the watchdog check.
621 */
622 pr_info("Override clocksource %s is not currently HRT compatible - deferring\n",
623 cs->name);
624 }
625 } else
626 /* Override clocksource can be used. */
627 best = cs;
628 break;
629 }
630
631 if (curr_clocksource != best && !timekeeping_notify(best)) {
632 pr_info("Switched to clocksource %s\n", best->name);
633 curr_clocksource = best;
634 }
635}

661/*
662 * clocksource_done_booting - Called near the end of core bootup
663 *
664 * Hack to avoid lots of clocksource churn at boot time.
665 * We use fs_initcall because we want this to start before
666 * device_initcall but after subsys_initcall.

686static void clocksource_enqueue(struct clocksource *cs)

700/**
701 * __clocksource_update_freq_scale - Used update clocksource with new freq
702 * @cs: clocksource to be registered
703 * @scale: Scale factor multiplied against freq to get clocksource hz
704 * @freq: clocksource frequency (cycles per second) divided by scale
705 *
706 * This should only be called from the clocksource->enable() method.
707 *
708 * This *SHOULD NOT* be called directly! Please use the
709 * __clocksource_update_freq_hz() or __clocksource_update_freq_khz() helper
710 * functions.
711 */
712void __clocksource_update_freq_scale(struct clocksource *cs, u32 scale, u32 freq)
713{
714 u64 sec;
715
716 /*
717 * Default clocksources are *special* and self-define their mult/shift.
718 * But, you're not special, so you should specify a freq value.
719 */
720 if (freq) {
721 /*
722 * Calc the maximum number of seconds which we can run before
723 * wrapping around. For clocksources which have a mask > 32-bit
724 * we need to limit the max sleep time to have a good
725 * conversion precision. 10 minutes is still a reasonable
726 * amount. That results in a shift value of 24 for a
727 * clocksource with mask >= 40-bit and f >= 4GHz. That maps to
728 * ~ 0.06ppm granularity for NTP.
729 */
730 sec = cs->mask;
731 do_div(sec, freq);
732 do_div(sec, scale);
733 if (!sec)
734 sec = 1;
735 else if (sec > 600 && cs->mask > UINT_MAX)
736 sec = 600;
737
738 clocks_calc_mult_shift(&cs->mult, &cs->shift, freq,
739 NSEC_PER_SEC / scale, sec * scale);
740 }
741 /*
742 * Ensure clocksources that have large 'mult' values don't overflow
743 * when adjusted.
744 */
745 cs->maxadj = clocksource_max_adjustment(cs);
746 while (freq && ((cs->mult + cs->maxadj < cs->mult)
747 || (cs->mult - cs->maxadj > cs->mult))) {
748 cs->mult >>= 1;
749 cs->shift--;
750 cs->maxadj = clocksource_max_adjustment(cs);
751 }
752
753 /*
754 * Only warn for *special* clocksources that self-define
755 * their mult/shift values and don't specify a freq.
756 */
757 WARN_ONCE(cs->mult + cs->maxadj < cs->mult,
758 "timekeeping: Clocksource %s might overflow on 11%% adjustment\n",
759 cs->name);
760
761 clocksource_update_max_deferment(cs);
762
763 pr_info("%s: mask: 0x%llx max_cycles: 0x%llx, max_idle_ns: %lld ns\n",
764 cs->name, cs->mask, cs->max_cycles, cs->max_idle_ns);
765}
766EXPORT_SYMBOL_GPL(__clocksource_update_freq_scale);
767
768/**
769 * __clocksource_register_scale - Used to install new clocksources
770 * @cs: clocksource to be registered
771 * @scale: Scale factor multiplied against freq to get clocksource hz
772 * @freq: clocksource frequency (cycles per second) divided by scale
773 *
774 * Returns -EBUSY if registration fails, zero otherwise.
775 *
776 * This *SHOULD NOT* be called directly! Please use the
777 * clocksource_register_hz() or clocksource_register_khz helper functions.
778 */
779int __clocksource_register_scale(struct clocksource *cs, u32 scale, u32 freq)
780{
781
782 /* Initialize mult/shift and max_idle_ns */
783 __clocksource_update_freq_scale(cs, scale, freq);
784
785 /* Add clocksource to the clocksource list */
786 mutex_lock(&clocksource_mutex);
787 clocksource_enqueue(cs);
788 clocksource_enqueue_watchdog(cs);
789 clocksource_select();
790 clocksource_select_watchdog(false);
791 mutex_unlock(&clocksource_mutex);
792 return 0;
793}
794EXPORT_SYMBOL_GPL(__clocksource_register_scale);

818/*
819 * Unbind clocksource @cs. Called with clocksource_mutex held
820 */
821static int clocksource_unbind(struct clocksource *cs)
822{
823 if (clocksource_is_watchdog(cs)) {
824 /* Select and try to install a replacement watchdog. */
825 clocksource_select_watchdog(true);
826 if (clocksource_is_watchdog(cs))
827 return -EBUSY;
828 }
829
830 if (cs == curr_clocksource) {
831 /* Select and try to install a replacement clock source */
832 clocksource_select_fallback();
833 if (curr_clocksource == cs)
834 return -EBUSY;
835 }
836 clocksource_dequeue_watchdog(cs);
837 list_del_init(&cs->list);
838 return 0;
839}

842 * clocksource_unregister - remove a registered clocksource
843 * @cs: clocksource to be unregistered

846{

848
849 mutex_lock(&clocksource_mutex);
850 if (!list_empty(&cs->list))
851 ret = clocksource_unbind(cs);
852 mutex_unlock(&clocksource_mutex);
853 return ret;
854}

879ssize_t sysfs_get_uname(const char *buf, char *dst, size_t cnt)
880{
881 size_t ret = cnt;
882
883 /* strings from sysfs write are not 0 terminated! */
884 if (!cnt || cnt >= CS_NAME_LEN)
885 return -EINVAL;
