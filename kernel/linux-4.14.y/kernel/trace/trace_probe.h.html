71/* data_rloc: data relative location, compatible with u32 */
72#define make_data_rloc(len, roffs) \
73 (((u32)(len) << 16) | ((u32)(roffs) & 0xffff))
74#define get_rloc_len(dl) ((u32)(dl) >> 16)
75#define get_rloc_offs(dl) ((u32)(dl) & 0xffff)
76
77/*
78 * Convert data_rloc to data_loc:
79 * data_rloc stores the offset from data_rloc itself, but data_loc
80 * stores the offset from event entry.
81 */
82#define convert_rloc_to_loc(dl, offs) ((u32)(dl) + (offs))
83
84static nokprobe_inline void *get_rloc_data(u32 *dl)
85{
86 return (u8 *)dl + get_rloc_offs(*dl);
87}
88
89/* For data_loc conversion */
90static nokprobe_inline void *get_loc_data(u32 *dl, void *ent)
91{
92 return (u8 *)ent + get_rloc_offs(*dl);
93}

323/* Check the name is good for event/group/fields */
324static inline bool is_good_name(const char *name)
325{
326 if (!isalpha(*name) && *name != '_')
327 return false;
328 while (*++name != '\0') {
329 if (!isalpha(*name) && !isdigit(*name) && *name != '_')
330 return false;
331 }
332 return true;
333}

335static inline struct event_file_link *
336find_event_file_link(struct trace_probe *tp, struct trace_event_file *file)
337{
338 struct event_file_link *link;
339
340 list_for_each_entry(link, &tp->files, list)
341 if (link->file == file)
342 return link;
343
344 return NULL;
345}

365/* Sum up total data length for dynamic arraies (strings) */
366static nokprobe_inline int
367__get_data_size(struct trace_probe *tp, struct pt_regs *regs)
368{
369 int i, ret = 0;
370 u32 len;
371
372 for (i = 0; i < tp->nr_args; i++)
373 if (unlikely(tp->args[i].fetch_size.fn)) {
374 call_fetch(&tp->args[i].fetch_size, regs, &len);
375 ret += len;
376 }
377
378 return ret;
379}

381/* Store the value of each argument */
382static nokprobe_inline void
383store_trace_args(int ent_size, struct trace_probe *tp, struct pt_regs *regs,
384 u8 *data, int maxlen)
385{
386 int i;
387 u32 end = tp->size;
388 u32 *dl; /* Data (relative) location */
389
390 for (i = 0; i < tp->nr_args; i++) {
391 if (unlikely(tp->args[i].fetch_size.fn)) {
392 /*
393 * First, we set the relative location and
394 * maximum data length to *dl
395 */
396 dl = (u32 *)(data + tp->args[i].offset);
397 *dl = make_data_rloc(maxlen, end - tp->args[i].offset);
398 /* Then try to fetch string or dynamic array data */
399 call_fetch(&tp->args[i].fetch, regs, dl);
400 /* Reduce maximum length */
401 end += get_rloc_len(*dl);
402 maxlen -= get_rloc_len(*dl);
403 /* Trick here, convert data_rloc to data_loc */
404 *dl = convert_rloc_to_loc(*dl,
405 ent_size + tp->args[i].offset);
406 } else
407 /* Just fetching data normally */
408 call_fetch(&tp->args[i].fetch, regs,
409 data + tp->args[i].offset);
410 }
