98void blk_queue_congestion_threshold(struct request_queue *q)
99{
100 int nr;
101
102 nr = q->nr_requests - (q->nr_requests / 8) + 1;
103 if (nr > q->nr_requests)
104 nr = q->nr_requests;
105 q->nr_congestion_on = nr;
106
107 nr = q->nr_requests - (q->nr_requests / 8) - (q->nr_requests / 16) - 1;
108 if (nr < 1)
109 nr = 1;
110 q->nr_congestion_off = nr;
111}

155blk_status_t errno_to_blk_status(int errno)
156{
157 int i;
158
159 for (i = 0; i < ARRAY_SIZE(blk_errors); i++) {
160 if (blk_errors[i].errno == errno)
161 return (__force blk_status_t)i;
162 }
163
164 return BLK_STS_IOERR;
165}

191static void req_bio_endio(struct request *rq, struct bio *bio,
192 unsigned int nbytes, blk_status_t error)
193{
194 if (error)
195 bio->bi_status = error;
196
197 if (unlikely(rq->rq_flags & RQF_QUIET))
198 bio_set_flag(bio, BIO_QUIET);
199
200 bio_advance(bio, nbytes);
201
202 /* don't actually finish bio if it's part of flush sequence */
203 if (bio->bi_iter.bi_size == 0 && !(rq->rq_flags & RQF_FLUSH_SEQ))
204 bio_endio(bio);
205}

291/**
292 * blk_stop_queue - stop a queue
293 * @q: The &struct request_queue in question
294 *
295 * Description:
296 * The Linux block layer assumes that a block driver will consume all
297 * entries on the request queue when the request_fn strategy is called.
298 * Often this will not happen, because of hardware limitations (queue
299 * depth settings). If a device driver gets a 'queue full' response,
300 * or if it simply chooses not to queue more I/O at one point, it can
301 * call this function to prevent the request_fn from being called until
302 * the driver has signalled it's ready to go again. This happens by calling
303 * blk_start_queue() to restart queue operations.
304 **/
305void blk_stop_queue(struct request_queue *q)
306{
307 lockdep_assert_held(q->queue_lock);
308 WARN_ON_ONCE(q->mq_ops);
309
310 cancel_delayed_work(&q->delay_work);
311 queue_flag_set(QUEUE_FLAG_STOPPED, q);
312}
313EXPORT_SYMBOL(blk_stop_queue);

315/**
316 * blk_sync_queue - cancel any pending callbacks on a queue
317 * @q: the queue
318 *
319 * Description:
320 * The block layer may perform asynchronous callback activity
321 * on a queue, such as calling the unplug function after a timeout.
322 * A block device may call blk_sync_queue to ensure that any
323 * such activity is cancelled, thus allowing it to release resources
324 * that the callbacks might use. The caller must already have made sure
325 * that its ->make_request_fn will not re-add plugging prior to calling
326 * this function.
327 *
328 * This function does not cancel any asynchronous activity arising
329 * out of elevator or throttling code. That would require elevator_exit()
330 * and blkcg_exit_queue() to be called with queue lock initialized.
331 *
332 */
333void blk_sync_queue(struct request_queue *q)
334{
335 del_timer_sync(&q->timeout);
336 cancel_work_sync(&q->timeout_work);
337
338 if (q->mq_ops) {
339 struct blk_mq_hw_ctx *hctx;
340 int i;
341
342 cancel_delayed_work_sync(&q->requeue_work);
343 queue_for_each_hw_ctx(q, hctx, i)
344 cancel_delayed_work_sync(&hctx->run_work);
345 } else {
346 cancel_delayed_work_sync(&q->delay_work);
347 }
348}
349EXPORT_SYMBOL(blk_sync_queue);

351/**
352 * __blk_run_queue_uncond - run a queue whether or not it has been stopped
353 * @q: The queue to run
354 *
355 * Description:
356 * Invoke request handling on a queue if there are any pending requests.
357 * May be used to restart request handling after a request has completed.
358 * This variant runs the queue whether or not the queue has been
359 * stopped. Must be called with the queue lock held and interrupts
360 * disabled. See also @blk_run_queue.
361 */
362inline void __blk_run_queue_uncond(struct request_queue *q)
363{
364 lockdep_assert_held(q->queue_lock);
365 WARN_ON_ONCE(q->mq_ops);
366
367 if (unlikely(blk_queue_dead(q)))
368 return;
369
370 /*
371 * Some request_fn implementations, e.g. scsi_request_fn(), unlock
372 * the queue lock internally. As a result multiple threads may be
373 * running such a request function concurrently. Keep track of the
374 * number of active request_fn invocations such that blk_drain_queue()
375 * can wait until all these request_fn calls have finished.
376 */
377 q->request_fn_active++;
378 q->request_fn(q);
379 q->request_fn_active--;
380}
381EXPORT_SYMBOL_GPL(__blk_run_queue_uncond);

403 * blk_run_queue_async - run a single device queue in workqueue context
404 * @q: The queue to run
405 *
406 * Description:
407 * Tells kblockd to perform the equivalent of @blk_run_queue on behalf

451/**
452 * __blk_drain_queue - drain requests from request_queue
453 * @q: queue to drain
454 * @drain_all: whether to drain all requests or only the ones w/ ELVPRIV
455 *
456 * Drain requests from @q. If @drain_all is set, all requests are drained.
457 * If not, only ELVPRIV requests are drained. The caller is responsible
458 * for ensuring that no new requests which need to be drained are queued.
459 */
460static void __blk_drain_queue(struct request_queue *q, bool drain_all)
461 __releases(q->queue_lock)
462 __acquires(q->queue_lock)
463{
464 int i;
465
466 lockdep_assert_held(q->queue_lock);
467 WARN_ON_ONCE(q->mq_ops);
468
469 while (true) {
470 bool drain = false;
471
472 /*
473 * The caller might be trying to drain @q before its
474 * elevator is initialized.
475 */
476 if (q->elevator)
477 elv_drain_elevator(q);
478
479 blkcg_drain_queue(q);
480
481 /*
482 * This function might be called on a queue which failed
483 * driver init after queue creation or is not yet fully
484 * active yet. Some drivers (e.g. fd and loop) get unhappy
485 * in such cases. Kick queue iff dispatch queue has
486 * something on it and @q has request_fn set.
487 */
488 if (!list_empty(&q->queue_head) && q->request_fn)
489 __blk_run_queue(q);
490
491 drain |= q->nr_rqs_elvpriv;
492 drain |= q->request_fn_active;
493
494 /*
495 * Unfortunately, requests are queued at and tracked from
496 * multiple places and there's no single counter which can
497 * be drained. Check all the queues and counters.
498 */
499 if (drain_all) {
500 struct blk_flush_queue *fq = blk_get_flush_queue(q, NULL);
501 drain |= !list_empty(&q->queue_head);
502 for (i = 0; i < 2; i++) {
503 drain |= q->nr_rqs[i];
504 drain |= q->in_flight[i];
505 if (fq)
506 drain |= !list_empty(&fq->flush_queue[i]);
507 }
508 }
509
510 if (!drain)
511 break;
512
513 spin_unlock_irq(q->queue_lock);
514
515 msleep(10);
516
517 spin_lock_irq(q->queue_lock);
518 }
519
520 /*
521 * With queue marked dead, any woken up waiter will fail the
522 * allocation path, so the wakeup chaining is lost and we're
523 * left with hung waiters. We need to wake up those waiters.
524 */
525 if (q->request_fn) {
526 struct request_list *rl;
527
528 blk_queue_for_each_rl(rl, q)
529 for (i = 0; i < ARRAY_SIZE(rl->wait); i++)
530 wake_up_all(&rl->wait[i]);
531 }
532}

541/**
542 * blk_queue_bypass_start - enter queue bypass mode
543 * @q: queue of interest
544 *
545 * In bypass mode, only the dispatch FIFO queue of @q is used. This
546 * function makes @q enter bypass mode and drains all requests which were
547 * throttled or issued before. On return, it's guaranteed that no request
548 * is being throttled or has ELVPRIV set and blk_queue_bypass() %true
549 * inside queue or RCU read lock.
550 */
551void blk_queue_bypass_start(struct request_queue *q)
552{
553 WARN_ON_ONCE(q->mq_ops);
554
555 spin_lock_irq(q->queue_lock);
556 q->bypass_depth++;
557 queue_flag_set(QUEUE_FLAG_BYPASS, q);
558 spin_unlock_irq(q->queue_lock);
559
560 /*
561 * Queues start drained. Skip actual draining till init is
562 * complete. This avoids lenghty delays during queue init which
563 * can happen many times during boot.
564 */
565 if (blk_queue_init_done(q)) {
566 spin_lock_irq(q->queue_lock);
567 __blk_drain_queue(q, false);
568 spin_unlock_irq(q->queue_lock);
569
570 /* ensure blk_queue_bypass() is %true inside RCU read lock */
571 synchronize_rcu();
572 }
573}
574EXPORT_SYMBOL_GPL(blk_queue_bypass_start);
575
576/**
577 * blk_queue_bypass_end - leave queue bypass mode
578 * @q: queue of interest
579 *
580 * Leave bypass mode and restore the normal queueing behavior.
581 *
582 * Note: although blk_queue_bypass_start() is only called for blk-sq queues,
583 * this function is called for both blk-sq and blk-mq queues.
584 */
585void blk_queue_bypass_end(struct request_queue *q)
586{
587 spin_lock_irq(q->queue_lock);
588 if (!--q->bypass_depth)
589 queue_flag_clear(QUEUE_FLAG_BYPASS, q);
590 WARN_ON_ONCE(q->bypass_depth < 0);
591 spin_unlock_irq(q->queue_lock);
592}
593EXPORT_SYMBOL_GPL(blk_queue_bypass_end);

595void blk_set_queue_dying(struct request_queue *q)
596{
597 spin_lock_irq(q->queue_lock);
598 queue_flag_set(QUEUE_FLAG_DYING, q);
599 spin_unlock_irq(q->queue_lock);
600
601 /*
602 * When queue DYING flag is set, we need to block new req
603 * entering queue, so we call blk_freeze_queue_start() to
604 * prevent I/O from crossing blk_queue_enter().
605 */
606 blk_freeze_queue_start(q);
607
608 if (q->mq_ops)
609 blk_mq_wake_waiters(q);
610 else {
611 struct request_list *rl;
612
613 spin_lock_irq(q->queue_lock);
614 blk_queue_for_each_rl(rl, q) {
615 if (rl->rq_pool) {
616 wake_up_all(&rl->wait[BLK_RW_SYNC]);
617 wake_up_all(&rl->wait[BLK_RW_ASYNC]);
618 }
619 }
620 spin_unlock_irq(q->queue_lock);
621 }
622}

625/**
626 * blk_cleanup_queue - shutdown a request queue
627 * @q: request queue to shutdown
628 *
629 * Mark @q DYING, drain all pending requests, mark @q DEAD, destroy and
630 * put it. All future requests will be failed immediately with -ENODEV.
631 */
632void blk_cleanup_queue(struct request_queue *q)
633{
634 spinlock_t *lock = q->queue_lock;
635
636 /* mark @q DYING, no new request or merges will be allowed afterwards */
637 mutex_lock(&q->sysfs_lock);
638 blk_set_queue_dying(q);
639 spin_lock_irq(lock);
640
641 /*
642 * A dying queue is permanently in bypass mode till released. Note
643 * that, unlike blk_queue_bypass_start(), we aren't performing
644 * synchronize_rcu() after entering bypass mode to avoid the delay
645 * as some drivers create and destroy a lot of queues while
646 * probing. This is still safe because blk_release_queue() will be
647 * called only after the queue refcnt drops to zero and nothing,
648 * RCU or not, would be traversing the queue by then.
649 */
650 q->bypass_depth++;
651 queue_flag_set(QUEUE_FLAG_BYPASS, q);
652
653 queue_flag_set(QUEUE_FLAG_NOMERGES, q);
654 queue_flag_set(QUEUE_FLAG_NOXMERGES, q);
655 queue_flag_set(QUEUE_FLAG_DYING, q);
656 spin_unlock_irq(lock);
657 mutex_unlock(&q->sysfs_lock);
658
659 /*
660 * Drain all requests queued before DYING marking. Set DEAD flag to
661 * prevent that q->request_fn() gets invoked after draining finished.
662 */
663 blk_freeze_queue(q);
664 spin_lock_irq(lock);
665 queue_flag_set(QUEUE_FLAG_DEAD, q);
666 spin_unlock_irq(lock);
667
668 /*
669 * make sure all in-progress dispatch are completed because
670 * blk_freeze_queue() can only complete all requests, and
671 * dispatch may still be in-progress since we dispatch requests
672 * from more than one contexts
673 */
674 if (q->mq_ops)
675 blk_mq_quiesce_queue(q);
676
677 /* for synchronous bio-based driver finish in-flight integrity i/o */
678 blk_flush_integrity();
679
680 /* @q won't process any more request, flush async actions */
681 del_timer_sync(&q->backing_dev_info->laptop_mode_wb_timer);
682 blk_sync_queue(q);
683
684 if (q->mq_ops)
685 blk_mq_free_queue(q);
686 percpu_ref_exit(&q->q_usage_counter);
687
688 spin_lock_irq(lock);
689 if (q->queue_lock != &q->__queue_lock)

779int blk_queue_enter(struct request_queue *q, bool nowait)
780{
781 while (true) {
782
783 if (percpu_ref_tryget_live(&q->q_usage_counter))
784 return 0;
785
786 if (nowait)
787 return -EBUSY;
788
789 /*
790 * read pair of barrier in blk_freeze_queue_start(),
791 * we need to order reading __PERCPU_REF_DEAD flag of
792 * .q_usage_counter and reading .mq_freeze_depth or
793 * queue dying flag, otherwise the following wait may
794 * never return if the two reads are reordered.
795 */
796 smp_rmb();
797
798 wait_event(q->mq_freeze_wq,
799 !atomic_read(&q->mq_freeze_depth) ||
800 blk_queue_dying(q));
801 if (blk_queue_dying(q))
802 return -ENODEV;
803 }
804}

806void blk_queue_exit(struct request_queue *q)
807{
808 percpu_ref_put(&q->q_usage_counter);
809}

813 struct request_queue *q =
814 container_of(ref, struct request_queue, q_usage_counter);
815
816 wake_up_all(&q->mq_freeze_wq);
817}

826struct request_queue *blk_alloc_queue_node(gfp_t gfp_mask, int node_id)
827{
828 struct request_queue *q;
829
830 q = kmem_cache_alloc_node(blk_requestq_cachep,
831 gfp_mask | __GFP_ZERO, node_id);
832 if (!q)
833 return NULL;
834
835 q->id = ida_simple_get(&blk_queue_ida, 0, 0, gfp_mask);
836 if (q->id < 0)
837 goto fail_q;
838
839 q->bio_split = bioset_create(BIO_POOL_SIZE, 0, BIOSET_NEED_BVECS);
840 if (!q->bio_split)
841 goto fail_id;
842
843 q->backing_dev_info = bdi_alloc_node(gfp_mask, node_id);
844 if (!q->backing_dev_info)
845 goto fail_split;
846
847 q->stats = blk_alloc_queue_stats();
848 if (!q->stats)
849 goto fail_stats;
850
851 q->backing_dev_info->ra_pages =
852 (VM_MAX_READAHEAD * 1024) / PAGE_SIZE;
853 q->backing_dev_info->capabilities = BDI_CAP_CGROUP_WRITEBACK;
854 q->backing_dev_info->name = "block";
855 q->node = node_id;
856
857 setup_timer(&q->backing_dev_info->laptop_mode_wb_timer,
858 laptop_mode_timer_fn, (unsigned long) q);
859 setup_timer(&q->timeout, blk_rq_timed_out_timer, (unsigned long) q);
860 INIT_WORK(&q->timeout_work, NULL);
861 INIT_LIST_HEAD(&q->queue_head);
862 INIT_LIST_HEAD(&q->timeout_list);
863 INIT_LIST_HEAD(&q->icq_list);
864#ifdef CONFIG_BLK_CGROUP
865 INIT_LIST_HEAD(&q->blkg_list);
866#endif
867 INIT_DELAYED_WORK(&q->delay_work, blk_delay_work);
868
869 kobject_init(&q->kobj, &blk_queue_ktype);
870
871#ifdef CONFIG_BLK_DEV_IO_TRACE
872 mutex_init(&q->blk_trace_mutex);
873#endif
874 mutex_init(&q->sysfs_lock);
875 spin_lock_init(&q->__queue_lock);
876
877 /*
878 * By default initialize queue_lock to internal lock and driver can
879 * override it later if need be.
880 */
881 q->queue_lock = &q->__queue_lock;
882
883 /*
884 * A queue starts its life with bypass turned on to avoid
885 * unnecessary bypass on/off overhead and nasty surprises during
886 * init. The initial bypass will be finished when the queue is
887 * registered by blk_register_queue().
888 */
889 q->bypass_depth = 1;
890 __set_bit(QUEUE_FLAG_BYPASS, &q->queue_flags);
891
892 init_waitqueue_head(&q->mq_freeze_wq);
893
894 /*
895 * Init percpu_ref in atomic mode so that it's faster to shutdown.
896 * See blk_register_queue() for details.
897 */
898 if (percpu_ref_init(&q->q_usage_counter,
899 blk_queue_usage_counter_release,
900 PERCPU_REF_INIT_ATOMIC, GFP_KERNEL))
901 goto fail_bdi;
902
903 if (blkcg_init_queue(q))
904 goto fail_ref;
905
906 return q;
907
908fail_ref:
909 percpu_ref_exit(&q->q_usage_counter);
910fail_bdi:
911 blk_free_queue_stats(q->stats);
912fail_stats:
913 bdi_put(q->backing_dev_info);
914fail_split:
915 bioset_free(q->bio_split);
916fail_id:
917 ida_simple_remove(&blk_queue_ida, q->id);
918fail_q:
919 kmem_cache_free(blk_requestq_cachep, q);
920 return NULL;
921}

924/**
925 * blk_init_queue - prepare a request queue for use with a block device
926 * @rfn: The function to be called to process requests that have been
927 * placed on the queue.
928 * @lock: Request queue spin lock
929 *
930 * Description:
931 * If a block device wishes to use the standard request handling procedures,
932 * which sorts requests and coalesces adjacent requests, then it must
933 * call blk_init_queue(). The function @rfn will be called when there
934 * are requests on the queue that need to be processed. If the device
935 * supports plugging, then @rfn may not be called immediately when requests
936 * are available on the queue, but may be called at some time later instead.
937 * Plugged queues are generally unplugged when a buffer belonging to one
938 * of the requests on the queue is needed, or due to memory pressure.
939 *
940 * @rfn is not required, or even expected, to remove all requests off the
941 * queue, but only as many as it can handle at a time. If it does leave
942 * requests on the queue, it is responsible for arranging that the requests
943 * get dealt with eventually.
944 *
945 * The queue spin lock must be held while manipulating the requests on the
946 * request queue; this lock will be taken also from interrupt context, so irq
947 * disabling is needed for it.
948 *
949 * Function returns a pointer to the initialized request queue, or %NULL if
950 * it didn't succeed.
951 *
952 * Note:
953 * blk_init_queue() must be paired with a blk_cleanup_queue() call
954 * when the block device is deactivated (such as at module unload).
955 **/
956
957struct request_queue *blk_init_queue(request_fn_proc *rfn, spinlock_t *lock)
958{
959 return blk_init_queue_node(rfn, lock, NUMA_NO_NODE);
960}
961EXPORT_SYMBOL(blk_init_queue);
962
963struct request_queue *
964blk_init_queue_node(request_fn_proc *rfn, spinlock_t *lock, int node_id)
965{
966 struct request_queue *q;
967
968 q = blk_alloc_queue_node(GFP_KERNEL, node_id);
969 if (!q)
970 return NULL;
971
972 q->request_fn = rfn;
973 if (lock)
974 q->queue_lock = lock;
975 if (blk_init_allocated_queue(q) < 0) {
976 blk_cleanup_queue(q);
977 return NULL;
978 }
979
980 return q;
981}

1054/*
1055 * ioc_batching returns true if the ioc is a valid batching request and
1056 * should be given priority access to a request.
1057 */
1058static inline int ioc_batching(struct request_queue *q, struct io_context *ioc)
1059{
1060 if (!ioc)
1061 return 0;
1062
1063 /*
1064 * Make sure the process is able to allocate at least 1 request
1065 * even if the batch times out, otherwise we could theoretically
1066 * lose wakeups.
1067 */
1068 return ioc->nr_batch_requests == q->nr_batching ||
1069 (ioc->nr_batch_requests > 0
1070 && time_before(jiffies, ioc->last_waited + BLK_BATCH_TIME));
1071}
1072
1073/*
1074 * ioc_set_batching sets ioc to be a new "batcher" if it is not one. This
1075 * will cause the process to be a "batcher" on all queues in the system. This
1076 * is the behaviour we want though - once it gets a wakeup it should be given
1077 * a nice run.
1078 */
1079static void ioc_set_batching(struct request_queue *q, struct io_context *ioc)
1080{
1081 if (!ioc || ioc_batching(q, ioc))
1082 return;
1083
1084 ioc->nr_batch_requests = q->nr_batching;
1085 ioc->last_waited = jiffies;
1086}

1088static void __freed_request(struct request_list *rl, int sync)
1089{
1090 struct request_queue *q = rl->q;
1091
1092 if (rl->count[sync] < queue_congestion_off_threshold(q))
1093 blk_clear_congested(rl, sync);

1103/*
1104 * A request has just been released. Account for it, update the full and
1105 * congestion status, wake up any waiters. Called under q->queue_lock.
1106 */
1107static void freed_request(struct request_list *rl, bool sync,
1108 req_flags_t rq_flags)
1109{
1110 struct request_queue *q = rl->q;
1111
1112 q->nr_rqs[sync]--;
1113 rl->count[sync]--;
1114 if (rq_flags & RQF_ELVPRIV)
1115 q->nr_rqs_elvpriv--;
1116
1117 __freed_request(rl, sync);
1118
1119 if (unlikely(rl->starved[sync ^ 1]))
1120 __freed_request(rl, sync ^ 1);
1121}

1123int blk_update_nr_requests(struct request_queue *q, unsigned int nr)
1124{
1125 struct request_list *rl;
1126 int on_thresh, off_thresh;
1127
1128 WARN_ON_ONCE(q->mq_ops);
1129
1130 spin_lock_irq(q->queue_lock);
1131 q->nr_requests = nr;
1132 blk_queue_congestion_threshold(q);
1133 on_thresh = queue_congestion_on_threshold(q);
1134 off_thresh = queue_congestion_off_threshold(q);
1135
1136 blk_queue_for_each_rl(rl, q) {
1137 if (rl->count[BLK_RW_SYNC] >= on_thresh)
1138 blk_set_congested(rl, BLK_RW_SYNC);
1139 else if (rl->count[BLK_RW_SYNC] < off_thresh)
1140 blk_clear_congested(rl, BLK_RW_SYNC);
1141
1142 if (rl->count[BLK_RW_ASYNC] >= on_thresh)
1143 blk_set_congested(rl, BLK_RW_ASYNC);
1144 else if (rl->count[BLK_RW_ASYNC] < off_thresh)
1145 blk_clear_congested(rl, BLK_RW_ASYNC);
1146
1147 if (rl->count[BLK_RW_SYNC] >= q->nr_requests) {
1148 blk_set_rl_full(rl, BLK_RW_SYNC);
1149 } else {
1150 blk_clear_rl_full(rl, BLK_RW_SYNC);
1151 wake_up(&rl->wait[BLK_RW_SYNC]);
1152 }
1153
1154 if (rl->count[BLK_RW_ASYNC] >= q->nr_requests) {
1155 blk_set_rl_full(rl, BLK_RW_ASYNC);
1156 } else {
1157 blk_clear_rl_full(rl, BLK_RW_ASYNC);
1158 wake_up(&rl->wait[BLK_RW_ASYNC]);
1159 }
1160 }
1161
1162 spin_unlock_irq(q->queue_lock);
1163 return 0;
1164}

1166/**
1167 * __get_request - get a free request
1168 * @rl: request list to allocate from
1169 * @op: operation and flags
1170 * @bio: bio to allocate request for (can be %NULL)
1171 * @gfp_mask: allocation mask
1172 *
1173 * Get a free request from @q. This function may fail under memory
1174 * pressure or if @q is dead.
1175 *
1176 * Must be called with @q->queue_lock held and,
1177 * Returns ERR_PTR on failure, with @q->queue_lock held.
1178 * Returns request pointer on success, with @q->queue_lock *not held*.
1179 */
1180static struct request *__get_request(struct request_list *rl, unsigned int op,
1181 struct bio *bio, gfp_t gfp_mask)
1182{
1183 struct request_queue *q = rl->q;
1184 struct request *rq;
1185 struct elevator_type *et = q->elevator->type;
1186 struct io_context *ioc = rq_ioc(bio);
1187 struct io_cq *icq = NULL;
1188 const bool is_sync = op_is_sync(op);
1189 int may_queue;
1190 req_flags_t rq_flags = RQF_ALLOCED;
1191
1192 lockdep_assert_held(q->queue_lock);
1193
1194 if (unlikely(blk_queue_dying(q)))
1195 return ERR_PTR(-ENODEV);
1196
1197 may_queue = elv_may_queue(q, op);
1198 if (may_queue == ELV_MQUEUE_NO)
1199 goto rq_starved;
1200
1201 if (rl->count[is_sync]+1 >= queue_congestion_on_threshold(q)) {
1202 if (rl->count[is_sync]+1 >= q->nr_requests) {
1203 /*
1204 * The queue will fill after this allocation, so set
1205 * it as full, and mark this process as "batching".
1206 * This process will be allowed to complete a batch of
1207 * requests, others will be blocked.
1208 */
1209 if (!blk_rl_full(rl, is_sync)) {
1210 ioc_set_batching(q, ioc);
1211 blk_set_rl_full(rl, is_sync);
1212 } else {
1213 if (may_queue != ELV_MQUEUE_MUST
1214 && !ioc_batching(q, ioc)) {
1215 /*
1216 * The queue is full and the allocating
1217 * process is not a "batcher", and not
1218 * exempted by the IO scheduler
1219 */
1220 return ERR_PTR(-ENOMEM);
1221 }
1222 }
1223 }
1224 blk_set_congested(rl, is_sync);
1225 }
1226
1227 /*
1228 * Only allow batching queuers to allocate up to 50% over the defined
1229 * limit of requests, otherwise we could have thousands of requests
1230 * allocated with any setting of ->nr_requests
1231 */
1232 if (rl->count[is_sync] >= (3 * q->nr_requests / 2))
1233 return ERR_PTR(-ENOMEM);
1234
1235 q->nr_rqs[is_sync]++;
1236 rl->count[is_sync]++;
1237 rl->starved[is_sync] = 0;
1238
1239 /*
1240 * Decide whether the new request will be managed by elevator. If
1241 * so, mark @rq_flags and increment elvpriv. Non-zero elvpriv will
1242 * prevent the current elevator from being destroyed until the new
1243 * request is freed. This guarantees icq's won't be destroyed and
1244 * makes creating new ones safe.
1245 *
1246 * Flush requests do not use the elevator so skip initialization.
1247 * This allows a request to share the flush and elevator data.
1248 *
1249 * Also, lookup icq while holding queue_lock. If it doesn't exist,
1250 * it will be created after releasing queue_lock.
1251 */
1252 if (!op_is_flush(op) && !blk_queue_bypass(q)) {
1253 rq_flags |= RQF_ELVPRIV;
1254 q->nr_rqs_elvpriv++;
1255 if (et->icq_cache && ioc)
1256 icq = ioc_lookup_icq(ioc, q);
1257 }
1258
1259 if (blk_queue_io_stat(q))
1260 rq_flags |= RQF_IO_STAT;
1261 spin_unlock_irq(q->queue_lock);
1262
1263 /* allocate and init request */
1264 rq = mempool_alloc(rl->rq_pool, gfp_mask);
1265 if (!rq)
1266 goto fail_alloc;
1267
1268 blk_rq_init(q, rq);
1269 blk_rq_set_rl(rq, rl);
1270 rq->cmd_flags = op;
1271 rq->rq_flags = rq_flags;
1272
1273 /* init elvpriv */
1274 if (rq_flags & RQF_ELVPRIV) {
1275 if (unlikely(et->icq_cache && !icq)) {
1276 if (ioc)
1277 icq = ioc_create_icq(ioc, q, gfp_mask);
1278 if (!icq)
1279 goto fail_elvpriv;
1280 }
1281
1282 rq->elv.icq = icq;
1283 if (unlikely(elv_set_request(q, rq, bio, gfp_mask)))
1284 goto fail_elvpriv;
1285
1286 /* @rq->elv.icq holds io_context until @rq is freed */
1287 if (icq)
1288 get_io_context(icq->ioc);
1289 }
1290out:
1291 /*
1292 * ioc may be NULL here, and ioc_batching will be false. That's
1293 * OK, if the queue is under the request limit then requests need
1294 * not count toward the nr_batch_requests limit. There will always
1295 * be some limit enforced by BLK_BATCH_TIME.
1296 */
1297 if (ioc_batching(q, ioc))
1298 ioc->nr_batch_requests--;
1299
1300 trace_block_getrq(q, bio, op);
1301 return rq;
1302
1303fail_elvpriv:
1304 /*
1305 * elvpriv init failed. ioc, icq and elvpriv aren't mempool backed
1306 * and may fail indefinitely under memory pressure and thus
1307 * shouldn't stall IO. Treat this request as !elvpriv. This will
1308 * disturb iosched and blkcg but weird is bettern than dead.
1309 */
1310 printk_ratelimited(KERN_WARNING "%s: dev %s: request aux data allocation failed, iosched may be disturbed\n",
1311 __func__, dev_name(q->backing_dev_info->dev));
1312
1313 rq->rq_flags &= ~RQF_ELVPRIV;
1314 rq->elv.icq = NULL;
1315
1316 spin_lock_irq(q->queue_lock);
1317 q->nr_rqs_elvpriv--;
1318 spin_unlock_irq(q->queue_lock);
1319 goto out;
1320
1321fail_alloc:
1322 /*
1323 * Allocation failed presumably due to memory. Undo anything we
1324 * might have messed up.
1325 *
1326 * Allocating task should really be put onto the front of the wait
1327 * queue, but this is pretty rare.
1328 */
1329 spin_lock_irq(q->queue_lock);
1330 freed_request(rl, is_sync, rq_flags);
1331
1332 /*
1333 * in the very unlikely event that allocation failed and no
1334 * requests for this direction was pending, mark us starved so that
1335 * freeing of a request in the other direction will notice
1336 * us. another possible fix would be to split the rq mempool into
1337 * READ and WRITE
1338 */
1339rq_starved:
1340 if (unlikely(rl->count[is_sync] == 0))
1341 rl->starved[is_sync] = 1;
1342 return ERR_PTR(-ENOMEM);
1343}
1344
1345/**
1346 * get_request - get a free request
1347 * @q: request_queue to allocate request from
1348 * @op: operation and flags
1349 * @bio: bio to allocate request for (can be %NULL)
1350 * @gfp_mask: allocation mask
1351 *
1352 * Get a free request from @q. If %__GFP_DIRECT_RECLAIM is set in @gfp_mask,
1353 * this function keeps retrying under memory pressure and fails iff @q is dead.
1354 *
1355 * Must be called with @q->queue_lock held and,
1356 * Returns ERR_PTR on failure, with @q->queue_lock held.
1357 * Returns request pointer on success, with @q->queue_lock *not held*.
1358 */
1359static struct request *get_request(struct request_queue *q, unsigned int op,
1360 struct bio *bio, gfp_t gfp_mask)
1361{
1362 const bool is_sync = op_is_sync(op);
1363 DEFINE_WAIT(wait);
1364 struct request_list *rl;
1365 struct request *rq;
1366
1367 lockdep_assert_held(q->queue_lock);
1368 WARN_ON_ONCE(q->mq_ops);
1369
1370 rl = blk_get_rl(q, bio); /* transferred to @rq on success */
1371retry:
1372 rq = __get_request(rl, op, bio, gfp_mask);
1373 if (!IS_ERR(rq))
1374 return rq;
1375
1376 if (op & REQ_NOWAIT) {
1377 blk_put_rl(rl);
1378 return ERR_PTR(-EAGAIN);
1379 }
1380
1381 if (!gfpflags_allow_blocking(gfp_mask) || unlikely(blk_queue_dying(q))) {
1382 blk_put_rl(rl);
1383 return rq;
1384 }
1385
1386 /* wait on @rl and retry */
1387 prepare_to_wait_exclusive(&rl->wait[is_sync], &wait,
1388 TASK_UNINTERRUPTIBLE);
1389
1390 trace_block_sleeprq(q, bio, op);
1391
1392 spin_unlock_irq(q->queue_lock);
1393 io_schedule();
1394
1395 /*
1396 * After sleeping, we become a "batching" process and will be able
1397 * to allocate at least one request, and up to a big batch of them
1398 * for a small period time. See ioc_batching, ioc_set_batching
1399 */
1400 ioc_set_batching(q, current->io_context);
1401
1402 spin_lock_irq(q->queue_lock);
1403 finish_wait(&rl->wait[is_sync], &wait);
1404
1405 goto retry;
1406}

1408static struct request *blk_old_get_request(struct request_queue *q,
1409 unsigned int op, gfp_t gfp_mask)
1410{
1411 struct request *rq;
1412
1413 WARN_ON_ONCE(q->mq_ops);
1414
1415 /* create ioc upfront */
1416 create_io_context(gfp_mask, q->node);
1417
1418 spin_lock_irq(q->queue_lock);
1419 rq = get_request(q, op, NULL, gfp_mask);
1420 if (IS_ERR(rq)) {
1421 spin_unlock_irq(q->queue_lock);
1422 return rq;
1423 }
1424
1425 /* q->queue_lock is unlocked at this point */
1426 rq->__data_len = 0;
1427 rq->__sector = (sector_t) -1;
1428 rq->bio = rq->biotail = NULL;
1429 return rq;
1430}

1432struct request *blk_get_request(struct request_queue *q, unsigned int op,
1433 gfp_t gfp_mask)
1434{
1435 struct request *req;
1436
1437 if (q->mq_ops) {
1438 req = blk_mq_alloc_request(q, op,
1439 (gfp_mask & __GFP_DIRECT_RECLAIM) ?
1440 0 : BLK_MQ_REQ_NOWAIT);

1501/**
1502 * part_round_stats() - Round off the performance stats on a struct disk_stats.
1503 * @q: target block queue
1504 * @cpu: cpu number for stats access
1505 * @part: target partition
1506 *
1507 * The average IO queue length and utilisation statistics are maintained
1508 * by observing the current state of the queue length and the amount of
1509 * time it has been in this state for.
1510 *
1511 * Normally, that accounting is done on IO completion, but that can result
1512 * in more than a second's worth of IO being accounted for within any one
1513 * second, leading to >100% utilisation. To deal with that, we call this
1514 * function to do a round-off before returning the results when reading
1515 * /proc/diskstats. This accounts immediately for all queue usage up to
1516 * the current jiffies and restarts the counters again.
1517 */
1518void part_round_stats(struct request_queue *q, int cpu, struct hd_struct *part)
1519{
1520 struct hd_struct *part2 = NULL;
1521 unsigned long now = jiffies;
1522 unsigned int inflight[2];
1523 int stats = 0;
1524
1525 if (part->stamp != now)
1526 stats |= 1;
1527
1528 if (part->partno) {
1529 part2 = &part_to_disk(part)->part0;
1530 if (part2->stamp != now)
1531 stats |= 2;
1532 }
1533
1534 if (!stats)
1535 return;
1536
1537 part_in_flight(q, part, inflight);
1538
1539 if (stats & 2)
1540 part_round_stats_single(q, cpu, part2, now, inflight[1]);
1541 if (stats & 1)
1542 part_round_stats_single(q, cpu, part, now, inflight[0]);
1543}
1544EXPORT_SYMBOL_GPL(part_round_stats);

1683/**
1684 * blk_attempt_plug_merge - try to merge with %current's plugged list
1685 * @q: request_queue new bio is being queued at
1686 * @bio: new bio being queued
1687 * @request_count: out parameter for number of traversed plugged requests
1688 * @same_queue_rq: pointer to &struct request that gets filled in when
1689 * another request associated with @q is found on the plug list
1690 * (optional, may be %NULL)
1691 *
1692 * Determine whether @bio being queued on @q can be merged with a request
1693 * on %current's plugged list. Returns %true if merge was successful,
1694 * otherwise %false.
1695 *
1696 * Plugging coalesces IOs from the same issuer for the same purpose without
1697 * going through @q->queue_lock. As such it's more of an issuing mechanism
1698 * than scheduling, and the request, while may have elvpriv data, is not
1699 * added on the elevator at this point. In addition, we don't have
1700 * reliable access to the elevator outside queue lock. Only check basic
1701 * merging parameters without querying the elevator.
1702 *
1703 * Caller must ensure !blk_queue_nomerges(q) beforehand.
1704 */
1705bool blk_attempt_plug_merge(struct request_queue *q, struct bio *bio,
1706 unsigned int *request_count,
1707 struct request **same_queue_rq)
1708{
1709 struct blk_plug *plug;
1710 struct request *rq;
1711 struct list_head *plug_list;
1712
1713 plug = current->plug;
1714 if (!plug)
1715 return false;
1716 *request_count = 0;
1717
1718 if (q->mq_ops)
1719 plug_list = &plug->mq_list;
1720 else
1721 plug_list = &plug->list;
1722
1723 list_for_each_entry_reverse(rq, plug_list, queuelist) {
1724 bool merged = false;
1725
1726 if (rq->q == q) {
1727 (*request_count)++;
1728 /*
1729 * Only blk-mq multiple hardware queues case checks the
1730 * rq in the same queue, there should be only one such
1731 * rq in a queue
1732 **/
1733 if (same_queue_rq)
1734 *same_queue_rq = rq;
1735 }
1736
1737 if (rq->q != q || !blk_rq_merge_ok(rq, bio))
1738 continue;
1739
1740 switch (blk_try_merge(rq, bio)) {
1741 case ELEVATOR_BACK_MERGE:
1742 merged = bio_attempt_back_merge(q, rq, bio);
1743 break;
1744 case ELEVATOR_FRONT_MERGE:
1745 merged = bio_attempt_front_merge(q, rq, bio);
1746 break;
1747 case ELEVATOR_DISCARD_MERGE:
1748 merged = bio_attempt_discard_merge(q, rq, bio);
1749 break;
1750 default:
1751 break;
1752 }
1753
1754 if (merged)
1755 return true;
1756 }
1757
1758 return false;
1759}

1804static blk_qc_t blk_queue_bio(struct request_queue *q, struct bio *bio)
1805{
1806 struct blk_plug *plug;
1807 int where = ELEVATOR_INSERT_SORT;
1808 struct request *req, *free;
1809 unsigned int request_count = 0;
1810 unsigned int wb_acct;
1811
1812 /*
1813 * low level driver can indicate that it wants pages above a
1814 * certain limit bounced to low memory (ie for highmem, or even
1815 * ISA dma in theory)
1816 */
1817 blk_queue_bounce(q, &bio);
1818
1819 blk_queue_split(q, &bio);
1820
1821 if (!bio_integrity_prep(bio))
1822 return BLK_QC_T_NONE;
1823
1824 if (op_is_flush(bio->bi_opf)) {
1825 spin_lock_irq(q->queue_lock);
1826 where = ELEVATOR_INSERT_FLUSH;
1827 goto get_rq;
1828 }
1829
1830 /*
1831 * Check if we can merge with the plugged list before grabbing
1832 * any locks.
1833 */
1834 if (!blk_queue_nomerges(q)) {
1835 if (blk_attempt_plug_merge(q, bio, &request_count, NULL))
1836 return BLK_QC_T_NONE;
1837 } else
1838 request_count = blk_plug_queued_count(q);
1839
1840 spin_lock_irq(q->queue_lock);
1841
1842 switch (elv_merge(q, &req, bio)) {
1843 case ELEVATOR_BACK_MERGE:
1844 if (!bio_attempt_back_merge(q, req, bio))
1845 break;
1846 elv_bio_merged(q, req, bio);
1847 free = attempt_back_merge(q, req);
1848 if (free)
1849 __blk_put_request(q, free);
1850 else
1851 elv_merged_request(q, req, ELEVATOR_BACK_MERGE);
1852 goto out_unlock;
1853 case ELEVATOR_FRONT_MERGE:
1854 if (!bio_attempt_front_merge(q, req, bio))
1855 break;
1856 elv_bio_merged(q, req, bio);
1857 free = attempt_front_merge(q, req);
1858 if (free)
1859 __blk_put_request(q, free);
1860 else
1861 elv_merged_request(q, req, ELEVATOR_FRONT_MERGE);
1862 goto out_unlock;
1863 default:
1864 break;
1865 }
1866
1867get_rq:
1868 wb_acct = wbt_wait(q->rq_wb, bio, q->queue_lock);
1869
1870 /*
1871 * Grab a free request. This is might sleep but can not fail.
1872 * Returns with the queue unlocked.
1873 */
1874 req = get_request(q, bio->bi_opf, bio, GFP_NOIO);
1875 if (IS_ERR(req)) {
1876 __wbt_done(q->rq_wb, wb_acct);
1877 if (PTR_ERR(req) == -ENOMEM)
1878 bio->bi_status = BLK_STS_RESOURCE;
1879 else
1880 bio->bi_status = BLK_STS_IOERR;
1881 bio_endio(bio);
1882 goto out_unlock;
1883 }
1884
1885 wbt_track(&req->issue_stat, wb_acct);
1886
1887 /*
1888 * After dropping the lock and possibly sleeping here, our request
1889 * may now be mergeable after it had proven unmergeable (above).
1890 * We don't worry about that case for efficiency. It won't happen
1891 * often, and the elevators are able to handle it.
1892 */
1893 blk_init_request_from_bio(req, bio);
1894
1895 if (test_bit(QUEUE_FLAG_SAME_COMP, &q->queue_flags))
1896 req->cpu = raw_smp_processor_id();
1897
1898 plug = current->plug;
1899 if (plug) {
1900 /*
1901 * If this is the first request added after a plug, fire
1902 * of a plug trace.
1903 *
1904 * @request_count may become stale because of schedule
1905 * out, so check plug list again.
1906 */
1907 if (!request_count || list_empty(&plug->list))
1908 trace_block_plug(q);
1909 else {
1910 struct request *last = list_entry_rq(plug->list.prev);
1911 if (request_count >= BLK_MAX_REQUEST_COUNT ||
1912 blk_rq_bytes(last) >= BLK_PLUG_FLUSH_SIZE) {
1913 blk_flush_plug_list(plug, false);
1914 trace_block_plug(q);
1915 }
1916 }
1917 list_add_tail(&req->queuelist, &plug->list);
1918 blk_account_io_start(req, true);
1919 } else {
1920 spin_lock_irq(q->queue_lock);
1921 add_acct_request(q, req, where);
1922 __blk_run_queue(q);
1923out_unlock:
1924 spin_unlock_irq(q->queue_lock);
1925 }
1926
1927 return BLK_QC_T_NONE;
1928}

1976/*
1977 * Remap block n of partition p to block n+start(p) of the disk.
1978 */
1979static inline int blk_partition_remap(struct bio *bio)
1980{
1981 struct hd_struct *p;
1982 int ret = 0;
1983
1984 /*
1985 * Zone reset does not include bi_size so bio_sectors() is always 0.
1986 * Include a test for the reset op code and perform the remap if needed.
1987 */
1988 if (!bio->bi_partno ||
1989 (!bio_sectors(bio) && bio_op(bio) != REQ_OP_ZONE_RESET))
1990 return 0;
1991
1992 rcu_read_lock();
1993 p = __disk_get_part(bio->bi_disk, bio->bi_partno);
1994 if (likely(p && !should_fail_request(p, bio->bi_iter.bi_size))) {
1995 bio->bi_iter.bi_sector += p->start_sect;
1996 bio->bi_partno = 0;
1997 trace_block_bio_remap(bio->bi_disk->queue, bio, part_devt(p),
1998 bio->bi_iter.bi_sector - p->start_sect);
1999 } else {
2000 printk("%s: fail for partition %d\n", __func__, bio->bi_partno);
2001 ret = -EIO;
2002 }
2003 rcu_read_unlock();
2004
2005 return ret;
2006}

2037static noinline_for_stack bool
2038generic_make_request_checks(struct bio *bio)
2039{
2040 struct request_queue *q;
2041 int nr_sectors = bio_sectors(bio);
2042 blk_status_t status = BLK_STS_IOERR;
2043 char b[BDEVNAME_SIZE];
2044
2045 might_sleep();
2046
2047 if (bio_check_eod(bio, nr_sectors))
2048 goto end_io;
2049
2050 q = bio->bi_disk->queue;
2051 if (unlikely(!q)) {
2052 printk(KERN_ERR
2053 "generic_make_request: Trying to access "
2054 "nonexistent block-device %s (%Lu)\n",
2055 bio_devname(bio, b), (long long)bio->bi_iter.bi_sector);
2056 goto end_io;
2057 }
2058
2059 /*
2060 * For a REQ_NOWAIT based request, return -EOPNOTSUPP
2061 * if queue is not a request based queue.
2062 */
2063
2064 if ((bio->bi_opf & REQ_NOWAIT) && !queue_is_rq_based(q))
2065 goto not_supported;
2066
2067 if (should_fail_request(&bio->bi_disk->part0, bio->bi_iter.bi_size))
2068 goto end_io;
2069
2070 if (blk_partition_remap(bio))
2071 goto end_io;
2072
2073 if (bio_check_eod(bio, nr_sectors))
2074 goto end_io;
2075
2076 /*
2077 * Filter flush bio's early so that make_request based
2078 * drivers without flush support don't have to worry
2079 * about them.
2080 */
2081 if (op_is_flush(bio->bi_opf) &&
2082 !test_bit(QUEUE_FLAG_WC, &q->queue_flags)) {
2083 bio->bi_opf &= ~(REQ_PREFLUSH | REQ_FUA);
2084 if (!nr_sectors) {
2085 status = BLK_STS_OK;
2086 goto end_io;
2087 }
2088 }
2089
2090 switch (bio_op(bio)) {
2091 case REQ_OP_DISCARD:
2092 if (!blk_queue_discard(q))
2093 goto not_supported;
2094 break;
2095 case REQ_OP_SECURE_ERASE:
2096 if (!blk_queue_secure_erase(q))
2097 goto not_supported;
2098 break;
2099 case REQ_OP_WRITE_SAME:
2100 if (!q->limits.max_write_same_sectors)
2101 goto not_supported;
2102 break;
2103 case REQ_OP_ZONE_REPORT:
2104 case REQ_OP_ZONE_RESET:
2105 if (!blk_queue_is_zoned(q))
2106 goto not_supported;
2107 break;
2108 case REQ_OP_WRITE_ZEROES:
2109 if (!q->limits.max_write_zeroes_sectors)
2110 goto not_supported;
2111 break;
2112 default:
2113 break;
2114 }
2115
2116 /*
2117 * Various block parts want %current->io_context and lazy ioc
2118 * allocation ends up trading a lot of pain for a small amount of
2119 * memory. Just allocate it upfront. This may fail and block
2120 * layer knows how to live with it.
2121 */
2122 create_io_context(GFP_ATOMIC, q->node);
2123
2124 if (!blkcg_bio_issue_check(q, bio))
2125 return false;
2126
2127 if (!bio_flagged(bio, BIO_TRACE_COMPLETION)) {
2128 trace_block_bio_queue(q, bio);
2129 /* Now that enqueuing has been traced, we need to trace
2130 * completion as well.
2131 */
2132 bio_set_flag(bio, BIO_TRACE_COMPLETION);
2133 }
2134 return true;
2135
2136not_supported:
2137 status = BLK_STS_NOTSUPP;
2138end_io:
2139 bio->bi_status = status;
2140 bio_endio(bio);
2141 return false;
2142}

2144/**
2145 * generic_make_request - hand a buffer to its device driver for I/O
2146 * @bio: The bio describing the location in memory and on the device.
2147 *
2148 * generic_make_request() is used to make I/O requests of block
2149 * devices. It is passed a &struct bio, which describes the I/O that needs
2150 * to be done.
2151 *
2152 * generic_make_request() does not return any status. The
2153 * success/failure status of the request, along with notification of
2154 * completion, is delivered asynchronously through the bio->bi_end_io
2155 * function described (one day) else where.
2156 *
2157 * The caller of generic_make_request must make sure that bi_io_vec
2158 * are set to describe the memory buffer, and that bi_dev and bi_sector are
2159 * set to describe the device address, and the
2160 * bi_end_io and optionally bi_private are set to describe how
2161 * completion notification should be signaled.
2162 *
2163 * generic_make_request and the drivers it calls may use bi_next if this
2164 * bio happens to be merged with someone else, and may resubmit the bio to
2165 * a lower device by calling into generic_make_request recursively, which
2166 * means the bio should NOT be touched after the call to ->make_request_fn.
2167 */
2168blk_qc_t generic_make_request(struct bio *bio)
2169{
2170 /*
2171 * bio_list_on_stack[0] contains bios submitted by the current
2172 * make_request_fn.
2173 * bio_list_on_stack[1] contains bios that were submitted before
2174 * the current make_request_fn, but that haven't been processed
2175 * yet.
2176 */
2177 struct bio_list bio_list_on_stack[2];
2178 blk_qc_t ret = BLK_QC_T_NONE;
2179
2180 if (!generic_make_request_checks(bio))
2181 goto out;
2182
2183 /*
2184 * We only want one ->make_request_fn to be active at a time, else
2185 * stack usage with stacked devices could be a problem. So use
2186 * current->bio_list to keep a list of requests submited by a
2187 * make_request_fn function. current->bio_list is also used as a
2188 * flag to say if generic_make_request is currently active in this
2189 * task or not. If it is NULL, then no make_request is active. If
2190 * it is non-NULL, then a make_request is active, and new requests
2191 * should be added at the tail
2192 */
2193 if (current->bio_list) {
2194 bio_list_add(&current->bio_list[0], bio);
2195 goto out;
2196 }
2197
2198 /* following loop may be a bit non-obvious, and so deserves some
2199 * explanation.
2200 * Before entering the loop, bio->bi_next is NULL (as all callers
2201 * ensure that) so we have a list with a single bio.
2202 * We pretend that we have just taken it off a longer list, so
2203 * we assign bio_list to a pointer to the bio_list_on_stack,
2204 * thus initialising the bio_list of new bios to be
2205 * added. ->make_request() may indeed add some more bios
2206 * through a recursive call to generic_make_request. If it
2207 * did, we find a non-NULL value in bio_list and re-enter the loop
2208 * from the top. In this case we really did just take the bio
2209 * of the top of the list (no pretending) and so remove it from
2210 * bio_list, and call into ->make_request() again.
2211 */
2212 BUG_ON(bio->bi_next);
2213 bio_list_init(&bio_list_on_stack[0]);
2214 current->bio_list = bio_list_on_stack;
2215 do {
2216 struct request_queue *q = bio->bi_disk->queue;
2217
2218 if (likely(blk_queue_enter(q, bio->bi_opf & REQ_NOWAIT) == 0)) {
2219 struct bio_list lower, same;
2220
2221 /* Create a fresh bio_list for all subordinate requests */
2222 bio_list_on_stack[1] = bio_list_on_stack[0];
2223 bio_list_init(&bio_list_on_stack[0]);
2224 ret = q->make_request_fn(q, bio);
2225
2226 blk_queue_exit(q);
2227
2228 /* sort new bios into those for a lower level
2229 * and those for the same level
2230 */
2231 bio_list_init(&lower);
2232 bio_list_init(&same);
2233 while ((bio = bio_list_pop(&bio_list_on_stack[0])) != NULL)
2234 if (q == bio->bi_disk->queue)
2235 bio_list_add(&same, bio);
2236 else
2237 bio_list_add(&lower, bio);
2238 /* now assemble so we handle the lowest level first */
2239 bio_list_merge(&bio_list_on_stack[0], &lower);
2240 bio_list_merge(&bio_list_on_stack[0], &same);
2241 bio_list_merge(&bio_list_on_stack[0], &bio_list_on_stack[1]);
2242 } else {
2243 if (unlikely(!blk_queue_dying(q) &&
2244 (bio->bi_opf & REQ_NOWAIT)))
2245 bio_wouldblock_error(bio);
2246 else
2247 bio_io_error(bio);
2248 }
2249 bio = bio_list_pop(&bio_list_on_stack[0]);
2250 } while (bio);
2251 current->bio_list = NULL; /* deactivate */
2252
2253out:
2254 return ret;
2255}
2256EXPORT_SYMBOL(generic_make_request);

2396 * blk_rq_err_bytes - determine number of bytes till the next failure boundary
2397 * @rq: request to examine
2398 *
2399 * Description:
2400 * A request could be merge of IOs which require different failure
2401 * handling. This function determines the number of bytes which
2402 * can be failed from the beginning of the request without
2403 * crossing into area which need to be retried further.
2404 *
2405 * Return:
2406 * The number of bytes to fail.
2407 */
2408unsigned int blk_rq_err_bytes(const struct request *rq)
2409{
2410 unsigned int ff = rq->cmd_flags & REQ_FAILFAST_MASK;
2411 unsigned int bytes = 0;
2412 struct bio *bio;
2413
2414 if (!(rq->rq_flags & RQF_MIXED_MERGE))
2415 return blk_rq_bytes(rq);
2416
2417 /*
2418 * Currently the only 'mixing' which can happen is between
2419 * different fastfail types. We can safely fail portions
2420 * which have all the failfast bits that the first one has -
2421 * the ones which are at least as eager to fail as the first
2422 * one.
2423 */

2425 if ((bio->bi_opf & ff) != ff)
2426 break;
2427 bytes += bio->bi_iter.bi_size;
2428 }
2429
2430 /* this could lead to infinite loop */
2431 BUG_ON(blk_rq_bytes(rq) && !bytes);

2534/**
2535 * blk_peek_request - peek at the top of a request queue
2536 * @q: request queue to peek at
2537 *
2538 * Description:
2539 * Return the request at the top of @q. The returned request
2540 * should be started using blk_start_request() before LLD starts
2541 * processing it.
2542 *
2543 * Return:
2544 * Pointer to the request at the top of @q if available. Null
2545 * otherwise.
2546 */
2547struct request *blk_peek_request(struct request_queue *q)
2548{
2549 struct request *rq;
2550 int ret;
2551
2552 lockdep_assert_held(q->queue_lock);
2553 WARN_ON_ONCE(q->mq_ops);
2554
2555 while ((rq = __elv_next_request(q)) != NULL) {
2556
2557 rq = blk_pm_peek_request(q, rq);
2558 if (!rq)
2559 break;
2560
2561 if (!(rq->rq_flags & RQF_STARTED)) {
2562 /*
2563 * This is the first time the device driver
2564 * sees this request (possibly after
2565 * requeueing). Notify IO scheduler.
2566 */
2567 if (rq->rq_flags & RQF_SORTED)
2568 elv_activate_rq(q, rq);
2569
2570 /*
2571 * just mark as started even if we don't start
2572 * it, a request that has been delayed should
2573 * not be passed by new incoming requests
2574 */
2575 rq->rq_flags |= RQF_STARTED;
2576 trace_block_rq_issue(q, rq);
2577 }
2578
2579 if (!q->boundary_rq || q->boundary_rq == rq) {
2580 q->end_sector = rq_end_sector(rq);
2581 q->boundary_rq = NULL;
2582 }
2583
2584 if (rq->rq_flags & RQF_DONTPREP)
2585 break;
2586
2587 if (q->dma_drain_size && blk_rq_bytes(rq)) {
2588 /*
2589 * make sure space for the drain appears we
2590 * know we can do this because max_hw_segments
2591 * has been adjusted to be one fewer than the
2592 * device can handle
2593 */
2594 rq->nr_phys_segments++;
2595 }
2596
2597 if (!q->prep_rq_fn)
2598 break;
2599
2600 ret = q->prep_rq_fn(q, rq);
2601 if (ret == BLKPREP_OK) {
2602 break;
2603 } else if (ret == BLKPREP_DEFER) {
2604 /*
2605 * the request may have been (partially) prepped.
2606 * we need to keep this request in the front to
2607 * avoid resource deadlock. RQF_STARTED will
2608 * prevent other fs requests from passing this one.
2609 */
2610 if (q->dma_drain_size && blk_rq_bytes(rq) &&
2611 !(rq->rq_flags & RQF_DONTPREP)) {
2612 /*
2613 * remove the space for the drain we added
2614 * so that we don't add it again
2615 */
2616 --rq->nr_phys_segments;
2617 }
2618
2619 rq = NULL;
2620 break;
2621 } else if (ret == BLKPREP_KILL || ret == BLKPREP_INVALID) {
2622 rq->rq_flags |= RQF_QUIET;
2623 /*
2624 * Mark this request as started so we don't trigger
2625 * any debug logic in the end I/O path.
2626 */
2627 blk_start_request(rq);
2628 __blk_end_request_all(rq, ret == BLKPREP_INVALID ?
2629 BLK_STS_TARGET : BLK_STS_IOERR);
2630 } else {
2631 printk(KERN_ERR "%s: bad return=%d\n", __func__, ret);
2632 break;
2633 }
2634 }
2635
2636 return rq;
2637}

2712/**
2713 * blk_update_request - Special helper function for request stacking drivers
2714 * @req: the request being processed
2715 * @error: block status code
2716 * @nr_bytes: number of bytes to complete @req
2717 *
2718 * Description:
2719 * Ends I/O on a number of bytes attached to @req, but doesn't complete
2720 * the request structure even if @req doesn't have leftover.
2721 * If @req has leftover, sets it up for the next range of segments.
2722 *
2723 * This special helper function is only for request stacking drivers
2724 * (e.g. request-based dm) so that they can handle partial completion.
2725 * Actual device drivers should use blk_end_request instead.
2726 *
2727 * Passing the result of blk_rq_bytes() as @nr_bytes guarantees
2728 * %false return from this function.
2729 *
2730 * Return:
2731 * %false - this request doesn't have any more data
2732 * %true - this request has more data
2733 **/
2734bool blk_update_request(struct request *req, blk_status_t error,
2735 unsigned int nr_bytes)
2736{
2737 int total_bytes;
2738
2739 trace_block_rq_complete(req, blk_status_to_errno(error), nr_bytes);
2740
2741 if (!req->bio)
2742 return false;
2743
2744 if (unlikely(error && !blk_rq_is_passthrough(req) &&
2745 !(req->rq_flags & RQF_QUIET)))
2746 print_req_error(req, error);
2747
2748 blk_account_io_completion(req, nr_bytes);
2749
2750 total_bytes = 0;
2751 while (req->bio) {
2752 struct bio *bio = req->bio;
2753 unsigned bio_bytes = min(bio->bi_iter.bi_size, nr_bytes);
2754
2755 if (bio_bytes == bio->bi_iter.bi_size)
2756 req->bio = bio->bi_next;
2757
2758 /* Completion has already been traced */
2759 bio_clear_flag(bio, BIO_TRACE_COMPLETION);
2760 req_bio_endio(req, bio, bio_bytes, error);
2761
2762 total_bytes += bio_bytes;
2763 nr_bytes -= bio_bytes;
2764
2765 if (!nr_bytes)
2766 break;
2767 }
2768
2769 /*
2770 * completely done
2771 */
2772 if (!req->bio) {
2773 /*
2774 * Reset counters so that the request stacking driver
2775 * can find how many bytes remain in the request
2776 * later.
2777 */
2778 req->__data_len = 0;
2779 return false;
2780 }
2781
2782 req->__data_len -= total_bytes;
2783
2784 /* update sector only for requests with clear definition of sector */
2785 if (!blk_rq_is_passthrough(req))
2786 req->__sector += total_bytes >> 9;
2787
2788 /* mixed attributes always follow the first bio */
2789 if (req->rq_flags & RQF_MIXED_MERGE) {
2790 req->cmd_flags &= ~REQ_FAILFAST_MASK;
2791 req->cmd_flags |= req->bio->bi_opf & REQ_FAILFAST_MASK;
2792 }
2793
2794 if (!(req->rq_flags & RQF_SPECIAL_PAYLOAD)) {
2795 /*
2796 * If total number of sectors is less than the first segment
2797 * size, something has gone terribly wrong.
2798 */
2799 if (blk_rq_bytes(req) < blk_rq_cur_bytes(req)) {
2800 blk_dump_rq_flags(req, "request botched");
2801 req->__data_len = blk_rq_cur_bytes(req);
2802 }
2803
2804 /* recalculate the number of segments */
2805 blk_recalc_rq_segments(req);
2806 }
2807
2808 return true;
2809}

3378
3379 /*
3380 * Save and disable interrupts here, to avoid doing it for every
3381 * queue lock we have to take.
3382 */
3383 local_irq_save(flags);
3384 while (!list_empty(&list)) {
3385 rq = list_entry_rq(list.next);
3386 list_del_init(&rq->queuelist);
3387 BUG_ON(!rq->q);
3388 if (rq->q != q) {
3389 /*
3390 * This drops the queue lock
3391 */
3392 if (q)
3393 queue_unplugged(q, depth, from_schedule);
3394 q = rq->q;
3395 depth = 0;
3396 spin_lock(q->queue_lock);
3397 }
3398
3399 /*
3400 * Short-circuit if @q is dead
3401 */
3402 if (unlikely(blk_queue_dying(q))) {
3403 __blk_end_request_all(rq, BLK_STS_IOERR);
3404 continue;
3405 }
3406
3407 /*
3408 * rq is already accounted, so use raw insert
3409 */
3410 if (op_is_flush(rq->cmd_flags))
3411 __elv_add_request(q, rq, ELEVATOR_INSERT_FLUSH);
3412 else
3413 __elv_add_request(q, rq, ELEVATOR_INSERT_SORT_MERGE);
3414
3415 depth++;
3416 }
3417
3418 /*
3419 * This drops the queue lock
3420 */
3421 if (q)
3422 queue_unplugged(q, depth, from_schedule);
3423
3424 local_irq_restore(flags);
3425}
