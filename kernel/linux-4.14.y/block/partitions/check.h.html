6/*
7 * add_gd_partition adds a partitions details to the devices partition
8 * description.
9 */
10struct parsed_partitions {
11 struct block_device *bdev;
12 char name[BDEVNAME_SIZE];
13 struct {
14 sector_t from;
15 sector_t size;
16 int flags;
17 bool has_info;
18 struct partition_meta_info info;
19 } *parts;
20 int next;
21 int limit;
22 bool access_beyond_eod;
23 char *pp_buf;
24};
25
26void free_partitions(struct parsed_partitions *state);
27
28struct parsed_partitions *
29check_partition(struct gendisk *, struct block_device *);
30
31static inline void *read_part_sector(struct parsed_partitions *state,
32 sector_t n, Sector *p)
33{
34 if (n >= get_capacity(state->bdev->bd_disk)) {
35 state->access_beyond_eod = true;
36 return NULL;
37 }
38 return read_dev_sector(state->bdev, n, p);
39}
40
41static inline void
42put_partition(struct parsed_partitions *p, int n, sector_t from, sector_t size)
43{
44 if (n < p->limit) {
45 char tmp[1 + BDEVNAME_SIZE + 10 + 1];
46
47 p->parts[n].from = from;
48 p->parts[n].size = size;
49 snprintf(tmp, sizeof(tmp), " %s%d", p->name, n);
50 strlcat(p->pp_buf, tmp, PAGE_SIZE);
51 }
52}
