38static inline bool
39blk_mq_sched_bio_merge(struct request_queue *q, struct bio *bio)
40{
41 if (blk_queue_nomerges(q) || !bio_mergeable(bio))
42 return false;
43
44 return __blk_mq_sched_bio_merge(q, bio);
45}
46
47static inline bool
48blk_mq_sched_allow_merge(struct request_queue *q, struct request *rq,
49 struct bio *bio)
50{
51 struct elevator_queue *e = q->elevator;
52
53 if (e && e->type->ops.mq.allow_merge)
54 return e->type->ops.mq.allow_merge(q, rq, bio);
55
56 return true;
57}
