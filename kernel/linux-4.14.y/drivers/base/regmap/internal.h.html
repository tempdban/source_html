261static inline const char *regmap_name(const struct regmap *map)
262{
263 if (map->dev)
264 return dev_name(map->dev);
265
266 return map->name;
267}
268
269static inline unsigned int regmap_get_offset(const struct regmap *map,
270 unsigned int index)
271{
272 if (map->reg_stride_order >= 0)
273 return index << map->reg_stride_order;
274 else
275 return index * map->reg_stride;
276}
277
278static inline unsigned int regcache_get_index_by_order(const struct regmap *map,
279 unsigned int reg)
280{
281 return reg >> map->reg_stride_order;
282}
