16/*
17 * Keytable for MK-F12 IR remote provided together with Pixelview
18 * Ultra Pro Remote Controller. Uses NEC extended format.
19 */
20static struct rc_map_table pixelview_mk12[] = {
21 { 0x866b03, KEY_TUNER }, /* Timeshift */
22 { 0x866b1e, KEY_POWER2 }, /* power */
23
24 { 0x866b01, KEY_1 },
25 { 0x866b0b, KEY_2 },
26 { 0x866b1b, KEY_3 },
27 { 0x866b05, KEY_4 },
28 { 0x866b09, KEY_5 },
29 { 0x866b15, KEY_6 },
30 { 0x866b06, KEY_7 },
31 { 0x866b0a, KEY_8 },
32 { 0x866b12, KEY_9 },
33 { 0x866b02, KEY_0 },
34
35 { 0x866b13, KEY_AGAIN }, /* loop */
36 { 0x866b10, KEY_DIGITS }, /* +100 */
37
38 { 0x866b00, KEY_VIDEO }, /* source */
39 { 0x866b18, KEY_MUTE }, /* mute */
40 { 0x866b19, KEY_CAMERA }, /* snapshot */
41 { 0x866b1a, KEY_SEARCH }, /* scan */
42
43 { 0x866b16, KEY_CHANNELUP }, /* chn + */
44 { 0x866b14, KEY_CHANNELDOWN }, /* chn - */
45 { 0x866b1f, KEY_VOLUMEUP }, /* vol + */
46 { 0x866b17, KEY_VOLUMEDOWN }, /* vol - */
47 { 0x866b1c, KEY_ZOOM }, /* zoom */
48
49 { 0x866b04, KEY_REWIND },
50 { 0x866b0e, KEY_RECORD },
51 { 0x866b0c, KEY_FORWARD },
52
53 { 0x866b1d, KEY_STOP },
54 { 0x866b08, KEY_PLAY },
55 { 0x866b0f, KEY_PAUSE },
56
57 { 0x866b0d, KEY_TV },
58 { 0x866b07, KEY_RADIO }, /* FM */
59};
60
61static struct rc_map_list pixelview_map = {
62 .map = {
63 .scan = pixelview_mk12,
64 .size = ARRAY_SIZE(pixelview_mk12),
65 .rc_proto = RC_PROTO_NEC,
66 .name = RC_MAP_PIXELVIEW_MK12,
67 }
68};
69
70static int __init init_rc_map_pixelview(void)
71{
72 return rc_map_register(&pixelview_map);
73}
74
75static void __exit exit_rc_map_pixelview(void)
76{
77 rc_map_unregister(&pixelview_map);
78}
79
80module_init(init_rc_map_pixelview)
81module_exit(exit_rc_map_pixelview)
