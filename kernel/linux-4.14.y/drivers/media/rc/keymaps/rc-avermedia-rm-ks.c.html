21#include <media/rc-map.h>
22#include <linux/module.h>
23
24/* Initial keytable is from Jose Alberto Reguero <jareguero@telefonica.net>
25 and Felipe Morales Moreno <felipe.morales.moreno@gmail.com> */
26/* Keytable fixed by Philippe Valembois <lephilousophe@users.sourceforge.net> */
27static struct rc_map_table avermedia_rm_ks[] = {
28 { 0x0501, KEY_POWER2 }, /* Power (RED POWER BUTTON) */
29 { 0x0502, KEY_CHANNELUP }, /* Channel+ */
30 { 0x0503, KEY_CHANNELDOWN }, /* Channel- */
31 { 0x0504, KEY_VOLUMEUP }, /* Volume+ */
32 { 0x0505, KEY_VOLUMEDOWN }, /* Volume- */
33 { 0x0506, KEY_MUTE }, /* Mute */
34 { 0x0507, KEY_AGAIN }, /* Recall */
35 { 0x0508, KEY_VIDEO }, /* Source */
36 { 0x0509, KEY_1 }, /* 1 */
37 { 0x050a, KEY_2 }, /* 2 */
38 { 0x050b, KEY_3 }, /* 3 */
39 { 0x050c, KEY_4 }, /* 4 */
40 { 0x050d, KEY_5 }, /* 5 */
41 { 0x050e, KEY_6 }, /* 6 */
42 { 0x050f, KEY_7 }, /* 7 */
43 { 0x0510, KEY_8 }, /* 8 */
44 { 0x0511, KEY_9 }, /* 9 */
45 { 0x0512, KEY_0 }, /* 0 */
46 { 0x0513, KEY_AUDIO }, /* Audio */
47 { 0x0515, KEY_EPG }, /* EPG */
48 { 0x0516, KEY_PLAYPAUSE }, /* Play/Pause */
49 { 0x0517, KEY_RECORD }, /* Record */
50 { 0x0518, KEY_STOP }, /* Stop */
51 { 0x051c, KEY_BACK }, /* << */
52 { 0x051d, KEY_FORWARD }, /* >> */
53 { 0x054d, KEY_INFO }, /* Display information */
54 { 0x0556, KEY_ZOOM }, /* Fullscreen */
55};
56
57static struct rc_map_list avermedia_rm_ks_map = {
58 .map = {
59 .scan = avermedia_rm_ks,
60 .size = ARRAY_SIZE(avermedia_rm_ks),
61 .rc_proto = RC_PROTO_NEC,
62 .name = RC_MAP_AVERMEDIA_RM_KS,
63 }
64};
65
66static int __init init_rc_map_avermedia_rm_ks(void)
67{
68 return rc_map_register(&avermedia_rm_ks_map);
69}
70
71static void __exit exit_rc_map_avermedia_rm_ks(void)
72{
73 rc_map_unregister(&avermedia_rm_ks_map);
74}
75
76module_init(init_rc_map_avermedia_rm_ks)
77module_exit(exit_rc_map_avermedia_rm_ks)
