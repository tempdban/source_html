25 { 0x0000, KEY_MUTE }, /* [symbol speaker] */

27 { 0x0002, KEY_POWER2 }, /* TV [power button] */
28 { 0x0003, KEY_2 },

31 { 0x0006, KEY_6 },
32 { 0x0007, KEY_7 },
33 { 0x0008, KEY_8 },
34 { 0x0009, KEY_NUMERIC_STAR }, /* [*] */
35 { 0x000a, KEY_0 },
36 { 0x000b, KEY_NUMERIC_POUND }, /* [#] */
37 { 0x000c, KEY_RIGHT }, /* [right arrow] */
38 { 0x000d, KEY_HOMEPAGE }, /* [symbol home] Start */
39 { 0x000e, KEY_RED }, /* [red] Videos */
40 { 0x0010, KEY_POWER }, /* PC [power button] */

49 { 0x0019, KEY_BLUE }, /* [blue] MyTV */

73 { 0x005a, KEY_PREVIOUS }, /* REPLAY */

75
76static struct rc_map_list digitalnow_tinytwin_map = {
77 .map = {
78 .scan = digitalnow_tinytwin,
79 .size = ARRAY_SIZE(digitalnow_tinytwin),
80 .rc_proto = RC_PROTO_NEC,
81 .name = RC_MAP_DIGITALNOW_TINYTWIN,
82 }
83};
84
85static int __init init_rc_map_digitalnow_tinytwin(void)
86{
87 return rc_map_register(&digitalnow_tinytwin_map);
88}
89
90static void __exit exit_rc_map_digitalnow_tinytwin(void)
91{
92 rc_map_unregister(&digitalnow_tinytwin_map);
93}
94
95module_init(init_rc_map_digitalnow_tinytwin)
96module_exit(exit_rc_map_digitalnow_tinytwin)
