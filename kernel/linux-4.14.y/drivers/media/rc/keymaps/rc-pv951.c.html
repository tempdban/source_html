56static struct rc_map_list pv951_map = {
57 .map = {
58 .scan = pv951,
59 .size = ARRAY_SIZE(pv951),
60 .rc_proto = RC_PROTO_UNKNOWN, /* Legacy IR type */
61 .name = RC_MAP_PV951,
62 }
63};

65static int __init init_rc_map_pv951(void)
66{
67 return rc_map_register(&pv951_map);
68}
69
70static void __exit exit_rc_map_pv951(void)
71{
72 rc_map_unregister(&pv951_map);
73}

75module_init(init_rc_map_pv951)
