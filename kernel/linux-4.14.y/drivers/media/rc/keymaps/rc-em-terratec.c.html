47static struct rc_map_list em_terratec_map = {
48 .map = {
49 .scan = em_terratec,
50 .size = ARRAY_SIZE(em_terratec),
51 .rc_proto = RC_PROTO_UNKNOWN, /* Legacy IR type */
52 .name = RC_MAP_EM_TERRATEC,
53 }
54};
55
56static int __init init_rc_map_em_terratec(void)
57{
58 return rc_map_register(&em_terratec_map);
59}
60
61static void __exit exit_rc_map_em_terratec(void)
62{
63 rc_map_unregister(&em_terratec_map);
64}
65
66module_init(init_rc_map_em_terratec)
67module_exit(exit_rc_map_em_terratec)
