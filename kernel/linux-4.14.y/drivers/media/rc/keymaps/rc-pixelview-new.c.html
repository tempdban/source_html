61static struct rc_map_list pixelview_new_map = {
62 .map = {
63 .scan = pixelview_new,
64 .size = ARRAY_SIZE(pixelview_new),
65 .rc_proto = RC_PROTO_UNKNOWN, /* Legacy IR type */
66 .name = RC_MAP_PIXELVIEW_NEW,

70static int __init init_rc_map_pixelview_new(void)
71{
72 return rc_map_register(&pixelview_new_map);
73}
74
75static void __exit exit_rc_map_pixelview_new(void)
76{
77 rc_map_unregister(&pixelview_new_map);
78}
