84static struct rc_map_list snapstream_firefly_map = {
85 .map = {
86 .scan = snapstream_firefly,
87 .size = ARRAY_SIZE(snapstream_firefly),
88 .rc_proto = RC_PROTO_OTHER,
89 .name = RC_MAP_SNAPSTREAM_FIREFLY,
90 }
91};

93static int __init init_rc_map_snapstream_firefly(void)
94{
95 return rc_map_register(&snapstream_firefly_map);
96}
97
98static void __exit exit_rc_map_snapstream_firefly(void)
99{
100 rc_map_unregister(&snapstream_firefly_map);
101}
