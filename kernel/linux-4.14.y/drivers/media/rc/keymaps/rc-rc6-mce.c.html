97static struct rc_map_list rc6_mce_map = {
98 .map = {

106static int __init init_rc_map_rc6_mce(void)
107{
108 return rc_map_register(&rc6_mce_map);
109}
110
111static void __exit exit_rc_map_rc6_mce(void)
112{
113 rc_map_unregister(&rc6_mce_map);
114}
