21#include <media/rc-map.h>
22#include <linux/module.h>
23
24/* MSI DIGIVOX mini III */
25/* Uses NEC extended 0x61d6. */
26/* This remote seems to be same as rc-kworld-315u.c. Anyhow, add new remote
27 since rc-kworld-315u.c lacks NEC extended address byte. */
28static struct rc_map_table msi_digivox_iii[] = {
29 { 0x61d601, KEY_VIDEO }, /* Source */
30 { 0x61d602, KEY_3 },
31 { 0x61d603, KEY_POWER }, /* ShutDown */
32 { 0x61d604, KEY_1 },
33 { 0x61d605, KEY_5 },
34 { 0x61d606, KEY_6 },
35 { 0x61d607, KEY_CHANNELDOWN }, /* CH- */
36 { 0x61d608, KEY_2 },
37 { 0x61d609, KEY_CHANNELUP }, /* CH+ */
38 { 0x61d60a, KEY_9 },
39 { 0x61d60b, KEY_ZOOM }, /* Zoom */
40 { 0x61d60c, KEY_7 },
41 { 0x61d60d, KEY_8 },
42 { 0x61d60e, KEY_VOLUMEUP }, /* Vol+ */
43 { 0x61d60f, KEY_4 },
44 { 0x61d610, KEY_ESC }, /* [back up arrow] */
45 { 0x61d611, KEY_0 },
46 { 0x61d612, KEY_OK }, /* [enter arrow] */
47 { 0x61d613, KEY_VOLUMEDOWN }, /* Vol- */
48 { 0x61d614, KEY_RECORD }, /* Rec */
49 { 0x61d615, KEY_STOP }, /* Stop */
50 { 0x61d616, KEY_PLAY }, /* Play */
51 { 0x61d617, KEY_MUTE }, /* Mute */
52 { 0x61d618, KEY_UP },
53 { 0x61d619, KEY_DOWN },
54 { 0x61d61a, KEY_LEFT },
55 { 0x61d61b, KEY_RIGHT },
56 { 0x61d61c, KEY_RED },
57 { 0x61d61d, KEY_GREEN },
58 { 0x61d61e, KEY_YELLOW },
59 { 0x61d61f, KEY_BLUE },
60 { 0x61d643, KEY_POWER2 }, /* [red power button] */
61};
62
63static struct rc_map_list msi_digivox_iii_map = {
64 .map = {
65 .scan = msi_digivox_iii,
66 .size = ARRAY_SIZE(msi_digivox_iii),
67 .rc_proto = RC_PROTO_NEC,
68 .name = RC_MAP_MSI_DIGIVOX_III,
69 }
70};
71
72static int __init init_rc_map_msi_digivox_iii(void)
73{
74 return rc_map_register(&msi_digivox_iii_map);
75}
76
77static void __exit exit_rc_map_msi_digivox_iii(void)
78{
79 rc_map_unregister(&msi_digivox_iii_map);
80}
81
82module_init(init_rc_map_msi_digivox_iii)
83module_exit(exit_rc_map_msi_digivox_iii)
