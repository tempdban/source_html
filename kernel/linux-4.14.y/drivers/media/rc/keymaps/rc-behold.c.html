24 * the button labels (several variants when appropriate)

119static struct rc_map_list behold_map = {
120 .map = {
121 .scan = behold,
122 .size = ARRAY_SIZE(behold),
123 .rc_proto = RC_PROTO_NEC,
124 .name = RC_MAP_BEHOLD,
125 }
126};
127
128static int __init init_rc_map_behold(void)
129{
130 return rc_map_register(&behold_map);
131}
132
133static void __exit exit_rc_map_behold(void)
134{
135 rc_map_unregister(&behold_map);
136}
137
138module_init(init_rc_map_behold)
139module_exit(exit_rc_map_behold)
