26#include <stdarg.h>
27#include <linux/seq_file.h>
28#include <drm/drmP.h>
29#include <drm/drm_print.h>
30
31void __drm_printfn_seq_file(struct drm_printer *p, struct va_format *vaf)
32{
33 seq_printf(p->arg, "%pV", vaf);
34}
35EXPORT_SYMBOL(__drm_printfn_seq_file);
36
37void __drm_printfn_info(struct drm_printer *p, struct va_format *vaf)
38{
39 dev_info(p->arg, "[" DRM_NAME "] %pV", vaf);
40}
41EXPORT_SYMBOL(__drm_printfn_info);
42
43void __drm_printfn_debug(struct drm_printer *p, struct va_format *vaf)
44{
45 pr_debug("%s %pV", p->prefix, vaf);
46}
47EXPORT_SYMBOL(__drm_printfn_debug);
48
49/**
50 * drm_printf - print to a &drm_printer stream
51 * @p: the &drm_printer
52 * @f: format string
53 */
54void drm_printf(struct drm_printer *p, const char *f, ...)
55{
56 struct va_format vaf;
57 va_list args;
58
59 va_start(args, f);
60 vaf.fmt = f;
61 vaf.va = &args;
62 p->printfn(p, &vaf);
63 va_end(args);
64}
65EXPORT_SYMBOL(drm_printf);
