61static int
62qxl_pci_probe(struct pci_dev *pdev, const struct pci_device_id *ent)
63{
64 struct qxl_device *qdev;
65 int ret;
66
67 if (pdev->revision < 4) {
68 DRM_ERROR("qxl too old, doesn't support client_monitors_config,"
69 " use xf86-video-qxl in user mode");
70 return -EINVAL; /* TODO: ENODEV ? */
71 }
72
73 qdev = kzalloc(sizeof(struct qxl_device), GFP_KERNEL);
74 if (!qdev)
75 return -ENOMEM;
76
77 ret = pci_enable_device(pdev);
78 if (ret)
79 goto free_dev;
80
81 ret = qxl_device_init(qdev, &qxl_driver, pdev);
82 if (ret)
83 goto disable_pci;
84
85 ret = qxl_modeset_init(qdev);
86 if (ret)
87 goto unload;
88
89 drm_kms_helper_poll_init(&qdev->ddev);
90
91 /* Complete initialization. */
92 ret = drm_dev_register(&qdev->ddev, ent->driver_data);
93 if (ret)
94 goto modeset_cleanup;
95
96 return 0;
97
98modeset_cleanup:
99 qxl_modeset_fini(qdev);
100unload:
101 qxl_device_fini(qdev);
102disable_pci:
103 pci_disable_device(pdev);
104free_dev:
105 kfree(qdev);
106 return ret;
107}

109static void
110qxl_pci_remove(struct pci_dev *pdev)
111{
112 struct drm_device *dev = pci_get_drvdata(pdev);
113 struct qxl_device *qdev = dev->dev_private;
114
115 drm_dev_unregister(dev);
116
117 qxl_modeset_fini(qdev);
118 qxl_device_fini(qdev);
119
120 dev->dev_private = NULL;
121 kfree(qdev);
122 drm_dev_unref(dev);
123}

135static int qxl_drm_freeze(struct drm_device *dev)
136{
137 struct pci_dev *pdev = dev->pdev;
138 struct qxl_device *qdev = dev->dev_private;
139 struct drm_crtc *crtc;
140
141 drm_kms_helper_poll_disable(dev);
142
143 console_lock();
144 qxl_fbdev_set_suspend(qdev, 1);
145 console_unlock();
146
147 /* unpin the front buffers */
148 list_for_each_entry(crtc, &dev->mode_config.crtc_list, head) {
149 const struct drm_crtc_helper_funcs *crtc_funcs = crtc->helper_private;
150 if (crtc->enabled)
151 (*crtc_funcs->disable)(crtc);
152 }
153
154 qxl_destroy_monitors_object(qdev);
155 qxl_surf_evict(qdev);
156 qxl_vram_evict(qdev);
157
158 while (!qxl_check_idle(qdev->command_ring));
159 while (!qxl_check_idle(qdev->release_ring))
160 qxl_queue_garbage_collect(qdev, 1);
161
162 pci_save_state(pdev);
163
164 return 0;
165}

167static int qxl_drm_resume(struct drm_device *dev, bool thaw)
168{
169 struct qxl_device *qdev = dev->dev_private;
170
171 qdev->ram_header->int_mask = QXL_INTERRUPT_MASK;
172 if (!thaw) {
173 qxl_reinit_memslots(qdev);
174 qxl_ring_init_hdr(qdev->release_ring);
175 }
176
177 qxl_create_monitors_object(qdev);
178 drm_helper_resume_force_mode(dev);
179
180 console_lock();
181 qxl_fbdev_set_suspend(qdev, 0);
182 console_unlock();
183
184 drm_kms_helper_poll_enable(dev);
185 return 0;
186}

188static int qxl_pm_suspend(struct device *dev)
189{
190 struct pci_dev *pdev = to_pci_dev(dev);
191 struct drm_device *drm_dev = pci_get_drvdata(pdev);
192 int error;
193
194 error = qxl_drm_freeze(drm_dev);
195 if (error)
196 return error;
197
198 pci_disable_device(pdev);
199 pci_set_power_state(pdev, PCI_D3hot);
200 return 0;
201}
202
203static int qxl_pm_resume(struct device *dev)
204{
205 struct pci_dev *pdev = to_pci_dev(dev);
206 struct drm_device *drm_dev = pci_get_drvdata(pdev);
207
208 pci_set_power_state(pdev, PCI_D0);
209 pci_restore_state(pdev);
210 if (pci_enable_device(pdev)) {
211 return -EIO;
212 }
213
214 return qxl_drm_resume(drm_dev, false);
215}

217static int qxl_pm_thaw(struct device *dev)
218{
219 struct pci_dev *pdev = to_pci_dev(dev);
220 struct drm_device *drm_dev = pci_get_drvdata(pdev);
221
222 return qxl_drm_resume(drm_dev, true);
223}

233static int qxl_pm_restore(struct device *dev)
234{
235 struct pci_dev *pdev = to_pci_dev(dev);
236 struct drm_device *drm_dev = pci_get_drvdata(pdev);
237 struct qxl_device *qdev = drm_dev->dev_private;
238
239 qxl_io_reset(qdev);
240 return qxl_drm_resume(drm_dev, false);
241}

294static int __init qxl_init(void)
295{
296 if (vgacon_text_force() && qxl_modeset == -1)
297 return -EINVAL;
298
299 if (qxl_modeset == 0)
300 return -EINVAL;
301 qxl_driver.num_ioctls = qxl_max_ioctls;
302 return pci_register_driver(&qxl_pci_driver);
303}
304
305static void __exit qxl_exit(void)
306{
307 pci_unregister_driver(&qxl_pci_driver);
308}
