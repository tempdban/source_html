59static int qxl_ttm_global_init(struct qxl_device *qdev)
60{
61 struct drm_global_reference *global_ref;
62 int r;
63
64 qdev->mman.mem_global_referenced = false;
65 global_ref = &qdev->mman.mem_global_ref;
66 global_ref->global_type = DRM_GLOBAL_TTM_MEM;
67 global_ref->size = sizeof(struct ttm_mem_global);
68 global_ref->init = &qxl_ttm_mem_global_init;
69 global_ref->release = &qxl_ttm_mem_global_release;
70
71 r = drm_global_item_ref(global_ref);
72 if (r != 0) {
73 DRM_ERROR("Failed setting up TTM memory accounting "
74 "subsystem.\n");
75 return r;
76 }
77
78 qdev->mman.bo_global_ref.mem_glob =
79 qdev->mman.mem_global_ref.object;
80 global_ref = &qdev->mman.bo_global_ref.ref;
81 global_ref->global_type = DRM_GLOBAL_TTM_BO;
82 global_ref->size = sizeof(struct ttm_bo_global);
83 global_ref->init = &ttm_bo_global_init;
84 global_ref->release = &ttm_bo_global_release;
85 r = drm_global_item_ref(global_ref);
86 if (r != 0) {
87 DRM_ERROR("Failed setting up TTM BO subsystem.\n");
88 drm_global_item_unref(&qdev->mman.mem_global_ref);
89 return r;
90 }
91
92 qdev->mman.mem_global_referenced = true;
93 return 0;
94}
95
96static void qxl_ttm_global_fini(struct qxl_device *qdev)
97{
98 if (qdev->mman.mem_global_referenced) {
99 drm_global_item_unref(&qdev->mman.bo_global_ref.ref);
100 drm_global_item_unref(&qdev->mman.mem_global_ref);
101 qdev->mman.mem_global_referenced = false;
102 }
103}

120int qxl_mmap(struct file *filp, struct vm_area_struct *vma)
121{
122 struct drm_file *file_priv;
123 struct qxl_device *qdev;
124 int r;
125
126 if (unlikely(vma->vm_pgoff < DRM_FILE_PAGE_OFFSET)) {
127 pr_info("%s: vma->vm_pgoff (%ld) < DRM_FILE_PAGE_OFFSET\n",
128 __func__, vma->vm_pgoff);
129 return -EINVAL;
130 }
131
132 file_priv = filp->private_data;
133 qdev = file_priv->minor->dev->dev_private;
134 if (qdev == NULL) {
135 DRM_ERROR(
136 "filp->private_data->minor->dev->dev_private == NULL\n");
137 return -EINVAL;
138 }
139 QXL_INFO(qdev, "%s: filp->private_data = 0x%p, vma->vm_pgoff = %lx\n",
140 __func__, filp->private_data, vma->vm_pgoff);
141
142 r = ttm_bo_mmap(filp, vma, &qdev->mman.bdev);
143 if (unlikely(r != 0))
144 return r;
145 if (unlikely(ttm_vm_ops == NULL)) {
146 ttm_vm_ops = vma->vm_ops;
147 qxl_ttm_vm_ops = *ttm_vm_ops;
148 qxl_ttm_vm_ops.fault = &qxl_ttm_fault;
149 }
150 vma->vm_ops = &qxl_ttm_vm_ops;
151 return 0;
152}

159static int qxl_init_mem_type(struct ttm_bo_device *bdev, uint32_t type,
160 struct ttm_mem_type_manager *man)
161{
162 switch (type) {
163 case TTM_PL_SYSTEM:
164 /* System memory */
165 man->flags = TTM_MEMTYPE_FLAG_MAPPABLE;
166 man->available_caching = TTM_PL_MASK_CACHING;
167 man->default_caching = TTM_PL_FLAG_CACHED;
168 break;
169 case TTM_PL_VRAM:
170 case TTM_PL_PRIV:
171 /* "On-card" video ram */
172 man->func = &ttm_bo_manager_func;
173 man->gpu_offset = 0;
174 man->flags = TTM_MEMTYPE_FLAG_FIXED |
175 TTM_MEMTYPE_FLAG_MAPPABLE;
176 man->available_caching = TTM_PL_MASK_CACHING;
177 man->default_caching = TTM_PL_FLAG_CACHED;
178 break;
179 default:
180 DRM_ERROR("Unsupported memory type %u\n", (unsigned)type);
181 return -EINVAL;
182 }
183 return 0;
184}

186static void qxl_evict_flags(struct ttm_buffer_object *bo,
187 struct ttm_placement *placement)
188{
189 struct qxl_bo *qbo;
190 static const struct ttm_place placements = {
191 .fpfn = 0,
192 .lpfn = 0,
193 .flags = TTM_PL_MASK_CACHING | TTM_PL_FLAG_SYSTEM
194 };
195
196 if (!qxl_ttm_bo_is_qxl_bo(bo)) {
197 placement->placement = &placements;
198 placement->busy_placement = &placements;
199 placement->num_placement = 1;
200 placement->num_busy_placement = 1;
201 return;
202 }
203 qbo = to_qxl_bo(bo);
204 qxl_ttm_placement_from_domain(qbo, QXL_GEM_DOMAIN_CPU, false);
205 *placement = qbo->placement;
206}

216static int qxl_ttm_io_mem_reserve(struct ttm_bo_device *bdev,
217 struct ttm_mem_reg *mem)
218{
219 struct ttm_mem_type_manager *man = &bdev->man[mem->mem_type];
220 struct qxl_device *qdev = qxl_get_qdev(bdev);
221
222 mem->bus.addr = NULL;
223 mem->bus.offset = 0;
224 mem->bus.size = mem->num_pages << PAGE_SHIFT;
225 mem->bus.base = 0;
226 mem->bus.is_iomem = false;
227 if (!(man->flags & TTM_MEMTYPE_FLAG_MAPPABLE))
228 return -EINVAL;
229 switch (mem->mem_type) {
230 case TTM_PL_SYSTEM:
231 /* system memory */
232 return 0;
233 case TTM_PL_VRAM:
234 mem->bus.is_iomem = true;
235 mem->bus.base = qdev->vram_base;
236 mem->bus.offset = mem->start << PAGE_SHIFT;
237 break;
238 case TTM_PL_PRIV:
239 mem->bus.is_iomem = true;
240 mem->bus.base = qdev->surfaceram_base;
241 mem->bus.offset = mem->start << PAGE_SHIFT;
242 break;
243 default:
244 return -EINVAL;
245 }
246 return 0;
247}

263static int qxl_ttm_backend_bind(struct ttm_tt *ttm,
264 struct ttm_mem_reg *bo_mem)
265{
266 struct qxl_ttm_tt *gtt = (void *)ttm;
267
268 gtt->offset = (unsigned long)(bo_mem->start << PAGE_SHIFT);
269 if (!ttm->num_pages) {
270 WARN(1, "nothing to bind %lu pages for mreg %p back %p!\n",
271 ttm->num_pages, bo_mem, ttm);
272 }
273 /* Not implemented */
274 return -1;

316static struct ttm_tt *qxl_ttm_tt_create(struct ttm_bo_device *bdev,
317 unsigned long size, uint32_t page_flags,
318 struct page *dummy_read_page)
319{
320 struct qxl_device *qdev;
321 struct qxl_ttm_tt *gtt;
322
323 qdev = qxl_get_qdev(bdev);
324 gtt = kzalloc(sizeof(struct qxl_ttm_tt), GFP_KERNEL);
325 if (gtt == NULL)
326 return NULL;
327 gtt->ttm.ttm.func = &qxl_backend_func;
328 gtt->qdev = qdev;
329 if (ttm_dma_tt_init(&gtt->ttm, bdev, size, page_flags,
330 dummy_read_page)) {
331 kfree(gtt);
332 return NULL;
333 }
334 return &gtt->ttm.ttm;
335}

347static int qxl_bo_move(struct ttm_buffer_object *bo,
348 bool evict, bool interruptible,
349 bool no_wait_gpu,
350 struct ttm_mem_reg *new_mem)
351{
352 struct ttm_mem_reg *old_mem = &bo->mem;
353 int ret;
354
355 ret = ttm_bo_wait(bo, interruptible, no_wait_gpu);
356 if (ret)
357 return ret;
358
359
360 if (old_mem->mem_type == TTM_PL_SYSTEM && bo->ttm == NULL) {
361 qxl_move_null(bo, new_mem);
362 return 0;
363 }
364 return ttm_bo_move_memcpy(bo, interruptible, no_wait_gpu,
365 new_mem);
366}

368static void qxl_bo_move_notify(struct ttm_buffer_object *bo,
369 bool evict,
370 struct ttm_mem_reg *new_mem)
371{
372 struct qxl_bo *qbo;
373 struct qxl_device *qdev;
374
375 if (!qxl_ttm_bo_is_qxl_bo(bo))
376 return;
377 qbo = to_qxl_bo(bo);
378 qdev = qbo->gem_base.dev->dev_private;
379
380 if (bo->mem.mem_type == TTM_PL_PRIV && qbo->surface_id)
381 qxl_surface_evict(qdev, qbo, new_mem ? true : false);
382}

400int qxl_ttm_init(struct qxl_device *qdev)
401{
402 int r;
403 int num_io_pages; /* != rom->num_io_pages, we include surface0 */
404
405 r = qxl_ttm_global_init(qdev);
406 if (r)
407 return r;
408 /* No others user of address space so set it to 0 */
409 r = ttm_bo_device_init(&qdev->mman.bdev,
410 qdev->mman.bo_global_ref.ref.object,
411 &qxl_bo_driver,
412 qdev->ddev.anon_inode->i_mapping,
413 DRM_FILE_PAGE_OFFSET, 0);
414 if (r) {
415 DRM_ERROR("failed initializing buffer object driver(%d).\n", r);
416 return r;
417 }
418 /* NOTE: this includes the framebuffer (aka surface 0) */
419 num_io_pages = qdev->rom->ram_header_offset / PAGE_SIZE;
420 r = ttm_bo_init_mm(&qdev->mman.bdev, TTM_PL_VRAM,
421 num_io_pages);
422 if (r) {
423 DRM_ERROR("Failed initializing VRAM heap.\n");
424 return r;
425 }
426 r = ttm_bo_init_mm(&qdev->mman.bdev, TTM_PL_PRIV,
427 qdev->surfaceram_size / PAGE_SIZE);
428 if (r) {
429 DRM_ERROR("Failed initializing Surfaces heap.\n");
430 return r;
431 }
432 DRM_INFO("qxl: %uM of VRAM memory size\n",
433 (unsigned)qdev->vram_size / (1024 * 1024));
434 DRM_INFO("qxl: %luM of IO pages memory ready (VRAM domain)\n",
435 ((unsigned)num_io_pages * PAGE_SIZE) / (1024 * 1024));
436 DRM_INFO("qxl: %uM of Surface memory size\n",
437 (unsigned)qdev->surfaceram_size / (1024 * 1024));
438 return 0;
439}

441void qxl_ttm_fini(struct qxl_device *qdev)
442{
443 ttm_bo_clean_mm(&qdev->mman.bdev, TTM_PL_VRAM);
444 ttm_bo_clean_mm(&qdev->mman.bdev, TTM_PL_PRIV);
445 ttm_bo_device_release(&qdev->mman.bdev);
446 qxl_ttm_global_fini(qdev);
447 DRM_INFO("qxl: ttm finalized\n");
448}

454static int qxl_mm_dump_table(struct seq_file *m, void *data)
455{
456 struct drm_info_node *node = (struct drm_info_node *)m->private;

458 struct drm_device *dev = node->minor->dev;

461 struct drm_printer p = drm_seq_file_printer(m);
462
463 spin_lock(&glob->lru_lock);
464 drm_mm_print(mm, &p);
465 spin_unlock(&glob->lru_lock);
466 return 0;
467}

470int qxl_ttm_debugfs_init(struct qxl_device *qdev)
471{
472#if defined(CONFIG_DEBUG_FS)
473 static struct drm_info_list qxl_mem_types_list[QXL_DEBUGFS_MEM_TYPES];
474 static char qxl_mem_types_names[QXL_DEBUGFS_MEM_TYPES][32];
475 unsigned i;
476
477 for (i = 0; i < QXL_DEBUGFS_MEM_TYPES; i++) {
478 if (i == 0)
479 sprintf(qxl_mem_types_names[i], "qxl_mem_mm");
480 else
481 sprintf(qxl_mem_types_names[i], "qxl_surf_mm");
482 qxl_mem_types_list[i].name = qxl_mem_types_names[i];
483 qxl_mem_types_list[i].show = &qxl_mm_dump_table;
484 qxl_mem_types_list[i].driver_features = 0;
485 if (i == 0)
486 qxl_mem_types_list[i].data = qdev->mman.bdev.man[TTM_PL_VRAM].priv;
487 else
488 qxl_mem_types_list[i].data = qdev->mman.bdev.man[TTM_PL_PRIV].priv;
489
490 }
491 return qxl_debugfs_add_files(qdev, qxl_mem_types_list, i);
492#else
493 return 0;
494#endif
495}
