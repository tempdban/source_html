33static char printable_char(int c)
34{
35 return isascii(c) && isprint(c) ? c : '?';
36}

38/**
39 * drm_mode_legacy_fb_format - compute drm fourcc code from legacy description
40 * @bpp: bits per pixels
41 * @depth: bit depth per pixel
42 *
43 * Computes a drm fourcc pixel format code for the given @bpp/@depth values.
44 * Useful in fbdev emulation code, since that deals in those values.
45 */
46uint32_t drm_mode_legacy_fb_format(uint32_t bpp, uint32_t depth)
47{
48 uint32_t fmt;
49
50 switch (bpp) {
51 case 8:
52 fmt = DRM_FORMAT_C8;
53 break;
54 case 16:
55 if (depth == 15)
56 fmt = DRM_FORMAT_XRGB1555;
57 else
58 fmt = DRM_FORMAT_RGB565;
59 break;
60 case 24:
61 fmt = DRM_FORMAT_RGB888;
62 break;
63 case 32:
64 if (depth == 24)
65 fmt = DRM_FORMAT_XRGB8888;
66 else if (depth == 30)
67 fmt = DRM_FORMAT_XRGB2101010;
68 else
69 fmt = DRM_FORMAT_ARGB8888;
70 break;
71 default:
72 DRM_ERROR("bad bpp, assuming x8r8g8b8 pixel format\n");
73 fmt = DRM_FORMAT_XRGB8888;
74 break;
75 }
76
77 return fmt;
78}
79EXPORT_SYMBOL(drm_mode_legacy_fb_format);

81/**
82 * drm_get_format_name - fill a string with a drm fourcc format's name
83 * @format: format to compute name of
84 * @buf: caller-supplied buffer
85 */
86const char *drm_get_format_name(uint32_t format, struct drm_format_name_buf *buf)
87{
88 snprintf(buf->str, sizeof(buf->str),
89 "%c%c%c%c %s-endian (0x%08x)",
90 printable_char(format & 0xff),
91 printable_char((format >> 8) & 0xff),
92 printable_char((format >> 16) & 0xff),
93 printable_char((format >> 24) & 0x7f),
94 format & DRM_FORMAT_BIG_ENDIAN ? "big" : "little",
95 format);
96
97 return buf->str;
98}
99EXPORT_SYMBOL(drm_get_format_name);

188/**
189 * drm_format_info - query information for a given format
190 * @format: pixel format (DRM_FORMAT_*)
191 *
192 * The caller should only pass a supported pixel format to this function.
193 * Unsupported pixel formats will generate a warning in the kernel log.
194 *
195 * Returns:
196 * The instance of struct drm_format_info that describes the pixel format, or
197 * NULL if the format is unsupported.
198 */
199const struct drm_format_info *drm_format_info(u32 format)
200{
201 const struct drm_format_info *info;
202
203 info = __drm_format_info(format);
204 WARN_ON(!info);
205 return info;
206}
207EXPORT_SYMBOL(drm_format_info);

209/**
210 * drm_get_format_info - query information for a given framebuffer configuration
211 * @dev: DRM device
212 * @mode_cmd: metadata from the userspace fb creation request
213 *
214 * Returns:
215 * The instance of struct drm_format_info that describes the pixel format, or
216 * NULL if the format is unsupported.
217 */
218const struct drm_format_info *
219drm_get_format_info(struct drm_device *dev,
220 const struct drm_mode_fb_cmd2 *mode_cmd)
221{
222 const struct drm_format_info *info = NULL;
223
224 if (dev->mode_config.funcs->get_format_info)
225 info = dev->mode_config.funcs->get_format_info(mode_cmd);
226
227 if (!info)
228 info = drm_format_info(mode_cmd->pixel_format);
229
230 return info;
231}
232EXPORT_SYMBOL(drm_get_format_info);

250/**
251 * drm_format_plane_cpp - determine the bytes per pixel value
252 * @format: pixel format (DRM_FORMAT_*)
253 * @plane: plane index
254 *
255 * Returns:
256 * The bytes per pixel value for the specified plane.
257 */
258int drm_format_plane_cpp(uint32_t format, int plane)
259{
260 const struct drm_format_info *info;
261
262 info = drm_format_info(format);
263 if (!info || plane >= info->num_planes)
264 return 0;
265
266 return info->cpp[plane];
267}

304/**
305 * drm_format_plane_width - width of the plane given the first plane
306 * @width: width of the first plane
307 * @format: pixel format
308 * @plane: plane index
309 *
310 * Returns:
311 * The width of @plane, given that the width of the first plane is @width.
312 */
313int drm_format_plane_width(int width, uint32_t format, int plane)
314{
315 const struct drm_format_info *info;
316
317 info = drm_format_info(format);
318 if (!info || plane >= info->num_planes)
319 return 0;
320
321 if (plane == 0)
322 return width;
323
324 return width / info->hsub;
325}
326EXPORT_SYMBOL(drm_format_plane_width);
