29static int bochs_ttm_global_init(struct bochs_device *bochs)
30{
31 struct drm_global_reference *global_ref;
32 int r;
33
34 global_ref = &bochs->ttm.mem_global_ref;
35 global_ref->global_type = DRM_GLOBAL_TTM_MEM;
36 global_ref->size = sizeof(struct ttm_mem_global);
37 global_ref->init = &bochs_ttm_mem_global_init;
38 global_ref->release = &bochs_ttm_mem_global_release;
39 r = drm_global_item_ref(global_ref);
40 if (r != 0) {
41 DRM_ERROR("Failed setting up TTM memory accounting "
42 "subsystem.\n");
43 return r;
44 }
45
46 bochs->ttm.bo_global_ref.mem_glob =
47 bochs->ttm.mem_global_ref.object;
48 global_ref = &bochs->ttm.bo_global_ref.ref;
49 global_ref->global_type = DRM_GLOBAL_TTM_BO;
50 global_ref->size = sizeof(struct ttm_bo_global);
51 global_ref->init = &ttm_bo_global_init;
52 global_ref->release = &ttm_bo_global_release;
53 r = drm_global_item_ref(global_ref);
54 if (r != 0) {
55 DRM_ERROR("Failed setting up TTM BO subsystem.\n");
56 drm_global_item_unref(&bochs->ttm.mem_global_ref);
57 return r;
58 }
59
60 return 0;
61}

63static void bochs_ttm_global_release(struct bochs_device *bochs)
64{
65 if (bochs->ttm.mem_global_ref.release == NULL)
66 return;
67
68 drm_global_item_unref(&bochs->ttm.bo_global_ref.ref);
69 drm_global_item_unref(&bochs->ttm.mem_global_ref);
70 bochs->ttm.mem_global_ref.release = NULL;
71}

90static int bochs_bo_init_mem_type(struct ttm_bo_device *bdev, uint32_t type,
91 struct ttm_mem_type_manager *man)
92{
93 switch (type) {
94 case TTM_PL_SYSTEM:
95 man->flags = TTM_MEMTYPE_FLAG_MAPPABLE;
96 man->available_caching = TTM_PL_MASK_CACHING;
97 man->default_caching = TTM_PL_FLAG_CACHED;
98 break;
99 case TTM_PL_VRAM:
100 man->func = &ttm_bo_manager_func;
101 man->flags = TTM_MEMTYPE_FLAG_FIXED |
102 TTM_MEMTYPE_FLAG_MAPPABLE;
103 man->available_caching = TTM_PL_FLAG_UNCACHED |
104 TTM_PL_FLAG_WC;
105 man->default_caching = TTM_PL_FLAG_WC;
106 break;
107 default:
108 DRM_ERROR("Unsupported memory type %u\n", (unsigned)type);
109 return -EINVAL;
110 }
111 return 0;
112}

135static int bochs_ttm_io_mem_reserve(struct ttm_bo_device *bdev,
136 struct ttm_mem_reg *mem)
137{
138 struct ttm_mem_type_manager *man = &bdev->man[mem->mem_type];
139 struct bochs_device *bochs = bochs_bdev(bdev);
140
141 mem->bus.addr = NULL;
142 mem->bus.offset = 0;
143 mem->bus.size = mem->num_pages << PAGE_SHIFT;
144 mem->bus.base = 0;
145 mem->bus.is_iomem = false;
146 if (!(man->flags & TTM_MEMTYPE_FLAG_MAPPABLE))
147 return -EINVAL;
148 switch (mem->mem_type) {
149 case TTM_PL_SYSTEM:
150 /* system memory */
151 return 0;
152 case TTM_PL_VRAM:
153 mem->bus.offset = mem->start << PAGE_SHIFT;
154 mem->bus.base = bochs->fb_base;
155 mem->bus.is_iomem = true;
156 break;
157 default:
158 return -EINVAL;
159 break;
160 }
161 return 0;
162}

179static struct ttm_tt *bochs_ttm_tt_create(struct ttm_bo_device *bdev,
180 unsigned long size,
181 uint32_t page_flags,
182 struct page *dummy_read_page)
183{
184 struct ttm_tt *tt;
185
186 tt = kzalloc(sizeof(struct ttm_tt), GFP_KERNEL);
187 if (tt == NULL)
188 return NULL;
189 tt->func = &bochs_tt_backend_func;
190 if (ttm_tt_init(tt, bdev, size, page_flags, dummy_read_page)) {
191 kfree(tt);
192 return NULL;
193 }
194 return tt;
195}

211int bochs_mm_init(struct bochs_device *bochs)
212{
213 struct ttm_bo_device *bdev = &bochs->ttm.bdev;
214 int ret;
215
216 ret = bochs_ttm_global_init(bochs);
217 if (ret)
218 return ret;
219
220 ret = ttm_bo_device_init(&bochs->ttm.bdev,
221 bochs->ttm.bo_global_ref.ref.object,
222 &bochs_bo_driver,
223 bochs->dev->anon_inode->i_mapping,
224 DRM_FILE_PAGE_OFFSET,
225 true);
226 if (ret) {
227 DRM_ERROR("Error initialising bo driver; %d\n", ret);
228 return ret;
229 }
230
231 ret = ttm_bo_init_mm(bdev, TTM_PL_VRAM,
232 bochs->fb_size >> PAGE_SHIFT);
233 if (ret) {
234 DRM_ERROR("Failed ttm VRAM init: %d\n", ret);
235 return ret;
236 }
237
238 bochs->ttm.initialized = true;
239 return 0;
240}

242void bochs_mm_fini(struct bochs_device *bochs)
243{
244 if (!bochs->ttm.initialized)
245 return;
246
247 ttm_bo_device_release(&bochs->ttm.bdev);
248 bochs_ttm_global_release(bochs);
249 bochs->ttm.initialized = false;
250}

252static void bochs_ttm_placement(struct bochs_bo *bo, int domain)
253{
254 unsigned i;
255 u32 c = 0;
256 bo->placement.placement = bo->placements;
257 bo->placement.busy_placement = bo->placements;
258 if (domain & TTM_PL_FLAG_VRAM) {
259 bo->placements[c++].flags = TTM_PL_FLAG_WC
260 | TTM_PL_FLAG_UNCACHED
261 | TTM_PL_FLAG_VRAM;
262 }
263 if (domain & TTM_PL_FLAG_SYSTEM) {
264 bo->placements[c++].flags = TTM_PL_MASK_CACHING
265 | TTM_PL_FLAG_SYSTEM;
266 }
267 if (!c) {
268 bo->placements[c++].flags = TTM_PL_MASK_CACHING
269 | TTM_PL_FLAG_SYSTEM;
270 }
271 for (i = 0; i < c; ++i) {
272 bo->placements[i].fpfn = 0;
273 bo->placements[i].lpfn = 0;
274 }
275 bo->placement.num_placement = c;
276 bo->placement.num_busy_placement = c;
277}

284int bochs_bo_pin(struct bochs_bo *bo, u32 pl_flag, u64 *gpu_addr)
285{
286 int i, ret;
287
288 if (bo->pin_count) {
289 bo->pin_count++;
290 if (gpu_addr)
291 *gpu_addr = bochs_bo_gpu_offset(bo);
292 return 0;
293 }
294
295 bochs_ttm_placement(bo, pl_flag);
296 for (i = 0; i < bo->placement.num_placement; i++)
297 bo->placements[i].flags |= TTM_PL_FLAG_NO_EVICT;
298 ret = ttm_bo_validate(&bo->bo, &bo->placement, false, false);
299 if (ret)
300 return ret;
301
302 bo->pin_count = 1;
303 if (gpu_addr)
304 *gpu_addr = bochs_bo_gpu_offset(bo);
305 return 0;
306}
307
308int bochs_bo_unpin(struct bochs_bo *bo)
309{
310 int i, ret;
311
312 if (!bo->pin_count) {
313 DRM_ERROR("unpin bad %p\n", bo);
314 return 0;
315 }
316 bo->pin_count--;
317
318 if (bo->pin_count)
319 return 0;
320
321 for (i = 0; i < bo->placement.num_placement; i++)
322 bo->placements[i].flags &= ~TTM_PL_FLAG_NO_EVICT;
323 ret = ttm_bo_validate(&bo->bo, &bo->placement, false, false);
324 if (ret)
325 return ret;
326
327 return 0;
328}

345static int bochs_bo_create(struct drm_device *dev, int size, int align,
346 uint32_t flags, struct bochs_bo **pbochsbo)
347{
348 struct bochs_device *bochs = dev->dev_private;
349 struct bochs_bo *bochsbo;
350 size_t acc_size;
351 int ret;
352
353 bochsbo = kzalloc(sizeof(struct bochs_bo), GFP_KERNEL);
354 if (!bochsbo)
355 return -ENOMEM;
356
357 ret = drm_gem_object_init(dev, &bochsbo->gem, size);
358 if (ret) {
359 kfree(bochsbo);
360 return ret;
361 }
362
363 bochsbo->bo.bdev = &bochs->ttm.bdev;
364 bochsbo->bo.bdev->dev_mapping = dev->anon_inode->i_mapping;
365
366 bochs_ttm_placement(bochsbo, TTM_PL_FLAG_VRAM | TTM_PL_FLAG_SYSTEM);
367
368 acc_size = ttm_bo_dma_acc_size(&bochs->ttm.bdev, size,
369 sizeof(struct bochs_bo));
370
371 ret = ttm_bo_init(&bochs->ttm.bdev, &bochsbo->bo, size,
372 ttm_bo_type_device, &bochsbo->placement,
373 align >> PAGE_SHIFT, false, NULL, acc_size,
374 NULL, NULL, bochs_bo_ttm_destroy);
375 if (ret)
376 return ret;
377
378 *pbochsbo = bochsbo;
379 return 0;
380}

382int bochs_gem_create(struct drm_device *dev, u32 size, bool iskernel,
383 struct drm_gem_object **obj)
384{
385 struct bochs_bo *bochsbo;
386 int ret;
387
388 *obj = NULL;
389
390 size = PAGE_ALIGN(size);
391 if (size == 0)
392 return -EINVAL;
393
394 ret = bochs_bo_create(dev, size, 0, 0, &bochsbo);
395 if (ret) {
396 if (ret != -ERESTARTSYS)
397 DRM_ERROR("failed to allocate GEM object\n");
398 return ret;
399 }
400 *obj = &bochsbo->gem;
401 return 0;
402}

404int bochs_dumb_create(struct drm_file *file, struct drm_device *dev,
405 struct drm_mode_create_dumb *args)
406{
407 struct drm_gem_object *gobj;
408 u32 handle;
409 int ret;
410
411 args->pitch = args->width * ((args->bpp + 7) / 8);
412 args->size = args->pitch * args->height;
413
414 ret = bochs_gem_create(dev, args->size, false,
415 &gobj);
416 if (ret)
417 return ret;
418
419 ret = drm_gem_handle_create(file, gobj, &handle);
420 drm_gem_object_unreference_unlocked(gobj);
421 if (ret)
422 return ret;
423
424 args->handle = handle;
425 return 0;
426}

496static struct drm_framebuffer *
497bochs_user_framebuffer_create(struct drm_device *dev,
498 struct drm_file *filp,
499 const struct drm_mode_fb_cmd2 *mode_cmd)
500{
501 struct drm_gem_object *obj;
502 struct bochs_framebuffer *bochs_fb;
503 int ret;
504
505 DRM_DEBUG_DRIVER("%dx%d, format %c%c%c%c\n",
506 mode_cmd->width, mode_cmd->height,
507 (mode_cmd->pixel_format) & 0xff,
508 (mode_cmd->pixel_format >> 8) & 0xff,
509 (mode_cmd->pixel_format >> 16) & 0xff,
510 (mode_cmd->pixel_format >> 24) & 0xff);
511
512 if (mode_cmd->pixel_format != DRM_FORMAT_XRGB8888)
513 return ERR_PTR(-ENOENT);
514
515 obj = drm_gem_object_lookup(filp, mode_cmd->handles[0]);
516 if (obj == NULL)
517 return ERR_PTR(-ENOENT);
518
519 bochs_fb = kzalloc(sizeof(*bochs_fb), GFP_KERNEL);
520 if (!bochs_fb) {
521 drm_gem_object_unreference_unlocked(obj);
522 return ERR_PTR(-ENOMEM);
523 }
524
525 ret = bochs_framebuffer_init(dev, bochs_fb, mode_cmd, obj);
526 if (ret) {
527 drm_gem_object_unreference_unlocked(obj);
528 kfree(bochs_fb);
529 return ERR_PTR(ret);
530 }
531 return &bochs_fb->base;
532}
