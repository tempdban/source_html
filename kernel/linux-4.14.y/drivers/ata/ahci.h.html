308struct ahci_port_priv {
309 struct ata_link *active_link;
310 struct ahci_cmd_hdr *cmd_slot;

312 void *cmd_tbl;
313 dma_addr_t cmd_tbl_dma;

327 char *irq_desc; /* desc in /proc/interrupts */
328};

346 u32 em_msg_type; /* EM message type */

351 * If platform uses PHYs. There is a 1:1 relation between the port number and

429static inline void __iomem *__ahci_port_base(struct ata_host *host,
430 unsigned int port_no)
431{
432 struct ahci_host_priv *hpriv = host->private_data;
433 void __iomem *mmio = hpriv->mmio;
434
435 return mmio + 0x100 + (port_no * 0x80);
436}
437
438static inline void __iomem *ahci_port_base(struct ata_port *ap)
439{
440 return __ahci_port_base(ap->host, ap->port_no);
441}

443static inline int ahci_nr_ports(u32 cap)
444{
445 return (cap & 0x1f) + 1;
446}
