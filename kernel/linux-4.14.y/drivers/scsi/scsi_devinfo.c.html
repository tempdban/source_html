273static struct scsi_dev_info_list_table *scsi_devinfo_lookup_by_key(int key)
274{
275 struct scsi_dev_info_list_table *devinfo_table;
276 int found = 0;
277
278 list_for_each_entry(devinfo_table, &scsi_dev_info_list, node)
279 if (devinfo_table->key == key) {
280 found = 1;
281 break;
282 }
283 if (!found)
284 return ERR_PTR(-EINVAL);
285
286 return devinfo_table;
287}

289/*
290 * scsi_strcpy_devinfo: called from scsi_dev_info_list_add to copy into
291 * devinfo vendor and model strings.
292 */
293static void scsi_strcpy_devinfo(char *name, char *to, size_t to_length,
294 char *from, int compatible)
295{
296 size_t from_length;
297
298 from_length = strlen(from);
299 strncpy(to, from, min(to_length, from_length));
300 if (from_length < to_length) {
301 if (compatible) {
302 /*
303 * NUL terminate the string if it is short.
304 */
305 to[from_length] = '\0';
306 } else {
307 /*
308 * space pad the string if it is short.
309 */
310 strncpy(&to[from_length], spaces,
311 to_length - from_length);
312 }
313 }
314 if (from_length > to_length)
315 printk(KERN_WARNING "%s: %s string '%s' is too long\n",
316 __func__, name, from);
317}

319/**
320 * scsi_dev_info_list_add - add one dev_info list entry.
321 * @compatible: if true, null terminate short strings. Otherwise space pad.
322 * @vendor: vendor string
323 * @model: model (product) string
324 * @strflags: integer string
325 * @flags: if strflags NULL, use this flag value
326 *
327 * Description:
328 * Create and add one dev_info entry for @vendor, @model, @strflags or
329 * @flag. If @compatible, add to the tail of the list, do not space
330 * pad, and set devinfo->compatible. The scsi_static_device_list entries
331 * are added with @compatible 1 and @clfags NULL.
332 *
333 * Returns: 0 OK, -error on failure.
334 **/
335static int scsi_dev_info_list_add(int compatible, char *vendor, char *model,
336 char *strflags, int flags)
337{
338 return scsi_dev_info_list_add_keyed(compatible, vendor, model,
339 strflags, flags,
340 SCSI_DEVINFO_GLOBAL);
341}
342
343/**
344 * scsi_dev_info_list_add_keyed - add one dev_info list entry.
345 * @compatible: if true, null terminate short strings. Otherwise space pad.
346 * @vendor: vendor string
347 * @model: model (product) string
348 * @strflags: integer string
349 * @flags: if strflags NULL, use this flag value
350 * @key: specify list to use
351 *
352 * Description:
353 * Create and add one dev_info entry for @vendor, @model,
354 * @strflags or @flag in list specified by @key. If @compatible,
355 * add to the tail of the list, do not space pad, and set
356 * devinfo->compatible. The scsi_static_device_list entries are
357 * added with @compatible 1 and @clfags NULL.
358 *
359 * Returns: 0 OK, -error on failure.
360 **/
361int scsi_dev_info_list_add_keyed(int compatible, char *vendor, char *model,
362 char *strflags, int flags, int key)
363{
364 struct scsi_dev_info_list *devinfo;
365 struct scsi_dev_info_list_table *devinfo_table =
366 scsi_devinfo_lookup_by_key(key);
367
368 if (IS_ERR(devinfo_table))
369 return PTR_ERR(devinfo_table);
370
371 devinfo = kmalloc(sizeof(*devinfo), GFP_KERNEL);
372 if (!devinfo) {
373 printk(KERN_ERR "%s: no memory\n", __func__);
374 return -ENOMEM;
375 }
376
377 scsi_strcpy_devinfo("vendor", devinfo->vendor, sizeof(devinfo->vendor),
378 vendor, compatible);
379 scsi_strcpy_devinfo("model", devinfo->model, sizeof(devinfo->model),
380 model, compatible);
381
382 if (strflags)
383 devinfo->flags = simple_strtoul(strflags, NULL, 0);
384 else
385 devinfo->flags = flags;
386
387 devinfo->compatible = compatible;
388
389 if (compatible)
390 list_add_tail(&devinfo->dev_info_list,
391 &devinfo_table->scsi_dev_info_list);
392 else
393 list_add(&devinfo->dev_info_list,
394 &devinfo_table->scsi_dev_info_list);
395
396 return 0;
397}
398EXPORT_SYMBOL(scsi_dev_info_list_add_keyed);

400/**
401 * scsi_dev_info_list_find - find a matching dev_info list entry.
402 * @vendor: vendor string
403 * @model: model (product) string
404 * @key: specify list to use
405 *
406 * Description:
407 * Finds the first dev_info entry matching @vendor, @model
408 * in list specified by @key.
409 *
410 * Returns: pointer to matching entry, or ERR_PTR on failure.
411 **/
412static struct scsi_dev_info_list *scsi_dev_info_list_find(const char *vendor,
413 const char *model, int key)
414{
415 struct scsi_dev_info_list *devinfo;
416 struct scsi_dev_info_list_table *devinfo_table =
417 scsi_devinfo_lookup_by_key(key);
418 size_t vmax, mmax;
419 const char *vskip, *mskip;
420
421 if (IS_ERR(devinfo_table))
422 return (struct scsi_dev_info_list *) devinfo_table;
423
424 /* Prepare for "compatible" matches */
425
426 /*
427 * XXX why skip leading spaces? If an odd INQUIRY
428 * value, that should have been part of the
429 * scsi_static_device_list[] entry, such as " FOO"
430 * rather than "FOO". Since this code is already
431 * here, and we don't know what device it is
432 * trying to work with, leave it as-is.
433 */
434 vmax = sizeof(devinfo->vendor);
435 vskip = vendor;
436 while (vmax > 0 && *vskip == ' ') {
437 vmax--;
438 vskip++;
439 }
440 /* Also skip trailing spaces */
441 while (vmax > 0 && vskip[vmax - 1] == ' ')
442 --vmax;
443
444 mmax = sizeof(devinfo->model);
445 mskip = model;
446 while (mmax > 0 && *mskip == ' ') {
447 mmax--;
448 mskip++;
449 }
450 while (mmax > 0 && mskip[mmax - 1] == ' ')
451 --mmax;
452
453 list_for_each_entry(devinfo, &devinfo_table->scsi_dev_info_list,
454 dev_info_list) {
455 if (devinfo->compatible) {
456 /*
457 * Behave like the older version of get_device_flags.
458 */
459 if (memcmp(devinfo->vendor, vskip, vmax) ||
460 (vmax < sizeof(devinfo->vendor) &&
461 devinfo->vendor[vmax]))
462 continue;
463 if (memcmp(devinfo->model, mskip, mmax) ||
464 (mmax < sizeof(devinfo->model) &&
465 devinfo->model[mmax]))
466 continue;
467 return devinfo;
468 } else {
469 if (!memcmp(devinfo->vendor, vendor,
470 sizeof(devinfo->vendor)) &&
471 !memcmp(devinfo->model, model,
472 sizeof(devinfo->model)))
473 return devinfo;
474 }
475 }
476
477 return ERR_PTR(-ENOENT);
478}

480/**
481 * scsi_dev_info_list_del_keyed - remove one dev_info list entry.
482 * @vendor: vendor string
483 * @model: model (product) string
484 * @key: specify list to use
485 *
486 * Description:
487 * Remove and destroy one dev_info entry for @vendor, @model
488 * in list specified by @key.
489 *
490 * Returns: 0 OK, -error on failure.
491 **/
492int scsi_dev_info_list_del_keyed(char *vendor, char *model, int key)
493{

496 found = scsi_dev_info_list_find(vendor, model, key);
497 if (IS_ERR(found))
498 return PTR_ERR(found);

502 return 0;

506/**
507 * scsi_dev_info_list_add_str - parse dev_list and add to the scsi_dev_info_list.
508 * @dev_list: string of device flags to add
509 *
510 * Description:
511 * Parse dev_list, and add entries to the scsi_dev_info_list.
512 * dev_list is of the form "vendor:product:flag,vendor:product:flag".
513 * dev_list is modified via strsep. Can be called for command line
514 * addition, for proc or mabye a sysfs interface.
515 *
516 * Returns: 0 if OK, -error on failure.
517 **/
518static int scsi_dev_info_list_add_str(char *dev_list)
519{
520 char *vendor, *model, *strflags, *next;
521 char *next_check;
522 int res = 0;
523
524 next = dev_list;
525 if (next && next[0] == '"') {
526 /*
527 * Ignore both the leading and trailing quote.
528 */
529 next++;
530 next_check = ",\"";
531 } else {
532 next_check = ",";
533 }
534
535 /*
536 * For the leading and trailing '"' case, the for loop comes
537 * through the last time with vendor[0] == '\0'.
538 */
539 for (vendor = strsep(&next, ":"); vendor && (vendor[0] != '\0')
540 && (res == 0); vendor = strsep(&next, ":")) {
541 strflags = NULL;
542 model = strsep(&next, ":");
543 if (model)
544 strflags = strsep(&next, next_check);
545 if (!model || !strflags) {
546 printk(KERN_ERR "%s: bad dev info string '%s' '%s'"
547 " '%s'\n", __func__, vendor, model,
548 strflags);
549 res = -EINVAL;
550 } else
551 res = scsi_dev_info_list_add(0 /* compatible */, vendor,
552 model, strflags, 0);
553 }
554 return res;
555}

578/**
579 * scsi_get_device_flags_keyed - get device specific flags from the dynamic device list
580 * @sdev: &scsi_device to get flags for
581 * @vendor: vendor name
582 * @model: model name
583 * @key: list to look up
584 *
585 * Description:
586 * Search the scsi_dev_info_list specified by @key for an entry
587 * matching @vendor and @model, if found, return the matching
588 * flags value, else return the host or global default settings.
589 * Called during scan time.
590 **/
591int scsi_get_device_flags_keyed(struct scsi_device *sdev,
592 const unsigned char *vendor,
593 const unsigned char *model,
594 int key)
595{
596 struct scsi_dev_info_list *devinfo;
597
598 devinfo = scsi_dev_info_list_find(vendor, model, key);
599 if (!IS_ERR(devinfo))
600 return devinfo->flags;
601
602 /* key or device not found: return nothing */
603 if (key != SCSI_DEVINFO_GLOBAL)
604 return 0;
605
606 /* except for the global list, where we have an exception */
607 if (sdev->sdev_bflags)
608 return sdev->sdev_bflags;
609
610 return scsi_default_dev_flags;
611}

620static int devinfo_seq_show(struct seq_file *m, void *v)
621{
622 struct double_list *dl = v;

624 list_entry(dl->top, struct scsi_dev_info_list_table, node);
625 struct scsi_dev_info_list *devinfo =

629 if (devinfo_table->scsi_dev_info_list.next == dl->bottom &&
630 devinfo_table->name)
631 seq_printf(m, "[%s]:\n", devinfo_table->name);

638static void *devinfo_seq_start(struct seq_file *m, loff_t *ppos)
639{
640 struct double_list *dl = kmalloc(sizeof(*dl), GFP_KERNEL);
641 loff_t pos = *ppos;
642
643 if (!dl)
644 return NULL;
645
646 list_for_each(dl->top, &scsi_dev_info_list) {
647 struct scsi_dev_info_list_table *devinfo_table =
648 list_entry(dl->top, struct scsi_dev_info_list_table,
649 node);
650 list_for_each(dl->bottom, &devinfo_table->scsi_dev_info_list)
651 if (pos-- == 0)
652 return dl;
653 }
654
655 kfree(dl);
656 return NULL;
657}
658
659static void *devinfo_seq_next(struct seq_file *m, void *v, loff_t *ppos)
660{
661 struct double_list *dl = v;
662 struct scsi_dev_info_list_table *devinfo_table =
663 list_entry(dl->top, struct scsi_dev_info_list_table, node);
664
665 ++*ppos;
666 dl->bottom = dl->bottom->next;
667 while (&devinfo_table->scsi_dev_info_list == dl->bottom) {
668 dl->top = dl->top->next;
669 if (dl->top == &scsi_dev_info_list) {
670 kfree(dl);
671 return NULL;
672 }
673 devinfo_table = list_entry(dl->top,
674 struct scsi_dev_info_list_table,
675 node);
676 dl->bottom = devinfo_table->scsi_dev_info_list.next;
677 }
678
679 return dl;
680}

699/*
700 * proc_scsi_dev_info_write - allow additions to scsi_dev_info_list via /proc.
701 *
702 * Description: Adds a black/white list entry for vendor and model with an
703 * integer value of flag to the scsi device info list.
704 * To use, echo "vendor:model:flag" > /proc/scsi/device_info
705 */
706static ssize_t proc_scsi_devinfo_write(struct file *file,
707 const char __user *buf,
708 size_t length, loff_t *ppos)
709{
710 char *buffer;
711 ssize_t err = length;
712
713 if (!buf || length>PAGE_SIZE)
714 return -EINVAL;
715 if (!(buffer = (char *) __get_free_page(GFP_KERNEL)))
716 return -ENOMEM;
717 if (copy_from_user(buffer, buf, length)) {
718 err =-EFAULT;
719 goto out;
720 }
721
722 if (length < PAGE_SIZE)
723 buffer[length] = '\0';
724 else if (buffer[PAGE_SIZE-1]) {
725 err = -EINVAL;
726 goto out;
727 }
728
729 scsi_dev_info_list_add_str(buffer);
730
731out:
732 free_page((unsigned long)buffer);
733 return err;
734}

768/**
769 * scsi_dev_info_add_list - add a new devinfo list
770 * @key: key of the list to add
771 * @name: Name of the list to add (for /proc/scsi/device_info)
772 *
773 * Adds the requested list, returns zero on success, -EEXIST if the
774 * key is already registered to a list, or other error on failure.
775 */
776int scsi_dev_info_add_list(int key, const char *name)
777{
778 struct scsi_dev_info_list_table *devinfo_table =
779 scsi_devinfo_lookup_by_key(key);
780
781 if (!IS_ERR(devinfo_table))
782 /* list already exists */
783 return -EEXIST;
784
785 devinfo_table = kmalloc(sizeof(*devinfo_table), GFP_KERNEL);
786
787 if (!devinfo_table)
788 return -ENOMEM;
789
790 INIT_LIST_HEAD(&devinfo_table->node);
791 INIT_LIST_HEAD(&devinfo_table->scsi_dev_info_list);
792 devinfo_table->name = name;
793 devinfo_table->key = key;
794 list_add_tail(&devinfo_table->node, &scsi_dev_info_list);
795
796 return 0;
797}

800/**
801 * scsi_dev_info_remove_list - destroy an added devinfo list
802 * @key: key of the list to destroy
803 *
804 * Iterates over the entire list first, freeing all the values, then
805 * frees the list itself. Returns 0 on success or -EINVAL if the key
806 * can't be found.
807 */
808int scsi_dev_info_remove_list(int key)
809{
810 struct list_head *lh, *lh_next;
811 struct scsi_dev_info_list_table *devinfo_table =
812 scsi_devinfo_lookup_by_key(key);
813
814 if (IS_ERR(devinfo_table))
815 /* no such list */
816 return -EINVAL;
817
818 /* remove from the master list */
819 list_del(&devinfo_table->node);
820
821 list_for_each_safe(lh, lh_next, &devinfo_table->scsi_dev_info_list) {
822 struct scsi_dev_info_list *devinfo;
823
824 devinfo = list_entry(lh, struct scsi_dev_info_list,
825 dev_info_list);
826 kfree(devinfo);
827 }
828 kfree(devinfo_table);
829
830 return 0;
831}

834/**
835 * scsi_init_devinfo - set up the dynamic device list.
836 *
837 * Description:
838 * Add command line entries from scsi_dev_flags, then add
839 * scsi_static_device_list entries to the scsi device info list.
840 */
841int __init scsi_init_devinfo(void)
842{
843#ifdef CONFIG_SCSI_PROC_FS
844 struct proc_dir_entry *p;
845#endif
846 int error, i;
847
848 error = scsi_dev_info_add_list(SCSI_DEVINFO_GLOBAL, NULL);
849 if (error)
850 return error;
851
852 error = scsi_dev_info_list_add_str(scsi_dev_flags);
853 if (error)
854 goto out;
855
856 for (i = 0; scsi_static_device_list[i].vendor; i++) {
857 error = scsi_dev_info_list_add(1 /* compatibile */,
858 scsi_static_device_list[i].vendor,
859 scsi_static_device_list[i].model,
860 NULL,
861 scsi_static_device_list[i].flags);
862 if (error)
863 goto out;
864 }
865
866#ifdef CONFIG_SCSI_PROC_FS
867 p = proc_create("scsi/device_info", 0, NULL, &scsi_devinfo_proc_fops);
868 if (!p) {
869 error = -ENOMEM;
870 goto out;
871 }
872#endif /* CONFIG_SCSI_PROC_FS */
873
874 out:
875 if (error)
876 scsi_exit_devinfo();
877 return error;
878}
