72{ \
73 if (IOC->logging_level & BITS) \
74 CMD; \
75}

146/* inline functions for dumping debug data*/

148/**
149 * _debug_dump_mf - print message frame contents
150 * @mpi_request: pointer to message frame
151 * @sz: number of dwords
152 */
153static inline void
154_debug_dump_mf(void *mpi_request, int sz)
155{
156 int i;
157 __le32 *mfp = (__le32 *)mpi_request;
158
159 pr_info("mf:\n\t");
160 for (i = 0; i < sz; i++) {
161 if (i && ((i % 8) == 0))
162 pr_info("\n\t");
163 pr_info("%08x ", le32_to_cpu(mfp[i]));
164 }
165 pr_info("\n");
166}
