188#define IOAT2_CHAINADDR_OFFSET 0x10 /* 64-bit Descriptor Chain Address Register */

190 ? IOAT1_CHAINADDR_OFFSET : IOAT2_CHAINADDR_OFFSET)
191#define IOAT1_CHAINADDR_OFFSET_LOW 0x0C
192#define IOAT2_CHAINADDR_OFFSET_LOW 0x10
193#define IOAT_CHAINADDR_OFFSET_LOW(ver) ((ver) < IOAT_VER_2_0 \
194 ? IOAT1_CHAINADDR_OFFSET_LOW : IOAT2_CHAINADDR_OFFSET_LOW)
195#define IOAT1_CHAINADDR_OFFSET_HIGH 0x10
196#define IOAT2_CHAINADDR_OFFSET_HIGH 0x14
197#define IOAT_CHAINADDR_OFFSET_HIGH(ver) ((ver) < IOAT_VER_2_0 \
198 ? IOAT1_CHAINADDR_OFFSET_HIGH : IOAT2_CHAINADDR_OFFSET_HIGH)
