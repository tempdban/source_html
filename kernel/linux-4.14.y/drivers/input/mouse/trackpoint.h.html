11#ifndef _TRACKPOINT_H
12#define _TRACKPOINT_H
13
14/*
15 * These constants are from the TrackPoint System
16 * Engineering documentation Version 4 from IBM Watson
17 * research:
18 * http://wwwcssrv.almaden.ibm.com/trackpoint/download.html
19 */
20
21#define TP_COMMAND 0xE2 /* Commands start with this */
22
23#define TP_READ_ID 0xE1 /* Sent for device identification */
24
25/*
26 * Valid first byte responses to the "Read Secondary ID" (0xE1) command.
27 * 0x01 was the original IBM trackpoint, others implement very limited
28 * subset of trackpoint features.
29 */
30#define TP_VARIANT_IBM 0x01
31#define TP_VARIANT_ALPS 0x02
32#define TP_VARIANT_ELAN 0x03
33#define TP_VARIANT_NXP 0x04
34
35/*
36 * Commands
37 */
38#define TP_RECALIB 0x51 /* Recalibrate */
39#define TP_POWER_DOWN 0x44 /* Can only be undone through HW reset */
40#define TP_EXT_DEV 0x21 /* Determines if external device is connected (RO) */
41#define TP_EXT_BTN 0x4B /* Read extended button status */
42#define TP_POR 0x7F /* Execute Power on Reset */
43#define TP_POR_RESULTS 0x25 /* Read Power on Self test results */
44#define TP_DISABLE_EXT 0x40 /* Disable external pointing device */
45#define TP_ENABLE_EXT 0x41 /* Enable external pointing device */
46
47/*
48 * Mode manipulation
49 */
50#define TP_SET_SOFT_TRANS 0x4E /* Set mode */
51#define TP_CANCEL_SOFT_TRANS 0xB9 /* Cancel mode */
52#define TP_SET_HARD_TRANS 0x45 /* Mode can only be set */
53
54
55/*
56 * Register oriented commands/properties
57 */
58#define TP_WRITE_MEM 0x81
59#define TP_READ_MEM 0x80 /* Not used in this implementation */
60
61/*
62* RAM Locations for properties
63 */
64#define TP_SENS 0x4A /* Sensitivity */
65#define TP_MB 0x4C /* Read Middle Button Status (RO) */
66#define TP_INERTIA 0x4D /* Negative Inertia */
67#define TP_SPEED 0x60 /* Speed of TP Cursor */
68#define TP_REACH 0x57 /* Backup for Z-axis press */
69#define TP_DRAGHYS 0x58 /* Drag Hysteresis */
70 /* (how hard it is to drag */
71 /* with Z-axis pressed) */
72
73#define TP_MINDRAG 0x59 /* Minimum amount of force needed */
74 /* to trigger dragging */
75
76#define TP_THRESH 0x5C /* Minimum value for a Z-axis press */
77#define TP_UP_THRESH 0x5A /* Used to generate a 'click' on Z-axis */
78#define TP_Z_TIME 0x5E /* How sharp of a press */
79#define TP_JENKS_CURV 0x5D /* Minimum curvature for double click */
80#define TP_DRIFT_TIME 0x5F /* How long a 'hands off' condition */
81 /* must last (x*107ms) for drift */
82 /* correction to occur */
83
84/*
85 * Toggling Flag bits
86 */
87#define TP_TOGGLE 0x47 /* Toggle command */
88
89#define TP_TOGGLE_MB 0x23 /* Disable/Enable Middle Button */
90#define TP_MASK_MB 0x01
91#define TP_TOGGLE_EXT_DEV 0x23 /* Disable external device */
92#define TP_MASK_EXT_DEV 0x02
93#define TP_TOGGLE_DRIFT 0x23 /* Drift Correction */
94#define TP_MASK_DRIFT 0x80
95#define TP_TOGGLE_BURST 0x28 /* Burst Mode */
96#define TP_MASK_BURST 0x80
97#define TP_TOGGLE_PTSON 0x2C /* Press to Select */
98#define TP_MASK_PTSON 0x01
99#define TP_TOGGLE_HARD_TRANS 0x2C /* Alternate method to set Hard Transparency */
100#define TP_MASK_HARD_TRANS 0x80
101#define TP_TOGGLE_TWOHAND 0x2D /* Two handed */
102#define TP_MASK_TWOHAND 0x01
103#define TP_TOGGLE_STICKY_TWO 0x2D /* Sticky two handed */
104#define TP_MASK_STICKY_TWO 0x04
105#define TP_TOGGLE_SKIPBACK 0x2D /* Suppress movement after drag release */
106#define TP_MASK_SKIPBACK 0x08
107#define TP_TOGGLE_SOURCE_TAG 0x20 /* Bit 3 of the first packet will be set to
108 to the origin of the packet (external or TP) */
109#define TP_MASK_SOURCE_TAG 0x80
110#define TP_TOGGLE_EXT_TAG 0x22 /* Bit 3 of the first packet coming from the
111 external device will be forced to 1 */
112#define TP_MASK_EXT_TAG 0x04
113
114
115/* Power on Self Test Results */
116#define TP_POR_SUCCESS 0x3B
117
118/*
119 * Default power on values
120 */
121#define TP_DEF_SENS 0x80
122#define TP_DEF_INERTIA 0x06
123#define TP_DEF_SPEED 0x61
124#define TP_DEF_REACH 0x0A
125
126#define TP_DEF_DRAGHYS 0xFF
127#define TP_DEF_MINDRAG 0x14
128
129#define TP_DEF_THRESH 0x08
130#define TP_DEF_UP_THRESH 0xFF
131#define TP_DEF_Z_TIME 0x26
132#define TP_DEF_JENKS_CURV 0x87
133#define TP_DEF_DRIFT_TIME 0x05
134
135/* Toggles */
136#define TP_DEF_MB 0x00
137#define TP_DEF_PTSON 0x00
138#define TP_DEF_SKIPBACK 0x00
139#define TP_DEF_EXT_DEV 0x00 /* 0 means enabled */
140#define TP_DEF_TWOHAND 0x00
141#define TP_DEF_SOURCE_TAG 0x00
142
143#define MAKE_PS2_CMD(params, results, cmd) ((params<<12) | (results<<8) | (cmd))
144
145struct trackpoint_data {
146 u8 variant_id;
147 u8 firmware_id;
148
149 u8 sensitivity, speed, inertia, reach;
150 u8 draghys, mindrag;
151 u8 thresh, upthresh;
152 u8 ztime, jenks;
153 u8 drift_time;
154
155 /* toggles */
156 bool press_to_select;
157 bool skipback;
158 bool ext_dev;
159};

162int trackpoint_detect(struct psmouse *psmouse, bool set_properties);
163#else
164inline int trackpoint_detect(struct psmouse *psmouse, bool set_properties)
165{
166 return -ENOSYS;
167}
168#endif /* CONFIG_MOUSE_PS2_TRACKPOINT */
169
170#endif /* _TRACKPOINT_H */
