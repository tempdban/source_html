76struct psmouse_protocol {
77 enum psmouse_type type;
78 bool maxproto;
79 bool ignore_parity; /* Protocol should ignore parity errors from KBC */

83 const char *alias;
84 int (*detect)(struct psmouse *, bool);
85 int (*init)(struct psmouse *);
86};
