45/* synaptics capability bits */

48#define SYN_CAP_PASS_THROUGH(c) ((c) & BIT(7))

58#define SYN_MEXT_CAP_BIT(m) ((m) & BIT(1))

60/*
61 * The following describes response for the 0x0c query.
62 *
63 * byte mask name meaning
64 * ---- ---- ------- ------------
65 * 1 0x01 adjustable threshold capacitive button sensitivity
66 * can be adjusted
67 * 1 0x02 report max query 0x0d gives max coord reported
68 * 1 0x04 clearpad sensor is ClearPad product
69 * 1 0x08 advanced gesture not particularly meaningful
70 * 1 0x10 clickpad bit 0 1-button ClickPad
71 * 1 0x60 multifinger mode identifies firmware finger counting
72 * (not reporting!) algorithm.
73 * Not particularly meaningful
74 * 1 0x80 covered pad W clipped to 14, 15 == pad mostly covered
75 * 2 0x01 clickpad bit 1 2-button ClickPad
76 * 2 0x02 deluxe LED controls touchpad support LED commands
77 * ala multimedia control bar
78 * 2 0x04 reduced filtering firmware does less filtering on
79 * position data, driver should watch
80 * for noise.
81 * 2 0x08 image sensor image sensor tracks 5 fingers, but only
82 * reports 2.
83 * 2 0x01 uniform clickpad whole clickpad moves instead of being
84 * hinged at the top.
85 * 2 0x20 report min query 0x0f gives min coord reported
86 */
87#define SYN_CAP_CLICKPAD(ex0c) ((ex0c) & BIT(20)) /* 1-button ClickPad */
88#define SYN_CAP_CLICKPAD2BTN(ex0c) ((ex0c) & BIT(8)) /* 2-button ClickPad */
89#define SYN_CAP_MAX_DIMENSIONS(ex0c) ((ex0c) & BIT(17))
90#define SYN_CAP_MIN_DIMENSIONS(ex0c) ((ex0c) & BIT(13))
91#define SYN_CAP_ADV_GESTURE(ex0c) ((ex0c) & BIT(19))
92#define SYN_CAP_REDUCED_FILTERING(ex0c) ((ex0c) & BIT(10))
93#define SYN_CAP_IMAGE_SENSOR(ex0c) ((ex0c) & BIT(11))
94#define SYN_CAP_INTERTOUCH(ex0c) ((ex0c) & BIT(14))
95
96/*
97 * The following descibes response for the 0x10 query.
98 *
99 * byte mask name meaning
100 * ---- ---- ------- ------------
101 * 1 0x01 ext buttons are stick buttons exported in the extended
102 * capability are actually meant to be used
103 * by the tracktick (pass-through).
104 * 1 0x02 SecurePad the touchpad is a SecurePad, so it
105 * contains a built-in fingerprint reader.
106 * 1 0xe0 more ext count how many more extented queries are
107 * available after this one.
108 * 2 0xff SecurePad width the width of the SecurePad fingerprint
109 * reader.
110 * 3 0xff SecurePad height the height of the SecurePad fingerprint
111 * reader.
112 */
113#define SYN_CAP_EXT_BUTTONS_STICK(ex10) ((ex10) & BIT(16))
114#define SYN_CAP_SECUREPAD(ex10) ((ex10) & BIT(17))
115
116#define SYN_EXT_BUTTON_STICK_L(eb) (((eb) & BIT(0)) >> 0)
117#define SYN_EXT_BUTTON_STICK_M(eb) (((eb) & BIT(1)) >> 1)
118#define SYN_EXT_BUTTON_STICK_R(eb) (((eb) & BIT(2)) >> 2)

120/* synaptics modes query bits */
121#define SYN_MODE_ABSOLUTE(m) ((m) & BIT(7))
122#define SYN_MODE_RATE(m) ((m) & BIT(6))
123#define SYN_MODE_BAUD_SLEEP(m) ((m) & BIT(3))
124#define SYN_MODE_DISABLE_GESTURE(m) ((m) & BIT(2))
125#define SYN_MODE_PACKSIZE(m) ((m) & BIT(1))
126#define SYN_MODE_WMODE(m) ((m) & BIT(0))
127
128/* synaptics identify query bits */
129#define SYN_ID_MODEL(i) (((i) & GENMASK(7, 4)) >> 4)
130#define SYN_ID_MAJOR(i) (((i) & GENMASK(3, 0)) >> 0)
131#define SYN_ID_MINOR(i) (((i) & GENMASK(23, 16)) >> 16)
132#define SYN_ID_FULL(i) ((SYN_ID_MAJOR(i) << 8) | SYN_ID_MINOR(i))
133#define SYN_ID_IS_SYNAPTICS(i) (((i) & GENMASK(15, 8)) == 0x004700U)
134#define SYN_ID_DISGEST_SUPPORTED(i) (SYN_ID_MAJOR(i) >= 4)

151/*
152 * A structure to describe the state of the touchpad hardware (buttons and pad)
153 */
154struct synaptics_hw_state {
155 int x;
156 int y;
157 int z;
158 int w;
159 unsigned int left:1;
160 unsigned int right:1;
161 unsigned int middle:1;
162 unsigned int up:1;
163 unsigned int down:1;
164 u8 ext_buttons;
165 s8 scroll;
166};
167
168/* Data read from the touchpad */
169struct synaptics_device_info {
170 u32 model_id; /* Model-ID */
171 u32 firmware_id; /* Firmware-ID */
172 u32 board_id; /* Board-ID */
173 u32 capabilities; /* Capabilities */
174 u32 ext_cap; /* Extended Capabilities */
175 u32 ext_cap_0c; /* Ext Caps from 0x0c query */
176 u32 ext_cap_10; /* Ext Caps from 0x10 query */
177 u32 identity; /* Identification */
178 u32 x_res, y_res; /* X/Y resolution in units/mm */
179 u32 x_max, y_max; /* Max coordinates (from FW) */
180 u32 x_min, y_min; /* Min coordinates (from FW) */
181};
182
183struct synaptics_data {
184 struct synaptics_device_info info;
185
186 enum synaptics_pkt_type pkt_type; /* packet type - old, new, etc */
187 u8 mode; /* current mode byte */
188 int scroll;
189
190 bool absolute_mode; /* run in Absolute mode */
191 bool disable_gesture; /* disable gestures */
192
193 struct serio *pt_port; /* Pass-through serio port */
194
195 /*
196 * Last received Advanced Gesture Mode (AGM) packet. An AGM packet
197 * contains position data for a second contact, at half resolution.
198 */
199 struct synaptics_hw_state agm;
200 unsigned int agm_count; /* finger count reported by agm */
201
202 /* ForcePad handling */
203 unsigned long press_start;
204 bool press;
205 bool report_press;
206 bool is_forcepad;
207};
