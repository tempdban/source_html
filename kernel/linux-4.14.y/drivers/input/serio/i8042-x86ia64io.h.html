17
18#define I8042_KBD_PHYS_DESC "isa0060/serio0"
19#define I8042_AUX_PHYS_DESC "isa0060/serio1"
20#define I8042_MUX_PHYS_DESC "isa0060/serio%d"
21
22/*
23 * IRQs.
24 */
25
26#if defined(__ia64__)
27# define I8042_MAP_IRQ(x) isa_irq_to_vector((x))
28#else
29# define I8042_MAP_IRQ(x) (x)
30#endif
31
32#define I8042_KBD_IRQ i8042_kbd_irq
33#define I8042_AUX_IRQ i8042_aux_irq
34
35static int i8042_kbd_irq;
36static int i8042_aux_irq;
37
38/*
39 * Register numbers.
40 */
41
42#define I8042_COMMAND_REG i8042_command_reg
43#define I8042_STATUS_REG i8042_command_reg
44#define I8042_DATA_REG i8042_data_reg
45
46static int i8042_command_reg = 0x64;
47static int i8042_data_reg = 0x60;
48
49
50static inline int i8042_read_data(void)
51{
52 return inb(I8042_DATA_REG);
53}
54
55static inline int i8042_read_status(void)
56{
57 return inb(I8042_STATUS_REG);
58}
59
60static inline void i8042_write_data(int val)
61{
62 outb(val, I8042_DATA_REG);
63}
64
65static inline void i8042_write_command(int val)
66{
67 outb(val, I8042_COMMAND_REG);
68}

909static int i8042_pnp_kbd_probe(struct pnp_dev *dev, const struct pnp_device_id *did)
910{
911 if (pnp_port_valid(dev, 0) && pnp_port_len(dev, 0) == 1)
912 i8042_pnp_data_reg = pnp_port_start(dev,0);
913
914 if (pnp_port_valid(dev, 1) && pnp_port_len(dev, 1) == 1)
915 i8042_pnp_command_reg = pnp_port_start(dev, 1);
916
917 if (pnp_irq_valid(dev,0))
918 i8042_pnp_kbd_irq = pnp_irq(dev, 0);
919
920 strlcpy(i8042_pnp_kbd_name, did->id, sizeof(i8042_pnp_kbd_name));
921 if (strlen(pnp_dev_name(dev))) {
922 strlcat(i8042_pnp_kbd_name, ":", sizeof(i8042_pnp_kbd_name));
923 strlcat(i8042_pnp_kbd_name, pnp_dev_name(dev), sizeof(i8042_pnp_kbd_name));
924 }
925 i8042_pnp_id_to_string(dev->id, i8042_kbd_firmware_id,
926 sizeof(i8042_kbd_firmware_id));
927
928 /* Keyboard ports are always supposed to be wakeup-enabled */
929 device_set_wakeup_enable(&dev->dev, true);
930
931 i8042_pnp_kbd_devices++;
932 return 0;
933}

1027static int __init i8042_pnp_init(void)
1028{
1029 char kbd_irq_str[4] = { 0 }, aux_irq_str[4] = { 0 };
1030 bool pnp_data_busted = false;
1031 int err;
1032
1033#ifdef CONFIG_X86
1034 if (dmi_check_system(i8042_dmi_nopnp_table))
1035 i8042_nopnp = true;
1036#endif
1037
1038 if (i8042_nopnp) {
1039 pr_info("PNP detection disabled\n");
1040 return 0;
1041 }
1042
1043 err = pnp_register_driver(&i8042_pnp_kbd_driver);
1044 if (!err)
1045 i8042_pnp_kbd_registered = true;
1046
1047 err = pnp_register_driver(&i8042_pnp_aux_driver);
1048 if (!err)
1049 i8042_pnp_aux_registered = true;
1050
1051 if (!i8042_pnp_kbd_devices && !i8042_pnp_aux_devices) {
1052 i8042_pnp_exit();
1053#if defined(__ia64__)
1054 return -ENODEV;
1055#else
1056 pr_info("PNP: No PS/2 controller found.\n");
1057 if (x86_platform.legacy.i8042 !=
1058 X86_LEGACY_I8042_EXPECTED_PRESENT)
1059 return -ENODEV;
1060 pr_info("Probing ports directly.\n");
1061 return 0;
1062#endif
1063 }
1064
1065 if (i8042_pnp_kbd_devices)
1066 snprintf(kbd_irq_str, sizeof(kbd_irq_str),
1067 "%d", i8042_pnp_kbd_irq);
1068 if (i8042_pnp_aux_devices)
1069 snprintf(aux_irq_str, sizeof(aux_irq_str),
1070 "%d", i8042_pnp_aux_irq);
1071
1072 pr_info("PNP: PS/2 Controller [%s%s%s] at %#x,%#x irq %s%s%s\n",
1073 i8042_pnp_kbd_name, (i8042_pnp_kbd_devices && i8042_pnp_aux_devices) ? "," : "",
1074 i8042_pnp_aux_name,
1075 i8042_pnp_data_reg, i8042_pnp_command_reg,
1076 kbd_irq_str, (i8042_pnp_kbd_devices && i8042_pnp_aux_devices) ? "," : "",
1077 aux_irq_str);
1078
1079#if defined(__ia64__)
1080 if (!i8042_pnp_kbd_devices)
1081 i8042_nokbd = true;
1082 if (!i8042_pnp_aux_devices)
1083 i8042_noaux = true;
1084#endif
1085
1086 if (((i8042_pnp_data_reg & ~0xf) == (i8042_data_reg & ~0xf) &&
1087 i8042_pnp_data_reg != i8042_data_reg) ||
1088 !i8042_pnp_data_reg) {
1089 pr_warn("PNP: PS/2 controller has invalid data port %#x; using default %#x\n",
1090 i8042_pnp_data_reg, i8042_data_reg);
1091 i8042_pnp_data_reg = i8042_data_reg;
1092 pnp_data_busted = true;
1093 }
1094
1095 if (((i8042_pnp_command_reg & ~0xf) == (i8042_command_reg & ~0xf) &&
1096 i8042_pnp_command_reg != i8042_command_reg) ||
1097 !i8042_pnp_command_reg) {
1098 pr_warn("PNP: PS/2 controller has invalid command port %#x; using default %#x\n",
1099 i8042_pnp_command_reg, i8042_command_reg);
1100 i8042_pnp_command_reg = i8042_command_reg;
1101 pnp_data_busted = true;
1102 }
1103
1104 if (!i8042_nokbd && !i8042_pnp_kbd_irq) {
1105 pr_warn("PNP: PS/2 controller doesn't have KBD irq; using default %d\n",
1106 i8042_kbd_irq);
1107 i8042_pnp_kbd_irq = i8042_kbd_irq;
1108 pnp_data_busted = true;
1109 }
1110
1111 if (!i8042_noaux && !i8042_pnp_aux_irq) {
1112 if (!pnp_data_busted && i8042_pnp_kbd_irq) {
1113 pr_warn("PNP: PS/2 appears to have AUX port disabled, "
1114 "if this is incorrect please boot with i8042.nopnp\n");
1115 i8042_noaux = true;
1116 } else {
1117 pr_warn("PNP: PS/2 controller doesn't have AUX irq; using default %d\n",
1118 i8042_aux_irq);
1119 i8042_pnp_aux_irq = i8042_aux_irq;
1120 }
1121 }
1122
1123 i8042_data_reg = i8042_pnp_data_reg;
1124 i8042_command_reg = i8042_pnp_command_reg;
1125 i8042_kbd_irq = i8042_pnp_kbd_irq;
1126 i8042_aux_irq = i8042_pnp_aux_irq;
1127
1128#ifdef CONFIG_X86
1129 i8042_bypass_aux_irq_test = !pnp_data_busted &&
1130 dmi_check_system(i8042_dmi_laptop_table);
1131#endif
1132
1133 return 0;
1134}

1141static int __init i8042_platform_init(void)
1142{
1143 int retval;
1144
1145#ifdef CONFIG_X86
1146 u8 a20_on = 0xdf;
1147 /* Just return if platform does not have i8042 controller */
1148 if (x86_platform.legacy.i8042 == X86_LEGACY_I8042_PLATFORM_ABSENT)
1149 return -ENODEV;
1150#endif
1151
1152/*
1153 * On ix86 platforms touching the i8042 data register region can do really
1154 * bad things. Because of this the region is always reserved on ix86 boxes.
1155 *
1156 * if (!request_region(I8042_DATA_REG, 16, "i8042"))
1157 * return -EBUSY;
1158 */
1159
1160 i8042_kbd_irq = I8042_MAP_IRQ(1);
1161 i8042_aux_irq = I8042_MAP_IRQ(12);
1162
1163 retval = i8042_pnp_init();
1164 if (retval)
1165 return retval;
1166
1167#if defined(__ia64__)
1168 i8042_reset = I8042_RESET_ALWAYS;
1169#endif
1170
1171#ifdef CONFIG_X86
1172 /* Honor module parameter when value is not default */
1173 if (i8042_reset == I8042_RESET_DEFAULT) {
1174 if (dmi_check_system(i8042_dmi_reset_table))
1175 i8042_reset = I8042_RESET_ALWAYS;
1176
1177 if (dmi_check_system(i8042_dmi_noselftest_table))
1178 i8042_reset = I8042_RESET_NEVER;
1179 }
1180
1181 if (dmi_check_system(i8042_dmi_noloop_table))
1182 i8042_noloop = true;
1183
1184 if (dmi_check_system(i8042_dmi_nomux_table))
1185 i8042_nomux = true;
1186
1187 if (dmi_check_system(i8042_dmi_forcemux_table))
1188 i8042_nomux = false;
1189
1190 if (dmi_check_system(i8042_dmi_notimeout_table))
1191 i8042_notimeout = true;
1192
1193 if (dmi_check_system(i8042_dmi_dritek_table))
1194 i8042_dritek = true;
1195
1196 if (dmi_check_system(i8042_dmi_kbdreset_table))
1197 i8042_kbdreset = true;
1198
1199 /*
1200 * A20 was already enabled during early kernel init. But some buggy
1201 * BIOSes (in MSI Laptops) require A20 to be enabled using 8042 to
1202 * resume from S3. So we do it here and hope that nothing breaks.
1203 */
1204 i8042_command(&a20_on, 0x10d1);
1205 i8042_command(NULL, 0x00ff); /* Null command for SMM firmware */
1206#endif /* CONFIG_X86 */
1207
1208 return retval;
1209}

1213 i8042_pnp_exit();
1214}
