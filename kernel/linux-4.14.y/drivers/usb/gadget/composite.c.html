56/**
57 * function_descriptors() - get function descriptors for speed
58 * @f: the function
59 * @speed: the speed
60 *
61 * Returns the descriptors or NULL if not set.
62 */
63static struct usb_descriptor_header **
64function_descriptors(struct usb_function *f,
65 enum usb_device_speed speed)
66{
67 struct usb_descriptor_header **descriptors;
68
69 /*
70 * NOTE: we try to help gadget drivers which might not be setting
71 * max_speed appropriately.
72 */
73
74 switch (speed) {
75 case USB_SPEED_SUPER_PLUS:
76 descriptors = f->ssp_descriptors;
77 if (descriptors)
78 break;
79 /* FALLTHROUGH */
80 case USB_SPEED_SUPER:
81 descriptors = f->ss_descriptors;
82 if (descriptors)
83 break;
84 /* FALLTHROUGH */
85 case USB_SPEED_HIGH:
86 descriptors = f->hs_descriptors;
87 if (descriptors)
88 break;
89 /* FALLTHROUGH */
90 default:
91 descriptors = f->fs_descriptors;
92 }
93
94 /*
95 * if we can't find any descriptors at all, then this gadget deserves to
96 * Oops with a NULL pointer dereference
97 */
98
99 return descriptors;
100}

102/**
103 * next_ep_desc() - advance to the next EP descriptor
104 * @t: currect pointer within descriptor array
105 *
106 * Return: next EP descriptor or NULL
107 *
108 * Iterate over @t until either EP descriptor found or
109 * NULL (that indicates end of list) encountered
110 */
111static struct usb_descriptor_header**
112next_ep_desc(struct usb_descriptor_header **t)
113{
114 for (; *t; t++) {
115 if ((*t)->bDescriptorType == USB_DT_ENDPOINT)
116 return t;
117 }
118 return NULL;
119}

121/*
122 * for_each_ep_desc()- iterate over endpoint descriptors in the
123 * descriptors list
124 * @start: pointer within descriptor array.
125 * @ep_desc: endpoint descriptor to use as the loop cursor
126 */
127#define for_each_ep_desc(start, ep_desc) \
128 for (ep_desc = next_ep_desc(start); \
129 ep_desc; ep_desc = next_ep_desc(ep_desc+1))

131/**
132 * config_ep_by_speed() - configures the given endpoint
133 * according to gadget speed.
134 * @g: pointer to the gadget
135 * @f: usb function
136 * @_ep: the endpoint to configure
137 *
138 * Return: error code, 0 on success
139 *
140 * This function chooses the right descriptors for a given
141 * endpoint according to gadget speed and saves it in the
142 * endpoint desc field. If the endpoint already has a descriptor
143 * assigned to it - overwrites it with currently corresponding
144 * descriptor. The endpoint maxpacket field is updated according
145 * to the chosen descriptor.
146 * Note: the supplied function should hold all the descriptors
147 * for supported speeds
148 */
149int config_ep_by_speed(struct usb_gadget *g,
150 struct usb_function *f,
151 struct usb_ep *_ep)
152{
153 struct usb_endpoint_descriptor *chosen_desc = NULL;
154 struct usb_descriptor_header **speed_desc = NULL;
155
156 struct usb_ss_ep_comp_descriptor *comp_desc = NULL;
157 int want_comp_desc = 0;
158
159 struct usb_descriptor_header **d_spd; /* cursor for speed desc */
160
161 if (!g || !f || !_ep)
162 return -EIO;
163
164 /* select desired speed */
165 switch (g->speed) {
166 case USB_SPEED_SUPER_PLUS:
167 if (gadget_is_superspeed_plus(g)) {
168 speed_desc = f->ssp_descriptors;
169 want_comp_desc = 1;
170 break;
171 }
172 /* else: Fall trough */
173 case USB_SPEED_SUPER:
174 if (gadget_is_superspeed(g)) {
175 speed_desc = f->ss_descriptors;
176 want_comp_desc = 1;
177 break;
178 }
179 /* else: Fall trough */
180 case USB_SPEED_HIGH:
181 if (gadget_is_dualspeed(g)) {
182 speed_desc = f->hs_descriptors;
183 break;
184 }
185 /* else: fall through */
186 default:
187 speed_desc = f->fs_descriptors;
188 }
189 /* find descriptors */
190 for_each_ep_desc(speed_desc, d_spd) {
191 chosen_desc = (struct usb_endpoint_descriptor *)*d_spd;
192 if (chosen_desc->bEndpointAddress == _ep->address)
193 goto ep_found;
194 }
195 return -EIO;
196
197ep_found:
198 /* commit results */
199 _ep->maxpacket = usb_endpoint_maxp(chosen_desc);
200 _ep->desc = chosen_desc;
201 _ep->comp_desc = NULL;
202 _ep->maxburst = 0;
203 _ep->mult = 1;
204
205 if (g->speed == USB_SPEED_HIGH && (usb_endpoint_xfer_isoc(_ep->desc) ||
206 usb_endpoint_xfer_int(_ep->desc)))
207 _ep->mult = usb_endpoint_maxp_mult(_ep->desc);
208
209 if (!want_comp_desc)
210 return 0;
211
212 /*
213 * Companion descriptor should follow EP descriptor
214 * USB 3.0 spec, #9.6.7
215 */
216 comp_desc = (struct usb_ss_ep_comp_descriptor *)*(++d_spd);
217 if (!comp_desc ||
218 (comp_desc->bDescriptorType != USB_DT_SS_ENDPOINT_COMP))
219 return -EIO;
220 _ep->comp_desc = comp_desc;
221 if (g->speed >= USB_SPEED_SUPER) {
222 switch (usb_endpoint_type(_ep->desc)) {
223 case USB_ENDPOINT_XFER_ISOC:
224 /* mult: bits 1:0 of bmAttributes */
225 _ep->mult = (comp_desc->bmAttributes & 0x3) + 1;
226 case USB_ENDPOINT_XFER_BULK:
227 case USB_ENDPOINT_XFER_INT:
228 _ep->maxburst = comp_desc->bMaxBurst + 1;
229 break;
230 default:
231 if (comp_desc->bMaxBurst != 0) {
232 struct usb_composite_dev *cdev;
233
234 cdev = get_gadget_data(g);
235 ERROR(cdev, "ep0 bMaxBurst must be 0\n");
236 }
237 _ep->maxburst = 1;
238 break;
239 }
240 }
241 return 0;
242}
243EXPORT_SYMBOL_GPL(config_ep_by_speed);

245/**
246 * usb_add_function() - add a function to a configuration
247 * @config: the configuration
248 * @function: the function being added
249 * Context: single threaded during gadget setup
250 *
251 * After initialization, each configuration must have one or more
252 * functions added to it. Adding a function involves calling its @bind()
253 * method to allocate resources such as interface and string identifiers
254 * and endpoints.
255 *
256 * This function returns the value of the function's bind(), which is
257 * zero for success else a negative errno value.
258 */
259int usb_add_function(struct usb_configuration *config,
260 struct usb_function *function)
261{
262 int value = -EINVAL;
263
264 DBG(config->cdev, "adding '%s'/%p to config '%s'/%p\n",
265 function->name, function,
266 config->label, config);
267
268 if (!function->set_alt || !function->disable)
269 goto done;
270
271 function->config = config;
272 list_add_tail(&function->list, &config->functions);
273
274 if (function->bind_deactivated) {
275 value = usb_function_deactivate(function);
276 if (value)
277 goto done;
278 }
279
280 /* REVISIT *require* function->bind? */
281 if (function->bind) {
282 value = function->bind(config, function);
283 if (value < 0) {
284 list_del(&function->list);
285 function->config = NULL;
286 }
287 } else
288 value = 0;
289
290 /* We allow configurations that don't work at both speeds.
291 * If we run into a lowspeed Linux system, treat it the same
292 * as full speed ... it's the function drivers that will need
293 * to avoid bulk and ISO transfers.
294 */
295 if (!config->fullspeed && function->fs_descriptors)
296 config->fullspeed = true;
297 if (!config->highspeed && function->hs_descriptors)
298 config->highspeed = true;
299 if (!config->superspeed && function->ss_descriptors)
300 config->superspeed = true;
301 if (!config->superspeed_plus && function->ssp_descriptors)
302 config->superspeed_plus = true;
303
304done:
305 if (value)
306 DBG(config->cdev, "adding '%s'/%p --> %d\n",
307 function->name, function, value);
308 return value;
309}
310EXPORT_SYMBOL_GPL(usb_add_function);

314 if (f->disable)

327/**
328 * usb_function_deactivate - prevent function and gadget enumeration
329 * @function: the function that isn't yet ready to respond
330 *
331 * Blocks response of the gadget driver to host enumeration by
332 * preventing the data line pullup from being activated. This is
333 * normally called during @bind() processing to change from the
334 * initial "ready to respond" state, or when a required resource
335 * becomes available.
336 *
337 * For example, drivers that serve as a passthrough to a userspace
338 * daemon can block enumeration unless that daemon (such as an OBEX,
339 * MTP, or print server) is ready to handle host requests.
340 *
341 * Not all systems support software control of their USB peripheral
342 * data pullups.
343 *
344 * Returns zero on success, else negative errno.
345 */
346int usb_function_deactivate(struct usb_function *function)
347{
348 struct usb_composite_dev *cdev = function->config->cdev;
349 unsigned long flags;
350 int status = 0;
351
352 spin_lock_irqsave(&cdev->lock, flags);
353
354 if (cdev->deactivations == 0)
355 status = usb_gadget_deactivate(cdev->gadget);
356 if (status == 0)
357 cdev->deactivations++;
358
359 spin_unlock_irqrestore(&cdev->lock, flags);
360 return status;
361}
362EXPORT_SYMBOL_GPL(usb_function_deactivate);

364/**
365 * usb_function_activate - allow function and gadget enumeration
366 * @function: function on which usb_function_activate() was called
367 *
368 * Reverses effect of usb_function_deactivate(). If no more functions
369 * are delaying their activation, the gadget driver will respond to
370 * host enumeration procedures.
371 *
372 * Returns zero on success, else negative errno.
373 */
374int usb_function_activate(struct usb_function *function)
375{
376 struct usb_composite_dev *cdev = function->config->cdev;
377 unsigned long flags;
378 int status = 0;
379
380 spin_lock_irqsave(&cdev->lock, flags);
381
382 if (WARN_ON(cdev->deactivations == 0))
383 status = -EINVAL;
384 else {
385 cdev->deactivations--;
386 if (cdev->deactivations == 0)
387 status = usb_gadget_activate(cdev->gadget);
388 }
389
390 spin_unlock_irqrestore(&cdev->lock, flags);
391 return status;
392}
393EXPORT_SYMBOL_GPL(usb_function_activate);

395/**
396 * usb_interface_id() - allocate an unused interface ID
397 * @config: configuration associated with the interface
398 * @function: function handling the interface
399 * Context: single threaded during gadget setup
400 *
401 * usb_interface_id() is called from usb_function.bind() callbacks to
402 * allocate new interface IDs. The function driver will then store that
403 * ID in interface, association, CDC union, and other descriptors. It
404 * will also handle any control requests targeted at that interface,
405 * particularly changing its altsetting via set_alt(). There may
406 * also be class-specific or vendor-specific requests to handle.
407 *
408 * All interface identifier should be allocated using this routine, to
409 * ensure that for example different functions don't wrongly assign
410 * different meanings to the same identifier. Note that since interface
411 * identifiers are configuration-specific, functions used in more than
412 * one configuration (or more than once in a given configuration) need
413 * multiple versions of the relevant descriptors.
414 *
415 * Returns the interface ID which was allocated; or -ENODEV if no
416 * more interface IDs can be allocated.
417 */
418int usb_interface_id(struct usb_configuration *config,
419 struct usb_function *function)
420{
421 unsigned id = config->next_interface_id;
422
423 if (id < MAX_CONFIG_INTERFACES) {
424 config->interface[id] = function;
425 config->next_interface_id = id + 1;
426 return id;
427 }
428 return -ENODEV;
429}
430EXPORT_SYMBOL_GPL(usb_interface_id);

432static u8 encode_bMaxPower(enum usb_device_speed speed,
433 struct usb_configuration *c)
434{
435 unsigned val;
436
437 if (c->MaxPower)
438 val = c->MaxPower;
439 else
440 val = CONFIG_USB_GADGET_VBUS_DRAW;
441 if (!val)
442 return 0;
443 switch (speed) {
444 case USB_SPEED_SUPER:
445 return DIV_ROUND_UP(val, 8);
446 default:
447 return DIV_ROUND_UP(val, 2);
448 }
449}

451static int config_buf(struct usb_configuration *config,
452 enum usb_device_speed speed, void *buf, u8 type)
453{
454 struct usb_config_descriptor *c = buf;
455 void *next = buf + USB_DT_CONFIG_SIZE;
456 int len;
457 struct usb_function *f;
458 int status;
459
460 len = USB_COMP_EP0_BUFSIZ - USB_DT_CONFIG_SIZE;
461 /* write the config descriptor */
462 c = buf;
463 c->bLength = USB_DT_CONFIG_SIZE;
464 c->bDescriptorType = type;
465 /* wTotalLength is written later */
466 c->bNumInterfaces = config->next_interface_id;
467 c->bConfigurationValue = config->bConfigurationValue;
468 c->iConfiguration = config->iConfiguration;
469 c->bmAttributes = USB_CONFIG_ATT_ONE | config->bmAttributes;
470 c->bMaxPower = encode_bMaxPower(speed, config);
471
472 /* There may be e.g. OTG descriptors */
473 if (config->descriptors) {
474 status = usb_descriptor_fillbuf(next, len,
475 config->descriptors);
476 if (status < 0)
477 return status;
478 len -= status;
479 next += status;
480 }
481
482 /* add each function's descriptors */
483 list_for_each_entry(f, &config->functions, list) {
484 struct usb_descriptor_header **descriptors;
485
486 descriptors = function_descriptors(f, speed);
487 if (!descriptors)
488 continue;
489 status = usb_descriptor_fillbuf(next, len,
490 (const struct usb_descriptor_header **) descriptors);
491 if (status < 0)
492 return status;
493 len -= status;
494 next += status;
495 }
496
497 len = next - buf;
498 c->wTotalLength = cpu_to_le16(len);
499 return len;
500}

502static int config_desc(struct usb_composite_dev *cdev, unsigned w_value)
503{
504 struct usb_gadget *gadget = cdev->gadget;
505 struct usb_configuration *c;
506 struct list_head *pos;
507 u8 type = w_value >> 8;
508 enum usb_device_speed speed = USB_SPEED_UNKNOWN;
509
510 if (gadget->speed >= USB_SPEED_SUPER)
511 speed = gadget->speed;
512 else if (gadget_is_dualspeed(gadget)) {
513 int hs = 0;
514 if (gadget->speed == USB_SPEED_HIGH)
515 hs = 1;
516 if (type == USB_DT_OTHER_SPEED_CONFIG)
517 hs = !hs;
518 if (hs)
519 speed = USB_SPEED_HIGH;
520
521 }
522
523 /* This is a lookup by config *INDEX* */
524 w_value &= 0xff;
525
526 pos = &cdev->configs;
527 c = cdev->os_desc_config;
528 if (c)
529 goto check_config;
530
531 while ((pos = pos->next) != &cdev->configs) {
532 c = list_entry(pos, typeof(*c), list);
533
534 /* skip OS Descriptors config which is handled separately */
535 if (c == cdev->os_desc_config)
536 continue;
537
538check_config:
539 /* ignore configs that won't work at this speed */
540 switch (speed) {
541 case USB_SPEED_SUPER_PLUS:
542 if (!c->superspeed_plus)
543 continue;
544 break;
545 case USB_SPEED_SUPER:
546 if (!c->superspeed)
547 continue;
548 break;
549 case USB_SPEED_HIGH:
550 if (!c->highspeed)
551 continue;
552 break;
553 default:
554 if (!c->fullspeed)
555 continue;
556 }
557
558 if (w_value == 0)
559 return config_buf(c, speed, cdev->req->buf, type);
560 w_value--;
561 }
562 return -EINVAL;
563}

565static int count_configs(struct usb_composite_dev *cdev, unsigned type)
566{
567 struct usb_gadget *gadget = cdev->gadget;
568 struct usb_configuration *c;
569 unsigned count = 0;
570 int hs = 0;
571 int ss = 0;
572 int ssp = 0;
573
574 if (gadget_is_dualspeed(gadget)) {
575 if (gadget->speed == USB_SPEED_HIGH)
576 hs = 1;
577 if (gadget->speed == USB_SPEED_SUPER)
578 ss = 1;
579 if (gadget->speed == USB_SPEED_SUPER_PLUS)
580 ssp = 1;
581 if (type == USB_DT_DEVICE_QUALIFIER)
582 hs = !hs;
583 }
584 list_for_each_entry(c, &cdev->configs, list) {
585 /* ignore configs that won't work at this speed */
586 if (ssp) {
587 if (!c->superspeed_plus)
588 continue;
589 } else if (ss) {
590 if (!c->superspeed)
591 continue;
592 } else if (hs) {
593 if (!c->highspeed)
594 continue;
595 } else {
596 if (!c->fullspeed)
597 continue;
598 }
599 count++;
600 }
601 return count;
602}

604/**
605 * bos_desc() - prepares the BOS descriptor.
606 * @cdev: pointer to usb_composite device to generate the bos
607 * descriptor for
608 *
609 * This function generates the BOS (Binary Device Object)
610 * descriptor and its device capabilities descriptors. The BOS
611 * descriptor should be supported by a SuperSpeed device.
612 */
613static int bos_desc(struct usb_composite_dev *cdev)
614{
615 struct usb_ext_cap_descriptor *usb_ext;
616 struct usb_dcd_config_params dcd_config_params;
617 struct usb_bos_descriptor *bos = cdev->req->buf;
618
619 bos->bLength = USB_DT_BOS_SIZE;
620 bos->bDescriptorType = USB_DT_BOS;
621
622 bos->wTotalLength = cpu_to_le16(USB_DT_BOS_SIZE);
623 bos->bNumDeviceCaps = 0;
624
625 /*
626 * A SuperSpeed device shall include the USB2.0 extension descriptor
627 * and shall support LPM when operating in USB2.0 HS mode.
628 */
629 usb_ext = cdev->req->buf + le16_to_cpu(bos->wTotalLength);
630 bos->bNumDeviceCaps++;
631 le16_add_cpu(&bos->wTotalLength, USB_DT_USB_EXT_CAP_SIZE);
632 usb_ext->bLength = USB_DT_USB_EXT_CAP_SIZE;
633 usb_ext->bDescriptorType = USB_DT_DEVICE_CAPABILITY;
634 usb_ext->bDevCapabilityType = USB_CAP_TYPE_EXT;
635 usb_ext->bmAttributes = cpu_to_le32(USB_LPM_SUPPORT | USB_BESL_SUPPORT);
636
637 /*
638 * The Superspeed USB Capability descriptor shall be implemented by all
639 * SuperSpeed devices.
640 */
641 if (gadget_is_superspeed(cdev->gadget)) {
642 struct usb_ss_cap_descriptor *ss_cap;
643
644 ss_cap = cdev->req->buf + le16_to_cpu(bos->wTotalLength);
645 bos->bNumDeviceCaps++;
646 le16_add_cpu(&bos->wTotalLength, USB_DT_USB_SS_CAP_SIZE);
647 ss_cap->bLength = USB_DT_USB_SS_CAP_SIZE;
648 ss_cap->bDescriptorType = USB_DT_DEVICE_CAPABILITY;
649 ss_cap->bDevCapabilityType = USB_SS_CAP_TYPE;
650 ss_cap->bmAttributes = 0; /* LTM is not supported yet */
651 ss_cap->wSpeedSupported = cpu_to_le16(USB_LOW_SPEED_OPERATION |
652 USB_FULL_SPEED_OPERATION |
653 USB_HIGH_SPEED_OPERATION |
654 USB_5GBPS_OPERATION);
655 ss_cap->bFunctionalitySupport = USB_LOW_SPEED_OPERATION;
656
657 /* Get Controller configuration */
658 if (cdev->gadget->ops->get_config_params) {
659 cdev->gadget->ops->get_config_params(
660 &dcd_config_params);
661 } else {
662 dcd_config_params.bU1devExitLat =
663 USB_DEFAULT_U1_DEV_EXIT_LAT;
664 dcd_config_params.bU2DevExitLat =
665 cpu_to_le16(USB_DEFAULT_U2_DEV_EXIT_LAT);
666 }
667 ss_cap->bU1devExitLat = dcd_config_params.bU1devExitLat;
668 ss_cap->bU2DevExitLat = dcd_config_params.bU2DevExitLat;
669 }
670
671 /* The SuperSpeedPlus USB Device Capability descriptor */
672 if (gadget_is_superspeed_plus(cdev->gadget)) {
673 struct usb_ssp_cap_descriptor *ssp_cap;
674
675 ssp_cap = cdev->req->buf + le16_to_cpu(bos->wTotalLength);
676 bos->bNumDeviceCaps++;
677
678 /*
679 * Report typical values.
680 */
681
682 le16_add_cpu(&bos->wTotalLength, USB_DT_USB_SSP_CAP_SIZE(1));
683 ssp_cap->bLength = USB_DT_USB_SSP_CAP_SIZE(1);
684 ssp_cap->bDescriptorType = USB_DT_DEVICE_CAPABILITY;
685 ssp_cap->bDevCapabilityType = USB_SSP_CAP_TYPE;
686 ssp_cap->bReserved = 0;
687 ssp_cap->wReserved = 0;
688
689 /* SSAC = 1 (2 attributes) */
690 ssp_cap->bmAttributes = cpu_to_le32(1);
691
692 /* Min RX/TX Lane Count = 1 */
693 ssp_cap->wFunctionalitySupport =
694 cpu_to_le16((1 << 8) | (1 << 12));
695
696 /*
697 * bmSublinkSpeedAttr[0]:
698 * ST = Symmetric, RX
699 * LSE = 3 (Gbps)
700 * LP = 1 (SuperSpeedPlus)
701 * LSM = 10 (10 Gbps)
702 */
703 ssp_cap->bmSublinkSpeedAttr[0] =
704 cpu_to_le32((3 << 4) | (1 << 14) | (0xa << 16));
705 /*
706 * bmSublinkSpeedAttr[1] =
707 * ST = Symmetric, TX
708 * LSE = 3 (Gbps)
709 * LP = 1 (SuperSpeedPlus)
710 * LSM = 10 (10 Gbps)
711 */
712 ssp_cap->bmSublinkSpeedAttr[1] =
713 cpu_to_le32((3 << 4) | (1 << 14) |
714 (0xa << 16) | (1 << 7));
715 }
716
717 return le16_to_cpu(bos->wTotalLength);
718}

720static void device_qual(struct usb_composite_dev *cdev)
721{
722 struct usb_qualifier_descriptor *qual = cdev->req->buf;
723
724 qual->bLength = sizeof(*qual);
725 qual->bDescriptorType = USB_DT_DEVICE_QUALIFIER;
726 /* POLICY: same bcdUSB and device type info at both speeds */
727 qual->bcdUSB = cdev->desc.bcdUSB;
728 qual->bDeviceClass = cdev->desc.bDeviceClass;
729 qual->bDeviceSubClass = cdev->desc.bDeviceSubClass;
730 qual->bDeviceProtocol = cdev->desc.bDeviceProtocol;
731 /* ASSUME same EP0 fifo size at both speeds */
732 qual->bMaxPacketSize0 = cdev->gadget->ep0->maxpacket;
733 qual->bNumConfigurations = count_configs(cdev, USB_DT_DEVICE_QUALIFIER);
734 qual->bRESERVED = 0;
735}

755static int set_config(struct usb_composite_dev *cdev,
756 const struct usb_ctrlrequest *ctrl, unsigned number)
757{
758 struct usb_gadget *gadget = cdev->gadget;
759 struct usb_configuration *c = NULL;
760 int result = -EINVAL;
761 unsigned power = gadget_is_otg(gadget) ? 8 : 100;
762 int tmp;
763
764 if (number) {
765 list_for_each_entry(c, &cdev->configs, list) {
766 if (c->bConfigurationValue == number) {
767 /*
768 * We disable the FDs of the previous
769 * configuration only if the new configuration
770 * is a valid one
771 */
772 if (cdev->config)
773 reset_config(cdev);
774 result = 0;
775 break;
776 }
777 }
778 if (result < 0)
779 goto done;
780 } else { /* Zero configuration value - need to reset the config */
781 if (cdev->config)
782 reset_config(cdev);
783 result = 0;
784 }
785
786 INFO(cdev, "%s config #%d: %s\n",
787 usb_speed_string(gadget->speed),
788 number, c ? c->label : "unconfigured");
789
790 if (!c)
791 goto done;
792
793 usb_gadget_set_state(gadget, USB_STATE_CONFIGURED);
794 cdev->config = c;
795
796 /* Initialize all interfaces by setting them to altsetting zero. */
797 for (tmp = 0; tmp < MAX_CONFIG_INTERFACES; tmp++) {
798 struct usb_function *f = c->interface[tmp];
799 struct usb_descriptor_header **descriptors;
800
801 if (!f)
802 break;
803
804 /*
805 * Record which endpoints are used by the function. This is used
806 * to dispatch control requests targeted at that endpoint to the
807 * function's setup callback instead of the current
808 * configuration's setup callback.
809 */
810 descriptors = function_descriptors(f, gadget->speed);
811
812 for (; *descriptors; ++descriptors) {
813 struct usb_endpoint_descriptor *ep;
814 int addr;
815
816 if ((*descriptors)->bDescriptorType != USB_DT_ENDPOINT)
817 continue;
818
819 ep = (struct usb_endpoint_descriptor *)*descriptors;
820 addr = ((ep->bEndpointAddress & 0x80) >> 3)
821 | (ep->bEndpointAddress & 0x0f);
822 set_bit(addr, f->endpoints);
823 }
824
825 result = f->set_alt(f, tmp, 0);
826 if (result < 0) {
827 DBG(cdev, "interface %d (%s/%p) alt 0 --> %d\n",
828 tmp, f->name, f, result);
829
830 reset_config(cdev);
831 goto done;
832 }
833
834 if (result == USB_GADGET_DELAYED_STATUS) {
835 DBG(cdev,
836 "%s: interface %d (%s) requested delayed status\n",
837 __func__, tmp, f->name);
838 cdev->delayed_status++;
839 DBG(cdev, "delayed_status count %d\n",
840 cdev->delayed_status);
841 }
842 }
843
844 /* when we return, be sure our power usage is valid */
845 power = c->MaxPower ? c->MaxPower : CONFIG_USB_GADGET_VBUS_DRAW;
846done:
847 usb_gadget_vbus_draw(gadget, power);
848 if (result >= 0 && cdev->delayed_status)
849 result = USB_GADGET_DELAYED_STATUS;
850 return result;
851}

878/**
879 * usb_add_config() - add a configuration to a device.
880 * @cdev: wraps the USB gadget
881 * @config: the configuration, with bConfigurationValue assigned
882 * @bind: the configuration's bind function
883 * Context: single threaded during gadget setup
884 *
885 * One of the main tasks of a composite @bind() routine is to
886 * add each of the configurations it supports, using this routine.
887 *
888 * This function returns the value of the configuration's @bind(), which
889 * is zero for success else a negative errno value. Binding configurations
890 * assigns global resources including string IDs, and per-configuration
891 * resources such as interface IDs and endpoints.
892 */
893int usb_add_config(struct usb_composite_dev *cdev,
894 struct usb_configuration *config,
895 int (*bind)(struct usb_configuration *))
896{
897 int status = -EINVAL;
898
899 if (!bind)
900 goto done;
901
902 DBG(cdev, "adding config #%u '%s'/%p\n",
903 config->bConfigurationValue,
904 config->label, config);
905
906 status = usb_add_config_only(cdev, config);
907 if (status)
908 goto done;
909
910 status = bind(config);
911 if (status < 0) {
912 while (!list_empty(&config->functions)) {
913 struct usb_function *f;
914
915 f = list_first_entry(&config->functions,
916 struct usb_function, list);
917 list_del(&f->list);
918 if (f->unbind) {
919 DBG(cdev, "unbind function '%s'/%p\n",
920 f->name, f);
921 f->unbind(config, f);
922 /* may free memory for "f" */
923 }
924 }
925 list_del(&config->list);
926 config->cdev = NULL;
927 } else {
928 unsigned i;
929
930 DBG(cdev, "cfg %d/%p speeds:%s%s%s%s\n",
931 config->bConfigurationValue, config,
932 config->superspeed_plus ? " superplus" : "",
933 config->superspeed ? " super" : "",
934 config->highspeed ? " high" : "",
935 config->fullspeed
936 ? (gadget_is_dualspeed(cdev->gadget)
937 ? " full"
938 : " full/low")
939 : "");
940
941 for (i = 0; i < MAX_CONFIG_INTERFACES; i++) {
942 struct usb_function *f = config->interface[i];
943
944 if (!f)
945 continue;
946 DBG(cdev, " interface %d = %s/%p\n",
947 i, f->name, f);
948 }
949 }
950
951 /* set_alt(), or next bind(), sets up ep->claimed as needed */
952 usb_ep_autoconfig_reset(cdev->gadget);
953
954done:
955 if (status)
956 DBG(cdev, "added config '%s'/%u --> %d\n", config->label,
957 config->bConfigurationValue, status);
958 return status;
959}
960EXPORT_SYMBOL_GPL(usb_add_config);

983 * @cdev: wraps the USB gadget
984 * @config: the configuration

1007/* We support strings in multiple languages ... string descriptor zero
1008 * says which languages are supported. The typical case will be that
1009 * only one language (probably English) is used, with i18n handled on
1010 * the host side.
1011 */
1012
1013static void collect_langs(struct usb_gadget_strings **sp, __le16 *buf)
1014{
1015 const struct usb_gadget_strings *s;
1016 __le16 language;
1017 __le16 *tmp;
1018
1019 while (*sp) {
1020 s = *sp;
1021 language = cpu_to_le16(s->language);
1022 for (tmp = buf; *tmp && tmp < &buf[126]; tmp++) {
1023 if (*tmp == language)
1024 goto repeat;
1025 }
1026 *tmp++ = language;
1027repeat:
1028 sp++;
1029 }
1030}

1032static int lookup_string(
1033 struct usb_gadget_strings **sp,
1034 void *buf,
1035 u16 language,
1036 int id
1037)
1038{
1039 struct usb_gadget_strings *s;
1040 int value;
1041
1042 while (*sp) {
1043 s = *sp++;

1045 continue;

1053static int get_string(struct usb_composite_dev *cdev,
1054 void *buf, u16 language, int id)
1055{
1056 struct usb_composite_driver *composite = cdev->driver;
1057 struct usb_gadget_string_container *uc;
1058 struct usb_configuration *c;
1059 struct usb_function *f;
1060 int len;
1061
1062 /* Yes, not only is USB's i18n support probably more than most
1063 * folk will ever care about ... also, it's all supported here.
1064 * (Except for UTF8 support for Unicode's "Astral Planes".)
1065 */
1066
1067 /* 0 == report all available language codes */
1068 if (id == 0) {
1069 struct usb_string_descriptor *s = buf;
1070 struct usb_gadget_strings **sp;
1071
1072 memset(s, 0, 256);
1073 s->bDescriptorType = USB_DT_STRING;
1074
1075 sp = composite->strings;
1076 if (sp)
1077 collect_langs(sp, s->wData);
1078
1079 list_for_each_entry(c, &cdev->configs, list) {
1080 sp = c->strings;
1081 if (sp)
1082 collect_langs(sp, s->wData);
1083
1084 list_for_each_entry(f, &c->functions, list) {
1085 sp = f->strings;
1086 if (sp)
1087 collect_langs(sp, s->wData);
1088 }
1089 }
1090 list_for_each_entry(uc, &cdev->gstrings, list) {
1091 struct usb_gadget_strings **sp;
1092
1093 sp = get_containers_gs(uc);
1094 collect_langs(sp, s->wData);
1095 }
1096
1097 for (len = 0; len <= 126 && s->wData[len]; len++)
1098 continue;
1099 if (!len)
1100 return -EINVAL;
1101
1102 s->bLength = 2 * (len + 1);
1103 return s->bLength;
1104 }
1105
1106 if (cdev->use_os_string && language == 0 && id == OS_STRING_IDX) {
1107 struct usb_os_string *b = buf;
1108 b->bLength = sizeof(*b);
1109 b->bDescriptorType = USB_DT_STRING;
1110 compiletime_assert(
1111 sizeof(b->qwSignature) == sizeof(cdev->qw_sign),
1112 "qwSignature size must be equal to qw_sign");
1113 memcpy(&b->qwSignature, cdev->qw_sign, sizeof(b->qwSignature));
1114 b->bMS_VendorCode = cdev->b_vendor_code;
1115 b->bPad = 0;
1116 return sizeof(*b);
1117 }
1118
1119 list_for_each_entry(uc, &cdev->gstrings, list) {
1120 struct usb_gadget_strings **sp;
1121
1122 sp = get_containers_gs(uc);
1123 len = lookup_string(sp, buf, language, id);
1124 if (len > 0)
1125 return len;
1126 }
1127
1128 /* String IDs are device-scoped, so we look up each string
1129 * table we're told about. These lookups are infrequent;
1130 * simpler-is-better here.
1131 */
1132 if (composite->strings) {
1133 len = lookup_string(composite->strings, buf, language, id);
1134 if (len > 0)
1135 return len;
1136 }
1137 list_for_each_entry(c, &cdev->configs, list) {
1138 if (c->strings) {
1139 len = lookup_string(c->strings, buf, language, id);
1140 if (len > 0)
1141 return len;
1142 }
1143 list_for_each_entry(f, &c->functions, list) {
1144 if (!f->strings)
1145 continue;
1146 len = lookup_string(f->strings, buf, language, id);
1147 if (len > 0)
1148 return len;
1149 }
1150 }
1151 return -EINVAL;
1152}

1154/**
1155 * usb_string_id() - allocate an unused string ID
1156 * @cdev: the device whose string descriptor IDs are being allocated
1157 * Context: single threaded during gadget setup
1158 *
1159 * @usb_string_id() is called from bind() callbacks to allocate
1160 * string IDs. Drivers for functions, configurations, or gadgets will
1161 * then store that ID in the appropriate descriptors and string table.
1162 *
1163 * All string identifier should be allocated using this,
1164 * @usb_string_ids_tab() or @usb_string_ids_n() routine, to ensure
1165 * that for example different functions don't wrongly assign different
1166 * meanings to the same identifier.
1167 */
1168int usb_string_id(struct usb_composite_dev *cdev)
1169{
1170 if (cdev->next_string_id < 254) {
1171 /* string id 0 is reserved by USB spec for list of
1172 * supported languages */
1173 /* 255 reserved as well? -- mina86 */
1174 cdev->next_string_id++;
1175 return cdev->next_string_id;
1176 }
1177 return -ENODEV;
1178}

1181/**
1182 * usb_string_ids() - allocate unused string IDs in batch
1183 * @cdev: the device whose string descriptor IDs are being allocated
1184 * @str: an array of usb_string objects to assign numbers to
1185 * Context: single threaded during gadget setup
1186 *
1187 * @usb_string_ids() is called from bind() callbacks to allocate
1188 * string IDs. Drivers for functions, configurations, or gadgets will
1189 * then copy IDs from the string table to the appropriate descriptors
1190 * and string table for other languages.
1191 *
1192 * All string identifier should be allocated using this,
1193 * @usb_string_id() or @usb_string_ids_n() routine, to ensure that for
1194 * example different functions don't wrongly assign different meanings
1195 * to the same identifier.
1196 */
1197int usb_string_ids_tab(struct usb_composite_dev *cdev, struct usb_string *str)
1198{
1199 int next = cdev->next_string_id;
1200
1201 for (; str->s; ++str) {
1202 if (unlikely(next >= 254))
1203 return -ENODEV;
1204 str->id = ++next;
1205 }
1206
1207 cdev->next_string_id = next;
1208
1209 return 0;
1210}

1213static struct usb_gadget_string_container *copy_gadget_strings(
1214 struct usb_gadget_strings **sp, unsigned n_gstrings,
1215 unsigned n_strings)
1216{
1217 struct usb_gadget_string_container *uc;
1218 struct usb_gadget_strings **gs_array;
1219 struct usb_gadget_strings *gs;
1220 struct usb_string *s;
1221 unsigned mem;
1222 unsigned n_gs;
1223 unsigned n_s;
1224 void *stash;
1225
1226 mem = sizeof(*uc);
1227 mem += sizeof(void *) * (n_gstrings + 1);
1228 mem += sizeof(struct usb_gadget_strings) * n_gstrings;
1229 mem += sizeof(struct usb_string) * (n_strings + 1) * (n_gstrings);
1230 uc = kmalloc(mem, GFP_KERNEL);
1231 if (!uc)
1232 return ERR_PTR(-ENOMEM);
1233 gs_array = get_containers_gs(uc);
1234 stash = uc->stash;
1235 stash += sizeof(void *) * (n_gstrings + 1);
1236 for (n_gs = 0; n_gs < n_gstrings; n_gs++) {
1237 struct usb_string *org_s;
1238
1239 gs_array[n_gs] = stash;
1240 gs = gs_array[n_gs];
1241 stash += sizeof(struct usb_gadget_strings);
1242 gs->language = sp[n_gs]->language;
1243 gs->strings = stash;
1244 org_s = sp[n_gs]->strings;
1245
1246 for (n_s = 0; n_s < n_strings; n_s++) {
1247 s = stash;
1248 stash += sizeof(struct usb_string);
1249 if (org_s->s)
1250 s->s = org_s->s;
1251 else
1252 s->s = "";
1253 org_s++;
1254 }
1255 s = stash;
1256 s->s = NULL;
1257 stash += sizeof(struct usb_string);
1258
1259 }
1260 gs_array[n_gs] = NULL;
1261 return uc;
1262}

1264/**
1265 * usb_gstrings_attach() - attach gadget strings to a cdev and assign ids
1266 * @cdev: the device whose string descriptor IDs are being allocated
1267 * and attached.
1268 * @sp: an array of usb_gadget_strings to attach.
1269 * @n_strings: number of entries in each usb_strings array (sp[]->strings)
1270 *
1271 * This function will create a deep copy of usb_gadget_strings and usb_string
1272 * and attach it to the cdev. The actual string (usb_string.s) will not be
1273 * copied but only a referenced will be made. The struct usb_gadget_strings
1274 * array may contain multiple languages and should be NULL terminated.
1275 * The ->language pointer of each struct usb_gadget_strings has to contain the
1276 * same amount of entries.
1277 * For instance: sp[0] is en-US, sp[1] is es-ES. It is expected that the first
1278 * usb_string entry of es-ES contains the translation of the first usb_string
1279 * entry of en-US. Therefore both entries become the same id assign.
1280 */
1281struct usb_string *usb_gstrings_attach(struct usb_composite_dev *cdev,
1282 struct usb_gadget_strings **sp, unsigned n_strings)
1283{
1284 struct usb_gadget_string_container *uc;
1285 struct usb_gadget_strings **n_gs;
1286 unsigned n_gstrings = 0;
1287 unsigned i;
1288 int ret;
1289
1290 for (i = 0; sp[i]; i++)
1291 n_gstrings++;
1292
1293 if (!n_gstrings)
1294 return ERR_PTR(-EINVAL);
1295
1296 uc = copy_gadget_strings(sp, n_gstrings, n_strings);
1297 if (IS_ERR(uc))
1298 return ERR_CAST(uc);
1299
1300 n_gs = get_containers_gs(uc);
1301 ret = usb_string_ids_tab(cdev, n_gs[0]->strings);
1302 if (ret)
1303 goto err;
1304
1305 for (i = 1; i < n_gstrings; i++) {
1306 struct usb_string *m_s;
1307 struct usb_string *s;
1308 unsigned n;
1309
1310 m_s = n_gs[0]->strings;
1311 s = n_gs[i]->strings;
1312 for (n = 0; n < n_strings; n++) {
1313 s->id = m_s->id;
1314 s++;
1315 m_s++;
1316 }
1317 }
1318 list_add_tail(&uc->list, &cdev->gstrings);
1319 return n_gs[0]->strings;
1320err:
1321 kfree(uc);
1322 return ERR_PTR(ret);
1323}

1404static int count_ext_compat(struct usb_configuration *c)
1405{
1406 int i, res;
1407
1408 res = 0;
1409 for (i = 0; i < c->next_interface_id; ++i) {
1410 struct usb_function *f;
1411 int j;
1412
1413 f = c->interface[i];
1414 for (j = 0; j < f->os_desc_n; ++j) {

1416

1418 continue;

1421 ++res;
1422 }
1423 }
1424 BUG_ON(res > 255);

1428static int fill_ext_compat(struct usb_configuration *c, u8 *buf)
1429{
1430 int i, count;
1431
1432 count = 16;
1433 for (i = 0; i < c->next_interface_id; ++i) {
1434 struct usb_function *f;
1435 int j;
1436
1437 f = c->interface[i];
1438 for (j = 0; j < f->os_desc_n; ++j) {
1439 struct usb_os_desc *d;
1440

1442 continue;

1445 *buf++ = i;

1450 ++buf;
1451 *buf = 0x01;
1452 buf += 23;
1453 }
1454 count += 24;
1455 if (count + 24 >= USB_COMP_EP0_OS_DESC_BUFSIZ)
1456 return count;
1457 }
1458 }
1459
1460 return count;
1461}

1463static int count_ext_prop(struct usb_configuration *c, int interface)
1464{
1465 struct usb_function *f;
1466 int j;
1467
1468 f = c->interface[interface];
1469 for (j = 0; j < f->os_desc_n; ++j) {
1470 struct usb_os_desc *d;
1471
1472 if (interface != f->os_desc_table[j].if_id)
1473 continue;
1474 d = f->os_desc_table[j].os_desc;
1475 if (d && d->ext_compat_id)
1476 return d->ext_prop_count;
1477 }
1478 return 0;

1481static int len_ext_prop(struct usb_configuration *c, int interface)
1482{
1483 struct usb_function *f;
1484 struct usb_os_desc *d;
1485 int j, res;
1486
1487 res = 10; /* header length */
1488 f = c->interface[interface];
1489 for (j = 0; j < f->os_desc_n; ++j) {
1490 if (interface != f->os_desc_table[j].if_id)
1491 continue;
1492 d = f->os_desc_table[j].os_desc;
1493 if (d)
1494 return min(res + d->ext_prop_len, 4096);
1495 }
1496 return res;

1499static int fill_ext_prop(struct usb_configuration *c, int interface, u8 *buf)
1500{
1501 struct usb_function *f;
1502 struct usb_os_desc *d;
1503 struct usb_os_desc_ext_prop *ext_prop;
1504 int j, count, n, ret;
1505
1506 f = c->interface[interface];
1507 count = 10; /* header length */
1508 for (j = 0; j < f->os_desc_n; ++j) {
1509 if (interface != f->os_desc_table[j].if_id)
1510 continue;
1511 d = f->os_desc_table[j].os_desc;
1512 if (d)
1513 list_for_each_entry(ext_prop, &d->ext_prop, entry) {
1514 n = ext_prop->data_len +
1515 ext_prop->name_len + 14;
1516 if (count + n >= USB_COMP_EP0_OS_DESC_BUFSIZ)
1517 return count;
1518 usb_ext_prop_put_size(buf, n);
1519 usb_ext_prop_put_type(buf, ext_prop->type);
1520 ret = usb_ext_prop_put_name(buf, ext_prop->name,
1521 ext_prop->name_len);
1522 if (ret < 0)
1523 return ret;
1524 switch (ext_prop->type) {
1525 case USB_EXT_PROP_UNICODE:
1526 case USB_EXT_PROP_UNICODE_ENV:
1527 case USB_EXT_PROP_UNICODE_LINK:
1528 usb_ext_prop_put_unicode(buf, ret,
1529 ext_prop->data,
1530 ext_prop->data_len);
1531 break;
1532 case USB_EXT_PROP_BINARY:
1533 usb_ext_prop_put_binary(buf, ret,
1534 ext_prop->data,
1535 ext_prop->data_len);
1536 break;
1537 case USB_EXT_PROP_LE32:
1538 /* not implemented */
1539 case USB_EXT_PROP_BE32:
1540 /* not implemented */
1541 default:
1542 return -EINVAL;
1543 }
1544 buf += n;
1545 count += n;
1546 }
1547 }
1548
1549 return count;
1550}

1552/*
1553 * The setup() callback implements all the ep0 functionality that's
1554 * not handled lower down, in hardware or the hardware driver(like
1555 * device and endpoint feature flags, and their status). It's all
1556 * housekeeping for the gadget function we're implementing. Most of
1557 * the work is in config and function specific setup.
1558 */
1559int
1560composite_setup(struct usb_gadget *gadget, const struct usb_ctrlrequest *ctrl)
1561{
1562 struct usb_composite_dev *cdev = get_gadget_data(gadget);
1563 struct usb_request *req = cdev->req;
1564 int value = -EOPNOTSUPP;
1565 int status = 0;
1566 u16 w_index = le16_to_cpu(ctrl->wIndex);
1567 u8 intf = w_index & 0xFF;
1568 u16 w_value = le16_to_cpu(ctrl->wValue);
1569 u16 w_length = le16_to_cpu(ctrl->wLength);
1570 struct usb_function *f = NULL;
1571 u8 endp;
1572
1573 /* partial re-init of the response message; the function or the
1574 * gadget might need to intercept e.g. a control-OUT completion
1575 * when we delegate to it.
1576 */
1577 req->zero = 0;
1578 req->context = cdev;
1579 req->complete = composite_setup_complete;
1580 req->length = 0;
1581 gadget->ep0->driver_data = cdev;
1582
1583 /*
1584 * Don't let non-standard requests match any of the cases below
1585 * by accident.
1586 */
1587 if ((ctrl->bRequestType & USB_TYPE_MASK) != USB_TYPE_STANDARD)
1588 goto unknown;
1589
1590 switch (ctrl->bRequest) {
1591
1592 /* we handle all standard USB descriptors */
1593 case USB_REQ_GET_DESCRIPTOR:
1594 if (ctrl->bRequestType != USB_DIR_IN)
1595 goto unknown;
1596 switch (w_value >> 8) {
1597
1598 case USB_DT_DEVICE:
1599 cdev->desc.bNumConfigurations =
1600 count_configs(cdev, USB_DT_DEVICE);
1601 cdev->desc.bMaxPacketSize0 =
1602 cdev->gadget->ep0->maxpacket;
1603 if (gadget_is_superspeed(gadget)) {
1604 if (gadget->speed >= USB_SPEED_SUPER) {
1605 cdev->desc.bcdUSB = cpu_to_le16(0x0310);
1606 cdev->desc.bMaxPacketSize0 = 9;
1607 } else {
1608 cdev->desc.bcdUSB = cpu_to_le16(0x0210);
1609 }
1610 } else {
1611 if (gadget->lpm_capable)
1612 cdev->desc.bcdUSB = cpu_to_le16(0x0201);
1613 else
1614 cdev->desc.bcdUSB = cpu_to_le16(0x0200);
1615 }
1616
1617 value = min(w_length, (u16) sizeof cdev->desc);
1618 memcpy(req->buf, &cdev->desc, value);
1619 break;
1620 case USB_DT_DEVICE_QUALIFIER:
1621 if (!gadget_is_dualspeed(gadget) ||
1622 gadget->speed >= USB_SPEED_SUPER)
1623 break;
1624 device_qual(cdev);
1625 value = min_t(int, w_length,
1626 sizeof(struct usb_qualifier_descriptor));
1627 break;
1628 case USB_DT_OTHER_SPEED_CONFIG:
1629 if (!gadget_is_dualspeed(gadget) ||
1630 gadget->speed >= USB_SPEED_SUPER)
1631 break;
1632 /* FALLTHROUGH */
1633 case USB_DT_CONFIG:
1634 value = config_desc(cdev, w_value);
1635 if (value >= 0)
1636 value = min(w_length, (u16) value);
1637 break;
1638 case USB_DT_STRING:
1639 value = get_string(cdev, req->buf,
1640 w_index, w_value & 0xff);
1641 if (value >= 0)
1642 value = min(w_length, (u16) value);
1643 break;
1644 case USB_DT_BOS:
1645 if (gadget_is_superspeed(gadget) ||
1646 gadget->lpm_capable) {
1647 value = bos_desc(cdev);
1648 value = min(w_length, (u16) value);
1649 }
1650 break;
1651 case USB_DT_OTG:
1652 if (gadget_is_otg(gadget)) {
1653 struct usb_configuration *config;
1654 int otg_desc_len = 0;
1655
1656 if (cdev->config)
1657 config = cdev->config;
1658 else
1659 config = list_first_entry(
1660 &cdev->configs,
1661 struct usb_configuration, list);
1662 if (!config)
1663 goto done;
1664
1665 if (gadget->otg_caps &&
1666 (gadget->otg_caps->otg_rev >= 0x0200))
1667 otg_desc_len += sizeof(
1668 struct usb_otg20_descriptor);
1669 else
1670 otg_desc_len += sizeof(
1671 struct usb_otg_descriptor);
1672
1673 value = min_t(int, w_length, otg_desc_len);
1674 memcpy(req->buf, config->descriptors[0], value);
1675 }
1676 break;
1677 }
1678 break;
1679
1680 /* any number of configs can work */
1681 case USB_REQ_SET_CONFIGURATION:
1682 if (ctrl->bRequestType != 0)
1683 goto unknown;
1684 if (gadget_is_otg(gadget)) {
1685 if (gadget->a_hnp_support)
1686 DBG(cdev, "HNP available\n");
1687 else if (gadget->a_alt_hnp_support)
1688 DBG(cdev, "HNP on another port\n");
1689 else
1690 VDBG(cdev, "HNP inactive\n");
1691 }
1692 spin_lock(&cdev->lock);
1693 value = set_config(cdev, ctrl, w_value);
1694 spin_unlock(&cdev->lock);
1695 break;
1696 case USB_REQ_GET_CONFIGURATION:
1697 if (ctrl->bRequestType != USB_DIR_IN)
1698 goto unknown;
1699 if (cdev->config)
1700 *(u8 *)req->buf = cdev->config->bConfigurationValue;
1701 else
1702 *(u8 *)req->buf = 0;
1703 value = min(w_length, (u16) 1);
1704 break;
1705
1706 /* function drivers must handle get/set altsetting */
1707 case USB_REQ_SET_INTERFACE:
1708 if (ctrl->bRequestType != USB_RECIP_INTERFACE)
1709 goto unknown;
1710 if (!cdev->config || intf >= MAX_CONFIG_INTERFACES)
1711 break;
1712 f = cdev->config->interface[intf];
1713 if (!f)
1714 break;
1715
1716 /*
1717 * If there's no get_alt() method, we know only altsetting zero
1718 * works. There is no need to check if set_alt() is not NULL
1719 * as we check this in usb_add_function().
1720 */
1721 if (w_value && !f->get_alt)
1722 break;
1723 value = f->set_alt(f, w_index, w_value);
1724 if (value == USB_GADGET_DELAYED_STATUS) {
1725 DBG(cdev,
1726 "%s: interface %d (%s) requested delayed status\n",
1727 __func__, intf, f->name);
1728 cdev->delayed_status++;
1729 DBG(cdev, "delayed_status count %d\n",
1730 cdev->delayed_status);
1731 }
1732 break;
1733 case USB_REQ_GET_INTERFACE:
1734 if (ctrl->bRequestType != (USB_DIR_IN|USB_RECIP_INTERFACE))
1735 goto unknown;
1736 if (!cdev->config || intf >= MAX_CONFIG_INTERFACES)
1737 break;
1738 f = cdev->config->interface[intf];
1739 if (!f)
1740 break;
1741 /* lots of interfaces only need altsetting zero... */
1742 value = f->get_alt ? f->get_alt(f, w_index) : 0;
1743 if (value < 0)
1744 break;
1745 *((u8 *)req->buf) = value;
1746 value = min(w_length, (u16) 1);
1747 break;
1748 case USB_REQ_GET_STATUS:
1749 if (gadget_is_otg(gadget) && gadget->hnp_polling_support &&
1750 (w_index == OTG_STS_SELECTOR)) {
1751 if (ctrl->bRequestType != (USB_DIR_IN |
1752 USB_RECIP_DEVICE))
1753 goto unknown;
1754 *((u8 *)req->buf) = gadget->host_request_flag;
1755 value = 1;
1756 break;
1757 }
1758
1759 /*
1760 * USB 3.0 additions:
1761 * Function driver should handle get_status request. If such cb
1762 * wasn't supplied we respond with default value = 0
1763 * Note: function driver should supply such cb only for the
1764 * first interface of the function
1765 */
1766 if (!gadget_is_superspeed(gadget))
1767 goto unknown;
1768 if (ctrl->bRequestType != (USB_DIR_IN | USB_RECIP_INTERFACE))
1769 goto unknown;
1770 value = 2; /* This is the length of the get_status reply */
1771 put_unaligned_le16(0, req->buf);
1772 if (!cdev->config || intf >= MAX_CONFIG_INTERFACES)
1773 break;
1774 f = cdev->config->interface[intf];
1775 if (!f)
1776 break;
1777 status = f->get_status ? f->get_status(f) : 0;
1778 if (status < 0)
1779 break;
1780 put_unaligned_le16(status & 0x0000ffff, req->buf);
1781 break;
1782 /*
1783 * Function drivers should handle SetFeature/ClearFeature
1784 * (FUNCTION_SUSPEND) request. function_suspend cb should be supplied
1785 * only for the first interface of the function
1786 */
1787 case USB_REQ_CLEAR_FEATURE:
1788 case USB_REQ_SET_FEATURE:
1789 if (!gadget_is_superspeed(gadget))
1790 goto unknown;
1791 if (ctrl->bRequestType != (USB_DIR_OUT | USB_RECIP_INTERFACE))
1792 goto unknown;
1793 switch (w_value) {
1794 case USB_INTRF_FUNC_SUSPEND:
1795 if (!cdev->config || intf >= MAX_CONFIG_INTERFACES)
1796 break;
1797 f = cdev->config->interface[intf];
1798 if (!f)
1799 break;
1800 value = 0;
1801 if (f->func_suspend)
1802 value = f->func_suspend(f, w_index >> 8);
1803 if (value < 0) {
1804 ERROR(cdev,
1805 "func_suspend() returned error %d\n",
1806 value);
1807 value = 0;
1808 }
1809 break;
1810 }
1811 break;
1812 default:
1813unknown:
1814 /*
1815 * OS descriptors handling
1816 */
1817 if (cdev->use_os_string && cdev->os_desc_config &&
1818 (ctrl->bRequestType & USB_TYPE_VENDOR) &&
1819 ctrl->bRequest == cdev->b_vendor_code) {
1820 struct usb_request *req;
1821 struct usb_configuration *os_desc_cfg;
1822 u8 *buf;
1823 int interface;
1824 int count = 0;
1825
1826 req = cdev->os_desc_req;
1827 req->context = cdev;
1828 req->complete = composite_setup_complete;
1829 buf = req->buf;
1830 os_desc_cfg = cdev->os_desc_config;
1831 w_length = min_t(u16, w_length, USB_COMP_EP0_OS_DESC_BUFSIZ);
1832 memset(buf, 0, w_length);
1833 buf[5] = 0x01;
1834 switch (ctrl->bRequestType & USB_RECIP_MASK) {
1835 case USB_RECIP_DEVICE:
1836 if (w_index != 0x4 || (w_value >> 8))
1837 break;
1838 buf[6] = w_index;
1839 if (w_length == 0x10) {
1840 /* Number of ext compat interfaces */
1841 count = count_ext_compat(os_desc_cfg);
1842 buf[8] = count;
1843 count *= 24; /* 24 B/ext compat desc */
1844 count += 16; /* header */
1845 put_unaligned_le32(count, buf);
1846 value = w_length;
1847 } else {
1848 /* "extended compatibility ID"s */
1849 count = count_ext_compat(os_desc_cfg);
1850 buf[8] = count;
1851 count *= 24; /* 24 B/ext compat desc */
1852 count += 16; /* header */
1853 put_unaligned_le32(count, buf);
1854 buf += 16;
1855 value = fill_ext_compat(os_desc_cfg, buf);
1856 value = min_t(u16, w_length, value);
1857 }
1858 break;
1859 case USB_RECIP_INTERFACE:
1860 if (w_index != 0x5 || (w_value >> 8))
1861 break;
1862 interface = w_value & 0xFF;
1863 buf[6] = w_index;
1864 if (w_length == 0x0A) {
1865 count = count_ext_prop(os_desc_cfg,
1866 interface);
1867 put_unaligned_le16(count, buf + 8);
1868 count = len_ext_prop(os_desc_cfg,
1869 interface);
1870 put_unaligned_le32(count, buf);
1871
1872 value = w_length;
1873 } else {
1874 count = count_ext_prop(os_desc_cfg,
1875 interface);
1876 put_unaligned_le16(count, buf + 8);
1877 count = len_ext_prop(os_desc_cfg,
1878 interface);
1879 put_unaligned_le32(count, buf);
1880 buf += 10;
1881 value = fill_ext_prop(os_desc_cfg,
1882 interface, buf);
1883 if (value < 0)
1884 return value;
1885 value = min_t(u16, w_length, value);
1886 }
1887 break;
1888 }
1889
1890 if (value >= 0) {
1891 req->length = value;
1892 req->context = cdev;
1893 req->zero = value < w_length;
1894 value = composite_ep0_queue(cdev, req,
1895 GFP_ATOMIC);
1896 if (value < 0) {
1897 DBG(cdev, "ep_queue --> %d\n", value);
1898 req->status = 0;
1899 composite_setup_complete(gadget->ep0,
1900 req);
1901 }
1902 }
1903 return value;
1904 }
1905
1906 VDBG(cdev,
1907 "non-core control req%02x.%02x v%04x i%04x l%d\n",
1908 ctrl->bRequestType, ctrl->bRequest,
1909 w_value, w_index, w_length);
1910
1911 /* functions always handle their interfaces and endpoints...
1912 * punt other recipients (other, WUSB, ...) to the current
1913 * configuration code.
1914 */
1915 if (cdev->config) {
1916 list_for_each_entry(f, &cdev->config->functions, list)
1917 if (f->req_match &&
1918 f->req_match(f, ctrl, false))
1919 goto try_fun_setup;
1920 } else {
1921 struct usb_configuration *c;
1922 list_for_each_entry(c, &cdev->configs, list)
1923 list_for_each_entry(f, &c->functions, list)
1924 if (f->req_match &&
1925 f->req_match(f, ctrl, true))
1926 goto try_fun_setup;
1927 }
1928 f = NULL;
1929
1930 switch (ctrl->bRequestType & USB_RECIP_MASK) {
1931 case USB_RECIP_INTERFACE:
1932 if (!cdev->config || intf >= MAX_CONFIG_INTERFACES)
1933 break;
1934 f = cdev->config->interface[intf];
1935 break;
1936
1937 case USB_RECIP_ENDPOINT:
1938 if (!cdev->config)
1939 break;
1940 endp = ((w_index & 0x80) >> 3) | (w_index & 0x0f);
1941 list_for_each_entry(f, &cdev->config->functions, list) {
1942 if (test_bit(endp, f->endpoints))
1943 break;
1944 }
1945 if (&f->list == &cdev->config->functions)
1946 f = NULL;
1947 break;
1948 }
1949try_fun_setup:
1950 if (f && f->setup)
1951 value = f->setup(f, ctrl);
1952 else {
1953 struct usb_configuration *c;
1954
1955 c = cdev->config;
1956 if (!c)
1957 goto done;
1958
1959 /* try current config's setup */
1960 if (c->setup) {
1961 value = c->setup(c, ctrl);
1962 goto done;
1963 }
1964
1965 /* try the only function in the current config */
1966 if (!list_is_singular(&c->functions))
1967 goto done;
1968 f = list_first_entry(&c->functions, struct usb_function,
1969 list);
1970 if (f->setup)
1971 value = f->setup(f, ctrl);
1972 }
1973
1974 goto done;
1975 }
1976
1977 /* respond with data transfer before status phase? */
1978 if (value >= 0 && value != USB_GADGET_DELAYED_STATUS) {
1979 req->length = value;
1980 req->context = cdev;
1981 req->zero = value < w_length;
1982 value = composite_ep0_queue(cdev, req, GFP_ATOMIC);
1983 if (value < 0) {
1984 DBG(cdev, "ep_queue --> %d\n", value);
1985 req->status = 0;
1986 composite_setup_complete(gadget->ep0, req);
1987 }
1988 } else if (value == USB_GADGET_DELAYED_STATUS && w_length != 0) {
1989 WARN(cdev,
1990 "%s: Delayed status not supported for w_length != 0",
1991 __func__);
1992 }
1993
1994done:
1995 /* device either stalls (value < 0) or reports success */
1996 return value;
1997}

2102int composite_dev_prepare(struct usb_composite_driver *composite,
2103 struct usb_composite_dev *cdev)
2104{
2105 struct usb_gadget *gadget = cdev->gadget;
2106 int ret = -ENOMEM;
2107
2108 /* preallocate control response and buffer */
2109 cdev->req = usb_ep_alloc_request(gadget->ep0, GFP_KERNEL);
2110 if (!cdev->req)
2111 return -ENOMEM;
2112
2113 cdev->req->buf = kmalloc(USB_COMP_EP0_BUFSIZ, GFP_KERNEL);
2114 if (!cdev->req->buf)
2115 goto fail;
2116
2117 ret = device_create_file(&gadget->dev, &dev_attr_suspended);
2118 if (ret)
2119 goto fail_dev;
2120
2121 cdev->req->complete = composite_setup_complete;
2122 cdev->req->context = cdev;
2123 gadget->ep0->driver_data = cdev;
2124
2125 cdev->driver = composite;
2126
