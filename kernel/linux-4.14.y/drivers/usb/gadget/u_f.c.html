17struct usb_request *alloc_ep_req(struct usb_ep *ep, size_t len)
18{
19 struct usb_request *req;
20
21 req = usb_ep_alloc_request(ep, GFP_ATOMIC);
22 if (req) {
23 req->length = usb_endpoint_dir_out(ep->desc) ?
24 usb_ep_align(ep, len) : len;
25 req->buf = kmalloc(req->length, GFP_ATOMIC);
26 if (!req->buf) {
27 usb_ep_free_request(ep, req);
28 req = NULL;
29 }
30 }
31 return req;
32}
33EXPORT_SYMBOL_GPL(alloc_ep_req);
