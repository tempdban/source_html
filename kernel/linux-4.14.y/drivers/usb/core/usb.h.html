49static inline unsigned usb_get_max_power(struct usb_device *udev,
50 struct usb_host_config *c)
51{
52 /* SuperSpeed power is in 8 mA units; others are in 2 mA units */
53 unsigned mul = (udev->speed >= USB_SPEED_SUPER ? 8 : 2);
54
55 return c->desc.bMaxPower * mul;
56}
