292static bool usb_match_any_interface(struct usb_device *udev,
293 const struct usb_device_id *id)
294{
295 unsigned int i;
296
297 for (i = 0; i < udev->descriptor.bNumConfigurations; ++i) {
298 struct usb_host_config *cfg = &udev->config[i];
299 unsigned int j;
300
301 for (j = 0; j < cfg->desc.bNumInterfaces; ++j) {
302 struct usb_interface_cache *cache;
303 struct usb_host_interface *intf;
304
305 cache = cfg->intf_cache[j];
306 if (cache->num_altsetting == 0)
307 continue;
308
309 intf = &cache->altsetting[0];
310 if (usb_match_one_id_intf(udev, intf, id))
311 return true;
312 }
313 }
314
315 return false;
316}

318static int usb_amd_resume_quirk(struct usb_device *udev)
319{
320 struct usb_hcd *hcd;
321
322 hcd = bus_to_hcd(udev->bus);
323 /* The device should be attached directly to root hub */
324 if (udev->level == 1 && hcd->amd_resume_bug == 1)
325 return 1;
326
327 return 0;
328}

330static u32 __usb_detect_quirks(struct usb_device *udev,
331 const struct usb_device_id *id)
332{
333 u32 quirks = 0;
334
335 for (; id->match_flags; id++) {
336 if (!usb_match_device(udev, id))
337 continue;
338
339 if ((id->match_flags & USB_DEVICE_ID_MATCH_INT_INFO) &&
340 !usb_match_any_interface(udev, id))
341 continue;
342
343 quirks |= (u32)(id->driver_info);
344 }
345
346 return quirks;
347}

349/*
350 * Detect any quirks the device has, and do any housekeeping for it if needed.
351 */
352void usb_detect_quirks(struct usb_device *udev)
353{
354 udev->quirks = __usb_detect_quirks(udev, usb_quirk_list);
355
356 /*
357 * Pixart-based mice would trigger remote wakeup issue on AMD
358 * Yangtze chipset, so set them as RESET_RESUME flag.
359 */
360 if (usb_amd_resume_quirk(udev))
361 udev->quirks |= __usb_detect_quirks(udev,
362 usb_amd_resume_quirk_list);
363
364 if (udev->quirks)
365 dev_dbg(&udev->dev, "USB quirks for this device: %x\n",
366 udev->quirks);
367
368#ifdef CONFIG_USB_DEFAULT_PERSIST
369 if (!(udev->quirks & USB_QUIRK_RESET))
370 udev->persist_enabled = 1;
371#else
372 /* Hubs are automatically enabled for USB-PERSIST */
373 if (udev->descriptor.bDeviceClass == USB_CLASS_HUB)
374 udev->persist_enabled = 1;
375#endif /* CONFIG_USB_DEFAULT_PERSIST */
376}
377
378void usb_detect_interface_quirks(struct usb_device *udev)
379{
380 u32 quirks;
381
382 quirks = __usb_detect_quirks(udev, usb_interface_quirk_list);
383 if (quirks == 0)
384 return;
385
386 dev_dbg(&udev->dev, "USB interface quirks for this device: %x\n",
387 quirks);
388 udev->quirks |= quirks;
389}
