30/**
31 * notify_user_space - Notifies user space about thermal events
32 * @tz - thermal_zone_device
33 * @trip - trip point index
34 *
35 * This function notifies the user space through UEvents.
36 */
37static int notify_user_space(struct thermal_zone_device *tz, int trip)
38{
39 char *thermal_prop[5];
40 int i;
41
42 mutex_lock(&tz->lock);
43 thermal_prop[0] = kasprintf(GFP_KERNEL, "NAME=%s", tz->type);
44 thermal_prop[1] = kasprintf(GFP_KERNEL, "TEMP=%d", tz->temperature);
45 thermal_prop[2] = kasprintf(GFP_KERNEL, "TRIP=%d", trip);
46 thermal_prop[3] = kasprintf(GFP_KERNEL, "EVENT=%d", tz->notify_event);
47 thermal_prop[4] = NULL;
48 kobject_uevent_env(&tz->device.kobj, KOBJ_CHANGE, thermal_prop);
49 for (i = 0; i < 4; ++i)
50 kfree(thermal_prop[i]);
51 mutex_unlock(&tz->lock);
52 return 0;
53}
54
55static struct thermal_governor thermal_gov_user_space = {
56 .name = "user_space",
57 .throttle = notify_user_space,
58};
59
60int thermal_gov_user_space_register(void)
61{
62 return thermal_register_governor(&thermal_gov_user_space);
63}
64
65void thermal_gov_user_space_unregister(void)
66{
67 thermal_unregister_governor(&thermal_gov_user_space);
68}
