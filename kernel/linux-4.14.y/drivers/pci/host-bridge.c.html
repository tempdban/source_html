11static struct pci_bus *find_pci_root_bus(struct pci_bus *bus)
12{
13 while (bus->parent)
14 bus = bus->parent;
15
16 return bus;
17}
18
19struct pci_host_bridge *pci_find_host_bridge(struct pci_bus *bus)
20{
21 struct pci_bus *root_bus = find_pci_root_bus(bus);
22
23 return to_pci_host_bridge(root_bus->bridge);
24}
25
26struct device *pci_get_host_bridge_device(struct pci_dev *dev)
27{
28 struct pci_bus *root_bus = find_pci_root_bus(dev->bus);
29 struct device *bridge = root_bus->bridge;
30
31 kobject_get(&bridge->kobj);
32 return bridge;
33}
34
35void pci_put_host_bridge_device(struct device *dev)
36{
37 kobject_put(&dev->kobj);
38}

49void pcibios_resource_to_bus(struct pci_bus *bus, struct pci_bus_region *region,
50 struct resource *res)
51{
52 struct pci_host_bridge *bridge = pci_find_host_bridge(bus);
53 struct resource_entry *window;
54 resource_size_t offset = 0;
55
56 resource_list_for_each_entry(window, &bridge->windows) {
57 if (resource_contains(window->res, res)) {
58 offset = window->offset;
59 break;
60 }
61 }
62
63 region->start = res->start - offset;
64 region->end = res->end - offset;
65}
66EXPORT_SYMBOL(pcibios_resource_to_bus);
67
68static bool region_contains(struct pci_bus_region *region1,
69 struct pci_bus_region *region2)
70{
71 return region1->start <= region2->start && region1->end >= region2->end;
72}
73
74void pcibios_bus_to_resource(struct pci_bus *bus, struct resource *res,
75 struct pci_bus_region *region)
76{
77 struct pci_host_bridge *bridge = pci_find_host_bridge(bus);
78 struct resource_entry *window;
79 resource_size_t offset = 0;
80
81 resource_list_for_each_entry(window, &bridge->windows) {
82 struct pci_bus_region bus_region;
83
84 if (resource_type(res) != resource_type(window->res))
85 continue;
86
87 bus_region.start = window->res->start - window->offset;
88 bus_region.end = window->res->end - window->offset;
89
90 if (region_contains(&bus_region, region)) {
91 offset = window->offset;
92 break;
93 }
94 }
95
96 res->start = region->start + offset;
97 res->end = region->end + offset;
98}
99EXPORT_SYMBOL(pcibios_bus_to_resource);
