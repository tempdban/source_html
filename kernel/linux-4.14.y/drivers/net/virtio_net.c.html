225/*
226 * private is used to chain pages for big packets, put the whole
227 * most recent used list in the beginning for reuse
228 */
229static void give_pages(struct receive_queue *rq, struct page *page)
230{
231 struct page *end;
232
233 /* Find end of list, sew whole thing into vi->rq.pages. */
234 for (end = page; end->private; end = (struct page *)end->private);
235 end->private = (unsigned long)rq->pages;
236 rq->pages = page;
237}

239static struct page *get_a_page(struct receive_queue *rq, gfp_t gfp_mask)
240{
241 struct page *p = rq->pages;
242
243 if (p) {
244 rq->pages = (struct page *)p->private;
245 /* clear private here, it is used to chain pages */
246 p->private = 0;
247 } else
248 p = alloc_page(gfp_mask);
249 return p;
250}

253 struct virtqueue *vq)

261static void virtqueue_napi_complete(struct napi_struct *napi,
262 struct virtqueue *vq, int processed)
263{
264 int opaque;
265
266 opaque = virtqueue_enable_cb_prepare(vq);
267 if (napi_complete_done(napi, processed)) {
268 if (unlikely(virtqueue_poll(vq, opaque)))
269 virtqueue_napi_schedule(napi, vq);
270 } else {
271 virtqueue_disable_cb(vq);
272 }
273}

275static void skb_xmit_done(struct virtqueue *vq)
276{
277 struct virtnet_info *vi = vq->vdev->priv;
278 struct napi_struct *napi = &vi->sq[vq2txq(vq)].napi;
279
280 /* Suppress further interrupts. */
281 virtqueue_disable_cb(vq);
282
283 if (napi->weight)
284 virtqueue_napi_schedule(napi, vq);
285 else
286 /* We were probably waiting for more output buffers. */
287 netif_wake_subqueue(vi->dev, vq2txq(vq));
288}

308static struct sk_buff *page_to_skb(struct virtnet_info *vi,
309 struct receive_queue *rq,
310 struct page *page, unsigned int offset,
311 unsigned int len, unsigned int truesize)
312{
313 struct sk_buff *skb;
314 struct virtio_net_hdr_mrg_rxbuf *hdr;
315 unsigned int copy, hdr_len, hdr_padded_len;
316 char *p;
317
318 p = page_address(page) + offset;
319
320 /* copy small packet so we can reuse these pages for small data */
321 skb = napi_alloc_skb(&rq->napi, GOOD_COPY_LEN);
322 if (unlikely(!skb))
323 return NULL;
324
325 hdr = skb_vnet_hdr(skb);
326
327 hdr_len = vi->hdr_len;
328 if (vi->mergeable_rx_bufs)
329 hdr_padded_len = sizeof(*hdr);
330 else
331 hdr_padded_len = sizeof(struct padded_vnet_hdr);
332
333 memcpy(hdr, p, hdr_len);
334
335 len -= hdr_len;
336 offset += hdr_padded_len;
337 p += hdr_padded_len;
338
339 copy = len;
340 if (copy > skb_tailroom(skb))
341 copy = skb_tailroom(skb);
342 skb_put_data(skb, p, copy);
343
344 len -= copy;
345 offset += copy;
346
347 if (vi->mergeable_rx_bufs) {
348 if (len)
349 skb_add_rx_frag(skb, 0, page, offset, len, truesize);
350 else
351 put_page(page);
352 return skb;
353 }
354
355 /*
356 * Verify that we can indeed put this data into a skb.
357 * This is here to handle cases when the device erroneously
358 * tries to receive more than is possible. This is usually
359 * the case of a broken device.
360 */
361 if (unlikely(len > MAX_SKB_FRAGS * PAGE_SIZE)) {
362 net_dbg_ratelimited("%s: too much data\n", skb->dev->name);
363 dev_kfree_skb(skb);
364 return NULL;
365 }
366 BUG_ON(offset >= PAGE_SIZE);
367 while (len) {
368 unsigned int frag_size = min((unsigned)PAGE_SIZE - offset, len);
369 skb_add_rx_frag(skb, skb_shinfo(skb)->nr_frags, page, offset,
370 frag_size, truesize);
371 len -= frag_size;
372 page = (struct page *)page->private;
373 offset = 0;
374 }
375
376 if (page)
377 give_pages(rq, page);
378
379 return skb;
380}

382static bool virtnet_xdp_xmit(struct virtnet_info *vi,
383 struct receive_queue *rq,
384 struct xdp_buff *xdp)
385{
386 struct virtio_net_hdr_mrg_rxbuf *hdr;
387 unsigned int len;
388 struct send_queue *sq;
389 unsigned int qp;
390 void *xdp_sent;
391 int err;
392
393 qp = vi->curr_queue_pairs - vi->xdp_queue_pairs + smp_processor_id();
394 sq = &vi->sq[qp];
395
396 /* Free up any pending old buffers before queueing new ones. */
397 while ((xdp_sent = virtqueue_get_buf(sq->vq, &len)) != NULL) {
398 struct page *sent_page = virt_to_head_page(xdp_sent);
399
400 put_page(sent_page);
401 }
402
403 xdp->data -= vi->hdr_len;
404 /* Zero header and leave csum up to XDP layers */
405 hdr = xdp->data;
406 memset(hdr, 0, vi->hdr_len);
407
408 sg_init_one(sq->sg, xdp->data, xdp->data_end - xdp->data);
409
410 err = virtqueue_add_outbuf(sq->vq, sq->sg, 1, xdp->data, GFP_ATOMIC);
411 if (unlikely(err)) {
412 struct page *page = virt_to_head_page(xdp->data);
413
414 put_page(page);
415 return false;
416 }
417
418 virtqueue_kick(sq->vq);
419 return true;
420}

427/* We copy the packet for XDP in the following cases:
428 *
429 * 1) Packet is scattered across multiple rx buffers.
430 * 2) Headroom space is insufficient.
431 *
432 * This is inefficient but it's a temporary condition that
433 * we hit right after XDP is enabled and until queue is refilled
434 * with large buffers with sufficient headroom - so it should affect
435 * at most queue size packets.
436 * Afterwards, the conditions to enable
437 * XDP should preclude the underlying device from sending packets
438 * across multiple buffers (num_buf > 1), and we make sure buffers
439 * have enough headroom.
440 */
441static struct page *xdp_linearize_page(struct receive_queue *rq,
442 u16 *num_buf,
443 struct page *p,
444 int offset,
445 int page_off,
446 unsigned int *len)
447{
448 struct page *page = alloc_page(GFP_ATOMIC);
449
450 if (!page)
451 return NULL;
452
453 memcpy(page_address(page) + page_off, page_address(p) + offset, *len);
454 page_off += *len;
455
456 while (--*num_buf) {
457 unsigned int buflen;
458 void *buf;
459 int off;
460
461 buf = virtqueue_get_buf(rq->vq, &buflen);
462 if (unlikely(!buf))
463 goto err_buf;
464
465 p = virt_to_head_page(buf);
466 off = buf - page_address(p);
467
468 /* guard against a misconfigured or uncooperative backend that
469 * is sending packet larger than the MTU.
470 */
471 if ((page_off + buflen) > PAGE_SIZE) {
472 put_page(p);
473 goto err_buf;
474 }
475
476 memcpy(page_address(page) + page_off,
477 page_address(p) + off, buflen);
478 page_off += buflen;
479 put_page(p);
480 }
481
482 /* Headroom does not contribute to packet length */
483 *len = page_off - VIRTIO_XDP_HEADROOM;
484 return page;
485err_buf:
486 __free_pages(page, 0);
487 return NULL;
488}

490static struct sk_buff *receive_small(struct net_device *dev,
491 struct virtnet_info *vi,
492 struct receive_queue *rq,
493 void *buf, void *ctx,
494 unsigned int len)
495{
496 struct sk_buff *skb;
497 struct bpf_prog *xdp_prog;
498 unsigned int xdp_headroom = (unsigned long)ctx;
499 unsigned int header_offset = VIRTNET_RX_PAD + xdp_headroom;
500 unsigned int headroom = vi->hdr_len + header_offset;
501 unsigned int buflen = SKB_DATA_ALIGN(GOOD_PACKET_LEN + headroom) +
502 SKB_DATA_ALIGN(sizeof(struct skb_shared_info));
503 struct page *page = virt_to_head_page(buf);
504 unsigned int delta = 0;
505 struct page *xdp_page;
506 len -= vi->hdr_len;
507
508 rcu_read_lock();
509 xdp_prog = rcu_dereference(rq->xdp_prog);
510 if (xdp_prog) {
511 struct virtio_net_hdr_mrg_rxbuf *hdr = buf + header_offset;
512 struct xdp_buff xdp;
513 void *orig_data;
514 u32 act;
515
516 if (unlikely(hdr->hdr.gso_type))
517 goto err_xdp;
518
519 if (unlikely(xdp_headroom < virtnet_get_headroom(vi))) {
520 int offset = buf - page_address(page) + header_offset;
521 unsigned int tlen = len + vi->hdr_len;
522 u16 num_buf = 1;
523
524 xdp_headroom = virtnet_get_headroom(vi);
525 header_offset = VIRTNET_RX_PAD + xdp_headroom;
526 headroom = vi->hdr_len + header_offset;
527 buflen = SKB_DATA_ALIGN(GOOD_PACKET_LEN + headroom) +
528 SKB_DATA_ALIGN(sizeof(struct skb_shared_info));
529 xdp_page = xdp_linearize_page(rq, &num_buf, page,
530 offset, header_offset,
531 &tlen);
532 if (!xdp_page)
533 goto err_xdp;
534
535 buf = page_address(xdp_page);
536 put_page(page);
537 page = xdp_page;
538 }
539
540 xdp.data_hard_start = buf + VIRTNET_RX_PAD + vi->hdr_len;
541 xdp.data = xdp.data_hard_start + xdp_headroom;
542 xdp.data_end = xdp.data + len;
543 orig_data = xdp.data;
544 act = bpf_prog_run_xdp(xdp_prog, &xdp);
545
546 switch (act) {
547 case XDP_PASS:
548 /* Recalculate length in case bpf program changed it */
549 delta = orig_data - xdp.data;
550 break;
551 case XDP_TX:
552 if (unlikely(!virtnet_xdp_xmit(vi, rq, &xdp)))
553 trace_xdp_exception(vi->dev, xdp_prog, act);
554 rcu_read_unlock();
555 goto xdp_xmit;
556 default:
557 bpf_warn_invalid_xdp_action(act);
558 case XDP_ABORTED:
559 trace_xdp_exception(vi->dev, xdp_prog, act);
560 case XDP_DROP:
561 goto err_xdp;
562 }
563 }
564 rcu_read_unlock();
565
566 skb = build_skb(buf, buflen);
567 if (!skb) {
568 put_page(page);
569 goto err;
570 }
571 skb_reserve(skb, headroom - delta);
572 skb_put(skb, len + delta);
573 if (!delta) {
574 buf += header_offset;
575 memcpy(skb_vnet_hdr(skb), buf, vi->hdr_len);
576 } /* keep zeroed vnet hdr since packet was changed by bpf */
577
578err:
579 return skb;
580
581err_xdp:
582 rcu_read_unlock();
583 dev->stats.rx_dropped++;
584 put_page(page);
585xdp_xmit:
586 return NULL;
587}

609static struct sk_buff *receive_mergeable(struct net_device *dev,
610 struct virtnet_info *vi,
611 struct receive_queue *rq,
612 void *buf,
613 void *ctx,
614 unsigned int len)
615{
616 struct virtio_net_hdr_mrg_rxbuf *hdr = buf;
617 u16 num_buf = virtio16_to_cpu(vi->vdev, hdr->num_buffers);
618 struct page *page = virt_to_head_page(buf);
619 int offset = buf - page_address(page);
620 struct sk_buff *head_skb, *curr_skb;
621 struct bpf_prog *xdp_prog;
622 unsigned int truesize;
623 unsigned int headroom = mergeable_ctx_to_headroom(ctx);
624
625 head_skb = NULL;
626
627 rcu_read_lock();
628 xdp_prog = rcu_dereference(rq->xdp_prog);
629 if (xdp_prog) {
630 struct page *xdp_page;
631 struct xdp_buff xdp;
632 void *data;
633 u32 act;
634
635 /* Transient failure which in theory could occur if
636 * in-flight packets from before XDP was enabled reach
637 * the receive path after XDP is loaded.
638 */
639 if (unlikely(hdr->hdr.gso_type))
640 goto err_xdp;
641
642 /* This happens when rx buffer size is underestimated */
643 if (unlikely(num_buf > 1 ||
644 headroom < virtnet_get_headroom(vi))) {
645 /* linearize data for XDP */
646 xdp_page = xdp_linearize_page(rq, &num_buf,
647 page, offset,
648 VIRTIO_XDP_HEADROOM,
649 &len);
650 if (!xdp_page)
651 goto err_xdp;
652 offset = VIRTIO_XDP_HEADROOM;
653 } else {
654 xdp_page = page;
655 }
656
657 /* Allow consuming headroom but reserve enough space to push
658 * the descriptor on if we get an XDP_TX return code.
659 */
660 data = page_address(xdp_page) + offset;
661 xdp.data_hard_start = data - VIRTIO_XDP_HEADROOM + vi->hdr_len;
662 xdp.data = data + vi->hdr_len;
663 xdp.data_end = xdp.data + (len - vi->hdr_len);
664 act = bpf_prog_run_xdp(xdp_prog, &xdp);
665
666 switch (act) {
667 case XDP_PASS:
668 /* recalculate offset to account for any header
669 * adjustments. Note other cases do not build an
670 * skb and avoid using offset
671 */
672 offset = xdp.data -
673 page_address(xdp_page) - vi->hdr_len;
674
675 /* We can only create skb based on xdp_page. */
676 if (unlikely(xdp_page != page)) {
677 rcu_read_unlock();
678 put_page(page);
679 head_skb = page_to_skb(vi, rq, xdp_page,
680 offset, len, PAGE_SIZE);
681 ewma_pkt_len_add(&rq->mrg_avg_pkt_len, len);
682 return head_skb;
683 }
684 break;
685 case XDP_TX:
686 if (unlikely(!virtnet_xdp_xmit(vi, rq, &xdp)))
687 trace_xdp_exception(vi->dev, xdp_prog, act);
688 ewma_pkt_len_add(&rq->mrg_avg_pkt_len, len);
689 if (unlikely(xdp_page != page))
690 put_page(page);
691 rcu_read_unlock();
692 goto xdp_xmit;
693 default:
694 bpf_warn_invalid_xdp_action(act);
695 case XDP_ABORTED:
696 trace_xdp_exception(vi->dev, xdp_prog, act);
697 case XDP_DROP:
698 if (unlikely(xdp_page != page))
699 __free_pages(xdp_page, 0);
700 ewma_pkt_len_add(&rq->mrg_avg_pkt_len, len);
701 goto err_xdp;
702 }
703 }
704 rcu_read_unlock();
705
706 truesize = mergeable_ctx_to_truesize(ctx);
707 if (unlikely(len > truesize)) {
708 pr_debug("%s: rx error: len %u exceeds truesize %lu\n",
709 dev->name, len, (unsigned long)ctx);
710 dev->stats.rx_length_errors++;
711 goto err_skb;
712 }
713
714 head_skb = page_to_skb(vi, rq, page, offset, len, truesize);
715 curr_skb = head_skb;
716
717 if (unlikely(!curr_skb))
718 goto err_skb;
719 while (--num_buf) {
720 int num_skb_frags;
721
722 buf = virtqueue_get_buf_ctx(rq->vq, &len, &ctx);
723 if (unlikely(!buf)) {
724 pr_debug("%s: rx error: %d buffers out of %d missing\n",
725 dev->name, num_buf,
726 virtio16_to_cpu(vi->vdev,
727 hdr->num_buffers));
728 dev->stats.rx_length_errors++;
729 goto err_buf;
730 }
731
732 page = virt_to_head_page(buf);
733
734 truesize = mergeable_ctx_to_truesize(ctx);
735 if (unlikely(len > truesize)) {
736 pr_debug("%s: rx error: len %u exceeds truesize %lu\n",
737 dev->name, len, (unsigned long)ctx);
738 dev->stats.rx_length_errors++;
739 goto err_skb;
740 }
741
742 num_skb_frags = skb_shinfo(curr_skb)->nr_frags;
743 if (unlikely(num_skb_frags == MAX_SKB_FRAGS)) {
744 struct sk_buff *nskb = alloc_skb(0, GFP_ATOMIC);
745
746 if (unlikely(!nskb))
747 goto err_skb;
748 if (curr_skb == head_skb)
749 skb_shinfo(curr_skb)->frag_list = nskb;
750 else
751 curr_skb->next = nskb;
752 curr_skb = nskb;
753 head_skb->truesize += nskb->truesize;
754 num_skb_frags = 0;
755 }
756 if (curr_skb != head_skb) {
757 head_skb->data_len += len;
758 head_skb->len += len;
759 head_skb->truesize += truesize;
760 }
761 offset = buf - page_address(page);
762 if (skb_can_coalesce(curr_skb, num_skb_frags, page, offset)) {
763 put_page(page);
764 skb_coalesce_rx_frag(curr_skb, num_skb_frags - 1,
765 len, truesize);
766 } else {
767 skb_add_rx_frag(curr_skb, num_skb_frags, page,
768 offset, len, truesize);
769 }
770 }
771
772 ewma_pkt_len_add(&rq->mrg_avg_pkt_len, head_skb->len);
773 return head_skb;
774
775err_xdp:
776 rcu_read_unlock();
777err_skb:
778 put_page(page);
779 while (num_buf-- > 1) {
780 buf = virtqueue_get_buf(rq->vq, &len);
781 if (unlikely(!buf)) {
782 pr_debug("%s: rx error: %d buffers missing\n",
783 dev->name, num_buf);
784 dev->stats.rx_length_errors++;
785 break;
786 }
787 page = virt_to_head_page(buf);
788 put_page(page);
789 }
790err_buf:
791 dev->stats.rx_dropped++;
792 dev_kfree_skb(head_skb);
793xdp_xmit:
794 return NULL;
795}

797static int receive_buf(struct virtnet_info *vi, struct receive_queue *rq,
798 void *buf, unsigned int len, void **ctx)
799{
800 struct net_device *dev = vi->dev;
801 struct sk_buff *skb;
802 struct virtio_net_hdr_mrg_rxbuf *hdr;
803 int ret;
804
805 if (unlikely(len < vi->hdr_len + ETH_HLEN)) {
806 pr_debug("%s: short packet %i\n", dev->name, len);
807 dev->stats.rx_length_errors++;
808 if (vi->mergeable_rx_bufs) {
809 put_page(virt_to_head_page(buf));
810 } else if (vi->big_packets) {
811 give_pages(rq, buf);
812 } else {
813 put_page(virt_to_head_page(buf));
814 }
815 return 0;
816 }
817
818 if (vi->mergeable_rx_bufs)
819 skb = receive_mergeable(dev, vi, rq, buf, ctx, len);
820 else if (vi->big_packets)
821 skb = receive_big(dev, vi, rq, buf, len);
822 else
823 skb = receive_small(dev, vi, rq, buf, ctx, len);
824
825 if (unlikely(!skb))
826 return 0;
827
828 hdr = skb_vnet_hdr(skb);
829
830 ret = skb->len;
831
832 if (hdr->hdr.flags & VIRTIO_NET_HDR_F_DATA_VALID)
833 skb->ip_summed = CHECKSUM_UNNECESSARY;
834
835 if (virtio_net_hdr_to_skb(skb, &hdr->hdr,
836 virtio_is_little_endian(vi->vdev))) {
837 net_warn_ratelimited("%s: bad gso: type: %u, size: %u\n",
838 dev->name, hdr->hdr.gso_type,
839 hdr->hdr.gso_size);
840 goto frame_err;
841 }
842
843 skb->protocol = eth_type_trans(skb, dev);
844 pr_debug("Receiving skb proto 0x%04x len %i type %i\n",
845 ntohs(skb->protocol), skb->len, skb->pkt_type);
846
847 napi_gro_receive(&rq->napi, skb);
848 return ret;
849
850frame_err:
851 dev->stats.rx_frame_errors++;
852 dev_kfree_skb(skb);
853 return 0;
854}

856/* Unlike mergeable buffers, all buffers are allocated to the
857 * same size, except for the headroom. For this reason we do
858 * not need to use mergeable_len_to_ctx here - it is enough
859 * to store the headroom as the context ignoring the truesize.
860 */
861static int add_recvbuf_small(struct virtnet_info *vi, struct receive_queue *rq,
862 gfp_t gfp)
863{
864 struct page_frag *alloc_frag = &rq->alloc_frag;
865 char *buf;
866 unsigned int xdp_headroom = virtnet_get_headroom(vi);
867 void *ctx = (void *)(unsigned long)xdp_headroom;
868 int len = vi->hdr_len + VIRTNET_RX_PAD + GOOD_PACKET_LEN + xdp_headroom;
869 int err;
870
871 len = SKB_DATA_ALIGN(len) +
872 SKB_DATA_ALIGN(sizeof(struct skb_shared_info));
873 if (unlikely(!skb_page_frag_refill(len, alloc_frag, gfp)))
874 return -ENOMEM;
875
876 buf = (char *)page_address(alloc_frag->page) + alloc_frag->offset;
877 get_page(alloc_frag->page);
878 alloc_frag->offset += len;
879 sg_init_one(rq->sg, buf + VIRTNET_RX_PAD + xdp_headroom,
880 vi->hdr_len + GOOD_PACKET_LEN);
881 err = virtqueue_add_inbuf_ctx(rq->vq, rq->sg, 1, buf, ctx, gfp);
882 if (err < 0)
883 put_page(virt_to_head_page(buf));
884 return err;
885}

887static int add_recvbuf_big(struct virtnet_info *vi, struct receive_queue *rq,
888 gfp_t gfp)
889{
890 struct page *first, *list = NULL;
891 char *p;
892 int i, err, offset;
893
894 sg_init_table(rq->sg, MAX_SKB_FRAGS + 2);
895
896 /* page in rq->sg[MAX_SKB_FRAGS + 1] is list tail */
897 for (i = MAX_SKB_FRAGS + 1; i > 1; --i) {
898 first = get_a_page(rq, gfp);
899 if (!first) {
900 if (list)
901 give_pages(rq, list);
902 return -ENOMEM;
903 }
904 sg_set_buf(&rq->sg[i], page_address(first), PAGE_SIZE);
905
906 /* chain new page in list head to match sg */
907 first->private = (unsigned long)list;
908 list = first;
909 }
910
911 first = get_a_page(rq, gfp);
912 if (!first) {
913 give_pages(rq, list);
914 return -ENOMEM;
915 }
916 p = page_address(first);
917
918 /* rq->sg[0], rq->sg[1] share the same page */
919 /* a separated rq->sg[0] for header - required in case !any_header_sg */
920 sg_set_buf(&rq->sg[0], p, vi->hdr_len);
921
922 /* rq->sg[1] for data packet, from offset */
923 offset = sizeof(struct padded_vnet_hdr);
924 sg_set_buf(&rq->sg[1], p + offset, PAGE_SIZE - offset);
925
926 /* chain first in list head */
927 first->private = (unsigned long)list;
928 err = virtqueue_add_inbuf(rq->vq, rq->sg, MAX_SKB_FRAGS + 2,
929 first, gfp);
930 if (err < 0)
931 give_pages(rq, first);
932
933 return err;
934}

936static unsigned int get_mergeable_buf_len(struct receive_queue *rq,
937 struct ewma_pkt_len *avg_pkt_len)
938{
939 const size_t hdr_len = sizeof(struct virtio_net_hdr_mrg_rxbuf);
940 unsigned int len;
941
942 len = hdr_len + clamp_t(unsigned int, ewma_pkt_len_read(avg_pkt_len),
943 rq->min_buf_len, PAGE_SIZE - hdr_len);
944 return ALIGN(len, L1_CACHE_BYTES);
945}

947static int add_recvbuf_mergeable(struct virtnet_info *vi,
948 struct receive_queue *rq, gfp_t gfp)
949{
950 struct page_frag *alloc_frag = &rq->alloc_frag;
951 unsigned int headroom = virtnet_get_headroom(vi);
952 char *buf;
953 void *ctx;
954 int err;
955 unsigned int len, hole;
956
957 len = get_mergeable_buf_len(rq, &rq->mrg_avg_pkt_len);
958 if (unlikely(!skb_page_frag_refill(len + headroom, alloc_frag, gfp)))
959 return -ENOMEM;
960
961 buf = (char *)page_address(alloc_frag->page) + alloc_frag->offset;
962 buf += headroom; /* advance address leaving hole at front of pkt */
963 get_page(alloc_frag->page);
964 alloc_frag->offset += len + headroom;
965 hole = alloc_frag->size - alloc_frag->offset;
966 if (hole < len + headroom) {
967 /* To avoid internal fragmentation, if there is very likely not
968 * enough space for another buffer, add the remaining space to
969 * the current buffer.
970 */
971 len += hole;
972 alloc_frag->offset += hole;
973 }
974
975 sg_init_one(rq->sg, buf, len);
976 ctx = mergeable_len_to_ctx(len, headroom);
977 err = virtqueue_add_inbuf_ctx(rq->vq, rq->sg, 1, buf, ctx, gfp);
978 if (err < 0)
979 put_page(virt_to_head_page(buf));
980
981 return err;
982}

984/*
985 * Returns false if we couldn't fill entirely (OOM).
986 *
987 * Normally run in the receive path, but can also be run from ndo_open
988 * before we're receiving packets, or from refill_work which is
989 * careful to disable receiving (using napi_disable).
990 */
991static bool try_fill_recv(struct virtnet_info *vi, struct receive_queue *rq,
992 gfp_t gfp)
993{
994 int err;
995 bool oom;
996
997 gfp |= __GFP_COLD;
998 do {
999 if (vi->mergeable_rx_bufs)
1000 err = add_recvbuf_mergeable(vi, rq, gfp);
1001 else if (vi->big_packets)
1002 err = add_recvbuf_big(vi, rq, gfp);
1003 else
1004 err = add_recvbuf_small(vi, rq, gfp);
1005
1006 oom = err == -ENOMEM;
1007 if (err)
1008 break;
1009 } while (rq->vq->num_free);
1010 virtqueue_kick(rq->vq);
1011 return !oom;
1012}

1059static void refill_work(struct work_struct *work)
1060{
1061 struct virtnet_info *vi =
1062 container_of(work, struct virtnet_info, refill.work);
1063 bool still_empty;
1064 int i;
1065
1066 for (i = 0; i < vi->curr_queue_pairs; i++) {
1067 struct receive_queue *rq = &vi->rq[i];
1068
1069 napi_disable(&rq->napi);
1070 still_empty = !try_fill_recv(vi, rq, GFP_KERNEL);
1071 virtnet_napi_enable(rq->vq, &rq->napi);

1074 * we will *never* try to fill again.
1075 */
1076 if (still_empty)
1077 schedule_delayed_work(&vi->refill, HZ/2);
1078 }

1081static int virtnet_receive(struct receive_queue *rq, int budget)
1082{
1083 struct virtnet_info *vi = rq->vq->vdev->priv;
1084 unsigned int len, received = 0, bytes = 0;
1085 void *buf;
1086 struct virtnet_stats *stats = this_cpu_ptr(vi->stats);
1087
1088 if (!vi->big_packets || vi->mergeable_rx_bufs) {
1089 void *ctx;
1090
1091 while (received < budget &&
1092 (buf = virtqueue_get_buf_ctx(rq->vq, &len, &ctx))) {
1093 bytes += receive_buf(vi, rq, buf, len, ctx);
1094 received++;
1095 }
1096 } else {
1097 while (received < budget &&
1098 (buf = virtqueue_get_buf(rq->vq, &len)) != NULL) {
1099 bytes += receive_buf(vi, rq, buf, len, NULL);
1100 received++;
1101 }
1102 }
1103
1104 if (rq->vq->num_free > virtqueue_get_vring_size(rq->vq) / 2) {
1105 if (!try_fill_recv(vi, rq, GFP_ATOMIC))
1106 schedule_delayed_work(&vi->refill, 0);
1107 }
1108
1109 u64_stats_update_begin(&stats->rx_syncp);
1110 stats->rx_bytes += bytes;
1111 stats->rx_packets += received;
1112 u64_stats_update_end(&stats->rx_syncp);
1113
1114 return received;
1115}

1121 struct virtnet_info *vi = sq->vq->vdev->priv;

1218static int xmit_skb(struct send_queue *sq, struct sk_buff *skb)
1219{
1220 struct virtio_net_hdr_mrg_rxbuf *hdr;
1221 const unsigned char *dest = ((struct ethhdr *)skb->data)->h_dest;
1222 struct virtnet_info *vi = sq->vq->vdev->priv;
1223 int num_sg;
1224 unsigned hdr_len = vi->hdr_len;
1225 bool can_push;
1226
1227 pr_debug("%s: xmit %p %pM\n", vi->dev->name, skb, dest);
1228
1229 can_push = vi->any_header_sg &&
1230 !((unsigned long)skb->data & (__alignof__(*hdr) - 1)) &&
1231 !skb_header_cloned(skb) && skb_headroom(skb) >= hdr_len;
1232 /* Even if we can, don't push here yet as this would skew
1233 * csum_start offset below. */
1234 if (can_push)
1235 hdr = (struct virtio_net_hdr_mrg_rxbuf *)(skb->data - hdr_len);
1236 else
1237 hdr = skb_vnet_hdr(skb);
1238
1239 if (virtio_net_hdr_from_skb(skb, &hdr->hdr,
1240 virtio_is_little_endian(vi->vdev), false,
1241 0))
1242 BUG();
1243
1244 if (vi->mergeable_rx_bufs)
1245 hdr->num_buffers = 0;
1246
1247 sg_init_table(sq->sg, skb_shinfo(skb)->nr_frags + (can_push ? 1 : 2));
1248 if (can_push) {
1249 __skb_push(skb, hdr_len);
1250 num_sg = skb_to_sgvec(skb, sq->sg, 0, skb->len);
1251 if (unlikely(num_sg < 0))
1252 return num_sg;
1253 /* Pull header back to avoid skew in tx bytes calculations. */
1254 __skb_pull(skb, hdr_len);
1255 } else {
1256 sg_set_buf(sq->sg, hdr, hdr_len);
1257 num_sg = skb_to_sgvec(skb, sq->sg + 1, 0, skb->len);
1258 if (unlikely(num_sg < 0))
1259 return num_sg;
1260 num_sg++;
1261 }
1262 return virtqueue_add_outbuf(sq->vq, sq->sg, num_sg, skb, GFP_ATOMIC);
1263}

1265static netdev_tx_t start_xmit(struct sk_buff *skb, struct net_device *dev)
1266{
1267 struct virtnet_info *vi = netdev_priv(dev);
1268 int qnum = skb_get_queue_mapping(skb);
1269 struct send_queue *sq = &vi->sq[qnum];
1270 int err;
1271 struct netdev_queue *txq = netdev_get_tx_queue(dev, qnum);
1272 bool kick = !skb->xmit_more;
1273 bool use_napi = sq->napi.weight;
1274
1275 /* Free up any pending old buffers before queueing new ones. */
1276 free_old_xmit_skbs(sq);
1277
1278 if (use_napi && kick)
1279 virtqueue_enable_cb_delayed(sq->vq);
1280
1281 /* timestamp packet in software */
1282 skb_tx_timestamp(skb);
1283
1284 /* Try to transmit */
1285 err = xmit_skb(sq, skb);
1286
1287 /* This should not happen! */
1288 if (unlikely(err)) {
1289 dev->stats.tx_fifo_errors++;
1290 if (net_ratelimit())
1291 dev_warn(&dev->dev,
1292 "Unexpected TXQ (%d) queue failure: %d\n", qnum, err);
1293 dev->stats.tx_dropped++;
1294 dev_kfree_skb_any(skb);
1295 return NETDEV_TX_OK;
1296 }
1297
1298 /* Don't wait up for transmitted skbs to be freed. */
1299 if (!use_napi) {
1300 skb_orphan(skb);
1301 nf_reset(skb);
1302 }
1303
1304 /* If running out of space, stop queue to avoid getting packets that we
1305 * are then unable to transmit.
1306 * An alternative would be to force queuing layer to requeue the skb by
1307 * returning NETDEV_TX_BUSY. However, NETDEV_TX_BUSY should not be
1308 * returned in a normal path of operation: it means that driver is not
1309 * maintaining the TX queue stop/start state properly, and causes
1310 * the stack to do a non-trivial amount of useless work.
1311 * Since most packets only take 1 or 2 ring slots, stopping the queue
1312 * early means 16 slots are typically wasted.
1313 */
1314 if (sq->vq->num_free < 2+MAX_SKB_FRAGS) {
1315 netif_stop_subqueue(dev, qnum);
1316 if (!use_napi &&
1317 unlikely(!virtqueue_enable_cb_delayed(sq->vq))) {
1318 /* More just got used, free them then recheck. */
1319 free_old_xmit_skbs(sq);
1320 if (sq->vq->num_free >= 2+MAX_SKB_FRAGS) {
1321 netif_start_subqueue(dev, qnum);
1322 virtqueue_disable_cb(sq->vq);
1323 }
1324 }
1325 }
1326
1327 if (kick || netif_xmit_stopped(txq))
1328 virtqueue_kick(sq->vq);
1329
1330 return NETDEV_TX_OK;
1331}

1333/*
1334 * Send command via the control virtqueue and check status. Commands
1335 * supported by the hypervisor, as indicated by feature bits, should
1336 * never fail unless improperly formatted.
1337 */
1338static bool virtnet_send_command(struct virtnet_info *vi, u8 class, u8 cmd,
1339 struct scatterlist *out)
1340{
1341 struct scatterlist *sgs[4], hdr, stat;
1342 unsigned out_num = 0, tmp;
1343
1344 /* Caller should know better */
1345 BUG_ON(!virtio_has_feature(vi->vdev, VIRTIO_NET_F_CTRL_VQ));
1346
1347 vi->ctrl->status = ~0;
1348 vi->ctrl->hdr.class = class;
1349 vi->ctrl->hdr.cmd = cmd;
1350 /* Add header */
1351 sg_init_one(&hdr, &vi->ctrl->hdr, sizeof(vi->ctrl->hdr));
1352 sgs[out_num++] = &hdr;
1353
1354 if (out)
1355 sgs[out_num++] = out;
1356
1357 /* Add return status. */
1358 sg_init_one(&stat, &vi->ctrl->status, sizeof(vi->ctrl->status));
1359 sgs[out_num] = &stat;
1360
1361 BUG_ON(out_num + 1 > ARRAY_SIZE(sgs));
1362 virtqueue_add_sgs(vi->cvq, sgs, out_num, 1, vi, GFP_ATOMIC);
1363
1364 if (unlikely(!virtqueue_kick(vi->cvq)))
1365 return vi->ctrl->status == VIRTIO_NET_OK;
1366
1367 /* Spin for a response, the kick causes an ioport write, trapping
1368 * into the hypervisor, so the request should be handled immediately.
1369 */
1370 while (!virtqueue_get_buf(vi->cvq, &tmp) &&
1371 !virtqueue_is_broken(vi->cvq))
1372 cpu_relax();
1373
1374 return vi->ctrl->status == VIRTIO_NET_OK;
1375}

1378{

1380 struct virtio_device *vdev = vi->vdev;

1391 goto out;

1529static void virtnet_set_rx_mode(struct net_device *dev)
1530{
1531 struct virtnet_info *vi = netdev_priv(dev);
1532 struct scatterlist sg[2];
1533 struct virtio_net_ctrl_mac *mac_data;
1534 struct netdev_hw_addr *ha;
1535 int uc_count;
1536 int mc_count;
1537 void *buf;
1538 int i;
1539
1540 /* We can't dynamically set ndo_set_rx_mode, so return gracefully */
1541 if (!virtio_has_feature(vi->vdev, VIRTIO_NET_F_CTRL_RX))
1542 return;
1543
1544 vi->ctrl->promisc = ((dev->flags & IFF_PROMISC) != 0);
1545 vi->ctrl->allmulti = ((dev->flags & IFF_ALLMULTI) != 0);
1546
1547 sg_init_one(sg, &vi->ctrl->promisc, sizeof(vi->ctrl->promisc));
1548
1549 if (!virtnet_send_command(vi, VIRTIO_NET_CTRL_RX,
1550 VIRTIO_NET_CTRL_RX_PROMISC, sg))
1551 dev_warn(&dev->dev, "Failed to %sable promisc mode.\n",
1552 vi->ctrl->promisc ? "en" : "dis");
1553
1554 sg_init_one(sg, &vi->ctrl->allmulti, sizeof(vi->ctrl->allmulti));
1555
1556 if (!virtnet_send_command(vi, VIRTIO_NET_CTRL_RX,
1557 VIRTIO_NET_CTRL_RX_ALLMULTI, sg))
1558 dev_warn(&dev->dev, "Failed to %sable allmulti mode.\n",
1559 vi->ctrl->allmulti ? "en" : "dis");
1560
1561 uc_count = netdev_uc_count(dev);
1562 mc_count = netdev_mc_count(dev);
1563 /* MAC filter - use one buffer for both lists */
1564 buf = kzalloc(((uc_count + mc_count) * ETH_ALEN) +
1565 (2 * sizeof(mac_data->entries)), GFP_ATOMIC);
1566 mac_data = buf;
1567 if (!buf)
1568 return;
1569
1570 sg_init_table(sg, 2);
1571
1572 /* Store the unicast list and count in the front of the buffer */
1573 mac_data->entries = cpu_to_virtio32(vi->vdev, uc_count);
1574 i = 0;
1575 netdev_for_each_uc_addr(ha, dev)
1576 memcpy(&mac_data->macs[i++][0], ha->addr, ETH_ALEN);
1577
1578 sg_set_buf(&sg[0], mac_data,
1579 sizeof(mac_data->entries) + (uc_count * ETH_ALEN));
1580
1581 /* multicast list and count fill the end */
1582 mac_data = (void *)&mac_data->macs[uc_count][0];
1583
1584 mac_data->entries = cpu_to_virtio32(vi->vdev, mc_count);
1585 i = 0;
1586 netdev_for_each_mc_addr(ha, dev)
1587 memcpy(&mac_data->macs[i++][0], ha->addr, ETH_ALEN);
1588
1589 sg_set_buf(&sg[1], mac_data,
1590 sizeof(mac_data->entries) + (mc_count * ETH_ALEN));
1591
1592 if (!virtnet_send_command(vi, VIRTIO_NET_CTRL_MAC,
1593 VIRTIO_NET_CTRL_MAC_TABLE_SET, sg))
1594 dev_warn(&dev->dev, "Failed to set MAC filter table.\n");
1595
1596 kfree(buf);
1597}

1957static int virtnet_xdp_set(struct net_device *dev, struct bpf_prog *prog,
1958 struct netlink_ext_ack *extack)
1959{
1960 unsigned long int max_sz = PAGE_SIZE - sizeof(struct padded_vnet_hdr);
1961 struct virtnet_info *vi = netdev_priv(dev);
1962 struct bpf_prog *old_prog;
1963 u16 xdp_qp = 0, curr_qp;
1964 int i, err;
1965
1966 if (!virtio_has_feature(vi->vdev, VIRTIO_NET_F_CTRL_GUEST_OFFLOADS)
1967 && (virtio_has_feature(vi->vdev, VIRTIO_NET_F_GUEST_TSO4) ||
1968 virtio_has_feature(vi->vdev, VIRTIO_NET_F_GUEST_TSO6) ||
1969 virtio_has_feature(vi->vdev, VIRTIO_NET_F_GUEST_ECN) ||
1970 virtio_has_feature(vi->vdev, VIRTIO_NET_F_GUEST_UFO))) {
1971 NL_SET_ERR_MSG_MOD(extack, "Can't set XDP while host is implementing LRO, disable LRO first");
1972 return -EOPNOTSUPP;
1973 }
1974
1975 if (vi->mergeable_rx_bufs && !vi->any_header_sg) {
1976 NL_SET_ERR_MSG_MOD(extack, "XDP expects header/data in single page, any_header_sg required");
1977 return -EINVAL;
1978 }
1979
1980 if (dev->mtu > max_sz) {
1981 NL_SET_ERR_MSG_MOD(extack, "MTU too large to enable XDP");
1982 netdev_warn(dev, "XDP requires MTU less than %lu\n", max_sz);
1983 return -EINVAL;
1984 }
1985
1986 curr_qp = vi->curr_queue_pairs - vi->xdp_queue_pairs;
1987 if (prog)
1988 xdp_qp = nr_cpu_ids;
1989
1990 /* XDP requires extra queues for XDP_TX */
1991 if (curr_qp + xdp_qp > vi->max_queue_pairs) {
1992 NL_SET_ERR_MSG_MOD(extack, "Too few free TX rings available");
1993 netdev_warn(dev, "request %i queues but max is %i\n",
1994 curr_qp + xdp_qp, vi->max_queue_pairs);
1995 return -ENOMEM;
1996 }
1997
1998 if (prog) {
1999 prog = bpf_prog_add(prog, vi->max_queue_pairs - 1);
2000 if (IS_ERR(prog))
2001 return PTR_ERR(prog);
2002 }
2003
2004 /* Make sure NAPI is not using any XDP TX queues for RX. */
2005 if (netif_running(dev))
2006 for (i = 0; i < vi->max_queue_pairs; i++)
2007 napi_disable(&vi->rq[i].napi);
2008
2009 netif_set_real_num_rx_queues(dev, curr_qp + xdp_qp);
2010 err = _virtnet_set_queues(vi, curr_qp + xdp_qp);
2011 if (err)
2012 goto err;
2013 vi->xdp_queue_pairs = xdp_qp;
2014
2015 for (i = 0; i < vi->max_queue_pairs; i++) {
2016 old_prog = rtnl_dereference(vi->rq[i].xdp_prog);
2017 rcu_assign_pointer(vi->rq[i].xdp_prog, prog);
2018 if (i == 0) {
2019 if (!old_prog)
2020 virtnet_clear_guest_offloads(vi);
2021 if (!prog)
2022 virtnet_restore_guest_offloads(vi);
2023 }
2024 if (old_prog)
2025 bpf_prog_put(old_prog);
2026 if (netif_running(dev))
2027 virtnet_napi_enable(vi->rq[i].vq, &vi->rq[i].napi);
2028 }
2029
2030 return 0;
2031
2032err:
2033 for (i = 0; i < vi->max_queue_pairs; i++)
2034 virtnet_napi_enable(vi->rq[i].vq, &vi->rq[i].napi);
2035 if (prog)
2036 bpf_prog_sub(prog, vi->max_queue_pairs - 1);
2037 return err;
2038}

2226/* How large should a single buffer be so a queue full of these can fit at
2227 * least one full packet?
2228 * Logic below assumes the mergeable buffer header is used.
2229 */
2230static unsigned int mergeable_min_buf_len(struct virtnet_info *vi, struct virtqueue *vq)
2231{
2232 const unsigned int hdr_len = sizeof(struct virtio_net_hdr_mrg_rxbuf);
2233 unsigned int rq_size = virtqueue_get_vring_size(vq);
2234 unsigned int packet_len = vi->big_packets ? IP_MAX_MTU : vi->dev->max_mtu;
2235 unsigned int buf_len = hdr_len + ETH_HLEN + VLAN_HLEN + packet_len;
2236 unsigned int min_buf_len = DIV_ROUND_UP(buf_len, rq_size);

2238 return max(max(min_buf_len, hdr_len) - hdr_len,
2239 (unsigned int)GOOD_PACKET_LEN);

2584 vi->hdr_len = sizeof(struct virtio_net_hdr_mrg_rxbuf);
2585 else
2586 vi->hdr_len = sizeof(struct virtio_net_hdr);
2587
2588 if (virtio_has_feature(vdev, VIRTIO_F_ANY_LAYOUT) ||
2589 virtio_has_feature(vdev, VIRTIO_F_VERSION_1))
2590 vi->any_header_sg = true;
2591
2592 if (virtio_has_feature(vdev, VIRTIO_NET_F_CTRL_VQ))
2593 vi->has_cvq = true;
2594
2595 if (virtio_has_feature(vdev, VIRTIO_NET_F_MTU)) {
2596 mtu = virtio_cread16(vdev,
2597 offsetof(struct virtio_net_config,
2598 mtu));
2599 if (mtu < dev->min_mtu) {
2600 /* Should never trigger: MTU was previously validated
2601 * in virtnet_validate.
2602 */
2603 dev_err(&vdev->dev, "device MTU appears to have changed "
2604 "it is now %d < %d", mtu, dev->min_mtu);
2605 goto free_stats;
2606 }
2607
2608 dev->mtu = mtu;
2609 dev->max_mtu = mtu;
2610
2611 /* TODO: size buffers correctly in this case. */
2612 if (dev->mtu > ETH_DATA_LEN)
2613 vi->big_packets = true;
2614 }
2615
2616 if (vi->any_header_sg)
2617 dev->needed_headroom = vi->hdr_len;
2618
2619 /* Enable multiqueue by default */
2620 if (num_online_cpus() >= max_queue_pairs)
2621 vi->curr_queue_pairs = max_queue_pairs;
2622 else
2623 vi->curr_queue_pairs = num_online_cpus();
2624 vi->max_queue_pairs = max_queue_pairs;
2625
2626 /* Allocate/initialize the rx/tx queues, and invoke find_vqs */
2627 err = init_vqs(vi);
2628 if (err)
2629 goto free_stats;
2630
2631#ifdef CONFIG_SYSFS
2632 if (vi->mergeable_rx_bufs)
2633 dev->sysfs_rx_queue_group = &virtio_net_mrg_rx_group;
2634#endif
2635 netif_set_real_num_tx_queues(dev, vi->curr_queue_pairs);
2636 netif_set_real_num_rx_queues(dev, vi->curr_queue_pairs);
2637
2638 virtnet_init_settings(dev);
2639
2640 err = register_netdev(dev);
2641 if (err) {
2642 pr_debug("virtio_net: registering device failed\n");
2643 goto free_vqs;
2644 }
2645
2646 virtio_device_ready(vdev);
2647
2648 err = virtnet_cpu_notif_add(vi);
2649 if (err) {
2650 pr_debug("virtio_net: registering cpu notifier failed\n");
2651 goto free_unregister_netdev;
2652 }
2653
2654 virtnet_set_queues(vi, vi->curr_queue_pairs);
2655
2656 /* Assume link up if device can't report link status,
2657 otherwise get link status from config. */
2658 netif_carrier_off(dev);
2659 if (virtio_has_feature(vi->vdev, VIRTIO_NET_F_STATUS)) {
2660 schedule_work(&vi->config_work);
2661 } else {
2662 vi->status = VIRTIO_NET_S_LINK_UP;
2663 netif_carrier_on(dev);
2664 }
2665
2666 for (i = 0; i < ARRAY_SIZE(guest_offloads); i++)
2667 if (virtio_has_feature(vi->vdev, guest_offloads[i]))
2668 set_bit(guest_offloads[i], &vi->guest_offloads);
2669
2670 pr_debug("virtnet: registered device %s with %d RX and TX vq's\n",
2671 dev->name, max_queue_pairs);
2672
2673 return 0;
2674
2675free_unregister_netdev:
2676 vi->vdev->config->reset(vdev);
2677
2678 unregister_netdev(dev);
2679free_vqs:
2680 cancel_delayed_work_sync(&vi->refill);
2681 free_receive_page_frags(vi);
2682 virtnet_del_vqs(vi);
2683free_stats:
2684 free_percpu(vi->stats);
2685free:
2686 free_netdev(dev);
2687 return err;
2688}
