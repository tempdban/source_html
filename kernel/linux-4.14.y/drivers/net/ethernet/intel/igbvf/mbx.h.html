27#ifndef _E1000_MBX_H_
28#define _E1000_MBX_H_
29
30#include "vf.h"
31
32#define E1000_V2PMAILBOX_REQ 0x00000001 /* Request for PF Ready bit */
33#define E1000_V2PMAILBOX_ACK 0x00000002 /* Ack PF message received */
34#define E1000_V2PMAILBOX_VFU 0x00000004 /* VF owns the mailbox buffer */
35#define E1000_V2PMAILBOX_PFU 0x00000008 /* PF owns the mailbox buffer */
36#define E1000_V2PMAILBOX_PFSTS 0x00000010 /* PF wrote a message in the MB */
37#define E1000_V2PMAILBOX_PFACK 0x00000020 /* PF ack the previous VF msg */
38#define E1000_V2PMAILBOX_RSTI 0x00000040 /* PF has reset indication */
39#define E1000_V2PMAILBOX_RSTD 0x00000080 /* PF has indicated reset done */
40#define E1000_V2PMAILBOX_R2C_BITS 0x000000B0 /* All read to clear bits */
41
42#define E1000_VFMAILBOX_SIZE 16 /* 16 32 bit words - 64 bytes */
43
44/* If it's a E1000_VF_* msg then it originates in the VF and is sent to the
45 * PF. The reverse is true if it is E1000_PF_*.
46 * Message ACK's are the value or'd with 0xF0000000
47 */
48/* Messages below or'd with this are the ACK */
49#define E1000_VT_MSGTYPE_ACK 0x80000000
50/* Messages below or'd with this are the NACK */
51#define E1000_VT_MSGTYPE_NACK 0x40000000
52/* Indicates that VF is still clear to send requests */
53#define E1000_VT_MSGTYPE_CTS 0x20000000
54
55/* We have a total wait time of 1s for vf mailbox posted messages */
56#define E1000_VF_MBX_INIT_TIMEOUT 2000 /* retry count for mbx timeout */
57#define E1000_VF_MBX_INIT_DELAY 500 /* usec delay between retries */
58
59#define E1000_VT_MSGINFO_SHIFT 16
60/* bits 23:16 are used for exra info for certain messages */
61#define E1000_VT_MSGINFO_MASK (0xFF << E1000_VT_MSGINFO_SHIFT)
62
63#define E1000_VF_RESET 0x01 /* VF requests reset */
64#define E1000_VF_SET_MAC_ADDR 0x02 /* VF requests PF to set MAC addr */
65/* VF requests PF to clear all unicast MAC filters */
66#define E1000_VF_MAC_FILTER_CLR (0x01 << E1000_VT_MSGINFO_SHIFT)
67/* VF requests PF to add unicast MAC filter */
68#define E1000_VF_MAC_FILTER_ADD (0x02 << E1000_VT_MSGINFO_SHIFT)
69#define E1000_VF_SET_MULTICAST 0x03 /* VF requests PF to set MC addr */
70#define E1000_VF_SET_VLAN 0x04 /* VF requests PF to set VLAN */
71#define E1000_VF_SET_LPE 0x05 /* VF requests PF to set VMOLR.LPE */
72
73#define E1000_PF_CONTROL_MSG 0x0100 /* PF control message */
74
75void e1000_init_mbx_ops_generic(struct e1000_hw *hw);
76s32 e1000_init_mbx_params_vf(struct e1000_hw *);
77
78#endif /* _E1000_MBX_H_ */
