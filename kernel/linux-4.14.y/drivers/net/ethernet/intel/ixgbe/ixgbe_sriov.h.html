66static inline void ixgbe_set_vmvir(struct ixgbe_adapter *adapter,
67 u16 vid, u16 qos, u32 vf)
68{
69 struct ixgbe_hw *hw = &adapter->hw;
70 u32 vmvir = vid | (qos << VLAN_PRIO_SHIFT) | IXGBE_VMVIR_VLANA_DEFAULT;
71
72 IXGBE_WRITE_REG(hw, IXGBE_VMVIR(vf), vmvir);
73}
