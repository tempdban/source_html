29#ifndef _IXGBE_MBX_H_
30#define _IXGBE_MBX_H_
31
32#include "ixgbe_type.h"
33
34#define IXGBE_VFMAILBOX_SIZE 16 /* 16 32 bit words - 64 bytes */
35#define IXGBE_ERR_MBX -100
36
37#define IXGBE_VFMAILBOX 0x002FC
38#define IXGBE_VFMBMEM 0x00200
39
40#define IXGBE_PFMAILBOX_STS 0x00000001 /* Initiate message send to VF */
41#define IXGBE_PFMAILBOX_ACK 0x00000002 /* Ack message recv'd from VF */
42#define IXGBE_PFMAILBOX_VFU 0x00000004 /* VF owns the mailbox buffer */
43#define IXGBE_PFMAILBOX_PFU 0x00000008 /* PF owns the mailbox buffer */
44#define IXGBE_PFMAILBOX_RVFU 0x00000010 /* Reset VFU - used when VF stuck */
45
46#define IXGBE_MBVFICR_VFREQ_MASK 0x0000FFFF /* bits for VF messages */
47#define IXGBE_MBVFICR_VFREQ_VF1 0x00000001 /* bit for VF 1 message */
48#define IXGBE_MBVFICR_VFACK_MASK 0xFFFF0000 /* bits for VF acks */
49#define IXGBE_MBVFICR_VFACK_VF1 0x00010000 /* bit for VF 1 ack */
50
51
52/* If it's a IXGBE_VF_* msg then it originates in the VF and is sent to the
53 * PF. The reverse is true if it is IXGBE_PF_*.
54 * Message ACK's are the value or'd with 0xF0000000
55 */
56#define IXGBE_VT_MSGTYPE_ACK 0x80000000 /* Messages below or'd with
57 * this are the ACK */
58#define IXGBE_VT_MSGTYPE_NACK 0x40000000 /* Messages below or'd with
59 * this are the NACK */
60#define IXGBE_VT_MSGTYPE_CTS 0x20000000 /* Indicates that VF is still
61 clear to send requests */
62#define IXGBE_VT_MSGINFO_SHIFT 16
63/* bits 23:16 are used for exra info for certain messages */
64#define IXGBE_VT_MSGINFO_MASK (0xFF << IXGBE_VT_MSGINFO_SHIFT)
65
66/* definitions to support mailbox API version negotiation */
67
68/*
69 * Each element denotes a version of the API; existing numbers may not
70 * change; any additions must go at the end
71 */
72enum ixgbe_pfvf_api_rev {
73 ixgbe_mbox_api_10, /* API version 1.0, linux/freebsd VF driver */
74 ixgbe_mbox_api_20, /* API version 2.0, solaris Phase1 VF driver */
75 ixgbe_mbox_api_11, /* API version 1.1, linux/freebsd VF driver */
76 ixgbe_mbox_api_12, /* API version 1.2, linux/freebsd VF driver */
77 ixgbe_mbox_api_13, /* API version 1.3, linux/freebsd VF driver */
78 /* This value should always be last */
79 ixgbe_mbox_api_unknown, /* indicates that API version is not known */
80};
81
82/* mailbox API, legacy requests */
83#define IXGBE_VF_RESET 0x01 /* VF requests reset */
84#define IXGBE_VF_SET_MAC_ADDR 0x02 /* VF requests PF to set MAC addr */
85#define IXGBE_VF_SET_MULTICAST 0x03 /* VF requests PF to set MC addr */
86#define IXGBE_VF_SET_VLAN 0x04 /* VF requests PF to set VLAN */
87
88/* mailbox API, version 1.0 VF requests */
89#define IXGBE_VF_SET_LPE 0x05 /* VF requests PF to set VMOLR.LPE */
90#define IXGBE_VF_SET_MACVLAN 0x06 /* VF requests PF for unicast filter */
91#define IXGBE_VF_API_NEGOTIATE 0x08 /* negotiate API version */
92
93/* mailbox API, version 1.1 VF requests */
94#define IXGBE_VF_GET_QUEUES 0x09 /* get queue configuration */
95
96/* GET_QUEUES return data indices within the mailbox */
97#define IXGBE_VF_TX_QUEUES 1 /* number of Tx queues supported */
98#define IXGBE_VF_RX_QUEUES 2 /* number of Rx queues supported */
99#define IXGBE_VF_TRANS_VLAN 3 /* Indication of port vlan */
100#define IXGBE_VF_DEF_QUEUE 4 /* Default queue offset */
101
102/* mailbox API, version 1.2 VF requests */
103#define IXGBE_VF_GET_RETA 0x0a /* VF request for RETA */
104#define IXGBE_VF_GET_RSS_KEY 0x0b /* get RSS key */
105
106#define IXGBE_VF_UPDATE_XCAST_MODE 0x0c
107
108/* length of permanent address message returned from PF */
109#define IXGBE_VF_PERMADDR_MSG_LEN 4
110/* word in permanent address message with the current multicast type */
111#define IXGBE_VF_MC_TYPE_WORD 3
112
113#define IXGBE_PF_CONTROL_MSG 0x0100 /* PF control message */
114
115#define IXGBE_VF_MBX_INIT_TIMEOUT 2000 /* number of retries on mailbox */
116#define IXGBE_VF_MBX_INIT_DELAY 500 /* microseconds between retries */
117

121s32 ixgbe_check_for_ack(struct ixgbe_hw *, u16);
122s32 ixgbe_check_for_rst(struct ixgbe_hw *, u16);
123#ifdef CONFIG_PCI_IOV
124void ixgbe_init_mbx_params_pf(struct ixgbe_hw *);
125#endif /* CONFIG_PCI_IOV */
