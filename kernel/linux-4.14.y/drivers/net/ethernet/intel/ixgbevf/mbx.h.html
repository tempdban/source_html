27#ifndef _IXGBE_MBX_H_
28#define _IXGBE_MBX_H_
29
30#include "vf.h"
31
32#define IXGBE_VFMAILBOX_SIZE 16 /* 16 32 bit words - 64 bytes */
33#define IXGBE_ERR_MBX -100
34
35#define IXGBE_VFMAILBOX 0x002FC
36#define IXGBE_VFMBMEM 0x00200
37
38/* Define mailbox register bits */
39#define IXGBE_VFMAILBOX_REQ 0x00000001 /* Request for PF Ready bit */
40#define IXGBE_VFMAILBOX_ACK 0x00000002 /* Ack PF message received */
41#define IXGBE_VFMAILBOX_VFU 0x00000004 /* VF owns the mailbox buffer */
42#define IXGBE_VFMAILBOX_PFU 0x00000008 /* PF owns the mailbox buffer */
43#define IXGBE_VFMAILBOX_PFSTS 0x00000010 /* PF wrote a message in the MB */
44#define IXGBE_VFMAILBOX_PFACK 0x00000020 /* PF ack the previous VF msg */
45#define IXGBE_VFMAILBOX_RSTI 0x00000040 /* PF has reset indication */
46#define IXGBE_VFMAILBOX_RSTD 0x00000080 /* PF has indicated reset done */
47#define IXGBE_VFMAILBOX_R2C_BITS 0x000000B0 /* All read to clear bits */
48
49#define IXGBE_PFMAILBOX(x) (0x04B00 + (4 * (x)))
50#define IXGBE_PFMBMEM(vfn) (0x13000 + (64 * (vfn)))
51
52#define IXGBE_PFMAILBOX_STS 0x00000001 /* Initiate message send to VF */
53#define IXGBE_PFMAILBOX_ACK 0x00000002 /* Ack message recv'd from VF */
54#define IXGBE_PFMAILBOX_VFU 0x00000004 /* VF owns the mailbox buffer */
55#define IXGBE_PFMAILBOX_PFU 0x00000008 /* PF owns the mailbox buffer */
56#define IXGBE_PFMAILBOX_RVFU 0x00000010 /* Reset VFU - used when VF stuck */
57
58#define IXGBE_MBVFICR_VFREQ_MASK 0x0000FFFF /* bits for VF messages */
59#define IXGBE_MBVFICR_VFREQ_VF1 0x00000001 /* bit for VF 1 message */
60#define IXGBE_MBVFICR_VFACK_MASK 0xFFFF0000 /* bits for VF acks */
61#define IXGBE_MBVFICR_VFACK_VF1 0x00010000 /* bit for VF 1 ack */
62
63/* If it's a IXGBE_VF_* msg then it originates in the VF and is sent to the
64 * PF. The reverse is true if it is IXGBE_PF_*.
65 * Message ACK's are the value or'd with 0xF0000000
66 */
67/* Messages below or'd with this are the ACK */
68#define IXGBE_VT_MSGTYPE_ACK 0x80000000
69/* Messages below or'd with this are the NACK */
70#define IXGBE_VT_MSGTYPE_NACK 0x40000000
71/* Indicates that VF is still clear to send requests */
72#define IXGBE_VT_MSGTYPE_CTS 0x20000000
73#define IXGBE_VT_MSGINFO_SHIFT 16
74/* bits 23:16 are used for exra info for certain messages */
75#define IXGBE_VT_MSGINFO_MASK (0xFF << IXGBE_VT_MSGINFO_SHIFT)
76
77/* definitions to support mailbox API version negotiation */
78
79/* each element denotes a version of the API; existing numbers may not
80 * change; any additions must go at the end
81 */
82enum ixgbe_pfvf_api_rev {
83 ixgbe_mbox_api_10, /* API version 1.0, linux/freebsd VF driver */
84 ixgbe_mbox_api_20, /* API version 2.0, solaris Phase1 VF driver */
85 ixgbe_mbox_api_11, /* API version 1.1, linux/freebsd VF driver */
86 ixgbe_mbox_api_12, /* API version 1.2, linux/freebsd VF driver */
87 ixgbe_mbox_api_13, /* API version 1.3, linux/freebsd VF driver */
88 /* This value should always be last */
89 ixgbe_mbox_api_unknown, /* indicates that API version is not known */
90};
91
92/* mailbox API, legacy requests */
93#define IXGBE_VF_RESET 0x01 /* VF requests reset */
94#define IXGBE_VF_SET_MAC_ADDR 0x02 /* VF requests PF to set MAC addr */
95#define IXGBE_VF_SET_MULTICAST 0x03 /* VF requests PF to set MC addr */
96#define IXGBE_VF_SET_VLAN 0x04 /* VF requests PF to set VLAN */
97
98/* mailbox API, version 1.0 VF requests */
99#define IXGBE_VF_SET_LPE 0x05 /* VF requests PF to set VMOLR.LPE */
100#define IXGBE_VF_SET_MACVLAN 0x06 /* VF requests PF for unicast filter */
101#define IXGBE_VF_API_NEGOTIATE 0x08 /* negotiate API version */
102
103/* mailbox API, version 1.1 VF requests */
104#define IXGBE_VF_GET_QUEUE 0x09 /* get queue configuration */
105
106/* GET_QUEUES return data indices within the mailbox */
107#define IXGBE_VF_TX_QUEUES 1 /* number of Tx queues supported */
108#define IXGBE_VF_RX_QUEUES 2 /* number of Rx queues supported */
109#define IXGBE_VF_TRANS_VLAN 3 /* Indication of port VLAN */
110#define IXGBE_VF_DEF_QUEUE 4 /* Default queue offset */
111
112/* mailbox API, version 1.2 VF requests */
113#define IXGBE_VF_GET_RETA 0x0a /* VF request for RETA */
114#define IXGBE_VF_GET_RSS_KEY 0x0b /* get RSS hash key */
115
116#define IXGBE_VF_UPDATE_XCAST_MODE 0x0c
117
118/* length of permanent address message returned from PF */
119#define IXGBE_VF_PERMADDR_MSG_LEN 4
120/* word in permanent address message with the current multicast type */
121#define IXGBE_VF_MC_TYPE_WORD 3
122
123#define IXGBE_PF_CONTROL_MSG 0x0100 /* PF control message */
124
125#define IXGBE_VF_MBX_INIT_TIMEOUT 2000 /* number of retries on mailbox */
126#define IXGBE_VF_MBX_INIT_DELAY 500 /* microseconds between retries */
127
128/* forward declaration of the HW struct */
129struct ixgbe_hw;
130
131#endif /* _IXGBE_MBX_H_ */
