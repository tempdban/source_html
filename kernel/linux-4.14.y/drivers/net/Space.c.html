37/* A unified ethernet device probe. This is the easiest way to have every
38 * ethernet adaptor have the name "eth[0123...]".
39 */
40
41struct devprobe2 {
42 struct net_device *(*probe)(int unit);
43 int status; /* non-zero if autoprobe has failed */
44};
45
46static int __init probe_list2(int unit, struct devprobe2 *p, int autoprobe)
47{
48 struct net_device *dev;
49
50 for (; p->probe; p++) {
51 if (autoprobe && p->status)
52 continue;
53 dev = p->probe(unit);
54 if (!IS_ERR(dev))
55 return 0;
56 if (autoprobe)
57 p->status = PTR_ERR(dev);
58 }
59 return -ENODEV;
60}

126/* Unified ethernet device probe, segmented per architecture and
127 * per bus interface. This drives the legacy devices only for now.
128 */
129
130static void __init ethif_probe2(int unit)
131{
132 unsigned long base_addr = netdev_boot_base("eth", unit);
133
134 if (base_addr == 1)
135 return;
136
137 (void)(probe_list2(unit, m68k_probes, base_addr == 0) &&
138 probe_list2(unit, isa_probes, base_addr == 0));
139}

141/* Statically configured drivers -- order matters here. */
142static int __init net_olddevs_init(void)
143{
144 int num;
145
146#ifdef CONFIG_SBNI
147 for (num = 0; num < 8; ++num)
148 sbni_probe(num);
149#endif
150 for (num = 0; num < 8; ++num)
151 ethif_probe2(num);
152
153#ifdef CONFIG_COPS
154 cops_probe(0);
155 cops_probe(1);
156 cops_probe(2);
157#endif
158#ifdef CONFIG_LTPC
159 ltpc_probe();
160#endif
161
162 return 0;
163}
164
165device_initcall(net_olddevs_init);
