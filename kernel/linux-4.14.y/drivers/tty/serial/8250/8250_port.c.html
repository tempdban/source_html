315/* Uart divisor latch read */
316static int default_serial_dl_read(struct uart_8250_port *up)
317{
318 return serial_in(up, UART_DLL) | serial_in(up, UART_DLM) << 8;
319}
320
321/* Uart divisor latch write */
322static void default_serial_dl_write(struct uart_8250_port *up, int value)
323{
324 serial_out(up, UART_DLL, value & 0xff);
325 serial_out(up, UART_DLM, value >> 8 & 0xff);
326}

330/* Au1x00/RT288x UART hardware has a weird register layout */
331static const s8 au_io_in_map[8] = {
332 0, /* UART_RX */
333 2, /* UART_IER */
334 3, /* UART_IIR */
335 5, /* UART_LCR */
336 6, /* UART_MCR */
337 7, /* UART_LSR */
338 8, /* UART_MSR */
339 -1, /* UART_SCR (unmapped) */
340};

342static const s8 au_io_out_map[8] = {
343 1, /* UART_TX */
344 2, /* UART_IER */
345 4, /* UART_FCR */
346 5, /* UART_LCR */
347 6, /* UART_MCR */
348 -1, /* UART_LSR (unmapped) */
349 -1, /* UART_MSR (unmapped) */
350 -1, /* UART_SCR (unmapped) */
351};

353unsigned int au_serial_in(struct uart_port *p, int offset)
354{
355 if (offset >= ARRAY_SIZE(au_io_in_map))
356 return UINT_MAX;
357 offset = au_io_in_map[offset];
358 if (offset < 0)
359 return UINT_MAX;
360 return __raw_readl(p->membase + (offset << p->regshift));
361}
362
363void au_serial_out(struct uart_port *p, int offset, int value)
364{
365 if (offset >= ARRAY_SIZE(au_io_out_map))
366 return;
367 offset = au_io_out_map[offset];
368 if (offset < 0)
369 return;
370 __raw_writel(value, p->membase + (offset << p->regshift));
371}

462static void set_io_from_upio(struct uart_port *p)
463{
464 struct uart_8250_port *up = up_to_u8250p(p);
465
466 up->dl_read = default_serial_dl_read;
467 up->dl_write = default_serial_dl_write;
468
469 switch (p->iotype) {
470 case UPIO_HUB6:
471 p->serial_in = hub6_serial_in;
472 p->serial_out = hub6_serial_out;
473 break;
474
475 case UPIO_MEM:
476 p->serial_in = mem_serial_in;
477 p->serial_out = mem_serial_out;
478 break;
479
480 case UPIO_MEM16:
481 p->serial_in = mem16_serial_in;
482 p->serial_out = mem16_serial_out;
483 break;
484
485 case UPIO_MEM32:
486 p->serial_in = mem32_serial_in;
487 p->serial_out = mem32_serial_out;
488 break;
489
490 case UPIO_MEM32BE:
491 p->serial_in = mem32be_serial_in;
492 p->serial_out = mem32be_serial_out;
493 break;
494
495#ifdef CONFIG_SERIAL_8250_RT288X
496 case UPIO_AU:
497 p->serial_in = au_serial_in;
498 p->serial_out = au_serial_out;
499 up->dl_read = au_serial_dl_read;
500 up->dl_write = au_serial_dl_write;
501 break;
502#endif
503
504 default:
505 p->serial_in = io_serial_in;
506 p->serial_out = io_serial_out;
507 break;
508 }
509 /* Remember loaded iotype */
510 up->cur_iotype = p->iotype;
511 p->handle_irq = serial8250_default_handle_irq;
512}

514static void
515serial_port_out_sync(struct uart_port *p, int offset, int value)
516{
517 switch (p->iotype) {
518 case UPIO_MEM:
519 case UPIO_MEM16:
520 case UPIO_MEM32:
521 case UPIO_MEM32BE:
522 case UPIO_AU:
523 p->serial_out(p, offset, value);
524 p->serial_in(p, UART_LCR); /* safe, no side-effects */
525 break;
526 default:
527 p->serial_out(p, offset, value);
528 }
529}

603/**
604 * serial8250_em485_init() - put uart_8250_port into rs485 emulating
605 * @p: uart_8250_port port instance
606 *
607 * The function is used to start rs485 software emulating on the
608 * &struct uart_8250_port* @p. Namely, RTS is switched before/after
609 * transmission. The function is idempotent, so it is safe to call it
610 * multiple times.
611 *
612 * The caller MUST enable interrupt on empty shift register before
613 * calling serial8250_em485_init(). This interrupt is not a part of
614 * 8250 standard, but implementation defined.
615 *
616 * The function is supposed to be called from .rs485_config callback
617 * or from any other callback protected with p->port.lock spinlock.
618 *
619 * See also serial8250_em485_destroy()
620 *
621 * Return 0 - success, -errno - otherwise
622 */
623int serial8250_em485_init(struct uart_8250_port *p)
624{
625 if (p->em485)
626 return 0;
627
628 p->em485 = kmalloc(sizeof(struct uart_8250_em485), GFP_ATOMIC);
629 if (!p->em485)
630 return -ENOMEM;
631
632 hrtimer_init(&p->em485->stop_tx_timer, CLOCK_MONOTONIC,
633 HRTIMER_MODE_REL);
634 hrtimer_init(&p->em485->start_tx_timer, CLOCK_MONOTONIC,
635 HRTIMER_MODE_REL);
636 p->em485->stop_tx_timer.function = &serial8250_em485_handle_stop_tx;
637 p->em485->start_tx_timer.function = &serial8250_em485_handle_start_tx;
638 p->em485->port = p;
639 p->em485->active_timer = NULL;
640 serial8250_em485_rts_after_send(p);
641
642 return 0;
643}

706/*
707 * IER sleep support. UARTs which have EFRs need the "extended
708 * capability" bit enabled. Note that on XR16C850s, we need to
709 * reset LCR to write to IER.
710 */
711static void serial8250_set_sleep(struct uart_8250_port *p, int sleep)
712{
713 unsigned char lcr = 0, efr = 0;
714 /*
715 * Exar UARTs have a SLEEP register that enables or disables
716 * each UART to enter sleep mode separately. On the XR17V35x the
717 * register is accessible to each UART at the UART_EXAR_SLEEP
718 * offset but the UART channel may only write to the corresponding
719 * bit.
720 */
721 serial8250_rpm_get(p);
722 if ((p->port.type == PORT_XR17V35X) ||
723 (p->port.type == PORT_XR17D15X)) {
724 serial_out(p, UART_EXAR_SLEEP, sleep ? 0xff : 0);
725 goto out;
726 }
727
728 if (p->capabilities & UART_CAP_SLEEP) {
729 if (p->capabilities & UART_CAP_EFR) {
730 lcr = serial_in(p, UART_LCR);
731 efr = serial_in(p, UART_EFR);
732 serial_out(p, UART_LCR, UART_LCR_CONF_MODE_B);
733 serial_out(p, UART_EFR, UART_EFR_ECB);
734 serial_out(p, UART_LCR, 0);
735 }
736 serial_out(p, UART_IER, sleep ? UART_IERX_SLEEP : 0);
737 if (p->capabilities & UART_CAP_EFR) {
738 serial_out(p, UART_LCR, UART_LCR_CONF_MODE_B);
739 serial_out(p, UART_EFR, efr);
740 serial_out(p, UART_LCR, lcr);
741 }
742 }
743out:
744 serial8250_rpm_put(p);
745}

816/*
817 * This is a quickie test to see how big the FIFO is.
818 * It doesn't work at all the time, more's the pity.
819 */
820static int size_fifo(struct uart_8250_port *up)
821{
822 unsigned char old_fcr, old_mcr, old_lcr;
823 unsigned short old_dl;
824 int count;
825
826 old_lcr = serial_in(up, UART_LCR);
827 serial_out(up, UART_LCR, 0);
828 old_fcr = serial_in(up, UART_FCR);
829 old_mcr = serial8250_in_MCR(up);
830 serial_out(up, UART_FCR, UART_FCR_ENABLE_FIFO |
831 UART_FCR_CLEAR_RCVR | UART_FCR_CLEAR_XMIT);
832 serial8250_out_MCR(up, UART_MCR_LOOP);
833 serial_out(up, UART_LCR, UART_LCR_CONF_MODE_A);
834 old_dl = serial_dl_read(up);
835 serial_dl_write(up, 0x0001);
836 serial_out(up, UART_LCR, 0x03);
837 for (count = 0; count < 256; count++)
838 serial_out(up, UART_TX, count);
839 mdelay(20);/* FIXME - schedule_timeout */
840 for (count = 0; (serial_in(up, UART_LSR) & UART_LSR_DR) &&
841 (count < 256); count++)
842 serial_in(up, UART_RX);
843 serial_out(up, UART_FCR, old_fcr);
844 serial8250_out_MCR(up, old_mcr);
845 serial_out(up, UART_LCR, UART_LCR_CONF_MODE_A);
846 serial_dl_write(up, old_dl);
847 serial_out(up, UART_LCR, old_lcr);
848
849 return count;
850}

853 * Read UART ID using the divisor method - set DLL and DLM to zero
854 * and the revision will be in DLL and device type in DLM. We
855 * preserve the device state across this.

874/*
875 * This is a helper routine to autodetect StarTech/Exar/Oxsemi UART's.
876 * When this function is called we know it is at least a StarTech
877 * 16650 V2, but it might be one of several StarTech UARTs, or one of
878 * its clones. (We treat the broken original StarTech 16650 V1 as a
879 * 16550, and why not? Startech doesn't seem to even acknowledge its
880 * existence.)
881 *
882 * What evil have men's minds wrought...
883 */
884static void autoconfig_has_efr(struct uart_8250_port *up)
885{
886 unsigned int id1, id2, id3, rev;
887
888 /*
889 * Everything with an EFR has SLEEP
890 */
891 up->capabilities |= UART_CAP_EFR | UART_CAP_SLEEP;
892
893 /*
894 * First we check to see if it's an Oxford Semiconductor UART.
895 *
896 * If we have to do this here because some non-National
897 * Semiconductor clone chips lock up if you try writing to the
898 * LSR register (which serial_icr_read does)
899 */
900
901 /*
902 * Check for Oxford Semiconductor 16C950.
903 *
904 * EFR [4] must be set else this test fails.
905 *
906 * This shouldn't be necessary, but Mike Hudson (Exoray@isys.ca)
907 * claims that it's needed for 952 dual UART's (which are not
908 * recommended for new designs).
909 */
910 up->acr = 0;
911 serial_out(up, UART_LCR, UART_LCR_CONF_MODE_B);
912 serial_out(up, UART_EFR, UART_EFR_ECB);
913 serial_out(up, UART_LCR, 0x00);
914 id1 = serial_icr_read(up, UART_ID1);
915 id2 = serial_icr_read(up, UART_ID2);
916 id3 = serial_icr_read(up, UART_ID3);
917 rev = serial_icr_read(up, UART_REV);
918
919 DEBUG_AUTOCONF("950id=%02x:%02x:%02x:%02x ", id1, id2, id3, rev);
920
921 if (id1 == 0x16 && id2 == 0xC9 &&
922 (id3 == 0x50 || id3 == 0x52 || id3 == 0x54)) {
923 up->port.type = PORT_16C950;
924
925 /*
926 * Enable work around for the Oxford Semiconductor 952 rev B
927 * chip which causes it to seriously miscalculate baud rates
928 * when DLL is 0.
929 */
930 if (id3 == 0x52 && rev == 0x01)
931 up->bugs |= UART_BUG_QUOT;
932 return;
933 }
934
935 /*
936 * We check for a XR16C850 by setting DLL and DLM to 0, and then
937 * reading back DLL and DLM. The chip type depends on the DLM
938 * value read back:
939 * 0x10 - XR16C850 and the DLL contains the chip revision.
940 * 0x12 - XR16C2850.
941 * 0x14 - XR16C854.
942 */
943 id1 = autoconfig_read_divisor_id(up);
944 DEBUG_AUTOCONF("850id=%04x ", id1);
945
946 id2 = id1 >> 8;
947 if (id2 == 0x10 || id2 == 0x12 || id2 == 0x14) {
948 up->port.type = PORT_16850;
949 return;
950 }
951
952 /*
953 * It wasn't an XR16C850.
954 *
955 * We distinguish between the '654 and the '650 by counting
956 * how many bytes are in the FIFO. I'm using this for now,
957 * since that's the technique that was sent to me in the
958 * serial driver update, but I'm not convinced this works.
959 * I've had problems doing this in the past. -TYT
960 */
961 if (size_fifo(up) == 64)
962 up->port.type = PORT_16654;
963 else
964 up->port.type = PORT_16650V2;
965}

967/*
968 * We detected a chip without a FIFO. Only two fall into
969 * this category - the original 8250 and the 16450. The
970 * 16450 has a scratch register (accessible with LCR=0)
971 */
972static void autoconfig_8250(struct uart_8250_port *up)
973{
974 unsigned char scratch, status1, status2;
975
976 up->port.type = PORT_8250;
977
978 scratch = serial_in(up, UART_SCR);
979 serial_out(up, UART_SCR, 0xa5);
980 status1 = serial_in(up, UART_SCR);
981 serial_out(up, UART_SCR, 0x5a);
982 status2 = serial_in(up, UART_SCR);
983 serial_out(up, UART_SCR, scratch);

989static int broken_efr(struct uart_8250_port *up)

991 /*
992 * Exar ST16C2550 "A2" devices incorrectly detect as

994 * http://linux.derkeiler.com/Mailing-Lists/Kernel/2004-11/4812.html

1002/*
1003 * We know that the chip has FIFOs. Does it have an EFR? The
1004 * EFR is located in the same register position as the IIR and
1005 * we know the top two bits of the IIR are currently set. The
1006 * EFR should contain zero. Try to read the EFR.
1007 */
1008static void autoconfig_16550a(struct uart_8250_port *up)
1009{
1010 unsigned char status1, status2;
1011 unsigned int iersave;
1012
1013 up->port.type = PORT_16550A;
1014 up->capabilities |= UART_CAP_FIFO;
1015
1016 /*
1017 * XR17V35x UARTs have an extra divisor register, DLD
1018 * that gets enabled with when DLAB is set which will
1019 * cause the device to incorrectly match and assign
1020 * port type to PORT_16650. The EFR for this UART is
1021 * found at offset 0x09. Instead check the Deice ID (DVID)
1022 * register for a 2, 4 or 8 port UART.
1023 */
1024 if (up->port.flags & UPF_EXAR_EFR) {
1025 status1 = serial_in(up, UART_EXAR_DVID);
1026 if (status1 == 0x82 || status1 == 0x84 || status1 == 0x88) {
1027 DEBUG_AUTOCONF("Exar XR17V35x ");
1028 up->port.type = PORT_XR17V35X;
1029 up->capabilities |= UART_CAP_AFE | UART_CAP_EFR |
1030 UART_CAP_SLEEP;
1031
1032 return;
1033 }
1034
1035 }
1036
1037 /*
1038 * Check for presence of the EFR when DLAB is set.
1039 * Only ST16C650V1 UARTs pass this test.
1040 */
1041 serial_out(up, UART_LCR, UART_LCR_CONF_MODE_A);
1042 if (serial_in(up, UART_EFR) == 0) {
1043 serial_out(up, UART_EFR, 0xA8);
1044 if (serial_in(up, UART_EFR) != 0) {
1045 DEBUG_AUTOCONF("EFRv1 ");
1046 up->port.type = PORT_16650;
1047 up->capabilities |= UART_CAP_EFR | UART_CAP_SLEEP;
1048 } else {
1049 serial_out(up, UART_LCR, 0);
1050 serial_out(up, UART_FCR, UART_FCR_ENABLE_FIFO |
1051 UART_FCR7_64BYTE);
1052 status1 = serial_in(up, UART_IIR) >> 5;
1053 serial_out(up, UART_FCR, 0);
1054 serial_out(up, UART_LCR, 0);
1055
1056 if (status1 == 7)
1057 up->port.type = PORT_16550A_FSL64;
1058 else
1059 DEBUG_AUTOCONF("Motorola 8xxx DUART ");
1060 }
1061 serial_out(up, UART_EFR, 0);
1062 return;
1063 }
1064
1065 /*
1066 * Maybe it requires 0xbf to be written to the LCR.
1067 * (other ST16C650V2 UARTs, TI16C752A, etc)
1068 */
1069 serial_out(up, UART_LCR, UART_LCR_CONF_MODE_B);
1070 if (serial_in(up, UART_EFR) == 0 && !broken_efr(up)) {
1071 DEBUG_AUTOCONF("EFRv2 ");
1072 autoconfig_has_efr(up);
1073 return;
1074 }
1075
1076 /*
1077 * Check for a National Semiconductor SuperIO chip.
1078 * Attempt to switch to bank 2, read the value of the LOOP bit
1079 * from EXCR1. Switch back to bank 0, change it in MCR. Then
1080 * switch back to bank 2, read it from EXCR1 again and check
1081 * it's changed. If so, set baud_base in EXCR2 to 921600. -- dwmw2
1082 */
1083 serial_out(up, UART_LCR, 0);
1084 status1 = serial8250_in_MCR(up);
1085 serial_out(up, UART_LCR, 0xE0);
1086 status2 = serial_in(up, 0x02); /* EXCR1 */
1087
1088 if (!((status2 ^ status1) & UART_MCR_LOOP)) {
1089 serial_out(up, UART_LCR, 0);
1090 serial8250_out_MCR(up, status1 ^ UART_MCR_LOOP);
1091 serial_out(up, UART_LCR, 0xE0);
1092 status2 = serial_in(up, 0x02); /* EXCR1 */
1093 serial_out(up, UART_LCR, 0);
1094 serial8250_out_MCR(up, status1);
1095
1096 if ((status2 ^ status1) & UART_MCR_LOOP) {
1097 unsigned short quot;
1098
1099 serial_out(up, UART_LCR, 0xE0);
1100
1101 quot = serial_dl_read(up);
1102 quot <<= 3;
1103
1104 if (ns16550a_goto_highspeed(up))
1105 serial_dl_write(up, quot);
1106
1107 serial_out(up, UART_LCR, 0);
1108
1109 up->port.uartclk = 921600*16;
1110 up->port.type = PORT_NS16550A;
1111 up->capabilities |= UART_NATSEMI;
1112 return;
1113 }
1114 }
1115
1116 /*
1117 * No EFR. Try to detect a TI16750, which only sets bit 5 of
1118 * the IIR when 64 byte FIFO mode is enabled when DLAB is set.
1119 * Try setting it with and without DLAB set. Cheap clones
1120 * set bit 5 without DLAB set.
1121 */
1122 serial_out(up, UART_LCR, 0);
1123 serial_out(up, UART_FCR, UART_FCR_ENABLE_FIFO | UART_FCR7_64BYTE);
1124 status1 = serial_in(up, UART_IIR) >> 5;
1125 serial_out(up, UART_FCR, UART_FCR_ENABLE_FIFO);
1126 serial_out(up, UART_LCR, UART_LCR_CONF_MODE_A);
1127 serial_out(up, UART_FCR, UART_FCR_ENABLE_FIFO | UART_FCR7_64BYTE);
1128 status2 = serial_in(up, UART_IIR) >> 5;
1129 serial_out(up, UART_FCR, UART_FCR_ENABLE_FIFO);
1130 serial_out(up, UART_LCR, 0);
1131
1132 DEBUG_AUTOCONF("iir1=%d iir2=%d ", status1, status2);
1133
1134 if (status1 == 6 && status2 == 7) {
1135 up->port.type = PORT_16750;
1136 up->capabilities |= UART_CAP_AFE | UART_CAP_SLEEP;
1137 return;
1138 }
1139
1140 /*
1141 * Try writing and reading the UART_IER_UUE bit (b6).
1142 * If it works, this is probably one of the Xscale platform's
1143 * internal UARTs.
1144 * We're going to explicitly set the UUE bit to 0 before
1145 * trying to write and read a 1 just to make sure it's not
1146 * already a 1 and maybe locked there before we even start start.
1147 */
1148 iersave = serial_in(up, UART_IER);
1149 serial_out(up, UART_IER, iersave & ~UART_IER_UUE);
1150 if (!(serial_in(up, UART_IER) & UART_IER_UUE)) {
1151 /*
1152 * OK it's in a known zero state, try writing and reading
1153 * without disturbing the current state of the other bits.
1154 */
1155 serial_out(up, UART_IER, iersave | UART_IER_UUE);
1156 if (serial_in(up, UART_IER) & UART_IER_UUE) {
1157 /*
1158 * It's an Xscale.
1159 * We'll leave the UART_IER_UUE bit set to 1 (enabled).
1160 */
1161 DEBUG_AUTOCONF("Xscale ");
1162 up->port.type = PORT_XSCALE;
1163 up->capabilities |= UART_CAP_UUE | UART_CAP_RTOIE;
1164 return;
1165 }
1166 } else {
1167 /*
1168 * If we got here we couldn't force the IER_UUE bit to 0.
1169 * Log it and continue.
1170 */
1171 DEBUG_AUTOCONF("Couldn't force IER_UUE to 0 ");
1172 }
1173 serial_out(up, UART_IER, iersave);
1174
1175 /*
1176 * Exar uarts have EFR in a weird location
1177 */
1178 if (up->port.flags & UPF_EXAR_EFR) {
1179 DEBUG_AUTOCONF("Exar XR17D15x ");
1180 up->port.type = PORT_XR17D15X;
1181 up->capabilities |= UART_CAP_AFE | UART_CAP_EFR |
1182 UART_CAP_SLEEP;
1183
1184 return;
1185 }
1186
1187 /*
1188 * We distinguish between 16550A and U6 16550A by counting
1189 * how many bytes are in the FIFO.
1190 */
1191 if (up->port.type == PORT_16550A && size_fifo(up) == 64) {
1192 up->port.type = PORT_U6_16550A;
1193 up->capabilities |= UART_CAP_AFE;
1194 }
1195}

1197/*
1198 * This routine is called by rs_init() to initialize a specific serial
1199 * port. It determines what type of UART chip this serial port is
1200 * using: 8250, 16450, 16550, 16550A. The important question is
1201 * whether or not this UART is a 16550A or not, since this will
1202 * determine whether or not we can use its FIFO features or not.
1203 */
1204static void autoconfig(struct uart_8250_port *up)
1205{
1206 unsigned char status1, scratch, scratch2, scratch3;
1207 unsigned char save_lcr, save_mcr;
1208 struct uart_port *port = &up->port;
1209 unsigned long flags;
1210 unsigned int old_capabilities;
1211
1212 if (!port->iobase && !port->mapbase && !port->membase)
1213 return;
1214
1215 DEBUG_AUTOCONF("ttyS%d: autoconf (0x%04lx, 0x%p): ",
1216 serial_index(port), port->iobase, port->membase);
1217
1218 /*
1219 * We really do need global IRQs disabled here - we're going to
1220 * be frobbing the chips IRQ enable register to see if it exists.
1221 */
1222 spin_lock_irqsave(&port->lock, flags);
1223
1224 up->capabilities = 0;
1225 up->bugs = 0;
1226
1227 if (!(port->flags & UPF_BUGGY_UART)) {
1228 /*
1229 * Do a simple existence test first; if we fail this,
1230 * there's no point trying anything else.
1231 *
1232 * 0x80 is used as a nonsense port to prevent against
1233 * false positives due to ISA bus float. The
1234 * assumption is that 0x80 is a non-existent port;
1235 * which should be safe since include/asm/io.h also
1236 * makes this assumption.
1237 *
1238 * Note: this is safe as long as MCR bit 4 is clear
1239 * and the device is in "PC" mode.
1240 */
1241 scratch = serial_in(up, UART_IER);
1242 serial_out(up, UART_IER, 0);
1243#ifdef __i386__
1244 outb(0xff, 0x080);
1245#endif
1246 /*
1247 * Mask out IER[7:4] bits for test as some UARTs (e.g. TL
1248 * 16C754B) allow only to modify them if an EFR bit is set.
1249 */
1250 scratch2 = serial_in(up, UART_IER) & 0x0f;
1251 serial_out(up, UART_IER, 0x0F);
1252#ifdef __i386__
1253 outb(0, 0x080);
1254#endif
1255 scratch3 = serial_in(up, UART_IER) & 0x0f;
1256 serial_out(up, UART_IER, scratch);
1257 if (scratch2 != 0 || scratch3 != 0x0F) {
1258 /*
1259 * We failed; there's nothing here
1260 */
1261 spin_unlock_irqrestore(&port->lock, flags);
1262 DEBUG_AUTOCONF("IER test failed (%02x, %02x) ",
1263 scratch2, scratch3);
1264 goto out;
1265 }
1266 }
1267
1268 save_mcr = serial8250_in_MCR(up);
1269 save_lcr = serial_in(up, UART_LCR);
1270
1271 /*
1272 * Check to see if a UART is really there. Certain broken
1273 * internal modems based on the Rockwell chipset fail this
1274 * test, because they apparently don't implement the loopback
1275 * test mode. So this test is skipped on the COM 1 through
1276 * COM 4 ports. This *should* be safe, since no board
1277 * manufacturer would be stupid enough to design a board
1278 * that conflicts with COM 1-4 --- we hope!
1279 */
1280 if (!(port->flags & UPF_SKIP_TEST)) {
1281 serial8250_out_MCR(up, UART_MCR_LOOP | 0x0A);
1282 status1 = serial_in(up, UART_MSR) & 0xF0;
1283 serial8250_out_MCR(up, save_mcr);
1284 if (status1 != 0x90) {
1285 spin_unlock_irqrestore(&port->lock, flags);
1286 DEBUG_AUTOCONF("LOOP test failed (%02x) ",
1287 status1);
1288 goto out;
1289 }
1290 }
1291
1292 /*
1293 * We're pretty sure there's a port here. Lets find out what
1294 * type of port it is. The IIR top two bits allows us to find
1295 * out if it's 8250 or 16450, 16550, 16550A or later. This
1296 * determines what we test for next.
1297 *
1298 * We also initialise the EFR (if any) to zero for later. The
1299 * EFR occupies the same register location as the FCR and IIR.
1300 */
1301 serial_out(up, UART_LCR, UART_LCR_CONF_MODE_B);
1302 serial_out(up, UART_EFR, 0);
1303 serial_out(up, UART_LCR, 0);
1304
1305 serial_out(up, UART_FCR, UART_FCR_ENABLE_FIFO);
1306 scratch = serial_in(up, UART_IIR) >> 6;
1307
1308 switch (scratch) {
1309 case 0:
1310 autoconfig_8250(up);
1311 break;
1312 case 1:
1313 port->type = PORT_UNKNOWN;
1314 break;
1315 case 2:
1316 port->type = PORT_16550;
1317 break;
1318 case 3:
1319 autoconfig_16550a(up);
1320 break;
1321 }
1322
1323#ifdef CONFIG_SERIAL_8250_RSA
1324 /*
1325 * Only probe for RSA ports if we got the region.
1326 */
1327 if (port->type == PORT_16550A && up->probe & UART_PROBE_RSA &&
1328 __enable_rsa(up))
1329 port->type = PORT_RSA;
1330#endif
1331
1332 serial_out(up, UART_LCR, save_lcr);
1333
1334 port->fifosize = uart_config[up->port.type].fifo_size;
1335 old_capabilities = up->capabilities;
1336 up->capabilities = uart_config[port->type].flags;
1337 up->tx_loadsz = uart_config[port->type].tx_loadsz;
1338
1339 if (port->type == PORT_UNKNOWN)
1340 goto out_lock;
1341
1342 /*
1343 * Reset the UART.
1344 */
1345#ifdef CONFIG_SERIAL_8250_RSA
1346 if (port->type == PORT_RSA)
1347 serial_out(up, UART_RSA_FRR, 0);
1348#endif
1349 serial8250_out_MCR(up, save_mcr);
1350 serial8250_clear_fifos(up);
1351 serial_in(up, UART_RX);
1352 if (up->capabilities & UART_CAP_UUE)
1353 serial_out(up, UART_IER, UART_IER_UUE);
1354 else
1355 serial_out(up, UART_IER, 0);
1356
1357out_lock:
1358 spin_unlock_irqrestore(&port->lock, flags);
1359
1360 /*
1361 * Check if the device is a Fintek F81216A
1362 */
1363 if (port->type == PORT_16550A && port->iotype == UPIO_PORT)
1364 fintek_8250_probe(up);
1365
1366 if (up->capabilities != old_capabilities) {
1367 pr_warn("ttyS%d: detected caps %08x should be %08x\n",
1368 serial_index(port), old_capabilities,
1369 up->capabilities);
1370 }
1371out:
1372 DEBUG_AUTOCONF("iir=%d ", scratch);
1373 DEBUG_AUTOCONF("type=%s\n", uart_config[port->type].name);
1374}

1376static void autoconfig_irq(struct uart_8250_port *up)
1377{
1378 struct uart_port *port = &up->port;
1379 unsigned char save_mcr, save_ier;
1380 unsigned char save_ICP = 0;
1381 unsigned int ICP = 0;
1382 unsigned long irqs;
1383 int irq;
1384
1385 if (port->flags & UPF_FOURPORT) {
1386 ICP = (port->iobase & 0xfe0) | 0x1f;
1387 save_ICP = inb_p(ICP);
1388 outb_p(0x80, ICP);
1389 inb_p(ICP);
1390 }
1391
1392 if (uart_console(port))
1393 console_lock();
1394
1395 /* forget possible initially masked and pending IRQ */
1396 probe_irq_off(probe_irq_on());
1397 save_mcr = serial8250_in_MCR(up);
1398 save_ier = serial_in(up, UART_IER);
1399 serial8250_out_MCR(up, UART_MCR_OUT1 | UART_MCR_OUT2);
1400
1401 irqs = probe_irq_on();
1402 serial8250_out_MCR(up, 0);
1403 udelay(10);
1404 if (port->flags & UPF_FOURPORT) {
1405 serial8250_out_MCR(up, UART_MCR_DTR | UART_MCR_RTS);
1406 } else {
1407 serial8250_out_MCR(up,
1408 UART_MCR_DTR | UART_MCR_RTS | UART_MCR_OUT2);
1409 }
1410 serial_out(up, UART_IER, 0x0f); /* enable all intrs */
1411 serial_in(up, UART_LSR);
1412 serial_in(up, UART_RX);
1413 serial_in(up, UART_IIR);
1414 serial_in(up, UART_MSR);
1415 serial_out(up, UART_TX, 0xFF);
1416 udelay(20);
1417 irq = probe_irq_off(irqs);
1418
1419 serial8250_out_MCR(up, save_mcr);
1420 serial_out(up, UART_IER, save_ier);
1421
1422 if (port->flags & UPF_FOURPORT)
1423 outb_p(save_ICP, ICP);
1424
1425 if (uart_console(port))
1426 console_unlock();
1427
1428 port->irq = (irq > 0) ? irq : 0;
1429}

1515static inline void __stop_tx(struct uart_8250_port *p)
1516{
1517 struct uart_8250_em485 *em485 = p->em485;
1518
1519 if (em485) {
1520 unsigned char lsr = serial_in(p, UART_LSR);
1521 /*
1522 * To provide required timeing and allow FIFO transfer,
1523 * __stop_tx_rs485() must be called only when both FIFO and
1524 * shift register are empty. It is for device driver to enable
1525 * interrupt on TEMT.
1526 */
1527 if ((lsr & BOTH_EMPTY) != BOTH_EMPTY)
1528 return;
1529
1530 em485->active_timer = NULL;
1531 hrtimer_cancel(&em485->start_tx_timer);
1532
1533 __stop_tx_rs485(p);
1534 }
1535 __do_stop_tx(p);
1536}

1555static inline void __start_tx(struct uart_port *port)
1556{
1557 struct uart_8250_port *up = up_to_u8250p(port);
1558
1559 if (up->dma && !up->dma->tx_dma(up))
1560 return;
1561
1562 if (!(up->ier & UART_IER_THRI)) {
1563 up->ier |= UART_IER_THRI;
1564 serial_port_out(port, UART_IER, up->ier);
1565
1566 if (up->bugs & UART_BUG_TXEN) {
1567 unsigned char lsr;
1568
1569 lsr = serial_in(up, UART_LSR);
1570 up->lsr_saved_flags |= lsr & LSR_SAVE_FLAGS;
1571 if (lsr & UART_LSR_THRE)
1572 serial8250_tx_chars(up);
1573 }
1574 }
1575
1576 /*
1577 * Re-enable the transmitter if we disabled it.
1578 */
1579 if (port->type == PORT_16C950 && up->acr & UART_ACR_TXDIS) {
1580 up->acr &= ~UART_ACR_TXDIS;
1581 serial_icr_write(up, UART_ACR, up->acr);
1582 }
1583}

1586{
1587 struct uart_8250_port *up = up_to_u8250p(port);
1588 struct uart_8250_em485 *em485 = up->em485;
1589 unsigned char mcr;
1590
1591 if (!(up->port.rs485.flags & SER_RS485_RX_DURING_TX))
1592 serial8250_stop_rx(&up->port);
1593
1594 em485->active_timer = NULL;
1595 if (hrtimer_is_queued(&em485->stop_tx_timer))
1596 hrtimer_cancel(&em485->stop_tx_timer);
1597
1598 mcr = serial8250_in_MCR(up);
1599 if (!!(up->port.rs485.flags & SER_RS485_RTS_ON_SEND) !=
1600 !!(mcr & UART_MCR_RTS)) {
1601 if (up->port.rs485.flags & SER_RS485_RTS_ON_SEND)
1602 mcr |= UART_MCR_RTS;
1603 else
1604 mcr &= ~UART_MCR_RTS;
1605 serial8250_out_MCR(up, mcr);
1606
1607 if (up->port.rs485.delay_rts_before_send > 0) {
1608 em485->active_timer = &em485->start_tx_timer;
1609 start_hrtimer_ms(&em485->start_tx_timer,
1610 up->port.rs485.delay_rts_before_send);
1611 return;
1612 }
1613 }
1614
1615 __start_tx(port);
1616}

1690static void serial8250_read_char(struct uart_8250_port *up, unsigned char lsr)
1691{
1692 struct uart_port *port = &up->port;
1693 unsigned char ch;
1694 char flag = TTY_NORMAL;
1695
1696 if (likely(lsr & UART_LSR_DR))
1697 ch = serial_in(up, UART_RX);
1698 else
1699 /*
1700 * Intel 82571 has a Serial Over Lan device that will
1701 * set UART_LSR_BI without setting UART_LSR_DR when
1702 * it receives a break. To avoid reading from the
1703 * receive buffer without UART_LSR_DR bit set, we
1704 * just force the read character to be 0
1705 */
1706 ch = 0;
1707
1708 port->icount.rx++;
1709
1710 lsr |= up->lsr_saved_flags;
1711 up->lsr_saved_flags = 0;
1712
1713 if (unlikely(lsr & UART_LSR_BRK_ERROR_BITS)) {
1714 if (lsr & UART_LSR_BI) {
1715 lsr &= ~(UART_LSR_FE | UART_LSR_PE);
1716 port->icount.brk++;
1717 /*
1718 * We do the SysRQ and SAK checking
1719 * here because otherwise the break
1720 * may get masked by ignore_status_mask
1721 * or read_status_mask.
1722 */
1723 if (uart_handle_break(port))
1724 return;
1725 } else if (lsr & UART_LSR_PE)
1726 port->icount.parity++;
1727 else if (lsr & UART_LSR_FE)
1728 port->icount.frame++;
1729 if (lsr & UART_LSR_OE)
1730 port->icount.overrun++;
1731
1732 /*
1733 * Mask off conditions which should be ignored.
1734 */
1735 lsr &= port->read_status_mask;
1736
1737 if (lsr & UART_LSR_BI) {
1738 pr_debug("%s: handling break\n", __func__);
1739 flag = TTY_BREAK;
1740 } else if (lsr & UART_LSR_PE)
1741 flag = TTY_PARITY;
1742 else if (lsr & UART_LSR_FE)
1743 flag = TTY_FRAME;
1744 }
1745 if (uart_handle_sysrq_char(port, ch))
1746 return;
1747
1748 uart_insert_char(port, lsr, UART_LSR_OE, ch, flag);
1749}

1751/*
1752 * serial8250_rx_chars: processes according to the passed in LSR
1753 * value, and returns the remaining LSR bits not handled
1754 * by this Rx routine.
1755 */
1756unsigned char serial8250_rx_chars(struct uart_8250_port *up, unsigned char lsr)
1757{
1758 struct uart_port *port = &up->port;
1759 int max_count = 256;
1760
1761 do {
1762 serial8250_read_char(up, lsr);
1763 if (--max_count == 0)
1764 break;
1765 lsr = serial_in(up, UART_LSR);
1766 } while (lsr & (UART_LSR_DR | UART_LSR_BI));
1767
1768 tty_flip_buffer_push(&port->state->port);
1769 return lsr;
1770}
1771EXPORT_SYMBOL_GPL(serial8250_rx_chars);

1773void serial8250_tx_chars(struct uart_8250_port *up)
1774{
1775 struct uart_port *port = &up->port;
1776 struct circ_buf *xmit = &port->state->xmit;
1777 int count;
1778
1779 if (port->x_char) {
1780 serial_out(up, UART_TX, port->x_char);
1781 port->icount.tx++;
1782 port->x_char = 0;
1783 return;
1784 }
1785 if (uart_tx_stopped(port)) {
1786 serial8250_stop_tx(port);
1787 return;
1788 }
1789 if (uart_circ_empty(xmit)) {
1790 __stop_tx(up);
1791 return;
1792 }
1793
1794 count = up->tx_loadsz;
1795 do {
1796 serial_out(up, UART_TX, xmit->buf[xmit->tail]);
1797 xmit->tail = (xmit->tail + 1) & (UART_XMIT_SIZE - 1);
1798 port->icount.tx++;
1799 if (uart_circ_empty(xmit))
1800 break;
1801 if ((up->capabilities & UART_CAP_HFIFO) &&
1802 (serial_in(up, UART_LSR) & BOTH_EMPTY) != BOTH_EMPTY)
1803 break;
1804 /* The BCM2835 MINI UART THRE bit is really a not-full bit. */
1805 if ((up->capabilities & UART_CAP_MINI) &&
1806 !(serial_in(up, UART_LSR) & UART_LSR_THRE))
1807 break;
1808 } while (--count > 0);
1809
1810 if (uart_circ_chars_pending(xmit) < WAKEUP_CHARS)
1811 uart_write_wakeup(port);
1812
1813 /*
1814 * With RPM enabled, we have to wait until the FIFO is empty before the
1815 * HW can go idle. So we get here once again with empty FIFO and disable
1816 * the interrupt and RPM in __stop_tx()
1817 */
1818 if (uart_circ_empty(xmit) && !(up->capabilities & UART_CAP_RPM))
1819 __stop_tx(up);
1820}
1821EXPORT_SYMBOL_GPL(serial8250_tx_chars);

1823/* Caller holds uart port lock */
1824unsigned int serial8250_modem_status(struct uart_8250_port *up)
1825{
1826 struct uart_port *port = &up->port;
1827 unsigned int status = serial_in(up, UART_MSR);
1828
1829 status |= up->msr_saved_flags;
1830 up->msr_saved_flags = 0;
1831 if (status & UART_MSR_ANY_DELTA && up->ier & UART_IER_MSI &&
1832 port->state != NULL) {
1833 if (status & UART_MSR_TERI)
1834 port->icount.rng++;
1835 if (status & UART_MSR_DDSR)
1836 port->icount.dsr++;
1837 if (status & UART_MSR_DDCD)
1838 uart_handle_dcd_change(port, status & UART_MSR_DCD);
1839 if (status & UART_MSR_DCTS)
1840 uart_handle_cts_change(port, status & UART_MSR_CTS);
1841
1842 wake_up_interruptible(&port->state->port.delta_msr_wait);
1843 }
1844
1845 return status;
1846}

1861/*
1862 * This handles the interrupt from one port.
1863 */
1864int serial8250_handle_irq(struct uart_port *port, unsigned int iir)
1865{
1866 unsigned char status;
1867 unsigned long flags;
1868 struct uart_8250_port *up = up_to_u8250p(port);
1869
1870 if (iir & UART_IIR_NO_INT)
1871 return 0;
1872
1873 spin_lock_irqsave(&port->lock, flags);
1874
1875 status = serial_port_in(port, UART_LSR);
1876
1877 if (status & (UART_LSR_DR | UART_LSR_BI) &&
1878 iir & UART_IIR_RDI) {
1879 if (!up->dma || handle_rx_dma(up, iir))
1880 status = serial8250_rx_chars(up, status);
1881 }
1882 serial8250_modem_status(up);
1883 if ((!up->dma || up->dma->tx_err) && (status & UART_LSR_THRE))
1884 serial8250_tx_chars(up);
1885
1886 spin_unlock_irqrestore(&port->lock, flags);
1887 return 1;
1888}

2024/*
2025 * Wait for transmitter & holding register to empty
2026 */
2027static void wait_for_xmitr(struct uart_8250_port *up, int bits)
2028{
2029 unsigned int status, tmout = 10000;
2030
2031 /* Wait up to 10ms for the character(s) to be sent. */
2032 for (;;) {
2033 status = serial_in(up, UART_LSR);
2034
2035 up->lsr_saved_flags |= status & LSR_SAVE_FLAGS;
2036
2037 if ((status & bits) == bits)
2038 break;
2039 if (--tmout == 0)
2040 break;
2041 udelay(1);
2042 touch_nmi_watchdog();
2043 }
2044
2045 /* Wait up to 1s for flow control if necessary */
2046 if (up->port.flags & UPF_CONS_FLOW) {
2047 for (tmout = 1000000; tmout; tmout--) {
2048 unsigned int msr = serial_in(up, UART_MSR);
2049 up->msr_saved_flags |= msr & MSR_SAVE_FLAGS;
2050 if (msr & UART_MSR_CTS)
2051 break;
2052 udelay(1);
2053 touch_nmi_watchdog();
2054 }
2055 }
2056}

2119int serial8250_do_startup(struct uart_port *port)
2120{
2121 struct uart_8250_port *up = up_to_u8250p(port);
2122 unsigned long flags;
2123 unsigned char lsr, iir;
2124 int retval;
2125
2126 if (!port->fifosize)
2127 port->fifosize = uart_config[port->type].fifo_size;
2128 if (!up->tx_loadsz)
2129 up->tx_loadsz = uart_config[port->type].tx_loadsz;
2130 if (!up->capabilities)
2131 up->capabilities = uart_config[port->type].flags;
2132 up->mcr = 0;
2133
2134 if (port->iotype != up->cur_iotype)
2135 set_io_from_upio(port);
2136
2137 serial8250_rpm_get(up);
2138 if (port->type == PORT_16C950) {
2139 /* Wake up and initialize UART */
2140 up->acr = 0;
2141 serial_port_out(port, UART_LCR, UART_LCR_CONF_MODE_B);
2142 serial_port_out(port, UART_EFR, UART_EFR_ECB);
2143 serial_port_out(port, UART_IER, 0);
2144 serial_port_out(port, UART_LCR, 0);
2145 serial_icr_write(up, UART_CSR, 0); /* Reset the UART */
2146 serial_port_out(port, UART_LCR, UART_LCR_CONF_MODE_B);
2147 serial_port_out(port, UART_EFR, UART_EFR_ECB);
2148 serial_port_out(port, UART_LCR, 0);
2149 }
2150
2151 if (port->type == PORT_DA830) {
2152 /* Reset the port */
2153 serial_port_out(port, UART_IER, 0);
2154 serial_port_out(port, UART_DA830_PWREMU_MGMT, 0);
2155 mdelay(10);
2156
2157 /* Enable Tx, Rx and free run mode */
2158 serial_port_out(port, UART_DA830_PWREMU_MGMT,
2159 UART_DA830_PWREMU_MGMT_UTRST |
2160 UART_DA830_PWREMU_MGMT_URRST |
2161 UART_DA830_PWREMU_MGMT_FREE);
2162 }
2163
2164 if (port->type == PORT_NPCM) {
2165 /*
2166 * Nuvoton calls the scratch register 'UART_TOR' (timeout
2167 * register). Enable it, and set TIOC (timeout interrupt
2168 * comparator) to be 0x20 for correct operation.
2169 */
2170 serial_port_out(port, UART_NPCM_TOR, UART_NPCM_TOIE | 0x20);
2171 }
2172
2173#ifdef CONFIG_SERIAL_8250_RSA
2174 /*
2175 * If this is an RSA port, see if we can kick it up to the
2176 * higher speed clock.
2177 */
2178 enable_rsa(up);
2179#endif
2180
2181 if (port->type == PORT_XR17V35X) {
2182 /*
2183 * First enable access to IER [7:5], ISR [5:4], FCR [5:4],
2184 * MCR [7:5] and MSR [7:0]
2185 */
2186 serial_port_out(port, UART_XR_EFR, UART_EFR_ECB);
2187
2188 /*
2189 * Make sure all interrups are masked until initialization is
2190 * complete and the FIFOs are cleared
2191 */
2192 serial_port_out(port, UART_IER, 0);
2193 }
2194
2195 /*
2196 * Clear the FIFO buffers and disable them.
2197 * (they will be reenabled in set_termios())
2198 */
2199 serial8250_clear_fifos(up);
2200
2201 /*
2202 * Clear the interrupt registers.
2203 */
2204 serial_port_in(port, UART_LSR);
2205 serial_port_in(port, UART_RX);
2206 serial_port_in(port, UART_IIR);
2207 serial_port_in(port, UART_MSR);
2208 if ((port->type == PORT_XR17V35X) || (port->type == PORT_XR17D15X))
2209 serial_port_in(port, UART_EXAR_INT0);
2210
2211 /*
2212 * At this point, there's no way the LSR could still be 0xff;
2213 * if it is, then bail out, because there's likely no UART
2214 * here.
2215 */
2216 if (!(port->flags & UPF_BUGGY_UART) &&
2217 (serial_port_in(port, UART_LSR) == 0xff)) {
2218 printk_ratelimited(KERN_INFO "ttyS%d: LSR safety check engaged!\n",
2219 serial_index(port));
2220 retval = -ENODEV;
2221 goto out;
2222 }
2223
2224 /*
2225 * For a XR16C850, we need to set the trigger levels
2226 */
2227 if (port->type == PORT_16850) {
2228 unsigned char fctr;
2229
2230 serial_out(up, UART_LCR, UART_LCR_CONF_MODE_B);
2231
2232 fctr = serial_in(up, UART_FCTR) & ~(UART_FCTR_RX|UART_FCTR_TX);
2233 serial_port_out(port, UART_FCTR,
2234 fctr | UART_FCTR_TRGD | UART_FCTR_RX);
2235 serial_port_out(port, UART_TRG, UART_TRG_96);
2236 serial_port_out(port, UART_FCTR,
2237 fctr | UART_FCTR_TRGD | UART_FCTR_TX);
2238 serial_port_out(port, UART_TRG, UART_TRG_96);
2239
2240 serial_port_out(port, UART_LCR, 0);
2241 }
2242
2243 /*
2244 * For the Altera 16550 variants, set TX threshold trigger level.
2245 */
2246 if (((port->type == PORT_ALTR_16550_F32) ||
2247 (port->type == PORT_ALTR_16550_F64) ||
2248 (port->type == PORT_ALTR_16550_F128)) && (port->fifosize > 1)) {
2249 /* Bounds checking of TX threshold (valid 0 to fifosize-2) */
2250 if ((up->tx_loadsz < 2) || (up->tx_loadsz > port->fifosize)) {
2251 pr_err("ttyS%d TX FIFO Threshold errors, skipping\n",
2252 serial_index(port));
2253 } else {
2254 serial_port_out(port, UART_ALTR_AFR,
2255 UART_ALTR_EN_TXFIFO_LW);
2256 serial_port_out(port, UART_ALTR_TX_LOW,
2257 port->fifosize - up->tx_loadsz);
2258 port->handle_irq = serial8250_tx_threshold_handle_irq;
2259 }
2260 }
2261
2262 if (port->irq && !(up->port.flags & UPF_NO_THRE_TEST)) {
2263 unsigned char iir1;
2264 /*
2265 * Test for UARTs that do not reassert THRE when the
2266 * transmitter is idle and the interrupt has already
2267 * been cleared. Real 16550s should always reassert
2268 * this interrupt whenever the transmitter is idle and
2269 * the interrupt is enabled. Delays are necessary to
2270 * allow register changes to become visible.
2271 */
2272 spin_lock_irqsave(&port->lock, flags);
2273 if (up->port.irqflags & IRQF_SHARED)
2274 disable_irq_nosync(port->irq);
2275
2276 wait_for_xmitr(up, UART_LSR_THRE);
2277 serial_port_out_sync(port, UART_IER, UART_IER_THRI);
2278 udelay(1); /* allow THRE to set */
2279 iir1 = serial_port_in(port, UART_IIR);
2280 serial_port_out(port, UART_IER, 0);
2281 serial_port_out_sync(port, UART_IER, UART_IER_THRI);
2282 udelay(1); /* allow a working UART time to re-assert THRE */
2283 iir = serial_port_in(port, UART_IIR);
2284 serial_port_out(port, UART_IER, 0);
2285
2286 if (port->irqflags & IRQF_SHARED)
2287 enable_irq(port->irq);
2288 spin_unlock_irqrestore(&port->lock, flags);
2289
2290 /*
2291 * If the interrupt is not reasserted, or we otherwise
2292 * don't trust the iir, setup a timer to kick the UART
2293 * on a regular basis.
2294 */
2295 if ((!(iir1 & UART_IIR_NO_INT) && (iir & UART_IIR_NO_INT)) ||
2296 up->port.flags & UPF_BUG_THRE) {
2297 up->bugs |= UART_BUG_THRE;
2298 }
2299 }
2300
2301 retval = up->ops->setup_irq(up);
2302 if (retval)
2303 goto out;
2304
2305 /*
2306 * Now, initialize the UART
2307 */
2308 serial_port_out(port, UART_LCR, UART_LCR_WLEN8);
2309
2310 spin_lock_irqsave(&port->lock, flags);
2311 if (up->port.flags & UPF_FOURPORT) {
2312 if (!up->port.irq)
2313 up->port.mctrl |= TIOCM_OUT1;
2314 } else
2315 /*
2316 * Most PC uarts need OUT2 raised to enable interrupts.
2317 */
2318 if (port->irq)
2319 up->port.mctrl |= TIOCM_OUT2;
2320
2321 serial8250_set_mctrl(port, port->mctrl);
2322
2323 /*
2324 * Serial over Lan (SoL) hack:
2325 * Intel 8257x Gigabit ethernet chips have a 16550 emulation, to be
2326 * used for Serial Over Lan. Those chips take a longer time than a
2327 * normal serial device to signalize that a transmission data was
2328 * queued. Due to that, the above test generally fails. One solution
2329 * would be to delay the reading of iir. However, this is not
2330 * reliable, since the timeout is variable. So, let's just don't
2331 * test if we receive TX irq. This way, we'll never enable
2332 * UART_BUG_TXEN.
2333 */
2334 if (up->port.quirks & UPQ_NO_TXEN_TEST)
2335 goto dont_test_tx_en;
2336
2337 /*
2338 * Do a quick test to see if we receive an interrupt when we enable
2339 * the TX irq.
2340 */
2341 serial_port_out(port, UART_IER, UART_IER_THRI);
2342 lsr = serial_port_in(port, UART_LSR);
2343 iir = serial_port_in(port, UART_IIR);
2344 serial_port_out(port, UART_IER, 0);
2345
2346 if (lsr & UART_LSR_TEMT && iir & UART_IIR_NO_INT) {
2347 if (!(up->bugs & UART_BUG_TXEN)) {
2348 up->bugs |= UART_BUG_TXEN;
2349 pr_debug("ttyS%d - enabling bad tx status workarounds\n",
2350 serial_index(port));
2351 }
2352 } else {
2353 up->bugs &= ~UART_BUG_TXEN;
2354 }
2355
2356dont_test_tx_en:
2357 spin_unlock_irqrestore(&port->lock, flags);
2358
2359 /*
2360 * Clear the interrupt registers again for luck, and clear the
2361 * saved flags to avoid getting false values from polling
2362 * routines or the previous session.
2363 */
2364 serial_port_in(port, UART_LSR);
2365 serial_port_in(port, UART_RX);
2366 serial_port_in(port, UART_IIR);
2367 serial_port_in(port, UART_MSR);
2368 if ((port->type == PORT_XR17V35X) || (port->type == PORT_XR17D15X))
2369 serial_port_in(port, UART_EXAR_INT0);
2370 up->lsr_saved_flags = 0;
2371 up->msr_saved_flags = 0;
2372
2373 /*
2374 * Request DMA channels for both RX and TX.
2375 */
2376 if (up->dma) {
2377 retval = serial8250_request_dma(up);
2378 if (retval) {
2379 pr_warn_ratelimited("ttyS%d - failed to request DMA\n",
2380 serial_index(port));
2381 up->dma = NULL;
2382 }
2383 }
2384
2385 /*
2386 * Set the IER shadow for rx interrupts but defer actual interrupt
2387 * enable until after the FIFOs are enabled; otherwise, an already-
2388 * active sender can swamp the interrupt handler with "too much work".
2389 */
2390 up->ier = UART_IER_RLSI | UART_IER_RDI;
2391
2392 if (port->flags & UPF_FOURPORT) {
2393 unsigned int icp;
2394 /*
2395 * Enable interrupts on the AST Fourport board
2396 */
2397 icp = (port->iobase & 0xfe0) | 0x01f;
2398 outb_p(0x80, icp);
2399 inb_p(icp);
2400 }
2401 retval = 0;
2402out:
2403 serial8250_rpm_put(up);
2404 return retval;
2405}

2504static unsigned int serial8250_get_divisor(struct uart_8250_port *up,
2505 unsigned int baud,
2506 unsigned int *frac)
2507{
2508 struct uart_port *port = &up->port;
2509 unsigned int quot;
2510
2511 /*
2512 * Handle magic divisors for baud rates above baud_base on
2513 * SMSC SuperIO chips.
2514 *
2515 */
2516 if ((port->flags & UPF_MAGIC_MULTIPLIER) &&
2517 baud == (port->uartclk/4))
2518 quot = 0x8001;
2519 else if ((port->flags & UPF_MAGIC_MULTIPLIER) &&
2520 baud == (port->uartclk/8))
2521 quot = 0x8002;
2522 else if (up->port.type == PORT_XR17V35X)
2523 quot = xr17v35x_get_divisor(up, baud, frac);
2524 else if (up->port.type == PORT_NPCM)
2525 quot = npcm_get_divisor(up, baud);
2526 else
2527 quot = uart_get_divisor(port, baud);

2538static unsigned char serial8250_compute_lcr(struct uart_8250_port *up,
2539 tcflag_t c_cflag)
2540{
2541 unsigned char cval;
2542
2543 switch (c_cflag & CSIZE) {
2544 case CS5:
2545 cval = UART_LCR_WLEN5;
2546 break;
2547 case CS6:
2548 cval = UART_LCR_WLEN6;
2549 break;
2550 case CS7:
2551 cval = UART_LCR_WLEN7;
2552 break;
2553 default:
2554 case CS8:
2555 cval = UART_LCR_WLEN8;
2556 break;
2557 }
2558
2559 if (c_cflag & CSTOPB)
2560 cval |= UART_LCR_STOP;
2561 if (c_cflag & PARENB) {
2562 cval |= UART_LCR_PARITY;
2563 if (up->bugs & UART_BUG_PARITY)
2564 up->fifo_bug = true;
2565 }
2566 if (!(c_cflag & PARODD))
2567 cval |= UART_LCR_EPAR;
2568#ifdef CMSPAR
2569 if (c_cflag & CMSPAR)
2570 cval |= UART_LCR_SPAR;
2571#endif
2572
2573 return cval;
2574}

2576static void serial8250_set_divisor(struct uart_port *port, unsigned int baud,
2577 unsigned int quot, unsigned int quot_frac)
2578{
2579 struct uart_8250_port *up = up_to_u8250p(port);
2580
2581 /* Workaround to enable 115200 baud on OMAP1510 internal ports */
2582 if (is_omap1510_8250(up)) {
2583 if (baud == 115200) {
2584 quot = 1;
2585 serial_port_out(port, UART_OMAP_OSC_12M_SEL, 1);
2586 } else
2587 serial_port_out(port, UART_OMAP_OSC_12M_SEL, 0);
2588 }
2589
2590 /*
2591 * For NatSemi, switch to bank 2 not bank 1, to avoid resetting EXCR2,
2592 * otherwise just set DLAB
2593 */
2594 if (up->capabilities & UART_NATSEMI)
2595 serial_port_out(port, UART_LCR, 0xe0);
2596 else
2597 serial_port_out(port, UART_LCR, up->lcr | UART_LCR_DLAB);
2598
2599 serial_dl_write(up, quot);
2600
2601 /* XR17V35x UARTs have an extra fractional divisor register (DLD) */
2602 if (up->port.type == PORT_XR17V35X) {
2603 /* Preserve bits not related to baudrate; DLD[7:4]. */
2604 quot_frac |= serial_port_in(port, 0x2) & 0xf0;
2605 serial_port_out(port, 0x2, quot_frac);
2606 }
2607}

2624void
2625serial8250_do_set_termios(struct uart_port *port, struct ktermios *termios,
2626 struct ktermios *old)
2627{
2628 struct uart_8250_port *up = up_to_u8250p(port);
2629 unsigned char cval;
2630 unsigned long flags;
2631 unsigned int baud, quot, frac = 0;
2632
2633 if (up->capabilities & UART_CAP_MINI) {
2634 termios->c_cflag &= ~(CSTOPB | PARENB | PARODD | CMSPAR);
2635 if ((termios->c_cflag & CSIZE) == CS5 ||
2636 (termios->c_cflag & CSIZE) == CS6)
2637 termios->c_cflag = (termios->c_cflag & ~CSIZE) | CS7;
2638 }
2639 cval = serial8250_compute_lcr(up, termios->c_cflag);
2640
2641 baud = serial8250_get_baud_rate(port, termios, old);
2642 quot = serial8250_get_divisor(up, baud, &frac);
2643
2644 /*
2645 * Ok, we're now changing the port state. Do it with
2646 * interrupts disabled.
2647 */
2648 serial8250_rpm_get(up);
2649 spin_lock_irqsave(&port->lock, flags);
2650
2651 up->lcr = cval; /* Save computed LCR */
2652
2653 if (up->capabilities & UART_CAP_FIFO && port->fifosize > 1) {
2654 /* NOTE: If fifo_bug is not set, a user can set RX_trigger. */
2655 if ((baud < 2400 && !up->dma) || up->fifo_bug) {
2656 up->fcr &= ~UART_FCR_TRIGGER_MASK;
2657 up->fcr |= UART_FCR_TRIGGER_1;
2658 }
2659 }
2660
2661 /*
2662 * MCR-based auto flow control. When AFE is enabled, RTS will be
2663 * deasserted when the receive FIFO contains more characters than
2664 * the trigger, or the MCR RTS bit is cleared.
2665 */
2666 if (up->capabilities & UART_CAP_AFE) {
2667 up->mcr &= ~UART_MCR_AFE;
2668 if (termios->c_cflag & CRTSCTS)
2669 up->mcr |= UART_MCR_AFE;
2670 }
2671
2672 /*
2673 * Update the per-port timeout.
2674 */
2675 uart_update_timeout(port, termios->c_cflag, baud);
2676
2677 port->read_status_mask = UART_LSR_OE | UART_LSR_THRE | UART_LSR_DR;
2678 if (termios->c_iflag & INPCK)
2679 port->read_status_mask |= UART_LSR_FE | UART_LSR_PE;
2680 if (termios->c_iflag & (IGNBRK | BRKINT | PARMRK))
2681 port->read_status_mask |= UART_LSR_BI;
2682
2683 /*
2684 * Characteres to ignore
2685 */
2686 port->ignore_status_mask = 0;
2687 if (termios->c_iflag & IGNPAR)
2688 port->ignore_status_mask |= UART_LSR_PE | UART_LSR_FE;
2689 if (termios->c_iflag & IGNBRK) {
2690 port->ignore_status_mask |= UART_LSR_BI;
2691 /*
2692 * If we're ignoring parity and break indicators,
2693 * ignore overruns too (for real raw support).
2694 */
2695 if (termios->c_iflag & IGNPAR)
2696 port->ignore_status_mask |= UART_LSR_OE;
2697 }
2698
2699 /*
2700 * ignore all characters if CREAD is not set
2701 */
2702 if ((termios->c_cflag & CREAD) == 0)
2703 port->ignore_status_mask |= UART_LSR_DR;
2704
2705 /*
2706 * CTS flow control flag and modem status interrupts
2707 */
2708 up->ier &= ~UART_IER_MSI;
2709 if (!(up->bugs & UART_BUG_NOMSR) &&
2710 UART_ENABLE_MS(&up->port, termios->c_cflag))
2711 up->ier |= UART_IER_MSI;
2712 if (up->capabilities & UART_CAP_UUE)
2713 up->ier |= UART_IER_UUE;
2714 if (up->capabilities & UART_CAP_RTOIE)
2715 up->ier |= UART_IER_RTOIE;
2716
2717 serial_port_out(port, UART_IER, up->ier);
2718
2719 if (up->capabilities & UART_CAP_EFR) {
2720 unsigned char efr = 0;
2721 /*
2722 * TI16C752/Startech hardware flow control. FIXME:
2723 * - TI16C752 requires control thresholds to be set.
2724 * - UART_MCR_RTS is ineffective if auto-RTS mode is enabled.
2725 */
2726 if (termios->c_cflag & CRTSCTS)
2727 efr |= UART_EFR_CTS;
2728
2729 serial_port_out(port, UART_LCR, UART_LCR_CONF_MODE_B);
2730 if (port->flags & UPF_EXAR_EFR)
2731 serial_port_out(port, UART_XR_EFR, efr);
2732 else
2733 serial_port_out(port, UART_EFR, efr);
2734 }
2735
2736 serial8250_set_divisor(port, baud, quot, frac);
2737
2738 /*
2739 * LCR DLAB must be set to enable 64-byte FIFO mode. If the FCR
2740 * is written without DLAB set, this mode will be disabled.
2741 */
2742 if (port->type == PORT_16750)
2743 serial_port_out(port, UART_FCR, up->fcr);
2744
2745 serial_port_out(port, UART_LCR, up->lcr); /* reset DLAB */
2746 if (port->type != PORT_16750) {
2747 /* emulated UARTs (Lucent Venus 167x) need two steps */
2748 if (up->fcr & UART_FCR_ENABLE_FIFO)
2749 serial_port_out(port, UART_FCR, UART_FCR_ENABLE_FIFO);
2750 serial_port_out(port, UART_FCR, up->fcr); /* set fcr */
2751 }
2752 serial8250_set_mctrl(port, port->mctrl);
2753 spin_unlock_irqrestore(&port->lock, flags);
2754 serial8250_rpm_put(up);
2755
2756 /* Don't rewrite B0 */
2757 if (tty_termios_baud_rate(termios))
2758 tty_termios_encode_baud_rate(termios, baud, baud);
2759}

3211 * Print a string to the serial port trying not to disturb
3212 * any possible real use of the port...
3213 *
3214 * The console_lock must be held when we get here.
3215 */
3216void serial8250_console_write(struct uart_8250_port *up, const char *s,
3217 unsigned int count)
3218{
3219 struct uart_port *port = &up->port;
3220 unsigned long flags;
3221 unsigned int ier;
3222 int locked = 1;
3223
3224 touch_nmi_watchdog();
3225
3226 serial8250_rpm_get(up);
3227
3228 if (port->sysrq)
3229 locked = 0;
3230 else if (oops_in_progress)
3231 locked = spin_trylock_irqsave(&port->lock, flags);
3232 else
3233 spin_lock_irqsave(&port->lock, flags);
3234
3235 /*
3236 * First save the IER then disable the interrupts
3237 */
3238 ier = serial_port_in(port, UART_IER);
3239
3240 if (up->capabilities & UART_CAP_UUE)
3241 serial_port_out(port, UART_IER, UART_IER_UUE);
3242 else
3243 serial_port_out(port, UART_IER, 0);
3244
3245 /* check scratch reg to see if port powered off during system sleep */
3246 if (up->canary && (up->canary != serial_port_in(port, UART_SCR))) {
3247 serial8250_console_restore(up);
3248 up->canary = 0;
3249 }
3250
3251 uart_console_write(port, s, count, serial8250_console_putchar);
3252
3253 /*
3254 * Finally, wait for transmitter to become empty
3255 * and restore the IER
3256 */
3257 wait_for_xmitr(up, BOTH_EMPTY);
3258 serial_port_out(port, UART_IER, ier);
3259
3260 /*
3261 * The receive handling will happen properly because the
3262 * receive ready bit will still be set; it is not cleared
3263 * on read. However, modem control will not, we must
3264 * call it if we have saved something in the saved flags
3265 * while processing with interrupts off.
3266 */
3267 if (up->msr_saved_flags)
3268 serial8250_modem_status(up);
3269
3270 if (locked)
3271 spin_unlock_irqrestore(&port->lock, flags);
3272 serial8250_rpm_put(up);
3273}
