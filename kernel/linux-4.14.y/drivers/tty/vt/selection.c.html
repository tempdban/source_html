36/* Variables for selection control. */
37/* Use a dynamic buffer, instead of static (Dec 1994) */
38struct vc_data *sel_cons; /* must not be deallocated */
39static int use_unicode;
40static volatile int sel_start = -1; /* cleared by clear_selection */
41static int sel_end;
42static int sel_buffer_lth;
43static char *sel_buffer;

45/* clear_selection, highlight and highlight_pointer can be called
46 from interrupt (via scrollback/front) */
47
48/* set reverse video on characters s-e of console with selection. */
49static inline void highlight(const int s, const int e)
50{
51 invert_screen(sel_cons, s, e-s+2, 1);
52}
53
54/* use complementary color to show the pointer */
55static inline void highlight_pointer(const int where)
56{
57 complement_pos(sel_cons, where);
58}
59
60static u16
61sel_pos(int n)
62{
63 return inverse_translate(sel_cons, screen_glyph(sel_cons, n),
64 use_unicode);
65}

67/**
68 * clear_selection - remove current selection
69 *
70 * Remove the current selection highlight, if any from the console
71 * holding the selection. The caller must hold the console lock.
72 */
73void clear_selection(void)
74{
75 highlight_pointer(-1); /* hide the pointer */
76 if (sel_start != -1) {
77 highlight(sel_start, sel_end);
78 sel_start = -1;
79 }
80}

82/*
83 * User settable table: what characters are to be considered alphabetic?
84 * 128 bits. Locked by the console lock.
85 */
86static u32 inwordLut[]={
87 0x00000000, /* control chars */
88 0x03FFE000, /* digits and "-./" */
89 0x87FFFFFE, /* uppercase and '_' */
90 0x07FFFFFE, /* lowercase */
91};
92
93static inline int inword(const u16 c) {
94 return c > 0x7f || (( inwordLut[c>>5] >> (c & 0x1F) ) & 1);
95}
96
97/**
98 * set loadlut - load the LUT table
99 * @p: user table
100 *
101 * Load the LUT table from user space. The caller must hold the console
102 * lock. Make a temporary copy so a partial update doesn't make a mess.
103 */
104int sel_loadlut(char __user *p)
105{
106 u32 tmplut[ARRAY_SIZE(inwordLut)];
107 if (copy_from_user(tmplut, (u32 __user *)(p+4), sizeof(inwordLut)))
108 return -EFAULT;
109 memcpy(inwordLut, tmplut, sizeof(inwordLut));
110 return 0;
111}

113/* does screen address p correspond to character at LH/RH edge of screen? */
114static inline int atedge(const int p, int size_row)
115{
116 return (!(p % size_row) || !((p + 2) % size_row));
117}
118
119/* constrain v such that v <= u */
120static inline unsigned short limit(const unsigned short v, const unsigned short u)
121{
122 return (v > u) ? u : v;
123}
124
125/* stores the char in UTF8 and returns the number of bytes used (1-3) */
126static int store_utf8(u16 c, char *p)
127{
128 if (c < 0x80) {
129 /* 0******* */
130 p[0] = c;
131 return 1;
132 } else if (c < 0x800) {
133 /* 110***** 10****** */
134 p[0] = 0xc0 | (c >> 6);
135 p[1] = 0x80 | (c & 0x3f);
136 return 2;
137 } else {
138 /* 1110**** 10****** 10****** */
139 p[0] = 0xe0 | (c >> 12);
140 p[1] = 0x80 | ((c >> 6) & 0x3f);
141 p[2] = 0x80 | (c & 0x3f);
142 return 3;
143 }
144}

146/**
147 * set_selection - set the current selection.
148 * @sel: user selection info
149 * @tty: the console tty
150 *
151 * Invoked by the ioctl handle for the vt layer.
152 *
153 * The entire selection process is managed under the console_lock. It's
154 * a lot under the lock but its hardly a performance path
155 */
156int set_selection(const struct tiocl_selection __user *sel, struct tty_struct *tty)
157{
158 struct vc_data *vc = vc_cons[fg_console].d;
159 int sel_mode, new_sel_start, new_sel_end, spc;
160 char *bp, *obp;
161 int i, ps, pe, multiplier;
162 u16 c;
163 int mode;
164
165 poke_blanked_console();
166
167 { unsigned short xs, ys, xe, ye;
168
169 if (!access_ok(VERIFY_READ, sel, sizeof(*sel)))
170 return -EFAULT;
171 __get_user(xs, &sel->xs);
172 __get_user(ys, &sel->ys);
173 __get_user(xe, &sel->xe);
174 __get_user(ye, &sel->ye);
175 __get_user(sel_mode, &sel->sel_mode);
176 xs--; ys--; xe--; ye--;
177 xs = limit(xs, vc->vc_cols - 1);
178 ys = limit(ys, vc->vc_rows - 1);
179 xe = limit(xe, vc->vc_cols - 1);
180 ye = limit(ye, vc->vc_rows - 1);
181 ps = ys * vc->vc_size_row + (xs << 1);
182 pe = ye * vc->vc_size_row + (xe << 1);
183
184 if (sel_mode == TIOCL_SELCLEAR) {
185 /* useful for screendump without selection highlights */
186 clear_selection();
187 return 0;
188 }
189
190 if (mouse_reporting() && (sel_mode & TIOCL_SELMOUSEREPORT)) {
191 mouse_report(tty, sel_mode & TIOCL_SELBUTTONMASK, xs, ys);
192 return 0;
193 }
194 }
195
196 if (ps > pe) /* make sel_start <= sel_end */
197 {
198 int tmp = ps;
199 ps = pe;
200 pe = tmp;
201 }
202
203 if (sel_cons != vc_cons[fg_console].d) {
204 clear_selection();
205 sel_cons = vc_cons[fg_console].d;
206 }
207 mode = vt_do_kdgkbmode(fg_console);
208 if (mode == K_UNICODE)
209 use_unicode = 1;
210 else
211 use_unicode = 0;
212
213 switch (sel_mode)
214 {
215 case TIOCL_SELCHAR: /* character-by-character selection */
216 new_sel_start = ps;
217 new_sel_end = pe;
218 break;
219 case TIOCL_SELWORD: /* word-by-word selection */
220 spc = isspace(sel_pos(ps));
221 for (new_sel_start = ps; ; ps -= 2)
222 {
223 if ((spc && !isspace(sel_pos(ps))) ||
224 (!spc && !inword(sel_pos(ps))))
225 break;
226 new_sel_start = ps;
227 if (!(ps % vc->vc_size_row))
228 break;
229 }
230 spc = isspace(sel_pos(pe));
231 for (new_sel_end = pe; ; pe += 2)
232 {
233 if ((spc && !isspace(sel_pos(pe))) ||
234 (!spc && !inword(sel_pos(pe))))
235 break;
236 new_sel_end = pe;
237 if (!((pe + 2) % vc->vc_size_row))
238 break;
239 }
240 break;
241 case TIOCL_SELLINE: /* line-by-line selection */
242 new_sel_start = ps - ps % vc->vc_size_row;
243 new_sel_end = pe + vc->vc_size_row
244 - pe % vc->vc_size_row - 2;
245 break;
246 case TIOCL_SELPOINTER:
247 highlight_pointer(pe);
248 return 0;
249 default:
250 return -EINVAL;
251 }
252
253 /* remove the pointer */
254 highlight_pointer(-1);
255
256 /* select to end of line if on trailing space */
257 if (new_sel_end > new_sel_start &&
258 !atedge(new_sel_end, vc->vc_size_row) &&
259 isspace(sel_pos(new_sel_end))) {
260 for (pe = new_sel_end + 2; ; pe += 2)
261 if (!isspace(sel_pos(pe)) ||
262 atedge(pe, vc->vc_size_row))
263 break;
264 if (isspace(sel_pos(pe)))
265 new_sel_end = pe;
266 }
267 if (sel_start == -1) /* no current selection */
268 highlight(new_sel_start, new_sel_end);
269 else if (new_sel_start == sel_start)
270 {
271 if (new_sel_end == sel_end) /* no action required */
272 return 0;
273 else if (new_sel_end > sel_end) /* extend to right */
274 highlight(sel_end + 2, new_sel_end);
275 else /* contract from right */
276 highlight(new_sel_end + 2, sel_end);
277 }
278 else if (new_sel_end == sel_end)
279 {
280 if (new_sel_start < sel_start) /* extend to left */
281 highlight(new_sel_start, sel_start - 2);
282 else /* contract from left */
283 highlight(sel_start, new_sel_start - 2);
284 }
285 else /* some other case; start selection from scratch */
286 {
287 clear_selection();
288 highlight(new_sel_start, new_sel_end);
289 }
290 sel_start = new_sel_start;
291 sel_end = new_sel_end;
292
293 /* Allocate a new buffer before freeing the old one ... */
294 multiplier = use_unicode ? 3 : 1; /* chars can take up to 3 bytes */
295 bp = kmalloc(((sel_end-sel_start)/2+1)*multiplier, GFP_KERNEL);
296 if (!bp) {
297 printk(KERN_WARNING "selection: kmalloc() failed\n");
298 clear_selection();
299 return -ENOMEM;
300 }
301 kfree(sel_buffer);
302 sel_buffer = bp;
303
304 obp = bp;
305 for (i = sel_start; i <= sel_end; i += 2) {
306 c = sel_pos(i);
307 if (use_unicode)
308 bp += store_utf8(c, bp);
309 else
310 *bp++ = c;
311 if (!isspace(c))
312 obp = bp;
313 if (! ((i + 2) % vc->vc_size_row)) {
314 /* strip trailing blanks from line and add newline,
315 unless non-space at end of line. */
316 if (obp != bp) {
317 bp = obp;
318 *bp++ = '\r';
319 }
320 obp = bp;
321 }
322 }
323 sel_buffer_lth = bp - sel_buffer;
324 return 0;
325}

327/* Insert the contents of the selection buffer into the
328 * queue of the tty associated with the current console.
329 * Invoked by ioctl().
330 *
331 * Locking: called without locks. Calls the ldisc wrongly with
332 * unsafe methods,
333 */
334int paste_selection(struct tty_struct *tty)
335{
336 struct vc_data *vc = tty->driver_data;
337 int pasted = 0;
338 unsigned int count;
339 struct tty_ldisc *ld;
340 DECLARE_WAITQUEUE(wait, current);
341
342 console_lock();
343 poke_blanked_console();
344 console_unlock();
345
346 ld = tty_ldisc_ref_wait(tty);
347 if (!ld)
348 return -EIO; /* ldisc was hung up */
349 tty_buffer_lock_exclusive(&vc->port);
350
351 add_wait_queue(&vc->paste_wait, &wait);
352 while (sel_buffer && sel_buffer_lth > pasted) {
353 set_current_state(TASK_INTERRUPTIBLE);
354 if (tty_throttled(tty)) {
355 schedule();
356 continue;
357 }
358 __set_current_state(TASK_RUNNING);
359 count = sel_buffer_lth - pasted;
360 count = tty_ldisc_receive_buf(ld, sel_buffer + pasted, NULL,
361 count);
362 pasted += count;
363 }
364 remove_wait_queue(&vc->paste_wait, &wait);
365 __set_current_state(TASK_RUNNING);
366
367 tty_buffer_unlock_exclusive(&vc->port);
368 tty_ldisc_deref(ld);
369 return 0;
370}
