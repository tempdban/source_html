27static const struct hid_device_id hid_table[] = {
28 { HID_DEVICE(HID_BUS_ANY, HID_GROUP_GENERIC, HID_ANY_ID, HID_ANY_ID) },
29 { }
30};

33static struct hid_driver hid_generic = {
34 .name = "hid-generic",
35 .id_table = hid_table,
