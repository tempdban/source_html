177/* Dynamic HID quirks list - specified at runtime */
178struct quirks_list_struct {
179 struct hid_blacklist hid_bl_item;
180 struct list_head node;
181};
182
183static LIST_HEAD(dquirks_list);
184static DECLARE_RWSEM(dquirks_rwsem);
185
186/* Runtime ("dynamic") quirks manipulation functions */
187
188/**
189 * usbhid_exists_dquirk: find any dynamic quirks for a USB HID device
190 * @idVendor: the 16-bit USB vendor ID, in native byteorder
191 * @idProduct: the 16-bit USB product ID, in native byteorder
192 *
193 * Description:
194 * Scans dquirks_list for a matching dynamic quirk and returns
195 * the pointer to the relevant struct hid_blacklist if found.
196 * Must be called with a read lock held on dquirks_rwsem.
197 *
198 * Returns: NULL if no quirk found, struct hid_blacklist * if found.
199 */
200static struct hid_blacklist *usbhid_exists_dquirk(const u16 idVendor,
201 const u16 idProduct)
202{
203 struct quirks_list_struct *q;
204 struct hid_blacklist *bl_entry = NULL;
205
206 list_for_each_entry(q, &dquirks_list, node) {
207 if (q->hid_bl_item.idVendor == idVendor &&
208 q->hid_bl_item.idProduct == idProduct) {
209 bl_entry = &q->hid_bl_item;
210 break;
211 }
212 }
213
214 if (bl_entry != NULL)
215 dbg_hid("Found dynamic quirk 0x%x for USB HID vendor 0x%hx prod 0x%hx\n",
216 bl_entry->quirks, bl_entry->idVendor,
217 bl_entry->idProduct);
218
219 return bl_entry;
220}
221
222
223/**
224 * usbhid_modify_dquirk: add/replace a HID quirk
225 * @idVendor: the 16-bit USB vendor ID, in native byteorder
226 * @idProduct: the 16-bit USB product ID, in native byteorder
227 * @quirks: the u32 quirks value to add/replace
228 *
229 * Description:
230 * If an dynamic quirk exists in memory for this (idVendor,
231 * idProduct) pair, replace its quirks value with what was
232 * provided. Otherwise, add the quirk to the dynamic quirks list.
233 *
234 * Returns: 0 OK, -error on failure.
235 */
236static int usbhid_modify_dquirk(const u16 idVendor, const u16 idProduct,
237 const u32 quirks)
238{
239 struct quirks_list_struct *q_new, *q;
240 int list_edited = 0;
241
242 if (!idVendor) {
243 dbg_hid("Cannot add a quirk with idVendor = 0\n");
244 return -EINVAL;
245 }
246
247 q_new = kmalloc(sizeof(struct quirks_list_struct), GFP_KERNEL);
248 if (!q_new)
249 return -ENOMEM;
250
251 q_new->hid_bl_item.idVendor = idVendor;
252 q_new->hid_bl_item.idProduct = idProduct;
253 q_new->hid_bl_item.quirks = quirks;
254
255 down_write(&dquirks_rwsem);
256
257 list_for_each_entry(q, &dquirks_list, node) {
258
259 if (q->hid_bl_item.idVendor == idVendor &&
260 q->hid_bl_item.idProduct == idProduct) {
261
262 list_replace(&q->node, &q_new->node);
263 kfree(q);
264 list_edited = 1;
265 break;
266
267 }
268
269 }
270
271 if (!list_edited)
272 list_add_tail(&q_new->node, &dquirks_list);
273
274 up_write(&dquirks_rwsem);
275
276 return 0;
277}
278
279/**
280 * usbhid_remove_all_dquirks: remove all runtime HID quirks from memory
281 *
282 * Description:
283 * Free all memory associated with dynamic quirks - called before
284 * module unload.
285 *
286 */
287static void usbhid_remove_all_dquirks(void)
288{
289 struct quirks_list_struct *q, *temp;
290
291 down_write(&dquirks_rwsem);
292 list_for_each_entry_safe(q, temp, &dquirks_list, node) {
293 list_del(&q->node);
294 kfree(q);
295 }
296 up_write(&dquirks_rwsem);
297
298}

300/**
301 * usbhid_quirks_init: apply USB HID quirks specified at module load time
302 */
303int usbhid_quirks_init(char **quirks_param)
304{
305 u16 idVendor, idProduct;
306 u32 quirks;
307 int n = 0, m;
308
309 for (; n < MAX_USBHID_BOOT_QUIRKS && quirks_param[n]; n++) {
310
311 m = sscanf(quirks_param[n], "0x%hx:0x%hx:0x%x",
312 &idVendor, &idProduct, &quirks);
313
314 if (m != 3 ||
315 usbhid_modify_dquirk(idVendor, idProduct, quirks) != 0) {
316 pr_warn("Could not parse HID quirk module param %s\n",
317 quirks_param[n]);
318 }
319 }
320
321 return 0;
322}

338/**
339 * usbhid_exists_squirk: return any static quirks for a USB HID device
340 * @idVendor: the 16-bit USB vendor ID, in native byteorder
341 * @idProduct: the 16-bit USB product ID, in native byteorder
342 *
343 * Description:
344 * Given a USB vendor ID and product ID, return a pointer to
345 * the hid_blacklist entry associated with that device.
346 *
347 * Returns: pointer if quirk found, or NULL if no quirks found.
348 */
349static const struct hid_blacklist *usbhid_exists_squirk(const u16 idVendor,
350 const u16 idProduct)
351{
352 const struct hid_blacklist *bl_entry = NULL;
353 int n = 0;
354
355 for (; hid_blacklist[n].idVendor; n++)
356 if (hid_blacklist[n].idVendor == idVendor &&
357 (hid_blacklist[n].idProduct == (__u16) HID_ANY_ID ||
358 hid_blacklist[n].idProduct == idProduct))
359 bl_entry = &hid_blacklist[n];
360
361 if (bl_entry != NULL)
362 dbg_hid("Found squirk 0x%x for USB HID vendor 0x%hx prod 0x%hx\n",
363 bl_entry->quirks, bl_entry->idVendor,
364 bl_entry->idProduct);
365 return bl_entry;
366}

368/**
369 * usbhid_lookup_quirk: return any quirks associated with a USB HID device
370 * @idVendor: the 16-bit USB vendor ID, in native byteorder
371 * @idProduct: the 16-bit USB product ID, in native byteorder
372 *
373 * Description:
374 * Given a USB vendor ID and product ID, return any quirks associated
375 * with that device.
376 *
377 * Returns: a u32 quirks value.
378 */
379u32 usbhid_lookup_quirk(const u16 idVendor, const u16 idProduct)
380{
381 u32 quirks = 0;
382 const struct hid_blacklist *bl_entry = NULL;
383
384 /* NCR devices must not be queried for reports */
385 if (idVendor == USB_VENDOR_ID_NCR &&
386 idProduct >= USB_DEVICE_ID_NCR_FIRST &&
387 idProduct <= USB_DEVICE_ID_NCR_LAST)
388 return HID_QUIRK_NO_INIT_REPORTS;
389
390 down_read(&dquirks_rwsem);
391 bl_entry = usbhid_exists_dquirk(idVendor, idProduct);
392 if (!bl_entry)
393 bl_entry = usbhid_exists_squirk(idVendor, idProduct);
394 if (bl_entry)
395 quirks = bl_entry->quirks;
396 up_read(&dquirks_rwsem);
397
398 return quirks;
399}
