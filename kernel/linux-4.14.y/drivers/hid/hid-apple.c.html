169static const struct apple_key_translation *apple_find_translation(
170 const struct apple_key_translation *table, u16 from)
171{
172 const struct apple_key_translation *trans;
173
174 /* Look for the translation */
175 for (trans = table; trans->from; trans++)
176 if (trans->from == from)
177 return trans;
178
179 return NULL;
180}

182static int hidinput_apple_event(struct hid_device *hid, struct input_dev *input,
183 struct hid_usage *usage, __s32 value)
184{
185 struct apple_sc *asc = hid_get_drvdata(hid);
186 const struct apple_key_translation *trans, *table;
187
188 if (usage->code == KEY_FN) {
189 asc->fn_on = !!value;
190 input_event(input, usage->type, usage->code, value);
191 return 1;
192 }
193
194 if (fnmode) {
195 int do_translate;
196
197 if (hid->product >= USB_DEVICE_ID_APPLE_WELLSPRING4_ANSI &&
198 hid->product <= USB_DEVICE_ID_APPLE_WELLSPRING4A_JIS)
199 table = macbookair_fn_keys;
200 else if (hid->product < 0x21d || hid->product >= 0x300)
201 table = powerbook_fn_keys;
202 else
203 table = apple_fn_keys;
204
205 trans = apple_find_translation (table, usage->code);
206
207 if (trans) {
208 if (test_bit(usage->code, asc->pressed_fn))
209 do_translate = 1;
210 else if (trans->flags & APPLE_FLAG_FKEY)
211 do_translate = (fnmode == 2 && asc->fn_on) ||
212 (fnmode == 1 && !asc->fn_on);
213 else
214 do_translate = asc->fn_on;
215
216 if (do_translate) {
217 if (value)
218 set_bit(usage->code, asc->pressed_fn);
219 else
220 clear_bit(usage->code, asc->pressed_fn);
221
222 input_event(input, usage->type, trans->to,
223 value);
224
225 return 1;
226 }
227 }
228
229 if (asc->quirks & APPLE_NUMLOCK_EMULATION &&
230 (test_bit(usage->code, asc->pressed_numlock) ||
231 test_bit(LED_NUML, input->led))) {
232 trans = apple_find_translation(powerbook_numlock_keys,
233 usage->code);
234
235 if (trans) {
236 if (value)
237 set_bit(usage->code,
238 asc->pressed_numlock);
239 else
240 clear_bit(usage->code,
241 asc->pressed_numlock);
242
243 input_event(input, usage->type, trans->to,
244 value);
245 }
246
247 return 1;
248 }
249 }
250
251 if (iso_layout) {
252 if (hid->country == HID_COUNTRY_INTERNATIONAL_ISO) {
253 trans = apple_find_translation(apple_iso_keyboard, usage->code);
254 if (trans) {
255 input_event(input, usage->type, trans->to, value);
256 return 1;
257 }
258 }
259 }
260
261 if (swap_opt_cmd) {
262 trans = apple_find_translation(swapped_option_cmd_keys, usage->code);
263 if (trans) {
264 input_event(input, usage->type, trans->to, value);
265 return 1;
266 }
267 }
268
269 return 0;
270}

272static int apple_event(struct hid_device *hdev, struct hid_field *field,
273 struct hid_usage *usage, __s32 value)
274{
275 struct apple_sc *asc = hid_get_drvdata(hdev);
276
277 if (!(hdev->claimed & HID_CLAIMED_INPUT) || !field->hidinput ||
278 !usage->type)
279 return 0;
280
281 if ((asc->quirks & APPLE_INVERT_HWHEEL) &&
282 usage->code == REL_HWHEEL) {
283 input_event(field->hidinput->input, usage->type, usage->code,
284 -value);
285 return 1;
286 }
287
288 if ((asc->quirks & APPLE_HAS_FN) &&
289 hidinput_apple_event(hdev, field->hidinput->input,
290 usage, value))
291 return 1;
292
293
294 return 0;
295}

297/*
298 * MacBook JIS keyboard has wrong logical maximum
299 */
300static __u8 *apple_report_fixup(struct hid_device *hdev, __u8 *rdesc,
301 unsigned int *rsize)
302{
303 struct apple_sc *asc = hid_get_drvdata(hdev);
304
305 if ((asc->quirks & APPLE_RDESC_JIS) && *rsize >= 60 &&
306 rdesc[53] == 0x65 && rdesc[59] == 0x65) {
307 hid_info(hdev,
308 "fixing up MacBook JIS keyboard report descriptor\n");
309 rdesc[53] = rdesc[59] = 0xe7;
310 }
311 return rdesc;
312}

314static void apple_setup_input(struct input_dev *input)
315{
316 const struct apple_key_translation *trans;
317
318 set_bit(KEY_NUMLOCK, input->keybit);
319
320 /* Enable all needed keys */
321 for (trans = apple_fn_keys; trans->from; trans++)
322 set_bit(trans->to, input->keybit);
323
324 for (trans = powerbook_fn_keys; trans->from; trans++)
325 set_bit(trans->to, input->keybit);
326
327 for (trans = powerbook_numlock_keys; trans->from; trans++)
328 set_bit(trans->to, input->keybit);
329
330 for (trans = apple_iso_keyboard; trans->from; trans++)
331 set_bit(trans->to, input->keybit);
332}

334static int apple_input_mapping(struct hid_device *hdev, struct hid_input *hi,
335 struct hid_field *field, struct hid_usage *usage,
336 unsigned long **bit, int *max)
337{
338 if (usage->hid == (HID_UP_CUSTOM | 0x0003)) {
339 /* The fn key on Apple USB keyboards */
340 set_bit(EV_REP, hi->input->evbit);
341 hid_map_usage_clear(hi, usage, bit, max, EV_KEY, KEY_FN);
342 apple_setup_input(hi->input);
343 return 1;
344 }
345
346 /* we want the hid layer to go through standard path (set and ignore) */
347 return 0;
348}

350static int apple_input_mapped(struct hid_device *hdev, struct hid_input *hi,
351 struct hid_field *field, struct hid_usage *usage,
352 unsigned long **bit, int *max)
353{
354 struct apple_sc *asc = hid_get_drvdata(hdev);
355
356 if (asc->quirks & APPLE_MIGHTYMOUSE) {
357 if (usage->hid == HID_GD_Z)
358 hid_map_usage(hi, usage, bit, max, EV_REL, REL_HWHEEL);
359 else if (usage->code == BTN_1)
360 hid_map_usage(hi, usage, bit, max, EV_KEY, BTN_2);
361 else if (usage->code == BTN_2)
362 hid_map_usage(hi, usage, bit, max, EV_KEY, BTN_1);
363 }
364
365 return 0;
366}

368static int apple_probe(struct hid_device *hdev,
369 const struct hid_device_id *id)
370{
371 unsigned long quirks = id->driver_data;
372 struct apple_sc *asc;
373 unsigned int connect_mask = HID_CONNECT_DEFAULT;
374 int ret;
375
376 asc = devm_kzalloc(&hdev->dev, sizeof(*asc), GFP_KERNEL);
377 if (asc == NULL) {
378 hid_err(hdev, "can't alloc apple descriptor\n");
379 return -ENOMEM;
380 }
381
382 asc->quirks = quirks;
383
384 hid_set_drvdata(hdev, asc);
385
386 ret = hid_parse(hdev);
387 if (ret) {
388 hid_err(hdev, "parse failed\n");
389 return ret;
390 }
391
392 if (quirks & APPLE_HIDDEV)
393 connect_mask |= HID_CONNECT_HIDDEV_FORCE;
394 if (quirks & APPLE_IGNORE_HIDINPUT)
395 connect_mask &= ~HID_CONNECT_HIDINPUT;
396
397 ret = hid_hw_start(hdev, connect_mask);
398 if (ret) {
399 hid_err(hdev, "hw start failed\n");
400 return ret;
401 }
402
403 return 0;
404}
