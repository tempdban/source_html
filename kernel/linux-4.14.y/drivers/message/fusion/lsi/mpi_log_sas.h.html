19/****************************************************************************/
20/* IOC LOGINFO defines, 0x00000000 - 0x0FFFFFFF */
21/* Format: */
22/* Bits 31-28: MPI_IOCLOGINFO_TYPE_SAS (3) */
23/* Bits 27-24: IOC_LOGINFO_ORIGINATOR: 0=IOP, 1=PL, 2=IR */
24/* Bits 23-16: LOGINFO_CODE */
25/* Bits 15-0: LOGINFO_CODE Specific */
