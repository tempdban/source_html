268typedef union _MPT_FRAME_TRACKER {
269 struct {
270 struct list_head list;
271 u32 arg1;
272 u32 pad;
273 void *argp1;
274 } linkage;
275 /*
276 * NOTE: When request frames are free, on the linkage structure
277 * contets are valid. All other values are invalid.
278 * In particular, do NOT reply on offset [2]
279 * (in words) being the * message context.
280 * The message context must be reset (computed via base address
281 * + an offset) prior to issuing any command.
282 *
283 * NOTE2: On non-32-bit systems, where pointers are LARGE,
284 * using the linkage pointers destroys our sacred MsgContext
285 * field contents. But we don't care anymore because these
286 * are now reset in mpt_put_msg_frame() just prior to sending
287 * a request off to the IOC.
288 */
289 struct {
290 u32 __hdr[2];
291 /*
292 * The following _MUST_ match the location of the
293 * MsgContext field in the MPT message headers.
294 */
295 union {
296 u32 MsgContext;
297 struct {
298 u16 req_idx; /* Request index */
299 u8 cb_idx; /* callback function index */
300 u8 rsvd;
301 } fld;
302 } msgctxu;
303 } hwhdr;
304 /*
305 * Remark: 32 bit identifier:
306 * 31-24: reserved
307 * 23-16: call back index
308 * 15-0 : request index
309 */
310} MPT_FRAME_TRACKER;
311
312/*
313 * We might want to view/access a frame as:
314 * 1) generic request header
315 * 2) SCSIIORequest
316 * 3) SCSIIOReply
317 * 4) MPIDefaultReply
318 * 5) frame tracker
319 */
320typedef struct _MPT_FRAME_HDR {

322 MPIHeader_t hdr;
323 SCSIIORequest_t scsireq;
324 SCSIIOReply_t sreply;
325 ConfigReply_t configreply;
326 MPIDefaultReply_t reply;
327 MPT_FRAME_TRACKER frame;
328 } u;
329} MPT_FRAME_HDR;
