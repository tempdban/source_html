12 * Copyright 2010 Cisco Systems, Inc. All rights reserved.
13 * Author: Tom Lyon, pugs@cisco.com
14 *
15 * We arbitrarily define a Type1 IOMMU as one matching the below code.
16 * It could be called the x86 IOMMU as it's designed for AMD-Vi & Intel
17 * VT-d, but that makes it harder to re-use as theoretically anyone
18 * implementing a similar IOMMU could make use of this. We expect the
19 * IOMMU to support the IOMMU API and have few to no restrictions around
20 * the IOVA range that can be mapped. The Type1 IOMMU is currently
21 * optimized for relatively static mappings of a userspace process with
22 * userpsace pages pinned into memory. We also assume devices and IOMMU
23 * domains are PCI based as the IOMMU API is still centered around a
24 * device/bus interface rather than a group interface.
25 */

111 * This code handles mapping and unmapping of user data buffers
112 * into DMA'ble space using the IOMMU

114
115static struct vfio_dma *vfio_find_dma(struct vfio_iommu *iommu,
116 dma_addr_t start, size_t size)
117{
118 struct rb_node *node = iommu->dma_list.rb_node;
119
120 while (node) {
121 struct vfio_dma *dma = rb_entry(node, struct vfio_dma, node);
122
123 if (start + size <= dma->iova)
124 node = node->rb_left;
125 else if (start >= dma->iova + dma->size)
126 node = node->rb_right;
127 else
128 return dma;
129 }
130
131 return NULL;
132}
133
134static void vfio_link_dma(struct vfio_iommu *iommu, struct vfio_dma *new)
135{
136 struct rb_node **link = &iommu->dma_list.rb_node, *parent = NULL;
137 struct vfio_dma *dma;
138
139 while (*link) {
140 parent = *link;
141 dma = rb_entry(parent, struct vfio_dma, node);
142
143 if (new->iova + new->size <= dma->iova)
144 link = &(*link)->rb_left;
145 else
146 link = &(*link)->rb_right;
147 }
148
149 rb_link_node(&new->node, parent, link);
150 rb_insert_color(&new->node, &iommu->dma_list);
151}
152
153static void vfio_unlink_dma(struct vfio_iommu *iommu, struct vfio_dma *old)
154{
155 rb_erase(&old->node, &iommu->dma_list);
156}

158/*
159 * Helper Functions for host iova-pfn list
160 */
161static struct vfio_pfn *vfio_find_vpfn(struct vfio_dma *dma, dma_addr_t iova)
162{
163 struct vfio_pfn *vpfn;
164 struct rb_node *node = dma->pfn_list.rb_node;
165
166 while (node) {
167 vpfn = rb_entry(node, struct vfio_pfn, node);
168
169 if (iova < vpfn->iova)
170 node = node->rb_left;
171 else if (iova > vpfn->iova)
172 node = node->rb_right;
173 else
174 return vpfn;
175 }
176 return NULL;
177}

179static void vfio_link_pfn(struct vfio_dma *dma,
180 struct vfio_pfn *new)
181{
182 struct rb_node **link, *parent = NULL;
183 struct vfio_pfn *vpfn;
184
185 link = &dma->pfn_list.rb_node;
186 while (*link) {
187 parent = *link;
188 vpfn = rb_entry(parent, struct vfio_pfn, node);
189
190 if (new->iova < vpfn->iova)
191 link = &(*link)->rb_left;
192 else
193 link = &(*link)->rb_right;
194 }
195
196 rb_link_node(&new->node, parent, link);
197 rb_insert_color(&new->node, &dma->pfn_list);
198}

200static void vfio_unlink_pfn(struct vfio_dma *dma, struct vfio_pfn *old)
201{
202 rb_erase(&old->node, &dma->pfn_list);
203}

205static int vfio_add_to_pfn_list(struct vfio_dma *dma, dma_addr_t iova,
206 unsigned long pfn)
207{
208 struct vfio_pfn *vpfn;
209
210 vpfn = kzalloc(sizeof(*vpfn), GFP_KERNEL);
211 if (!vpfn)
212 return -ENOMEM;
213
214 vpfn->iova = iova;
215 vpfn->pfn = pfn;
216 atomic_set(&vpfn->ref_count, 1);
217 vfio_link_pfn(dma, vpfn);
218 return 0;
219}
220
221static void vfio_remove_from_pfn_list(struct vfio_dma *dma,
222 struct vfio_pfn *vpfn)
223{
224 vfio_unlink_pfn(dma, vpfn);
225 kfree(vpfn);
226}

228static struct vfio_pfn *vfio_iova_get_vfio_pfn(struct vfio_dma *dma,
229 unsigned long iova)
230{
231 struct vfio_pfn *vpfn = vfio_find_vpfn(dma, iova);
232
233 if (vpfn)
234 atomic_inc(&vpfn->ref_count);
235 return vpfn;
236}

238static int vfio_iova_put_vfio_pfn(struct vfio_dma *dma, struct vfio_pfn *vpfn)
239{
240 int ret = 0;
241
242 if (atomic_dec_and_test(&vpfn->ref_count)) {
243 ret = put_pfn(vpfn->pfn, dma->prot);
244 vfio_remove_from_pfn_list(dma, vpfn);
245 }
246 return ret;
247}

249static int vfio_lock_acct(struct task_struct *task, long npage, bool *lock_cap)
250{
251 struct mm_struct *mm;
252 bool is_current;
253 int ret;
254
255 if (!npage)
256 return 0;
257
258 is_current = (task->mm == current->mm);
259
260 mm = is_current ? task->mm : get_task_mm(task);
261 if (!mm)
262 return -ESRCH; /* process exited */
263
264 ret = down_write_killable(&mm->mmap_sem);
265 if (!ret) {
266 if (npage > 0) {
267 if (lock_cap ? !*lock_cap :
268 !has_capability(task, CAP_IPC_LOCK)) {
269 unsigned long limit;
270
271 limit = task_rlimit(task,
272 RLIMIT_MEMLOCK) >> PAGE_SHIFT;
273
274 if (mm->locked_vm + npage > limit)
275 ret = -ENOMEM;
276 }
277 }
278
279 if (!ret)
280 mm->locked_vm += npage;
281
282 up_write(&mm->mmap_sem);
283 }
284
285 if (!is_current)
286 mmput(mm);
287
288 return ret;
289}

291/*
292 * Some mappings aren't backed by a struct page, for example an mmap'd
293 * MMIO range for our own or another device. These use a different
294 * pfn conversion and shouldn't be tracked as locked pages.
295 */
296static bool is_invalid_reserved_pfn(unsigned long pfn)
297{
298 if (pfn_valid(pfn)) {
299 bool reserved;
300 struct page *tail = pfn_to_page(pfn);
301 struct page *head = compound_head(tail);
302 reserved = !!(PageReserved(head));
303 if (head != tail) {
304 /*
305 * "head" is not a dangling pointer
306 * (compound_head takes care of that)
307 * but the hugepage may have been split
308 * from under us (and we may not hold a
309 * reference count on the head page so it can
310 * be reused before we run PageReferenced), so
311 * we've to check PageTail before returning
312 * what we just read.
313 */
314 smp_rmb();
315 if (PageTail(tail))
316 return reserved;
317 }
318 return PageReserved(tail);
319 }
320
321 return true;
322}
323
324static int put_pfn(unsigned long pfn, int prot)
325{
326 if (!is_invalid_reserved_pfn(pfn)) {
327 struct page *page = pfn_to_page(pfn);
328 if (prot & IOMMU_WRITE)
329 SetPageDirty(page);
330 put_page(page);
331 return 1;
332 }
333 return 0;
334}

336static int vaddr_get_pfn(struct mm_struct *mm, unsigned long vaddr,
337 int prot, unsigned long *pfn)
338{
339 struct page *page[1];
340 struct vm_area_struct *vma;
341 struct vm_area_struct *vmas[1];
342 unsigned int flags = 0;
343 int ret;
344
345 if (prot & IOMMU_WRITE)
346 flags |= FOLL_WRITE;
347
348 down_read(&mm->mmap_sem);
349 if (mm == current->mm) {
350 ret = get_user_pages_longterm(vaddr, 1, flags, page, vmas);
351 } else {
352 ret = get_user_pages_remote(NULL, mm, vaddr, 1, flags, page,
353 vmas, NULL);
354 /*
355 * The lifetime of a vaddr_get_pfn() page pin is
356 * userspace-controlled. In the fs-dax case this could
357 * lead to indefinite stalls in filesystem operations.
358 * Disallow attempts to pin fs-dax pages via this
359 * interface.
360 */
361 if (ret > 0 && vma_is_fsdax(vmas[0])) {
362 ret = -EOPNOTSUPP;
363 put_page(page[0]);
364 }
365 }
366 up_read(&mm->mmap_sem);
367
368 if (ret == 1) {
369 *pfn = page_to_pfn(page[0]);
370 return 0;
371 }
372
373 down_read(&mm->mmap_sem);
374
375 vma = find_vma_intersection(mm, vaddr, vaddr + 1);
376
377 if (vma && vma->vm_flags & VM_PFNMAP) {
378 *pfn = ((vaddr - vma->vm_start) >> PAGE_SHIFT) + vma->vm_pgoff;
379 if (is_invalid_reserved_pfn(*pfn))
380 ret = 0;
381 }
382
383 up_read(&mm->mmap_sem);
384 return ret;
385}

387/*
388 * Attempt to pin pages. We really don't want to track all the pfns and
389 * the iommu can only map chunks of consecutive pfns anyway, so get the
390 * first page and all consecutive pages with the same locking.
391 */
392static long vfio_pin_pages_remote(struct vfio_dma *dma, unsigned long vaddr,
393 long npage, unsigned long *pfn_base,
394 bool lock_cap, unsigned long limit)
395{
396 unsigned long pfn = 0;
397 long ret, pinned = 0, lock_acct = 0;
398 bool rsvd;
399 dma_addr_t iova = vaddr - dma->vaddr + dma->iova;
400
401 /* This code path is only user initiated */
402 if (!current->mm)
403 return -ENODEV;
404
405 ret = vaddr_get_pfn(current->mm, vaddr, dma->prot, pfn_base);
406 if (ret)
407 return ret;
408
409 pinned++;
410 rsvd = is_invalid_reserved_pfn(*pfn_base);
411
412 /*
413 * Reserved pages aren't counted against the user, externally pinned
414 * pages are already counted against the user.
415 */
416 if (!rsvd && !vfio_find_vpfn(dma, iova)) {
417 if (!lock_cap && current->mm->locked_vm + 1 > limit) {
418 put_pfn(*pfn_base, dma->prot);
419 pr_warn("%s: RLIMIT_MEMLOCK (%ld) exceeded\n", __func__,
420 limit << PAGE_SHIFT);
421 return -ENOMEM;
422 }
423 lock_acct++;
424 }
425
426 if (unlikely(disable_hugepages))
427 goto out;
428
429 /* Lock all the consecutive pages from pfn_base */
430 for (vaddr += PAGE_SIZE, iova += PAGE_SIZE; pinned < npage;
431 pinned++, vaddr += PAGE_SIZE, iova += PAGE_SIZE) {
432 ret = vaddr_get_pfn(current->mm, vaddr, dma->prot, &pfn);
433 if (ret)
434 break;
435
436 if (pfn != *pfn_base + pinned ||
437 rsvd != is_invalid_reserved_pfn(pfn)) {
438 put_pfn(pfn, dma->prot);
439 break;
440 }
441
442 if (!rsvd && !vfio_find_vpfn(dma, iova)) {
443 if (!lock_cap &&
444 current->mm->locked_vm + lock_acct + 1 > limit) {
445 put_pfn(pfn, dma->prot);
446 pr_warn("%s: RLIMIT_MEMLOCK (%ld) exceeded\n",
447 __func__, limit << PAGE_SHIFT);
448 ret = -ENOMEM;
449 goto unpin_out;
450 }
451 lock_acct++;
452 }
453 }
454
455out:
456 ret = vfio_lock_acct(current, lock_acct, &lock_cap);
457
458unpin_out:
459 if (ret) {
460 if (!rsvd) {
461 for (pfn = *pfn_base ; pinned ; pfn++, pinned--)
462 put_pfn(pfn, dma->prot);
463 }
464
465 return ret;
466 }
467
468 return pinned;
469}

471static long vfio_unpin_pages_remote(struct vfio_dma *dma, dma_addr_t iova,
472 unsigned long pfn, long npage,
473 bool do_accounting)
474{
475 long unlocked = 0, locked = 0;
476 long i;
477
478 for (i = 0; i < npage; i++, iova += PAGE_SIZE) {
479 if (put_pfn(pfn++, dma->prot)) {
480 unlocked++;
481 if (vfio_find_vpfn(dma, iova))
482 locked++;
483 }
484 }
485
486 if (do_accounting)
487 vfio_lock_acct(dma->task, locked - unlocked, NULL);
488
489 return unlocked;
490}

492static int vfio_pin_page_external(struct vfio_dma *dma, unsigned long vaddr,
493 unsigned long *pfn_base, bool do_accounting)
494{
495 struct mm_struct *mm;
496 int ret;
497
498 mm = get_task_mm(dma->task);
499 if (!mm)
500 return -ENODEV;
501
502 ret = vaddr_get_pfn(mm, vaddr, dma->prot, pfn_base);
503 if (!ret && do_accounting && !is_invalid_reserved_pfn(*pfn_base)) {
504 ret = vfio_lock_acct(dma->task, 1, NULL);
505 if (ret) {
506 put_pfn(*pfn_base, dma->prot);
507 if (ret == -ENOMEM)
508 pr_warn("%s: Task %s (%d) RLIMIT_MEMLOCK "
509 "(%ld) exceeded\n", __func__,
510 dma->task->comm, task_pid_nr(dma->task),
511 task_rlimit(dma->task, RLIMIT_MEMLOCK));
512 }
513 }
514
515 mmput(mm);
516 return ret;
517}
518
519static int vfio_unpin_page_external(struct vfio_dma *dma, dma_addr_t iova,
520 bool do_accounting)
521{
522 int unlocked;
523 struct vfio_pfn *vpfn = vfio_find_vpfn(dma, iova);
524
525 if (!vpfn)
526 return 0;
527
528 unlocked = vfio_iova_put_vfio_pfn(dma, vpfn);
529
530 if (do_accounting)
531 vfio_lock_acct(dma->task, -unlocked, NULL);
532
533 return unlocked;
534}

536static int vfio_iommu_type1_pin_pages(void *iommu_data,
537 unsigned long *user_pfn,
538 int npage, int prot,
539 unsigned long *phys_pfn)
540{
541 struct vfio_iommu *iommu = iommu_data;
542 int i, j, ret;
543 unsigned long remote_vaddr;
544 struct vfio_dma *dma;
545 bool do_accounting;
546
547 if (!iommu || !user_pfn || !phys_pfn)
548 return -EINVAL;
549
550 /* Supported for v2 version only */
551 if (!iommu->v2)
552 return -EACCES;
553
554 mutex_lock(&iommu->lock);
555
556 /* Fail if notifier list is empty */
557 if ((!iommu->external_domain) || (!iommu->notifier.head)) {
558 ret = -EINVAL;
559 goto pin_done;
560 }
561
562 /*
563 * If iommu capable domain exist in the container then all pages are
564 * already pinned and accounted. Accouting should be done if there is no
565 * iommu capable domain in the container.
566 */
567 do_accounting = !IS_IOMMU_CAP_DOMAIN_IN_CONTAINER(iommu);
568
569 for (i = 0; i < npage; i++) {
570 dma_addr_t iova;
571 struct vfio_pfn *vpfn;
572
573 iova = user_pfn[i] << PAGE_SHIFT;
574 dma = vfio_find_dma(iommu, iova, PAGE_SIZE);
575 if (!dma) {
576 ret = -EINVAL;
577 goto pin_unwind;
578 }
579
580 if ((dma->prot & prot) != prot) {
581 ret = -EPERM;
582 goto pin_unwind;
583 }
584
585 vpfn = vfio_iova_get_vfio_pfn(dma, iova);
586 if (vpfn) {
587 phys_pfn[i] = vpfn->pfn;
588 continue;
589 }
590
591 remote_vaddr = dma->vaddr + iova - dma->iova;
592 ret = vfio_pin_page_external(dma, remote_vaddr, &phys_pfn[i],
593 do_accounting);
594 if (ret)
595 goto pin_unwind;
596
597 ret = vfio_add_to_pfn_list(dma, iova, phys_pfn[i]);
598 if (ret) {
599 vfio_unpin_page_external(dma, iova, do_accounting);
600 goto pin_unwind;
601 }
602 }
603
604 ret = i;
605 goto pin_done;
606
607pin_unwind:
608 phys_pfn[i] = 0;
609 for (j = 0; j < i; j++) {
610 dma_addr_t iova;
611
612 iova = user_pfn[j] << PAGE_SHIFT;
613 dma = vfio_find_dma(iommu, iova, PAGE_SIZE);
614 vfio_unpin_page_external(dma, iova, do_accounting);
615 phys_pfn[j] = 0;
616 }
617pin_done:
618 mutex_unlock(&iommu->lock);
619 return ret;
620}

622static int vfio_iommu_type1_unpin_pages(void *iommu_data,
623 unsigned long *user_pfn,
624 int npage)
625{
626 struct vfio_iommu *iommu = iommu_data;
627 bool do_accounting;
628 int i;
629
630 if (!iommu || !user_pfn)
631 return -EINVAL;
632
633 /* Supported for v2 version only */
634 if (!iommu->v2)
635 return -EACCES;
636
637 mutex_lock(&iommu->lock);
638
639 if (!iommu->external_domain) {
640 mutex_unlock(&iommu->lock);
641 return -EINVAL;
642 }
643
644 do_accounting = !IS_IOMMU_CAP_DOMAIN_IN_CONTAINER(iommu);
645 for (i = 0; i < npage; i++) {
646 struct vfio_dma *dma;
647 dma_addr_t iova;
648
649 iova = user_pfn[i] << PAGE_SHIFT;
650 dma = vfio_find_dma(iommu, iova, PAGE_SIZE);
651 if (!dma)
652 goto unpin_exit;
653 vfio_unpin_page_external(dma, iova, do_accounting);
654 }
655
656unpin_exit:
657 mutex_unlock(&iommu->lock);
658 return i > npage ? npage : (i > 0 ? i : -EINVAL);
659}

661static long vfio_unmap_unpin(struct vfio_iommu *iommu, struct vfio_dma *dma,
662 bool do_accounting)
663{
664 dma_addr_t iova = dma->iova, end = dma->iova + dma->size;
665 struct vfio_domain *domain, *d;
666 long unlocked = 0;
667
668 if (!dma->size)
669 return 0;
670
671 if (!IS_IOMMU_CAP_DOMAIN_IN_CONTAINER(iommu))
672 return 0;
673
674 /*
675 * We use the IOMMU to track the physical addresses, otherwise we'd
676 * need a much more complicated tracking system. Unfortunately that
677 * means we need to use one of the iommu domains to figure out the
678 * pfns to unpin. The rest need to be unmapped in advance so we have
679 * no iommu translations remaining when the pages are unpinned.
680 */
681 domain = d = list_first_entry(&iommu->domain_list,
682 struct vfio_domain, next);
683
684 list_for_each_entry_continue(d, &iommu->domain_list, next) {
685 iommu_unmap(d->domain, dma->iova, dma->size);
686 cond_resched();
687 }
688
689 while (iova < end) {
690 size_t unmapped, len;
691 phys_addr_t phys, next;
692
693 phys = iommu_iova_to_phys(domain->domain, iova);
694 if (WARN_ON(!phys)) {
695 iova += PAGE_SIZE;
696 continue;
697 }
698
699 /*
700 * To optimize for fewer iommu_unmap() calls, each of which
701 * may require hardware cache flushing, try to find the
702 * largest contiguous physical memory chunk to unmap.
703 */
704 for (len = PAGE_SIZE;
705 !domain->fgsp && iova + len < end; len += PAGE_SIZE) {
706 next = iommu_iova_to_phys(domain->domain, iova + len);
707 if (next != phys + len)
708 break;
709 }
710
711 unmapped = iommu_unmap(domain->domain, iova, len);
712 if (WARN_ON(!unmapped))
713 break;
714
715 unlocked += vfio_unpin_pages_remote(dma, iova,
716 phys >> PAGE_SHIFT,
717 unmapped >> PAGE_SHIFT,
718 false);
719 iova += unmapped;
720
721 cond_resched();
722 }
723
724 dma->iommu_mapped = false;
725 if (do_accounting) {
726 vfio_lock_acct(dma->task, -unlocked, NULL);
727 return 0;
728 }
729 return unlocked;
730}

732static void vfio_remove_dma(struct vfio_iommu *iommu, struct vfio_dma *dma)
733{
734 vfio_unmap_unpin(iommu, dma, true);
735 vfio_unlink_dma(iommu, dma);
736 put_task_struct(dma->task);
737 kfree(dma);
738}

740static unsigned long vfio_pgsize_bitmap(struct vfio_iommu *iommu)
741{
742 struct vfio_domain *domain;
743 unsigned long bitmap = ULONG_MAX;
744
745 mutex_lock(&iommu->lock);
746 list_for_each_entry(domain, &iommu->domain_list, next)
747 bitmap &= domain->domain->pgsize_bitmap;
748 mutex_unlock(&iommu->lock);
749
750 /*
751 * In case the IOMMU supports page sizes smaller than PAGE_SIZE
752 * we pretend PAGE_SIZE is supported and hide sub-PAGE_SIZE sizes.
753 * That way the user will be able to map/unmap buffers whose size/
754 * start address is aligned with PAGE_SIZE. Pinning code uses that
755 * granularity while iommu driver can use the sub-PAGE_SIZE size
756 * to map the buffer.
757 */
758 if (bitmap & ~PAGE_MASK) {
759 bitmap &= PAGE_MASK;
760 bitmap |= PAGE_SIZE;
761 }
762
763 return bitmap;
764}
765
766static int vfio_dma_do_unmap(struct vfio_iommu *iommu,
767 struct vfio_iommu_type1_dma_unmap *unmap)
768{
769 uint64_t mask;
770 struct vfio_dma *dma, *dma_last = NULL;
771 size_t unmapped = 0;
772 int ret = 0, retries = 0;
773
774 mask = ((uint64_t)1 << __ffs(vfio_pgsize_bitmap(iommu))) - 1;
775
776 if (unmap->iova & mask)
777 return -EINVAL;
778 if (!unmap->size || unmap->size & mask)
779 return -EINVAL;
780
781 WARN_ON(mask & PAGE_MASK);
782again:
783 mutex_lock(&iommu->lock);
784
785 /*
786 * vfio-iommu-type1 (v1) - User mappings were coalesced together to
787 * avoid tracking individual mappings. This means that the granularity
788 * of the original mapping was lost and the user was allowed to attempt
789 * to unmap any range. Depending on the contiguousness of physical
790 * memory and page sizes supported by the IOMMU, arbitrary unmaps may
791 * or may not have worked. We only guaranteed unmap granularity
792 * matching the original mapping; even though it was untracked here,
793 * the original mappings are reflected in IOMMU mappings. This
794 * resulted in a couple unusual behaviors. First, if a range is not
795 * able to be unmapped, ex. a set of 4k pages that was mapped as a
796 * 2M hugepage into the IOMMU, the unmap ioctl returns success but with
797 * a zero sized unmap. Also, if an unmap request overlaps the first
798 * address of a hugepage, the IOMMU will unmap the entire hugepage.
799 * This also returns success and the returned unmap size reflects the
800 * actual size unmapped.
801 *
802 * We attempt to maintain compatibility with this "v1" interface, but
803 * we take control out of the hands of the IOMMU. Therefore, an unmap
804 * request offset from the beginning of the original mapping will
805 * return success with zero sized unmap. And an unmap request covering
806 * the first iova of mapping will unmap the entire range.
807 *
808 * The v2 version of this interface intends to be more deterministic.
809 * Unmap requests must fully cover previous mappings. Multiple
810 * mappings may still be unmaped by specifying large ranges, but there
811 * must not be any previous mappings bisected by the range. An error
812 * will be returned if these conditions are not met. The v2 interface
813 * will only return success and a size of zero if there were no
814 * mappings within the range.
815 */
816 if (iommu->v2) {
817 dma = vfio_find_dma(iommu, unmap->iova, 1);
818 if (dma && dma->iova != unmap->iova) {
819 ret = -EINVAL;
820 goto unlock;
821 }
822 dma = vfio_find_dma(iommu, unmap->iova + unmap->size - 1, 0);
823 if (dma && dma->iova + dma->size != unmap->iova + unmap->size) {
824 ret = -EINVAL;
825 goto unlock;
826 }
827 }
828
829 while ((dma = vfio_find_dma(iommu, unmap->iova, unmap->size))) {
830 if (!iommu->v2 && unmap->iova > dma->iova)
831 break;
832 /*
833 * Task with same address space who mapped this iova range is
834 * allowed to unmap the iova range.
835 */
836 if (dma->task->mm != current->mm)
837 break;
838
839 if (!RB_EMPTY_ROOT(&dma->pfn_list)) {
840 struct vfio_iommu_type1_dma_unmap nb_unmap;
841
842 if (dma_last == dma) {
843 BUG_ON(++retries > 10);
844 } else {
845 dma_last = dma;
846 retries = 0;
847 }
848
849 nb_unmap.iova = dma->iova;
850 nb_unmap.size = dma->size;
851
852 /*
853 * Notify anyone (mdev vendor drivers) to invalidate and
854 * unmap iovas within the range we're about to unmap.
855 * Vendor drivers MUST unpin pages in response to an
856 * invalidation.
857 */
858 mutex_unlock(&iommu->lock);
859 blocking_notifier_call_chain(&iommu->notifier,
860 VFIO_IOMMU_NOTIFY_DMA_UNMAP,
861 &nb_unmap);
862 goto again;
863 }
864 unmapped += dma->size;
865 vfio_remove_dma(iommu, dma);
866 }
867
868unlock:
869 mutex_unlock(&iommu->lock);
870
871 /* Report how much was unmapped */
872 unmap->size = unmapped;
873
874 return ret;
875}
876
877/*
878 * Turns out AMD IOMMU has a page table bug where it won't map large pages
879 * to a region that previously mapped smaller pages. This should be fixed
880 * soon, so this is just a temporary workaround to break mappings down into
881 * PAGE_SIZE. Better to map smaller pages than nothing.
882 */
883static int map_try_harder(struct vfio_domain *domain, dma_addr_t iova,
884 unsigned long pfn, long npage, int prot)
885{
886 long i;
887 int ret = 0;
888
889 for (i = 0; i < npage; i++, pfn++, iova += PAGE_SIZE) {
890 ret = iommu_map(domain->domain, iova,
891 (phys_addr_t)pfn << PAGE_SHIFT,
892 PAGE_SIZE, prot | domain->prot);
893 if (ret)
894 break;
895 }
896
897 for (; i < npage && i > 0; i--, iova -= PAGE_SIZE)
898 iommu_unmap(domain->domain, iova, PAGE_SIZE);
899
900 return ret;
901}

903static int vfio_iommu_map(struct vfio_iommu *iommu, dma_addr_t iova,
904 unsigned long pfn, long npage, int prot)
905{
906 struct vfio_domain *d;
907 int ret;
908
909 list_for_each_entry(d, &iommu->domain_list, next) {
910 ret = iommu_map(d->domain, iova, (phys_addr_t)pfn << PAGE_SHIFT,
911 npage << PAGE_SHIFT, prot | d->prot);
912 if (ret) {
913 if (ret != -EBUSY ||
914 map_try_harder(d, iova, pfn, npage, prot))
915 goto unwind;
916 }
917
918 cond_resched();
919 }
920
921 return 0;
922
923unwind:
924 list_for_each_entry_continue_reverse(d, &iommu->domain_list, next)
925 iommu_unmap(d->domain, iova, npage << PAGE_SHIFT);
926
927 return ret;
928}

930static int vfio_pin_map_dma(struct vfio_iommu *iommu, struct vfio_dma *dma,
931 size_t map_size)
932{
933 dma_addr_t iova = dma->iova;
934 unsigned long vaddr = dma->vaddr;
935 size_t size = map_size;
936 long npage;
937 unsigned long pfn, limit = rlimit(RLIMIT_MEMLOCK) >> PAGE_SHIFT;
938 bool lock_cap = capable(CAP_IPC_LOCK);
939 int ret = 0;
940
941 while (size) {
942 /* Pin a contiguous chunk of memory */
943 npage = vfio_pin_pages_remote(dma, vaddr + dma->size,
944 size >> PAGE_SHIFT, &pfn,
945 lock_cap, limit);
946 if (npage <= 0) {
947 WARN_ON(!npage);
948 ret = (int)npage;
949 break;
950 }
951
952 /* Map it! */
953 ret = vfio_iommu_map(iommu, iova + dma->size, pfn, npage,
954 dma->prot);
955 if (ret) {
956 vfio_unpin_pages_remote(dma, iova + dma->size, pfn,
957 npage, true);
958 break;
959 }
960
961 size -= npage << PAGE_SHIFT;
962 dma->size += npage << PAGE_SHIFT;
963 }
964
965 dma->iommu_mapped = true;
966
967 if (ret)
968 vfio_remove_dma(iommu, dma);
969
970 return ret;
971}
972
973static int vfio_dma_do_map(struct vfio_iommu *iommu,
974 struct vfio_iommu_type1_dma_map *map)
975{
976 dma_addr_t iova = map->iova;
977 unsigned long vaddr = map->vaddr;
978 size_t size = map->size;
979 int ret = 0, prot = 0;
980 uint64_t mask;
981 struct vfio_dma *dma;
982
983 /* Verify that none of our __u64 fields overflow */
984 if (map->size != size || map->vaddr != vaddr || map->iova != iova)
985 return -EINVAL;
986
987 mask = ((uint64_t)1 << __ffs(vfio_pgsize_bitmap(iommu))) - 1;
988
989 WARN_ON(mask & PAGE_MASK);
990
991 /* READ/WRITE from device perspective */
992 if (map->flags & VFIO_DMA_MAP_FLAG_WRITE)
993 prot |= IOMMU_WRITE;
994 if (map->flags & VFIO_DMA_MAP_FLAG_READ)
995 prot |= IOMMU_READ;
996
997 if (!prot || !size || (size | iova | vaddr) & mask)
998 return -EINVAL;
999
1000 /* Don't allow IOVA or virtual address wrap */
1001 if (iova + size - 1 < iova || vaddr + size - 1 < vaddr)
1002 return -EINVAL;
1003
1004 mutex_lock(&iommu->lock);
1005
1006 if (vfio_find_dma(iommu, iova, size)) {
1007 ret = -EEXIST;
1008 goto out_unlock;
1009 }
1010
1011 dma = kzalloc(sizeof(*dma), GFP_KERNEL);
1012 if (!dma) {
1013 ret = -ENOMEM;
1014 goto out_unlock;
1015 }
1016
1017 dma->iova = iova;
1018 dma->vaddr = vaddr;
1019 dma->prot = prot;
1020 get_task_struct(current);
1021 dma->task = current;
1022 dma->pfn_list = RB_ROOT;
1023
1024 /* Insert zero-sized and grow as we map chunks of it */
1025 vfio_link_dma(iommu, dma);
1026
1027 /* Don't pin and map if container doesn't contain IOMMU capable domain*/
1028 if (!IS_IOMMU_CAP_DOMAIN_IN_CONTAINER(iommu))
1029 dma->size = size;
1030 else
1031 ret = vfio_pin_map_dma(iommu, dma, size);
1032
1033out_unlock:
1034 mutex_unlock(&iommu->lock);
1035 return ret;
1036}

1050static int vfio_iommu_replay(struct vfio_iommu *iommu,
1051 struct vfio_domain *domain)
1052{
1053 struct vfio_domain *d;
1054 struct rb_node *n;
1055 unsigned long limit = rlimit(RLIMIT_MEMLOCK) >> PAGE_SHIFT;
1056 bool lock_cap = capable(CAP_IPC_LOCK);
1057 int ret;
1058
1059 /* Arbitrarily pick the first domain in the list for lookups */
1060 d = list_first_entry(&iommu->domain_list, struct vfio_domain, next);
1061 n = rb_first(&iommu->dma_list);
1062
1063 for (; n; n = rb_next(n)) {
1064 struct vfio_dma *dma;
1065 dma_addr_t iova;
1066
1067 dma = rb_entry(n, struct vfio_dma, node);
1068 iova = dma->iova;
1069
1070 while (iova < dma->iova + dma->size) {
1071 phys_addr_t phys;
1072 size_t size;
1073
1074 if (dma->iommu_mapped) {
1075 phys_addr_t p;
1076 dma_addr_t i;
1077
1078 phys = iommu_iova_to_phys(d->domain, iova);
1079
1080 if (WARN_ON(!phys)) {
1081 iova += PAGE_SIZE;
1082 continue;
1083 }
1084
1085 size = PAGE_SIZE;
1086 p = phys + size;
1087 i = iova + size;
1088 while (i < dma->iova + dma->size &&
1089 p == iommu_iova_to_phys(d->domain, i)) {
1090 size += PAGE_SIZE;
1091 p += PAGE_SIZE;
1092 i += PAGE_SIZE;
1093 }
1094 } else {
1095 unsigned long pfn;
1096 unsigned long vaddr = dma->vaddr +
1097 (iova - dma->iova);
1098 size_t n = dma->iova + dma->size - iova;
1099 long npage;
1100
1101 npage = vfio_pin_pages_remote(dma, vaddr,
1102 n >> PAGE_SHIFT,
1103 &pfn, lock_cap,
1104 limit);
1105 if (npage <= 0) {
1106 WARN_ON(!npage);
1107 ret = (int)npage;
1108 return ret;
1109 }
1110
1111 phys = pfn << PAGE_SHIFT;
1112 size = npage << PAGE_SHIFT;
1113 }
1114
1115 ret = iommu_map(domain->domain, iova, phys,
1116 size, dma->prot | domain->prot);
1117 if (ret)
1118 return ret;
1119
1120 iova += size;
1121 }
1122 dma->iommu_mapped = true;
1123 }
1124 return 0;
1125}

1127/*
1128 * We change our unmap behavior slightly depending on whether the IOMMU
1129 * supports fine-grained superpages. IOMMUs like AMD-Vi will use a superpage
1130 * for practically any contiguous power-of-two mapping we give it. This means
1131 * we don't need to look for contiguous chunks ourselves to make unmapping
1132 * more efficient. On IOMMUs with coarse-grained super pages, like Intel VT-d
1133 * with discrete 2M/1G/512G/1T superpages, identifying contiguous chunks
1134 * significantly boosts non-hugetlbfs mappings and doesn't seem to hurt when
1135 * hugetlbfs is in use.
1136 */
1137static void vfio_test_domain_fgsp(struct vfio_domain *domain)
1138{
1139 struct page *pages;
1140 int ret, order = get_order(PAGE_SIZE * 2);
1141
1142 pages = alloc_pages(GFP_KERNEL | __GFP_ZERO, order);
1143 if (!pages)
1144 return;
1145
1146 ret = iommu_map(domain->domain, 0, page_to_phys(pages), PAGE_SIZE * 2,
1147 IOMMU_READ | IOMMU_WRITE | domain->prot);
1148 if (!ret) {
1149 size_t unmapped = iommu_unmap(domain->domain, 0, PAGE_SIZE);
1150
1151 if (unmapped == PAGE_SIZE)
1152 iommu_unmap(domain->domain, PAGE_SIZE, PAGE_SIZE);
1153 else
1154 domain->fgsp = true;
1155 }
1156
1157 __free_pages(pages, order);
1158}

1175 struct list_head group_resv_regions;

1198 kfree(region);

1202static int vfio_iommu_type1_attach_group(void *iommu_data,
1203 struct iommu_group *iommu_group)
1204{
1205 struct vfio_iommu *iommu = iommu_data;
1206 struct vfio_group *group;
1207 struct vfio_domain *domain, *d;
1208 struct bus_type *bus = NULL, *mdev_bus;
1209 int ret;
1210 bool resv_msi, msi_remap;
1211 phys_addr_t resv_msi_base;
1212
1213 mutex_lock(&iommu->lock);
1214
1215 list_for_each_entry(d, &iommu->domain_list, next) {
1216 if (find_iommu_group(d, iommu_group)) {
1217 mutex_unlock(&iommu->lock);
1218 return -EINVAL;
1219 }
1220 }
1221
1222 if (iommu->external_domain) {
1223 if (find_iommu_group(iommu->external_domain, iommu_group)) {
1224 mutex_unlock(&iommu->lock);
1225 return -EINVAL;
1226 }
1227 }
1228
1229 group = kzalloc(sizeof(*group), GFP_KERNEL);
1230 domain = kzalloc(sizeof(*domain), GFP_KERNEL);
1231 if (!group || !domain) {
1232 ret = -ENOMEM;
1233 goto out_free;
1234 }
1235
1236 group->iommu_group = iommu_group;
1237
1238 /* Determine bus_type in order to allocate a domain */
1239 ret = iommu_group_for_each_dev(iommu_group, &bus, vfio_bus_type);
1240 if (ret)
1241 goto out_free;
1242
1243 mdev_bus = symbol_get(mdev_bus_type);
1244
1245 if (mdev_bus) {
1246 if ((bus == mdev_bus) && !iommu_present(bus)) {
1247 symbol_put(mdev_bus_type);
1248 if (!iommu->external_domain) {
1249 INIT_LIST_HEAD(&domain->group_list);
1250 iommu->external_domain = domain;
1251 } else
1252 kfree(domain);
1253
1254 list_add(&group->next,
1255 &iommu->external_domain->group_list);
1256 mutex_unlock(&iommu->lock);
1257 return 0;
1258 }
1259 symbol_put(mdev_bus_type);
1260 }
1261
1262 domain->domain = iommu_domain_alloc(bus);
1263 if (!domain->domain) {
1264 ret = -EIO;
1265 goto out_free;
1266 }
1267
1268 if (iommu->nesting) {
1269 int attr = 1;
1270
1271 ret = iommu_domain_set_attr(domain->domain, DOMAIN_ATTR_NESTING,
1272 &attr);
1273 if (ret)
1274 goto out_domain;
1275 }
1276
1277 ret = iommu_attach_group(domain->domain, iommu_group);
1278 if (ret)
1279 goto out_domain;
1280
1281 resv_msi = vfio_iommu_has_sw_msi(iommu_group, &resv_msi_base);
1282
1283 INIT_LIST_HEAD(&domain->group_list);
1284 list_add(&group->next, &domain->group_list);
1285
1286 msi_remap = irq_domain_check_msi_remap() ||
1287 iommu_capable(bus, IOMMU_CAP_INTR_REMAP);
1288
1289 if (!allow_unsafe_interrupts && !msi_remap) {
1290 pr_warn("%s: No interrupt remapping support. Use the module param \"allow_unsafe_interrupts\" to enable VFIO IOMMU support on this platform\n",
1291 __func__);
1292 ret = -EPERM;
1293 goto out_detach;
1294 }
1295
1296 if (iommu_capable(bus, IOMMU_CAP_CACHE_COHERENCY))
1297 domain->prot |= IOMMU_CACHE;
1298
1299 /*
1300 * Try to match an existing compatible domain. We don't want to
1301 * preclude an IOMMU driver supporting multiple bus_types and being
1302 * able to include different bus_types in the same IOMMU domain, so
1303 * we test whether the domains use the same iommu_ops rather than
1304 * testing if they're on the same bus_type.
1305 */
1306 list_for_each_entry(d, &iommu->domain_list, next) {
1307 if (d->domain->ops == domain->domain->ops &&
1308 d->prot == domain->prot) {
1309 iommu_detach_group(domain->domain, iommu_group);
1310 if (!iommu_attach_group(d->domain, iommu_group)) {
1311 list_add(&group->next, &d->group_list);
1312 iommu_domain_free(domain->domain);
1313 kfree(domain);
1314 mutex_unlock(&iommu->lock);
1315 return 0;
1316 }
1317
1318 ret = iommu_attach_group(domain->domain, iommu_group);
1319 if (ret)
1320 goto out_domain;
1321 }
1322 }
1323
1324 vfio_test_domain_fgsp(domain);
1325
1326 /* replay mappings on new domains */
1327 ret = vfio_iommu_replay(iommu, domain);
1328 if (ret)
1329 goto out_detach;
1330
1331 if (resv_msi) {
1332 ret = iommu_get_msi_cookie(domain->domain, resv_msi_base);
1333 if (ret)
1334 goto out_detach;
1335 }
1336
1337 list_add(&domain->next, &iommu->domain_list);
1338
1339 mutex_unlock(&iommu->lock);
1340
1341 return 0;
1342
1343out_detach:
1344 iommu_detach_group(domain->domain, iommu_group);
1345out_domain:
1346 iommu_domain_free(domain->domain);
1347out_free:
1348 kfree(domain);
1349 kfree(group);
1350 mutex_unlock(&iommu->lock);
1351 return ret;
1352}

1402static void vfio_iommu_type1_detach_group(void *iommu_data,
1403 struct iommu_group *iommu_group)
1404{
1405 struct vfio_iommu *iommu = iommu_data;
1406 struct vfio_domain *domain;
1407 struct vfio_group *group;
1408
1409 mutex_lock(&iommu->lock);
1410
1411 if (iommu->external_domain) {
1412 group = find_iommu_group(iommu->external_domain, iommu_group);
1413 if (group) {
1414 list_del(&group->next);

1416
1417 if (list_empty(&iommu->external_domain->group_list)) {
1418 vfio_sanity_check_pfn_list(iommu);
1419
1420 if (!IS_IOMMU_CAP_DOMAIN_IN_CONTAINER(iommu))
1421 vfio_iommu_unmap_unpin_all(iommu);
1422
1423 kfree(iommu->external_domain);
1424 iommu->external_domain = NULL;
1425 }
1426 goto detach_group_done;
1427 }
1428 }
1429
1430 list_for_each_entry(domain, &iommu->domain_list, next) {
1431 group = find_iommu_group(domain, iommu_group);
1432 if (!group)
1433 continue;
1434
1435 iommu_detach_group(domain->domain, iommu_group);
1436 list_del(&group->next);
1437 kfree(group);
1438 /*
1439 * Group ownership provides privilege, if the group list is
1440 * empty, the domain goes away. If it's the last domain with
1441 * iommu and external domain doesn't exist, then all the
1442 * mappings go away too. If it's the last domain with iommu and
1443 * external domain exist, update accounting
1444 */
1445 if (list_empty(&domain->group_list)) {
1446 if (list_is_singular(&iommu->domain_list)) {
1447 if (!iommu->external_domain)
1448 vfio_iommu_unmap_unpin_all(iommu);
1449 else
1450 vfio_iommu_unmap_unpin_reaccount(iommu);
1451 }
1452 iommu_domain_free(domain->domain);
1453 list_del(&domain->next);
1454 kfree(domain);
1455 }
1456 break;
1457 }
1458
1459detach_group_done:
1460 mutex_unlock(&iommu->lock);
1461}

1463static void *vfio_iommu_type1_open(unsigned long arg)
1464{
1465 struct vfio_iommu *iommu;
1466
1467 iommu = kzalloc(sizeof(*iommu), GFP_KERNEL);
1468 if (!iommu)
1469 return ERR_PTR(-ENOMEM);
1470
1471 switch (arg) {
1472 case VFIO_TYPE1_IOMMU:
1473 break;
1474 case VFIO_TYPE1_NESTING_IOMMU:
1475 iommu->nesting = true;
1476 case VFIO_TYPE1v2_IOMMU:
1477 iommu->v2 = true;
1478 break;
1479 default:
1480 kfree(iommu);
1481 return ERR_PTR(-EINVAL);
1482 }
1483
1484 INIT_LIST_HEAD(&iommu->domain_list);
1485 iommu->dma_list = RB_ROOT;
1486 mutex_init(&iommu->lock);
1487 BLOCKING_INIT_NOTIFIER_HEAD(&iommu->notifier);
1488
1489 return iommu;
1490}

1547static long vfio_iommu_type1_ioctl(void *iommu_data,
1548 unsigned int cmd, unsigned long arg)
1549{
1550 struct vfio_iommu *iommu = iommu_data;
1551 unsigned long minsz;
1552
1553 if (cmd == VFIO_CHECK_EXTENSION) {
1554 switch (arg) {
1555 case VFIO_TYPE1_IOMMU:
1556 case VFIO_TYPE1v2_IOMMU:
1557 case VFIO_TYPE1_NESTING_IOMMU:
1558 return 1;
1559 case VFIO_DMA_CC_IOMMU:
1560 if (!iommu)
1561 return 0;
1562 return vfio_domains_have_iommu_cache(iommu);
1563 default:
1564 return 0;
1565 }
1566 } else if (cmd == VFIO_IOMMU_GET_INFO) {
1567 struct vfio_iommu_type1_info info;
1568
1569 minsz = offsetofend(struct vfio_iommu_type1_info, iova_pgsizes);
1570
1571 if (copy_from_user(&info, (void __user *)arg, minsz))
1572 return -EFAULT;
1573
1574 if (info.argsz < minsz)
1575 return -EINVAL;
1576
1577 info.flags = VFIO_IOMMU_INFO_PGSIZES;
1578
1579 info.iova_pgsizes = vfio_pgsize_bitmap(iommu);
1580
1581 return copy_to_user((void __user *)arg, &info, minsz) ?
1582 -EFAULT : 0;
1583
1584 } else if (cmd == VFIO_IOMMU_MAP_DMA) {
1585 struct vfio_iommu_type1_dma_map map;
1586 uint32_t mask = VFIO_DMA_MAP_FLAG_READ |
1587 VFIO_DMA_MAP_FLAG_WRITE;
1588
1589 minsz = offsetofend(struct vfio_iommu_type1_dma_map, size);
1590
1591 if (copy_from_user(&map, (void __user *)arg, minsz))
1592 return -EFAULT;
1593
1594 if (map.argsz < minsz || map.flags & ~mask)
1595 return -EINVAL;
1596
1597 return vfio_dma_do_map(iommu, &map);
1598
1599 } else if (cmd == VFIO_IOMMU_UNMAP_DMA) {
1600 struct vfio_iommu_type1_dma_unmap unmap;
1601 long ret;
1602
1603 minsz = offsetofend(struct vfio_iommu_type1_dma_unmap, size);
1604
1605 if (copy_from_user(&unmap, (void __user *)arg, minsz))
1606 return -EFAULT;
1607
1608 if (unmap.argsz < minsz || unmap.flags)
1609 return -EINVAL;
1610
1611 ret = vfio_dma_do_unmap(iommu, &unmap);
1612 if (ret)
1613 return ret;
1614
1615 return copy_to_user((void __user *)arg, &unmap, minsz) ?
1616 -EFAULT : 0;
1617 }
1618
1619 return -ENOTTY;
1620}
