70/*
71 * Our VGA arbiter participation is limited since we don't know anything
72 * about the device itself. However, if the device is the only VGA device
73 * downstream of a bridge and VFIO VGA support is disabled, then we can
74 * safely return legacy VGA IO and memory as not decoded since the user
75 * has no way to get to it and routing can be disabled externally at the
76 * bridge.
77 */
78static unsigned int vfio_pci_set_vga_decode(void *opaque, bool single_vga)
79{
80 struct vfio_pci_device *vdev = opaque;
81 struct pci_dev *tmp = NULL, *pdev = vdev->pdev;
82 unsigned char max_busnr;
83 unsigned int decodes;
84
85 if (single_vga || !vfio_vga_disabled() || pci_is_root_bus(pdev->bus))
86 return VGA_RSRC_NORMAL_IO | VGA_RSRC_NORMAL_MEM |
87 VGA_RSRC_LEGACY_IO | VGA_RSRC_LEGACY_MEM;
88
89 max_busnr = pci_bus_max_busnr(pdev->bus);
90 decodes = VGA_RSRC_NORMAL_IO | VGA_RSRC_NORMAL_MEM;
91
92 while ((tmp = pci_get_class(PCI_CLASS_DISPLAY_VGA << 8, tmp)) != NULL) {
93 if (tmp == pdev ||
94 pci_domain_nr(tmp->bus) != pci_domain_nr(pdev->bus) ||
95 pci_is_root_bus(tmp->bus))
96 continue;
97
98 if (tmp->bus->number >= pdev->bus->number &&
99 tmp->bus->number <= max_busnr) {
100 pci_dev_put(tmp);
101 decodes |= VGA_RSRC_LEGACY_IO | VGA_RSRC_LEGACY_MEM;
102 break;
103 }
104 }
105
106 return decodes;
107}

114static void vfio_pci_probe_mmaps(struct vfio_pci_device *vdev)
115{
116 struct resource *res;
117 int bar;
118 struct vfio_pci_dummy_resource *dummy_res;
119
120 INIT_LIST_HEAD(&vdev->dummy_resources_list);
121
122 for (bar = PCI_STD_RESOURCES; bar <= PCI_STD_RESOURCE_END; bar++) {
123 res = vdev->pdev->resource + bar;
124
125 if (!IS_ENABLED(CONFIG_VFIO_PCI_MMAP))
126 goto no_mmap;
127
128 if (!(res->flags & IORESOURCE_MEM))
129 goto no_mmap;
130
131 /*
132 * The PCI core shouldn't set up a resource with a
133 * type but zero size. But there may be bugs that
134 * cause us to do that.
135 */
136 if (!resource_size(res))
137 goto no_mmap;
138
139 if (resource_size(res) >= PAGE_SIZE) {
140 vdev->bar_mmap_supported[bar] = true;
141 continue;
142 }
143
144 if (!(res->start & ~PAGE_MASK)) {
145 /*
146 * Add a dummy resource to reserve the remainder
147 * of the exclusive page in case that hot-add
148 * device's bar is assigned into it.
149 */
150 dummy_res = kzalloc(sizeof(*dummy_res), GFP_KERNEL);
151 if (dummy_res == NULL)
152 goto no_mmap;
153
154 dummy_res->resource.name = "vfio sub-page reserved";
155 dummy_res->resource.start = res->end + 1;
156 dummy_res->resource.end = res->start + PAGE_SIZE - 1;
157 dummy_res->resource.flags = res->flags;
158 if (request_resource(res->parent,
159 &dummy_res->resource)) {
160 kfree(dummy_res);
161 goto no_mmap;
162 }
163 dummy_res->index = bar;
164 list_add(&dummy_res->res_next,
165 &vdev->dummy_resources_list);
166 vdev->bar_mmap_supported[bar] = true;
167 continue;
168 }
169 /*
170 * Here we don't handle the case when the BAR is not page
171 * aligned because we can't expect the BAR will be
172 * assigned into the same location in a page in guest
173 * when we passthrough the BAR. And it's hard to access
174 * this BAR in userspace because we have no way to get
175 * the BAR's location in a page.
176 */
177no_mmap:
178 vdev->bar_mmap_supported[bar] = false;
179 }
180}

185/*
186 * INTx masking requires the ability to disable INTx signaling via PCI_COMMAND
187 * _and_ the ability detect when the device is asserting INTx via PCI_STATUS.
188 * If a device implements the former but not the latter we would typically
189 * expect broken_intx_masking be set and require an exclusive interrupt.
190 * However since we do have control of the device's ability to assert INTx,
191 * we can instead pretend that the device does not implement INTx, virtualizing
192 * the pin register to report zero and maintaining DisINTx set on the host.
193 */
194static bool vfio_pci_nointx(struct pci_dev *pdev)
195{
196 switch (pdev->vendor) {
197 case PCI_VENDOR_ID_INTEL:
198 switch (pdev->device) {
199 /* All i40e (XL710/X710/XXV710) 10/20/25/40GbE NICs */
200 case 0x1572:
201 case 0x1574:
202 case 0x1580 ... 0x1581:
203 case 0x1583 ... 0x158b:
204 case 0x37d0 ... 0x37d2:
205 return true;
206 default:
207 return false;
208 }
209 }
210
211 return false;
212}

214static int vfio_pci_enable(struct vfio_pci_device *vdev)
215{
216 struct pci_dev *pdev = vdev->pdev;
217 int ret;
218 u16 cmd;
219 u8 msix_pos;
220
221 pci_set_power_state(pdev, PCI_D0);
222
223 /* Don't allow our initial saved state to include busmaster */
224 pci_clear_master(pdev);
225
226 ret = pci_enable_device(pdev);
227 if (ret)
228 return ret;
229
230 /* If reset fails because of the device lock, fail this path entirely */
231 ret = pci_try_reset_function(pdev);
232 if (ret == -EAGAIN) {
233 pci_disable_device(pdev);
234 return ret;
235 }
236
237 vdev->reset_works = !ret;
238 pci_save_state(pdev);
239 vdev->pci_saved_state = pci_store_saved_state(pdev);
240 if (!vdev->pci_saved_state)
241 pr_debug("%s: Couldn't store %s saved state\n",
242 __func__, dev_name(&pdev->dev));
243
244 if (likely(!nointxmask)) {
245 if (vfio_pci_nointx(pdev)) {
246 dev_info(&pdev->dev, "Masking broken INTx support\n");
247 vdev->nointx = true;
248 pci_intx(pdev, 0);
249 } else
250 vdev->pci_2_3 = pci_intx_mask_supported(pdev);
251 }
252
253 pci_read_config_word(pdev, PCI_COMMAND, &cmd);
254 if (vdev->pci_2_3 && (cmd & PCI_COMMAND_INTX_DISABLE)) {
255 cmd &= ~PCI_COMMAND_INTX_DISABLE;
256 pci_write_config_word(pdev, PCI_COMMAND, cmd);
257 }
258
259 ret = vfio_config_init(vdev);
260 if (ret) {
261 kfree(vdev->pci_saved_state);
262 vdev->pci_saved_state = NULL;
263 pci_disable_device(pdev);
264 return ret;
265 }
266
267 msix_pos = pdev->msix_cap;
268 if (msix_pos) {
269 u16 flags;
270 u32 table;
271
272 pci_read_config_word(pdev, msix_pos + PCI_MSIX_FLAGS, &flags);
273 pci_read_config_dword(pdev, msix_pos + PCI_MSIX_TABLE, &table);
274
275 vdev->msix_bar = table & PCI_MSIX_TABLE_BIR;
276 vdev->msix_offset = table & PCI_MSIX_TABLE_OFFSET;
277 vdev->msix_size = ((flags & PCI_MSIX_FLAGS_QSIZE) + 1) * 16;
278 } else
279 vdev->msix_bar = 0xFF;
280
281 if (!vfio_vga_disabled() && vfio_pci_is_vga(pdev))
282 vdev->has_vga = true;
283
284
285 if (vfio_pci_is_vga(pdev) &&
286 pdev->vendor == PCI_VENDOR_ID_INTEL &&
287 IS_ENABLED(CONFIG_VFIO_PCI_IGD)) {
288 ret = vfio_pci_igd_init(vdev);
289 if (ret) {
290 dev_warn(&vdev->pdev->dev,
291 "Failed to setup Intel IGD regions\n");
292 vfio_pci_disable(vdev);
293 return ret;
294 }
295 }
296
297 vfio_pci_probe_mmaps(vdev);
298
299 return 0;
300}

302static void vfio_pci_disable(struct vfio_pci_device *vdev)
303{
304 struct pci_dev *pdev = vdev->pdev;
305 struct vfio_pci_dummy_resource *dummy_res, *tmp;
306 int i, bar;
307
308 /* Stop the device from further DMA */
309 pci_clear_master(pdev);
310
311 vfio_pci_set_irqs_ioctl(vdev, VFIO_IRQ_SET_DATA_NONE |
312 VFIO_IRQ_SET_ACTION_TRIGGER,
313 vdev->irq_type, 0, 0, NULL);
314
315 vdev->virq_disabled = false;
316
317 for (i = 0; i < vdev->num_regions; i++)
318 vdev->region[i].ops->release(vdev, &vdev->region[i]);
319
320 vdev->num_regions = 0;
321 kfree(vdev->region);
322 vdev->region = NULL; /* don't krealloc a freed pointer */
323
324 vfio_config_free(vdev);
325
326 for (bar = PCI_STD_RESOURCES; bar <= PCI_STD_RESOURCE_END; bar++) {
327 if (!vdev->barmap[bar])
328 continue;
329 pci_iounmap(pdev, vdev->barmap[bar]);
330 pci_release_selected_regions(pdev, 1 << bar);
331 vdev->barmap[bar] = NULL;
332 }
333
334 list_for_each_entry_safe(dummy_res, tmp,
335 &vdev->dummy_resources_list, res_next) {
336 list_del(&dummy_res->res_next);
337 release_resource(&dummy_res->resource);
338 kfree(dummy_res);
339 }
340
341 vdev->needs_reset = true;
342
343 /*
344 * If we have saved state, restore it. If we can reset the device,
345 * even better. Resetting with current state seems better than
346 * nothing, but saving and restoring current state without reset
347 * is just busy work.
348 */
349 if (pci_load_and_free_saved_state(pdev, &vdev->pci_saved_state)) {
350 pr_info("%s: Couldn't reload %s saved state\n",
351 __func__, dev_name(&pdev->dev));
352
353 if (!vdev->reset_works)
354 goto out;
355
356 pci_save_state(pdev);
357 }
358
359 /*
360 * Disable INTx and MSI, presumably to avoid spurious interrupts
361 * during reset. Stolen from pci_reset_function()
362 */
363 pci_write_config_word(pdev, PCI_COMMAND, PCI_COMMAND_INTX_DISABLE);
364
365 /*
366 * Try to reset the device. The success of this is dependent on
367 * being able to lock the device, which is not always possible.
368 */
369 if (vdev->reset_works && !pci_try_reset_function(pdev))
370 vdev->needs_reset = false;
371
372 pci_restore_state(pdev);
373out:
374 pci_disable_device(pdev);
375
376 vfio_pci_try_bus_reset(vdev);
377
378 if (!disable_idle_d3)
379 pci_set_power_state(pdev, PCI_D3hot);
380}

382static void vfio_pci_release(void *device_data)
383{
384 struct vfio_pci_device *vdev = device_data;
385
386 mutex_lock(&driver_lock);
387
388 if (!(--vdev->refcnt)) {
389 vfio_spapr_pci_eeh_release(vdev->pdev);
390 vfio_pci_disable(vdev);
391 }
392
393 mutex_unlock(&driver_lock);
394
395 module_put(THIS_MODULE);
396}

398static int vfio_pci_open(void *device_data)
399{
400 struct vfio_pci_device *vdev = device_data;
401 int ret = 0;
402
403 if (!try_module_get(THIS_MODULE))
404 return -ENODEV;
405
406 mutex_lock(&driver_lock);
407
408 if (!vdev->refcnt) {
409 ret = vfio_pci_enable(vdev);
410 if (ret)
411 goto error;
412
413 vfio_spapr_pci_eeh_open(vdev->pdev);
414 }
415 vdev->refcnt++;
416error:
417 mutex_unlock(&driver_lock);
418 if (ret)
419 module_put(THIS_MODULE);
420 return ret;
421}

423static int vfio_pci_get_irq_count(struct vfio_pci_device *vdev, int irq_type)
424{
425 if (irq_type == VFIO_PCI_INTX_IRQ_INDEX) {
426 u8 pin;
427 pci_read_config_byte(vdev->pdev, PCI_INTERRUPT_PIN, &pin);
428 if (IS_ENABLED(CONFIG_VFIO_PCI_INTX) && !vdev->nointx && pin)
429 return 1;
430
431 } else if (irq_type == VFIO_PCI_MSI_IRQ_INDEX) {
432 u8 pos;
433 u16 flags;
434
435 pos = vdev->pdev->msi_cap;
436 if (pos) {
437 pci_read_config_word(vdev->pdev,
438 pos + PCI_MSI_FLAGS, &flags);
439 return 1 << ((flags & PCI_MSI_FLAGS_QMASK) >> 1);
440 }
441 } else if (irq_type == VFIO_PCI_MSIX_IRQ_INDEX) {
442 u8 pos;
443 u16 flags;
444
445 pos = vdev->pdev->msix_cap;
446 if (pos) {
447 pci_read_config_word(vdev->pdev,
448 pos + PCI_MSIX_FLAGS, &flags);
449
450 return (flags & PCI_MSIX_FLAGS_QSIZE) + 1;
451 }
452 } else if (irq_type == VFIO_PCI_ERR_IRQ_INDEX) {
453 if (pci_is_pcie(vdev->pdev))
454 return 1;
455 } else if (irq_type == VFIO_PCI_REQ_IRQ_INDEX) {
456 return 1;
457 }
458
459 return 0;
460}

471 struct vfio_pci_dependent_device *devices;

474static int vfio_pci_fill_devs(struct pci_dev *pdev, void *data)
475{
476 struct vfio_pci_fill_info *fill = data;
477 struct iommu_group *iommu_group;
478
479 if (fill->cur == fill->max)
480 return -EAGAIN; /* Something changed, try again */
481
482 iommu_group = iommu_group_get(&pdev->dev);
483 if (!iommu_group)
484 return -EPERM; /* Cannot reset non-isolated devices */
485
486 fill->devices[fill->cur].group_id = iommu_group_id(iommu_group);
487 fill->devices[fill->cur].segment = pci_domain_nr(pdev->bus);
488 fill->devices[fill->cur].bus = pdev->bus->number;
489 fill->devices[fill->cur].devfn = pdev->devfn;
490 fill->cur++;
491 iommu_group_put(iommu_group);
492 return 0;
493}

505static int vfio_pci_validate_devs(struct pci_dev *pdev, void *data)
506{
507 struct vfio_pci_group_info *info = data;
508 struct iommu_group *group;
509 int id, i;
510
511 group = iommu_group_get(&pdev->dev);
512 if (!group)
513 return -EPERM;
514
515 id = iommu_group_id(group);
516
517 for (i = 0; i < info->count; i++)
518 if (info->groups[i].id == id)
519 break;
520
521 iommu_group_put(group);
522
523 return (i == info->count) ? -EINVAL : 0;
524}

542static int vfio_pci_walk_wrapper(struct pci_dev *pdev, void *data)
543{
544 struct vfio_pci_walk_info *walk = data;
545
546 if (!walk->slot || vfio_pci_dev_below_slot(pdev, walk->pdev->slot))

566static int msix_sparse_mmap_cap(struct vfio_pci_device *vdev,
567 struct vfio_info_cap *caps)
568{
569 struct vfio_region_info_cap_sparse_mmap *sparse;
570 size_t end, size;
571 int nr_areas = 2, i = 0, ret;
572
573 end = pci_resource_len(vdev->pdev, vdev->msix_bar);
574
575 /* If MSI-X table is aligned to the start or end, only one area */
576 if (((vdev->msix_offset & PAGE_MASK) == 0) ||
577 (PAGE_ALIGN(vdev->msix_offset + vdev->msix_size) >= end))
578 nr_areas = 1;
579
580 size = sizeof(*sparse) + (nr_areas * sizeof(*sparse->areas));
581
582 sparse = kzalloc(size, GFP_KERNEL);
583 if (!sparse)
584 return -ENOMEM;
585
586 sparse->nr_areas = nr_areas;
587
588 if (vdev->msix_offset & PAGE_MASK) {
589 sparse->areas[i].offset = 0;
590 sparse->areas[i].size = vdev->msix_offset & PAGE_MASK;
591 i++;
592 }
593
594 if (PAGE_ALIGN(vdev->msix_offset + vdev->msix_size) < end) {
595 sparse->areas[i].offset = PAGE_ALIGN(vdev->msix_offset +
596 vdev->msix_size);
597 sparse->areas[i].size = end - sparse->areas[i].offset;
598 i++;
599 }
600
601 ret = vfio_info_add_capability(caps, VFIO_REGION_INFO_CAP_SPARSE_MMAP,
602 sparse);
603 kfree(sparse);
604
605 return ret;
606}

608int vfio_pci_register_dev_region(struct vfio_pci_device *vdev,
609 unsigned int type, unsigned int subtype,
610 const struct vfio_pci_regops *ops,
611 size_t size, u32 flags, void *data)
612{
613 struct vfio_pci_region *region;
614
615 region = krealloc(vdev->region,
616 (vdev->num_regions + 1) * sizeof(*region),
617 GFP_KERNEL);
618 if (!region)
619 return -ENOMEM;
620
621 vdev->region = region;
622 vdev->region[vdev->num_regions].type = type;
623 vdev->region[vdev->num_regions].subtype = subtype;
624 vdev->region[vdev->num_regions].ops = ops;
625 vdev->region[vdev->num_regions].size = size;
626 vdev->region[vdev->num_regions].flags = flags;
627 vdev->region[vdev->num_regions].data = data;
628
629 vdev->num_regions++;
630
631 return 0;
632}

634static long vfio_pci_ioctl(void *device_data,
635 unsigned int cmd, unsigned long arg)
636{
637 struct vfio_pci_device *vdev = device_data;
638 unsigned long minsz;
639
640 if (cmd == VFIO_DEVICE_GET_INFO) {
641 struct vfio_device_info info;
642
643 minsz = offsetofend(struct vfio_device_info, num_irqs);
644
645 if (copy_from_user(&info, (void __user *)arg, minsz))
646 return -EFAULT;
647
648 if (info.argsz < minsz)
649 return -EINVAL;
650
651 info.flags = VFIO_DEVICE_FLAGS_PCI;
652
653 if (vdev->reset_works)
654 info.flags |= VFIO_DEVICE_FLAGS_RESET;
655
656 info.num_regions = VFIO_PCI_NUM_REGIONS + vdev->num_regions;
657 info.num_irqs = VFIO_PCI_NUM_IRQS;
658
659 return copy_to_user((void __user *)arg, &info, minsz) ?
660 -EFAULT : 0;
661
662 } else if (cmd == VFIO_DEVICE_GET_REGION_INFO) {
663 struct pci_dev *pdev = vdev->pdev;
664 struct vfio_region_info info;
665 struct vfio_info_cap caps = { .buf = NULL, .size = 0 };
666 int i, ret;
667
668 minsz = offsetofend(struct vfio_region_info, offset);
669
670 if (copy_from_user(&info, (void __user *)arg, minsz))
671 return -EFAULT;
672
673 if (info.argsz < minsz)
674 return -EINVAL;
675
676 switch (info.index) {
677 case VFIO_PCI_CONFIG_REGION_INDEX:
678 info.offset = VFIO_PCI_INDEX_TO_OFFSET(info.index);
679 info.size = pdev->cfg_size;
680 info.flags = VFIO_REGION_INFO_FLAG_READ |
681 VFIO_REGION_INFO_FLAG_WRITE;
682 break;
683 case VFIO_PCI_BAR0_REGION_INDEX ... VFIO_PCI_BAR5_REGION_INDEX:
684 info.offset = VFIO_PCI_INDEX_TO_OFFSET(info.index);
685 info.size = pci_resource_len(pdev, info.index);
686 if (!info.size) {
687 info.flags = 0;
688 break;
689 }
690
691 info.flags = VFIO_REGION_INFO_FLAG_READ |
692 VFIO_REGION_INFO_FLAG_WRITE;
693 if (vdev->bar_mmap_supported[info.index]) {
694 info.flags |= VFIO_REGION_INFO_FLAG_MMAP;
695 if (info.index == vdev->msix_bar) {
696 ret = msix_sparse_mmap_cap(vdev, &caps);
697 if (ret)
698 return ret;
699 }
700 }
701
702 break;
703 case VFIO_PCI_ROM_REGION_INDEX:
704 {
705 void __iomem *io;
706 size_t size;
707
708 info.offset = VFIO_PCI_INDEX_TO_OFFSET(info.index);
709 info.flags = 0;
710
711 /* Report the BAR size, not the ROM size */
712 info.size = pci_resource_len(pdev, info.index);
713 if (!info.size) {
714 /* Shadow ROMs appear as PCI option ROMs */
715 if (pdev->resource[PCI_ROM_RESOURCE].flags &
716 IORESOURCE_ROM_SHADOW)
717 info.size = 0x20000;
718 else
719 break;
720 }
721
722 /* Is it really there? */
723 io = pci_map_rom(pdev, &size);
724 if (!io || !size) {
725 info.size = 0;
726 break;
727 }
728 pci_unmap_rom(pdev, io);
729
730 info.flags = VFIO_REGION_INFO_FLAG_READ;
731 break;
732 }
733 case VFIO_PCI_VGA_REGION_INDEX:
734 if (!vdev->has_vga)
735 return -EINVAL;
736
737 info.offset = VFIO_PCI_INDEX_TO_OFFSET(info.index);
738 info.size = 0xc0000;
739 info.flags = VFIO_REGION_INFO_FLAG_READ |
740 VFIO_REGION_INFO_FLAG_WRITE;
741
742 break;
743 default:
744 {
745 struct vfio_region_info_cap_type cap_type;
746
747 if (info.index >=
748 VFIO_PCI_NUM_REGIONS + vdev->num_regions)
749 return -EINVAL;
750 info.index = array_index_nospec(info.index,
751 VFIO_PCI_NUM_REGIONS +
752 vdev->num_regions);
753
754 i = info.index - VFIO_PCI_NUM_REGIONS;
755
756 info.offset = VFIO_PCI_INDEX_TO_OFFSET(info.index);
757 info.size = vdev->region[i].size;
758 info.flags = vdev->region[i].flags;
759
760 cap_type.type = vdev->region[i].type;
761 cap_type.subtype = vdev->region[i].subtype;
762
763 ret = vfio_info_add_capability(&caps,
764 VFIO_REGION_INFO_CAP_TYPE,
765 &cap_type);
766 if (ret)
767 return ret;
768
769 }
770 }
771
772 if (caps.size) {
773 info.flags |= VFIO_REGION_INFO_FLAG_CAPS;
774 if (info.argsz < sizeof(info) + caps.size) {
775 info.argsz = sizeof(info) + caps.size;
776 info.cap_offset = 0;
777 } else {
778 vfio_info_cap_shift(&caps, sizeof(info));
779 if (copy_to_user((void __user *)arg +
780 sizeof(info), caps.buf,
781 caps.size)) {
782 kfree(caps.buf);
783 return -EFAULT;
784 }
785 info.cap_offset = sizeof(info);
786 }
787
788 kfree(caps.buf);
789 }
790
791 return copy_to_user((void __user *)arg, &info, minsz) ?
792 -EFAULT : 0;
793
794 } else if (cmd == VFIO_DEVICE_GET_IRQ_INFO) {
795 struct vfio_irq_info info;
796
797 minsz = offsetofend(struct vfio_irq_info, count);
798
799 if (copy_from_user(&info, (void __user *)arg, minsz))
800 return -EFAULT;
801
802 if (info.argsz < minsz || info.index >= VFIO_PCI_NUM_IRQS)
803 return -EINVAL;
804
805 switch (info.index) {
806 case VFIO_PCI_INTX_IRQ_INDEX ... VFIO_PCI_MSIX_IRQ_INDEX:
807 case VFIO_PCI_REQ_IRQ_INDEX:
808 break;
809 case VFIO_PCI_ERR_IRQ_INDEX:
810 if (pci_is_pcie(vdev->pdev))
811 break;
812 /* pass thru to return error */
813 default:
814 return -EINVAL;
815 }
816
817 info.flags = VFIO_IRQ_INFO_EVENTFD;
818
819 info.count = vfio_pci_get_irq_count(vdev, info.index);
820
821 if (info.index == VFIO_PCI_INTX_IRQ_INDEX)
822 info.flags |= (VFIO_IRQ_INFO_MASKABLE |
823 VFIO_IRQ_INFO_AUTOMASKED);
824 else
825 info.flags |= VFIO_IRQ_INFO_NORESIZE;
826
827 return copy_to_user((void __user *)arg, &info, minsz) ?
828 -EFAULT : 0;
829
830 } else if (cmd == VFIO_DEVICE_SET_IRQS) {
831 struct vfio_irq_set hdr;
832 u8 *data = NULL;
833 int max, ret = 0;
834 size_t data_size = 0;
835
836 minsz = offsetofend(struct vfio_irq_set, count);
837
838 if (copy_from_user(&hdr, (void __user *)arg, minsz))
839 return -EFAULT;
840
841 max = vfio_pci_get_irq_count(vdev, hdr.index);
842
843 ret = vfio_set_irqs_validate_and_prepare(&hdr, max,
844 VFIO_PCI_NUM_IRQS, &data_size);
845 if (ret)
846 return ret;
847
848 if (data_size) {
849 data = memdup_user((void __user *)(arg + minsz),
850 data_size);
851 if (IS_ERR(data))
852 return PTR_ERR(data);
853 }
854
855 mutex_lock(&vdev->igate);
856
857 ret = vfio_pci_set_irqs_ioctl(vdev, hdr.flags, hdr.index,
858 hdr.start, hdr.count, data);
859
860 mutex_unlock(&vdev->igate);
861 kfree(data);
862
863 return ret;
864
865 } else if (cmd == VFIO_DEVICE_RESET) {
866 return vdev->reset_works ?
867 pci_try_reset_function(vdev->pdev) : -EINVAL;
868
869 } else if (cmd == VFIO_DEVICE_GET_PCI_HOT_RESET_INFO) {
870 struct vfio_pci_hot_reset_info hdr;
871 struct vfio_pci_fill_info fill = { 0 };
872 struct vfio_pci_dependent_device *devices = NULL;
873 bool slot = false;
874 int ret = 0;
875
876 minsz = offsetofend(struct vfio_pci_hot_reset_info, count);
877
878 if (copy_from_user(&hdr, (void __user *)arg, minsz))
879 return -EFAULT;
880
881 if (hdr.argsz < minsz)
882 return -EINVAL;
883
884 hdr.flags = 0;
885
886 /* Can we do a slot or bus reset or neither? */
887 if (!pci_probe_reset_slot(vdev->pdev->slot))
888 slot = true;
889 else if (pci_probe_reset_bus(vdev->pdev->bus))
890 return -ENODEV;
891
892 /* How many devices are affected? */
893 ret = vfio_pci_for_each_slot_or_bus(vdev->pdev,
894 vfio_pci_count_devs,
895 &fill.max, slot);
896 if (ret)
897 return ret;
898
899 WARN_ON(!fill.max); /* Should always be at least one */
900
901 /*
902 * If there's enough space, fill it now, otherwise return
903 * -ENOSPC and the number of devices affected.
904 */
905 if (hdr.argsz < sizeof(hdr) + (fill.max * sizeof(*devices))) {
906 ret = -ENOSPC;
907 hdr.count = fill.max;
908 goto reset_info_exit;
909 }
910
911 devices = kcalloc(fill.max, sizeof(*devices), GFP_KERNEL);
912 if (!devices)
913 return -ENOMEM;
914
915 fill.devices = devices;
916
917 ret = vfio_pci_for_each_slot_or_bus(vdev->pdev,
918 vfio_pci_fill_devs,
919 &fill, slot);
920
921 /*
922 * If a device was removed between counting and filling,
923 * we may come up short of fill.max. If a device was
924 * added, we'll have a return of -EAGAIN above.
925 */
926 if (!ret)
927 hdr.count = fill.cur;
928
929reset_info_exit:
930 if (copy_to_user((void __user *)arg, &hdr, minsz))
931 ret = -EFAULT;
932
933 if (!ret) {
934 if (copy_to_user((void __user *)(arg + minsz), devices,
935 hdr.count * sizeof(*devices)))
936 ret = -EFAULT;
937 }
938
939 kfree(devices);
940 return ret;
941
942 } else if (cmd == VFIO_DEVICE_PCI_HOT_RESET) {
943 struct vfio_pci_hot_reset hdr;
944 int32_t *group_fds;
945 struct vfio_pci_group_entry *groups;
946 struct vfio_pci_group_info info;
947 bool slot = false;
948 int i, count = 0, ret = 0;
949
950 minsz = offsetofend(struct vfio_pci_hot_reset, count);
951
952 if (copy_from_user(&hdr, (void __user *)arg, minsz))
953 return -EFAULT;
954
955 if (hdr.argsz < minsz || hdr.flags)
956 return -EINVAL;
957
958 /* Can we do a slot or bus reset or neither? */
959 if (!pci_probe_reset_slot(vdev->pdev->slot))
960 slot = true;
961 else if (pci_probe_reset_bus(vdev->pdev->bus))
962 return -ENODEV;
963
964 /*
965 * We can't let userspace give us an arbitrarily large
966 * buffer to copy, so verify how many we think there
967 * could be. Note groups can have multiple devices so
968 * one group per device is the max.
969 */
970 ret = vfio_pci_for_each_slot_or_bus(vdev->pdev,
971 vfio_pci_count_devs,
972 &count, slot);
973 if (ret)
974 return ret;
975
976 /* Somewhere between 1 and count is OK */
977 if (!hdr.count || hdr.count > count)
978 return -EINVAL;
979
980 group_fds = kcalloc(hdr.count, sizeof(*group_fds), GFP_KERNEL);
981 groups = kcalloc(hdr.count, sizeof(*groups), GFP_KERNEL);
982 if (!group_fds || !groups) {
983 kfree(group_fds);
984 kfree(groups);
985 return -ENOMEM;
986 }
987
988 if (copy_from_user(group_fds, (void __user *)(arg + minsz),
989 hdr.count * sizeof(*group_fds))) {
990 kfree(group_fds);
991 kfree(groups);
992 return -EFAULT;
993 }
994
995 /*
996 * For each group_fd, get the group through the vfio external
997 * user interface and store the group and iommu ID. This
998 * ensures the group is held across the reset.
999 */
1000 for (i = 0; i < hdr.count; i++) {
1001 struct vfio_group *group;
1002 struct fd f = fdget(group_fds[i]);
1003 if (!f.file) {
1004 ret = -EBADF;
1005 break;
1006 }
1007
1008 group = vfio_group_get_external_user(f.file);
1009 fdput(f);
1010 if (IS_ERR(group)) {
1011 ret = PTR_ERR(group);
1012 break;
1013 }
1014
1015 groups[i].group = group;
1016 groups[i].id = vfio_external_user_iommu_id(group);
1017 }
1018
1019 kfree(group_fds);
1020
1021 /* release reference to groups on error */
1022 if (ret)
1023 goto hot_reset_release;
1024
1025 info.count = hdr.count;
1026 info.groups = groups;
1027
1028 /*
1029 * Test whether all the affected devices are contained
1030 * by the set of groups provided by the user.
1031 */
1032 ret = vfio_pci_for_each_slot_or_bus(vdev->pdev,
1033 vfio_pci_validate_devs,
1034 &info, slot);
1035 if (!ret)
1036 /* User has access, do the reset */
1037 ret = slot ? pci_try_reset_slot(vdev->pdev->slot) :
1038 pci_try_reset_bus(vdev->pdev->bus);
1039
1040hot_reset_release:
1041 for (i--; i >= 0; i--)
1042 vfio_group_put_external_user(groups[i].group);
1043
1044 kfree(groups);
1045 return ret;
1046 }
1047
1048 return -ENOTTY;
1049}

1051static ssize_t vfio_pci_rw(void *device_data, char __user *buf,
1052 size_t count, loff_t *ppos, bool iswrite)
1053{
1054 unsigned int index = VFIO_PCI_OFFSET_TO_INDEX(*ppos);
1055 struct vfio_pci_device *vdev = device_data;
1056
1057 if (index >= VFIO_PCI_NUM_REGIONS + vdev->num_regions)
1058 return -EINVAL;
1059
1060 switch (index) {
1061 case VFIO_PCI_CONFIG_REGION_INDEX:
1062 return vfio_pci_config_rw(vdev, buf, count, ppos, iswrite);
1063
1064 case VFIO_PCI_ROM_REGION_INDEX:
1065 if (iswrite)
1066 return -EINVAL;
1067 return vfio_pci_bar_rw(vdev, buf, count, ppos, false);
1068
1069 case VFIO_PCI_BAR0_REGION_INDEX ... VFIO_PCI_BAR5_REGION_INDEX:
1070 return vfio_pci_bar_rw(vdev, buf, count, ppos, iswrite);
1071
1072 case VFIO_PCI_VGA_REGION_INDEX:
1073 return vfio_pci_vga_rw(vdev, buf, count, ppos, iswrite);
1074 default:
1075 index -= VFIO_PCI_NUM_REGIONS;
1076 return vdev->region[index].ops->rw(vdev, buf,
1077 count, ppos, iswrite);
1078 }
1079
1080 return -EINVAL;
1081}

1101static int vfio_pci_mmap(void *device_data, struct vm_area_struct *vma)
1102{
1103 struct vfio_pci_device *vdev = device_data;
1104 struct pci_dev *pdev = vdev->pdev;
1105 unsigned int index;
1106 u64 phys_len, req_len, pgoff, req_start;
1107 int ret;
1108
1109 index = vma->vm_pgoff >> (VFIO_PCI_OFFSET_SHIFT - PAGE_SHIFT);
1110
1111 if (vma->vm_end < vma->vm_start)
1112 return -EINVAL;
1113 if ((vma->vm_flags & VM_SHARED) == 0)
1114 return -EINVAL;
1115 if (index >= VFIO_PCI_ROM_REGION_INDEX)
1116 return -EINVAL;
1117 if (!vdev->bar_mmap_supported[index])
1118 return -EINVAL;
1119
1120 phys_len = PAGE_ALIGN(pci_resource_len(pdev, index));
1121 req_len = vma->vm_end - vma->vm_start;
1122 pgoff = vma->vm_pgoff &
1123 ((1U << (VFIO_PCI_OFFSET_SHIFT - PAGE_SHIFT)) - 1);
1124 req_start = pgoff << PAGE_SHIFT;
1125
1126 if (req_start + req_len > phys_len)
1127 return -EINVAL;
1128
1129 if (index == vdev->msix_bar) {
1130 /*
1131 * Disallow mmaps overlapping the MSI-X table; users don't
1132 * get to touch this directly. We could find somewhere
1133 * else to map the overlap, but page granularity is only
1134 * a recommendation, not a requirement, so the user needs
1135 * to know which bits are real. Requiring them to mmap
1136 * around the table makes that clear.
1137 */
1138
1139 /* If neither entirely above nor below, then it overlaps */
1140 if (!(req_start >= vdev->msix_offset + vdev->msix_size ||
1141 req_start + req_len <= vdev->msix_offset))
1142 return -EINVAL;
1143 }
1144
1145 /*
1146 * Even though we don't make use of the barmap for the mmap,
1147 * we need to request the region and the barmap tracks that.
1148 */
1149 if (!vdev->barmap[index]) {
1150 ret = pci_request_selected_regions(pdev,
1151 1 << index, "vfio-pci");
1152 if (ret)
1153 return ret;
1154
1155 vdev->barmap[index] = pci_iomap(pdev, index, 0);
1156 if (!vdev->barmap[index]) {
1157 pci_release_selected_regions(pdev, 1 << index);
1158 return -ENOMEM;
1159 }
1160 }
1161
1162 vma->vm_private_data = vdev;
1163 vma->vm_page_prot = pgprot_noncached(vma->vm_page_prot);
1164 vma->vm_pgoff = (pci_resource_start(pdev, index) >> PAGE_SHIFT) + pgoff;
1165
1166 return remap_pfn_range(vma, vma->vm_start, vma->vm_pgoff,
1167 req_len, vma->vm_page_prot);
1168}

1170static void vfio_pci_request(void *device_data, unsigned int count)
1171{
1172 struct vfio_pci_device *vdev = device_data;
1173
1174 mutex_lock(&vdev->igate);
1175
1176 if (vdev->req_trigger) {
1177 if (!(count % 10))
1178 dev_notice_ratelimited(&vdev->pdev->dev,
1179 "Relaying device request to user (#%u)\n",
1180 count);
1181 eventfd_signal(vdev->req_trigger, 1);
1182 } else if (count == 0) {
1183 dev_warn(&vdev->pdev->dev,
1184 "No device request channel registered, blocked until released by user\n");
1185 }
1186
1187 mutex_unlock(&vdev->igate);
1188}

1201static int vfio_pci_probe(struct pci_dev *pdev, const struct pci_device_id *id)
1202{
1203 struct vfio_pci_device *vdev;
1204 struct iommu_group *group;
1205 int ret;
1206
1207 if (pdev->hdr_type != PCI_HEADER_TYPE_NORMAL)
1208 return -EINVAL;
1209
1210 group = vfio_iommu_group_get(&pdev->dev);
1211 if (!group)
1212 return -EINVAL;
1213
1214 vdev = kzalloc(sizeof(*vdev), GFP_KERNEL);
1215 if (!vdev) {
1216 vfio_iommu_group_put(group, &pdev->dev);
1217 return -ENOMEM;
1218 }
1219
1220 vdev->pdev = pdev;
1221 vdev->irq_type = VFIO_PCI_NUM_IRQS;
1222 mutex_init(&vdev->igate);
1223 spin_lock_init(&vdev->irqlock);
1224
1225 ret = vfio_add_group_dev(&pdev->dev, &vfio_pci_ops, vdev);
1226 if (ret) {
1227 vfio_iommu_group_put(group, &pdev->dev);
1228 kfree(vdev);
1229 return ret;
1230 }
1231
1232 if (vfio_pci_is_vga(pdev)) {
1233 vga_client_register(pdev, vdev, NULL, vfio_pci_set_vga_decode);
1234 vga_set_legacy_decoding(pdev,
1235 vfio_pci_set_vga_decode(vdev, false));
1236 }
1237
1238 if (!disable_idle_d3) {
1239 /*
1240 * pci-core sets the device power state to an unknown value at
1241 * bootup and after being removed from a driver. The only
1242 * transition it allows from this unknown state is to D0, which
1243 * typically happens when a driver calls pci_enable_device().
1244 * We're not ready to enable the device yet, but we do want to
1245 * be able to get to D3. Therefore first do a D0 transition
1246 * before going to D3.
1247 */
1248 pci_set_power_state(pdev, PCI_D0);
1249 pci_set_power_state(pdev, PCI_D3hot);
1250 }
1251
1252 return ret;
1253}

1255static void vfio_pci_remove(struct pci_dev *pdev)
1256{
1257 struct vfio_pci_device *vdev;
1258
1259 vdev = vfio_del_group_dev(&pdev->dev);
1260 if (!vdev)
1261 return;
1262
1263 vfio_iommu_group_put(pdev->dev.iommu_group, &pdev->dev);
1264 kfree(vdev->region);
1265 kfree(vdev);
1266
1267 if (vfio_pci_is_vga(pdev)) {
1268 vga_client_register(pdev, NULL, NULL, NULL);
1269 vga_set_legacy_decoding(pdev,
1270 VGA_RSRC_NORMAL_IO | VGA_RSRC_NORMAL_MEM |
1271 VGA_RSRC_LEGACY_IO | VGA_RSRC_LEGACY_MEM);
1272 }
1273
1274 if (!disable_idle_d3)
1275 pci_set_power_state(pdev, PCI_D0);
1276}

1279 pci_channel_state_t state)

1345/*
1346 * Attempt to do a bus/slot reset if there are devices affected by a reset for
1347 * this device that are needs_reset and all of the affected devices are unused
1348 * (!refcnt). Callers are required to hold driver_lock when calling this to
1349 * prevent device opens and concurrent bus reset attempts. We prevent device
1350 * unbinds by acquiring and holding a reference to the vfio_device.
1351 *
1352 * NB: vfio-core considers a group to be viable even if some devices are
1353 * bound to drivers like pci-stub or pcieport. Here we require all devices
1354 * to be bound to vfio_pci since that's the only way we can be sure they
1355 * stay put.
1356 */
1357static void vfio_pci_try_bus_reset(struct vfio_pci_device *vdev)
1358{
1359 struct vfio_devices devs = { .cur_index = 0 };
1360 int i = 0, ret = -EINVAL;
1361 bool needs_reset = false, slot = false;
1362 struct vfio_pci_device *tmp;
1363
1364 if (!pci_probe_reset_slot(vdev->pdev->slot))
1365 slot = true;
1366 else if (pci_probe_reset_bus(vdev->pdev->bus))
1367 return;
1368
1369 if (vfio_pci_for_each_slot_or_bus(vdev->pdev, vfio_pci_count_devs,
1370 &i, slot) || !i)
1371 return;
1372
1373 devs.max_index = i;
1374 devs.devices = kcalloc(i, sizeof(struct vfio_device *), GFP_KERNEL);
1375 if (!devs.devices)
1376 return;
1377
1378 if (vfio_pci_for_each_slot_or_bus(vdev->pdev,
1379 vfio_pci_get_devs, &devs, slot))
1380 goto put_devs;
1381
1382 for (i = 0; i < devs.cur_index; i++) {
1383 tmp = vfio_device_data(devs.devices[i]);
1384 if (tmp->needs_reset)
1385 needs_reset = true;
1386 if (tmp->refcnt)
1387 goto put_devs;
1388 }
1389
1390 if (needs_reset)
1391 ret = slot ? pci_try_reset_slot(vdev->pdev->slot) :
1392 pci_try_reset_bus(vdev->pdev->bus);
1393
1394put_devs:
1395 for (i = 0; i < devs.cur_index; i++) {
1396 tmp = vfio_device_data(devs.devices[i]);
1397 if (!ret)
1398 tmp->needs_reset = false;
1399
1400 if (!tmp->refcnt && !disable_idle_d3)
1401 pci_set_power_state(tmp->pdev, PCI_D3hot);
1402
1403 vfio_device_put(devs.devices[i]);
1404 }
1405
1406 kfree(devs.devices);
1407}

1411 pci_unregister_driver(&vfio_pci_driver);
1412 vfio_pci_uninit_perm_bits();

1415static void __init vfio_pci_fill_ids(void)
1416{
1417 char *p, *id;
1418 int rc;
1419
1420 /* no ids passed actually */
1421 if (ids[0] == '\0')
1422 return;
1423
1424 /* add ids specified in the module parameter */
1425 p = ids;
1426 while ((id = strsep(&p, ","))) {
1427 unsigned int vendor, device, subvendor = PCI_ANY_ID,
1428 subdevice = PCI_ANY_ID, class = 0, class_mask = 0;
1429 int fields;
1430
1431 if (!strlen(id))
1432 continue;
1433
1434 fields = sscanf(id, "%x:%x:%x:%x:%x:%x",
1435 &vendor, &device, &subvendor, &subdevice,
1436 &class, &class_mask);
1437
1438 if (fields < 2) {
1439 pr_warn("invalid id string \"%s\"\n", id);
1440 continue;
1441 }
1442
1443 rc = pci_add_dynid(&vfio_pci_driver, vendor, device,
1444 subvendor, subdevice, class, class_mask, 0);
1445 if (rc)
1446 pr_warn("failed to add dynamic id [%04hx:%04hx[%04hx:%04hx]] class %#08x/%08x (%d)\n",
1447 vendor, device, subvendor, subdevice,
1448 class, class_mask, rc);
1449 else
1450 pr_info("add [%04hx:%04hx[%04hx:%04hx]] class %#08x/%08x\n",
1451 vendor, device, subvendor, subdevice,
1452 class, class_mask);
1453 }
1454}

1456static int __init vfio_pci_init(void)
1457{
1458 int ret;
1459
1460 /* Allocate shared config space permision data used by all devices */
1461 ret = vfio_pci_init_perm_bits();
1462 if (ret)
1463 return ret;
1464
1465 /* Register and scan for devices */
1466 ret = pci_register_driver(&vfio_pci_driver);
1467 if (ret)
1468 goto out_driver;
1469
1470 vfio_pci_fill_ids();
1471
1472 return 0;
1473
1474out_driver:
1475 vfio_pci_uninit_perm_bits();
1476 return ret;
1477}
