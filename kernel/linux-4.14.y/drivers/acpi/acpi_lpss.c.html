100/*
101 * LPSS_QUIRK_ALWAYS_POWER_ON: override power state for LPSS DMA device.
102 *
103 * The LPSS DMA controller has neither _PS0 nor _PS3 method. Moreover
104 * it can be powered off automatically whenever the last LPSS device goes down.
105 * In case of no power any access to the DMA controller will hang the system.
106 * The behaviour is reproduced on some HP laptops based on Intel BayTrail as
107 * well as on ASuS T100TA transformer.
108 *
109 * This quirk overrides power state of entire LPSS island to keep DMA powered
110 * on whenever we have at least one other device in use.
111 */
112#define LPSS_QUIRK_ALWAYS_POWER_ON BIT(0)

114/* UART Component Parameter Register */
115#define LPSS_UART_CPR 0xF4
116#define LPSS_UART_CPR_AFCE BIT(4)

118static void lpss_uart_setup(struct lpss_private_data *pdata)
119{
120 unsigned int offset;
121 u32 val;
122
123 offset = pdata->dev_desc->prv_offset + LPSS_TX_INT;
124 val = readl(pdata->mmio_base + offset);
125 writel(val | LPSS_TX_INT_MASK, pdata->mmio_base + offset);
126
127 val = readl(pdata->mmio_base + LPSS_UART_CPR);
128 if (!(val & LPSS_UART_CPR_AFCE)) {
129 offset = pdata->dev_desc->prv_offset + LPSS_GENERAL;
130 val = readl(pdata->mmio_base + offset);
131 val |= LPSS_GENERAL_UART_RTS_OVRD;
132 writel(val, pdata->mmio_base + offset);
133 }
134}

136static void lpss_deassert_reset(struct lpss_private_data *pdata)
137{
138 unsigned int offset;
139 u32 val;
140
141 offset = pdata->dev_desc->prv_offset + LPSS_RESETS;
142 val = readl(pdata->mmio_base + offset);
143 val |= LPSS_RESETS_RESET_APB | LPSS_RESETS_RESET_FUNC;
144 writel(val, pdata->mmio_base + offset);
145}

362static int register_device_clock(struct acpi_device *adev,
363 struct lpss_private_data *pdata)
364{
365 const struct lpss_device_desc *dev_desc = pdata->dev_desc;
366 const char *devname = dev_name(&adev->dev);
367 struct clk *clk = ERR_PTR(-ENODEV);
368 struct lpss_clk_data *clk_data;
369 const char *parent, *clk_name;
370 void __iomem *prv_base;
371
372 if (!lpss_clk_dev)
373 lpt_register_clock_device();
374
375 clk_data = platform_get_drvdata(lpss_clk_dev);
376 if (!clk_data)
377 return -ENODEV;
378 clk = clk_data->clk;
379
380 if (!pdata->mmio_base
381 || pdata->mmio_size < dev_desc->prv_offset + LPSS_CLK_SIZE)
382 return -ENODATA;
383
384 parent = clk_data->name;
385 prv_base = pdata->mmio_base + dev_desc->prv_offset;
386
387 if (pdata->fixed_clk_rate) {
388 clk = clk_register_fixed_rate(NULL, devname, parent, 0,
389 pdata->fixed_clk_rate);
390 goto out;
391 }
392
393 if (dev_desc->flags & LPSS_CLK_GATE) {
394 clk = clk_register_gate(NULL, devname, parent, 0,
395 prv_base, 0, 0, NULL);
396 parent = devname;
397 }
398
399 if (dev_desc->flags & LPSS_CLK_DIVIDER) {
400 /* Prevent division by zero */
401 if (!readl(prv_base))
402 writel(LPSS_CLK_DIVIDER_DEF_MASK, prv_base);
403
404 clk_name = kasprintf(GFP_KERNEL, "%s-div", devname);
405 if (!clk_name)
406 return -ENOMEM;
407 clk = clk_register_fractional_divider(NULL, clk_name, parent,
408 0, prv_base,
409 1, 15, 16, 15, 0, NULL);
410 parent = clk_name;
411
412 clk_name = kasprintf(GFP_KERNEL, "%s-update", devname);
413 if (!clk_name) {
414 kfree(parent);
415 return -ENOMEM;
416 }
417 clk = clk_register_gate(NULL, clk_name, parent,
418 CLK_SET_RATE_PARENT | CLK_SET_RATE_GATE,
419 prv_base, 31, 0, NULL);
420 kfree(parent);
421 kfree(clk_name);
422 }
423out:
424 if (IS_ERR(clk))
425 return PTR_ERR(clk);
426
427 pdata->clk = clk;
428 clk_register_clkdev(clk, dev_desc->clk_con_id, devname);
429 return 0;
430}

432static int acpi_lpss_create_device(struct acpi_device *adev,
433 const struct acpi_device_id *id)
434{
435 const struct lpss_device_desc *dev_desc;
436 struct lpss_private_data *pdata;
437 struct resource_entry *rentry;
438 struct list_head resource_list;
439 struct platform_device *pdev;
440 int ret;
441
442 dev_desc = (const struct lpss_device_desc *)id->driver_data;
443 if (!dev_desc) {
444 pdev = acpi_create_platform_device(adev, NULL);
445 return IS_ERR_OR_NULL(pdev) ? PTR_ERR(pdev) : 1;
446 }
447 pdata = kzalloc(sizeof(*pdata), GFP_KERNEL);
448 if (!pdata)
449 return -ENOMEM;
450
451 INIT_LIST_HEAD(&resource_list);
452 ret = acpi_dev_get_resources(adev, &resource_list, is_memory, NULL);
453 if (ret < 0)
454 goto err_out;
455
456 list_for_each_entry(rentry, &resource_list, node)
457 if (resource_type(rentry->res) == IORESOURCE_MEM) {
458 if (dev_desc->prv_size_override)
459 pdata->mmio_size = dev_desc->prv_size_override;
460 else
461 pdata->mmio_size = resource_size(rentry->res);
462 pdata->mmio_base = ioremap(rentry->res->start,
463 pdata->mmio_size);
464 break;
465 }
466
467 acpi_dev_free_resource_list(&resource_list);
468
469 if (!pdata->mmio_base) {
470 /* Avoid acpi_bus_attach() instantiating a pdev for this dev. */
471 adev->pnp.type.platform_id = 0;
472 /* Skip the device, but continue the namespace scan. */
473 ret = 0;
474 goto err_out;
475 }
476
477 pdata->adev = adev;
478 pdata->dev_desc = dev_desc;
479
480 if (dev_desc->setup)
481 dev_desc->setup(pdata);
482
483 if (dev_desc->flags & LPSS_CLK) {
484 ret = register_device_clock(adev, pdata);
485 if (ret) {
486 /* Skip the device, but continue the namespace scan. */
487 ret = 0;
488 goto err_out;
489 }
490 }
491
492 /*
493 * This works around a known issue in ACPI tables where LPSS devices
494 * have _PS0 and _PS3 without _PSC (and no power resources), so
495 * acpi_bus_init_power() will assume that the BIOS has put them into D0.
496 */
497 ret = acpi_device_fix_up_power(adev);
498 if (ret) {
499 /* Skip the device, but continue the namespace scan. */
500 ret = 0;
501 goto err_out;
502 }
503
504 adev->driver_data = pdata;
505 pdev = acpi_create_platform_device(adev, dev_desc->properties);
506 if (!IS_ERR_OR_NULL(pdev)) {
507 return 1;
508 }
509
510 ret = PTR_ERR(pdev);
511 adev->driver_data = NULL;
512
513 err_out:
514 kfree(pdata);
515 return ret;
516}

518static u32 __lpss_reg_read(struct lpss_private_data *pdata, unsigned int reg)
519{
520 return readl(pdata->mmio_base + pdata->dev_desc->prv_offset + reg);
521}

523static void __lpss_reg_write(u32 val, struct lpss_private_data *pdata,
524 unsigned int reg)
525{
526 writel(val, pdata->mmio_base + pdata->dev_desc->prv_offset + reg);
527}
528
529static int lpss_reg_read(struct device *dev, unsigned int reg, u32 *val)
530{
531 struct acpi_device *adev;
532 struct lpss_private_data *pdata;
533 unsigned long flags;
534 int ret;
535
536 ret = acpi_bus_get_device(ACPI_HANDLE(dev), &adev);
537 if (WARN_ON(ret))
538 return ret;
539
540 spin_lock_irqsave(&dev->power.lock, flags);
541 if (pm_runtime_suspended(dev)) {
542 ret = -EAGAIN;
543 goto out;
544 }
545 pdata = acpi_driver_data(adev);
546 if (WARN_ON(!pdata || !pdata->mmio_base)) {
547 ret = -ENODEV;
548 goto out;
549 }
550 *val = __lpss_reg_read(pdata, reg);
551
552 out:
553 spin_unlock_irqrestore(&dev->power.lock, flags);
554 return ret;
555}

557static ssize_t lpss_ltr_show(struct device *dev, struct device_attribute *attr,
558 char *buf)
559{
560 u32 ltr_value = 0;

562 int ret;
563
564 reg = strcmp(attr->attr.name, "auto_ltr") ? LPSS_SW_LTR : LPSS_AUTO_LTR;
565 ret = lpss_reg_read(dev, reg, &ltr_value);

567 return ret;
568
569 return snprintf(buf, PAGE_SIZE, "%08x\n", ltr_value);

603static void acpi_lpss_set_ltr(struct device *dev, s32 val)
604{
605 struct lpss_private_data *pdata = acpi_driver_data(ACPI_COMPANION(dev));
606 u32 ltr_mode, ltr_val;
607
608 ltr_mode = __lpss_reg_read(pdata, LPSS_GENERAL);
609 if (val < 0) {
610 if (ltr_mode & LPSS_GENERAL_LTR_MODE_SW) {
611 ltr_mode &= ~LPSS_GENERAL_LTR_MODE_SW;
612 __lpss_reg_write(ltr_mode, pdata, LPSS_GENERAL);
613 }
614 return;
615 }
616 ltr_val = __lpss_reg_read(pdata, LPSS_SW_LTR) & ~LPSS_LTR_SNOOP_MASK;
617 if (val >= LPSS_LTR_SNOOP_LAT_CUTOFF) {
618 ltr_val |= LPSS_LTR_SNOOP_LAT_32US;
619 val = LPSS_LTR_MAX_VAL;
620 } else if (val > LPSS_LTR_MAX_VAL) {
621 ltr_val |= LPSS_LTR_SNOOP_LAT_32US | LPSS_LTR_SNOOP_REQ;
622 val >>= LPSS_LTR_SNOOP_LAT_SHIFT;
623 } else {
624 ltr_val |= LPSS_LTR_SNOOP_LAT_1US | LPSS_LTR_SNOOP_REQ;
625 }
626 ltr_val |= val;
627 __lpss_reg_write(ltr_val, pdata, LPSS_SW_LTR);
628 if (!(ltr_mode & LPSS_GENERAL_LTR_MODE_SW)) {
629 ltr_mode |= LPSS_GENERAL_LTR_MODE_SW;
630 __lpss_reg_write(ltr_mode, pdata, LPSS_GENERAL);
631 }
632}

634#ifdef CONFIG_PM
635/**
636 * acpi_lpss_save_ctx() - Save the private registers of LPSS device
637 * @dev: LPSS device
638 * @pdata: pointer to the private data of the LPSS device
639 *
640 * Most LPSS devices have private registers which may loose their context when
641 * the device is powered down. acpi_lpss_save_ctx() saves those registers into
642 * prv_reg_ctx array.
643 */
644static void acpi_lpss_save_ctx(struct device *dev,
645 struct lpss_private_data *pdata)
646{
647 unsigned int i;
648
649 for (i = 0; i < LPSS_PRV_REG_COUNT; i++) {
650 unsigned long offset = i * sizeof(u32);
651
652 pdata->prv_reg_ctx[i] = __lpss_reg_read(pdata, offset);
653 dev_dbg(dev, "saving 0x%08x from LPSS reg at offset 0x%02lx\n",
654 pdata->prv_reg_ctx[i], offset);
655 }
656}
657
658/**
659 * acpi_lpss_restore_ctx() - Restore the private registers of LPSS device
660 * @dev: LPSS device
661 * @pdata: pointer to the private data of the LPSS device
662 *
663 * Restores the registers that were previously stored with acpi_lpss_save_ctx().
664 */
665static void acpi_lpss_restore_ctx(struct device *dev,
666 struct lpss_private_data *pdata)
667{
668 unsigned int i;
669
670 for (i = 0; i < LPSS_PRV_REG_COUNT; i++) {
671 unsigned long offset = i * sizeof(u32);
672
673 __lpss_reg_write(pdata->prv_reg_ctx[i], pdata, offset);
674 dev_dbg(dev, "restoring 0x%08x to LPSS reg at offset 0x%02lx\n",
675 pdata->prv_reg_ctx[i], offset);
676 }
677}

679static void acpi_lpss_d3_to_d0_delay(struct lpss_private_data *pdata)
680{
681 /*
682 * The following delay is needed or the subsequent write operations may
683 * fail. The LPSS devices are actually PCI devices and the PCI spec
684 * expects 10ms delay before the device can be accessed after D3 to D0
685 * transition. However some platforms like BSW does not need this delay.
686 */
687 unsigned int delay = 10; /* default 10ms delay */
688
689 if (pdata->dev_desc->flags & LPSS_NO_D3_DELAY)
690 delay = 0;
691
692 msleep(delay);
693}

698 int ret;
699
700 ret = acpi_dev_runtime_resume(dev);
701 if (ret)
702 return ret;
703
704 acpi_lpss_d3_to_d0_delay(pdata);

709 * we have to deassert reset line to be sure that ->probe() will
710 * recognize the device.

713 lpss_deassert_reset(pdata);

757/* IOSF SB for LPSS island */
758#define LPSS_IOSF_UNIT_LPIOEP 0xA0
759#define LPSS_IOSF_UNIT_LPIO1 0xAB
760#define LPSS_IOSF_UNIT_LPIO2 0xAC
761
762#define LPSS_IOSF_PMCSR 0x84
763#define LPSS_PMCSR_D0 0
764#define LPSS_PMCSR_D3hot 3
765#define LPSS_PMCSR_Dx_MASK GENMASK(1, 0)
766
767#define LPSS_IOSF_GPIODEF0 0x154
768#define LPSS_GPIODEF0_DMA1_D3 BIT(2)
769#define LPSS_GPIODEF0_DMA2_D3 BIT(3)
770#define LPSS_GPIODEF0_DMA_D3_MASK GENMASK(3, 2)
771#define LPSS_GPIODEF0_DMA_LLP BIT(13)
772
773static DEFINE_MUTEX(lpss_iosf_mutex);
774
775static void lpss_iosf_enter_d3_state(void)
776{
777 u32 value1 = 0;
778 u32 mask1 = LPSS_GPIODEF0_DMA_D3_MASK | LPSS_GPIODEF0_DMA_LLP;
779 u32 value2 = LPSS_PMCSR_D3hot;
780 u32 mask2 = LPSS_PMCSR_Dx_MASK;
781 /*
782 * PMC provides an information about actual status of the LPSS devices.
783 * Here we read the values related to LPSS power island, i.e. LPSS
784 * devices, excluding both LPSS DMA controllers, along with SCC domain.
785 */
786 u32 func_dis, d3_sts_0, pmc_status, pmc_mask = 0xfe000ffe;
787 int ret;
788
789 ret = pmc_atom_read(PMC_FUNC_DIS, &func_dis);
790 if (ret)
791 return;
792
793 mutex_lock(&lpss_iosf_mutex);
794
795 ret = pmc_atom_read(PMC_D3_STS_0, &d3_sts_0);
796 if (ret)
797 goto exit;
798
799 /*
800 * Get the status of entire LPSS power island per device basis.
801 * Shutdown both LPSS DMA controllers if and only if all other devices
802 * are already in D3hot.
803 */
804 pmc_status = (~(d3_sts_0 | func_dis)) & pmc_mask;
805 if (pmc_status)
806 goto exit;
807
808 iosf_mbi_modify(LPSS_IOSF_UNIT_LPIO1, MBI_CFG_WRITE,
809 LPSS_IOSF_PMCSR, value2, mask2);
810
811 iosf_mbi_modify(LPSS_IOSF_UNIT_LPIO2, MBI_CFG_WRITE,
812 LPSS_IOSF_PMCSR, value2, mask2);
813
814 iosf_mbi_modify(LPSS_IOSF_UNIT_LPIOEP, MBI_CR_WRITE,
815 LPSS_IOSF_GPIODEF0, value1, mask1);
816exit:
817 mutex_unlock(&lpss_iosf_mutex);
818}
819
820static void lpss_iosf_exit_d3_state(void)
821{
822 u32 value1 = LPSS_GPIODEF0_DMA1_D3 | LPSS_GPIODEF0_DMA2_D3 |
823 LPSS_GPIODEF0_DMA_LLP;
824 u32 mask1 = LPSS_GPIODEF0_DMA_D3_MASK | LPSS_GPIODEF0_DMA_LLP;
825 u32 value2 = LPSS_PMCSR_D0;
826 u32 mask2 = LPSS_PMCSR_Dx_MASK;
827
828 mutex_lock(&lpss_iosf_mutex);
829
830 iosf_mbi_modify(LPSS_IOSF_UNIT_LPIOEP, MBI_CR_WRITE,
831 LPSS_IOSF_GPIODEF0, value1, mask1);
832
833 iosf_mbi_modify(LPSS_IOSF_UNIT_LPIO2, MBI_CFG_WRITE,
834 LPSS_IOSF_PMCSR, value2, mask2);
835
836 iosf_mbi_modify(LPSS_IOSF_UNIT_LPIO1, MBI_CFG_WRITE,
837 LPSS_IOSF_PMCSR, value2, mask2);
838
839 mutex_unlock(&lpss_iosf_mutex);
840}

842static int acpi_lpss_runtime_suspend(struct device *dev)
843{
844 struct lpss_private_data *pdata = acpi_driver_data(ACPI_COMPANION(dev));
845 int ret;
846
847 ret = pm_generic_runtime_suspend(dev);
848 if (ret)
849 return ret;
850
851 if (pdata->dev_desc->flags & LPSS_SAVE_CTX)
852 acpi_lpss_save_ctx(dev, pdata);
853
854 ret = acpi_dev_runtime_suspend(dev);
855
856 /*
857 * This call must be last in the sequence, otherwise PMC will return
858 * wrong status for devices being about to be powered off. See
859 * lpss_iosf_enter_d3_state() for further information.
860 */
861 if (lpss_quirks & LPSS_QUIRK_ALWAYS_POWER_ON && iosf_mbi_available())
862 lpss_iosf_enter_d3_state();
863
864 return ret;
865}

867static int acpi_lpss_runtime_resume(struct device *dev)
868{
869 struct lpss_private_data *pdata = acpi_driver_data(ACPI_COMPANION(dev));
870 int ret;
871
872 /*
873 * This call is kept first to be in symmetry with
874 * acpi_lpss_runtime_suspend() one.
875 */
876 if (lpss_quirks & LPSS_QUIRK_ALWAYS_POWER_ON && iosf_mbi_available())
877 lpss_iosf_exit_d3_state();
878
879 ret = acpi_dev_runtime_resume(dev);
880 if (ret)
881 return ret;
882
883 acpi_lpss_d3_to_d0_delay(pdata);
884
885 if (pdata->dev_desc->flags & LPSS_SAVE_CTX)
886 acpi_lpss_restore_ctx(dev, pdata);
887
888 return pm_generic_runtime_resume(dev);
889}

916static int acpi_lpss_platform_notify(struct notifier_block *nb,
917 unsigned long action, void *data)
918{
919 struct platform_device *pdev = to_platform_device(data);
920 struct lpss_private_data *pdata;
921 struct acpi_device *adev;
922 const struct acpi_device_id *id;
923
924 id = acpi_match_device(acpi_lpss_device_ids, &pdev->dev);
925 if (!id || !id->driver_data)
926 return 0;
927
928 if (acpi_bus_get_device(ACPI_HANDLE(&pdev->dev), &adev))
929 return 0;
930
931 pdata = acpi_driver_data(adev);
932 if (!pdata)
933 return 0;
934
935 if (pdata->mmio_base &&
936 pdata->mmio_size < pdata->dev_desc->prv_offset + LPSS_LTR_SIZE) {
937 dev_err(&pdev->dev, "MMIO size insufficient to access LTR\n");
938 return 0;
939 }
940
941 switch (action) {
942 case BUS_NOTIFY_BIND_DRIVER:
943 dev_pm_domain_set(&pdev->dev, &acpi_lpss_pm_domain);
944 break;
945 case BUS_NOTIFY_DRIVER_NOT_BOUND:
946 case BUS_NOTIFY_UNBOUND_DRIVER:
947 dev_pm_domain_set(&pdev->dev, NULL);
948 break;
949 case BUS_NOTIFY_ADD_DEVICE:
950 dev_pm_domain_set(&pdev->dev, &acpi_lpss_pm_domain);
951 if (pdata->dev_desc->flags & LPSS_LTR)
952 return sysfs_create_group(&pdev->dev.kobj,
953 &lpss_attr_group);
954 break;
955 case BUS_NOTIFY_DEL_DEVICE:
956 if (pdata->dev_desc->flags & LPSS_LTR)
957 sysfs_remove_group(&pdev->dev.kobj, &lpss_attr_group);
958 dev_pm_domain_set(&pdev->dev, NULL);
959 break;
960 default:
961 break;
962 }
963
964 return 0;
965}
