349/*******************************************************************************
350 *
351 * FUNCTION: acpi_ex_dump_object
352 *
353 * PARAMETERS: obj_desc - Descriptor to dump
354 * info - Info table corresponding to this object
355 * type
356 *
357 * RETURN: None
358 *
359 * DESCRIPTION: Walk the info table for this object
360 *
361 ******************************************************************************/
362
363static void
364acpi_ex_dump_object(union acpi_operand_object *obj_desc,
365 struct acpi_exdump_info *info)
366{
367 u8 *target;
368 const char *name;
369 u8 count;
370 union acpi_operand_object *start;
371 union acpi_operand_object *data = NULL;
372 union acpi_operand_object *next;
373 struct acpi_namespace_node *node;
374
375 if (!info) {
376 acpi_os_printf
377 ("ExDumpObject: Display not implemented for object type %s\n",
378 acpi_ut_get_object_type_name(obj_desc));
379 return;
380 }
381
382 /* First table entry must contain the table length (# of table entries) */
383
384 count = info->offset;
385
386 while (count) {
387 if (!obj_desc) {
388 return;
389 }
390
391 target = ACPI_ADD_PTR(u8, obj_desc, info->offset);
392 name = info->name;
393
394 switch (info->opcode) {
395 case ACPI_EXD_INIT:
396
397 break;
398
399 case ACPI_EXD_TYPE:
400
401 acpi_os_printf("%20s : %2.2X [%s]\n", "Type",
402 obj_desc->common.type,
403 acpi_ut_get_object_type_name(obj_desc));
404 break;
405
406 case ACPI_EXD_UINT8:
407
408 acpi_os_printf("%20s : %2.2X\n", name, *target);
409 break;
410
411 case ACPI_EXD_UINT16:
412
413 acpi_os_printf("%20s : %4.4X\n", name,
414 ACPI_GET16(target));
415 break;
416
417 case ACPI_EXD_UINT32:
418
419 acpi_os_printf("%20s : %8.8X\n", name,
420 ACPI_GET32(target));
421 break;
422
423 case ACPI_EXD_UINT64:
424
425 acpi_os_printf("%20s : %8.8X%8.8X\n", "Value",
426 ACPI_FORMAT_UINT64(ACPI_GET64(target)));
427 break;
428
429 case ACPI_EXD_POINTER:
430 case ACPI_EXD_ADDRESS:
431
432 acpi_ex_out_pointer(name,
433 *ACPI_CAST_PTR(void *, target));
434 break;
435
436 case ACPI_EXD_STRING:
437
438 acpi_ut_print_string(obj_desc->string.pointer,
439 ACPI_UINT8_MAX);
440 acpi_os_printf("\n");
441 break;
442
443 case ACPI_EXD_BUFFER:
444
445 ACPI_DUMP_BUFFER(obj_desc->buffer.pointer,
446 obj_desc->buffer.length);
447 break;
448
449 case ACPI_EXD_PACKAGE:
450
451 /* Dump the package contents */
452
453 acpi_os_printf("\nPackage Contents:\n");
454 acpi_ex_dump_package_obj(obj_desc, 0, 0);
455 break;
456
457 case ACPI_EXD_FIELD:
458
459 acpi_ex_dump_object(obj_desc,
460 acpi_ex_dump_field_common);
461 break;
462
463 case ACPI_EXD_REFERENCE:
464
465 acpi_ex_out_string("Class Name",
466 acpi_ut_get_reference_name
467 (obj_desc));
468 acpi_ex_dump_reference_obj(obj_desc);
469 break;
470
471 case ACPI_EXD_LIST:
472
473 start = *ACPI_CAST_PTR(void *, target);
474 next = start;
475
476 acpi_os_printf("%20s : %p ", name, next);
477 if (next) {
478 acpi_os_printf("%s (Type %2.2X)",
479 acpi_ut_get_object_type_name
480 (next), next->common.type);
481
482 while (next->common.next_object) {
483 if ((next->common.type ==
484 ACPI_TYPE_LOCAL_DATA) && !data) {
485 data = next;
486 }
487
488 next = next->common.next_object;
489 acpi_os_printf("->%p(%s %2.2X)", next,
490 acpi_ut_get_object_type_name
491 (next),
492 next->common.type);
493
494 if ((next == start) || (next == data)) {
495 acpi_os_printf
496 ("\n**** Error: Object list appears to be circular linked");
497 break;
498 }
499 }
500 } else {
501 acpi_os_printf("- No attached objects");
502 }
503
504 acpi_os_printf("\n");
505 break;
506
507 case ACPI_EXD_HDLR_LIST:
508
509 start = *ACPI_CAST_PTR(void *, target);
510 next = start;
511
512 acpi_os_printf("%20s : %p", name, next);
513 if (next) {
514 acpi_os_printf("(%s %2.2X)",
515 acpi_ut_get_object_type_name
516 (next),
517 next->address_space.space_id);
518
519 while (next->address_space.next) {
520 if ((next->common.type ==
521 ACPI_TYPE_LOCAL_DATA) && !data) {
522 data = next;
523 }
524
525 next = next->address_space.next;
526 acpi_os_printf("->%p(%s %2.2X)", next,
527 acpi_ut_get_object_type_name
528 (next),
529 next->address_space.
530 space_id);
531
532 if ((next == start) || (next == data)) {
533 acpi_os_printf
534 ("\n**** Error: Handler list appears to be circular linked");
535 break;
536 }
537 }
538 }
539
540 acpi_os_printf("\n");
541 break;
542
543 case ACPI_EXD_RGN_LIST:
544
545 start = *ACPI_CAST_PTR(void *, target);
546 next = start;
547
548 acpi_os_printf("%20s : %p", name, next);
549 if (next) {
550 acpi_os_printf("(%s %2.2X)",
551 acpi_ut_get_object_type_name
552 (next), next->common.type);
553
554 while (next->region.next) {
555 if ((next->common.type ==
556 ACPI_TYPE_LOCAL_DATA) && !data) {
557 data = next;
558 }
559
560 next = next->region.next;
561 acpi_os_printf("->%p(%s %2.2X)", next,
562 acpi_ut_get_object_type_name
563 (next),
564 next->common.type);
565
566 if ((next == start) || (next == data)) {
567 acpi_os_printf
568 ("\n**** Error: Region list appears to be circular linked");
569 break;
570 }
571 }
572 }
573
574 acpi_os_printf("\n");
575 break;
576
577 case ACPI_EXD_NODE:
578
579 node =
580 *ACPI_CAST_PTR(struct acpi_namespace_node *,
581 target);
582
583 acpi_os_printf("%20s : %p", name, node);
584 if (node) {
585 acpi_os_printf(" [%4.4s]", node->name.ascii);
586 }
587 acpi_os_printf("\n");
588 break;
589
590 default:
591
592 acpi_os_printf("**** Invalid table opcode [%X] ****\n",
593 info->opcode);
594 return;
595 }
596
597 info++;
598 count--;
599 }
600}

602/*******************************************************************************
603 *
604 * FUNCTION: acpi_ex_dump_operand
605 *
606 * PARAMETERS: *obj_desc - Pointer to entry to be dumped
607 * depth - Current nesting depth
608 *
609 * RETURN: None
610 *
611 * DESCRIPTION: Dump an operand object
612 *
613 ******************************************************************************/
614
615void acpi_ex_dump_operand(union acpi_operand_object *obj_desc, u32 depth)
616{
617 u32 length;
618 u32 index;
619
620 ACPI_FUNCTION_NAME(ex_dump_operand)
621
622 /* Check if debug output enabled */
623 if (!ACPI_IS_DEBUG_ENABLED(ACPI_LV_EXEC, _COMPONENT)) {
624 return;
625 }
626
627 if (!obj_desc) {
628
629 /* This could be a null element of a package */
630
631 ACPI_DEBUG_PRINT((ACPI_DB_EXEC, "Null Object Descriptor\n"));
632 return;
633 }
634
635 if (ACPI_GET_DESCRIPTOR_TYPE(obj_desc) == ACPI_DESC_TYPE_NAMED) {
636 ACPI_DEBUG_PRINT((ACPI_DB_EXEC, "%p Namespace Node: ",
637 obj_desc));
638 ACPI_DUMP_ENTRY(obj_desc, ACPI_LV_EXEC);
639 return;
640 }
641
642 if (ACPI_GET_DESCRIPTOR_TYPE(obj_desc) != ACPI_DESC_TYPE_OPERAND) {
643 ACPI_DEBUG_PRINT((ACPI_DB_EXEC,
644 "%p is not a node or operand object: [%s]\n",
645 obj_desc,
646 acpi_ut_get_descriptor_name(obj_desc)));
647 ACPI_DUMP_BUFFER(obj_desc, sizeof(union acpi_operand_object));
648 return;
649 }
650
651 /* obj_desc is a valid object */
652
653 if (depth > 0) {
654 ACPI_DEBUG_PRINT((ACPI_DB_EXEC, "%*s[%u] %p Refs=%u ",
655 depth, " ", depth, obj_desc,
656 obj_desc->common.reference_count));
657 } else {
658 ACPI_DEBUG_PRINT((ACPI_DB_EXEC, "%p Refs=%u ",
659 obj_desc, obj_desc->common.reference_count));
660 }
661
662 /* Decode object type */
663
664 switch (obj_desc->common.type) {
665 case ACPI_TYPE_LOCAL_REFERENCE:
666
667 acpi_os_printf("Reference: [%s] ",
668 acpi_ut_get_reference_name(obj_desc));
669
670 switch (obj_desc->reference.class) {
671 case ACPI_REFCLASS_DEBUG:
672
673 acpi_os_printf("\n");
674 break;
675
676 case ACPI_REFCLASS_INDEX:
677
678 acpi_os_printf("%p\n", obj_desc->reference.object);
679 break;
680
681 case ACPI_REFCLASS_TABLE:
682
683 acpi_os_printf("Table Index %X\n",
684 obj_desc->reference.value);
685 break;
686
687 case ACPI_REFCLASS_REFOF:
688
689 acpi_os_printf("%p [%s]\n", obj_desc->reference.object,
690 acpi_ut_get_type_name(((union
691 acpi_operand_object
692 *)
693 obj_desc->
694 reference.
695 object)->common.
696 type));
697 break;
698
699 case ACPI_REFCLASS_NAME:
700
701 acpi_ut_repair_name(obj_desc->reference.node->name.
702 ascii);
703 acpi_os_printf("- [%4.4s] (Node %p)\n",
704 obj_desc->reference.node->name.ascii,
705 obj_desc->reference.node);
706 break;
707
708 case ACPI_REFCLASS_ARG:
709 case ACPI_REFCLASS_LOCAL:
710
711 acpi_os_printf("%X\n", obj_desc->reference.value);
712 break;
713
714 default: /* Unknown reference class */
715
716 acpi_os_printf("%2.2X\n", obj_desc->reference.class);
717 break;
718 }
719 break;
720
721 case ACPI_TYPE_BUFFER:
722
723 acpi_os_printf("Buffer length %.2X @ %p\n",
724 obj_desc->buffer.length,
725 obj_desc->buffer.pointer);
726
727 /* Debug only -- dump the buffer contents */
728
729 if (obj_desc->buffer.pointer) {
730 length = obj_desc->buffer.length;
731 if (length > 128) {
732 length = 128;
733 }
734
735 acpi_os_printf
736 ("Buffer Contents: (displaying length 0x%.2X)\n",
737 length);
738 ACPI_DUMP_BUFFER(obj_desc->buffer.pointer, length);
739 }
740 break;
741
742 case ACPI_TYPE_INTEGER:
743
744 acpi_os_printf("Integer %8.8X%8.8X\n",
745 ACPI_FORMAT_UINT64(obj_desc->integer.value));
746 break;
747
748 case ACPI_TYPE_PACKAGE:
749
750 acpi_os_printf("Package [Len %X] ElementArray %p\n",
751 obj_desc->package.count,
752 obj_desc->package.elements);
753
754 /*
755 * If elements exist, package element pointer is valid,
756 * and debug_level exceeds 1, dump package's elements.
757 */
758 if (obj_desc->package.count &&
759 obj_desc->package.elements && acpi_dbg_level > 1) {
760 for (index = 0; index < obj_desc->package.count;
761 index++) {
762 acpi_ex_dump_operand(obj_desc->package.
763 elements[index],
764 depth + 1);
765 }
766 }
767 break;
768
769 case ACPI_TYPE_REGION:
770
771 acpi_os_printf("Region %s (%X)",
772 acpi_ut_get_region_name(obj_desc->region.
773 space_id),
774 obj_desc->region.space_id);
775
776 /*
777 * If the address and length have not been evaluated,
778 * don't print them.
779 */
780 if (!(obj_desc->region.flags & AOPOBJ_DATA_VALID)) {
781 acpi_os_printf("\n");
782 } else {
783 acpi_os_printf(" base %8.8X%8.8X Length %X\n",
784 ACPI_FORMAT_UINT64(obj_desc->region.
785 address),
786 obj_desc->region.length);
787 }
788 break;
789
790 case ACPI_TYPE_STRING:
791
792 acpi_os_printf("String length %X @ %p ",
793 obj_desc->string.length,
794 obj_desc->string.pointer);
795
796 acpi_ut_print_string(obj_desc->string.pointer, ACPI_UINT8_MAX);
797 acpi_os_printf("\n");
798 break;
799
800 case ACPI_TYPE_LOCAL_BANK_FIELD:
801
802 acpi_os_printf("BankField\n");
803 break;
804
805 case ACPI_TYPE_LOCAL_REGION_FIELD:
806
807 acpi_os_printf
808 ("RegionField: Bits=%X AccWidth=%X Lock=%X Update=%X at "
809 "byte=%X bit=%X of below:\n", obj_desc->field.bit_length,
810 obj_desc->field.access_byte_width,
811 obj_desc->field.field_flags & AML_FIELD_LOCK_RULE_MASK,
812 obj_desc->field.field_flags & AML_FIELD_UPDATE_RULE_MASK,
813 obj_desc->field.base_byte_offset,
814 obj_desc->field.start_field_bit_offset);
815
816 acpi_ex_dump_operand(obj_desc->field.region_obj, depth + 1);
817 break;
818
819 case ACPI_TYPE_LOCAL_INDEX_FIELD:
820
821 acpi_os_printf("IndexField\n");
822 break;
823
824 case ACPI_TYPE_BUFFER_FIELD:
825
826 acpi_os_printf("BufferField: %X bits at byte %X bit %X of\n",
827 obj_desc->buffer_field.bit_length,
828 obj_desc->buffer_field.base_byte_offset,
829 obj_desc->buffer_field.start_field_bit_offset);
830
831 if (!obj_desc->buffer_field.buffer_obj) {
832 ACPI_DEBUG_PRINT((ACPI_DB_EXEC, "*NULL*\n"));
833 } else if ((obj_desc->buffer_field.buffer_obj)->common.type !=
834 ACPI_TYPE_BUFFER) {
835 acpi_os_printf("*not a Buffer*\n");
836 } else {
837 acpi_ex_dump_operand(obj_desc->buffer_field.buffer_obj,
838 depth + 1);
839 }
840 break;
841
842 case ACPI_TYPE_EVENT:
843
844 acpi_os_printf("Event\n");
845 break;
846
847 case ACPI_TYPE_METHOD:
848
849 acpi_os_printf("Method(%X) @ %p:%X\n",
850 obj_desc->method.param_count,
851 obj_desc->method.aml_start,
852 obj_desc->method.aml_length);
853 break;
854
855 case ACPI_TYPE_MUTEX:
856
857 acpi_os_printf("Mutex\n");
858 break;
859
860 case ACPI_TYPE_DEVICE:
861
862 acpi_os_printf("Device\n");
863 break;
864
865 case ACPI_TYPE_POWER:
866
867 acpi_os_printf("Power\n");
868 break;
869
870 case ACPI_TYPE_PROCESSOR:
871
872 acpi_os_printf("Processor\n");
873 break;
874
875 case ACPI_TYPE_THERMAL:
876
877 acpi_os_printf("Thermal\n");
878 break;
879
880 default:
881
882 /* Unknown Type */
883
884 acpi_os_printf("Unknown Type %X\n", obj_desc->common.type);
885 break;
886 }
887
888 return;
889}

893 * FUNCTION: acpi_ex_dump_operands
894 *
895 * PARAMETERS: operands - A list of Operand objects
896 * opcode_name - AML opcode name
897 * num_operands - Operand count for this opcode
898 *
899 * DESCRIPTION: Dump the operands associated with the opcode

904acpi_ex_dump_operands(union acpi_operand_object **operands,
905 const char *opcode_name, u32 num_operands)
906{
907 ACPI_FUNCTION_NAME(ex_dump_operands);
908
909 if (!opcode_name) {
910 opcode_name = "UNKNOWN";
911 }
912
913 ACPI_DEBUG_PRINT((ACPI_DB_EXEC,
914 "**** Start operand dump for opcode [%s], %u operands\n",
915 opcode_name, num_operands));
916
917 if (num_operands == 0) {
918 num_operands = 1;
919 }
920
921 /* Dump the individual operands */
922
923 while (num_operands) {
924 acpi_ex_dump_operand(*operands, 0);
925 operands++;
926 num_operands--;
927 }
928
929 ACPI_DEBUG_PRINT((ACPI_DB_EXEC,
930 "**** End operand dump for [%s]\n", opcode_name));
931 return;
932}

1050/*******************************************************************************
1051 *
1052 * FUNCTION: acpi_ex_dump_package_obj
1053 *
1054 * PARAMETERS: obj_desc - Descriptor to dump
1055 * level - Indentation Level
1056 * index - Package index for this object
1057 *
1058 * DESCRIPTION: Dumps the elements of the package
1059 *
1060 ******************************************************************************/
1061
1062static void
1063acpi_ex_dump_package_obj(union acpi_operand_object *obj_desc,
1064 u32 level, u32 index)
1065{
1066 u32 i;
1067
1068 /* Indentation and index output */
1069
1070 if (level > 0) {
1071 for (i = 0; i < level; i++) {
1072 acpi_os_printf(" ");
1073 }
1074
1075 acpi_os_printf("[%.2d] ", index);
1076 }
1077
1078 acpi_os_printf("%p ", obj_desc);
1079
1080 /* Null package elements are allowed */
1081
1082 if (!obj_desc) {
1083 acpi_os_printf("[Null Object]\n");
1084 return;
1085 }
1086
1087 /* Packages may only contain a few object types */
1088
1089 switch (obj_desc->common.type) {
1090 case ACPI_TYPE_INTEGER:
1091
1092 acpi_os_printf("[Integer] = %8.8X%8.8X\n",
1093 ACPI_FORMAT_UINT64(obj_desc->integer.value));
1094 break;
1095
1096 case ACPI_TYPE_STRING:
1097
1098 acpi_os_printf("[String] Value: ");
1099 acpi_ut_print_string(obj_desc->string.pointer, ACPI_UINT8_MAX);
1100 acpi_os_printf("\n");
1101 break;
1102
1103 case ACPI_TYPE_BUFFER:
1104
1105 acpi_os_printf("[Buffer] Length %.2X = ",
1106 obj_desc->buffer.length);
1107 if (obj_desc->buffer.length) {
1108 acpi_ut_debug_dump_buffer(ACPI_CAST_PTR
1109 (u8,
1110 obj_desc->buffer.pointer),
1111 obj_desc->buffer.length,
1112 DB_DWORD_DISPLAY, _COMPONENT);
1113 } else {
1114 acpi_os_printf("\n");
1115 }
1116 break;
1117
1118 case ACPI_TYPE_PACKAGE:
1119
1120 acpi_os_printf("[Package] Contains %u Elements:\n",
1121 obj_desc->package.count);
1122
1123 for (i = 0; i < obj_desc->package.count; i++) {
1124 acpi_ex_dump_package_obj(obj_desc->package.elements[i],
1125 level + 1, i);
1126 }
1127 break;
1128
1129 case ACPI_TYPE_LOCAL_REFERENCE:
1130
1131 acpi_os_printf("[Object Reference] Class [%s]",
1132 acpi_ut_get_reference_name(obj_desc));
1133 acpi_ex_dump_reference_obj(obj_desc);
1134 break;
1135
1136 default:
1137
1138 acpi_os_printf("[%s] Type: %2.2X\n",
1139 acpi_ut_get_type_name(obj_desc->common.type),
1140 obj_desc->common.type);
1141 break;
1142 }
1143}

1145/*******************************************************************************
1146 *
1147 * FUNCTION: acpi_ex_dump_object_descriptor
1148 *
1149 * PARAMETERS: obj_desc - Descriptor to dump
1150 * flags - Force display if TRUE
1151 *
1152 * DESCRIPTION: Dumps the members of the object descriptor given.
1153 *
1154 ******************************************************************************/
1155
1156void
1157acpi_ex_dump_object_descriptor(union acpi_operand_object *obj_desc, u32 flags)
1158{
1159 ACPI_FUNCTION_TRACE(ex_dump_object_descriptor);
1160
1161 if (!obj_desc) {
1162 return_VOID;
1163 }
1164
1165 if (!flags) {
1166
1167 /* Check if debug output enabled */
1168
1169 if (!ACPI_IS_DEBUG_ENABLED(ACPI_LV_OBJECTS, _COMPONENT)) {
1170 return_VOID;
1171 }
1172 }
1173
1174 if (ACPI_GET_DESCRIPTOR_TYPE(obj_desc) == ACPI_DESC_TYPE_NAMED) {
1175 acpi_ex_dump_namespace_node((struct acpi_namespace_node *)
1176 obj_desc, flags);
1177
1178 obj_desc = ((struct acpi_namespace_node *)obj_desc)->object;
1179 if (!obj_desc) {
1180 return_VOID;
1181 }
1182
1183 acpi_os_printf("\nAttached Object %p", obj_desc);
1184 if (ACPI_GET_DESCRIPTOR_TYPE(obj_desc) == ACPI_DESC_TYPE_NAMED) {
1185 acpi_os_printf(" - Namespace Node");
1186 }
1187
1188 acpi_os_printf(":\n");
1189 goto dump_object;
1190 }
1191
1192 if (ACPI_GET_DESCRIPTOR_TYPE(obj_desc) != ACPI_DESC_TYPE_OPERAND) {
1193 acpi_os_printf("%p is not an ACPI operand object: [%s]\n",
1194 obj_desc, acpi_ut_get_descriptor_name(obj_desc));
1195 return_VOID;
1196 }
1197
1198 /* Validate the object type */
1199
1200 if (obj_desc->common.type > ACPI_TYPE_LOCAL_MAX) {
1201 acpi_os_printf("Not a known object type: %2.2X\n",
1202 obj_desc->common.type);
1203 return_VOID;
1204 }
1205
1206dump_object:
1207
1208 if (!obj_desc) {
1209 return_VOID;
1210 }
1211
1212 /* Common Fields */
1213
1214 acpi_ex_dump_object(obj_desc, acpi_ex_dump_common);
1215
1216 /* Object-specific fields */
1217
1218 acpi_ex_dump_object(obj_desc, acpi_ex_dump_info[obj_desc->common.type]);
1219
1220 if (obj_desc->common.type == ACPI_TYPE_REGION) {
1221 obj_desc = obj_desc->common.next_object;
1222 if (obj_desc->common.type > ACPI_TYPE_LOCAL_MAX) {
1223 acpi_os_printf
1224 ("Secondary object is not a known object type: %2.2X\n",
1225 obj_desc->common.type);
1226
1227 return_VOID;
1228 }
1229
1230 acpi_os_printf("\nExtra attached Object (%p):\n", obj_desc);
1231 acpi_ex_dump_object(obj_desc,
1232 acpi_ex_dump_info[obj_desc->common.type]);
1233 }
1234
1235 return_VOID;
1236}
