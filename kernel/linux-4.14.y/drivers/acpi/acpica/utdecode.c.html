90/*******************************************************************************
91 *
92 * FUNCTION: acpi_ut_get_region_name
93 *
94 * PARAMETERS: Space ID - ID for the region
95 *
96 * RETURN: Decoded region space_id name
97 *
98 * DESCRIPTION: Translate a Space ID into a name string (Debug only)
99 *
100 ******************************************************************************/
101
102/* Region type decoding */
103
104const char *acpi_gbl_region_types[ACPI_NUM_PREDEFINED_REGIONS] = {
105 "SystemMemory", /* 0x00 */
106 "SystemIO", /* 0x01 */
107 "PCI_Config", /* 0x02 */
108 "EmbeddedControl", /* 0x03 */
109 "SMBus", /* 0x04 */
110 "SystemCMOS", /* 0x05 */
111 "PCIBARTarget", /* 0x06 */
112 "IPMI", /* 0x07 */
113 "GeneralPurposeIo", /* 0x08 */
114 "GenericSerialBus", /* 0x09 */
115 "PCC" /* 0x0A */
116};
117
118const char *acpi_ut_get_region_name(u8 space_id)
119{
120
121 if (space_id >= ACPI_USER_REGION_BEGIN) {
122 return ("UserDefinedRegion");
123 } else if (space_id == ACPI_ADR_SPACE_DATA_TABLE) {
124 return ("DataTable");
125 } else if (space_id == ACPI_ADR_SPACE_FIXED_HARDWARE) {
126 return ("FunctionalFixedHW");
127 } else if (space_id >= ACPI_NUM_PREDEFINED_REGIONS) {
128 return ("InvalidSpaceId");
129 }
130
131 return (acpi_gbl_region_types[space_id]);
132}

178/*
179 * Elements of acpi_gbl_ns_type_names below must match
180 * one-to-one with values of acpi_object_type
181 *
182 * The type ACPI_TYPE_ANY (Untyped) is used as a "don't care" when searching;
183 * when stored in a table it really means that we have thus far seen no
184 * evidence to indicate what type is actually going to be stored for this
185 & entry.
186 */
187static const char acpi_gbl_bad_type[] = "UNDEFINED";
188
189/* Printable names of the ACPI object types */
190
191static const char *acpi_gbl_ns_type_names[] = {
192 /* 00 */ "Untyped",
193 /* 01 */ "Integer",
194 /* 02 */ "String",
195 /* 03 */ "Buffer",
196 /* 04 */ "Package",
197 /* 05 */ "FieldUnit",
198 /* 06 */ "Device",
199 /* 07 */ "Event",
200 /* 08 */ "Method",
201 /* 09 */ "Mutex",
202 /* 10 */ "Region",
203 /* 11 */ "Power",
204 /* 12 */ "Processor",
205 /* 13 */ "Thermal",
206 /* 14 */ "BufferField",
207 /* 15 */ "DdbHandle",
208 /* 16 */ "DebugObject",
209 /* 17 */ "RegionField",
210 /* 18 */ "BankField",
211 /* 19 */ "IndexField",

225const char *acpi_ut_get_type_name(acpi_object_type type)
226{

229 return (acpi_gbl_bad_type);
230 }
231
232 return (acpi_gbl_ns_type_names[type]);
233}
234
235const char *acpi_ut_get_object_type_name(union acpi_operand_object *obj_desc)
236{
237 ACPI_FUNCTION_TRACE(ut_get_object_type_name);
238
239 if (!obj_desc) {
240 ACPI_DEBUG_PRINT((ACPI_DB_EXEC, "Null Object Descriptor\n"));
241 return_STR("[NULL Object Descriptor]");
242 }
243
244 /* These descriptor types share a common area */
245
246 if ((ACPI_GET_DESCRIPTOR_TYPE(obj_desc) != ACPI_DESC_TYPE_OPERAND) &&
247 (ACPI_GET_DESCRIPTOR_TYPE(obj_desc) != ACPI_DESC_TYPE_NAMED)) {
248 ACPI_DEBUG_PRINT((ACPI_DB_EXEC,
249 "Invalid object descriptor type: 0x%2.2X [%s] (%p)\n",
250 ACPI_GET_DESCRIPTOR_TYPE(obj_desc),
251 acpi_ut_get_descriptor_name(obj_desc),
252 obj_desc));
253
254 return_STR("Invalid object");
255 }
256
257 return_STR(acpi_ut_get_type_name(obj_desc->common.type));
258}

260/*******************************************************************************
261 *
262 * FUNCTION: acpi_ut_get_node_name
263 *
264 * PARAMETERS: object - A namespace node
265 *
266 * RETURN: ASCII name of the node
267 *
268 * DESCRIPTION: Validate the node and return the node's ACPI name.
269 *
270 ******************************************************************************/
271
272const char *acpi_ut_get_node_name(void *object)
273{
274 struct acpi_namespace_node *node = (struct acpi_namespace_node *)object;
275
276 /* Must return a string of exactly 4 characters == ACPI_NAME_SIZE */
277
278 if (!object) {
279 return ("NULL");
280 }
281
282 /* Check for Root node */
283
284 if ((object == ACPI_ROOT_OBJECT) || (object == acpi_gbl_root_node)) {
285 return ("\"\\\" ");
286 }
287
288 /* Descriptor must be a namespace node */
289
290 if (ACPI_GET_DESCRIPTOR_TYPE(node) != ACPI_DESC_TYPE_NAMED) {
291 return ("####");
292 }
293
294 /*
295 * Ensure name is valid. The name was validated/repaired when the node
296 * was created, but make sure it has not been corrupted.
297 */
298 acpi_ut_repair_name(node->name.ascii);
299
300 /* Return the name */
301
302 return (node->name.ascii);
303}

322 /* 02 */ "State-Generic",
323 /* 03 */ "State-Update",

352/*******************************************************************************
353 *
354 * FUNCTION: acpi_ut_get_reference_name
355 *
356 * PARAMETERS: object - An ACPI reference object
357 *
358 * RETURN: Decoded name of the type of reference
359 *
360 * DESCRIPTION: Decode a reference object sub-type to a string.
361 *
362 ******************************************************************************/
363
364/* Printable names of reference object sub-types */
365
366static const char *acpi_gbl_ref_class_names[] = {
367 /* 00 */ "Local",
368 /* 01 */ "Argument",
369 /* 02 */ "RefOf",
370 /* 03 */ "Index",
371 /* 04 */ "DdbHandle",
372 /* 05 */ "Named Object",
373 /* 06 */ "Debug"
374};
375
376const char *acpi_ut_get_reference_name(union acpi_operand_object *object)
377{
378
379 if (!object) {
380 return ("NULL Object");
381 }
382
383 if (ACPI_GET_DESCRIPTOR_TYPE(object) != ACPI_DESC_TYPE_OPERAND) {
384 return ("Not an Operand object");
385 }
386
387 if (object->common.type != ACPI_TYPE_LOCAL_REFERENCE) {
388 return ("Not a Reference object");
389 }
390
391 if (object->reference.class > ACPI_REFCLASS_MAX) {
392 return ("Unknown Reference class");
393 }
394
395 return (acpi_gbl_ref_class_names[object->reference.class]);
396}

436/*******************************************************************************
437 *
438 * FUNCTION: acpi_ut_get_notify_name
439 *
440 * PARAMETERS: notify_value - Value from the Notify() request
441 *
442 * RETURN: Decoded name for the notify value
443 *
444 * DESCRIPTION: Translate a Notify Value to a notify namestring.
445 *
446 ******************************************************************************/
447
448/* Names for Notify() values, used for debug output */
449
450static const char *acpi_gbl_generic_notify[ACPI_GENERIC_NOTIFY_MAX + 1] = {
451 /* 00 */ "Bus Check",
452 /* 01 */ "Device Check",
453 /* 02 */ "Device Wake",
454 /* 03 */ "Eject Request",
455 /* 04 */ "Device Check Light",
456 /* 05 */ "Frequency Mismatch",
457 /* 06 */ "Bus Mode Mismatch",
458 /* 07 */ "Power Fault",
459 /* 08 */ "Capabilities Check",
460 /* 09 */ "Device PLD Check",
461 /* 0A */ "Reserved",
462 /* 0B */ "System Locality Update",
463 /* 0C */ "Reserved (was previously Shutdown Request)",
464 /* Reserved in ACPI 6.0 */
465 /* 0D */ "System Resource Affinity Update",
466 /* 0E */ "Heterogeneous Memory Attributes Update"
467 /* ACPI 6.2 */
468};
469
470static const char *acpi_gbl_device_notify[5] = {
471 /* 80 */ "Status Change",
472 /* 81 */ "Information Change",
473 /* 82 */ "Device-Specific Change",
474 /* 83 */ "Device-Specific Change",
475 /* 84 */ "Reserved"
476};
477
478static const char *acpi_gbl_processor_notify[5] = {
479 /* 80 */ "Performance Capability Change",
480 /* 81 */ "C-State Change",
481 /* 82 */ "Throttling Capability Change",
482 /* 83 */ "Guaranteed Change",
483 /* 84 */ "Minimum Excursion"
484};
485
486static const char *acpi_gbl_thermal_notify[5] = {
487 /* 80 */ "Thermal Status Change",
488 /* 81 */ "Thermal Trip Point Change",
489 /* 82 */ "Thermal Device List Change",
490 /* 83 */ "Thermal Relationship Change",
491 /* 84 */ "Reserved"
492};
493
494const char *acpi_ut_get_notify_name(u32 notify_value, acpi_object_type type)
495{
496
497 /* 00 - 0D are "common to all object types" (from ACPI Spec) */
498
499 if (notify_value <= ACPI_GENERIC_NOTIFY_MAX) {
500 return (acpi_gbl_generic_notify[notify_value]);
501 }
502
503 /* 0E - 7F are reserved */
504
505 if (notify_value <= ACPI_MAX_SYS_NOTIFY) {
506 return ("Reserved");
507 }
508
509 /* 80 - 84 are per-object-type */
510
511 if (notify_value <= ACPI_SPECIFIC_NOTIFY_MAX) {
512 switch (type) {
513 case ACPI_TYPE_ANY:
514 case ACPI_TYPE_DEVICE:
515 return (acpi_gbl_device_notify[notify_value - 0x80]);
516
517 case ACPI_TYPE_PROCESSOR:
518 return (acpi_gbl_processor_notify[notify_value - 0x80]);
519
520 case ACPI_TYPE_THERMAL:
521 return (acpi_gbl_thermal_notify[notify_value - 0x80]);
522
523 default:
524 return ("Target object type does not support notifies");
525 }
526 }
527
528 /* 84 - BF are device-specific */
529
530 if (notify_value <= ACPI_MAX_DEVICE_SPECIFIC_NOTIFY) {
531 return ("Device-Specific");
532 }
533
534 /* C0 and above are hardware-specific */
535
536 return ("Hardware-Specific");
537}

591 *
592 * PARAMETERS: type - Object type to be validated

597 *

601{
602
603 if (type > ACPI_TYPE_LOCAL_MAX) {
604
605 /* Note: Assumes all TYPEs are contiguous (external/local) */
606
607 return (FALSE);
608 }
609
610 return (TRUE);
611}
