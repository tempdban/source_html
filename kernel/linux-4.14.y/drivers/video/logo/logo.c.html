13#include <linux/stddef.h>
14#include <linux/module.h>

16#ifdef CONFIG_M68K
17#include <asm/setup.h>
18#endif
19
20static bool nologo;
21module_param(nologo, bool, 0);
22MODULE_PARM_DESC(nologo, "Disables startup logo");
23
24/*
25 * Logos are located in the initdata, and will be freed in kernel_init.
26 * Use late_init to mark the logos as freed to prevent any further use.
27 */
28
29static bool logos_freed;
30
31static int __init fb_logo_late_init(void)
32{
33 logos_freed = true;
34 return 0;
35}
36
37late_initcall_sync(fb_logo_late_init);
38
39/* logo's are marked __initdata. Use __ref to tell
40 * modpost that it is intended that this function uses data
41 * marked __initdata.
42 */
43const struct linux_logo * __ref fb_find_logo(int depth)
44{
45 const struct linux_logo *logo = NULL;
46
47 if (nologo || logos_freed)
48 return NULL;
49
50 if (depth >= 1) {
51#ifdef CONFIG_LOGO_LINUX_MONO
52 /* Generic Linux logo */
53 logo = &logo_linux_mono;
54#endif
55#ifdef CONFIG_LOGO_SUPERH_MONO
56 /* SuperH Linux logo */
57 logo = &logo_superh_mono;
58#endif
59 }
60
61 if (depth >= 4) {
62#ifdef CONFIG_LOGO_LINUX_VGA16
63 /* Generic Linux logo */
64 logo = &logo_linux_vga16;
65#endif
66#ifdef CONFIG_LOGO_BLACKFIN_VGA16
67 /* Blackfin processor logo */
68 logo = &logo_blackfin_vga16;
69#endif
70#ifdef CONFIG_LOGO_SUPERH_VGA16
71 /* SuperH Linux logo */
72 logo = &logo_superh_vga16;
73#endif
74 }
75
76 if (depth >= 8) {
77#ifdef CONFIG_LOGO_LINUX_CLUT224
78 /* Generic Linux logo */
79 logo = &logo_linux_clut224;
80#endif
81#ifdef CONFIG_LOGO_BLACKFIN_CLUT224
82 /* Blackfin Linux logo */
83 logo = &logo_blackfin_clut224;
84#endif
85#ifdef CONFIG_LOGO_DEC_CLUT224
86 /* DEC Linux logo on MIPS/MIPS64 or ALPHA */
87 logo = &logo_dec_clut224;
88#endif
89#ifdef CONFIG_LOGO_MAC_CLUT224
90 /* Macintosh Linux logo on m68k */
91 if (MACH_IS_MAC)
92 logo = &logo_mac_clut224;
93#endif
94#ifdef CONFIG_LOGO_PARISC_CLUT224
95 /* PA-RISC Linux logo */
96 logo = &logo_parisc_clut224;
97#endif
98#ifdef CONFIG_LOGO_SGI_CLUT224
99 /* SGI Linux logo on MIPS/MIPS64 */
100 logo = &logo_sgi_clut224;
101#endif
102#ifdef CONFIG_LOGO_SUN_CLUT224
103 /* Sun Linux logo */
104 logo = &logo_sun_clut224;
105#endif
106#ifdef CONFIG_LOGO_SUPERH_CLUT224
107 /* SuperH Linux logo */
108 logo = &logo_superh_clut224;
109#endif
110#ifdef CONFIG_LOGO_M32R_CLUT224
111 /* M32R Linux logo */
112 logo = &logo_m32r_clut224;
113#endif
114 }
115 return logo;
116}
117EXPORT_SYMBOL_GPL(fb_find_logo);
