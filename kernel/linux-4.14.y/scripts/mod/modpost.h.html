55{

77static inline void __endian(const void *src, void *dest, unsigned int size)
78{
79 unsigned int i;
80 for (i = 0; i < size; i++)
81 ((unsigned char*)dest)[i] = ((unsigned char*)src)[size - i-1];
82}
83
84#define TO_NATIVE(x) \
85({ \
86 typeof(x) __x; \
87 __endian(&(x), &(__x), sizeof(__x)); \
88 __x; \
89})

151static inline int is_shndx_special(unsigned int i)
152{
153 return i != SHN_XINDEX && i >= SHN_LORESERVE && i <= SHN_HIRESERVE;
154}
155
156/*
157 * Move reserved section indices SHN_LORESERVE..SHN_HIRESERVE out of
158 * the way to -256..-1, to avoid conflicting with real section
159 * indices.
160 */
161#define SPECIAL(i) ((i) - (SHN_HIRESERVE + 1))
162
163/* Accessor for sym->st_shndx, hides ugliness of "64k sections" */
164static inline unsigned int get_secindex(const struct elf_info *info,
165 const Elf_Sym *sym)
166{
167 if (is_shndx_special(sym->st_shndx))
168 return SPECIAL(sym->st_shndx);
169 if (sym->st_shndx != SHN_XINDEX)
170 return sym->st_shndx;
171 return info->symtab_shndx_start[sym - info->symtab_start];
172}
