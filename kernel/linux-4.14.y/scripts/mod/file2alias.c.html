46/* This array collects all instances that use the generic do_table */
47struct devtable {
48 const char *device_id; /* name of table, __mod_<name>__*_device_table. */
49 unsigned long id_size;
50 void *function;
51};

53#define ___cat(a,b) a ## b
54#define __cat(a,b) ___cat(a,b)

56/* we need some special handling for this host tool running eventually on
57 * Darwin. The Mach-O section handling is a bit different than ELF section
58 * handling. The differnces in detail are:
59 * a) we have segments which have sections
60 * b) we need a API call to get the respective section symbols */
61#if defined(__MACH__)
62#include <mach-o/getsect.h>
63
64#define INIT_SECTION(name) do { \
65 unsigned long name ## _len; \
66 char *__cat(pstart_,name) = getsectdata("__TEXT", \
67 #name, &__cat(name,_len)); \
68 char *__cat(pstop_,name) = __cat(pstart_,name) + \
69 __cat(name, _len); \
70 __cat(__start_,name) = (void *)__cat(pstart_,name); \
71 __cat(__stop_,name) = (void *)__cat(pstop_,name); \
72 } while (0)
73#define SECTION(name) __attribute__((section("__TEXT, " #name)))
74
75struct devtable **__start___devtable, **__stop___devtable;
76#else
77#define INIT_SECTION(name) /* no-op for ELF */
78#define SECTION(name) __attribute__((section(#name)))
79
80/* We construct a table of pointers in an ELF section (pointers generally
81 * go unpadded by gcc). ld creates boundary syms for us. */
82extern struct devtable *__start___devtable[], *__stop___devtable[];
83#endif /* __MACH__ */
84
85#if !defined(__used)
86# if __GNUC__ == 3 && __GNUC_MINOR__ < 3
87# define __used __attribute__((__unused__))
88# else
89# define __used __attribute__((__used__))
90# endif
91#endif

93/* Define a variable f that holds the value of field f of struct devid
94 * based at address m.
95 */
96#define DEF_FIELD(m, devid, f) \
97 typeof(((struct devid *)0)->f) f = TO_NATIVE(*(typeof(f) *)((m) + OFF_##devid##_##f))
98/* Define a variable f that holds the address of field f of struct devid
99 * based at address m. Due to the way typeof works, for a field of type
100 * T[N] the variable has type T(*)[N], _not_ T*.
101 */
102#define DEF_FIELD_ADDR(m, devid, f) \
103 typeof(((struct devid *)0)->f) *f = ((m) + OFF_##devid##_##f)
104
105/* Add a table entry. We test function type matches while we're here. */
106#define ADD_TO_DEVTABLE(device_id, type, function) \
107 static struct devtable __cat(devtable,__LINE__) = { \
108 device_id + 0*sizeof((function)((const char *)NULL, \
109 (void *)NULL, \
110 (char *)NULL)), \
111 SIZE_##type, (function) }; \
112 static struct devtable *SECTION(__devtable) __used \
113 __cat(devtable_ptr,__LINE__) = &__cat(devtable,__LINE__)
114
115#define ADD(str, sep, cond, field) \
116do { \
117 strcat(str, sep); \
118 if (cond) \
119 sprintf(str + strlen(str), \
120 sizeof(field) == 1 ? "%02X" : \
121 sizeof(field) == 2 ? "%04X" : \
122 sizeof(field) == 4 ? "%08X" : "", \
123 field); \
124 else \
125 sprintf(str + strlen(str), "*"); \
126} while(0)
127
128/* End in a wildcard, for future extension */
129static inline void add_wildcard(char *str)
130{
131 int len = strlen(str);
132
133 if (str[len - 1] != '*')
134 strcat(str + len, "*");
135}

137static inline void add_uuid(char *str, uuid_le uuid)
138{
139 int len = strlen(str);
140
141 sprintf(str + len, "%02x%02x%02x%02x-%02x%02x-%02x%02x-%02x%02x-%02x%02x%02x%02x%02x%02x",
142 uuid.b[3], uuid.b[2], uuid.b[1], uuid.b[0],
143 uuid.b[5], uuid.b[4], uuid.b[7], uuid.b[6],
144 uuid.b[8], uuid.b[9], uuid.b[10], uuid.b[11],
145 uuid.b[12], uuid.b[13], uuid.b[14], uuid.b[15]);
146}

148/**
149 * Check that sizeof(device_id type) are consistent with size of section
150 * in .o file. If in-consistent then userspace and kernel does not agree
151 * on actual size which is a bug.
152 * Also verify that the final entry in the table is all zeros.
153 * Ignore both checks if build host differ from target host and size differs.
154 **/
155static void device_id_check(const char *modname, const char *device_id,
156 unsigned long size, unsigned long id_size,
157 void *symval)
158{
159 int i;
160
161 if (size % id_size || size < id_size) {
162 fatal("%s: sizeof(struct %s_device_id)=%lu is not a modulo "
163 "of the size of "
164 "section __mod_%s__<identifier>_device_table=%lu.\n"
165 "Fix definition of struct %s_device_id "
166 "in mod_devicetable.h\n",
167 modname, device_id, id_size, device_id, size, device_id);
168 }
169 /* Verify last one is a terminator */
170 for (i = 0; i < id_size; i++ ) {
171 if (*(uint8_t*)(symval+size-id_size+i)) {
172 fprintf(stderr,"%s: struct %s_device_id is %lu bytes. "
173 "The last of %lu is:\n",
174 modname, device_id, id_size, size / id_size);
175 for (i = 0; i < id_size; i++ )
176 fprintf(stderr,"0x%02x ",
177 *(uint8_t*)(symval+size-id_size+i) );
178 fprintf(stderr,"\n");
179 fatal("%s: struct %s_device_id is not terminated "
180 "with a NULL entry!\n", modname, device_id);
181 }
182 }
183}

185/* USB is special because the bcdDevice can be matched against a numeric range */
186/* Looks like "usb:vNpNdNdcNdscNdpNicNiscNipNinN" */
187static void do_usb_entry(void *symval,
188 unsigned int bcdDevice_initial, int bcdDevice_initial_digits,
189 unsigned char range_lo, unsigned char range_hi,
190 unsigned char max, struct module *mod)
191{
192 char alias[500];
193 DEF_FIELD(symval, usb_device_id, match_flags);
194 DEF_FIELD(symval, usb_device_id, idVendor);
195 DEF_FIELD(symval, usb_device_id, idProduct);
196 DEF_FIELD(symval, usb_device_id, bcdDevice_lo);
197 DEF_FIELD(symval, usb_device_id, bDeviceClass);
198 DEF_FIELD(symval, usb_device_id, bDeviceSubClass);
199 DEF_FIELD(symval, usb_device_id, bDeviceProtocol);
200 DEF_FIELD(symval, usb_device_id, bInterfaceClass);
201 DEF_FIELD(symval, usb_device_id, bInterfaceSubClass);
202 DEF_FIELD(symval, usb_device_id, bInterfaceProtocol);
203 DEF_FIELD(symval, usb_device_id, bInterfaceNumber);
204
205 strcpy(alias, "usb:");
206 ADD(alias, "v", match_flags&USB_DEVICE_ID_MATCH_VENDOR,
207 idVendor);
208 ADD(alias, "p", match_flags&USB_DEVICE_ID_MATCH_PRODUCT,
209 idProduct);
210
211 strcat(alias, "d");
212 if (bcdDevice_initial_digits)
213 sprintf(alias + strlen(alias), "%0*X",
214 bcdDevice_initial_digits, bcdDevice_initial);
215 if (range_lo == range_hi)
216 sprintf(alias + strlen(alias), "%X", range_lo);
217 else if (range_lo > 0 || range_hi < max) {
218 if (range_lo > 0x9 || range_hi < 0xA)
219 sprintf(alias + strlen(alias),
220 "[%X-%X]",
221 range_lo,
222 range_hi);
223 else {
224 sprintf(alias + strlen(alias),
225 range_lo < 0x9 ? "[%X-9" : "[%X",
226 range_lo);
227 sprintf(alias + strlen(alias),
228 range_hi > 0xA ? "A-%X]" : "%X]",
229 range_hi);
230 }
231 }
232 if (bcdDevice_initial_digits < (sizeof(bcdDevice_lo) * 2 - 1))
233 strcat(alias, "*");
234
235 ADD(alias, "dc", match_flags&USB_DEVICE_ID_MATCH_DEV_CLASS,
236 bDeviceClass);
237 ADD(alias, "dsc", match_flags&USB_DEVICE_ID_MATCH_DEV_SUBCLASS,
238 bDeviceSubClass);
239 ADD(alias, "dp", match_flags&USB_DEVICE_ID_MATCH_DEV_PROTOCOL,
240 bDeviceProtocol);
241 ADD(alias, "ic", match_flags&USB_DEVICE_ID_MATCH_INT_CLASS,
242 bInterfaceClass);
243 ADD(alias, "isc", match_flags&USB_DEVICE_ID_MATCH_INT_SUBCLASS,
244 bInterfaceSubClass);
245 ADD(alias, "ip", match_flags&USB_DEVICE_ID_MATCH_INT_PROTOCOL,
246 bInterfaceProtocol);
247 ADD(alias, "in", match_flags&USB_DEVICE_ID_MATCH_INT_NUMBER,
248 bInterfaceNumber);
249
250 add_wildcard(alias);
251 buf_printf(&mod->dev_table_buf,
252 "MODULE_ALIAS(\"%s\");\n", alias);
253}

255/* Handles increment/decrement of BCD formatted integers */
256/* Returns the previous value, so it works like i++ or i-- */
257static unsigned int incbcd(unsigned int *bcd,
258 int inc,
259 unsigned char max,
260 size_t chars)
261{
262 unsigned int init = *bcd, i, j;
263 unsigned long long c, dec = 0;
264
265 /* If bcd is not in BCD format, just increment */
266 if (max > 0x9) {
267 *bcd += inc;
268 return init;
269 }
270
271 /* Convert BCD to Decimal */
272 for (i=0 ; i < chars ; i++) {
273 c = (*bcd >> (i << 2)) & 0xf;
274 c = c > 9 ? 9 : c; /* force to bcd just in case */
275 for (j=0 ; j < i ; j++)
276 c = c * 10;
277 dec += c;
278 }
279
280 /* Do our increment/decrement */
281 dec += inc;
282 *bcd = 0;
283
284 /* Convert back to BCD */
285 for (i=0 ; i < chars ; i++) {
286 for (c=1,j=0 ; j < i ; j++)
287 c = c * 10;
288 c = (dec / c) % 10;
289 *bcd += c << (i << 2);
290 }
291 return init;
292}

294static void do_usb_entry_multi(void *symval, struct module *mod)
295{
296 unsigned int devlo, devhi;
297 unsigned char chi, clo, max;
298 int ndigits;
299
300 DEF_FIELD(symval, usb_device_id, match_flags);
301 DEF_FIELD(symval, usb_device_id, idVendor);
302 DEF_FIELD(symval, usb_device_id, idProduct);
303 DEF_FIELD(symval, usb_device_id, bcdDevice_lo);
304 DEF_FIELD(symval, usb_device_id, bcdDevice_hi);
305 DEF_FIELD(symval, usb_device_id, bDeviceClass);
306 DEF_FIELD(symval, usb_device_id, bInterfaceClass);
307
308 devlo = match_flags & USB_DEVICE_ID_MATCH_DEV_LO ?
309 bcdDevice_lo : 0x0U;
310 devhi = match_flags & USB_DEVICE_ID_MATCH_DEV_HI ?
311 bcdDevice_hi : ~0x0U;
312
313 /* Figure out if this entry is in bcd or hex format */
314 max = 0x9; /* Default to decimal format */
315 for (ndigits = 0 ; ndigits < sizeof(bcdDevice_lo) * 2 ; ndigits++) {
316 clo = (devlo >> (ndigits << 2)) & 0xf;
317 chi = ((devhi > 0x9999 ? 0x9999 : devhi) >> (ndigits << 2)) & 0xf;
318 if (clo > max || chi > max) {
319 max = 0xf;
320 break;
321 }
322 }
323
324 /*
325 * Some modules (visor) have empty slots as placeholder for
326 * run-time specification that results in catch-all alias
327 */
328 if (!(idVendor | idProduct | bDeviceClass | bInterfaceClass))
329 return;
330
331 /* Convert numeric bcdDevice range into fnmatch-able pattern(s) */
332 for (ndigits = sizeof(bcdDevice_lo) * 2 - 1; devlo <= devhi; ndigits--) {
333 clo = devlo & 0xf;
334 chi = devhi & 0xf;
335 if (chi > max) /* If we are in bcd mode, truncate if necessary */
336 chi = max;
337 devlo >>= 4;
338 devhi >>= 4;
339
340 if (devlo == devhi || !ndigits) {
341 do_usb_entry(symval, devlo, ndigits, clo, chi, max, mod);
342 break;
343 }
344
345 if (clo > 0x0)
346 do_usb_entry(symval,
347 incbcd(&devlo, 1, max,
348 sizeof(bcdDevice_lo) * 2),
349 ndigits, clo, max, max, mod);
350
351 if (chi < max)
352 do_usb_entry(symval,
353 incbcd(&devhi, -1, max,
354 sizeof(bcdDevice_lo) * 2),
355 ndigits, 0x0, chi, max, mod);
356 }
357}

359static void do_usb_table(void *symval, unsigned long size,
360 struct module *mod)
361{
362 unsigned int i;
363 const unsigned long id_size = SIZE_usb_device_id;
364
365 device_id_check(mod->name, "usb", size, id_size, symval);
366
367 /* Leave last one: it's the terminator. */
368 size -= id_size;
369
370 for (i = 0; i < size; i += id_size)
371 do_usb_entry_multi(symval + i, mod);
372}

374static void do_of_entry_multi(void *symval, struct module *mod)
375{
376 char alias[500];
377 int len;
378 char *tmp;
379
380 DEF_FIELD_ADDR(symval, of_device_id, name);
381 DEF_FIELD_ADDR(symval, of_device_id, type);
382 DEF_FIELD_ADDR(symval, of_device_id, compatible);
383
384 len = sprintf(alias, "of:N%sT%s", (*name)[0] ? *name : "*",
385 (*type)[0] ? *type : "*");
386
387 if ((*compatible)[0])
388 sprintf(&alias[len], "%sC%s", (*type)[0] ? "*" : "",
389 *compatible);
390
391 /* Replace all whitespace with underscores */
392 for (tmp = alias; tmp && *tmp; tmp++)
393 if (isspace(*tmp))
394 *tmp = '_';
395
396 buf_printf(&mod->dev_table_buf, "MODULE_ALIAS(\"%s\");\n", alias);
397 strcat(alias, "C");
398 add_wildcard(alias);
399 buf_printf(&mod->dev_table_buf, "MODULE_ALIAS(\"%s\");\n", alias);
400}

461/* Looks like: pci:vNdNsvNsdNbcNscNiN. */
462static int do_pci_entry(const char *filename,
463 void *symval, char *alias)
464{
465 /* Class field can be divided into these three. */
466 unsigned char baseclass, subclass, interface,
467 baseclass_mask, subclass_mask, interface_mask;
468
469 DEF_FIELD(symval, pci_device_id, vendor);
470 DEF_FIELD(symval, pci_device_id, device);
471 DEF_FIELD(symval, pci_device_id, subvendor);
472 DEF_FIELD(symval, pci_device_id, subdevice);
473 DEF_FIELD(symval, pci_device_id, class);
474 DEF_FIELD(symval, pci_device_id, class_mask);
475
476 strcpy(alias, "pci:");
477 ADD(alias, "v", vendor != PCI_ANY_ID, vendor);
478 ADD(alias, "d", device != PCI_ANY_ID, device);
479 ADD(alias, "sv", subvendor != PCI_ANY_ID, subvendor);
480 ADD(alias, "sd", subdevice != PCI_ANY_ID, subdevice);
481
482 baseclass = (class) >> 16;
483 baseclass_mask = (class_mask) >> 16;
484 subclass = (class) >> 8;
485 subclass_mask = (class_mask) >> 8;
486 interface = class;
487 interface_mask = class_mask;
488
489 if ((baseclass_mask != 0 && baseclass_mask != 0xFF)
490 || (subclass_mask != 0 && subclass_mask != 0xFF)
491 || (interface_mask != 0 && interface_mask != 0xFF)) {
492 warn("Can't handle masks in %s:%04X\n",
493 filename, class_mask);
494 return 0;
495 }
496
497 ADD(alias, "bc", baseclass_mask == 0xFF, baseclass);
498 ADD(alias, "sc", subclass_mask == 0xFF, subclass);
499 ADD(alias, "i", interface_mask == 0xFF, interface);
500 add_wildcard(alias);
501 return 1;
502}
503ADD_TO_DEVTABLE("pci", pci_device_id, do_pci_entry);

571/* looks like: "acpi:ACPI0003" or "acpi:PNP0C0B" or "acpi:LNXVIDEO" or
572 * "acpi:bbsspp" (bb=base-class, ss=sub-class, pp=prog-if)
573 *
574 * NOTE: Each driver should use one of the following : _HID, _CIDs
575 * or _CLS. Also, bb, ss, and pp can be substituted with ??
576 * as don't care byte.
577 */
578static int do_acpi_entry(const char *filename,
579 void *symval, char *alias)
580{
581 DEF_FIELD_ADDR(symval, acpi_device_id, id);
582 DEF_FIELD_ADDR(symval, acpi_device_id, cls);
583 DEF_FIELD_ADDR(symval, acpi_device_id, cls_msk);
584
585 if (id && strlen((const char *)*id))
586 sprintf(alias, "acpi*:%s:*", *id);
587 else if (cls) {
588 int i, byte_shift, cnt = 0;
589 unsigned int msk;
590
591 sprintf(&alias[cnt], "acpi*:");
592 cnt = 6;
593 for (i = 1; i <= 3; i++) {
594 byte_shift = 8 * (3-i);
595 msk = (*cls_msk >> byte_shift) & 0xFF;
596 if (msk)
597 sprintf(&alias[cnt], "%02x",
598 (*cls >> byte_shift) & 0xFF);
599 else
600 sprintf(&alias[cnt], "??");
601 cnt += 2;
602 }
603 sprintf(&alias[cnt], ":*");
604 }
605 return 1;
606}
607ADD_TO_DEVTABLE("acpi", acpi_device_id, do_acpi_entry);

609/* looks like: "pnp:dD" */
610static void do_pnp_device_entry(void *symval, unsigned long size,
611 struct module *mod)
612{
613 const unsigned long id_size = SIZE_pnp_device_id;
614 const unsigned int count = (size / id_size)-1;
615 unsigned int i;
616
617 device_id_check(mod->name, "pnp", size, id_size, symval);
618
619 for (i = 0; i < count; i++) {
620 DEF_FIELD_ADDR(symval + i*id_size, pnp_device_id, id);
621 char acpi_id[sizeof(*id)];
622 int j;
623

635/* looks like: "pnp:dD" for every device of the card */
636static void do_pnp_card_entries(void *symval, unsigned long size,
637 struct module *mod)
638{
639 const unsigned long id_size = SIZE_pnp_card_device_id;
640 const unsigned int count = (size / id_size)-1;
641 unsigned int i;
642
643 device_id_check(mod->name, "pnp", size, id_size, symval);
644
645 for (i = 0; i < count; i++) {
646 unsigned int j;
647 DEF_FIELD_ADDR(symval + i*id_size, pnp_card_device_id, devs);
648
649 for (j = 0; j < PNP_MAX_DEVICES; j++) {
650 const char *id = (char *)(*devs)[j].id;
651 int i2, j2;
652 int dup = 0;
653
654 if (!id[0])
655 break;
656
657 /* find duplicate, already added value */
658 for (i2 = 0; i2 < i && !dup; i2++) {
659 DEF_FIELD_ADDR(symval + i2*id_size, pnp_card_device_id, devs);
660
661 for (j2 = 0; j2 < PNP_MAX_DEVICES; j2++) {
662 const char *id2 = (char *)(*devs)[j2].id;
663
664 if (!id2[0])
665 break;
666
667 if (!strcmp(id, id2)) {
668 dup = 1;
669 break;
670 }
671 }
672 }
673
674 /* add an individual alias for every device entry */
675 if (!dup) {
676 char acpi_id[PNP_ID_LEN];
677 int k;
678
679 buf_printf(&mod->dev_table_buf,
680 "MODULE_ALIAS(\"pnp:d%s*\");\n", id);
681
682 /* fix broken pnp bus lowercasing */
683 for (k = 0; k < sizeof(acpi_id); k++)
684 acpi_id[k] = toupper(id[k]);
685 buf_printf(&mod->dev_table_buf,
686 "MODULE_ALIAS(\"acpi*:%s:*\");\n", acpi_id);
687 }
688 }
689 }

988static void dmi_ascii_filter(char *d, const char *s)
989{
990 /* Filter out characters we don't want to see in the modalias string */
991 for (; *s; s++)
992 if (*s > ' ' && *s < 127 && *s != ':')
993 *(d++) = *s;
994
995 *d = 0;

1101/*
1102 * Append a match expression for a single masked hex digit.
1103 * outp points to a pointer to the character at which to append.
1104 * *outp is updated on return to point just after the appended text,
1105 * to facilitate further appending.
1106 */
1107static void append_nibble_mask(char **outp,
1108 unsigned int nibble, unsigned int mask)
1109{
1110 char *p = *outp;
1111 unsigned int i;
1112
1113 switch (mask) {
1114 case 0:
1115 *p++ = '?';
1116 break;
1117
1118 case 0xf:
1119 p += sprintf(p, "%X", nibble);
1120 break;
1121
1122 default:
1123 /*
1124 * Dumbly emit a match pattern for all possible matching
1125 * digits. This could be improved in some cases using ranges,
1126 * but it has the advantage of being trivially correct, and is
1127 * often optimal.
1128 */
1129 *p++ = '[';
1130 for (i = 0; i < 0x10; i++)
1131 if ((i & mask) == nibble)
1132 p += sprintf(p, "%X", i);
1133 *p++ = ']';
1134 }
1135
1136 /* Ensure that the string remains NUL-terminated: */
1137 *p = '\0';
1138
1139 /* Advance the caller's end-of-string pointer: */
1140 *outp = p;
1141}

1143/*
1144 * looks like: "amba:dN"
1145 *
1146 * N is exactly 8 digits, where each is an upper-case hex digit, or
1147 * a ? or [] pattern matching exactly one digit.
1148 */
1149static int do_amba_entry(const char *filename,
1150 void *symval, char *alias)
1151{
1152 unsigned int digit;
1153 char *p = alias;
1154 DEF_FIELD(symval, amba_id, id);
1155 DEF_FIELD(symval, amba_id, mask);
1156
1157 if ((id & mask) != id)
1158 fatal("%s: Masked-off bit(s) of AMBA device ID are non-zero: "
1159 "id=0x%08X, mask=0x%08X. Please fix this driver.\n",
1160 filename, id, mask);
1161
1162 p += sprintf(alias, "amba:d");
1163 for (digit = 0; digit < 8; digit++)
1164 append_nibble_mask(&p,
1165 (id >> (4 * (7 - digit))) & 0xf,
1166 (mask >> (4 * (7 - digit))) & 0xf);
1167
1168 return 1;
1169}
1170ADD_TO_DEVTABLE("amba", amba_id, do_amba_entry);

1304/* Does namelen bytes of name exactly match the symbol? */
1305static bool sym_is(const char *name, unsigned namelen, const char *symbol)
1306{
1307 if (namelen != strlen(symbol))
1308 return false;
1309
1310 return memcmp(name, symbol, namelen) == 0;

1313static void do_table(void *symval, unsigned long size,
1314 unsigned long id_size,
1315 const char *device_id,
1316 void *function,
1317 struct module *mod)
1318{
1319 unsigned int i;
1320 char alias[500];
1321 int (*do_entry)(const char *, void *entry, char *alias) = function;
1322
1323 device_id_check(mod->name, device_id, size, id_size, symval);
1324 /* Leave last one: it's the terminator. */
1325 size -= id_size;
1326
1327 for (i = 0; i < size; i += id_size) {
1328 if (do_entry(mod->name, symval+i, alias)) {
1329 buf_printf(&mod->dev_table_buf,
1330 "MODULE_ALIAS(\"%s\");\n", alias);
1331 }
1332 }
1333}

1335/* Create MODULE_ALIAS() statements.
1336 * At this time, we cannot write the actual output C source yet,
1337 * so we write into the mod->dev_table_buf buffer. */
1338void handle_moddevtable(struct module *mod, struct elf_info *info,
1339 Elf_Sym *sym, const char *symname)
1340{
1341 void *symval;
1342 char *zeros = NULL;
1343 const char *name, *identifier;
1344 unsigned int namelen;
1345
1346 /* We're looking for a section relative symbol */
1347 if (!sym->st_shndx || get_secindex(info, sym) >= info->num_sections)
1348 return;
1349
1350 /* We're looking for an object */
1351 if (ELF_ST_TYPE(sym->st_info) != STT_OBJECT)
1352 return;
1353
1354 /* All our symbols are of form <prefix>__mod_<name>__<identifier>_device_table. */
1355 name = strstr(symname, "__mod_");
1356 if (!name)
1357 return;
1358 name += strlen("__mod_");
1359 namelen = strlen(name);
1360 if (namelen < strlen("_device_table"))
1361 return;
1362 if (strcmp(name + namelen - strlen("_device_table"), "_device_table"))
1363 return;
1364 identifier = strstr(name, "__");
1365 if (!identifier)
1366 return;
1367 namelen = identifier - name;
1368
1369 /* Handle all-NULL symbols allocated into .bss */
1370 if (info->sechdrs[get_secindex(info, sym)].sh_type & SHT_NOBITS) {
1371 zeros = calloc(1, sym->st_size);
1372 symval = zeros;
1373 } else {
1374 symval = (void *)info->hdr
1375 + info->sechdrs[get_secindex(info, sym)].sh_offset
1376 + sym->st_value;
1377 }
1378
1379 /* First handle the "special" cases */
1380 if (sym_is(name, namelen, "usb"))
1381 do_usb_table(symval, sym->st_size, mod);
1382 if (sym_is(name, namelen, "of"))
1383 do_of_table(symval, sym->st_size, mod);
1384 else if (sym_is(name, namelen, "pnp"))
1385 do_pnp_device_entry(symval, sym->st_size, mod);
1386 else if (sym_is(name, namelen, "pnp_card"))
1387 do_pnp_card_entries(symval, sym->st_size, mod);
1388 else {
1389 struct devtable **p;
1390 INIT_SECTION(__devtable);
1391
1392 for (p = __start___devtable; p < __stop___devtable; p++) {
1393 if (sym_is(name, namelen, (*p)->device_id)) {
1394 do_table(symval, sym->st_size, (*p)->id_size,
1395 (*p)->device_id, (*p)->function, mod);
1396 break;
1397 }
1398 }
1399 }
1400 free(zeros);
1401}

1403/* Now add out buffered information to the generated C source */
1404void add_moddevtable(struct buffer *buf, struct module *mod)
1405{
1406 buf_printf(buf, "\n");
1407 buf_write(buf, mod->dev_table_buf.p, mod->dev_table_buf.pos);
1408 free(mod->dev_table_buf.p);
1409}
