82static struct expr *menu_check_dep(struct expr *e)
83{
84 if (!e)
85 return e;
86
87 switch (e->type) {
88 case E_NOT:
89 e->left.expr = menu_check_dep(e->left.expr);
90 break;
91 case E_OR:
92 case E_AND:
93 e->left.expr = menu_check_dep(e->left.expr);
94 e->right.expr = menu_check_dep(e->right.expr);
95 break;
96 case E_SYMBOL:
97 /* change 'm' into 'm' && MODULES */
98 if (e->left.sym == &symbol_mod)
99 return expr_alloc_and(e, expr_alloc_symbol(modules_sym));
100 break;
101 default:
102 break;
103 }
104 return e;
105}

112void menu_set_type(int type)
113{
114 struct symbol *sym = current_entry->sym;
115
116 if (sym->type == type)
117 return;
118 if (sym->type == S_UNKNOWN) {
119 sym->type = type;
120 return;
121 }
122 menu_warn(current_entry,
123 "ignoring type redefinition of '%s' from '%s' to '%s'",
124 sym->name ? sym->name : "<choice>",
125 sym_type_name(sym->type), sym_type_name(type));
126}

128static struct property *menu_add_prop(enum prop_type type, char *prompt, struct expr *expr, struct expr *dep)
129{
130 struct property *prop = prop_alloc(type, current_entry->sym);
131
132 prop->menu = current_entry;
133 prop->expr = expr;
134 prop->visible.expr = menu_check_dep(dep);
135
136 if (prompt) {
137 if (isspace(*prompt)) {
138 prop_warn(prop, "leading whitespace ignored");
139 while (isspace(*prompt))
140 prompt++;
141 }
142 if (current_entry->prompt && current_entry != &rootmenu)
143 prop_warn(prop, "prompt redefined");
144
145 /* Apply all upper menus' visibilities to actual prompts. */
146 if(type == P_PROMPT) {
147 struct menu *menu = current_entry;
148
149 while ((menu = menu->parent) != NULL) {
150 struct expr *dup_expr;
151
152 if (!menu->visibility)
153 continue;
154 /*
155 * Do not add a reference to the
156 * menu's visibility expression but
157 * use a copy of it. Otherwise the
158 * expression reduction functions
159 * will modify expressions that have
160 * multiple references which can
161 * cause unwanted side effects.
162 */
163 dup_expr = expr_copy(menu->visibility);
164
165 prop->visible.expr
166 = expr_alloc_and(prop->visible.expr,
167 dup_expr);
168 }
169 }
170
171 current_entry->prompt = prop;
172 }
173 prop->text = prompt;
174
175 return prop;
176}

199void menu_add_option(int token, char *arg)
200{
201 switch (token) {
202 case T_OPT_MODULES:
203 if (modules_sym)
204 zconf_error("symbol '%s' redefines option 'modules'"
205 " already defined by symbol '%s'",
206 current_entry->sym->name,
207 modules_sym->name
208 );
209 modules_sym = current_entry->sym;
210 break;
211 case T_OPT_DEFCONFIG_LIST:
212 if (!sym_defconfig_list)
213 sym_defconfig_list = current_entry->sym;
214 else if (sym_defconfig_list != current_entry->sym)
215 zconf_error("trying to redefine defconfig symbol");
216 break;
217 case T_OPT_ENV:
218 prop_add_env(arg);
219 break;
220 case T_OPT_ALLNOCONFIG_Y:
221 current_entry->sym->flags |= SYMBOL_ALLNOCONFIG_Y;
222 break;
223 }
224}

226static int menu_validate_number(struct symbol *sym, struct symbol *sym2)
227{
228 return sym2->type == S_INT || sym2->type == S_HEX ||
229 (sym2->type == S_UNKNOWN && sym_string_valid(sym, sym2->name));

232static void sym_check_prop(struct symbol *sym)
233{
234 struct property *prop;
235 struct symbol *sym2;
236 char *use;
237
238 for (prop = sym->prop; prop; prop = prop->next) {
239 switch (prop->type) {
240 case P_DEFAULT:
241 if ((sym->type == S_STRING || sym->type == S_INT || sym->type == S_HEX) &&
242 prop->expr->type != E_SYMBOL)
243 prop_warn(prop,
244 "default for config symbol '%s'"
245 " must be a single symbol", sym->name);
246 if (prop->expr->type != E_SYMBOL)
247 break;
248 sym2 = prop_get_symbol(prop);
249 if (sym->type == S_HEX || sym->type == S_INT) {
250 if (!menu_validate_number(sym, sym2))
251 prop_warn(prop,
252 "'%s': number is invalid",
253 sym->name);
254 }
255 break;
256 case P_SELECT:
257 case P_IMPLY:
258 use = prop->type == P_SELECT ? "select" : "imply";
259 sym2 = prop_get_symbol(prop);
260 if (sym->type != S_BOOLEAN && sym->type != S_TRISTATE)
261 prop_warn(prop,
262 "config symbol '%s' uses %s, but is "
263 "not boolean or tristate", sym->name, use);
264 else if (sym2->type != S_UNKNOWN &&
265 sym2->type != S_BOOLEAN &&
266 sym2->type != S_TRISTATE)
267 prop_warn(prop,
268 "'%s' has wrong type. '%s' only "
269 "accept arguments of boolean and "
270 "tristate type", sym2->name, use);
271 break;
272 case P_RANGE:
273 if (sym->type != S_INT && sym->type != S_HEX)
274 prop_warn(prop, "range is only allowed "
275 "for int or hex symbols");
276 if (!menu_validate_number(sym, prop->expr->left.sym) ||
277 !menu_validate_number(sym, prop->expr->right.sym))
278 prop_warn(prop, "range is invalid");
279 break;
280 default:
281 ;
282 }
283 }

286void menu_finalize(struct menu *parent)
287{
288 struct menu *menu, *last_menu;
289 struct symbol *sym;
290 struct property *prop;
291 struct expr *parentdep, *basedep, *dep, *dep2, **ep;
292
293 sym = parent->sym;
294 if (parent->list) {
295 if (sym && sym_is_choice(sym)) {
296 if (sym->type == S_UNKNOWN) {
297 /* find the first choice value to find out choice type */
298 current_entry = parent;
299 for (menu = parent->list; menu; menu = menu->next) {
300 if (menu->sym && menu->sym->type != S_UNKNOWN) {
301 menu_set_type(menu->sym->type);
302 break;
303 }
304 }
305 }
306 /* set the type of the remaining choice values */
307 for (menu = parent->list; menu; menu = menu->next) {
308 current_entry = menu;
309 if (menu->sym && menu->sym->type == S_UNKNOWN)
310 menu_set_type(sym->type);
311 }
312 parentdep = expr_alloc_symbol(sym);
313 } else if (parent->prompt)
314 parentdep = parent->prompt->visible.expr;
315 else
316 parentdep = parent->dep;
317
318 for (menu = parent->list; menu; menu = menu->next) {
319 basedep = expr_transform(menu->dep);
320 basedep = expr_alloc_and(expr_copy(parentdep), basedep);
321 basedep = expr_eliminate_dups(basedep);
322 menu->dep = basedep;
323 if (menu->sym)
324 prop = menu->sym->prop;
325 else
326 prop = menu->prompt;
327 for (; prop; prop = prop->next) {
328 if (prop->menu != menu)
329 continue;
330 dep = expr_transform(prop->visible.expr);
331 dep = expr_alloc_and(expr_copy(basedep), dep);
332 dep = expr_eliminate_dups(dep);
333 if (menu->sym && menu->sym->type != S_TRISTATE)
334 dep = expr_trans_bool(dep);
335 prop->visible.expr = dep;
336 if (prop->type == P_SELECT) {
337 struct symbol *es = prop_get_symbol(prop);
338 es->rev_dep.expr = expr_alloc_or(es->rev_dep.expr,
339 expr_alloc_and(expr_alloc_symbol(menu->sym), expr_copy(dep)));
340 } else if (prop->type == P_IMPLY) {
341 struct symbol *es = prop_get_symbol(prop);
342 es->implied.expr = expr_alloc_or(es->implied.expr,
343 expr_alloc_and(expr_alloc_symbol(menu->sym), expr_copy(dep)));
344 }
345 }
346 }
347 for (menu = parent->list; menu; menu = menu->next)
348 menu_finalize(menu);
349 } else if (sym) {
350 basedep = parent->prompt ? parent->prompt->visible.expr : NULL;
351 basedep = expr_trans_compare(basedep, E_UNEQUAL, &symbol_no);
352 basedep = expr_eliminate_dups(expr_transform(basedep));
353 last_menu = NULL;
354 for (menu = parent->next; menu; menu = menu->next) {
355 dep = menu->prompt ? menu->prompt->visible.expr : menu->dep;
356 if (!expr_contains_symbol(dep, sym))
357 break;
358 if (expr_depends_symbol(dep, sym))
359 goto next;
360 dep = expr_trans_compare(dep, E_UNEQUAL, &symbol_no);
361 dep = expr_eliminate_dups(expr_transform(dep));
362 dep2 = expr_copy(basedep);
363 expr_eliminate_eq(&dep, &dep2);
364 expr_free(dep);
365 if (!expr_is_yes(dep2)) {
366 expr_free(dep2);
367 break;
368 }
369 expr_free(dep2);
370 next:
371 menu_finalize(menu);
372 menu->parent = parent;
373 last_menu = menu;
374 }
375 expr_free(basedep);
376 if (last_menu) {
377 parent->list = parent->next;
378 parent->next = last_menu->next;
379 last_menu->next = NULL;
380 }
381
382 sym->dir_dep.expr = expr_alloc_or(sym->dir_dep.expr, parent->dep);
383 }
384 for (menu = parent->list; menu; menu = menu->next) {
385 if (sym && sym_is_choice(sym) &&
386 menu->sym && !sym_is_choice_value(menu->sym)) {
387 current_entry = menu;
388 menu->sym->flags |= SYMBOL_CHOICEVAL;
389 if (!menu->prompt)
390 menu_warn(menu, "choice value must have a prompt");
391 for (prop = menu->sym->prop; prop; prop = prop->next) {
392 if (prop->type == P_DEFAULT)
393 prop_warn(prop, "defaults for choice "
394 "values not supported");
395 if (prop->menu == menu)
396 continue;
397 if (prop->type == P_PROMPT &&
398 prop->menu->parent->sym != sym)
399 prop_warn(prop, "choice value used outside its choice group");
400 }
401 /* Non-tristate choice values of tristate choices must
402 * depend on the choice being set to Y. The choice
403 * values' dependencies were propagated to their
404 * properties above, so the change here must be re-
405 * propagated.
406 */
407 if (sym->type == S_TRISTATE && menu->sym->type != S_TRISTATE) {
408 basedep = expr_alloc_comp(E_EQUAL, sym, &symbol_yes);
409 menu->dep = expr_alloc_and(basedep, menu->dep);
410 for (prop = menu->sym->prop; prop; prop = prop->next) {
411 if (prop->menu != menu)
412 continue;
413 prop->visible.expr = expr_alloc_and(expr_copy(basedep),
414 prop->visible.expr);
415 }
416 }
417 menu_add_symbol(P_CHOICE, sym, NULL);
418 prop = sym_get_choice_prop(sym);
419 for (ep = &prop->expr; *ep; ep = &(*ep)->left.expr)
420 ;
421 *ep = expr_alloc_one(E_LIST, NULL);
422 (*ep)->right.sym = menu->sym;
423 }
424 if (menu->list && (!menu->prompt || !menu->prompt->text)) {
425 for (last_menu = menu->list; ; last_menu = last_menu->next) {
426 last_menu->parent = parent;
427 if (!last_menu->next)
428 break;
429 }
430 last_menu->next = menu->next;
431 menu->next = menu->list;
432 menu->list = NULL;
433 }
434 }
435
436 if (sym && !(sym->flags & SYMBOL_WARNED)) {
437 if (sym->type == S_UNKNOWN)
438 menu_warn(parent, "config symbol defined without type");
439
440 if (sym_is_choice(sym) && !parent->prompt)
441 menu_warn(parent, "choice must have a prompt");
442
443 /* Check properties connected to this symbol */
444 sym_check_prop(sym);
445 sym->flags |= SYMBOL_WARNED;
446 }
447
448 if (sym && !sym_is_optional(sym) && parent->prompt) {
449 sym->rev_dep.expr = expr_alloc_or(sym->rev_dep.expr,
450 expr_alloc_and(parent->prompt->visible.expr,
451 expr_alloc_symbol(&symbol_mod)));
452 }
453}

462/*
463 * Determine if a menu is empty.
464 * A menu is considered empty if it contains no or only
465 * invisible entries.
466 */
467bool menu_is_empty(struct menu *menu)
468{
469 struct menu *child;
470
471 for (child = menu->list; child; child = child->next) {
472 if (menu_is_visible(child))
473 return(false);
474 }
475 return(true);
476}
477
478bool menu_is_visible(struct menu *menu)
479{
480 struct menu *child;
481 struct symbol *sym;
482 tristate visible;
483
484 if (!menu->prompt)
485 return false;
486
487 if (menu->visibility) {
488 if (expr_calc_value(menu->visibility) == no)
489 return false;
490 }
491
492 sym = menu->sym;
493 if (sym) {
494 sym_calc_value(sym);
495 visible = menu->prompt->visible.tri;
496 } else
497 visible = menu->prompt->visible.tri = expr_calc_value(menu->prompt->visible.expr);
498
499 if (visible != no)
500 return true;
501
502 if (!sym || sym_get_tristate_value(menu->sym) == no)
503 return false;
504
505 for (child = menu->list; child; child = child->next) {
506 if (menu_is_visible(child)) {
507 if (sym)
508 sym->flags |= SYMBOL_DEF_USER;
509 return true;
510 }
511 }
512
513 return false;
514}

555static void get_prompt_str(struct gstr *r, struct property *prop,
556 struct list_head *head)
557{
558 int i, j;
559 struct menu *submenu[8], *menu, *location = NULL;
560 struct jump_key *jump = NULL;
561
562 str_printf(r, _("Prompt: %s\n"), _(prop->text));
563 menu = prop->menu->parent;
564 for (i = 0; menu != &rootmenu && i < 8; menu = menu->parent) {
565 bool accessible = menu_is_visible(menu);
566
567 submenu[i++] = menu;
568 if (location == NULL && accessible)
569 location = menu;
570 }
571 if (head && location) {
572 jump = xmalloc(sizeof(struct jump_key));
573
574 if (menu_is_visible(prop->menu)) {
575 /*
576 * There is not enough room to put the hint at the
577 * beginning of the "Prompt" line. Put the hint on the
578 * last "Location" line even when it would belong on
579 * the former.
580 */
581 jump->target = prop->menu;
582 } else
583 jump->target = location;
584
585 if (list_empty(head))
586 jump->index = 0;
587 else
588 jump->index = list_entry(head->prev, struct jump_key,
589 entries)->index + 1;
590
591 list_add_tail(&jump->entries, head);
592 }
593
594 if (i > 0) {
595 str_printf(r, _(" Location:\n"));
596 for (j = 4; --i >= 0; j += 2) {
597 menu = submenu[i];
598 if (jump && menu == location)
599 jump->offset = strlen(r->s);
600 str_printf(r, "%*c-> %s", j, ' ',
601 _(menu_get_prompt(menu)));
602 if (menu->sym) {
603 str_printf(r, " (%s [=%s])", menu->sym->name ?
604 menu->sym->name : _("<choice>"),
605 sym_get_string_value(menu->sym));
606 }
607 str_append(r, "\n");
608 }
609 }
610}

624static void get_symbol_props_str(struct gstr *r, struct symbol *sym,
625 enum prop_type tok, const char *prefix)
626{
627 bool hit = false;
628 struct property *prop;
629
630 for_all_properties(sym, prop, tok) {
631 if (!hit) {
632 str_append(r, prefix);
633 hit = true;
634 } else
635 str_printf(r, " && ");
636 expr_gstr_print(prop->expr, r);
637 }
638 if (hit)
639 str_append(r, "\n");
640}

642/*
643 * head is optional and may be NULL
644 */
645static void get_symbol_str(struct gstr *r, struct symbol *sym,
646 struct list_head *head)
647{
648 struct property *prop;
649
650 if (sym && sym->name) {
651 str_printf(r, "Symbol: %s [=%s]\n", sym->name,
652 sym_get_string_value(sym));
653 str_printf(r, "Type : %s\n", sym_type_name(sym->type));
654 if (sym->type == S_INT || sym->type == S_HEX) {
655 prop = sym_get_range_prop(sym);
656 if (prop) {
657 str_printf(r, "Range : ");
658 expr_gstr_print(prop->expr, r);
659 str_append(r, "\n");
660 }
661 }
662 }
663 for_all_prompts(sym, prop)
664 get_prompt_str(r, prop, head);
665
666 prop = get_symbol_prop(sym);
667 if (prop) {
668 str_printf(r, _(" Defined at %s:%d\n"), prop->menu->file->name,
669 prop->menu->lineno);
670 if (!expr_is_yes(prop->visible.expr)) {
671 str_append(r, _(" Depends on: "));
672 expr_gstr_print(prop->visible.expr, r);
673 str_append(r, "\n");
674 }
675 }
676
677 get_symbol_props_str(r, sym, P_SELECT, _(" Selects: "));
678 if (sym->rev_dep.expr) {
679 str_append(r, _(" Selected by: "));
680 expr_gstr_print(sym->rev_dep.expr, r);
681 str_append(r, "\n");
682 }
683
684 get_symbol_props_str(r, sym, P_IMPLY, _(" Implies: "));
685 if (sym->implied.expr) {
686 str_append(r, _(" Implied by: "));
687 expr_gstr_print(sym->implied.expr, r);
688 str_append(r, "\n");
689 }
690
691 str_append(r, "\n\n");
692}

694struct gstr get_relations_str(struct symbol **sym_arr, struct list_head *head)
695{
696 struct symbol *sym;
697 struct gstr res = str_new();
698 int i;
699
700 for (i = 0; sym_arr && (sym = sym_arr[i]); i++)
701 get_symbol_str(&res, sym, head);
702 if (!i)
703 str_append(&res, _("No matches found.\n"));
704 return res;
705}

708void menu_get_ext_help(struct menu *menu, struct gstr *help)
709{
710 struct symbol *sym = menu->sym;
711 const char *help_text = nohelp_text;
712
713 if (menu_has_help(menu)) {
714 if (sym->name)
715 str_printf(help, "%s%s:\n\n", CONFIG_, sym->name);
716 help_text = menu_get_help(menu);
717 }
718 str_printf(help, "%s\n", _(help_text));
719 if (sym)
720 get_symbol_str(help, sym, NULL);
721}
