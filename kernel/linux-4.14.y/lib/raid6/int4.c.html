24 * This is the C data type to use

27/* Change this from BITS_PER_LONG if there is something better... */
28#if BITS_PER_LONG == 64
29# define NBYTES(x) ((x) * 0x0101010101010101UL)
30# define NSIZE 8
31# define NSHIFT 3
32# define NSTRING "64"
33typedef u64 unative_t;
34#else
35# define NBYTES(x) ((x) * 0x01010101U)
36# define NSIZE 4
37# define NSHIFT 2
38# define NSTRING "32"
39typedef u32 unative_t;
40#endif

44/*
45 * IA-64 wants insane amounts of unrolling. On other architectures that
46 * is just a waste of space.
47 */

51/*
52 * These sub-operations are separate inlines since they can sometimes be
53 * specially optimized using architecture-specific hacks.
54 */
55
56/*
57 * The SHLBYTE() operation shifts each byte left by 1, *not*
58 * rolling over into the next byte
59 */
60static inline __attribute_const__ unative_t SHLBYTE(unative_t v)
61{
62 unative_t vv;
63
64 vv = (v << 1) & NBYTES(0xfe);
65 return vv;
66}
67
68/*
69 * The MASK() operation returns 0xFF in any byte for which the high
70 * bit is 1, 0x00 for any byte for which the high bit is 0.
71 */
72static inline __attribute_const__ unative_t MASK(unative_t v)
73{
74 unative_t vv;
75
76 vv = v & NBYTES(0x80);
77 vv = (vv << 1) - (vv >> 7); /* Overflow on the top bit is OK */
78 return vv;
79}

82static void raid6_int4_gen_syndrome(int disks, size_t bytes, void **ptrs)
83{
84 u8 **dptr = (u8 **)ptrs;
85 u8 *p, *q;
86 int d, z, z0;
87
88 unative_t wd0, wq0, wp0, w10, w20;
89 unative_t wd1, wq1, wp1, w11, w21;
90 unative_t wd2, wq2, wp2, w12, w22;
91 unative_t wd3, wq3, wp3, w13, w23;
92
93 z0 = disks - 3; /* Highest data disk */
94 p = dptr[z0+1]; /* XOR parity */
95 q = dptr[z0+2]; /* RS syndrome */
96
97 for ( d = 0 ; d < bytes ; d += NSIZE*4 ) {
98 wq0 = wp0 = *(unative_t *)&dptr[z0][d+0*NSIZE];
99 wq1 = wp1 = *(unative_t *)&dptr[z0][d+1*NSIZE];
100 wq2 = wp2 = *(unative_t *)&dptr[z0][d+2*NSIZE];
101 wq3 = wp3 = *(unative_t *)&dptr[z0][d+3*NSIZE];
102 for ( z = z0-1 ; z >= 0 ; z-- ) {
103 wd0 = *(unative_t *)&dptr[z][d+0*NSIZE];
104 wd1 = *(unative_t *)&dptr[z][d+1*NSIZE];
105 wd2 = *(unative_t *)&dptr[z][d+2*NSIZE];
106 wd3 = *(unative_t *)&dptr[z][d+3*NSIZE];
107 wp0 ^= wd0;
108 wp1 ^= wd1;
109 wp2 ^= wd2;
110 wp3 ^= wd3;
111 w20 = MASK(wq0);
112 w21 = MASK(wq1);
113 w22 = MASK(wq2);
114 w23 = MASK(wq3);
115 w10 = SHLBYTE(wq0);
116 w11 = SHLBYTE(wq1);
117 w12 = SHLBYTE(wq2);
118 w13 = SHLBYTE(wq3);
119 w20 &= NBYTES(0x1d);
120 w21 &= NBYTES(0x1d);
121 w22 &= NBYTES(0x1d);
122 w23 &= NBYTES(0x1d);
123 w10 ^= w20;
124 w11 ^= w21;
125 w12 ^= w22;
126 w13 ^= w23;
127 wq0 = w10 ^ wd0;
128 wq1 = w11 ^ wd1;
129 wq2 = w12 ^ wd2;
130 wq3 = w13 ^ wd3;
131 }
132 *(unative_t *)&p[d+NSIZE*0] = wp0;
133 *(unative_t *)&p[d+NSIZE*1] = wp1;
134 *(unative_t *)&p[d+NSIZE*2] = wp2;
135 *(unative_t *)&p[d+NSIZE*3] = wp3;
136 *(unative_t *)&q[d+NSIZE*0] = wq0;
137 *(unative_t *)&q[d+NSIZE*1] = wq1;
138 *(unative_t *)&q[d+NSIZE*2] = wq2;
139 *(unative_t *)&q[d+NSIZE*3] = wq3;
140 }
141}

143static void raid6_int4_xor_syndrome(int disks, int start, int stop,
144 size_t bytes, void **ptrs)
145{
146 u8 **dptr = (u8 **)ptrs;
147 u8 *p, *q;
148 int d, z, z0;
149
150 unative_t wd0, wq0, wp0, w10, w20;
151 unative_t wd1, wq1, wp1, w11, w21;
152 unative_t wd2, wq2, wp2, w12, w22;
153 unative_t wd3, wq3, wp3, w13, w23;
154
155 z0 = stop; /* P/Q right side optimization */
156 p = dptr[disks-2]; /* XOR parity */
157 q = dptr[disks-1]; /* RS syndrome */
158
159 for ( d = 0 ; d < bytes ; d += NSIZE*4 ) {
160 /* P/Q data pages */
161 wq0 = wp0 = *(unative_t *)&dptr[z0][d+0*NSIZE];
162 wq1 = wp1 = *(unative_t *)&dptr[z0][d+1*NSIZE];
163 wq2 = wp2 = *(unative_t *)&dptr[z0][d+2*NSIZE];
164 wq3 = wp3 = *(unative_t *)&dptr[z0][d+3*NSIZE];
165 for ( z = z0-1 ; z >= start ; z-- ) {
166 wd0 = *(unative_t *)&dptr[z][d+0*NSIZE];
167 wd1 = *(unative_t *)&dptr[z][d+1*NSIZE];
168 wd2 = *(unative_t *)&dptr[z][d+2*NSIZE];
169 wd3 = *(unative_t *)&dptr[z][d+3*NSIZE];
170 wp0 ^= wd0;
171 wp1 ^= wd1;
172 wp2 ^= wd2;
173 wp3 ^= wd3;
174 w20 = MASK(wq0);
175 w21 = MASK(wq1);
176 w22 = MASK(wq2);
177 w23 = MASK(wq3);
178 w10 = SHLBYTE(wq0);
179 w11 = SHLBYTE(wq1);
180 w12 = SHLBYTE(wq2);
181 w13 = SHLBYTE(wq3);
182 w20 &= NBYTES(0x1d);
183 w21 &= NBYTES(0x1d);
184 w22 &= NBYTES(0x1d);
185 w23 &= NBYTES(0x1d);
186 w10 ^= w20;
187 w11 ^= w21;
188 w12 ^= w22;
189 w13 ^= w23;
190 wq0 = w10 ^ wd0;
191 wq1 = w11 ^ wd1;
192 wq2 = w12 ^ wd2;
193 wq3 = w13 ^ wd3;
194 }
195 /* P/Q left side optimization */
196 for ( z = start-1 ; z >= 0 ; z-- ) {
197 w20 = MASK(wq0);
198 w21 = MASK(wq1);
199 w22 = MASK(wq2);
200 w23 = MASK(wq3);
201 w10 = SHLBYTE(wq0);
202 w11 = SHLBYTE(wq1);
203 w12 = SHLBYTE(wq2);
204 w13 = SHLBYTE(wq3);
205 w20 &= NBYTES(0x1d);
206 w21 &= NBYTES(0x1d);
207 w22 &= NBYTES(0x1d);
208 w23 &= NBYTES(0x1d);
209 wq0 = w10 ^ w20;
210 wq1 = w11 ^ w21;
211 wq2 = w12 ^ w22;
212 wq3 = w13 ^ w23;
213 }
214 *(unative_t *)&p[d+NSIZE*0] ^= wp0;
215 *(unative_t *)&p[d+NSIZE*1] ^= wp1;
216 *(unative_t *)&p[d+NSIZE*2] ^= wp2;
217 *(unative_t *)&p[d+NSIZE*3] ^= wp3;
218 *(unative_t *)&q[d+NSIZE*0] ^= wq0;
219 *(unative_t *)&q[d+NSIZE*1] ^= wq1;
220 *(unative_t *)&q[d+NSIZE*2] ^= wq2;
221 *(unative_t *)&q[d+NSIZE*3] ^= wq3;
222 }
