27/* Change this from BITS_PER_LONG if there is something better... */
28#if BITS_PER_LONG == 64
29# define NBYTES(x) ((x) * 0x0101010101010101UL)
30# define NSIZE 8
31# define NSHIFT 3
32# define NSTRING "64"
33typedef u64 unative_t;
34#else
35# define NBYTES(x) ((x) * 0x01010101U)
36# define NSIZE 4
37# define NSHIFT 2
38# define NSTRING "32"
39typedef u32 unative_t;
40#endif

51/*
52 * These sub-operations are separate inlines since they can sometimes be
53 * specially optimized using architecture-specific hacks.
54 */
55
56/*
57 * The SHLBYTE() operation shifts each byte left by 1, *not*
58 * rolling over into the next byte
59 */
60static inline __attribute_const__ unative_t SHLBYTE(unative_t v)
61{
62 unative_t vv;
63
64 vv = (v << 1) & NBYTES(0xfe);
65 return vv;
66}
67
68/*
69 * The MASK() operation returns 0xFF in any byte for which the high
70 * bit is 1, 0x00 for any byte for which the high bit is 0.
71 */
72static inline __attribute_const__ unative_t MASK(unative_t v)
73{
74 unative_t vv;
75
76 vv = v & NBYTES(0x80);
77 vv = (vv << 1) - (vv >> 7); /* Overflow on the top bit is OK */
78 return vv;
79}

82static void raid6_int2_gen_syndrome(int disks, size_t bytes, void **ptrs)
83{
84 u8 **dptr = (u8 **)ptrs;
85 u8 *p, *q;
86 int d, z, z0;
87
88 unative_t wd0, wq0, wp0, w10, w20;
89 unative_t wd1, wq1, wp1, w11, w21;
90
91 z0 = disks - 3; /* Highest data disk */
92 p = dptr[z0+1]; /* XOR parity */
93 q = dptr[z0+2]; /* RS syndrome */
94
95 for ( d = 0 ; d < bytes ; d += NSIZE*2 ) {
96 wq0 = wp0 = *(unative_t *)&dptr[z0][d+0*NSIZE];
97 wq1 = wp1 = *(unative_t *)&dptr[z0][d+1*NSIZE];
98 for ( z = z0-1 ; z >= 0 ; z-- ) {
99 wd0 = *(unative_t *)&dptr[z][d+0*NSIZE];
100 wd1 = *(unative_t *)&dptr[z][d+1*NSIZE];
101 wp0 ^= wd0;
102 wp1 ^= wd1;
103 w20 = MASK(wq0);
104 w21 = MASK(wq1);
105 w10 = SHLBYTE(wq0);
106 w11 = SHLBYTE(wq1);
107 w20 &= NBYTES(0x1d);
108 w21 &= NBYTES(0x1d);
109 w10 ^= w20;
110 w11 ^= w21;
111 wq0 = w10 ^ wd0;
112 wq1 = w11 ^ wd1;
113 }
114 *(unative_t *)&p[d+NSIZE*0] = wp0;
115 *(unative_t *)&p[d+NSIZE*1] = wp1;
116 *(unative_t *)&q[d+NSIZE*0] = wq0;
117 *(unative_t *)&q[d+NSIZE*1] = wq1;
118 }
119}

121static void raid6_int2_xor_syndrome(int disks, int start, int stop,
122 size_t bytes, void **ptrs)
123{
124 u8 **dptr = (u8 **)ptrs;
125 u8 *p, *q;
126 int d, z, z0;
127
128 unative_t wd0, wq0, wp0, w10, w20;
129 unative_t wd1, wq1, wp1, w11, w21;
130
131 z0 = stop; /* P/Q right side optimization */
132 p = dptr[disks-2]; /* XOR parity */
133 q = dptr[disks-1]; /* RS syndrome */
134
135 for ( d = 0 ; d < bytes ; d += NSIZE*2 ) {
136 /* P/Q data pages */
137 wq0 = wp0 = *(unative_t *)&dptr[z0][d+0*NSIZE];
138 wq1 = wp1 = *(unative_t *)&dptr[z0][d+1*NSIZE];
139 for ( z = z0-1 ; z >= start ; z-- ) {
140 wd0 = *(unative_t *)&dptr[z][d+0*NSIZE];
141 wd1 = *(unative_t *)&dptr[z][d+1*NSIZE];
142 wp0 ^= wd0;
143 wp1 ^= wd1;
144 w20 = MASK(wq0);
145 w21 = MASK(wq1);
146 w10 = SHLBYTE(wq0);
147 w11 = SHLBYTE(wq1);
148 w20 &= NBYTES(0x1d);
149 w21 &= NBYTES(0x1d);
150 w10 ^= w20;
151 w11 ^= w21;
152 wq0 = w10 ^ wd0;
153 wq1 = w11 ^ wd1;
154 }
155 /* P/Q left side optimization */
156 for ( z = start-1 ; z >= 0 ; z-- ) {
157 w20 = MASK(wq0);
158 w21 = MASK(wq1);
159 w10 = SHLBYTE(wq0);
160 w11 = SHLBYTE(wq1);
161 w20 &= NBYTES(0x1d);
162 w21 &= NBYTES(0x1d);
163 wq0 = w10 ^ w20;
164 wq1 = w11 ^ w21;
165 }
166 *(unative_t *)&p[d+NSIZE*0] ^= wp0;
167 *(unative_t *)&p[d+NSIZE*1] ^= wp1;
168 *(unative_t *)&q[d+NSIZE*0] ^= wq0;
169 *(unative_t *)&q[d+NSIZE*1] ^= wq1;
170 }
