49/**
50 * crc16 - compute the CRC-16 for the data buffer
51 * @crc: previous CRC value
52 * @buffer: data pointer
53 * @len: number of bytes in the buffer
54 *
55 * Returns the updated CRC value.
56 */
57u16 crc16(u16 crc, u8 const *buffer, size_t len)
58{
59 while (len--)
60 crc = crc16_byte(crc, *buffer++);
61 return crc;
62}
