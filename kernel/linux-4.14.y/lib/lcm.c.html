6/* Lowest common multiple */
7unsigned long lcm(unsigned long a, unsigned long b)
8{
9 if (a && b)
10 return (a / gcd(a, b)) * b;
11 else
12 return 0;
13}

16unsigned long lcm_not_zero(unsigned long a, unsigned long b)
17{
18 unsigned long l = lcm(a, b);
19
20 if (l)

22

24}
