22/**
23 * strstarts - does @str start with @prefix?
24 * @str: string to examine
25 * @prefix: prefix to look for.
26 */
27static inline bool strstarts(const char *str, const char *prefix)
28{
29 return strncmp(str, prefix, strlen(prefix)) == 0;
30}
