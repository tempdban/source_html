15#define DOUBLEFAULT_STACKSIZE (1024)
16static unsigned long doublefault_stack[DOUBLEFAULT_STACKSIZE];
17#define STACK_START (unsigned long)(doublefault_stack+DOUBLEFAULT_STACKSIZE)
18
19#define ptr_ok(x) ((x) > PAGE_OFFSET && (x) < PAGE_OFFSET + MAXMEM)
20
21static void doublefault_fn(void)
22{
23 struct desc_ptr gdt_desc = {0, 0};
24 unsigned long gdt, tss;
25
26 native_store_gdt(&gdt_desc);
27 gdt = gdt_desc.address;
28
29 printk(KERN_EMERG "PANIC: double fault, gdt at %08lx [%d bytes]\n", gdt, gdt_desc.size);
30
31 if (ptr_ok(gdt)) {
32 gdt += GDT_ENTRY_TSS << 3;
33 tss = get_desc_base((struct desc_struct *)gdt);
34 printk(KERN_EMERG "double fault, tss at %08lx\n", tss);
35
36 if (ptr_ok(tss)) {
37 struct x86_hw_tss *t = (struct x86_hw_tss *)tss;
38
39 printk(KERN_EMERG "eip = %08lx, esp = %08lx\n",
40 t->ip, t->sp);
41
42 printk(KERN_EMERG "eax = %08lx, ebx = %08lx, ecx = %08lx, edx = %08lx\n",
43 t->ax, t->bx, t->cx, t->dx);
44 printk(KERN_EMERG "esi = %08lx, edi = %08lx\n",
45 t->si, t->di);
46 }
47 }
48
49 for (;;)
50 cpu_relax();
51}
52
53struct x86_hw_tss doublefault_tss __cacheline_aligned = {
54 .sp0 = STACK_START,
55 .ss0 = __KERNEL_DS,
56 .ldt = 0,
57 .io_bitmap_base = INVALID_IO_BITMAP_OFFSET,
58
59 .ip = (unsigned long) doublefault_fn,
60 /* 0x2 bit is always set */
61 .flags = X86_EFLAGS_SF | 0x2,
62 .sp = STACK_START,
63 .es = __USER_DS,
64 .cs = __KERNEL_CS,
65 .ss = __KERNEL_DS,
66 .ds = __USER_DS,
67 .fs = __KERNEL_PERCPU,
68
69 .__cr3 = __pa_nodebug(swapper_pg_dir),
70};
