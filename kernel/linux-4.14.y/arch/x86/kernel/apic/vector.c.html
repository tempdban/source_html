53static struct apic_chip_data *apic_chip_data(struct irq_data *irq_data)
54{
55 if (!irq_data)
56 return NULL;
57
58 while (irq_data->parent_data)
59 irq_data = irq_data->parent_data;
60
61 return irq_data->chip_data;
62}

77static struct apic_chip_data *alloc_apic_chip_data(int node)
78{
79 struct apic_chip_data *data;
80
81 data = kzalloc_node(sizeof(*data), GFP_KERNEL, node);
82 if (!data)
83 return NULL;
84 if (!zalloc_cpumask_var_node(&data->domain, GFP_KERNEL, node))
85 goto out_data;
86 if (!zalloc_cpumask_var_node(&data->old_domain, GFP_KERNEL, node))
87 goto out_domain;
88 return data;
89out_domain:
90 free_cpumask_var(data->domain);
91out_data:
92 kfree(data);
93 return NULL;
94}

96static void free_apic_chip_data(struct apic_chip_data *data)
97{
98 if (data) {
99 free_cpumask_var(data->domain);
100 free_cpumask_var(data->old_domain);
101 kfree(data);
102 }

105static int __assign_irq_vector(int irq, struct apic_chip_data *d,
106 const struct cpumask *mask,
107 struct irq_data *irqdata)
108{
109 /*
110 * NOTE! The local APIC isn't very good at handling
111 * multiple interrupts at the same interrupt level.
112 * As the interrupt level is determined by taking the
113 * vector number and shifting that right by 4, we
114 * want to spread these out a bit so that they don't
115 * all fall in the same interrupt level.
116 *
117 * Also, we've got to be careful not to trash gate
118 * 0x80, because int 0x80 is hm, kind of importantish. ;)
119 */
120 static int current_vector = FIRST_EXTERNAL_VECTOR + VECTOR_OFFSET_START;
121 static int current_offset = VECTOR_OFFSET_START % 16;
122 int cpu, vector;
123
124 /*
125 * If there is still a move in progress or the previous move has not
126 * been cleaned up completely, tell the caller to come back later.
127 */
128 if (d->move_in_progress ||
129 cpumask_intersects(d->old_domain, cpu_online_mask))
130 return -EBUSY;
131
132 /* Only try and allocate irqs on cpus that are present */
133 cpumask_clear(d->old_domain);
134 cpumask_clear(searched_cpumask);
135 cpu = cpumask_first_and(mask, cpu_online_mask);
136 while (cpu < nr_cpu_ids) {
137 int new_cpu, offset;
138
139 /* Get the possible target cpus for @mask/@cpu from the apic */
140 apic->vector_allocation_domain(cpu, vector_cpumask, mask);
141
142 /*
143 * Clear the offline cpus from @vector_cpumask for searching
144 * and verify whether the result overlaps with @mask. If true,
145 * then the call to apic->cpu_mask_to_apicid() will
146 * succeed as well. If not, no point in trying to find a
147 * vector in this mask.
148 */
149 cpumask_and(vector_searchmask, vector_cpumask, cpu_online_mask);
150 if (!cpumask_intersects(vector_searchmask, mask))
151 goto next_cpu;
152
153 if (cpumask_subset(vector_cpumask, d->domain)) {
154 if (cpumask_equal(vector_cpumask, d->domain))
155 goto success;
156 /*
157 * Mark the cpus which are not longer in the mask for
158 * cleanup.
159 */
160 cpumask_andnot(d->old_domain, d->domain, vector_cpumask);
161 vector = d->cfg.vector;
162 goto update;
163 }
164
165 vector = current_vector;
166 offset = current_offset;
167next:
168 vector += 16;
169 if (vector >= FIRST_SYSTEM_VECTOR) {
170 offset = (offset + 1) % 16;
171 vector = FIRST_EXTERNAL_VECTOR + offset;
172 }
173
174 /* If the search wrapped around, try the next cpu */
175 if (unlikely(current_vector == vector))
176 goto next_cpu;
177
178 if (test_bit(vector, used_vectors))
179 goto next;
180
181 for_each_cpu(new_cpu, vector_searchmask) {
182 if (!IS_ERR_OR_NULL(per_cpu(vector_irq, new_cpu)[vector]))
183 goto next;
184 }
185 /* Found one! */
186 current_vector = vector;
187 current_offset = offset;
188 /* Schedule the old vector for cleanup on all cpus */
189 if (d->cfg.vector)
190 cpumask_copy(d->old_domain, d->domain);
191 for_each_cpu(new_cpu, vector_searchmask)
192 per_cpu(vector_irq, new_cpu)[vector] = irq_to_desc(irq);
193 goto update;
194
195next_cpu:
196 /*
197 * We exclude the current @vector_cpumask from the requested
198 * @mask and try again with the next online cpu in the
199 * result. We cannot modify @mask, so we use @vector_cpumask
200 * as a temporary buffer here as it will be reassigned when
201 * calling apic->vector_allocation_domain() above.
202 */
203 cpumask_or(searched_cpumask, searched_cpumask, vector_cpumask);
204 cpumask_andnot(vector_cpumask, mask, searched_cpumask);
205 cpu = cpumask_first_and(vector_cpumask, cpu_online_mask);
206 continue;
207 }
208 return -ENOSPC;
209
210update:
211 /*
212 * Exclude offline cpus from the cleanup mask and set the
213 * move_in_progress flag when the result is not empty.
214 */
215 cpumask_and(d->old_domain, d->old_domain, cpu_online_mask);
216 d->move_in_progress = !cpumask_empty(d->old_domain);
217 d->cfg.old_vector = d->move_in_progress ? d->cfg.vector : 0;
218 d->cfg.vector = vector;
219 cpumask_copy(d->domain, vector_cpumask);
220success:
221 /*
222 * Cache destination APIC IDs into cfg->dest_apicid. This cannot fail
223 * as we already established, that mask & d->domain & cpu_online_mask
224 * is not empty.
225 *
226 * vector_searchmask is a subset of d->domain and has the offline
227 * cpus masked out.
228 */
229 cpumask_and(vector_searchmask, vector_searchmask, mask);
230 BUG_ON(apic->cpu_mask_to_apicid(vector_searchmask, irqdata,
231 &d->cfg.dest_apicid));
232 return 0;
233}

235static int assign_irq_vector(int irq, struct apic_chip_data *data,
236 const struct cpumask *mask,
237 struct irq_data *irqdata)
238{
239 int err;
240 unsigned long flags;
241
242 raw_spin_lock_irqsave(&vector_lock, flags);
243 err = __assign_irq_vector(irq, data, mask, irqdata);
244 raw_spin_unlock_irqrestore(&vector_lock, flags);
245 return err;
246}

248static int assign_irq_vector_policy(int irq, int node,
249 struct apic_chip_data *data,
250 struct irq_alloc_info *info,
251 struct irq_data *irqdata)
252{
253 if (info && info->mask)
254 return assign_irq_vector(irq, data, info->mask, irqdata);
255 if (node != NUMA_NO_NODE &&
256 assign_irq_vector(irq, data, cpumask_of_node(node), irqdata) == 0)
257 return 0;
258 return assign_irq_vector(irq, data, apic->target_cpus(), irqdata);
259}

261static void clear_irq_vector(int irq, struct apic_chip_data *data)
262{
263 struct irq_desc *desc;
264 int cpu, vector;
265
266 if (!data->cfg.vector)
267 return;
268
269 vector = data->cfg.vector;
270 for_each_cpu_and(cpu, data->domain, cpu_online_mask)
271 per_cpu(vector_irq, cpu)[vector] = VECTOR_UNUSED;
272
273 data->cfg.vector = 0;
274 cpumask_clear(data->domain);
275
276 /*
277 * If move is in progress or the old_domain mask is not empty,
278 * i.e. the cleanup IPI has not been processed yet, we need to remove
279 * the old references to desc from all cpus vector tables.
280 */
281 if (!data->move_in_progress && cpumask_empty(data->old_domain))
282 return;
283
284 desc = irq_to_desc(irq);
285 for_each_cpu_and(cpu, data->old_domain, cpu_online_mask) {
286 for (vector = FIRST_EXTERNAL_VECTOR; vector < NR_VECTORS;
287 vector++) {
288 if (per_cpu(vector_irq, cpu)[vector] != desc)
289 continue;
290 per_cpu(vector_irq, cpu)[vector] = VECTOR_UNUSED;
291 break;
292 }
293 }
294 data->move_in_progress = 0;
295}

312static void x86_vector_free_irqs(struct irq_domain *domain,
313 unsigned int virq, unsigned int nr_irqs)
314{
315 struct apic_chip_data *apic_data;
316 struct irq_data *irq_data;
317 unsigned long flags;
318 int i;
319
320 for (i = 0; i < nr_irqs; i++) {
321 irq_data = irq_domain_get_irq_data(x86_vector_domain, virq + i);
322 if (irq_data && irq_data->chip_data) {
323 raw_spin_lock_irqsave(&vector_lock, flags);
324 clear_irq_vector(virq + i, irq_data->chip_data);
325 apic_data = irq_data->chip_data;
326 irq_domain_reset_irq_data(irq_data);
327 raw_spin_unlock_irqrestore(&vector_lock, flags);
328 free_apic_chip_data(apic_data);
329#ifdef CONFIG_X86_IO_APIC
330 if (virq + i < nr_legacy_irqs())
331 legacy_irq_data[virq + i] = NULL;
332#endif
333 }
334 }
335}

337static int x86_vector_alloc_irqs(struct irq_domain *domain, unsigned int virq,
338 unsigned int nr_irqs, void *arg)
339{
340 struct irq_alloc_info *info = arg;
341 struct apic_chip_data *data;
342 struct irq_data *irq_data;
343 int i, err, node;
344
345 if (disable_apic)
346 return -ENXIO;
347
348 /* Currently vector allocator can't guarantee contiguous allocations */
349 if ((info->flags & X86_IRQ_ALLOC_CONTIGUOUS_VECTORS) && nr_irqs > 1)
350 return -ENOSYS;
351
352 for (i = 0; i < nr_irqs; i++) {
353 irq_data = irq_domain_get_irq_data(domain, virq + i);
354 BUG_ON(!irq_data);
355 node = irq_data_get_node(irq_data);
356#ifdef CONFIG_X86_IO_APIC
357 if (virq + i < nr_legacy_irqs() && legacy_irq_data[virq + i])
358 data = legacy_irq_data[virq + i];
359 else
360#endif
361 data = alloc_apic_chip_data(node);
362 if (!data) {
363 err = -ENOMEM;
364 goto error;
365 }
366
367 irq_data->chip = &lapic_controller;
368 irq_data->chip_data = data;
369 irq_data->hwirq = virq + i;
370 err = assign_irq_vector_policy(virq + i, node, data, info,
371 irq_data);
372 if (err) {
373 irq_data->chip_data = NULL;
374 free_apic_chip_data(data);
375 goto error;
376 }
377 /*
378 * If the apic destination mode is physical, then the
379 * effective affinity is restricted to a single target
380 * CPU. Mark the interrupt accordingly.
381 */
382 if (!apic->irq_dest_mode)
383 irqd_set_single_target(irq_data);
384 }
385
386 return 0;
387
388error:
389 x86_vector_free_irqs(domain, virq, i);
390 return err;
391}

398int __init arch_probe_nr_irqs(void)
399{
400 int nr;
401
402 if (nr_irqs > (NR_VECTORS * nr_cpu_ids))
403 nr_irqs = NR_VECTORS * nr_cpu_ids;
404
405 nr = (gsi_top + nr_legacy_irqs()) + 8 * nr_cpu_ids;
406#if defined(CONFIG_PCI_MSI) || defined(CONFIG_HT_IRQ)
407 /*
408 * for MSI and HT dyn irq
409 */
410 if (gsi_top <= NR_IRQS_LEGACY)
411 nr += 8 * nr_cpu_ids;
412 else
413 nr += gsi_top * 16;
414#endif
415 if (nr < nr_irqs)
416 nr_irqs = nr;
417
418 /*
419 * We don't know if PIC is present at this point so we need to do
420 * probe() to get the right number of legacy IRQs.
421 */
422 return legacy_pic->probe();
423}

442 }

472/* Initialize vector_irq on a new cpu */
473static void __setup_vector_irq(int cpu)
474{
475 struct apic_chip_data *data;
476 struct irq_desc *desc;
477 int irq, vector;
478
479 /* Mark the inuse vectors */
480 for_each_irq_desc(irq, desc) {
481 struct irq_data *idata = irq_desc_get_irq_data(desc);
482
483 data = apic_chip_data(idata);
484 if (!data || !cpumask_test_cpu(cpu, data->domain))
485 continue;
486 vector = data->cfg.vector;
487 per_cpu(vector_irq, cpu)[vector] = desc;
488 }
489 /* Mark the free vectors */
490 for (vector = 0; vector < NR_VECTORS; ++vector) {
491 desc = per_cpu(vector_irq, cpu)[vector];
492 if (IS_ERR_OR_NULL(desc))
493 continue;
494
495 data = apic_chip_data(irq_desc_get_irq_data(desc));
496 if (!cpumask_test_cpu(cpu, data->domain))
497 per_cpu(vector_irq, cpu)[vector] = VECTOR_UNUSED;
498 }
499}

501/*
502 * Setup the vector to irq mappings. Must be called with vector_lock held.
503 */
504void setup_vector_irq(int cpu)
505{
506 int irq;
507
508 lockdep_assert_held(&vector_lock);
509 /*
510 * On most of the platforms, legacy PIC delivers the interrupts on the
511 * boot cpu. But there are certain platforms where PIC interrupts are
512 * delivered to multiple cpu's. If the legacy IRQ is handled by the
513 * legacy PIC, for the new cpu that is coming online, setup the static
514 * legacy vector to irq mapping:
515 */
516 for (irq = 0; irq < nr_legacy_irqs(); irq++)
517 per_cpu(vector_irq, cpu)[ISA_IRQ_VECTOR(irq)] = irq_to_desc(irq);
518
519 __setup_vector_irq(cpu);
520}

522static int apic_retrigger_irq(struct irq_data *irq_data)
523{
524 struct apic_chip_data *data = apic_chip_data(irq_data);
525 unsigned long flags;
526 int cpu;
527
528 raw_spin_lock_irqsave(&vector_lock, flags);
529 cpu = cpumask_first_and(data->domain, cpu_online_mask);
530 apic->send_IPI_mask(cpumask_of(cpu), data->cfg.vector);
531 raw_spin_unlock_irqrestore(&vector_lock, flags);
532
533 return 1;
534}

543static int apic_set_affinity(struct irq_data *irq_data,
544 const struct cpumask *dest, bool force)
545{
546 struct apic_chip_data *data = irq_data->chip_data;
547 int err, irq = irq_data->irq;
548
549 if (!IS_ENABLED(CONFIG_SMP))
550 return -EPERM;
551
552 if (!cpumask_intersects(dest, cpu_online_mask))
553 return -EINVAL;
554
555 err = assign_irq_vector(irq, data, dest, irq_data);
556 return err ? err : IRQ_SET_MASK_OK;
557}

568{
569 raw_spin_lock(&vector_lock);

586asmlinkage __visible void __irq_entry smp_irq_move_cleanup_interrupt(void)
587{
588 unsigned vector, me;
589
590 entering_ack_irq();
591
592 /* Prevent vectors vanishing under us */
593 raw_spin_lock(&vector_lock);
594
595 me = smp_processor_id();
596 for (vector = FIRST_EXTERNAL_VECTOR; vector < NR_VECTORS; vector++) {
597 struct apic_chip_data *data;
598 struct irq_desc *desc;
599 unsigned int irr;
600
601 retry:
602 desc = __this_cpu_read(vector_irq[vector]);
603 if (IS_ERR_OR_NULL(desc))
604 continue;
605
606 if (!raw_spin_trylock(&desc->lock)) {
607 raw_spin_unlock(&vector_lock);
608 cpu_relax();
609 raw_spin_lock(&vector_lock);
610 goto retry;
611 }
612
613 data = apic_chip_data(irq_desc_get_irq_data(desc));
614 if (!data)
615 goto unlock;
616
617 /*
618 * Nothing to cleanup if irq migration is in progress
619 * or this cpu is not set in the cleanup mask.
620 */
621 if (data->move_in_progress ||
622 !cpumask_test_cpu(me, data->old_domain))
623 goto unlock;
624
625 /*
626 * We have two cases to handle here:
627 * 1) vector is unchanged but the target mask got reduced
628 * 2) vector and the target mask has changed
629 *
630 * #1 is obvious, but in #2 we have two vectors with the same
631 * irq descriptor: the old and the new vector. So we need to
632 * make sure that we only cleanup the old vector. The new
633 * vector has the current @vector number in the config and
634 * this cpu is part of the target mask. We better leave that
635 * one alone.
636 */
637 if (vector == data->cfg.vector &&
638 cpumask_test_cpu(me, data->domain))
639 goto unlock;
640
641 irr = apic_read(APIC_IRR + (vector / 32 * 0x10));
642 /*
643 * Check if the vector that needs to be cleanedup is
644 * registered at the cpu's IRR. If so, then this is not
645 * the best time to clean it up. Lets clean it up in the
646 * next attempt by sending another IRQ_MOVE_CLEANUP_VECTOR
647 * to myself.
648 */
649 if (irr & (1 << (vector % 32))) {
650 apic->send_IPI_self(IRQ_MOVE_CLEANUP_VECTOR);
651 goto unlock;
652 }
653 __this_cpu_write(vector_irq[vector], VECTOR_UNUSED);
654 cpumask_clear_cpu(me, data->old_domain);
655unlock:
656 raw_spin_unlock(&desc->lock);
657 }
658
659 raw_spin_unlock(&vector_lock);
660
661 exiting_irq();
662}

664static void __irq_complete_move(struct irq_cfg *cfg, unsigned vector)
665{
666 unsigned me;
667 struct apic_chip_data *data;
668
669 data = container_of(cfg, struct apic_chip_data, cfg);
670 if (likely(!data->move_in_progress))
671 return;
672
673 me = smp_processor_id();
674 if (vector == data->cfg.vector && cpumask_test_cpu(me, data->domain))
675 __send_cleanup_vector(data);
676}

683/*
684 * Called from fixup_irqs() with @desc->lock held and interrupts disabled.
685 */
686void irq_force_complete_move(struct irq_desc *desc)
687{
688 struct irq_data *irqdata;
689 struct apic_chip_data *data;
690 struct irq_cfg *cfg;
691 unsigned int cpu;
692
693 /*
694 * The function is called for all descriptors regardless of which
695 * irqdomain they belong to. For example if an IRQ is provided by
696 * an irq_chip as part of a GPIO driver, the chip data for that
697 * descriptor is specific to the irq_chip in question.
698 *
699 * Check first that the chip_data is what we expect
700 * (apic_chip_data) before touching it any further.
701 */
702 irqdata = irq_domain_get_irq_data(x86_vector_domain,
703 irq_desc_get_irq(desc));
704 if (!irqdata)
705 return;
706
707 data = apic_chip_data(irqdata);
708 cfg = data ? &data->cfg : NULL;
709
710 if (!cfg)
711 return;
712
713 /*
714 * This is tricky. If the cleanup of @data->old_domain has not been
715 * done yet, then the following setaffinity call will fail with
716 * -EBUSY. This can leave the interrupt in a stale state.
717 *
718 * All CPUs are stuck in stop machine with interrupts disabled so
719 * calling __irq_complete_move() would be completely pointless.
720 */
721 raw_spin_lock(&vector_lock);
722 /*
723 * Clean out all offline cpus (including the outgoing one) from the
724 * old_domain mask.
725 */
726 cpumask_and(data->old_domain, data->old_domain, cpu_online_mask);
727
728 /*
729 * If move_in_progress is cleared and the old_domain mask is empty,
730 * then there is nothing to cleanup. fixup_irqs() will take care of
731 * the stale vectors on the outgoing cpu.
732 */
733 if (!data->move_in_progress && cpumask_empty(data->old_domain)) {
734 raw_spin_unlock(&vector_lock);
735 return;
736 }
737
738 /*
739 * 1) The interrupt is in move_in_progress state. That means that we
740 * have not seen an interrupt since the io_apic was reprogrammed to
741 * the new vector.
742 *
743 * 2) The interrupt has fired on the new vector, but the cleanup IPIs
744 * have not been processed yet.
745 */
746 if (data->move_in_progress) {
747 /*
748 * In theory there is a race:
749 *
750 * set_ioapic(new_vector) <-- Interrupt is raised before update
751 * is effective, i.e. it's raised on
752 * the old vector.
753 *
754 * So if the target cpu cannot handle that interrupt before
755 * the old vector is cleaned up, we get a spurious interrupt
756 * and in the worst case the ioapic irq line becomes stale.
757 *
758 * But in case of cpu hotplug this should be a non issue
759 * because if the affinity update happens right before all
760 * cpus rendevouz in stop machine, there is no way that the
761 * interrupt can be blocked on the target cpu because all cpus
762 * loops first with interrupts enabled in stop machine, so the
763 * old vector is not yet cleaned up when the interrupt fires.
764 *
765 * So the only way to run into this issue is if the delivery
766 * of the interrupt on the apic/system bus would be delayed
767 * beyond the point where the target cpu disables interrupts
768 * in stop machine. I doubt that it can happen, but at least
769 * there is a theroretical chance. Virtualization might be
770 * able to expose this, but AFAICT the IOAPIC emulation is not
771 * as stupid as the real hardware.
772 *
773 * Anyway, there is nothing we can do about that at this point
774 * w/o refactoring the whole fixup_irq() business completely.
775 * We print at least the irq number and the old vector number,
776 * so we have the necessary information when a problem in that
777 * area arises.
778 */
779 pr_warn("IRQ fixup: irq %d move in progress, old vector %d\n",
780 irqdata->irq, cfg->old_vector);
781 }
782 /*
783 * If old_domain is not empty, then other cpus still have the irq
784 * descriptor set in their vector array. Clean it up.
785 */
786 for_each_cpu(cpu, data->old_domain)
787 per_cpu(vector_irq, cpu)[cfg->old_vector] = VECTOR_UNUSED;
788
789 /* Cleanup the left overs of the (half finished) move */
790 cpumask_clear(data->old_domain);
791 data->move_in_progress = 0;
792 raw_spin_unlock(&vector_lock);
793}

808static void __init print_local_APIC(void *dummy)
809{
810 unsigned int i, v, ver, maxlvt;
811 u64 icr;
812
813 pr_debug("printing local APIC contents on CPU#%d/%d:\n",
814 smp_processor_id(), hard_smp_processor_id());
815 v = apic_read(APIC_ID);
816 pr_info("... APIC ID: %08x (%01x)\n", v, read_apic_id());
817 v = apic_read(APIC_LVR);
818 pr_info("... APIC VERSION: %08x\n", v);
819 ver = GET_APIC_VERSION(v);
820 maxlvt = lapic_get_maxlvt();
821
822 v = apic_read(APIC_TASKPRI);
823 pr_debug("... APIC TASKPRI: %08x (%02x)\n", v, v & APIC_TPRI_MASK);
824
825 /* !82489DX */
826 if (APIC_INTEGRATED(ver)) {
827 if (!APIC_XAPIC(ver)) {
828 v = apic_read(APIC_ARBPRI);
829 pr_debug("... APIC ARBPRI: %08x (%02x)\n",
830 v, v & APIC_ARBPRI_MASK);
831 }
832 v = apic_read(APIC_PROCPRI);
833 pr_debug("... APIC PROCPRI: %08x\n", v);
834 }
835
836 /*
837 * Remote read supported only in the 82489DX and local APIC for
838 * Pentium processors.
839 */
840 if (!APIC_INTEGRATED(ver) || maxlvt == 3) {
841 v = apic_read(APIC_RRR);
842 pr_debug("... APIC RRR: %08x\n", v);
843 }
844
845 v = apic_read(APIC_LDR);
846 pr_debug("... APIC LDR: %08x\n", v);
847 if (!x2apic_enabled()) {
848 v = apic_read(APIC_DFR);
849 pr_debug("... APIC DFR: %08x\n", v);
850 }
851 v = apic_read(APIC_SPIV);
852 pr_debug("... APIC SPIV: %08x\n", v);
853
854 pr_debug("... APIC ISR field:\n");
855 print_APIC_field(APIC_ISR);
856 pr_debug("... APIC TMR field:\n");
857 print_APIC_field(APIC_TMR);
858 pr_debug("... APIC IRR field:\n");
859 print_APIC_field(APIC_IRR);
860
861 /* !82489DX */
862 if (APIC_INTEGRATED(ver)) {
863 /* Due to the Pentium erratum 3AP. */
864 if (maxlvt > 3)
865 apic_write(APIC_ESR, 0);
866
867 v = apic_read(APIC_ESR);
868 pr_debug("... APIC ESR: %08x\n", v);
869 }
870
871 icr = apic_icr_read();
872 pr_debug("... APIC ICR: %08x\n", (u32)icr);
873 pr_debug("... APIC ICR2: %08x\n", (u32)(icr >> 32));
874
875 v = apic_read(APIC_LVTT);
876 pr_debug("... APIC LVTT: %08x\n", v);
877
878 if (maxlvt > 3) {
879 /* PC is LVT#4. */
880 v = apic_read(APIC_LVTPC);
881 pr_debug("... APIC LVTPC: %08x\n", v);
882 }
883 v = apic_read(APIC_LVT0);
884 pr_debug("... APIC LVT0: %08x\n", v);
885 v = apic_read(APIC_LVT1);
886 pr_debug("... APIC LVT1: %08x\n", v);
887
888 if (maxlvt > 2) {
889 /* ERR is LVT#3. */
890 v = apic_read(APIC_LVTERR);
891 pr_debug("... APIC LVTERR: %08x\n", v);
892 }
893
894 v = apic_read(APIC_TMICT);
895 pr_debug("... APIC TMICT: %08x\n", v);
896 v = apic_read(APIC_TMCCT);
897 pr_debug("... APIC TMCCT: %08x\n", v);
898 v = apic_read(APIC_TDCR);
899 pr_debug("... APIC TDCR: %08x\n", v);
900
901 if (boot_cpu_has(X86_FEATURE_EXTAPIC)) {
902 v = apic_read(APIC_EFEAT);
903 maxlvt = (v >> 16) & 0xff;
904 pr_debug("... APIC EFEAT: %08x\n", v);
905 v = apic_read(APIC_ECTRL);
906 pr_debug("... APIC ECTRL: %08x\n", v);
907 for (i = 0; i < maxlvt; i++) {
908 v = apic_read(APIC_EILVTn(i));
909 pr_debug("... APIC EILVT%d: %08x\n", i, v);
910 }
911 }
912 pr_cont("\n");
913}

915static void __init print_local_APICs(int maxcpu)
916{
917 int cpu;
918
919 if (!maxcpu)
920 return;

923 for_each_online_cpu(cpu) {

931static void __init print_PIC(void)
932{
933 unsigned int v;
934 unsigned long flags;
935
936 if (!nr_legacy_irqs())
937 return;
938
939 pr_debug("\nprinting PIC contents\n");
940
941 raw_spin_lock_irqsave(&i8259A_lock, flags);
942
943 v = inb(0xa1) << 8 | inb(0x21);
944 pr_debug("... PIC IMR: %04x\n", v);
945
946 v = inb(0xa0) << 8 | inb(0x20);
947 pr_debug("... PIC IRR: %04x\n", v);
948
949 outb(0x0b, 0xa0);
950 outb(0x0b, 0x20);
951 v = inb(0xa0) << 8 | inb(0x20);
952 outb(0x0a, 0xa0);
953 outb(0x0a, 0x20);
954
955 raw_spin_unlock_irqrestore(&i8259A_lock, flags);
956
957 pr_debug("... PIC ISR: %04x\n", v);
958
959 v = inb(0x4d1) << 8 | inb(0x4d0);
960 pr_debug("... PIC ELCR: %04x\n", v);
961}
