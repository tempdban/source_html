32int audit_classify_arch(int arch)
33{
34#ifdef CONFIG_IA32_EMULATION
35 if (arch == AUDIT_ARCH_I386)
36 return 1;
37#endif
38 return 0;
39}
40
41int audit_classify_syscall(int abi, unsigned syscall)
42{
43#ifdef CONFIG_IA32_EMULATION
44 extern int ia32_classify_syscall(unsigned);
45 if (abi == AUDIT_ARCH_I386)
46 return ia32_classify_syscall(syscall);
47#endif
48 switch(syscall) {
49 case __NR_open:
50 return 2;
51 case __NR_openat:
52 return 3;
53 case __NR_execve:
54 case __NR_execveat:
55 return 5;
56 default:
57 return 0;
58 }
59}

61static int __init audit_classes_init(void)
62{
63#ifdef CONFIG_IA32_EMULATION
64 extern __u32 ia32_dir_class[];
65 extern __u32 ia32_write_class[];
66 extern __u32 ia32_read_class[];
67 extern __u32 ia32_chattr_class[];
68 extern __u32 ia32_signal_class[];
69 audit_register_class(AUDIT_CLASS_WRITE_32, ia32_write_class);
70 audit_register_class(AUDIT_CLASS_READ_32, ia32_read_class);
71 audit_register_class(AUDIT_CLASS_DIR_WRITE_32, ia32_dir_class);
72 audit_register_class(AUDIT_CLASS_CHATTR_32, ia32_chattr_class);
73 audit_register_class(AUDIT_CLASS_SIGNAL_32, ia32_signal_class);
74#endif
75 audit_register_class(AUDIT_CLASS_WRITE, write_class);
76 audit_register_class(AUDIT_CLASS_READ, read_class);
77 audit_register_class(AUDIT_CLASS_DIR_WRITE, dir_class);
78 audit_register_class(AUDIT_CLASS_CHATTR, chattr_class);
79 audit_register_class(AUDIT_CLASS_SIGNAL, signal_class);
80 return 0;
81}

83__initcall(audit_classes_init);
