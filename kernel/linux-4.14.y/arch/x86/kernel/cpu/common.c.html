225/* Standard macro to see if a specific flag is changeable */
226static inline int flag_is_changeable_p(u32 flag)
227{
228 u32 f1, f2;
229
230 /*
231 * Cyrix and IDT cpus allow disabling of CPUID
232 * so the code below may return different results
233 * when it is executed before and after enabling
234 * the CPUID. Add "volatile" to not allow gcc to
235 * optimize the subsequent calls to this function.
236 */
237 asm volatile ("pushfl \n\t"
238 "pushfl \n\t"
239 "popl %0 \n\t"
240 "movl %0, %1 \n\t"
241 "xorl %2, %0 \n\t"
242 "pushl %0 \n\t"
243 "popfl \n\t"
244 "pushfl \n\t"
245 "popl %0 \n\t"
246 "popfl \n\t"
247
248 : "=&r" (f1), "=&r" (f2)
249 : "ir" (flag));
250
251 return ((f1^f2) & flag) != 0;
252}

380/*
381 * Some CPU features depend on higher CPUID levels, which may not always
382 * be available due to CPUID level capping or broken virtualization
383 * software. Add those features to this table to auto-disable them.
384 */
385struct cpuid_dependent_feature {
386 u32 feature;
387 u32 level;
388};
389
390static const struct cpuid_dependent_feature
391cpuid_dependent_features[] = {
392 { X86_FEATURE_MWAIT, 0x00000005 },
393 { X86_FEATURE_DCA, 0x00000009 },
394 { X86_FEATURE_XSAVE, 0x0000000d },
395 { 0, 0 }
396};
397
398static void filter_cpuid_features(struct cpuinfo_x86 *c, bool warn)
399{
400 const struct cpuid_dependent_feature *df;
401
402 for (df = cpuid_dependent_features; df->feature; df++) {
403
404 if (!cpu_has(c, df->feature))
405 continue;
406 /*
407 * Note: cpuid_level is set to -1 if unavailable, but
408 * extended_extended_level is set to 0 if unavailable
409 * and the legitimate extended levels are all negative
410 * when signed; hence the weird messing around with
411 * signs here...
412 */
413 if (!((s32)df->level < 0 ?
414 (u32)df->level > (u32)c->extended_cpuid_level :
415 (s32)df->level > (s32)c->cpuid_level))
416 continue;
417
418 clear_cpu_cap(c, df->feature);
419 if (!warn)
420 continue;
421
422 pr_warn("CPU: CPU feature " X86_CAP_FMT " disabled, no CPUID level 0x%x\n",
423 x86_cap_flag(df->feature), df->level);
424 }
425}

525static void get_model_name(struct cpuinfo_x86 *c)
526{
527 unsigned int *v;
528 char *p, *q, *s;
529
530 if (c->extended_cpuid_level < 0x80000004)
531 return;
532
533 v = (unsigned int *)c->x86_model_id;
534 cpuid(0x80000002, &v[0], &v[1], &v[2], &v[3]);
535 cpuid(0x80000003, &v[4], &v[5], &v[6], &v[7]);
536 cpuid(0x80000004, &v[8], &v[9], &v[10], &v[11]);
537 c->x86_model_id[48] = 0;
538
539 /* Trim whitespace */
540 p = q = s = &c->x86_model_id[0];
541
542 while (*p == ' ')
543 p++;
544
545 while (*p) {
546 /* Note the last non-whitespace index */
547 if (!isspace(*p))
548 s = q;
549
550 *q++ = *p++;
551 }
552
553 *(s + 1) = '\0';
554}

556void cpu_detect_cache_sizes(struct cpuinfo_x86 *c)
557{
558 unsigned int n, dummy, ebx, ecx, edx, l2size;
559
560 n = c->extended_cpuid_level;
561
562 if (n >= 0x80000005) {
563 cpuid(0x80000005, &dummy, &ebx, &ecx, &edx);
564 c->x86_cache_size = (ecx>>24) + (edx>>24);
565#ifdef CONFIG_X86_64
566 /* On K8 L1 TLB is inclusive, so don't count it */
567 c->x86_tlbsize = 0;
568#endif
569 }
570
571 if (n < 0x80000006) /* Some chips just has a large L1. */
572 return;
573
574 cpuid(0x80000006, &dummy, &ebx, &ecx, &edx);
575 l2size = ecx >> 16;
576
577#ifdef CONFIG_X86_64
578 c->x86_tlbsize += ((ebx >> 16) & 0xfff) + (ebx & 0xfff);
579#else
580 /* do processor-specific cache resizing */
581 if (this_cpu->legacy_cache_size)
582 l2size = this_cpu->legacy_cache_size(c, l2size);
583
584 /* Allow user to override all this if necessary. */
585 if (cachesize_override != -1)
586 l2size = cachesize_override;
587
588 if (l2size == 0)
589 return; /* Again, no L2 cache is possible */
590#endif
591
592 c->x86_cache_size = l2size;
593}

617void detect_ht(struct cpuinfo_x86 *c)
618{
619#ifdef CONFIG_SMP
620 u32 eax, ebx, ecx, edx;
621 int index_msb, core_bits;
622 static bool printed;
623
624 if (!cpu_has(c, X86_FEATURE_HT))
625 return;
626
627 if (cpu_has(c, X86_FEATURE_CMP_LEGACY))
628 goto out;
629
630 if (cpu_has(c, X86_FEATURE_XTOPOLOGY))
631 return;
632
633 cpuid(1, &eax, &ebx, &ecx, &edx);
634
635 smp_num_siblings = (ebx & 0xff0000) >> 16;
636
637 if (smp_num_siblings == 1) {
638 pr_info_once("CPU0: Hyper-Threading is disabled\n");
639 goto out;
640 }
641
642 if (smp_num_siblings <= 1)
643 goto out;
644
645 index_msb = get_count_order(smp_num_siblings);
646 c->phys_proc_id = apic->phys_pkg_id(c->initial_apicid, index_msb);
647
648 smp_num_siblings = smp_num_siblings / c->x86_max_cores;
649
650 index_msb = get_count_order(smp_num_siblings);
651
652 core_bits = get_count_order(c->x86_max_cores);
653
654 c->cpu_core_id = apic->phys_pkg_id(c->initial_apicid, index_msb) &
655 ((1 << core_bits) - 1);
656
657out:
658 if (!printed && (c->x86_max_cores * smp_num_siblings) > 1) {
659 pr_info("CPU: Physical Processor ID: %d\n",
660 c->phys_proc_id);
661 pr_info("CPU: Processor Core ID: %d\n",
662 c->cpu_core_id);
663 printed = 1;
664 }
665#endif
666}

674 if (!cpu_devs[i])
675 break;
676
677 if (!strcmp(v, cpu_devs[i]->c_ident[0]) ||
678 (cpu_devs[i]->c_ident[1] &&
679 !strcmp(v, cpu_devs[i]->c_ident[1]))) {
680
681 this_cpu = cpu_devs[i];
682 c->x86_vendor = this_cpu->c_x86_vendor;
683 return;
684 }
685 }
686
687 pr_err_once("CPU: vendor_id '%s' unknown, using generic init.\n" \
688 "CPU: Your system may be unstable.\n", v);
689
690 c->x86_vendor = X86_VENDOR_UNKNOWN;
691 this_cpu = &default_cpu;
692}

694void cpu_detect(struct cpuinfo_x86 *c)
695{
696 /* Get vendor name */
697 cpuid(0x00000000, (unsigned int *)&c->cpuid_level,
698 (unsigned int *)&c->x86_vendor_id[0],
699 (unsigned int *)&c->x86_vendor_id[8],
700 (unsigned int *)&c->x86_vendor_id[4]);
701
702 c->x86 = 4;
703 /* Intel-defined flags: level 0x00000001 */
704 if (c->cpuid_level >= 0x00000001) {
705 u32 junk, tfms, cap0, misc;
706
707 cpuid(0x00000001, &tfms, &misc, &junk, &cap0);
708 c->x86 = x86_family(tfms);
709 c->x86_model = x86_model(tfms);
710 c->x86_stepping = x86_stepping(tfms);
711
712 if (cap0 & (1<<19)) {
713 c->x86_clflush_size = ((misc >> 8) & 0xff) * 8;
714 c->x86_cache_alignment = c->x86_clflush_size;
715 }
716 }

729static void init_speculation_control(struct cpuinfo_x86 *c)
730{
731 /*
732 * The Intel SPEC_CTRL CPUID bit implies IBRS and IBPB support,
733 * and they also have a different bit for STIBP support. Also,
734 * a hypervisor might have set the individual AMD bits even on
735 * Intel CPUs, for finer-grained selection of what's available.
736 */
737 if (cpu_has(c, X86_FEATURE_SPEC_CTRL)) {
738 set_cpu_cap(c, X86_FEATURE_IBRS);
739 set_cpu_cap(c, X86_FEATURE_IBPB);
740 set_cpu_cap(c, X86_FEATURE_MSR_SPEC_CTRL);
741 }
742
743 if (cpu_has(c, X86_FEATURE_INTEL_STIBP))
744 set_cpu_cap(c, X86_FEATURE_STIBP);
745
746 if (cpu_has(c, X86_FEATURE_SPEC_CTRL_SSBD) ||
747 cpu_has(c, X86_FEATURE_VIRT_SSBD))
748 set_cpu_cap(c, X86_FEATURE_SSBD);
749
750 if (cpu_has(c, X86_FEATURE_AMD_IBRS)) {
751 set_cpu_cap(c, X86_FEATURE_IBRS);
752 set_cpu_cap(c, X86_FEATURE_MSR_SPEC_CTRL);
753 }
754
755 if (cpu_has(c, X86_FEATURE_AMD_IBPB))
756 set_cpu_cap(c, X86_FEATURE_IBPB);
757
758 if (cpu_has(c, X86_FEATURE_AMD_STIBP)) {
759 set_cpu_cap(c, X86_FEATURE_STIBP);
760 set_cpu_cap(c, X86_FEATURE_MSR_SPEC_CTRL);
761 }

764void get_cpu_cap(struct cpuinfo_x86 *c)
765{
766 u32 eax, ebx, ecx, edx;
767
768 /* Intel-defined flags: level 0x00000001 */
769 if (c->cpuid_level >= 0x00000001) {
770 cpuid(0x00000001, &eax, &ebx, &ecx, &edx);
771
772 c->x86_capability[CPUID_1_ECX] = ecx;
773 c->x86_capability[CPUID_1_EDX] = edx;
774 }
775
776 /* Thermal and Power Management Leaf: level 0x00000006 (eax) */
777 if (c->cpuid_level >= 0x00000006)
778 c->x86_capability[CPUID_6_EAX] = cpuid_eax(0x00000006);
779
780 /* Additional Intel-defined flags: level 0x00000007 */
781 if (c->cpuid_level >= 0x00000007) {
782 cpuid_count(0x00000007, 0, &eax, &ebx, &ecx, &edx);
783 c->x86_capability[CPUID_7_0_EBX] = ebx;
784 c->x86_capability[CPUID_7_ECX] = ecx;
785 c->x86_capability[CPUID_7_EDX] = edx;
786 }
787
788 /* Extended state features: level 0x0000000d */
789 if (c->cpuid_level >= 0x0000000d) {
790 cpuid_count(0x0000000d, 1, &eax, &ebx, &ecx, &edx);
791
792 c->x86_capability[CPUID_D_1_EAX] = eax;
793 }
794
795 /* Additional Intel-defined flags: level 0x0000000F */
796 if (c->cpuid_level >= 0x0000000F) {
797
798 /* QoS sub-leaf, EAX=0Fh, ECX=0 */
799 cpuid_count(0x0000000F, 0, &eax, &ebx, &ecx, &edx);
800 c->x86_capability[CPUID_F_0_EDX] = edx;
801
802 if (cpu_has(c, X86_FEATURE_CQM_LLC)) {
803 /* will be overridden if occupancy monitoring exists */
804 c->x86_cache_max_rmid = ebx;
805
806 /* QoS sub-leaf, EAX=0Fh, ECX=1 */
807 cpuid_count(0x0000000F, 1, &eax, &ebx, &ecx, &edx);
808 c->x86_capability[CPUID_F_1_EDX] = edx;
809
810 if ((cpu_has(c, X86_FEATURE_CQM_OCCUP_LLC)) ||
811 ((cpu_has(c, X86_FEATURE_CQM_MBM_TOTAL)) ||
812 (cpu_has(c, X86_FEATURE_CQM_MBM_LOCAL)))) {
813 c->x86_cache_max_rmid = ecx;
814 c->x86_cache_occ_scale = ebx;
815 }
816 } else {
817 c->x86_cache_max_rmid = -1;
818 c->x86_cache_occ_scale = -1;
819 }
820 }
821
822 /* AMD-defined flags: level 0x80000001 */
823 eax = cpuid_eax(0x80000000);
824 c->extended_cpuid_level = eax;
825
826 if ((eax & 0xffff0000) == 0x80000000) {
827 if (eax >= 0x80000001) {
828 cpuid(0x80000001, &eax, &ebx, &ecx, &edx);
829
830 c->x86_capability[CPUID_8000_0001_ECX] = ecx;
831 c->x86_capability[CPUID_8000_0001_EDX] = edx;
832 }
833 }
834
835 if (c->extended_cpuid_level >= 0x80000007) {
836 cpuid(0x80000007, &eax, &ebx, &ecx, &edx);
837
838 c->x86_capability[CPUID_8000_0007_EBX] = ebx;
839 c->x86_power = edx;
840 }
841
842 if (c->extended_cpuid_level >= 0x80000008) {
843 cpuid(0x80000008, &eax, &ebx, &ecx, &edx);
844
845 c->x86_virt_bits = (eax >> 8) & 0xff;
846 c->x86_phys_bits = eax & 0xff;
847 c->x86_capability[CPUID_8000_0008_EBX] = ebx;
848 }
849#ifdef CONFIG_X86_32
850 else if (cpu_has(c, X86_FEATURE_PAE) || cpu_has(c, X86_FEATURE_PSE36))
851 c->x86_phys_bits = 36;
852#endif
853
854 if (c->extended_cpuid_level >= 0x8000000a)
855 c->x86_capability[CPUID_8000_000A_EDX] = cpuid_edx(0x8000000a);
856
857 init_scattered_cpuid_features(c);
858 init_speculation_control(c);
859
860 /*
861 * Clear/Set all flags overridden by options, after probe.
862 * This needs to happen each time we re-probe, which may happen
863 * several times during CPU initialization.
864 */
865 apply_forced_caps(c);
866}

1074static void detect_null_seg_behavior(struct cpuinfo_x86 *c)
1075{
1076#ifdef CONFIG_X86_64
1077 /*
1078 * Empirically, writing zero to a segment selector on AMD does
1079 * not clear the base, whereas writing zero to a segment
1080 * selector on Intel does clear the base. Intel's behavior
1081 * allows slightly faster context switches in the common case
1082 * where GS is unused by the prev and next threads.
1083 *
1084 * Since neither vendor documents this anywhere that I can see,
1085 * detect it directly instead of hardcoding the choice by
1086 * vendor.
1087 *
1088 * I've designated AMD's behavior as the "bug" because it's
1089 * counterintuitive and less friendly.
1090 */
1091
1092 unsigned long old_base, tmp;
1093 rdmsrl(MSR_FS_BASE, old_base);
1094 wrmsrl(MSR_FS_BASE, 1);
1095 loadsegment(fs, 0);
1096 rdmsrl(MSR_FS_BASE, tmp);
1097 if (tmp != 0)
1098 set_cpu_bug(c, X86_BUG_NULL_SEG);
1099 wrmsrl(MSR_FS_BASE, old_base);
1100#endif
1101}

1103static void generic_identify(struct cpuinfo_x86 *c)
1104{
1105 c->extended_cpuid_level = 0;
1106
1107 if (!have_cpuid_p())
1108 identify_cpu_without_cpuid(c);
1109
1110 /* cyrix could have cpuid enabled via c_identify()*/
1111 if (!have_cpuid_p())
1112 return;
1113
1114 cpu_detect(c);
1115
1116 get_cpu_vendor(c);
1117
1118 get_cpu_cap(c);
1119
1120 if (c->cpuid_level >= 0x00000001) {
1121 c->initial_apicid = (cpuid_ebx(1) >> 24) & 0xFF;
1122#ifdef CONFIG_X86_32
1123# ifdef CONFIG_SMP
1124 c->apicid = apic->phys_pkg_id(c->initial_apicid, 0);
1125# else
1126 c->apicid = c->initial_apicid;
1127# endif
1128#endif
1129 c->phys_proc_id = c->initial_apicid;
1130 }
1131
1132 get_model_name(c); /* Default name */
1133
1134 detect_nopl(c);
1135
1136 detect_null_seg_behavior(c);
1137
1138 /*
1139 * ESPFIX is a strange bug. All real CPUs have it. Paravirt
1140 * systems that run Linux at CPL > 0 may or may not have the
1141 * issue, but, even if they have the issue, there's absolutely
1142 * nothing we can do about it because we can't use the real IRET
1143 * instruction.
1144 *
1145 * NB: For the time being, only 32-bit kernels support
1146 * X86_BUG_ESPFIX as such. 64-bit kernels directly choose
1147 * whether to apply espfix using paravirt hooks. If any
1148 * non-paravirt system ever shows up that does *not* have the
1149 * ESPFIX issue, we can change this.
1150 */
1151#ifdef CONFIG_X86_32
1152# ifdef CONFIG_PARAVIRT
1153 do {
1154 extern void native_iret(void);
1155 if (pv_cpu_ops.iret == native_iret)
1156 set_cpu_bug(c, X86_BUG_ESPFIX);
1157 } while (0);
1158# else
1159 set_cpu_bug(c, X86_BUG_ESPFIX);
1160# endif
1161#endif
1162}

1200 * This does the hard work of actually picking apart the CPU stuff...
1201 */
1202static void identify_cpu(struct cpuinfo_x86 *c)
1203{
1204 int i;
1205
1206 c->loops_per_jiffy = loops_per_jiffy;
1207 c->x86_cache_size = 0;
1208 c->x86_vendor = X86_VENDOR_UNKNOWN;
1209 c->x86_model = c->x86_stepping = 0; /* So far unknown... */
1210 c->x86_vendor_id[0] = '\0'; /* Unset */
1211 c->x86_model_id[0] = '\0'; /* Unset */
1212 c->x86_max_cores = 1;
1213 c->x86_coreid_bits = 0;
1214 c->cu_id = 0xff;
1215#ifdef CONFIG_X86_64
1216 c->x86_clflush_size = 64;
1217 c->x86_phys_bits = 36;
1218 c->x86_virt_bits = 48;
1219#else
1220 c->cpuid_level = -1; /* CPUID not detected */
1221 c->x86_clflush_size = 32;
1222 c->x86_phys_bits = 32;
1223 c->x86_virt_bits = 32;
1224#endif
1225 c->x86_cache_alignment = c->x86_clflush_size;
1226 memset(&c->x86_capability, 0, sizeof c->x86_capability);
1227
1228 generic_identify(c);
1229
1230 if (this_cpu->c_identify)
1231 this_cpu->c_identify(c);
1232
1233 /* Clear/Set all flags overridden by options, after probe */
1234 apply_forced_caps(c);
1235
1236#ifdef CONFIG_X86_64
1237 c->apicid = apic->phys_pkg_id(c->initial_apicid, 0);
1238#endif
1239
1240 /*
1241 * Vendor-specific initialization. In this section we
1242 * canonicalize the feature flags, meaning if there are
1243 * features a certain CPU supports which CPUID doesn't
1244 * tell us, CPUID claiming incorrect flags, or other bugs,
1245 * we handle them here.
1246 *
1247 * At the end of this section, c->x86_capability better
1248 * indicate the features this CPU genuinely supports!
1249 */
1250 if (this_cpu->c_init)
1251 this_cpu->c_init(c);
1252
1253 /* Disable the PN if appropriate */
1254 squash_the_stupid_serial_number(c);
1255
1256 /* Set up SMEP/SMAP */
1257 setup_smep(c);
1258 setup_smap(c);
1259
1260 /*
1261 * The vendor-specific functions might have changed features.
1262 * Now we do "generic changes."
1263 */
1264
1265 /* Filter out anything that depends on CPUID levels we don't have */
1266 filter_cpuid_features(c, true);
1267
1268 /* If the model name is still unset, do table lookup. */
1269 if (!c->x86_model_id[0]) {
1270 const char *p;
1271 p = table_lookup_model(c);
1272 if (p)
1273 strcpy(c->x86_model_id, p);
1274 else
1275 /* Last resort... */
1276 sprintf(c->x86_model_id, "%02x/%02x",
1277 c->x86, c->x86_model);
1278 }
1279
1280#ifdef CONFIG_X86_64
1281 detect_ht(c);

1579/*
1580 * cpu_init() initializes state that is per-CPU. Some data is already
1581 * initialized (naturally) in the bootstrap process, such as the GDT
1582 * and IDT. We reload them nevertheless, this function acts as a
1583 * 'CPU state barrier', nothing should get across.
1584 * A lot of state is already set up in PDA init for 64 bit
1585 */
1586#ifdef CONFIG_X86_64
1587
1588void cpu_init(void)
1589{
1590 struct orig_ist *oist;
1591 struct task_struct *me;
1592 struct tss_struct *t;
1593 unsigned long v;
1594 int cpu = raw_smp_processor_id();
1595 int i;
1596
1597 wait_for_master_cpu(cpu);
1598
1599 /*
1600 * Initialize the CR4 shadow before doing anything that could
1601 * try to read it.
1602 */
1603 cr4_init_shadow();
1604
1605 if (cpu)
1606 load_ucode_ap();
1607
1608 t = &per_cpu(cpu_tss_rw, cpu);
1609 oist = &per_cpu(orig_ist, cpu);
1610
1611#ifdef CONFIG_NUMA
1612 if (this_cpu_read(numa_node) == 0 &&
1613 early_cpu_to_node(cpu) != NUMA_NO_NODE)
1614 set_numa_node(early_cpu_to_node(cpu));
1615#endif
1616
1617 me = current;
1618
1619 pr_debug("Initializing CPU#%d\n", cpu);
1620
1621 cr4_clear_bits(X86_CR4_VME|X86_CR4_PVI|X86_CR4_TSD|X86_CR4_DE);
1622
1623 /*
1624 * Initialize the per-CPU GDT with the boot GDT,
1625 * and set up the GDT descriptor:
1626 */
1627
1628 switch_to_new_gdt(cpu);
1629 loadsegment(fs, 0);
1630
1631 load_current_idt();
1632
1633 memset(me->thread.tls_array, 0, GDT_ENTRY_TLS_ENTRIES * 8);
1634 syscall_init();
1635
1636 wrmsrl(MSR_FS_BASE, 0);
1637 wrmsrl(MSR_KERNEL_GS_BASE, 0);
1638 barrier();
1639
1640 x86_configure_nx();
1641 x2apic_setup();
1642
1643 /*
1644 * set up and load the per-CPU TSS
1645 */
1646 if (!oist->ist[0]) {
1647 char *estacks = get_cpu_entry_area(cpu)->exception_stacks;
1648
1649 for (v = 0; v < N_EXCEPTION_STACKS; v++) {
1650 estacks += exception_stack_sizes[v];
1651 oist->ist[v] = t->x86_tss.ist[v] =
1652 (unsigned long)estacks;
1653 if (v == DEBUG_STACK-1)
1654 per_cpu(debug_stack_addr, cpu) = (unsigned long)estacks;
1655 }
1656 }
1657
1658 t->x86_tss.io_bitmap_base = IO_BITMAP_OFFSET;
1659
1660 /*
1661 * <= is required because the CPU will access up to
1662 * 8 bits beyond the end of the IO permission bitmap.
1663 */
1664 for (i = 0; i <= IO_BITMAP_LONGS; i++)
1665 t->io_bitmap[i] = ~0UL;
1666
1667 mmgrab(&init_mm);
1668 me->active_mm = &init_mm;
1669 BUG_ON(me->mm);
1670 initialize_tlbstate_and_flush();
1671 enter_lazy_tlb(&init_mm, me);
1672
1673 /*
1674 * Initialize the TSS. sp0 points to the entry trampoline stack
1675 * regardless of what task is running.
1676 */
1677 set_tss_desc(cpu, &get_cpu_entry_area(cpu)->tss.x86_tss);
1678 load_TR_desc();
1679 load_sp0((unsigned long)(cpu_entry_stack(cpu) + 1));
1680
1681 load_mm_ldt(&init_mm);
1682
1683 clear_all_debug_regs();
1684 dbg_restore_debug_regs();
1685
1686 fpu__init_cpu();
1687
1688 if (is_uv_system())
1689 uv_cpu_init();
1690
1691 load_fixmap_gdt(cpu);
1692}
