23#include <asm/processor.h>
24#include <asm/archrandom.h>
25#include <asm/sections.h>
26
27static int __init x86_rdrand_setup(char *s)
28{
29 setup_clear_cpu_cap(X86_FEATURE_RDRAND);
30 setup_clear_cpu_cap(X86_FEATURE_RDSEED);
31 return 1;
32}
33__setup("nordrand", x86_rdrand_setup);
34
35/*
36 * RDRAND has Built-In-Self-Test (BIST) that runs on every invocation.
37 * Run the instruction a few times as a sanity check.
38 * If it fails, it is simple to disable RDRAND here.
39 */
40#define SANITY_CHECK_LOOPS 8
41
42#ifdef CONFIG_ARCH_RANDOM
43void x86_init_rdrand(struct cpuinfo_x86 *c)
44{
45 unsigned long tmp;
46 int i;
47
48 if (!cpu_has(c, X86_FEATURE_RDRAND))
49 return;
50
51 for (i = 0; i < SANITY_CHECK_LOOPS; i++) {
52 if (!rdrand_long(&tmp)) {
53 clear_cpu_cap(c, X86_FEATURE_RDRAND);
54 pr_warn_once("rdrand: disabled\n");
55 return;
56 }
57 }
58}
59#endif
