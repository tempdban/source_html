2 * Routines to identify additional cpu features that are scattered in
3 * cpuid space.

12struct cpuid_bit {
13 u16 feature;
14 u8 reg;
15 u8 bit;
16 u32 level;
17 u32 sub_leaf;
18};
19
20/* Please keep the leaf sorted by cpuid_bit.level for faster search. */
21static const struct cpuid_bit cpuid_bits[] = {
22 { X86_FEATURE_APERFMPERF, CPUID_ECX, 0, 0x00000006, 0 },
23 { X86_FEATURE_EPB, CPUID_ECX, 3, 0x00000006, 0 },
24 { X86_FEATURE_CAT_L3, CPUID_EBX, 1, 0x00000010, 0 },
25 { X86_FEATURE_CAT_L2, CPUID_EBX, 2, 0x00000010, 0 },
26 { X86_FEATURE_CDP_L3, CPUID_ECX, 2, 0x00000010, 1 },
27 { X86_FEATURE_MBA, CPUID_EBX, 3, 0x00000010, 0 },
28 { X86_FEATURE_HW_PSTATE, CPUID_EDX, 7, 0x80000007, 0 },
29 { X86_FEATURE_CPB, CPUID_EDX, 9, 0x80000007, 0 },
30 { X86_FEATURE_PROC_FEEDBACK, CPUID_EDX, 11, 0x80000007, 0 },
31 { X86_FEATURE_SME, CPUID_EAX, 0, 0x8000001f, 0 },
32 { 0, 0, 0, 0, 0 }
33};

35void init_scattered_cpuid_features(struct cpuinfo_x86 *c)
36{
37 u32 max_level;
38 u32 regs[4];
39 const struct cpuid_bit *cb;
40
41 for (cb = cpuid_bits; cb->feature; cb++) {
42
43 /* Verify that the level is valid */
44 max_level = cpuid_eax(cb->level & 0xffff0000);
45 if (max_level < cb->level ||
46 max_level > (cb->level | 0xffff))
47 continue;
48
49 cpuid_count(cb->level, cb->sub_leaf, &regs[CPUID_EAX],
50 &regs[CPUID_EBX], &regs[CPUID_ECX],
51 &regs[CPUID_EDX]);
52
53 if (regs[cb->reg] & (1 << cb->bit))
54 set_cpu_cap(c, cb->feature);
55 }
56}
57
58u32 get_scattered_cpuid_leaf(unsigned int level, unsigned int sub_leaf,
59 enum cpuid_regs_idx reg)
60{
61 const struct cpuid_bit *cb;
62 u32 cpuid_val = 0;
63
64 for (cb = cpuid_bits; cb->feature; cb++) {
65
66 if (level > cb->level)
67 continue;
68
69 if (level < cb->level)
70 break;
71
72 if (reg == cb->reg && sub_leaf == cb->sub_leaf) {
73 if (cpu_has(&boot_cpu_data, cb->feature))
74 cpuid_val |= BIT(cb->bit);
75 }
76 }
77
78 return cpuid_val;
79}
80EXPORT_SYMBOL_GPL(get_scattered_cpuid_leaf);
