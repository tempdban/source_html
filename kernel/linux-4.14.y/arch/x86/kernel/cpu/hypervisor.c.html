29extern const struct hypervisor_x86 x86_hyper_vmware;
30extern const struct hypervisor_x86 x86_hyper_ms_hyperv;
31extern const struct hypervisor_x86 x86_hyper_xen_pv;
32extern const struct hypervisor_x86 x86_hyper_xen_hvm;
33extern const struct hypervisor_x86 x86_hyper_kvm;
34
35static const __initconst struct hypervisor_x86 * const hypervisors[] =
36{
37#ifdef CONFIG_XEN_PV
38 &x86_hyper_xen_pv,
39#endif
40#ifdef CONFIG_XEN_PVHVM
41 &x86_hyper_xen_hvm,
42#endif
43 &x86_hyper_vmware,
44 &x86_hyper_ms_hyperv,
45#ifdef CONFIG_KVM_GUEST
46 &x86_hyper_kvm,
47#endif
48};
49
50enum x86_hypervisor_type x86_hyper_type;
51EXPORT_SYMBOL(x86_hyper_type);
52
53static inline const struct hypervisor_x86 * __init
54detect_hypervisor_vendor(void)
55{
56 const struct hypervisor_x86 *h = NULL, * const *p;
57 uint32_t pri, max_pri = 0;
58
59 for (p = hypervisors; p < hypervisors + ARRAY_SIZE(hypervisors); p++) {
60 pri = (*p)->detect();
61 if (pri > max_pri) {
62 max_pri = pri;
63 h = *p;
64 }
65 }
66
67 if (h)
68 pr_info("Hypervisor detected: %s\n", h->name);
69
70 return h;
71}
72
73static void __init copy_array(const void *src, void *target, unsigned int size)
74{
75 unsigned int i, n = size / sizeof(void *);
76 const void * const *from = (const void * const *)src;
77 const void **to = (const void **)target;
78
79 for (i = 0; i < n; i++)
80 if (from[i])
81 to[i] = from[i];
82}
83
84void __init init_hypervisor_platform(void)
85{
86 const struct hypervisor_x86 *h;
87
88 h = detect_hypervisor_vendor();
89
90 if (!h)
91 return;
92
93 copy_array(&h->init, &x86_init.hyper, sizeof(h->init));
94 copy_array(&h->runtime, &x86_platform.hyper, sizeof(h->runtime));
95
96 x86_hyper_type = h->type;
97 x86_init.hyper.init_platform();
98}
