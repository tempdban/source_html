29int ia32_classify_syscall(unsigned syscall)
30{
31 switch (syscall) {
32 case __NR_open:
33 return 2;
34 case __NR_openat:
35 return 3;
36 case __NR_socketcall:
37 return 4;
38 case __NR_execve:
39 case __NR_execveat:
40 return 5;
41 default:
42 return 1;
43 }
44}
