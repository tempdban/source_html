2/*
3 * Count the digits of @val including a possible sign.
4 *
5 * (Typed on and submitted from hpa's mobile phone.)
6 */
7int num_digits(int val)
8{
9 int m = 10;
10 int d = 1;
11
12 if (val < 0) {
13 d++;
14 val = -val;
15 }
16
17 while (val >= m) {
18 m *= 10;
19 d++;
20 }
21 return d;
22}
