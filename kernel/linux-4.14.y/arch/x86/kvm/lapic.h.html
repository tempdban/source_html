19struct kvm_timer {

22 ktime_t target_expiration;

31struct kvm_lapic {
32 unsigned long base_address;
33 struct kvm_io_device dev;
34 struct kvm_timer lapic_timer;
35 u32 divide_count;
36 struct kvm_vcpu *vcpu;
37 bool sw_enabled;
38 bool irr_pending;
39 bool lvt0_in_nmi_mode;
40 /* Number of bits set in ISR. */
41 s16 isr_count;
42 /* The highest vector set in ISR; if -1 - invalid, must scan ISR. */
43 int highest_isr_cache;
44 /**
45 * APIC register page. The layout matches the register layout seen by
46 * the guest 1:1, because it is accessed by the vmx microcode.
47 * Note: Only one register, the TPR, is used by the microcode.
48 */
49 void *regs;
50 gpa_t vapic_addr;
51 struct gfn_to_hva_cache vapic_cache;
52 unsigned long pending_events;
53 unsigned int sipi_vector;
54};

119#define VEC_POS(v) ((v) & (32 - 1))
120#define REG_POS(v) (((v) >> 5) << 4)
121
122static inline void kvm_lapic_set_vector(int vec, void *bitmap)
123{
124 set_bit(VEC_POS(vec), (bitmap) + REG_POS(vec));
125}
126
127static inline void kvm_lapic_set_irr(int vec, struct kvm_lapic *apic)
128{
129 kvm_lapic_set_vector(vec, apic->regs + APIC_IRR);
130 /*
131 * irr_pending must be true if any interrupt is pending; set it after
132 * APIC_IRR to avoid race with apic_clear_irr
133 */
134 apic->irr_pending = true;
135}

147extern struct static_key kvm_no_apic_vcpu;

149static inline bool lapic_in_kernel(struct kvm_vcpu *vcpu)

151 if (static_key_false(&kvm_no_apic_vcpu))

156extern struct static_key_deferred apic_hw_disabled;
157
158static inline int kvm_apic_hw_enabled(struct kvm_lapic *apic)
159{
160 if (static_key_false(&apic_hw_disabled.key))
161 return apic->vcpu->arch.apic_base & MSR_IA32_APICBASE_ENABLE;
162 return MSR_IA32_APICBASE_ENABLE;
163}
164
165extern struct static_key_deferred apic_sw_disabled;
166
167static inline bool kvm_apic_sw_enabled(struct kvm_lapic *apic)
168{
169 if (static_key_false(&apic_sw_disabled.key))
170 return apic->sw_enabled;
171 return true;
172}

174static inline bool kvm_apic_present(struct kvm_vcpu *vcpu)
175{
176 return lapic_in_kernel(vcpu) && kvm_apic_hw_enabled(vcpu->arch.apic);
177}
