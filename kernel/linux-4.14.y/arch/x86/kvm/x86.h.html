57static inline bool is_64_bit_mode(struct kvm_vcpu *vcpu)
58{
59 int cs_db, cs_l;
60
61 if (!is_long_mode(vcpu))
62 return false;
63 kvm_x86_ops->get_cs_db_l_bits(vcpu, &cs_db, &cs_l);
64 return cs_l;
65}

102static inline u8 vcpu_virt_addr_bits(struct kvm_vcpu *vcpu)

112static inline u64 get_canonical(u64 la, u8 vaddr_bits)
113{
114 return ((int64_t)la << (64 - vaddr_bits)) >> (64 - vaddr_bits);
115}

136static inline void vcpu_cache_mmio_info(struct kvm_vcpu *vcpu,
137 gva_t gva, gfn_t gfn, unsigned access)
138{
139 /*
140 * If this is a shadow nested page table, the "GVA" is
141 * actually a nGPA.
142 */
143 vcpu->arch.mmio_gva = mmu_is_nested(vcpu) ? 0 : gva & PAGE_MASK;
144 vcpu->arch.access = access;
145 vcpu->arch.mmio_gfn = gfn;
146 vcpu->arch.mmio_gen = kvm_memslots(vcpu->kvm)->generation;
147}
148
149static inline bool vcpu_match_mmio_gen(struct kvm_vcpu *vcpu)
150{
151 return vcpu->arch.mmio_gen == kvm_memslots(vcpu->kvm)->generation;
152}
153
154/*
155 * Clear the mmio cache info for the given gva. If gva is MMIO_GVA_ANY, we
156 * clear all mmio cache info.
157 */
158#define MMIO_GVA_ANY (~(gva_t)0)
159
160static inline void vcpu_clear_mmio_info(struct kvm_vcpu *vcpu, gva_t gva)
161{
162 if (gva != MMIO_GVA_ANY && vcpu->arch.mmio_gva != (gva & PAGE_MASK))
163 return;
164
165 vcpu->arch.mmio_gva = 0;
166}
167
168static inline bool vcpu_match_mmio_gva(struct kvm_vcpu *vcpu, unsigned long gva)
169{
170 if (vcpu_match_mmio_gen(vcpu) && vcpu->arch.mmio_gva &&
171 vcpu->arch.mmio_gva == (gva & PAGE_MASK))
172 return true;
173
174 return false;
175}
176
177static inline bool vcpu_match_mmio_gpa(struct kvm_vcpu *vcpu, gpa_t gpa)
178{
179 if (vcpu_match_mmio_gen(vcpu) && vcpu->arch.mmio_gfn &&
180 vcpu->arch.mmio_gfn == gpa >> PAGE_SHIFT)
181 return true;
182
183 return false;
184}

253/* Same "calling convention" as do_div:
254 * - divide (n << 32) by base
255 * - put result in n
256 * - return remainder
257 */
258#define do_shl32_div32(n, base) \
259 ({ \
260 u32 __quot, __rem; \
261 asm("divl %2" : "=a" (__quot), "=d" (__rem) \
262 : "rm" (base), "0" (0), "1" ((u32) n)); \
263 n = __quot; \
264 __rem; \
265 })

267static inline bool kvm_mwait_in_guest(void)
268{
269 unsigned int eax, ebx, ecx, edx;
270
271 if (!cpu_has(&boot_cpu_data, X86_FEATURE_MWAIT))
272 return false;
273
274 switch (boot_cpu_data.x86_vendor) {
275 case X86_VENDOR_AMD:
276 /* All AMD CPUs have a working MWAIT implementation */
277 return true;
278 case X86_VENDOR_INTEL:
279 /* Handle Intel below */
280 break;
281 default:
282 return false;
283 }
284
285 /*
286 * Intel CPUs without CPUID5_ECX_INTERRUPT_BREAK are problematic as
287 * they would allow guest to stop the CPU completely by disabling
288 * interrupts then invoking MWAIT.
289 */
290 if (boot_cpu_data.cpuid_level < CPUID_MWAIT_LEAF)
291 return false;
292
293 cpuid(CPUID_MWAIT_LEAF, &eax, &ebx, &ecx, &edx);
294
295 if (!(ecx & CPUID5_ECX_INTERRUPT_BREAK))
296 return false;
297
298 return true;
299}
