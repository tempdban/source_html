109static inline int ioapic_in_kernel(struct kvm *kvm)
110{
111 int mode = kvm->arch.irqchip_mode;
112
113 /* Matches smp_wmb() when setting irqchip_mode */
114 smp_rmb();
115 return mode == KVM_IRQCHIP_KERNEL;
116}
