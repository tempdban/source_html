126static inline bool kvm_apic_map_get_logical_dest(struct kvm_apic_map *map,
127 u32 dest_id, struct kvm_lapic ***cluster, u16 *mask) {
128 switch (map->mode) {
129 case KVM_APIC_MODE_X2APIC: {
130 u32 offset = (dest_id >> 16) * 16;
131 u32 max_apic_id = map->max_apic_id;
132
133 if (offset <= max_apic_id) {
134 u8 cluster_size = min(max_apic_id - offset + 1, 16U);
135
136 *cluster = &map->phys_map[offset];
137 *mask = dest_id & (0xffff >> (16 - cluster_size));
138 } else {
139 *mask = 0;
140 }
141
142 return true;
143 }
144 case KVM_APIC_MODE_XAPIC_FLAT:
145 *cluster = map->xapic_flat_map;
146 *mask = dest_id & 0xff;
147 return true;
148 case KVM_APIC_MODE_XAPIC_CLUSTER:
149 *cluster = map->xapic_cluster_map[(dest_id >> 4) & 0xf];
150 *mask = dest_id & 0xf;
151 return true;
152 default:
153 /* Not optimized. */
154 return false;
155 }
156}

165static void recalculate_apic_map(struct kvm *kvm)
166{
167 struct kvm_apic_map *new, *old = NULL;
168 struct kvm_vcpu *vcpu;
169 int i;
170 u32 max_id = 255; /* enough space for any xAPIC ID */
171
172 mutex_lock(&kvm->arch.apic_map_lock);
173
174 kvm_for_each_vcpu(i, vcpu, kvm)
175 if (kvm_apic_present(vcpu))
176 max_id = max(max_id, kvm_x2apic_id(vcpu->arch.apic));
177
178 new = kvzalloc(sizeof(struct kvm_apic_map) +
179 sizeof(struct kvm_lapic *) * ((u64)max_id + 1), GFP_KERNEL);
180
181 if (!new)
182 goto out;
183
184 new->max_apic_id = max_id;
185
186 kvm_for_each_vcpu(i, vcpu, kvm) {
187 struct kvm_lapic *apic = vcpu->arch.apic;
188 struct kvm_lapic **cluster;
189 u16 mask;
190 u32 ldr;
191 u8 xapic_id;
192 u32 x2apic_id;
193
194 if (!kvm_apic_present(vcpu))
195 continue;
196
197 xapic_id = kvm_xapic_id(apic);
198 x2apic_id = kvm_x2apic_id(apic);
199
200 /* Hotplug hack: see kvm_apic_match_physical_addr(), ... */
201 if ((apic_x2apic_mode(apic) || x2apic_id > 0xff) &&
202 x2apic_id <= new->max_apic_id)
203 new->phys_map[x2apic_id] = apic;
204 /*
205 * ... xAPIC ID of VCPUs with APIC ID > 0xff will wrap-around,
206 * prevent them from masking VCPUs with APIC ID <= 0xff.
207 */
208 if (!apic_x2apic_mode(apic) && !new->phys_map[xapic_id])
209 new->phys_map[xapic_id] = apic;
210
211 ldr = kvm_lapic_get_reg(apic, APIC_LDR);
212
213 if (apic_x2apic_mode(apic)) {
214 new->mode |= KVM_APIC_MODE_X2APIC;
215 } else if (ldr) {
216 ldr = GET_APIC_LOGICAL_ID(ldr);
217 if (kvm_lapic_get_reg(apic, APIC_DFR) == APIC_DFR_FLAT)
218 new->mode |= KVM_APIC_MODE_XAPIC_FLAT;
219 else
220 new->mode |= KVM_APIC_MODE_XAPIC_CLUSTER;
221 }
222
223 if (!kvm_apic_map_get_logical_dest(new, ldr, &cluster, &mask))
224 continue;
225
226 if (mask)
227 cluster[ffs(mask) - 1] = apic;
228 }
229out:
230 old = rcu_dereference_protected(kvm->arch.apic_map,
231 lockdep_is_held(&kvm->arch.apic_map_lock));
232 rcu_assign_pointer(kvm->arch.apic_map, new);
233 mutex_unlock(&kvm->arch.apic_map_lock);
234
235 if (old)
236 call_rcu(&old->rcu, kvm_apic_map_free);
237
238 kvm_make_scan_ioapic_request(kvm);
239}

315void kvm_apic_set_version(struct kvm_vcpu *vcpu)
316{
317 struct kvm_lapic *apic = vcpu->arch.apic;
318 struct kvm_cpuid_entry2 *feat;
319 u32 v = APIC_VERSION;
320
321 if (!lapic_in_kernel(vcpu))
322 return;
323
324 /*
325 * KVM emulates 82093AA datasheet (with in-kernel IOAPIC implementation)
326 * which doesn't have EOI register; Some buggy OSes (e.g. Windows with
327 * Hyper-V role) disable EOI broadcast in lapic not checking for IOAPIC
328 * version first and level-triggered interrupts never get EOIed in
329 * IOAPIC.
330 */
331 feat = kvm_find_cpuid_entry(apic->vcpu, 0x1, 0);
332 if (feat && (feat->ecx & (1 << (X86_FEATURE_X2APIC & 31))) &&
333 !ioapic_in_kernel(vcpu->kvm))
334 v |= APIC_LVR_DIRECTED_EOI;
335 kvm_lapic_set_reg(apic, APIC_LVR, v);
336}

346static int find_highest_vector(void *bitmap)
347{
348 int vec;
349 u32 *reg;
350
351 for (vec = MAX_APIC_VECTOR - APIC_VECTORS_PER_REG;
352 vec >= 0; vec -= APIC_VECTORS_PER_REG) {
353 reg = bitmap + REG_POS(vec);
354 if (*reg)
355 return __fls(*reg) + vec;
356 }
357
358 return -1;
359}

361static u8 count_vectors(void *bitmap)
362{
363 int vec;
364 u32 *reg;
365 u8 count = 0;
366
367 for (vec = 0; vec < MAX_APIC_VECTOR; vec += APIC_VECTORS_PER_REG) {
368 reg = bitmap + REG_POS(vec);
369 count += hweight32(*reg);
370 }
371
372 return count;
373}

375int __kvm_apic_update_irr(u32 *pir, void *regs)
376{
377 u32 i, vec;
378 u32 pir_val, irr_val;
379 int max_irr = -1;
380
381 for (i = vec = 0; i <= 7; i++, vec += 32) {
382 pir_val = READ_ONCE(pir[i]);
383 irr_val = *((u32 *)(regs + APIC_IRR + i * 0x10));
384 if (pir_val) {
385 irr_val |= xchg(&pir[i], 0);
386 *((u32 *)(regs + APIC_IRR + i * 0x10)) = irr_val;
387 }
388 if (irr_val)
389 max_irr = __fls(irr_val) + vec;
390 }
391
392 return max_irr;
393}

409static inline int apic_find_highest_irr(struct kvm_lapic *apic)
410{
411 int result;
412
413 /*
414 * Note that irr_pending is just a hint. It will be always
415 * true with virtual interrupt delivery enabled.
416 */
417 if (!apic->irr_pending)
418 return -1;
419
420 result = apic_search_irr(apic);
421 ASSERT(result == -1 || result >= 16);
422
423 return result;
424}

426static inline void apic_clear_irr(int vec, struct kvm_lapic *apic)
427{
428 struct kvm_vcpu *vcpu;
429
430 vcpu = apic->vcpu;
431
432 if (unlikely(vcpu->arch.apicv_active)) {
433 /* need to update RVI */
434 apic_clear_vector(vec, apic->regs + APIC_IRR);
435 kvm_x86_ops->hwapic_irr_update(vcpu,
436 apic_find_highest_irr(apic));
437 } else {
438 apic->irr_pending = false;
439 apic_clear_vector(vec, apic->regs + APIC_IRR);
440 if (apic_search_irr(apic) != -1)
441 apic->irr_pending = true;
442 }
443}

445static inline void apic_set_isr(int vec, struct kvm_lapic *apic)
446{
447 struct kvm_vcpu *vcpu;
448
449 if (__apic_test_and_set_vector(vec, apic->regs + APIC_ISR))
450 return;
451
452 vcpu = apic->vcpu;
453
454 /*
455 * With APIC virtualization enabled, all caching is disabled
456 * because the processor can modify ISR under the hood. Instead
457 * just set SVI.
458 */
459 if (unlikely(vcpu->arch.apicv_active))
460 kvm_x86_ops->hwapic_isr_update(vcpu, vec);
461 else {
462 ++apic->isr_count;
463 BUG_ON(apic->isr_count > MAX_APIC_VECTOR);
464 /*
465 * ISR (in service register) bit is set when injecting an interrupt.
466 * The highest vector is injected. Thus the latest bit set matches
467 * the highest bit in ISR.
468 */
469 apic->highest_isr_cache = vec;
470 }
471}

473static inline int apic_find_highest_isr(struct kvm_lapic *apic)
474{
475 int result;
476
477 /*
478 * Note that isr_count is always 1, and highest_isr_cache
479 * is always -1, with APIC virtualization enabled.
480 */
481 if (!apic->isr_count)
482 return -1;
483 if (likely(apic->highest_isr_cache != -1))
484 return apic->highest_isr_cache;
485
486 result = find_highest_vector(apic->regs + APIC_ISR);
487 ASSERT(result == -1 || result >= 16);
488
489 return result;
490}

492static inline void apic_clear_isr(int vec, struct kvm_lapic *apic)
493{
494 struct kvm_vcpu *vcpu;
495 if (!__apic_test_and_clear_vector(vec, apic->regs + APIC_ISR))
496 return;
497
498 vcpu = apic->vcpu;
499
500 /*
501 * We do get here for APIC virtualization enabled if the guest
502 * uses the Hyper-V APIC enlightenment. In this case we may need
503 * to trigger a new interrupt delivery by writing the SVI field;
504 * on the other hand isr_count and highest_isr_cache are unused
505 * and must be left alone.
506 */
507 if (unlikely(vcpu->arch.apicv_active))
508 kvm_x86_ops->hwapic_isr_update(vcpu,
509 apic_find_highest_isr(apic));
510 else {
511 --apic->isr_count;
512 BUG_ON(apic->isr_count < 0);
513 apic->highest_isr_cache = -1;
514 }
515}

517int kvm_lapic_find_highest_irr(struct kvm_vcpu *vcpu)
518{
519 /* This may race with setting of irr in __apic_accept_irq() and
520 * value returned may be wrong, but kvm_vcpu_kick() in __apic_accept_irq
521 * will cause vmexit immediately and the value will be recalculated
522 * on the next vmentry.
523 */
524 return apic_find_highest_irr(vcpu->arch.apic);
525}

589static int apic_has_interrupt_for_ppr(struct kvm_lapic *apic, u32 ppr)
590{
591 int highest_irr;
592 if (kvm_x86_ops->sync_pir_to_irr && apic->vcpu->arch.apicv_active)
593 highest_irr = kvm_x86_ops->sync_pir_to_irr(apic->vcpu);
594 else
595 highest_irr = apic_find_highest_irr(apic);
596 if (highest_irr == -1 || (highest_irr & 0xF0) <= ppr)
597 return -1;
598 return highest_irr;
599}

601static bool __apic_update_ppr(struct kvm_lapic *apic, u32 *new_ppr)
602{
603 u32 tpr, isrv, ppr, old_ppr;
604 int isr;
605
606 old_ppr = kvm_lapic_get_reg(apic, APIC_PROCPRI);
607 tpr = kvm_lapic_get_reg(apic, APIC_TASKPRI);
608 isr = apic_find_highest_isr(apic);
609 isrv = (isr != -1) ? isr : 0;
610
611 if ((tpr & 0xf0) >= (isrv & 0xf0))
612 ppr = tpr & 0xff;
613 else
614 ppr = isrv & 0xf0;
615
616 apic_debug("vlapic %p, ppr 0x%x, isr 0x%x, isrv 0x%x",
617 apic, ppr, isr, isrv);
618
619 *new_ppr = ppr;
620 if (old_ppr != ppr)
621 kvm_lapic_set_reg(apic, APIC_PROCPRI, ppr);
622
623 return ppr < old_ppr;
624}

626static void apic_update_ppr(struct kvm_lapic *apic)
627{
628 u32 ppr;
629
630 if (__apic_update_ppr(apic, &ppr) &&

647static bool kvm_apic_broadcast(struct kvm_lapic *apic, u32 mda)
648{
649 return mda == (apic_x2apic_mode(apic) ?
650 X2APIC_BROADCAST : APIC_BROADCAST);
651}

653static bool kvm_apic_match_physical_addr(struct kvm_lapic *apic, u32 mda)
654{
655 if (kvm_apic_broadcast(apic, mda))
656 return true;
657
658 if (apic_x2apic_mode(apic))
659 return mda == kvm_x2apic_id(apic);
660
661 /*
662 * Hotplug hack: Make LAPIC in xAPIC mode also accept interrupts as if
663 * it were in x2APIC mode. Hotplugged VCPUs start in xAPIC mode and
664 * this allows unique addressing of VCPUs with APIC ID over 0xff.
665 * The 0xff condition is needed because writeable xAPIC ID.
666 */
667 if (kvm_x2apic_id(apic) > 0xff && mda == kvm_x2apic_id(apic))
668 return true;
669
670 return mda == kvm_xapic_id(apic);
671}
672
673static bool kvm_apic_match_logical_addr(struct kvm_lapic *apic, u32 mda)
674{
675 u32 logical_id;
676
677 if (kvm_apic_broadcast(apic, mda))
678 return true;
679
680 logical_id = kvm_lapic_get_reg(apic, APIC_LDR);
681
682 if (apic_x2apic_mode(apic))
683 return ((logical_id >> 16) == (mda >> 16))
684 && (logical_id & mda & 0xffff) != 0;
685
686 logical_id = GET_APIC_LOGICAL_ID(logical_id);
687
688 switch (kvm_lapic_get_reg(apic, APIC_DFR)) {
689 case APIC_DFR_FLAT:
690 return (logical_id & mda) != 0;
691 case APIC_DFR_CLUSTER:
692 return ((logical_id >> 4) == (mda >> 4))
693 && (logical_id & mda & 0xf) != 0;
694 default:
695 apic_debug("Bad DFR vcpu %d: %08x\n",
696 apic->vcpu->vcpu_id, kvm_lapic_get_reg(apic, APIC_DFR));
697 return false;
698 }
699}

701/* The KVM local APIC implementation has two quirks:
702 *
703 * - Real hardware delivers interrupts destined to x2APIC ID > 0xff to LAPICs
704 * in xAPIC mode if the "destination & 0xff" matches its xAPIC ID.
705 * KVM doesn't do that aliasing.
706 *
707 * - in-kernel IOAPIC messages have to be delivered directly to
708 * x2APIC, because the kernel does not support interrupt remapping.
709 * In order to support broadcast without interrupt remapping, x2APIC
710 * rewrites the destination of non-IPI messages from APIC_BROADCAST
711 * to X2APIC_BROADCAST.
712 *
713 * The broadcast quirk can be disabled with KVM_CAP_X2APIC_API. This is
714 * important when userspace wants to use x2APIC-format MSIs, because
715 * APIC_BROADCAST (0xff) is a legal route for "cluster 0, CPUs 0-7".
716 */
717static u32 kvm_apic_mda(struct kvm_vcpu *vcpu, unsigned int dest_id,
718 struct kvm_lapic *source, struct kvm_lapic *target)
719{
720 bool ipi = source != NULL;
721
722 if (!vcpu->kvm->arch.x2apic_broadcast_quirk_disabled &&
723 !ipi && dest_id == APIC_BROADCAST && apic_x2apic_mode(target))
724 return X2APIC_BROADCAST;
725
726 return dest_id;
727}
728
729bool kvm_apic_match_dest(struct kvm_vcpu *vcpu, struct kvm_lapic *source,
730 int short_hand, unsigned int dest, int dest_mode)
731{
732 struct kvm_lapic *target = vcpu->arch.apic;
733 u32 mda = kvm_apic_mda(vcpu, dest, source, target);
734
735 apic_debug("target %p, source %p, dest 0x%x, "
736 "dest_mode 0x%x, short_hand 0x%x\n",
737 target, source, dest, dest_mode, short_hand);
738
739 ASSERT(target);
740 switch (short_hand) {
741 case APIC_DEST_NOSHORT:
742 if (dest_mode == APIC_DEST_PHYSICAL)
743 return kvm_apic_match_physical_addr(target, mda);
744 else
745 return kvm_apic_match_logical_addr(target, mda);
746 case APIC_DEST_SELF:
747 return target == source;
748 case APIC_DEST_ALLINC:
749 return true;
750 case APIC_DEST_ALLBUT:
751 return target != source;
752 default:
753 apic_debug("kvm: apic: Bad dest shorthand value %x\n",
754 short_hand);
755 return false;
756 }
757}
758EXPORT_SYMBOL_GPL(kvm_apic_match_dest);

760int kvm_vector_to_index(u32 vector, u32 dest_vcpus,
761 const unsigned long *bitmap, u32 bitmap_size)
762{
763 u32 mod;
764 int i, idx = -1;
765
766 mod = vector % dest_vcpus;
767
768 for (i = 0; i <= mod; i++) {
769 idx = find_next_bit(bitmap, bitmap_size, idx + 1);
770 BUG_ON(idx == bitmap_size);
771 }
772
773 return idx;
774}

785static bool kvm_apic_is_broadcast_dest(struct kvm *kvm, struct kvm_lapic **src,
786 struct kvm_lapic_irq *irq, struct kvm_apic_map *map)
787{
788 if (kvm->arch.x2apic_broadcast_quirk_disabled) {
789 if ((irq->dest_id == APIC_BROADCAST &&
790 map->mode != KVM_APIC_MODE_X2APIC))
791 return true;
792 if (irq->dest_id == X2APIC_BROADCAST)
793 return true;
794 } else {
795 bool x2apic_ipi = src && *src && apic_x2apic_mode(*src);
796 if (irq->dest_id == (x2apic_ipi ?
797 X2APIC_BROADCAST : APIC_BROADCAST))
798 return true;
799 }
800
801 return false;
802}

804/* Return true if the interrupt can be handled by using *bitmap as index mask
805 * for valid destinations in *dst array.
806 * Return false if kvm_apic_map_get_dest_lapic did nothing useful.
807 * Note: we may have zero kvm_lapic destinations when we return true, which
808 * means that the interrupt should be dropped. In this case, *bitmap would be
809 * zero and *dst undefined.
810 */
811static inline bool kvm_apic_map_get_dest_lapic(struct kvm *kvm,
812 struct kvm_lapic **src, struct kvm_lapic_irq *irq,
813 struct kvm_apic_map *map, struct kvm_lapic ***dst,
814 unsigned long *bitmap)
815{
816 int i, lowest;
817
818 if (irq->shorthand == APIC_DEST_SELF && src) {
819 *dst = src;
820 *bitmap = 1;
821 return true;
822 } else if (irq->shorthand)
823 return false;
824
825 if (!map || kvm_apic_is_broadcast_dest(kvm, src, irq, map))
826 return false;
827
828 if (irq->dest_mode == APIC_DEST_PHYSICAL) {
829 if (irq->dest_id > map->max_apic_id) {
830 *bitmap = 0;
831 } else {
832 *dst = &map->phys_map[irq->dest_id];
833 *bitmap = 1;
834 }
835 return true;
836 }
837
838 *bitmap = 0;
839 if (!kvm_apic_map_get_logical_dest(map, irq->dest_id, dst,
840 (u16 *)bitmap))
841 return false;
842
843 if (!kvm_lowest_prio_delivery(irq))
844 return true;
845
846 if (!kvm_vector_hashing_enabled()) {
847 lowest = -1;
848 for_each_set_bit(i, bitmap, 16) {
849 if (!(*dst)[i])
850 continue;
851 if (lowest < 0)
852 lowest = i;
853 else if (kvm_apic_compare_prio((*dst)[i]->vcpu,
854 (*dst)[lowest]->vcpu) < 0)
855 lowest = i;
856 }
857 } else {
858 if (!*bitmap)
859 return true;
860
861 lowest = kvm_vector_to_index(irq->vector, hweight16(*bitmap),
862 bitmap, 16);
863
864 if (!(*dst)[lowest]) {
865 kvm_apic_disabled_lapic_found(kvm);
866 *bitmap = 0;
867 return true;
868 }
869 }
870
871 *bitmap = (lowest >= 0) ? 1 << lowest : 0;
872
873 return true;
874}

876bool kvm_irq_delivery_to_apic_fast(struct kvm *kvm, struct kvm_lapic *src,
877 struct kvm_lapic_irq *irq, int *r, struct dest_map *dest_map)
878{
879 struct kvm_apic_map *map;
880 unsigned long bitmap;
881 struct kvm_lapic **dst = NULL;
882 int i;
883 bool ret;
884
885 *r = -1;
886
887 if (irq->shorthand == APIC_DEST_SELF) {
888 *r = kvm_apic_set_irq(src->vcpu, irq, dest_map);
889 return true;
890 }
891
892 rcu_read_lock();
893 map = rcu_dereference(kvm->arch.apic_map);
894
895 ret = kvm_apic_map_get_dest_lapic(kvm, &src, irq, map, &dst, &bitmap);
896 if (ret)
897 for_each_set_bit(i, &bitmap, 16) {
898 if (!dst[i])
899 continue;
900 if (*r < 0)
901 *r = 0;
902 *r += kvm_apic_set_irq(dst[i]->vcpu, irq, dest_map);
903 }
904
905 rcu_read_unlock();
906 return ret;
907}

909/*
910 * This routine tries to handler interrupts in posted mode, here is how
911 * it deals with different cases:
912 * - For single-destination interrupts, handle it in posted mode
913 * - Else if vector hashing is enabled and it is a lowest-priority
914 * interrupt, handle it in posted mode and use the following mechanism
915 * to find the destinaiton vCPU.
916 * 1. For lowest-priority interrupts, store all the possible
917 * destination vCPUs in an array.
918 * 2. Use "guest vector % max number of destination vCPUs" to find
919 * the right destination vCPU in the array for the lowest-priority
920 * interrupt.
921 * - Otherwise, use remapped mode to inject the interrupt.
922 */
923bool kvm_intr_is_single_vcpu_fast(struct kvm *kvm, struct kvm_lapic_irq *irq,
924 struct kvm_vcpu **dest_vcpu)
925{
926 struct kvm_apic_map *map;
927 unsigned long bitmap;
928 struct kvm_lapic **dst = NULL;
929 bool ret = false;
930
931 if (irq->shorthand)
932 return false;
933
934 rcu_read_lock();
935 map = rcu_dereference(kvm->arch.apic_map);
936
937 if (kvm_apic_map_get_dest_lapic(kvm, NULL, irq, map, &dst, &bitmap) &&
938 hweight16(bitmap) == 1) {
939 unsigned long i = find_first_bit(&bitmap, 16);
940
941 if (dst[i]) {
942 *dest_vcpu = dst[i]->vcpu;
943 ret = true;
944 }
945 }
946
947 rcu_read_unlock();
948 return ret;
949}

951/*
952 * Add a pending IRQ into lapic.
953 * Return 1 if successfully added and 0 if discarded.
954 */
955static int __apic_accept_irq(struct kvm_lapic *apic, int delivery_mode,
956 int vector, int level, int trig_mode,
957 struct dest_map *dest_map)
958{
959 int result = 0;
960 struct kvm_vcpu *vcpu = apic->vcpu;
961
962 trace_kvm_apic_accept_irq(vcpu->vcpu_id, delivery_mode,
963 trig_mode, vector);
964 switch (delivery_mode) {
965 case APIC_DM_LOWEST:
966 vcpu->arch.apic_arb_prio++;
967 case APIC_DM_FIXED:
968 if (unlikely(trig_mode && !level))
969 break;
970
971 /* FIXME add logic for vcpu on reset */
972 if (unlikely(!apic_enabled(apic)))
973 break;
974
975 result = 1;
976
977 if (dest_map) {
978 __set_bit(vcpu->vcpu_id, dest_map->map);
979 dest_map->vectors[vcpu->vcpu_id] = vector;
980 }
981
982 if (apic_test_vector(vector, apic->regs + APIC_TMR) != !!trig_mode) {
983 if (trig_mode)
984 kvm_lapic_set_vector(vector, apic->regs + APIC_TMR);
985 else
986 apic_clear_vector(vector, apic->regs + APIC_TMR);
987 }
988
989 if (vcpu->arch.apicv_active)
990 kvm_x86_ops->deliver_posted_interrupt(vcpu, vector);
991 else {
992 kvm_lapic_set_irr(vector, apic);
993
994 kvm_make_request(KVM_REQ_EVENT, vcpu);
995 kvm_vcpu_kick(vcpu);
996 }
997 break;
998
999 case APIC_DM_REMRD:
1000 result = 1;
1001 vcpu->arch.pv.pv_unhalted = 1;
1002 kvm_make_request(KVM_REQ_EVENT, vcpu);
1003 kvm_vcpu_kick(vcpu);
1004 break;
1005
1006 case APIC_DM_SMI:
1007 result = 1;
1008 kvm_make_request(KVM_REQ_SMI, vcpu);
1009 kvm_vcpu_kick(vcpu);
1010 break;
1011
1012 case APIC_DM_NMI:
1013 result = 1;
1014 kvm_inject_nmi(vcpu);
1015 kvm_vcpu_kick(vcpu);
1016 break;
1017
1018 case APIC_DM_INIT:
1019 if (!trig_mode || level) {
1020 result = 1;
1021 /* assumes that there are only KVM_APIC_INIT/SIPI */
1022 apic->pending_events = (1UL << KVM_APIC_INIT);
1023 /* make sure pending_events is visible before sending
1024 * the request */
1025 smp_wmb();
1026 kvm_make_request(KVM_REQ_EVENT, vcpu);
1027 kvm_vcpu_kick(vcpu);
1028 } else {
1029 apic_debug("Ignoring de-assert INIT to vcpu %d\n",
1030 vcpu->vcpu_id);
1031 }
1032 break;
1033
1034 case APIC_DM_STARTUP:
1035 apic_debug("SIPI to vcpu %d vector 0x%02x\n",
1036 vcpu->vcpu_id, vector);
1037 result = 1;
1038 apic->sipi_vector = vector;
1039 /* make sure sipi_vector is visible for the receiver */
1040 smp_wmb();
1041 set_bit(KVM_APIC_SIPI, &apic->pending_events);
1042 kvm_make_request(KVM_REQ_EVENT, vcpu);
1043 kvm_vcpu_kick(vcpu);
1044 break;
1045
1046 case APIC_DM_EXTINT:
1047 /*
1048 * Should only be called by kvm_apic_local_deliver() with LVT0,
1049 * before NMI watchdog was enabled. Already handled by
1050 * kvm_apic_accept_pic_intr().
1051 */
1052 break;
1053
1054 default:
1055 printk(KERN_ERR "TODO: unsupported delivery mode %x\n",
1056 delivery_mode);
1057 break;
1058 }
1059 return result;
1060}

1165static u32 apic_get_tmcct(struct kvm_lapic *apic)
1166{
1167 ktime_t remaining, now;
1168 s64 ns;
1169 u32 tmcct;
1170
1171 ASSERT(apic != NULL);
1172
1173 /* if initial count is 0, current count should also be 0 */
1174 if (kvm_lapic_get_reg(apic, APIC_TMICT) == 0 ||
1175 apic->lapic_timer.period == 0)
1176 return 0;
1177
1178 now = ktime_get();
1179 remaining = ktime_sub(apic->lapic_timer.target_expiration, now);
1180 if (ktime_to_ns(remaining) < 0)
1181 remaining = 0;
1182
1183 ns = mod_64(ktime_to_ns(remaining), apic->lapic_timer.period);
1184 tmcct = div64_u64(ns,
1185 (APIC_BUS_CYCLE_NS * apic->divide_count));
1186
1187 return tmcct;
1188}

1206static u32 __apic_read(struct kvm_lapic *apic, unsigned int offset)
1207{
1208 u32 val = 0;
1209
1210 if (offset >= LAPIC_MMIO_LENGTH)
1211 return 0;
1212
1213 switch (offset) {
1214 case APIC_ARBPRI:
1215 apic_debug("Access APIC ARBPRI register which is for P6\n");
1216 break;
1217
1218 case APIC_TMCCT: /* Timer CCR */
1219 if (apic_lvtt_tscdeadline(apic))
1220 return 0;
1221
1222 val = apic_get_tmcct(apic);
1223 break;
1224 case APIC_PROCPRI:
1225 apic_update_ppr(apic);
1226 val = kvm_lapic_get_reg(apic, offset);
1227 break;
1228 case APIC_TASKPRI:
1229 report_tpr_access(apic, false);
1230 /* fall thru */
1231 default:
1232 val = kvm_lapic_get_reg(apic, offset);
1233 break;
1234 }
1235
1236 return val;
1237}

1244int kvm_lapic_reg_read(struct kvm_lapic *apic, u32 offset, int len,
1245 void *data)
1246{
1247 unsigned char alignment = offset & 0xf;
1248 u32 result;
1249 /* this bitmask has a bit cleared for each reserved register */
1250 static const u64 rmask = 0x43ff01ffffffe70cULL;
1251
1252 if ((alignment + len) > 4) {
1253 apic_debug("KVM_APIC_READ: alignment error %x %d\n",
1254 offset, len);
1255 return 1;
1256 }
1257
1258 if (offset > 0x3f0 || !(rmask & (1ULL << (offset >> 4)))) {
1259 apic_debug("KVM_APIC_READ: read reserved register %x\n",
1260 offset);
1261 return 1;
1262 }
1263
1264 result = __apic_read(apic, offset & ~0xf);
1265
1266 trace_kvm_apic_read(offset, result);
1267
1268 switch (len) {
1269 case 1:
1270 case 2:
1271 case 4:
1272 memcpy(data, (char *)&result + alignment, len);
1273 break;
1274 default:

1304static void update_divide_count(struct kvm_lapic *apic)
1305{
1306 u32 tmp1, tmp2, tdcr;
1307

1328static void apic_timer_expired(struct kvm_lapic *apic)
1329{
1330 struct kvm_vcpu *vcpu = apic->vcpu;
1331 struct swait_queue_head *q = &vcpu->wq;
1332 struct kvm_timer *ktimer = &apic->lapic_timer;
1333
1334 if (atomic_read(&apic->lapic_timer.pending))
1335 return;

1341 * For x86, the atomic_inc() is serialized, thus
1342 * using swait_active() is safe.
1343 */

1345 swake_up(q);
1346
1347 if (apic_lvtt_tscdeadline(apic))
1348 ktimer->expired_tscdeadline = ktimer->tscdeadline;
1349}

1374void wait_lapic_expire(struct kvm_vcpu *vcpu)
1375{
1376 struct kvm_lapic *apic = vcpu->arch.apic;
1377 u64 guest_tsc, tsc_deadline;
1378
1379 if (!lapic_in_kernel(vcpu))
1380 return;
1381
1382 if (apic->lapic_timer.expired_tscdeadline == 0)
1383 return;
1384
1385 if (!lapic_timer_int_injected(vcpu))
1386 return;
1387
1388 tsc_deadline = apic->lapic_timer.expired_tscdeadline;
1389 apic->lapic_timer.expired_tscdeadline = 0;
1390 guest_tsc = kvm_read_l1_tsc(vcpu, rdtsc());
1391 trace_kvm_wait_lapic_expire(vcpu->vcpu_id, guest_tsc - tsc_deadline);
1392
1393 /* __delay is delay_tsc whenever the hardware has TSC, thus always. */
1394 if (guest_tsc < tsc_deadline)
1395 __delay(min(tsc_deadline - guest_tsc,
1396 nsec_to_cycles(vcpu, lapic_timer_advance_ns)));
1397}

1399static void start_sw_tscdeadline(struct kvm_lapic *apic)
1400{
1401 u64 guest_tsc, tscdeadline = apic->lapic_timer.tscdeadline;
1402 u64 ns = 0;
1403 ktime_t expire;
1404 struct kvm_vcpu *vcpu = apic->vcpu;
1405 unsigned long this_tsc_khz = vcpu->arch.virtual_tsc_khz;
1406 unsigned long flags;
1407 ktime_t now;
1408
1409 if (unlikely(!tscdeadline || !this_tsc_khz))
1410 return;
1411
1412 local_irq_save(flags);
1413
1414 now = ktime_get();
1415 guest_tsc = kvm_read_l1_tsc(vcpu, rdtsc());
1416 if (likely(tscdeadline > guest_tsc)) {
1417 ns = (tscdeadline - guest_tsc) * 1000000ULL;
1418 do_div(ns, this_tsc_khz);
1419 expire = ktime_add_ns(now, ns);
1420 expire = ktime_sub_ns(expire, lapic_timer_advance_ns);
1421 hrtimer_start(&apic->lapic_timer.timer,
1422 expire, HRTIMER_MODE_ABS_PINNED);
1423 } else
1424 apic_timer_expired(apic);
1425
1426 local_irq_restore(flags);
1427}

1429static bool set_target_expiration(struct kvm_lapic *apic)
1430{
1431 ktime_t now;
1432 u64 tscl = rdtsc();
1433
1434 now = ktime_get();
1435 apic->lapic_timer.period = (u64)kvm_lapic_get_reg(apic, APIC_TMICT)
1436 * APIC_BUS_CYCLE_NS * apic->divide_count;
1437
1438 if (!apic->lapic_timer.period)
1439 return false;
1440
1441 /*
1442 * Do not allow the guest to program periodic timers with small
1443 * interval, since the hrtimers are not throttled by the host
1444 * scheduler.
1445 */
1446 if (apic_lvtt_period(apic)) {
1447 s64 min_period = min_timer_period_us * 1000LL;
1448
1449 if (apic->lapic_timer.period < min_period) {
1450 pr_info_ratelimited(
1451 "kvm: vcpu %i: requested %lld ns "
1452 "lapic timer period limited to %lld ns\n",
1453 apic->vcpu->vcpu_id,
1454 apic->lapic_timer.period, min_period);
1455 apic->lapic_timer.period = min_period;
1456 }
1457 }
1458
1459 apic_debug("%s: bus cycle is %" PRId64 "ns, now 0x%016"
1460 PRIx64 ", "
1461 "timer initial count 0x%x, period %lldns, "
1462 "expire @ 0x%016" PRIx64 ".\n", __func__,
1463 APIC_BUS_CYCLE_NS, ktime_to_ns(now),
1464 kvm_lapic_get_reg(apic, APIC_TMICT),
1465 apic->lapic_timer.period,
1466 ktime_to_ns(ktime_add_ns(now,
1467 apic->lapic_timer.period)));
1468
1469 apic->lapic_timer.tscdeadline = kvm_read_l1_tsc(apic->vcpu, tscl) +
1470 nsec_to_cycles(apic->vcpu, apic->lapic_timer.period);
1471 apic->lapic_timer.target_expiration = ktime_add_ns(now, apic->lapic_timer.period);
1472
1473 return true;
1474}
1475
1476static void advance_periodic_target_expiration(struct kvm_lapic *apic)
1477{
1478 ktime_t now = ktime_get();
1479 u64 tscl = rdtsc();
1480 ktime_t delta;
1481
1482 /*
1483 * Synchronize both deadlines to the same time source or
1484 * differences in the periods (caused by differences in the
1485 * underlying clocks or numerical approximation errors) will
1486 * cause the two to drift apart over time as the errors
1487 * accumulate.
1488 */
1489 apic->lapic_timer.target_expiration =
1490 ktime_add_ns(apic->lapic_timer.target_expiration,
1491 apic->lapic_timer.period);
1492 delta = ktime_sub(apic->lapic_timer.target_expiration, now);
1493 apic->lapic_timer.tscdeadline = kvm_read_l1_tsc(apic->vcpu, tscl) +
1494 nsec_to_cycles(apic->vcpu, delta);
1495}

1497static void start_sw_period(struct kvm_lapic *apic)
1498{
1499 if (!apic->lapic_timer.period)
1500 return;
1501
1502 if (ktime_after(ktime_get(),
1503 apic->lapic_timer.target_expiration)) {
1504 apic_timer_expired(apic);
1505
1506 if (apic_lvtt_oneshot(apic))
1507 return;
1508
1509 advance_periodic_target_expiration(apic);
1510 }
1511
1512 hrtimer_start(&apic->lapic_timer.timer,
1513 apic->lapic_timer.target_expiration,
1514 HRTIMER_MODE_ABS_PINNED);
1515}

1534static bool start_hv_timer(struct kvm_lapic *apic)
1535{
1536 struct kvm_timer *ktimer = &apic->lapic_timer;
1537 int r;
1538
1539 WARN_ON(preemptible());
1540 if (!kvm_x86_ops->set_hv_timer)
1541 return false;
1542
1543 if (!apic_lvtt_period(apic) && atomic_read(&ktimer->pending))
1544 return false;
1545
1546 r = kvm_x86_ops->set_hv_timer(apic->vcpu, ktimer->tscdeadline);
1547 if (r < 0)
1548 return false;
1549
1550 ktimer->hv_timer_in_use = true;
1551 hrtimer_cancel(&ktimer->timer);
1552
1553 /*
1554 * Also recheck ktimer->pending, in case the sw timer triggered in
1555 * the window. For periodic timer, leave the hv timer running for
1556 * simplicity, and the deadline will be recomputed on the next vmexit.
1557 */
1558 if (!apic_lvtt_period(apic) && (r || atomic_read(&ktimer->pending))) {
1559 if (r)
1560 apic_timer_expired(apic);
1561 return false;
1562 }
1563
1564 trace_kvm_hv_timer_state(apic->vcpu->vcpu_id, true);
1565 return true;
1566}

1569{
1570 struct kvm_timer *ktimer = &apic->lapic_timer;
1571
1572 WARN_ON(preemptible());
1573 if (apic->lapic_timer.hv_timer_in_use)
1574 cancel_hv_timer(apic);
1575 if (!apic_lvtt_period(apic) && atomic_read(&ktimer->pending))
1576 return;
1577
1578 if (apic_lvtt_period(apic) || apic_lvtt_oneshot(apic))
1579 start_sw_period(apic);
1580 else if (apic_lvtt_tscdeadline(apic))
1581 start_sw_tscdeadline(apic);
1582 trace_kvm_hv_timer_state(apic->vcpu->vcpu_id, false);
1583}

1666int kvm_lapic_reg_write(struct kvm_lapic *apic, u32 reg, u32 val)
1667{
1668 int ret = 0;
1669
1670 trace_kvm_apic_write(reg, val);
1671
1672 switch (reg) {
1673 case APIC_ID: /* Local APIC ID */
1674 if (!apic_x2apic_mode(apic))
1675 kvm_apic_set_xapic_id(apic, val >> 24);
1676 else
1677 ret = 1;
1678 break;
1679
1680 case APIC_TASKPRI:
1681 report_tpr_access(apic, true);
1682 apic_set_tpr(apic, val & 0xff);
1683 break;
1684
1685 case APIC_EOI:
1686 apic_set_eoi(apic);
1687 break;
1688
1689 case APIC_LDR:
1690 if (!apic_x2apic_mode(apic))
1691 kvm_apic_set_ldr(apic, val & APIC_LDR_MASK);
1692 else
1693 ret = 1;
1694 break;
1695
1696 case APIC_DFR:
1697 if (!apic_x2apic_mode(apic)) {
1698 kvm_lapic_set_reg(apic, APIC_DFR, val | 0x0FFFFFFF);
1699 recalculate_apic_map(apic->vcpu->kvm);
1700 } else
1701 ret = 1;
1702 break;
1703
1704 case APIC_SPIV: {
1705 u32 mask = 0x3ff;
1706 if (kvm_lapic_get_reg(apic, APIC_LVR) & APIC_LVR_DIRECTED_EOI)
1707 mask |= APIC_SPIV_DIRECTED_EOI;
1708 apic_set_spiv(apic, val & mask);
1709 if (!(val & APIC_SPIV_APIC_ENABLED)) {
1710 int i;
1711 u32 lvt_val;
1712
1713 for (i = 0; i < KVM_APIC_LVT_NUM; i++) {
1714 lvt_val = kvm_lapic_get_reg(apic,
1715 APIC_LVTT + 0x10 * i);
1716 kvm_lapic_set_reg(apic, APIC_LVTT + 0x10 * i,
1717 lvt_val | APIC_LVT_MASKED);
1718 }
1719 apic_update_lvtt(apic);
1720 atomic_set(&apic->lapic_timer.pending, 0);
1721
1722 }
1723 break;
1724 }
1725 case APIC_ICR:
1726 /* No delay here, so we always clear the pending bit */
1727 kvm_lapic_set_reg(apic, APIC_ICR, val & ~(1 << 12));
1728 apic_send_ipi(apic);
1729 break;
1730
1731 case APIC_ICR2:
1732 if (!apic_x2apic_mode(apic))
1733 val &= 0xff000000;
1734 kvm_lapic_set_reg(apic, APIC_ICR2, val);
1735 break;
1736
1737 case APIC_LVT0:
1738 apic_manage_nmi_watchdog(apic, val);
1739 case APIC_LVTTHMR:
1740 case APIC_LVTPC:
1741 case APIC_LVT1:
1742 case APIC_LVTERR:
1743 /* TODO: Check vector */
1744 if (!kvm_apic_sw_enabled(apic))
1745 val |= APIC_LVT_MASKED;
1746
1747 val &= apic_lvt_mask[(reg - APIC_LVTT) >> 4];
1748 kvm_lapic_set_reg(apic, reg, val);
1749
1750 break;
1751
1752 case APIC_LVTT:
1753 if (!kvm_apic_sw_enabled(apic))
1754 val |= APIC_LVT_MASKED;
1755 val &= (apic_lvt_mask[0] | apic->lapic_timer.timer_mode_mask);
1756 kvm_lapic_set_reg(apic, APIC_LVTT, val);
1757 apic_update_lvtt(apic);
1758 break;
1759
1760 case APIC_TMICT:
1761 if (apic_lvtt_tscdeadline(apic))
1762 break;
1763
1764 hrtimer_cancel(&apic->lapic_timer.timer);
1765 kvm_lapic_set_reg(apic, APIC_TMICT, val);
1766 start_apic_timer(apic);
1767 break;
1768
1769 case APIC_TDCR:
1770 if (val & 4)
1771 apic_debug("KVM_WRITE:TDCR %x\n", val);
1772 kvm_lapic_set_reg(apic, APIC_TDCR, val);
1773 update_divide_count(apic);
1774 break;
1775
1776 case APIC_ESR:
1777 if (apic_x2apic_mode(apic) && val != 0) {
1778 apic_debug("KVM_WRITE:ESR not zero %x\n", val);
1779 ret = 1;
1780 }
1781 break;
1782
1783 case APIC_SELF_IPI:
1784 if (apic_x2apic_mode(apic)) {
1785 kvm_lapic_reg_write(apic, APIC_ICR, 0x40000 | (val & 0xff));
1786 } else
1787 ret = 1;
1788 break;
1789 default:
1790 ret = 1;
1791 break;
1792 }
1793 if (ret)
1794 apic_debug("Local APIC Write to read-only register %x\n", reg);
1795 return ret;
1796}

1799static int apic_mmio_write(struct kvm_vcpu *vcpu, struct kvm_io_device *this,
1800 gpa_t address, int len, const void *data)
1801{
1802 struct kvm_lapic *apic = to_lapic(this);
1803 unsigned int offset = address - apic->base_address;
1804 u32 val;
1805
1806 if (!apic_mmio_in_range(apic, address))
1807 return -EOPNOTSUPP;
1808
1809 /*
1810 * APIC register must be aligned on 128-bits boundary.
1811 * 32/64/128 bits registers must be accessed thru 32 bits.
1812 * Refer SDM 8.4.1
1813 */
1814 if (len != 4 || (offset & 0xf)) {
1815 /* Don't shout loud, $infamous_os would cause only noise. */
1816 apic_debug("apic write: bad size=%d %lx\n", len, (long)address);
1817 return 0;
1818 }
1819
1820 val = *(u32*)data;
1821
1822 /* too common printing */
1823 if (offset != APIC_EOI)
1824 apic_debug("%s: offset 0x%x with length 0x%x, and value is "
1825 "0x%x\n", __func__, offset, len, val);
1826
1827 kvm_lapic_reg_write(apic, offset & 0xff0, val);
1828
1829 return 0;
1830}

1921{
1922 u64 old_value = vcpu->arch.apic_base;
1923 struct kvm_lapic *apic = vcpu->arch.apic;
1924
1925 if (!apic)
1926 value |= MSR_IA32_APICBASE_BSP;
1927
1928 vcpu->arch.apic_base = value;
1929

1931 kvm_update_cpuid(vcpu);
1932
1933 if (!apic)

1943 recalculate_apic_map(vcpu->kvm);

1953 }

1961

1963 apic_debug("apic base msr is 0x%016" PRIx64 ", and base address is "

1968void kvm_lapic_reset(struct kvm_vcpu *vcpu, bool init_event)
1969{
1970 struct kvm_lapic *apic = vcpu->arch.apic;
1971 int i;
1972
1973 if (!apic)
1974 return;
1975
1976 apic_debug("%s\n", __func__);
1977
1978 /* Stop the timer in case it's a reset to an active apic */
1979 hrtimer_cancel(&apic->lapic_timer.timer);
1980
1981 if (!init_event) {
1982 kvm_lapic_set_base(vcpu, APIC_DEFAULT_PHYS_BASE |
1983 MSR_IA32_APICBASE_ENABLE);
1984 kvm_apic_set_xapic_id(apic, vcpu->vcpu_id);
1985 }
1986 kvm_apic_set_version(apic->vcpu);
1987
1988 for (i = 0; i < KVM_APIC_LVT_NUM; i++)
1989 kvm_lapic_set_reg(apic, APIC_LVTT + 0x10 * i, APIC_LVT_MASKED);
1990 apic_update_lvtt(apic);
1991 if (kvm_vcpu_is_reset_bsp(vcpu) &&
1992 kvm_check_has_quirk(vcpu->kvm, KVM_X86_QUIRK_LINT0_REENABLED))
1993 kvm_lapic_set_reg(apic, APIC_LVT0,
1994 SET_APIC_DELIVERY_MODE(0, APIC_MODE_EXTINT));
1995 apic_manage_nmi_watchdog(apic, kvm_lapic_get_reg(apic, APIC_LVT0));
1996
1997 kvm_lapic_set_reg(apic, APIC_DFR, 0xffffffffU);
1998 apic_set_spiv(apic, 0xff);
1999 kvm_lapic_set_reg(apic, APIC_TASKPRI, 0);
2000 if (!apic_x2apic_mode(apic))
2001 kvm_apic_set_ldr(apic, 0);
2002 kvm_lapic_set_reg(apic, APIC_ESR, 0);
2003 kvm_lapic_set_reg(apic, APIC_ICR, 0);
2004 kvm_lapic_set_reg(apic, APIC_ICR2, 0);
2005 kvm_lapic_set_reg(apic, APIC_TDCR, 0);
2006 kvm_lapic_set_reg(apic, APIC_TMICT, 0);
2007 for (i = 0; i < 8; i++) {
2008 kvm_lapic_set_reg(apic, APIC_IRR + 0x10 * i, 0);
2009 kvm_lapic_set_reg(apic, APIC_ISR + 0x10 * i, 0);
2010 kvm_lapic_set_reg(apic, APIC_TMR + 0x10 * i, 0);
2011 }
2012 apic->irr_pending = vcpu->arch.apicv_active;
2013 apic->isr_count = vcpu->arch.apicv_active ? 1 : 0;
2014 apic->highest_isr_cache = -1;
2015 update_divide_count(apic);
2016 atomic_set(&apic->lapic_timer.pending, 0);
2017 if (kvm_vcpu_is_bsp(vcpu))
2018 kvm_lapic_set_base(vcpu,
2019 vcpu->arch.apic_base | MSR_IA32_APICBASE_BSP);
2020 vcpu->arch.pv_eoi.msr_val = 0;
2021 apic_update_ppr(apic);
2022 if (vcpu->arch.apicv_active) {
2023 kvm_x86_ops->apicv_post_state_restore(vcpu);
2024 kvm_x86_ops->hwapic_irr_update(vcpu, -1);
2025 kvm_x86_ops->hwapic_isr_update(vcpu, -1);
2026 }
2027
2028 vcpu->arch.apic_arb_prio = 0;
2029 vcpu->arch.apic_attention = 0;
2030
2031 apic_debug("%s: vcpu=%p, id=0x%x, base_msr="
2032 "0x%016" PRIx64 ", base_address=0x%0lx.\n", __func__,
2033 vcpu, kvm_lapic_get_reg(apic, APIC_ID),
2034 vcpu->arch.apic_base, apic->base_address);
2035}

2308 * apic_sync_pv_eoi_from_guest - called on vmexit or cancel interrupt
2309 *
2310 * Detect whether guest triggered PV EOI since the
2311 * last entry. If yes, set EOI on guests's behalf.
2312 * Clear PV EOI in guest memory in any case.
2313 */

2315 struct kvm_lapic *apic)
2316{

2318 int vector;
2319 /*
2320 * PV EOI state is derived from KVM_APIC_PV_EOI_PENDING in host
2321 * and KVM_PV_EOI_ENABLED in guest memory as follows:
2322 *
2323 * KVM_APIC_PV_EOI_PENDING is unset:
2324 * -> host disabled PV EOI.
2325 * KVM_APIC_PV_EOI_PENDING is set, KVM_PV_EOI_ENABLED is set:
2326 * -> host enabled PV EOI, guest did not execute EOI yet.
2327 * KVM_APIC_PV_EOI_PENDING is set, KVM_PV_EOI_ENABLED is unset:
2328 * -> host enabled PV EOI, guest executed EOI.
2329 */
2330 BUG_ON(!pv_eoi_enabled(vcpu));

2335 * this makes sure pv eoi is only enabled when we know it's safe.
2336 */
2337 pv_eoi_clr_pending(vcpu);
2338 if (pending)
2339 return;
2340 vector = apic_set_eoi(apic);
2341 trace_kvm_pv_eoi(apic, vector);

2456 }

2462
2463 *data = (((u64)high) << 32) | low;
2464
2465 return 0;
2466}

2512void kvm_apic_accept_events(struct kvm_vcpu *vcpu)
2513{
2514 struct kvm_lapic *apic = vcpu->arch.apic;
2515 u8 sipi_vector;
2516 unsigned long pe;
2517
2518 if (!lapic_in_kernel(vcpu) || !apic->pending_events)
2519 return;
2520
2521 /*
2522 * INITs are latched while in SMM. Because an SMM CPU cannot
2523 * be in KVM_MP_STATE_INIT_RECEIVED state, just eat SIPIs
2524 * and delay processing of INIT until the next RSM.
2525 */
2526 if (is_smm(vcpu)) {
2527 WARN_ON_ONCE(vcpu->arch.mp_state == KVM_MP_STATE_INIT_RECEIVED);
2528 if (test_bit(KVM_APIC_SIPI, &apic->pending_events))
2529 clear_bit(KVM_APIC_SIPI, &apic->pending_events);
2530 return;
2531 }
2532
2533 pe = xchg(&apic->pending_events, 0);
2534 if (test_bit(KVM_APIC_INIT, &pe)) {
2535 kvm_vcpu_reset(vcpu, true);
2536 if (kvm_vcpu_is_bsp(apic->vcpu))
2537 vcpu->arch.mp_state = KVM_MP_STATE_RUNNABLE;
2538 else
2539 vcpu->arch.mp_state = KVM_MP_STATE_INIT_RECEIVED;
2540 }
2541 if (test_bit(KVM_APIC_SIPI, &pe) &&
2542 vcpu->arch.mp_state == KVM_MP_STATE_INIT_RECEIVED) {
2543 /* evaluate pending_events before reading the vector */
2544 smp_rmb();
2545 sipi_vector = apic->sipi_vector;
2546 apic_debug("vcpu %d received sipi with vector # %x\n",
2547 vcpu->vcpu_id, sipi_vector);
2548 kvm_vcpu_deliver_sipi_vector(vcpu, sipi_vector);
2549 vcpu->arch.mp_state = KVM_MP_STATE_RUNNABLE;
2550 }
2551}
