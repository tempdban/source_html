27static inline struct kvm_vcpu_hv *vcpu_to_hv_vcpu(struct kvm_vcpu *vcpu)
28{
29 return &vcpu->arch.hyperv;
30}
31
32static inline struct kvm_vcpu *hv_vcpu_to_vcpu(struct kvm_vcpu_hv *hv_vcpu)
33{
34 struct kvm_vcpu_arch *arch;
35
36 arch = container_of(hv_vcpu, struct kvm_vcpu_arch, hyperv);
37 return container_of(arch, struct kvm_vcpu, arch);
38}
39
40static inline struct kvm_vcpu_hv_synic *vcpu_to_synic(struct kvm_vcpu *vcpu)
41{
42 return &vcpu->arch.hyperv.synic;
43}
44
45static inline struct kvm_vcpu *synic_to_vcpu(struct kvm_vcpu_hv_synic *synic)
46{
47 return hv_vcpu_to_vcpu(container_of(synic, struct kvm_vcpu_hv, synic));
48}

65static inline struct kvm_vcpu_hv_stimer *vcpu_to_stimer(struct kvm_vcpu *vcpu,
66 int timer_index)
67{
68 return &vcpu_to_hv_vcpu(vcpu)->stimer[timer_index];
69}
70
71static inline struct kvm_vcpu *stimer_to_vcpu(struct kvm_vcpu_hv_stimer *stimer)
72{
73 struct kvm_vcpu_hv *hv_vcpu;
74
75 hv_vcpu = container_of(stimer - stimer->index, struct kvm_vcpu_hv,
76 stimer[0]);
77 return hv_vcpu_to_vcpu(hv_vcpu);
78}

80static inline bool kvm_hv_has_stimer_pending(struct kvm_vcpu *vcpu)
81{
82 return !bitmap_empty(vcpu->arch.hyperv.stimer_pending_bitmap,
83 HV_SYNIC_STIMER_COUNT);
84}
