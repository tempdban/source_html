103/*
104 * 64-bit SYSCALL instruction entry. Up to 6 arguments in registers.
105 *
106 * This is the only entry point used for 64-bit system calls. The
107 * hardware interface is reasonably well designed and the register to
108 * argument mapping Linux uses fits well with the registers that are
109 * available when SYSCALL is used.
110 *
111 * SYSCALL instructions can be found inlined in libc implementations as
112 * well as some other programs and libraries. There are also a handful
113 * of SYSCALL instructions in the vDSO used, for example, as a
114 * clock_gettimeofday fallback.
115 *
116 * 64-bit SYSCALL saves rip to rcx, clears rflags.RF, then saves rflags to r11,
117 * then loads new ss, cs, and rip from previously programmed MSRs.
118 * rflags gets masked by a value from another MSR (so CLD and CLAC
119 * are not needed). SYSCALL does not save anything on the stack
120 * and does not change rsp.
121 *
122 * Registers on entry:
123 * rax system call number
124 * rcx return address
125 * r11 saved rflags (note: r11 is callee-clobbered register in C ABI)
126 * rdi arg0
127 * rsi arg1
128 * rdx arg2
129 * r10 arg3 (needs to be moved to rcx to conform to C ABI)
130 * r8 arg4
131 * r9 arg5
132 * (note: r12-r15, rbp, rbx are callee-preserved in C ABI)
133 *
134 * Only called from user space.
135 *
136 * When user can change pt_regs->foo always force IRET. That is because
137 * it deals with uncanonical addresses better. SYSRET has trouble
138 * with them due to bugs in both AMD and Intel CPUs.
139 */
140
141 .pushsection .entry_trampoline, "ax"

143/*
144 * The code in here gets remapped into cpu_entry_area's trampoline. This means
145 * that the assembler and linker have the wrong idea as to where this code
146 * lives (and, in fact, it's mapped more than once, so it's not even at a
147 * fixed address). So we can't reference any symbols outside the entry
148 * trampoline and expect it to work.
149 *
150 * Instead, we carefully abuse %rip-relative addressing.
151 * _entry_trampoline(%rip) refers to the start of the remapped) entry
152 * trampoline. We can thus find cpu_entry_area with this macro:
153 */
154
155#define CPU_ENTRY_AREA \
156 _entry_trampoline - CPU_ENTRY_AREA_entry_trampoline(%rip)
157
158/* The top word of the SYSENTER stack is hot and is usable as scratch space. */
159#define RSP_SCRATCH CPU_ENTRY_AREA_entry_stack + \
160 SIZEOF_entry_stack - 8 + CPU_ENTRY_AREA
161
162ENTRY(entry_SYSCALL_64_trampoline)
163 UNWIND_HINT_EMPTY
164 swapgs
165
166 /* Stash the user RSP. */
167 movq %rsp, RSP_SCRATCH
168
169 /* Note: using %rsp as a scratch reg. */
170 SWITCH_TO_KERNEL_CR3 scratch_reg=%rsp
171
172 /* Load the top of the task stack into RSP */
173 movq CPU_ENTRY_AREA_tss + TSS_sp1 + CPU_ENTRY_AREA, %rsp
174
175 /* Start building the simulated IRET frame. */
176 pushq $__USER_DS /* pt_regs->ss */
177 pushq RSP_SCRATCH /* pt_regs->sp */
178 pushq %r11 /* pt_regs->flags */
179 pushq $__USER_CS /* pt_regs->cs */
180 pushq %rcx /* pt_regs->ip */
181
182 /*
183 * x86 lacks a near absolute jump, and we can't jump to the real
184 * entry text with a relative jump. We could push the target
185 * address and then use retq, but this destroys the pipeline on
186 * many CPUs (wasting over 20 cycles on Sandy Bridge). Instead,
187 * spill RDI and restore it in a second-stage trampoline.
188 */
189 pushq %rdi
190 movq $entry_SYSCALL_64_stage2, %rdi
191 JMP_NOSPEC %rdi
192END(entry_SYSCALL_64_trampoline)
193
194 .popsection
195
196ENTRY(entry_SYSCALL_64_stage2)
197 UNWIND_HINT_EMPTY
198 popq %rdi
199 jmp entry_SYSCALL_64_after_hwframe
200END(entry_SYSCALL_64_stage2)
201
202ENTRY(entry_SYSCALL_64)
203 UNWIND_HINT_EMPTY
204 /*
205 * Interrupts are off on entry.
206 * We do not frame this tiny irq-off block with TRACE_IRQS_OFF/ON,
207 * it is too small to ever cause noticeable irq latency.
208 */
209
210 swapgs
211 /*
212 * This path is only taken when PAGE_TABLE_ISOLATION is disabled so it
213 * is not required to switch CR3.
214 */
215 movq %rsp, PER_CPU_VAR(rsp_scratch)
216 movq PER_CPU_VAR(cpu_current_top_of_stack), %rsp
217
218 /* Construct struct pt_regs on stack */
219 pushq $__USER_DS /* pt_regs->ss */
220 pushq PER_CPU_VAR(rsp_scratch) /* pt_regs->sp */
221 pushq %r11 /* pt_regs->flags */
222 pushq $__USER_CS /* pt_regs->cs */
223 pushq %rcx /* pt_regs->ip */
224GLOBAL(entry_SYSCALL_64_after_hwframe)
225 pushq %rax /* pt_regs->orig_ax */
226
227 PUSH_AND_CLEAR_REGS rax=$-ENOSYS
228
229 TRACE_IRQS_OFF
230
231 /* IRQs are off. */
232 movq %rsp, %rdi
233 call do_syscall_64 /* returns with IRQs disabled */
234
235 TRACE_IRQS_IRETQ /* we're about to change IF */
236
237 /*
238 * Try to use SYSRET instead of IRET if we're returning to
239 * a completely clean 64-bit userspace context. If we're not,
240 * go to the slow exit path.
241 */
242 movq RCX(%rsp), %rcx
243 movq RIP(%rsp), %r11
244
245 cmpq %rcx, %r11 /* SYSRET requires RCX == RIP */
246 jne swapgs_restore_regs_and_return_to_usermode
247
248 /*
249 * On Intel CPUs, SYSRET with non-canonical RCX/RIP will #GP
250 * in kernel space. This essentially lets the user take over
251 * the kernel, since userspace controls RSP.
252 *
253 * If width of "canonical tail" ever becomes variable, this will need
254 * to be updated to remain correct on both old and new CPUs.
255 *
256 * Change top bits to match most significant bit (47th or 56th bit
257 * depending on paging mode) in the address.
258 */
259 shl $(64 - (__VIRTUAL_MASK_SHIFT+1)), %rcx
260 sar $(64 - (__VIRTUAL_MASK_SHIFT+1)), %rcx
261
262 /* If this changed %rcx, it was not canonical */
263 cmpq %rcx, %r11
264 jne swapgs_restore_regs_and_return_to_usermode
265
266 cmpq $__USER_CS, CS(%rsp) /* CS must match SYSRET */
267 jne swapgs_restore_regs_and_return_to_usermode
268
269 movq R11(%rsp), %r11
270 cmpq %r11, EFLAGS(%rsp) /* R11 == RFLAGS */
271 jne swapgs_restore_regs_and_return_to_usermode
272
273 /*
274 * SYSCALL clears RF when it saves RFLAGS in R11 and SYSRET cannot
275 * restore RF properly. If the slowpath sets it for whatever reason, we
276 * need to restore it correctly.
277 *
278 * SYSRET can restore TF, but unlike IRET, restoring TF results in a
279 * trap from userspace immediately after SYSRET. This would cause an
280 * infinite loop whenever #DB happens with register state that satisfies
281 * the opportunistic SYSRET conditions. For example, single-stepping
282 * this user code:
283 *
284 * movq $stuck_here, %rcx
285 * pushfq
286 * popq %r11
287 * stuck_here:
288 *
289 * would never get past 'stuck_here'.
290 */
291 testq $(X86_EFLAGS_RF|X86_EFLAGS_TF), %r11
292 jnz swapgs_restore_regs_and_return_to_usermode
293
294 /* nothing to check for RSP */
295
296 cmpq $__USER_DS, SS(%rsp) /* SS must match SYSRET */
297 jne swapgs_restore_regs_and_return_to_usermode
298
299 /*
300 * We win! This label is here just for ease of understanding
301 * perf profiles. Nothing jumps here.
302 */
303syscall_return_via_sysret:
304 /* rcx and r11 are already restored (see code above) */
305 UNWIND_HINT_EMPTY
306 POP_REGS pop_rdi=0 skip_r11rcx=1
307
308 /*
309 * Now all regs are restored except RSP and RDI.
310 * Save old stack pointer and switch to trampoline stack.
311 */
312 movq %rsp, %rdi
313 movq PER_CPU_VAR(cpu_tss_rw + TSS_sp0), %rsp
314
315 pushq RSP-RDI(%rdi) /* RSP */
316 pushq (%rdi) /* RDI */
317
318 /*
319 * We are on the trampoline stack. All regs except RDI are live.
320 * We can do future final exit work right here.
321 */
322 SWITCH_TO_USER_CR3_STACK scratch_reg=%rdi
323
324 popq %rdi
325 popq %rsp
326 USERGS_SYSRET64
327END(entry_SYSCALL_64)

336 * Save callee-saved registers

341 pushq %r12

343 pushq %r14

440/*
441 * Enters the IRQ stack if we're not already using it. NMI-safe. Clobbers
442 * flags and puts old RSP into old_rsp, and leaves all other GPRs alone.
443 * Requires kernel GSBASE.
444 *
445 * The invariant is that, if irq_count != -1, then the IRQ stack is in use.
446 */
447.macro ENTER_IRQ_STACK regs=1 old_rsp
448 DEBUG_ENTRY_ASSERT_IRQS_OFF
449 movq %rsp, \old_rsp
450
451 .if \regs
452 UNWIND_HINT_REGS base=\old_rsp
453 .endif
454
455 incl PER_CPU_VAR(irq_count)
456 jnz .Lirq_stack_push_old_rsp_\@
457
458 /*
459 * Right now, if we just incremented irq_count to zero, we've
460 * claimed the IRQ stack but we haven't switched to it yet.
461 *
462 * If anything is added that can interrupt us here without using IST,
463 * it must be *extremely* careful to limit its stack usage. This
464 * could include kprobes and a hypothetical future IST-less #DB
465 * handler.
466 *
467 * The OOPS unwinder relies on the word at the top of the IRQ
468 * stack linking back to the previous RSP for the entire time we're
469 * on the IRQ stack. For this to work reliably, we need to write
470 * it before we actually move ourselves to the IRQ stack.
471 */
472
473 movq \old_rsp, PER_CPU_VAR(irq_stack_union + IRQ_STACK_SIZE - 8)
474 movq PER_CPU_VAR(irq_stack_ptr), %rsp
475
476#ifdef CONFIG_DEBUG_ENTRY
477 /*
478 * If the first movq above becomes wrong due to IRQ stack layout
479 * changes, the only way we'll notice is if we try to unwind right
480 * here. Assert that we set up the stack right to catch this type
481 * of bug quickly.
482 */
483 cmpq -8(%rsp), \old_rsp
484 je .Lirq_stack_okay\@
485 ud2
486 .Lirq_stack_okay\@:
487#endif
488
489.Lirq_stack_push_old_rsp_\@:
490 pushq \old_rsp
491
492 .if \regs
493 UNWIND_HINT_REGS indirect=1
494 .endif
495.endm
496
497/*
498 * Undoes ENTER_IRQ_STACK.
499 */
500.macro LEAVE_IRQ_STACK regs=1
501 DEBUG_ENTRY_ASSERT_IRQS_OFF
502 /* We need to be off the IRQ stack before decrementing irq_count. */
503 popq %rsp
504
505 .if \regs
506 UNWIND_HINT_REGS
507 .endif
508
509 /*
510 * As in ENTER_IRQ_STACK, irq_count == 0, we are still claiming
511 * the irq stack but we're not on it.
512 */
513
514 decl PER_CPU_VAR(irq_count)
515.endm

517/*
518 * Interrupt entry/exit.
519 *
520 * Interrupt entry points save only callee clobbered registers in fast path.
521 *
522 * Entry runs with interrupts off.
523 */
524
525/* 0(%rsp): ~(interrupt number) */
526 .macro interrupt func
527 cld
528
529 testb $3, CS-ORIG_RAX(%rsp)
530 jz 1f
531 SWAPGS
532 call switch_to_thread_stack
5331:
534
535 PUSH_AND_CLEAR_REGS
536 ENCODE_FRAME_POINTER
537
538 testb $3, CS(%rsp)
539 jz 1f
540
541 /*
542 * IRQ from user mode.
543 *
544 * We need to tell lockdep that IRQs are off. We can't do this until
545 * we fix gsbase, and we should do it before enter_from_user_mode
546 * (which can take locks). Since TRACE_IRQS_OFF idempotent,
547 * the simplest way to handle it is to just call it twice if
548 * we enter from user mode. There's no reason to optimize this since
549 * TRACE_IRQS_OFF is a no-op if lockdep is off.
550 */
551 TRACE_IRQS_OFF
552
553 CALL_enter_from_user_mode
554
5551:
556 ENTER_IRQ_STACK old_rsp=%rdi
557 /* We entered an interrupt context - irqs are off: */
558 TRACE_IRQS_OFF
559
560 call \func /* rdi points to pt_regs */
561 .endm

563 /*
564 * The interrupt stubs push (~vector+0x80) onto the stack and
565 * then jump to common_interrupt.
566 */
567 .p2align CONFIG_X86_L1_CACHE_SHIFT
568common_interrupt:
569 ASM_CLAC
570 addq $-0x80, (%rsp) /* Adjust vector to [-256, -1] range */
571 interrupt do_IRQ
572 /* 0(%rsp): old RSP */
573ret_from_intr:
574 DISABLE_INTERRUPTS(CLBR_ANY)
575 TRACE_IRQS_OFF
576
577 LEAVE_IRQ_STACK
578
579 testb $3, CS(%rsp)
580 jz retint_kernel
581
582 /* Interrupt came from user space */
583GLOBAL(retint_user)
584 mov %rsp,%rdi
585 call prepare_exit_to_usermode
586 TRACE_IRQS_IRETQ
587
588GLOBAL(swapgs_restore_regs_and_return_to_usermode)
589#ifdef CONFIG_DEBUG_ENTRY
590 /* Assert that pt_regs indicates user mode. */
591 testb $3, CS(%rsp)
592 jnz 1f
593 ud2
5941:
595#endif
596 POP_REGS pop_rdi=0
597
598 /*
599 * The stack is now user RDI, orig_ax, RIP, CS, EFLAGS, RSP, SS.
600 * Save old stack pointer and switch to trampoline stack.
601 */
602 movq %rsp, %rdi
603 movq PER_CPU_VAR(cpu_tss_rw + TSS_sp0), %rsp
604
605 /* Copy the IRET frame to the trampoline stack. */
606 pushq 6*8(%rdi) /* SS */
607 pushq 5*8(%rdi) /* RSP */
608 pushq 4*8(%rdi) /* EFLAGS */
609 pushq 3*8(%rdi) /* CS */
610 pushq 2*8(%rdi) /* RIP */
611
612 /* Push user RDI on the trampoline stack. */
613 pushq (%rdi)
614
615 /*
616 * We are on the trampoline stack. All regs except RDI are live.
617 * We can do future final exit work right here.
618 */
619
620 SWITCH_TO_USER_CR3_STACK scratch_reg=%rdi
621
622 /* Restore RDI. */
623 popq %rdi
624 SWAPGS
625 INTERRUPT_RETURN

658ENTRY(native_iret)
659 UNWIND_HINT_IRET_REGS
660 /*
661 * Are we returning to a stack segment from the LDT? Note: in
662 * 64-bit mode SS:RSP on the exception stack is always valid.
663 */
664#ifdef CONFIG_X86_ESPFIX64
665 testb $4, (SS-RIP)(%rsp)
666 jnz native_irq_return_ldt
667#endif
668
669.global native_irq_return_iret
670native_irq_return_iret:
671 /*
672 * This may fault. Non-paranoid faults on return to userspace are
673 * handled by fixup_bad_iret. These include #SS, #GP, and #NP.
674 * Double-faults due to espfix64 are handled in do_double_fault.
675 * Other faults here are fatal.
676 */
677 iretq
678
679#ifdef CONFIG_X86_ESPFIX64
680native_irq_return_ldt:
681 /*
682 * We are running with user GSBASE. All GPRs contain their user
683 * values. We have a percpu ESPFIX stack that is eight slots
684 * long (see ESPFIX_STACK_SIZE). espfix_waddr points to the bottom
685 * of the ESPFIX stack.
686 *
687 * We clobber RAX and RDI in this code. We stash RDI on the
688 * normal stack and RAX on the ESPFIX stack.
689 *
690 * The ESPFIX stack layout we set up looks like this:
691 *
692 * --- top of ESPFIX stack ---
693 * SS
694 * RSP
695 * RFLAGS
696 * CS
697 * RIP <-- RSP points here when we're done
698 * RAX <-- espfix_waddr points here
699 * --- bottom of ESPFIX stack ---
700 */
701
702 pushq %rdi /* Stash user RDI */
703 SWAPGS /* to kernel GS */
704 SWITCH_TO_KERNEL_CR3 scratch_reg=%rdi /* to kernel CR3 */
705
706 movq PER_CPU_VAR(espfix_waddr), %rdi
707 movq %rax, (0*8)(%rdi) /* user RAX */
708 movq (1*8)(%rsp), %rax /* user RIP */
709 movq %rax, (1*8)(%rdi)
710 movq (2*8)(%rsp), %rax /* user CS */
711 movq %rax, (2*8)(%rdi)
712 movq (3*8)(%rsp), %rax /* user RFLAGS */
713 movq %rax, (3*8)(%rdi)
714 movq (5*8)(%rsp), %rax /* user SS */
715 movq %rax, (5*8)(%rdi)
716 movq (4*8)(%rsp), %rax /* user RSP */
717 movq %rax, (4*8)(%rdi)
718 /* Now RAX == RSP. */
719
720 andl $0xffff0000, %eax /* RAX = (RSP & 0xffff0000) */
721
722 /*
723 * espfix_stack[31:16] == 0. The page tables are set up such that
724 * (espfix_stack | (X & 0xffff0000)) points to a read-only alias of
725 * espfix_waddr for any X. That is, there are 65536 RO aliases of
726 * the same page. Set up RSP so that RSP[31:16] contains the
727 * respective 16 bits of the /userspace/ RSP and RSP nonetheless
728 * still points to an RO alias of the ESPFIX stack.
729 */
730 orq PER_CPU_VAR(espfix_stack), %rax
731
732 SWITCH_TO_USER_CR3_STACK scratch_reg=%rdi
733 SWAPGS /* to user GS */
734 popq %rdi /* Restore user RDI */
735
736 movq %rax, %rsp
737 UNWIND_HINT_IRET_REGS offset=8
738
739 /*
740 * At this point, we cannot write to the stack any more, but we can
741 * still read.
742 */
743 popq %rax /* Restore user RAX */
744
745 /*
746 * RSP now points to an ordinary IRET frame, except that the page
747 * is read-only and RSP[31:16] are preloaded with the userspace
748 * values. We can now IRET back to userspace.
749 */
750 jmp native_irq_return_iret
751#endif
752END(common_interrupt)

1005/*
1006 * A note on the "critical region" in our callback handler.
1007 * We want to avoid stacking callback handlers due to events occurring
1008 * during handling of the last event. To do this, we keep events disabled
1009 * until we've done all processing. HOWEVER, we must enable events before
1010 * popping the stack frame (can't be done atomically) and so it would still
1011 * be possible to get enough handler activations to overflow the stack.
1012 * Although unlikely, bugs of that kind are hard to track down, so we'd
1013 * like to avoid the possibility.
1014 * So, on entry to the handler we detect whether we interrupted an
1015 * existing activation in its critical region -- if so, we pop the current
1016 * activation and restart the handler using the previous one.
1017 */

1019
1020/*
1021 * Since we don't modify %rdi, evtchn_do_upall(struct *pt_regs) will
1022 * see the correct pointer to the pt_regs

1033 call xen_maybe_preempt_hcall
1034#endif

1038/*
1039 * Hypervisor uses this for application faults while it executes.
1040 * We get here for two reasons:
1041 * 1. Fault while reloading DS, ES, FS or GS
1042 * 2. Fault while executing IRET
1043 * Category 1 we do not need to fix up as Xen has already reloaded all segment
1044 * registers that could be reloaded and zeroed the others.
1045 * Category 2 we fix up by killing the current process. We cannot use the
1046 * normal Linux return path in this case because if we use the IRET hypercall
1047 * to pop the stack frame we end up in an infinite loop of failsafe callbacks.
1048 * We distinguish between categories by comparing each saved segment register
1049 * with its current contents: any discrepancy means we in category 1.
1050 */

1052 UNWIND_HINT_EMPTY
1053 movl %ds, %ecx
1054 cmpw %cx, 0x10(%rsp)
1055 jne 1f
1056 movl %es, %ecx
1057 cmpw %cx, 0x18(%rsp)
1058 jne 1f
1059 movl %fs, %ecx
1060 cmpw %cx, 0x20(%rsp)
1061 jne 1f
1062 movl %gs, %ecx
1063 cmpw %cx, 0x28(%rsp)
1064 jne 1f

1071 jmp general_protection
10721: /* Segment mismatch => Category 1 (Bad segment). Retry the IRET. */
1073 movq (%rsp), %rcx

1075 addq $0x30, %rsp
1076 UNWIND_HINT_IRET_REGS
1077 pushq $-1 /* orig_ax = -1 => not a system call */
1078 PUSH_AND_CLEAR_REGS
1079 ENCODE_FRAME_POINTER
1080 jmp error_exit

1275/*
1276 * Runs on exception stack. Xen PV does not go through this path at all,
1277 * so we can use real assembly here.
1278 *
1279 * Registers:
1280 * %r14: Used to save/restore the CR3 of the interrupted context
1281 * when PAGE_TABLE_ISOLATION is in use. Do not clobber.
1282 */
1283ENTRY(nmi)
1284 UNWIND_HINT_IRET_REGS
1285
1286 /*
1287 * We allow breakpoints in NMIs. If a breakpoint occurs, then
1288 * the iretq it performs will take us out of NMI context.
1289 * This means that we can have nested NMIs where the next
1290 * NMI is using the top of the stack of the previous NMI. We
1291 * can't let it execute because the nested NMI will corrupt the
1292 * stack of the previous NMI. NMI handlers are not re-entrant
1293 * anyway.
1294 *
1295 * To handle this case we do the following:
1296 * Check the a special location on the stack that contains
1297 * a variable that is set when NMIs are executing.
1298 * The interrupted task's stack is also checked to see if it
1299 * is an NMI stack.
1300 * If the variable is not set and the stack is not the NMI
1301 * stack then:
1302 * o Set the special variable on the stack
1303 * o Copy the interrupt frame into an "outermost" location on the
1304 * stack
1305 * o Copy the interrupt frame into an "iret" location on the stack
1306 * o Continue processing the NMI
1307 * If the variable is set or the previous stack is the NMI stack:
1308 * o Modify the "iret" location to jump to the repeat_nmi
1309 * o return back to the first NMI
1310 *
1311 * Now on exit of the first NMI, we first clear the stack variable
1312 * The NMI stack will tell any nested NMIs at that point that it is
1313 * nested. Then we pop the stack normally with iret, and if there was
1314 * a nested NMI that updated the copy interrupt stack frame, a
1315 * jump will be made to the repeat_nmi code that will handle the second
1316 * NMI.
1317 *
1318 * However, espfix prevents us from directly returning to userspace
1319 * with a single IRET instruction. Similarly, IRET to user mode
1320 * can fault. We therefore handle NMIs from user space like
1321 * other IST entries.
1322 */
1323
1324 ASM_CLAC
1325
1326 /* Use %rdx as our temp variable throughout */
1327 pushq %rdx
1328
1329 testb $3, CS-RIP+8(%rsp)
1330 jz .Lnmi_from_kernel
1331
1332 /*
1333 * NMI from user mode. We need to run on the thread stack, but we
1334 * can't go through the normal entry paths: NMIs are masked, and
1335 * we don't want to enable interrupts, because then we'll end
1336 * up in an awkward situation in which IRQs are on but NMIs
1337 * are off.
1338 *
1339 * We also must not push anything to the stack before switching
1340 * stacks lest we corrupt the "NMI executing" variable.
1341 */
1342
1343 swapgs
1344 cld
1345 SWITCH_TO_KERNEL_CR3 scratch_reg=%rdx
1346 movq %rsp, %rdx
1347 movq PER_CPU_VAR(cpu_current_top_of_stack), %rsp
1348 UNWIND_HINT_IRET_REGS base=%rdx offset=8
1349 pushq 5*8(%rdx) /* pt_regs->ss */
1350 pushq 4*8(%rdx) /* pt_regs->rsp */
1351 pushq 3*8(%rdx) /* pt_regs->flags */
1352 pushq 2*8(%rdx) /* pt_regs->cs */
1353 pushq 1*8(%rdx) /* pt_regs->rip */
1354 UNWIND_HINT_IRET_REGS
1355 pushq $-1 /* pt_regs->orig_ax */
1356 PUSH_AND_CLEAR_REGS rdx=(%rdx)
1357 ENCODE_FRAME_POINTER
1358
1359 /*
1360 * At this point we no longer need to worry about stack damage
1361 * due to nesting -- we're on the normal thread stack and we're
1362 * done with the NMI stack.
1363 */
1364
1365 movq %rsp, %rdi
1366 movq $-1, %rsi
1367 call do_nmi
1368
1369 /*
1370 * Return back to user mode. We must *not* do the normal exit
1371 * work, because we don't want to enable interrupts.
1372 */
1373 jmp swapgs_restore_regs_and_return_to_usermode
1374
1375.Lnmi_from_kernel:
1376 /*
1377 * Here's what our stack frame will look like:
1378 * +---------------------------------------------------------+
1379 * | original SS |
1380 * | original Return RSP |
1381 * | original RFLAGS |
1382 * | original CS |
1383 * | original RIP |
1384 * +---------------------------------------------------------+
1385 * | temp storage for rdx |
1386 * +---------------------------------------------------------+
1387 * | "NMI executing" variable |
1388 * +---------------------------------------------------------+
1389 * | iret SS } Copied from "outermost" frame |
1390 * | iret Return RSP } on each loop iteration; overwritten |
1391 * | iret RFLAGS } by a nested NMI to force another |
1392 * | iret CS } iteration if needed. |
1393 * | iret RIP } |
1394 * +---------------------------------------------------------+
1395 * | outermost SS } initialized in first_nmi; |
1396 * | outermost Return RSP } will not be changed before |
1397 * | outermost RFLAGS } NMI processing is done. |
1398 * | outermost CS } Copied to "iret" frame on each |
1399 * | outermost RIP } iteration. |
1400 * +---------------------------------------------------------+
1401 * | pt_regs |
1402 * +---------------------------------------------------------+
1403 *
1404 * The "original" frame is used by hardware. Before re-enabling
1405 * NMIs, we need to be done with it, and we need to leave enough
1406 * space for the asm code here.
1407 *
1408 * We return by executing IRET while RSP points to the "iret" frame.
1409 * That will either return for real or it will loop back into NMI
1410 * processing.
1411 *
1412 * The "outermost" frame is copied to the "iret" frame on each
1413 * iteration of the loop, so each iteration starts with the "iret"
1414 * frame pointing to the final return target.
1415 */
1416
1417 /*
1418 * Determine whether we're a nested NMI.
1419 *
1420 * If we interrupted kernel code between repeat_nmi and
1421 * end_repeat_nmi, then we are a nested NMI. We must not
1422 * modify the "iret" frame because it's being written by
1423 * the outer NMI. That's okay; the outer NMI handler is
1424 * about to about to call do_nmi anyway, so we can just
1425 * resume the outer NMI.
1426 */
1427
1428 movq $repeat_nmi, %rdx
1429 cmpq 8(%rsp), %rdx
1430 ja 1f
1431 movq $end_repeat_nmi, %rdx
1432 cmpq 8(%rsp), %rdx
1433 ja nested_nmi_out
14341:
1435
1436 /*
1437 * Now check "NMI executing". If it's set, then we're nested.
1438 * This will not detect if we interrupted an outer NMI just
1439 * before IRET.
1440 */
1441 cmpl $1, -8(%rsp)
1442 je nested_nmi
1443
1444 /*
1445 * Now test if the previous stack was an NMI stack. This covers
1446 * the case where we interrupt an outer NMI after it clears
1447 * "NMI executing" but before IRET. We need to be careful, though:
1448 * there is one case in which RSP could point to the NMI stack
1449 * despite there being no NMI active: naughty userspace controls
1450 * RSP at the very beginning of the SYSCALL targets. We can
1451 * pull a fast one on naughty userspace, though: we program
1452 * SYSCALL to mask DF, so userspace cannot cause DF to be set
1453 * if it controls the kernel's RSP. We set DF before we clear
1454 * "NMI executing".
1455 */
1456 lea 6*8(%rsp), %rdx
1457 /* Compare the NMI stack (rdx) with the stack we came from (4*8(%rsp)) */
1458 cmpq %rdx, 4*8(%rsp)
1459 /* If the stack pointer is above the NMI stack, this is a normal NMI */
1460 ja first_nmi
1461
1462 subq $EXCEPTION_STKSZ, %rdx
1463 cmpq %rdx, 4*8(%rsp)
1464 /* If it is below the NMI stack, it is a normal NMI */
1465 jb first_nmi
1466
1467 /* Ah, it is within the NMI stack. */
1468
1469 testb $(X86_EFLAGS_DF >> 8), (3*8 + 1)(%rsp)
1470 jz first_nmi /* RSP was user controlled. */
1471
1472 /* This is a nested NMI. */
1473
1474nested_nmi:
1475 /*
1476 * Modify the "iret" frame to point to repeat_nmi, forcing another
1477 * iteration of NMI handling.
1478 */
1479 subq $8, %rsp
1480 leaq -10*8(%rsp), %rdx
1481 pushq $__KERNEL_DS
1482 pushq %rdx
1483 pushfq
1484 pushq $__KERNEL_CS
1485 pushq $repeat_nmi
1486
1487 /* Put stack back */
1488 addq $(6*8), %rsp
1489
1490nested_nmi_out:
1491 popq %rdx
1492
1493 /* We are returning to kernel mode, so this cannot result in a fault. */
1494 iretq
1495
1496first_nmi:
1497 /* Restore rdx. */
1498 movq (%rsp), %rdx
1499
1500 /* Make room for "NMI executing". */
1501 pushq $0
1502
1503 /* Leave room for the "iret" frame */
1504 subq $(5*8), %rsp
1505
1506 /* Copy the "original" frame to the "outermost" frame */
1507 .rept 5
1508 pushq 11*8(%rsp)
1509 .endr
1510 UNWIND_HINT_IRET_REGS
1511
1512 /* Everything up to here is safe from nested NMIs */
1513
1514#ifdef CONFIG_DEBUG_ENTRY
1515 /*
1516 * For ease of testing, unmask NMIs right away. Disabled by
1517 * default because IRET is very expensive.
1518 */
1519 pushq $0 /* SS */
1520 pushq %rsp /* RSP (minus 8 because of the previous push) */
1521 addq $8, (%rsp) /* Fix up RSP */
1522 pushfq /* RFLAGS */
1523 pushq $__KERNEL_CS /* CS */
1524 pushq $1f /* RIP */
1525 iretq /* continues at repeat_nmi below */
1526 UNWIND_HINT_IRET_REGS
15271:
1528#endif
1529
1530repeat_nmi:
1531 /*
1532 * If there was a nested NMI, the first NMI's iret will return
1533 * here. But NMIs are still enabled and we can take another
1534 * nested NMI. The nested NMI checks the interrupted RIP to see
1535 * if it is between repeat_nmi and end_repeat_nmi, and if so
1536 * it will just return, as we are about to repeat an NMI anyway.
1537 * This makes it safe to copy to the stack frame that a nested
1538 * NMI will update.
1539 *
1540 * RSP is pointing to "outermost RIP". gsbase is unknown, but, if
1541 * we're repeating an NMI, gsbase has the same value that it had on
1542 * the first iteration. paranoid_entry will load the kernel
1543 * gsbase if needed before we call do_nmi. "NMI executing"
1544 * is zero.
1545 */
1546 movq $1, 10*8(%rsp) /* Set "NMI executing". */
1547
1548 /*
1549 * Copy the "outermost" frame to the "iret" frame. NMIs that nest
1550 * here must not modify the "iret" frame while we're writing to
1551 * it or it will end up containing garbage.
1552 */
1553 addq $(10*8), %rsp
1554 .rept 5
1555 pushq -6*8(%rsp)
1556 .endr
1557 subq $(5*8), %rsp
1558end_repeat_nmi:
1559
1560 /*
1561 * Everything below this point can be preempted by a nested NMI.
1562 * If this happens, then the inner NMI will change the "iret"
1563 * frame to point back to repeat_nmi.
1564 */
1565 pushq $-1 /* ORIG_RAX: no syscall to restart */
1566
1567 /*
1568 * Use paranoid_entry to handle SWAPGS, but no need to use paranoid_exit
1569 * as we should not be calling schedule in NMI context.
1570 * Even with normal interrupts enabled. An NMI should not be
1571 * setting NEED_RESCHED or anything that normal interrupts and
1572 * exceptions might do.
1573 */
1574 call paranoid_entry
1575 UNWIND_HINT_REGS
1576
1577 /* paranoidentry do_nmi, 0; without TRACE_IRQS_OFF */
1578 movq %rsp, %rdi
1579 movq $-1, %rsi
1580 call do_nmi
1581
1582 RESTORE_CR3 scratch_reg=%r15 save_reg=%r14
1583
1584 testl %ebx, %ebx /* swapgs needed? */
1585 jnz nmi_restore
1586nmi_swapgs:
1587 SWAPGS_UNSAFE_STACK
1588nmi_restore:
1589 POP_REGS
1590
1591 /*
1592 * Skip orig_ax and the "outermost" frame to point RSP at the "iret"
1593 * at the "iret" frame.
1594 */
1595 addq $6*8, %rsp
1596
1597 /*
1598 * Clear "NMI executing". Set DF first so that we can easily
1599 * distinguish the remaining code between here and IRET from
1600 * the SYSCALL entry and exit paths.
1601 *
1602 * We arguably should just inspect RIP instead, but I (Andy) wrote
1603 * this code when I had the misapprehension that Xen PV supported
1604 * NMIs, and Xen PV would break that approach.
1605 */
1606 std
1607 movq $0, 5*8(%rsp) /* clear "NMI executing" */
1608
1609 /*
1610 * iretq reads the "iret" frame and exits the NMI stack in a
1611 * single instruction. We are returning to kernel mode, so this
1612 * cannot result in a fault. Similarly, we don't need to worry
1613 * about espfix64 on the way back to kernel mode.
1614 */
1615 iretq
1616END(nmi)
