13notrace long
14__vdso_getcpu(unsigned *cpu, unsigned *node, struct getcpu_cache *unused)
15{
16 unsigned int p;
17
18 p = __getcpu();
19
20 if (cpu)
21 *cpu = p & VGETCPU_CPU_MASK;
22 if (node)
23 *node = p >> 12;
24 return 0;
25}
