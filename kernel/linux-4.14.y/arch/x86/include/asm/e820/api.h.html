45/*
46 * Returns true iff the specified range [start,end) is completely contained inside
47 * the ISA region.
48 */
49static inline bool is_ISA_range(u64 start, u64 end)
50{
51 return start >= ISA_START_ADDRESS && end <= ISA_END_ADDRESS;
52}
