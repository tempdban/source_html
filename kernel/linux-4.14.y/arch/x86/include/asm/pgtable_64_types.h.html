51#endif /* CONFIG_X86_5LEVEL */

53/*
54 * 3rd level page
55 */
56#define PUD_SHIFT 30

58
59/*
60 * PMD_SHIFT determines the size of the area a middle-level

63#define PMD_SHIFT 21
64#define PTRS_PER_PMD 512

66/*
67 * entries per page directory level
