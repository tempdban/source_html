18 __field(u64, xcomp_bv)

20
21 TP_fast_assign(
22 __entry->fpu = fpu;
23 __entry->initialized = fpu->initialized;
24 if (boot_cpu_has(X86_FEATURE_OSXSAVE)) {
25 __entry->xfeatures = fpu->state.xsave.header.xfeatures;
26 __entry->xcomp_bv = fpu->state.xsave.header.xcomp_bv;
27 }
28 ),
29 TP_printk("x86/fpu: %p initialized: %d xfeatures: %llx xcomp_bv: %llx",
30 __entry->fpu,
31 __entry->initialized,
32 __entry->xfeatures,
33 __entry->xcomp_bv
34 )
35);
