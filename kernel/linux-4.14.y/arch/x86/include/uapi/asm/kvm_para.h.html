8/* This CPUID returns the signature 'KVMKVMKVM' in ebx, ecx, and edx. It
9 * should be used to determine that a VM is running under KVM.
10 */
11#define KVM_CPUID_SIGNATURE 0x40000000
12
13/* This CPUID returns a feature bitmap in eax. Before enabling a particular
14 * paravirtualization, the appropriate feature bit should be checked.
15 */
16#define KVM_CPUID_FEATURES 0x40000001
17#define KVM_FEATURE_CLOCKSOURCE 0
18#define KVM_FEATURE_NOP_IO_DELAY 1
19#define KVM_FEATURE_MMU_OP 2
20/* This indicates that the new set of kvmclock msrs
21 * are available. The use of 0x11 and 0x12 is deprecated
22 */
23#define KVM_FEATURE_CLOCKSOURCE2 3
24#define KVM_FEATURE_ASYNC_PF 4
25#define KVM_FEATURE_STEAL_TIME 5
26#define KVM_FEATURE_PV_EOI 6
27#define KVM_FEATURE_PV_UNHALT 7
28#define KVM_FEATURE_ASYNC_PF_VMEXIT 10
29
30/* The last 8 bits are used to indicate how to interpret the flags field
31 * in pvclock structure. If no bits are set, all flags are ignored.
32 */
33#define KVM_FEATURE_CLOCKSOURCE_STABLE_BIT 24
34
35#define MSR_KVM_WALL_CLOCK 0x11
36#define MSR_KVM_SYSTEM_TIME 0x12
37
38#define KVM_MSR_ENABLED 1
39/* Custom MSRs falls in the range 0x4b564d00-0x4b564dff */
40#define MSR_KVM_WALL_CLOCK_NEW 0x4b564d00
41#define MSR_KVM_SYSTEM_TIME_NEW 0x4b564d01
42#define MSR_KVM_ASYNC_PF_EN 0x4b564d02
43#define MSR_KVM_STEAL_TIME 0x4b564d03
44#define MSR_KVM_PV_EOI_EN 0x4b564d04

46struct kvm_steal_time {
47 __u64 steal;
48 __u32 version;
49 __u32 flags;
50 __u8 preempted;
51 __u8 u8_pad[3];
52 __u32 pad[11];
53};

55#define KVM_CLOCK_PAIRING_WALLCLOCK 0
56struct kvm_clock_pairing {
57 __s64 sec;
58 __s64 nsec;
59 __u64 tsc;
60 __u32 flags;
61 __u32 pad[9];
62};

64#define KVM_STEAL_ALIGNMENT_BITS 5
65#define KVM_STEAL_VALID_BITS ((-1ULL << (KVM_STEAL_ALIGNMENT_BITS + 1)))
66#define KVM_STEAL_RESERVED_MASK (((1 << KVM_STEAL_ALIGNMENT_BITS) - 1 ) << 1)

68#define KVM_MAX_MMU_OP_BATCH 32
69
70#define KVM_ASYNC_PF_ENABLED (1 << 0)
71#define KVM_ASYNC_PF_SEND_ALWAYS (1 << 1)
72#define KVM_ASYNC_PF_DELIVERY_AS_PF_VMEXIT (1 << 2)

74/* Operations for KVM_HC_MMU_OP */
75#define KVM_MMU_OP_WRITE_PTE 1
76#define KVM_MMU_OP_FLUSH_TLB 2
77#define KVM_MMU_OP_RELEASE_PT 3
78
79/* Payload for KVM_HC_MMU_OP */
80struct kvm_mmu_op_header {
81 __u32 op;
82 __u32 pad;
83};
84
85struct kvm_mmu_op_write_pte {
86 struct kvm_mmu_op_header header;
87 __u64 pte_phys;
88 __u64 pte_val;
89};
90
91struct kvm_mmu_op_flush_tlb {
92 struct kvm_mmu_op_header header;
93};
94
95struct kvm_mmu_op_release_pt {
96 struct kvm_mmu_op_header header;
97 __u64 pt_phys;
98};
