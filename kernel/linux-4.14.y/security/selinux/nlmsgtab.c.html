141static int nlmsg_perm(u16 nlmsg_type, u32 *perm, const struct nlmsg_perm *tab, size_t tabsize)
142{
143 int i, err = -EINVAL;
144
145 for (i = 0; i < tabsize/sizeof(struct nlmsg_perm); i++)
146 if (nlmsg_type == tab[i].nlmsg_type) {
147 *perm = tab[i].perm;
148 err = 0;
149 break;
150 }
151
152 return err;
153}
154
155int selinux_nlmsg_lookup(u16 sclass, u16 nlmsg_type, u32 *perm)
156{
157 int err = 0;
158
159 switch (sclass) {
160 case SECCLASS_NETLINK_ROUTE_SOCKET:
161 /* RTM_MAX always point to RTM_SETxxxx, ie RTM_NEWxxx + 3 */
162 BUILD_BUG_ON(RTM_MAX != (RTM_NEWCACHEREPORT + 3));
163 err = nlmsg_perm(nlmsg_type, perm, nlmsg_route_perms,
164 sizeof(nlmsg_route_perms));
165 break;
166
167 case SECCLASS_NETLINK_TCPDIAG_SOCKET:
168 err = nlmsg_perm(nlmsg_type, perm, nlmsg_tcpdiag_perms,
169 sizeof(nlmsg_tcpdiag_perms));
170 break;
171
172 case SECCLASS_NETLINK_XFRM_SOCKET:
173 BUILD_BUG_ON(XFRM_MSG_MAX != XFRM_MSG_MAPPING);
174 err = nlmsg_perm(nlmsg_type, perm, nlmsg_xfrm_perms,
175 sizeof(nlmsg_xfrm_perms));
176 break;
177
178 case SECCLASS_NETLINK_AUDIT_SOCKET:
179 if ((nlmsg_type >= AUDIT_FIRST_USER_MSG &&
180 nlmsg_type <= AUDIT_LAST_USER_MSG) ||
181 (nlmsg_type >= AUDIT_FIRST_USER_MSG2 &&
182 nlmsg_type <= AUDIT_LAST_USER_MSG2)) {
183 *perm = NETLINK_AUDIT_SOCKET__NLMSG_RELAY;
184 } else {
185 err = nlmsg_perm(nlmsg_type, perm, nlmsg_audit_perms,
186 sizeof(nlmsg_audit_perms));
187 }
188 break;
189
190 /* No messaging from userspace, or class unknown/unhandled */
191 default:
192 err = -ENOENT;
193 break;
194 }
195
196 return err;
197}
