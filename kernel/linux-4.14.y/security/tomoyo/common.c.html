187/**
188 * tomoyo_addprintf - strncat()-like-snprintf().
189 *
190 * @buffer: Buffer to write to. Must be '\0'-terminated.
191 * @len: Size of @buffer.
192 * @fmt: The printf()'s format string, followed by parameters.
193 *
194 * Returns nothing.
195 */
196static void tomoyo_addprintf(char *buffer, int len, const char *fmt, ...)
197{
198 va_list args;
199 const int pos = strlen(buffer);
200 va_start(args, fmt);
201 vsnprintf(buffer + pos, len - pos - 1, fmt, args);
202 va_end(args);
203}

205/**
206 * tomoyo_flush - Flush queued string to userspace's buffer.
207 *
208 * @head: Pointer to "struct tomoyo_io_buffer".
209 *
210 * Returns true if all data was flushed, false otherwise.
211 */
212static bool tomoyo_flush(struct tomoyo_io_buffer *head)
213{
214 while (head->r.w_pos) {
215 const char *w = head->r.w[0];
216 size_t len = strlen(w);
217 if (len) {
218 if (len > head->read_user_buf_avail)
219 len = head->read_user_buf_avail;
220 if (!len)
221 return false;
222 if (copy_to_user(head->read_user_buf, w, len))
223 return false;
224 head->read_user_buf_avail -= len;
225 head->read_user_buf += len;
226 w += len;
227 }
228 head->r.w[0] = w;
229 if (*w)
230 return false;
231 /* Add '\0' for audit logs and query. */
232 if (head->poll) {
233 if (!head->read_user_buf_avail ||
234 copy_to_user(head->read_user_buf, "", 1))
235 return false;
236 head->read_user_buf_avail--;
237 head->read_user_buf++;
238 }
239 head->r.w_pos--;
240 for (len = 0; len < head->r.w_pos; len++)
241 head->r.w[len] = head->r.w[len + 1];
242 }
243 head->r.avail = 0;
244 return true;
245}

247/**
248 * tomoyo_set_string - Queue string to "struct tomoyo_io_buffer" structure.
249 *
250 * @head: Pointer to "struct tomoyo_io_buffer".
251 * @string: String to print.
252 *
253 * Note that @string has to be kept valid until @head is kfree()d.
254 * This means that char[] allocated on stack memory cannot be passed to
255 * this function. Use tomoyo_io_printf() for char[] allocated on stack memory.
256 */
257static void tomoyo_set_string(struct tomoyo_io_buffer *head, const char *string)
258{
259 if (head->r.w_pos < TOMOYO_MAX_IO_READ_QUEUE) {
260 head->r.w[head->r.w_pos++] = string;
261 tomoyo_flush(head);
262 } else
263 WARN_ON(1);
264}

269/**
270 * tomoyo_io_printf - printf() to "struct tomoyo_io_buffer" structure.
271 *
272 * @head: Pointer to "struct tomoyo_io_buffer".
273 * @fmt: The printf()'s format string, followed by parameters.
274 */
275static void tomoyo_io_printf(struct tomoyo_io_buffer *head, const char *fmt,
276 ...)
277{
278 va_list args;
279 size_t len;
280 size_t pos = head->r.avail;
281 int size = head->readbuf_size - pos;
282 if (size <= 0)
283 return;
284 va_start(args, fmt);
285 len = vsnprintf(head->read_buf + pos, size, fmt, args) + 1;
286 va_end(args);
287 if (pos + len >= head->readbuf_size) {
288 WARN_ON(1);
289 return;
290 }
291 head->r.avail += len;
292 tomoyo_set_string(head, head->read_buf + pos);
293}

415/**
416 * tomoyo_print_number_union_nospace - Print a tomoyo_number_union without a space.
417 *
418 * @head: Pointer to "struct tomoyo_io_buffer".
419 * @ptr: Pointer to "struct tomoyo_number_union".
420 *
421 * Returns nothing.
422 */
423static void tomoyo_print_number_union_nospace
424(struct tomoyo_io_buffer *head, const struct tomoyo_number_union *ptr)
425{
426 if (ptr->group) {
427 tomoyo_set_string(head, "@");
428 tomoyo_set_string(head, ptr->group->group_name->name);
429 } else {
430 int i;
431 unsigned long min = ptr->values[0];
432 const unsigned long max = ptr->values[1];
433 u8 min_type = ptr->value_type[0];
434 const u8 max_type = ptr->value_type[1];
435 char buffer[128];
436 buffer[0] = '\0';
437 for (i = 0; i < 2; i++) {
438 switch (min_type) {
439 case TOMOYO_VALUE_TYPE_HEXADECIMAL:
440 tomoyo_addprintf(buffer, sizeof(buffer),
441 "0x%lX", min);
442 break;
443 case TOMOYO_VALUE_TYPE_OCTAL:
444 tomoyo_addprintf(buffer, sizeof(buffer),
445 "0%lo", min);
446 break;
447 default:
448 tomoyo_addprintf(buffer, sizeof(buffer), "%lu",
449 min);
450 break;
451 }
452 if (min == max && min_type == max_type)
453 break;
454 tomoyo_addprintf(buffer, sizeof(buffer), "-");
455 min_type = max_type;
456 min = max;
457 }
458 tomoyo_io_printf(head, "%s", buffer);
459 }
460}

477/**
478 * tomoyo_assign_profile - Create a new profile.
479 *
480 * @ns: Pointer to "struct tomoyo_policy_namespace".
481 * @profile: Profile number to create.
482 *
483 * Returns pointer to "struct tomoyo_profile" on success, NULL otherwise.
484 */
485static struct tomoyo_profile *tomoyo_assign_profile
486(struct tomoyo_policy_namespace *ns, const unsigned int profile)
487{
488 struct tomoyo_profile *ptr;
489 struct tomoyo_profile *entry;
490 if (profile >= TOMOYO_MAX_PROFILES)
491 return NULL;
492 ptr = ns->profile_ptr[profile];
493 if (ptr)
494 return ptr;
495 entry = kzalloc(sizeof(*entry), GFP_NOFS);
496 if (mutex_lock_interruptible(&tomoyo_policy_lock))
497 goto out;
498 ptr = ns->profile_ptr[profile];
499 if (!ptr && tomoyo_memory_ok(entry)) {
500 ptr = entry;
501 ptr->default_config = TOMOYO_CONFIG_DISABLED |
502 TOMOYO_CONFIG_WANT_GRANT_LOG |
503 TOMOYO_CONFIG_WANT_REJECT_LOG;
504 memset(ptr->config, TOMOYO_CONFIG_USE_DEFAULT,
505 sizeof(ptr->config));
506 ptr->pref[TOMOYO_PREF_MAX_AUDIT_LOG] =
507 CONFIG_SECURITY_TOMOYO_MAX_AUDIT_LOG;
508 ptr->pref[TOMOYO_PREF_MAX_LEARNING_ENTRY] =
509 CONFIG_SECURITY_TOMOYO_MAX_ACCEPT_ENTRY;
510 mb(); /* Avoid out-of-order execution. */
511 ns->profile_ptr[profile] = ptr;
512 entry = NULL;
513 }
514 mutex_unlock(&tomoyo_policy_lock);
515 out:
516 kfree(entry);
517 return ptr;
518}

576/**
577 * tomoyo_set_mode - Set mode for specified profile.
578 *
579 * @name: Name of functionality.
580 * @value: Mode for @name.
581 * @profile: Pointer to "struct tomoyo_profile".
582 *
583 * Returns 0 on success, negative value otherwise.
584 */
585static int tomoyo_set_mode(char *name, const char *value,
586 struct tomoyo_profile *profile)
587{
588 u8 i;
589 u8 config;
590 if (!strcmp(name, "CONFIG")) {
591 i = TOMOYO_MAX_MAC_INDEX + TOMOYO_MAX_MAC_CATEGORY_INDEX;
592 config = profile->default_config;
593 } else if (tomoyo_str_starts(&name, "CONFIG::")) {
594 config = 0;
595 for (i = 0; i < TOMOYO_MAX_MAC_INDEX
596 + TOMOYO_MAX_MAC_CATEGORY_INDEX; i++) {
597 int len = 0;
598 if (i < TOMOYO_MAX_MAC_INDEX) {
599 const u8 c = tomoyo_index2category[i];
600 const char *category =
601 tomoyo_category_keywords[c];
602 len = strlen(category);
603 if (strncmp(name, category, len) ||
604 name[len++] != ':' || name[len++] != ':')
605 continue;
606 }
607 if (strcmp(name + len, tomoyo_mac_keywords[i]))
608 continue;
609 config = profile->config[i];
610 break;
611 }
612 if (i == TOMOYO_MAX_MAC_INDEX + TOMOYO_MAX_MAC_CATEGORY_INDEX)
613 return -EINVAL;
614 } else {
615 return -EINVAL;
616 }
617 if (strstr(value, "use_default")) {
618 config = TOMOYO_CONFIG_USE_DEFAULT;
619 } else {
620 u8 mode;
621 for (mode = 0; mode < 4; mode++)
622 if (strstr(value, tomoyo_mode[mode]))
623 /*
624 * Update lower 3 bits in order to distinguish
625 * 'config' from 'TOMOYO_CONFIG_USE_DEAFULT'.
626 */
627 config = (config & ~7) | mode;
628 if (config != TOMOYO_CONFIG_USE_DEFAULT) {
629 switch (tomoyo_find_yesno(value, "grant_log")) {
630 case 1:
631 config |= TOMOYO_CONFIG_WANT_GRANT_LOG;
632 break;
633 case 0:
634 config &= ~TOMOYO_CONFIG_WANT_GRANT_LOG;
635 break;
636 }
637 switch (tomoyo_find_yesno(value, "reject_log")) {
638 case 1:
639 config |= TOMOYO_CONFIG_WANT_REJECT_LOG;
640 break;
641 case 0:
642 config &= ~TOMOYO_CONFIG_WANT_REJECT_LOG;
643 break;
644 }
645 }
646 }
647 if (i < TOMOYO_MAX_MAC_INDEX + TOMOYO_MAX_MAC_CATEGORY_INDEX)
648 profile->config[i] = config;
649 else if (config != TOMOYO_CONFIG_USE_DEFAULT)
650 profile->default_config = config;
651 return 0;
652}

722/**
723 * tomoyo_read_profile - Read profile table.
724 *
725 * @head: Pointer to "struct tomoyo_io_buffer".
726 *
727 * Returns nothing.
728 */
729static void tomoyo_read_profile(struct tomoyo_io_buffer *head)
730{
731 u8 index;
732 struct tomoyo_policy_namespace *ns =
733 container_of(head->r.ns, typeof(*ns), namespace_list);
734 const struct tomoyo_profile *profile;
735 if (head->r.eof)
736 return;
737 next:
738 index = head->r.index;
739 profile = ns->profile_ptr[index];
740 switch (head->r.step) {
741 case 0:
742 tomoyo_print_namespace(head);
743 tomoyo_io_printf(head, "PROFILE_VERSION=%u\n",
744 ns->profile_version);
745 head->r.step++;
746 break;
747 case 1:
748 for ( ; head->r.index < TOMOYO_MAX_PROFILES;
749 head->r.index++)
750 if (ns->profile_ptr[head->r.index])
751 break;
752 if (head->r.index == TOMOYO_MAX_PROFILES) {
753 head->r.eof = true;
754 return;
755 }
756 head->r.step++;
757 break;
758 case 2:
759 {
760 u8 i;
761 const struct tomoyo_path_info *comment =
762 profile->comment;
763 tomoyo_print_namespace(head);
764 tomoyo_io_printf(head, "%u-COMMENT=", index);
765 tomoyo_set_string(head, comment ? comment->name : "");
766 tomoyo_set_lf(head);
767 tomoyo_print_namespace(head);
768 tomoyo_io_printf(head, "%u-PREFERENCE={ ", index);
769 for (i = 0; i < TOMOYO_MAX_PREF; i++)
770 tomoyo_io_printf(head, "%s=%u ",
771 tomoyo_pref_keywords[i],
772 profile->pref[i]);
773 tomoyo_set_string(head, "}\n");
774 head->r.step++;
775 }
776 break;
777 case 3:
778 {
779 tomoyo_print_namespace(head);
780 tomoyo_io_printf(head, "%u-%s", index, "CONFIG");
781 tomoyo_print_config(head, profile->default_config);
782 head->r.bit = 0;
783 head->r.step++;
784 }
785 break;
786 case 4:
787 for ( ; head->r.bit < TOMOYO_MAX_MAC_INDEX
788 + TOMOYO_MAX_MAC_CATEGORY_INDEX; head->r.bit++) {
789 const u8 i = head->r.bit;
790 const u8 config = profile->config[i];
791 if (config == TOMOYO_CONFIG_USE_DEFAULT)
792 continue;
793 tomoyo_print_namespace(head);
794 if (i < TOMOYO_MAX_MAC_INDEX)
795 tomoyo_io_printf(head, "%u-CONFIG::%s::%s",
796 index,
797 tomoyo_category_keywords
798 [tomoyo_index2category[i]],
799 tomoyo_mac_keywords[i]);
800 else
801 tomoyo_io_printf(head, "%u-CONFIG::%s", index,
802 tomoyo_mac_keywords[i]);
803 tomoyo_print_config(head, config);
804 head->r.bit++;
805 break;
806 }
807 if (head->r.bit == TOMOYO_MAX_MAC_INDEX
808 + TOMOYO_MAX_MAC_CATEGORY_INDEX) {
809 head->r.index++;
810 head->r.step = 1;
811 }
812 break;
813 }
814 if (tomoyo_flush(head))
815 goto next;
816}

911/**
912 * tomoyo_manager - Check whether the current process is a policy manager.
913 *
914 * Returns true if the current process is permitted to modify policy
915 * via /sys/kernel/security/tomoyo/ interface.
916 *
917 * Caller holds tomoyo_read_lock().
918 */
919static bool tomoyo_manager(void)
920{
921 struct tomoyo_manager *ptr;
922 const char *exe;
923 const struct task_struct *task = current;
924 const struct tomoyo_path_info *domainname = tomoyo_domain()->domainname;
925 bool found = false;
926
927 if (!tomoyo_policy_loaded)
928 return true;
929 if (!tomoyo_manage_by_non_root &&
930 (!uid_eq(task->cred->uid, GLOBAL_ROOT_UID) ||
931 !uid_eq(task->cred->euid, GLOBAL_ROOT_UID)))
932 return false;
933 exe = tomoyo_get_exe();
934 if (!exe)
935 return false;
936 list_for_each_entry_rcu(ptr, &tomoyo_kernel_namespace.
937 policy_list[TOMOYO_ID_MANAGER], head.list) {
938 if (!ptr->head.is_deleted &&
939 (!tomoyo_pathcmp(domainname, ptr->manager) ||
940 !strcmp(exe, ptr->manager->name))) {
941 found = true;
942 break;
943 }
944 }
945 if (!found) { /* Reduce error messages. */
946 static pid_t last_pid;
947 const pid_t pid = current->pid;
948 if (last_pid != pid) {
949 printk(KERN_WARNING "%s ( %s ) is not permitted to "
950 "update policies.\n", domainname->name, exe);
951 last_pid = pid;
952 }
953 }
954 kfree(exe);
955 return found;
956}

973{

976 bool global_pid = false;

978 return false;
979 data += 7;

981 (global_pid = true, sscanf(data, "global-pid=%u", &pid) == 1)) {

1009 if (domain && domain->is_deleted)
1010 tomoyo_io_printf(head, "# This is a deleted domain.\n");

1012}

1193/**
1194 * tomoyo_print_condition - Print condition part.
1195 *
1196 * @head: Pointer to "struct tomoyo_io_buffer".
1197 * @cond: Pointer to "struct tomoyo_condition".
1198 *
1199 * Returns true on success, false otherwise.
1200 */
1201static bool tomoyo_print_condition(struct tomoyo_io_buffer *head,
1202 const struct tomoyo_condition *cond)
1203{
1204 switch (head->r.cond_step) {
1205 case 0:
1206 head->r.cond_index = 0;
1207 head->r.cond_step++;
1208 if (cond->transit) {
1209 tomoyo_set_space(head);
1210 tomoyo_set_string(head, cond->transit->name);
1211 }
1212 /* fall through */
1213 case 1:
1214 {
1215 const u16 condc = cond->condc;
1216 const struct tomoyo_condition_element *condp =
1217 (typeof(condp)) (cond + 1);
1218 const struct tomoyo_number_union *numbers_p =
1219 (typeof(numbers_p)) (condp + condc);
1220 const struct tomoyo_name_union *names_p =
1221 (typeof(names_p))
1222 (numbers_p + cond->numbers_count);
1223 const struct tomoyo_argv *argv =
1224 (typeof(argv)) (names_p + cond->names_count);
1225 const struct tomoyo_envp *envp =
1226 (typeof(envp)) (argv + cond->argc);
1227 u16 skip;
1228 for (skip = 0; skip < head->r.cond_index; skip++) {
1229 const u8 left = condp->left;
1230 const u8 right = condp->right;
1231 condp++;
1232 switch (left) {
1233 case TOMOYO_ARGV_ENTRY:
1234 argv++;
1235 continue;
1236 case TOMOYO_ENVP_ENTRY:
1237 envp++;
1238 continue;
1239 case TOMOYO_NUMBER_UNION:
1240 numbers_p++;
1241 break;
1242 }
1243 switch (right) {
1244 case TOMOYO_NAME_UNION:
1245 names_p++;
1246 break;
1247 case TOMOYO_NUMBER_UNION:
1248 numbers_p++;
1249 break;
1250 }
1251 }
1252 while (head->r.cond_index < condc) {
1253 const u8 match = condp->equals;
1254 const u8 left = condp->left;
1255 const u8 right = condp->right;
1256 if (!tomoyo_flush(head))
1257 return false;
1258 condp++;
1259 head->r.cond_index++;
1260 tomoyo_set_space(head);
1261 switch (left) {
1262 case TOMOYO_ARGV_ENTRY:
1263 tomoyo_io_printf(head,
1264 "exec.argv[%lu]%s=\"",
1265 argv->index, argv->
1266 is_not ? "!" : "");
1267 tomoyo_set_string(head,
1268 argv->value->name);
1269 tomoyo_set_string(head, "\"");
1270 argv++;
1271 continue;
1272 case TOMOYO_ENVP_ENTRY:
1273 tomoyo_set_string(head,
1274 "exec.envp[\"");
1275 tomoyo_set_string(head,
1276 envp->name->name);
1277 tomoyo_io_printf(head, "\"]%s=", envp->
1278 is_not ? "!" : "");
1279 if (envp->value) {
1280 tomoyo_set_string(head, "\"");
1281 tomoyo_set_string(head, envp->
1282 value->name);
1283 tomoyo_set_string(head, "\"");
1284 } else {
1285 tomoyo_set_string(head,
1286 "NULL");
1287 }
1288 envp++;
1289 continue;
1290 case TOMOYO_NUMBER_UNION:
1291 tomoyo_print_number_union_nospace
1292 (head, numbers_p++);
1293 break;
1294 default:
1295 tomoyo_set_string(head,
1296 tomoyo_condition_keyword[left]);
1297 break;
1298 }
1299 tomoyo_set_string(head, match ? "=" : "!=");

1308 break;

1311 tomoyo_condition_keyword[right]);
1312 break;
1313 }
1314 }
1315 }
1316 head->r.cond_step++;
1317 /* fall through */
1318 case 2:
1319 if (!tomoyo_flush(head))
1320 break;
1321 head->r.cond_step++;
1322 /* fall through */
1323 case 3:
1324 if (cond->grant_log != TOMOYO_GRANTLOG_AUTO)
1325 tomoyo_io_printf(head, " grant_log=%s",
1326 tomoyo_yesno(cond->grant_log ==
1327 TOMOYO_GRANTLOG_YES));

1329 return true;
1330 }
1331 return false;

1985/**
1986 * tomoyo_supervisor - Ask for the supervisor's decision.
1987 *
1988 * @r: Pointer to "struct tomoyo_request_info".
1989 * @fmt: The printf()'s format string, followed by parameters.
1990 *
1991 * Returns 0 if the supervisor decided to permit the access request which
1992 * violated the policy in enforcing mode, TOMOYO_RETRY_REQUEST if the
1993 * supervisor decided to retry the access request which violated the policy in
1994 * enforcing mode, 0 if it is not in enforcing mode, -EPERM otherwise.
1995 */
1996int tomoyo_supervisor(struct tomoyo_request_info *r, const char *fmt, ...)
1997{
1998 va_list args;
1999 int error;
2000 int len;
2001 static unsigned int tomoyo_serial;
2002 struct tomoyo_query entry = { };
2003 bool quota_exceeded = false;
2004 va_start(args, fmt);
2005 len = vsnprintf((char *) &len, 1, fmt, args) + 1;
2006 va_end(args);
2007 /* Write /sys/kernel/security/tomoyo/audit. */
2008 va_start(args, fmt);
2009 tomoyo_write_log2(r, len, fmt, args);
2010 va_end(args);
2011 /* Nothing more to do if granted. */
2012 if (r->granted)
2013 return 0;
2014 if (r->mode)
2015 tomoyo_update_stat(r->mode);
2016 switch (r->mode) {
2017 case TOMOYO_CONFIG_ENFORCING:
2018 error = -EPERM;
2019 if (atomic_read(&tomoyo_query_observers))
2020 break;
2021 goto out;
2022 case TOMOYO_CONFIG_LEARNING:
2023 error = 0;
2024 /* Check max_learning_entry parameter. */
2025 if (tomoyo_domain_quota_is_ok(r))
2026 break;
2027 /* fall through */
2028 default:
2029 return 0;
2030 }
2031 /* Get message. */
2032 va_start(args, fmt);
2033 entry.query = tomoyo_init_log(r, len, fmt, args);
2034 va_end(args);
2035 if (!entry.query)
2036 goto out;
2037 entry.query_len = strlen(entry.query) + 1;
2038 if (!error) {
2039 tomoyo_add_entry(r->domain, entry.query);
2040 goto out;
2041 }
2042 len = tomoyo_round2(entry.query_len);
2043 entry.domain = r->domain;
2044 spin_lock(&tomoyo_query_list_lock);
2045 if (tomoyo_memory_quota[TOMOYO_MEMORY_QUERY] &&
2046 tomoyo_memory_used[TOMOYO_MEMORY_QUERY] + len
2047 >= tomoyo_memory_quota[TOMOYO_MEMORY_QUERY]) {
2048 quota_exceeded = true;
2049 } else {
2050 entry.serial = tomoyo_serial++;
2051 entry.retry = r->retry;
2052 tomoyo_memory_used[TOMOYO_MEMORY_QUERY] += len;
2053 list_add_tail(&entry.list, &tomoyo_query_list);
2054 }
2055 spin_unlock(&tomoyo_query_list_lock);
2056 if (quota_exceeded)
2057 goto out;
2058 /* Give 10 seconds for supervisor's opinion. */
2059 while (entry.timer < 10) {
2060 wake_up_all(&tomoyo_query_wait);
2061 if (wait_event_interruptible_timeout
2062 (tomoyo_answer_wait, entry.answer ||
2063 !atomic_read(&tomoyo_query_observers), HZ))
2064 break;
2065 else
2066 entry.timer++;
2067 }
2068 spin_lock(&tomoyo_query_list_lock);
2069 list_del(&entry.list);
2070 tomoyo_memory_used[TOMOYO_MEMORY_QUERY] -= len;
2071 spin_unlock(&tomoyo_query_list_lock);
2072 switch (entry.answer) {
2073 case 3: /* Asked to retry by administrator. */
2074 error = TOMOYO_RETRY_REQUEST;
2075 r->retry++;
2076 break;
2077 case 1:
2078 /* Granted by administrator. */
2079 error = 0;
2080 break;
2081 default:
2082 /* Timed out or rejected by administrator. */
2083 break;
2084 }
2085out:
2086 kfree(entry.query);
2087 return error;
2088}

2647 /* fall through */

2650 error = -EPERM;
