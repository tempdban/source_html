70enum evm_ima_xattr_type {
71 IMA_XATTR_DIGEST = 0x01,
72 EVM_XATTR_HMAC,
73 EVM_IMA_XATTR_DIGSIG,
74 IMA_XATTR_DIGEST_NG,
75 EVM_XATTR_PORTABLE_DIGSIG,
76 IMA_XATTR_LAST

79struct evm_ima_xattr_data {
80 u8 type;
81 u8 digest[SHA1_DIGEST_SIZE];
82} __packed;
83
84#define IMA_MAX_DIGEST_SIZE 64
85
86struct ima_digest_data {
87 u8 algo;
88 u8 length;
89 union {
90 struct {
91 u8 unused;
92 u8 type;
93 } sha1;
94 struct {
95 u8 type;
96 u8 algo;
97 } ng;
98 u8 data[2];
99 } xattr;
100 u8 digest[0];
101} __packed;
102
103/*
104 * signature format v2 - for using with asymmetric keys
105 */
106struct signature_v2_hdr {
107 uint8_t type; /* xattr type */
108 uint8_t version; /* signature format version */
109 uint8_t hash_algo; /* Digest algorithm [enum hash_algo] */
110 __be32 keyid; /* IMA key identifier - not X509/PGP specific */
111 __be16 sig_size; /* signature size */
112 uint8_t sig[0]; /* signature payload */
113} __packed;
114
115/* integrity data associated with an inode */
116struct integrity_iint_cache {
117 struct rb_node rb_node; /* rooted in integrity_iint_tree */
118 struct mutex mutex; /* protects: version, flags, digest */
119 struct inode *inode; /* back pointer to inode in question */
120 u64 version; /* track inode changes */
121 unsigned long flags;
122 unsigned long measured_pcrs;
123 unsigned long atomic_flags;
124 enum integrity_status ima_file_status:4;
125 enum integrity_status ima_mmap_status:4;
126 enum integrity_status ima_bprm_status:4;
127 enum integrity_status ima_read_status:4;
128 enum integrity_status ima_creds_status:4;
129 enum integrity_status evm_status:4;
130 struct ima_digest_data *ima_hash;
131};
