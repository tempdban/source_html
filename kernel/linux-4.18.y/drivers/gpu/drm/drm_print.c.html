33void __drm_printfn_seq_file(struct drm_printer *p, struct va_format *vaf)
34{
35 seq_printf(p->arg, "%pV", vaf);
36}
37EXPORT_SYMBOL(__drm_printfn_seq_file);

39void __drm_printfn_info(struct drm_printer *p, struct va_format *vaf)
40{
41 dev_info(p->arg, "[" DRM_NAME "] %pV", vaf);

45void __drm_printfn_debug(struct drm_printer *p, struct va_format *vaf)

51/**
52 * drm_printf - print to a &drm_printer stream
53 * @p: the &drm_printer
54 * @f: format string
55 */
56void drm_printf(struct drm_printer *p, const char *f, ...)
57{
58 va_list args;
59
60 va_start(args, f);
61 drm_vprintf(p, f, &args);
62 va_end(args);
63}
64EXPORT_SYMBOL(drm_printf);
65
66void drm_dev_printk(const struct device *dev, const char *level,
67 const char *format, ...)
68{
69 struct va_format vaf;
70 va_list args;
71
72 va_start(args, format);
73 vaf.fmt = format;
74 vaf.va = &args;
75
76 if (dev)
77 dev_printk(level, dev, "[" DRM_NAME ":%ps] %pV",
78 __builtin_return_address(0), &vaf);
79 else
80 printk("%s" "[" DRM_NAME ":%ps] %pV",
81 level, __builtin_return_address(0), &vaf);
82
83 va_end(args);
84}
85EXPORT_SYMBOL(drm_dev_printk);
86
87void drm_dev_dbg(const struct device *dev, unsigned int category,
88 const char *format, ...)
89{
90 struct va_format vaf;
91 va_list args;
92
93 if (!(drm_debug & category))
94 return;
95
96 va_start(args, format);
97 vaf.fmt = format;
98 vaf.va = &args;
99
100 if (dev)
101 dev_printk(KERN_DEBUG, dev, "[" DRM_NAME ":%ps] %pV",
102 __builtin_return_address(0), &vaf);
103 else
104 printk(KERN_DEBUG "[" DRM_NAME ":%ps] %pV",
105 __builtin_return_address(0), &vaf);
106
107 va_end(args);
108}
109EXPORT_SYMBOL(drm_dev_dbg);
