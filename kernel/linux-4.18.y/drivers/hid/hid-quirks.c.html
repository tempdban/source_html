949bool hid_ignore(struct hid_device *hdev)
950{
951 if (hdev->quirks & HID_QUIRK_NO_IGNORE)
952 return false;
953 if (hdev->quirks & HID_QUIRK_IGNORE)
954 return true;
955
956 switch (hdev->vendor) {
957 case USB_VENDOR_ID_CODEMERCS:
958 /* ignore all Code Mercenaries IOWarrior devices */
959 if (hdev->product >= USB_DEVICE_ID_CODEMERCS_IOW_FIRST &&
960 hdev->product <= USB_DEVICE_ID_CODEMERCS_IOW_LAST)
961 return true;
962 break;
963 case USB_VENDOR_ID_LOGITECH:
964 if (hdev->product >= USB_DEVICE_ID_LOGITECH_HARMONY_FIRST &&
965 hdev->product <= USB_DEVICE_ID_LOGITECH_HARMONY_LAST)
966 return true;
967 /*
968 * The Keene FM transmitter USB device has the same USB ID as
969 * the Logitech AudioHub Speaker, but it should ignore the hid.
970 * Check if the name is that of the Keene device.
971 * For reference: the name of the AudioHub is
972 * "HOLTEK AudioHub Speaker".
973 */
974 if (hdev->product == USB_DEVICE_ID_LOGITECH_AUDIOHUB &&
975 !strcmp(hdev->name, "HOLTEK B-LINK USB Audio "))
976 return true;
977 break;
978 case USB_VENDOR_ID_SOUNDGRAPH:
979 if (hdev->product >= USB_DEVICE_ID_SOUNDGRAPH_IMON_FIRST &&
980 hdev->product <= USB_DEVICE_ID_SOUNDGRAPH_IMON_LAST)
981 return true;
982 break;
983 case USB_VENDOR_ID_HANWANG:
984 if (hdev->product >= USB_DEVICE_ID_HANWANG_TABLET_FIRST &&
985 hdev->product <= USB_DEVICE_ID_HANWANG_TABLET_LAST)
986 return true;
987 break;
988 case USB_VENDOR_ID_JESS:
989 if (hdev->product == USB_DEVICE_ID_JESS_YUREX &&
990 hdev->type == HID_TYPE_USBNONE)
991 return true;
992 break;
993 case USB_VENDOR_ID_VELLEMAN:
994 /* These are not HID devices. They are handled by comedi. */
995 if ((hdev->product >= USB_DEVICE_ID_VELLEMAN_K8055_FIRST &&
996 hdev->product <= USB_DEVICE_ID_VELLEMAN_K8055_LAST) ||
997 (hdev->product >= USB_DEVICE_ID_VELLEMAN_K8061_FIRST &&
998 hdev->product <= USB_DEVICE_ID_VELLEMAN_K8061_LAST))
999 return true;
1000 break;
1001 case USB_VENDOR_ID_ATMEL_V_USB:
1002 /* Masterkit MA901 usb radio based on Atmel tiny85 chip and
1003 * it has the same USB ID as many Atmel V-USB devices. This
1004 * usb radio is handled by radio-ma901.c driver so we want
1005 * ignore the hid. Check the name, bus, product and ignore
1006 * if we have MA901 usb radio.
1007 */
1008 if (hdev->product == USB_DEVICE_ID_ATMEL_V_USB &&
1009 hdev->bus == BUS_USB &&
1010 strncmp(hdev->name, "www.masterkit.ru MA901", 22) == 0)
1011 return true;
1012 break;
1013 case USB_VENDOR_ID_ELAN:
1014 /*
1015 * Many Elan devices have a product id of 0x0401 and are handled
1016 * by the elan_i2c input driver. But the ACPI HID ELAN0800 dev
1017 * is not (and cannot be) handled by that driver ->
1018 * Ignore all 0x0401 devs except for the ELAN0800 dev.
1019 */
1020 if (hdev->product == 0x0401 &&
1021 strncmp(hdev->name, "ELAN0800", 8) != 0)
1022 return true;
1023 break;
1024 }
1025
1026 if (hdev->type == HID_TYPE_USBMOUSE &&
1027 hid_match_id(hdev, hid_mouse_ignore_list))
1028 return true;
1029
1030 return !!hid_match_id(hdev, hid_ignore_list);
1031}

1045/**
1046 * hid_exists_dquirk: find any dynamic quirks for a HID device
1047 * @hdev: the HID device to match
1048 *
1049 * Description:
1050 * Scans dquirks_list for a matching dynamic quirk and returns
1051 * the pointer to the relevant struct hid_device_id if found.
1052 * Must be called with a read lock held on dquirks_lock.
1053 *
1054 * Returns: NULL if no quirk found, struct hid_device_id * if found.
1055 */
1056static struct hid_device_id *hid_exists_dquirk(const struct hid_device *hdev)
1057{
1058 struct quirks_list_struct *q;
1059 struct hid_device_id *bl_entry = NULL;
1060
1061 list_for_each_entry(q, &dquirks_list, node) {
1062 if (hid_match_one_id(hdev, &q->hid_bl_item)) {
1063 bl_entry = &q->hid_bl_item;
1064 break;
1065 }
1066 }
1067
1068 if (bl_entry != NULL)
1069 dbg_hid("Found dynamic quirk 0x%lx for HID device 0x%hx:0x%hx\n",
1070 bl_entry->driver_data, bl_entry->vendor,
1071 bl_entry->product);
1072
1073 return bl_entry;
1074}

1077/**
1078 * hid_modify_dquirk: add/replace a HID quirk
1079 * @id: the HID device to match
1080 * @quirks: the unsigned long quirks value to add/replace
1081 *
1082 * Description:
1083 * If an dynamic quirk exists in memory for this device, replace its
1084 * quirks value with what was provided. Otherwise, add the quirk
1085 * to the dynamic quirks list.
1086 *
1087 * Returns: 0 OK, -error on failure.
1088 */
1089static int hid_modify_dquirk(const struct hid_device_id *id,
1090 const unsigned long quirks)
1091{
1092 struct hid_device *hdev;
1093 struct quirks_list_struct *q_new, *q;
1094 int list_edited = 0;
1095 int ret = 0;
1096
1097 hdev = kzalloc(sizeof(*hdev), GFP_KERNEL);
1098 if (!hdev)
1099 return -ENOMEM;
1100
1101 q_new = kmalloc(sizeof(struct quirks_list_struct), GFP_KERNEL);
1102 if (!q_new) {
1103 ret = -ENOMEM;
1104 goto out;
1105 }
1106
1107 hdev->bus = q_new->hid_bl_item.bus = id->bus;
1108 hdev->group = q_new->hid_bl_item.group = id->group;
1109 hdev->vendor = q_new->hid_bl_item.vendor = id->vendor;
1110 hdev->product = q_new->hid_bl_item.product = id->product;
1111 q_new->hid_bl_item.driver_data = quirks;
1112
1113 mutex_lock(&dquirks_lock);
1114
1115 list_for_each_entry(q, &dquirks_list, node) {
1116
1117 if (hid_match_one_id(hdev, &q->hid_bl_item)) {
1118
1119 list_replace(&q->node, &q_new->node);
1120 kfree(q);
1121 list_edited = 1;
1122 break;
1123
1124 }
1125
1126 }
1127
1128 if (!list_edited)
1129 list_add_tail(&q_new->node, &dquirks_list);
1130
1131 mutex_unlock(&dquirks_lock);
1132
1133 out:
1134 kfree(hdev);
1135 return ret;
1136}

1139 * hid_remove_all_dquirks: remove all runtime HID quirks from memory
1140 * @bus: bus to match against. Use HID_BUS_ANY if all need to be removed.
1141 *
1142 * Description:
1143 * Free all memory associated with dynamic quirks - called before
1144 * module unload.
1145 *
1146 */
1147static void hid_remove_all_dquirks(__u16 bus)
1148{

1150

1152 list_for_each_entry_safe(q, temp, &dquirks_list, node) {
1153 if (bus == HID_BUS_ANY || bus == q->hid_bl_item.bus) {
1154 list_del(&q->node);
1155 kfree(q);
1156 }
1157 }
1158 mutex_unlock(&dquirks_lock);
1159
1160}

1165int hid_quirks_init(char **quirks_param, __u16 bus, int count)
1166{
1167 struct hid_device_id id = { 0 };
1168 int n = 0, m;
1169 unsigned short int vendor, product;
1170 u32 quirks;
1171
1172 id.bus = bus;
1173
1174 for (; n < count && quirks_param[n]; n++) {
1175

1177 &vendor, &product, &quirks);

1180 id.product = (__u16)product;

1183 hid_modify_dquirk(&id, quirks) != 0) {

1187 }

1211 * hid_gets_squirk: return any static quirks for a HID device
1212 * @hdev: the HID device to match
1213 *
1214 * Description:
1215 * Given a HID device, return a pointer to the quirked hid_device_id entry
1216 * associated with that device.
1217 *
1218 * Returns: the quirks.
1219 */
1220static unsigned long hid_gets_squirk(const struct hid_device *hdev)
1221{
1222 const struct hid_device_id *bl_entry;
1223 unsigned long quirks = 0;
1224
1225 if (hid_match_id(hdev, hid_ignore_list))
1226 quirks |= HID_QUIRK_IGNORE;

1229 quirks |= HID_QUIRK_HAVE_SPECIAL_DRIVER;

1232 if (bl_entry != NULL)
1233 quirks |= bl_entry->driver_data;
1234
1235 if (quirks)
1236 dbg_hid("Found squirk 0x%lx for HID device 0x%hx:0x%hx\n",
1237 quirks, hdev->vendor, hdev->product);

1241/**
1242 * hid_lookup_quirk: return any quirks associated with a HID device
1243 * @hdev: the HID device to look for
1244 *
1245 * Description:
1246 * Given a HID device, return any quirks associated with that device.
1247 *
1248 * Returns: an unsigned long quirks value.
1249 */
1250unsigned long hid_lookup_quirk(const struct hid_device *hdev)
1251{
1252 unsigned long quirks = 0;
1253 const struct hid_device_id *quirk_entry = NULL;
1254
1255 /* NCR devices must not be queried for reports */
1256 if (hdev->bus == BUS_USB &&
1257 hdev->vendor == USB_VENDOR_ID_NCR &&
1258 hdev->product >= USB_DEVICE_ID_NCR_FIRST &&
1259 hdev->product <= USB_DEVICE_ID_NCR_LAST)
1260 return HID_QUIRK_NO_INIT_REPORTS;
1261
1262 /* These devices must be ignored if version (bcdDevice) is too old */
1263 if (hdev->bus == BUS_USB && hdev->vendor == USB_VENDOR_ID_JABRA) {
1264 switch (hdev->product) {
1265 case USB_DEVICE_ID_JABRA_SPEAK_410:
1266 if (hdev->version < 0x0111)
1267 return HID_QUIRK_IGNORE;
1268 break;
1269 case USB_DEVICE_ID_JABRA_SPEAK_510:
1270 if (hdev->version < 0x0214)
1271 return HID_QUIRK_IGNORE;
1272 break;
1273 }
1274 }
1275
1276 mutex_lock(&dquirks_lock);
1277 quirk_entry = hid_exists_dquirk(hdev);
1278 if (quirk_entry)
1279 quirks = quirk_entry->driver_data;
1280 else
1281 quirks = hid_gets_squirk(hdev);
1282 mutex_unlock(&dquirks_lock);
1283
1284 return quirks;
1285}
