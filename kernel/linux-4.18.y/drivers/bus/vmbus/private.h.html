110/* Amount of space available for write */
111static inline uint32_t
112vmbus_br_availwrite(const struct vmbus_br *br, uint32_t windex)
113{
114 uint32_t rindex = br->vbr->rindex;
115
116 if (windex >= rindex)
117 return br->dsize - (windex - rindex);
118 else
119 return rindex - windex;
120}
121
122static inline uint32_t
123vmbus_br_availread(const struct vmbus_br *br)
124{
125 return br->dsize - vmbus_br_availwrite(br, br->vbr->windex);
126}
