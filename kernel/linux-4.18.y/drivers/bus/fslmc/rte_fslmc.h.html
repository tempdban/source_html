41/** Device driver supports link state interrupt */

56extern struct rte_fslmc_bus rte_fslmc_bus;
57
58enum rte_dpaa2_dev_type {
59 /* Devices backed by DPDK driver */
60 DPAA2_ETH, /**< DPNI type device*/
61 DPAA2_CRYPTO, /**< DPSECI type device */
62 DPAA2_CON, /**< DPCONC type device */
63 /* Devices not backed by a DPDK driver: DPIO, DPBP, DPCI, DPMCP */
64 DPAA2_BPOOL, /**< DPBP type device */
65 DPAA2_IO, /**< DPIO type device */
66 DPAA2_CI, /**< DPCI type device */
67 DPAA2_MPORTAL, /**< DPMCP type device */
68 DPAA2_QDMA, /**< DPDMAI type device */
69 /* Unknown device placeholder */
70 DPAA2_UNKNOWN,
71 DPAA2_DEVTYPE_MAX,
72};
73
74TAILQ_HEAD(rte_dpaa2_object_list, rte_dpaa2_object);
75
76typedef int (*rte_dpaa2_obj_create_t)(int vdev_fd,
77 struct vfio_device_info *obj_info,
78 int object_id);
79
80/**
81 * A structure describing a DPAA2 object.
82 */
83struct rte_dpaa2_object {
84 TAILQ_ENTRY(rte_dpaa2_object) next; /**< Next in list. */
85 const char *name; /**< Name of Object. */
86 enum rte_dpaa2_dev_type dev_type; /**< Type of device */
87 rte_dpaa2_obj_create_t create;
88};
89
90/**
91 * A structure describing a DPAA2 device.
92 */
93struct rte_dpaa2_device {
94 TAILQ_ENTRY(rte_dpaa2_device) next; /**< Next probed DPAA2 device. */
95 struct rte_device device; /**< Inherit core device */
96 union {
97 struct rte_eth_dev *eth_dev; /**< ethernet device */
98 struct rte_cryptodev *cryptodev; /**< Crypto Device */
99 struct rte_rawdev *rawdev; /**< Raw Device */
100 };
101 enum rte_dpaa2_dev_type dev_type; /**< Device Type */
102 uint16_t object_id; /**< DPAA2 Object ID */
103 struct rte_intr_handle intr_handle; /**< Interrupt handle */
104 struct rte_dpaa2_driver *driver; /**< Associated driver */
105 char name[FSLMC_OBJECT_MAX_LEN]; /**< DPAA2 Object name*/
106};
107
108typedef int (*rte_dpaa2_probe_t)(struct rte_dpaa2_driver *dpaa2_drv,
109 struct rte_dpaa2_device *dpaa2_dev);
110typedef int (*rte_dpaa2_remove_t)(struct rte_dpaa2_device *dpaa2_dev);
111
112/**
113 * A structure describing a DPAA2 driver.
114 */
115struct rte_dpaa2_driver {
116 TAILQ_ENTRY(rte_dpaa2_driver) next; /**< Next in list. */
117 struct rte_driver driver; /**< Inherit core driver. */
118 struct rte_fslmc_bus *fslmc_bus; /**< FSLMC bus reference */
119 uint32_t drv_flags; /**< Flags for controlling device.*/
120 enum rte_dpaa2_dev_type drv_type; /**< Driver Type */
121 rte_dpaa2_probe_t probe;
122 rte_dpaa2_remove_t remove;
123};
124
125/*
126 * FSLMC bus
127 */
128struct rte_fslmc_bus {
129 struct rte_bus bus; /**< Generic Bus object */
130 struct rte_fslmc_device_list device_list;
131 /**< FSLMC DPAA2 Device list */
132 struct rte_fslmc_driver_list driver_list;
133 /**< FSLMC DPAA2 Driver list */
134 int device_count[DPAA2_DEVTYPE_MAX];
135 /**< Count of all devices scanned */
136};

174/** Helper for DPAA2 device registration from driver (eth, crypto) instance */
175#define RTE_PMD_REGISTER_DPAA2(nm, dpaa2_drv) \
176RTE_INIT(dpaa2initfn_ ##nm) \
177{\
178 (dpaa2_drv).driver.name = RTE_STR(nm);\
179 rte_fslmc_driver_register(&dpaa2_drv); \
180} \
181RTE_PMD_EXPORT_NAME(nm, __COUNTER__)
182
183/**
184 * Register a DPAA2 MC Object driver.
185 *
186 * @param mc_object
187 * A pointer to a rte_dpaa_object structure describing the mc object
188 * to be registered.
189 */
190void rte_fslmc_object_register(struct rte_dpaa2_object *object);

192/**
193 * Count of a particular type of DPAA2 device scanned on the bus.
194 *
195 * @param dev_type

199 * invalid device type.

201uint32_t rte_fslmc_get_device_count(enum rte_dpaa2_dev_type device_type);
