586/**
587 * fb_try_mode - test a video mode
588 * @var: frame buffer user defined part of display
589 * @info: frame buffer info structure
590 * @mode: frame buffer video mode structure
591 * @bpp: color depth in bits per pixel
592 *
593 * Tries a video mode to test it's validity for device @info.
594 *
595 * Returns 1 on success.
596 *
597 */
598
599static int fb_try_mode(struct fb_var_screeninfo *var, struct fb_info *info,
600 const struct fb_videomode *mode, unsigned int bpp)
601{
602 int err = 0;
603
604 DPRINTK("Trying mode %s %dx%d-%d@%d\n",
605 mode->name ? mode->name : "noname",
606 mode->xres, mode->yres, bpp, mode->refresh);
607 var->xres = mode->xres;
608 var->yres = mode->yres;
609 var->xres_virtual = mode->xres;
610 var->yres_virtual = mode->yres;
611 var->xoffset = 0;
612 var->yoffset = 0;
613 var->bits_per_pixel = bpp;
614 var->activate |= FB_ACTIVATE_TEST;
615 var->pixclock = mode->pixclock;
616 var->left_margin = mode->left_margin;
617 var->right_margin = mode->right_margin;
618 var->upper_margin = mode->upper_margin;
619 var->lower_margin = mode->lower_margin;
620 var->hsync_len = mode->hsync_len;
621 var->vsync_len = mode->vsync_len;
622 var->sync = mode->sync;
623 var->vmode = mode->vmode;
624 if (info->fbops->fb_check_var)
625 err = info->fbops->fb_check_var(var, info);
626 var->activate &= ~FB_ACTIVATE_TEST;
627 return err;
628}

630/**
631 * fb_find_mode - finds a valid video mode
632 * @var: frame buffer user defined part of display
633 * @info: frame buffer info structure
634 * @mode_option: string video mode to find
635 * @db: video mode database
636 * @dbsize: size of @db
637 * @default_mode: default video mode to fall back to
638 * @default_bpp: default color depth in bits per pixel
639 *
640 * Finds a suitable video mode, starting with the specified mode
641 * in @mode_option with fallback to @default_mode. If
642 * @default_mode fails, all modes in the video mode database will
643 * be tried.
644 *
645 * Valid mode specifiers for @mode_option:
646 *
647 * <xres>x<yres>[M][R][-<bpp>][@<refresh>][i][p][m] or
648 * <name>[-<bpp>][@<refresh>]
649 *
650 * with <xres>, <yres>, <bpp> and <refresh> decimal numbers and
651 * <name> a string.
652 *
653 * If 'M' is present after yres (and before refresh/bpp if present),
654 * the function will compute the timings using VESA(tm) Coordinated
655 * Video Timings (CVT). If 'R' is present after 'M', will compute with
656 * reduced blanking (for flatpanels). If 'i' or 'p' are present, compute
657 * interlaced or progressive mode. If 'm' is present, add margins equal
658 * to 1.8% of xres rounded down to 8 pixels, and 1.8% of yres. The chars
659 * 'i', 'p' and 'm' must be after 'M' and 'R'. Example:
660 *
661 * 1024x768MR-8@60m - Reduced blank with margins at 60Hz.
662 *
663 * NOTE: The passed struct @var is _not_ cleared! This allows you
664 * to supply values for e.g. the grayscale and accel_flags fields.
665 *
666 * Returns zero for failure, 1 if using specified @mode_option,
667 * 2 if using specified @mode_option with an ignored refresh rate,
668 * 3 if default mode is used, 4 if fall back to any valid mode.
669 *
670 */
671
672int fb_find_mode(struct fb_var_screeninfo *var,
673 struct fb_info *info, const char *mode_option,
674 const struct fb_videomode *db, unsigned int dbsize,
675 const struct fb_videomode *default_mode,
676 unsigned int default_bpp)
677{
678 int i;
679
680 /* Set up defaults */
681 if (!db) {
682 db = modedb;
683 dbsize = ARRAY_SIZE(modedb);
684 }
685
686 if (!default_mode)
687 default_mode = &db[0];
688
689 if (!default_bpp)
690 default_bpp = 8;
691
692 /* Did the user specify a video mode? */
693 if (!mode_option)
694 mode_option = fb_mode_option;
695 if (mode_option) {
696 const char *name = mode_option;
697 unsigned int namelen = strlen(name);
698 int res_specified = 0, bpp_specified = 0, refresh_specified = 0;
699 unsigned int xres = 0, yres = 0, bpp = default_bpp, refresh = 0;
700 int yres_specified = 0, cvt = 0, rb = 0;
701 int interlace_specified = 0, interlace = 0;
702 int margins = 0;
703 u32 best, diff, tdiff;
704
705 for (i = namelen-1; i >= 0; i--) {
706 switch (name[i]) {
707 case '@':
708 namelen = i;
709 if (!refresh_specified && !bpp_specified &&
710 !yres_specified) {
711 refresh = simple_strtol(&name[i+1], NULL,
712 10);
713 refresh_specified = 1;
714 if (cvt || rb)
715 cvt = 0;
716 } else
717 goto done;
718 break;
719 case '-':
720 namelen = i;
721 if (!bpp_specified && !yres_specified) {
722 bpp = simple_strtol(&name[i+1], NULL,
723 10);
724 bpp_specified = 1;
725 if (cvt || rb)
726 cvt = 0;
727 } else
728 goto done;
729 break;
730 case 'x':
731 if (!yres_specified) {
732 yres = simple_strtol(&name[i+1], NULL,
733 10);
734 yres_specified = 1;
735 } else
736 goto done;
737 break;
738 case '0' ... '9':
739 break;
740 case 'M':
741 if (!yres_specified)
742 cvt = 1;
743 break;
744 case 'R':
745 if (!cvt)
746 rb = 1;
747 break;
748 case 'm':
749 if (!cvt)
750 margins = 1;
751 break;
752 case 'p':
753 if (!cvt) {
754 interlace = 0;
755 interlace_specified = 1;
756 }
757 break;
758 case 'i':
759 if (!cvt) {
760 interlace = 1;
761 interlace_specified = 1;
762 }
763 break;
764 default:
765 goto done;
766 }
767 }
768 if (i < 0 && yres_specified) {
769 xres = simple_strtol(name, NULL, 10);
770 res_specified = 1;
771 }
772done:
773 if (cvt) {
774 struct fb_videomode cvt_mode;
775 int ret;
776
777 DPRINTK("CVT mode %dx%d@%dHz%s%s%s\n", xres, yres,
778 (refresh) ? refresh : 60,
779 (rb) ? " reduced blanking" : "",
780 (margins) ? " with margins" : "",
781 (interlace) ? " interlaced" : "");
782
783 memset(&cvt_mode, 0, sizeof(cvt_mode));
784 cvt_mode.xres = xres;
785 cvt_mode.yres = yres;
786 cvt_mode.refresh = (refresh) ? refresh : 60;
787
788 if (interlace)
789 cvt_mode.vmode |= FB_VMODE_INTERLACED;
790 else
791 cvt_mode.vmode &= ~FB_VMODE_INTERLACED;
792
793 ret = fb_find_mode_cvt(&cvt_mode, margins, rb);
794
795 if (!ret && !fb_try_mode(var, info, &cvt_mode, bpp)) {
796 DPRINTK("modedb CVT: CVT mode ok\n");
797 return 1;
798 }
799
800 DPRINTK("CVT mode invalid, getting mode from database\n");
801 }
802
803 DPRINTK("Trying specified video mode%s %ix%i\n",
804 refresh_specified ? "" : " (ignoring refresh rate)",
805 xres, yres);
806
807 if (!refresh_specified) {
808 /*
809 * If the caller has provided a custom mode database and
810 * a valid monspecs structure, we look for the mode with
811 * the highest refresh rate. Otherwise we play it safe
812 * it and try to find a mode with a refresh rate closest
813 * to the standard 60 Hz.
814 */
815 if (db != modedb &&
816 info->monspecs.vfmin && info->monspecs.vfmax &&
817 info->monspecs.hfmin && info->monspecs.hfmax &&
818 info->monspecs.dclkmax) {
819 refresh = 1000;
820 } else {
821 refresh = 60;
822 }
823 }
824
825 diff = -1;
826 best = -1;
827 for (i = 0; i < dbsize; i++) {
828 if ((name_matches(db[i], name, namelen) ||
829 (res_specified && res_matches(db[i], xres, yres))) &&
830 !fb_try_mode(var, info, &db[i], bpp)) {
831 const int db_interlace = (db[i].vmode &
832 FB_VMODE_INTERLACED ? 1 : 0);
833 int score = abs(db[i].refresh - refresh);
834
835 if (interlace_specified)
836 score += abs(db_interlace - interlace);
837
838 if (!interlace_specified ||
839 db_interlace == interlace)
840 if (refresh_specified &&
841 db[i].refresh == refresh)
842 return 1;
843
844 if (score < diff) {
845 diff = score;
846 best = i;
847 }
848 }
849 }
850 if (best != -1) {
851 fb_try_mode(var, info, &db[best], bpp);
852 return (refresh_specified) ? 2 : 1;
853 }
854
855 diff = 2 * (xres + yres);
856 best = -1;
857 DPRINTK("Trying best-fit modes\n");
858 for (i = 0; i < dbsize; i++) {
859 DPRINTK("Trying %ix%i\n", db[i].xres, db[i].yres);
860 if (!fb_try_mode(var, info, &db[i], bpp)) {
861 tdiff = abs(db[i].xres - xres) +
862 abs(db[i].yres - yres);
863
864 /*
865 * Penalize modes with resolutions smaller
866 * than requested.
867 */
868 if (xres > db[i].xres || yres > db[i].yres)
869 tdiff += xres + yres;
870
871 if (diff > tdiff) {
872 diff = tdiff;
873 best = i;
874 }
875 }
876 }
877 if (best != -1) {
878 fb_try_mode(var, info, &db[best], bpp);
879 return 5;
880 }
881 }
882
883 DPRINTK("Trying default video mode\n");
884 if (!fb_try_mode(var, info, default_mode, default_bpp))
885 return 3;
886
887 DPRINTK("Trying all modes\n");
888 for (i = 0; i < dbsize; i++)
889 if (!fb_try_mode(var, info, &db[i], default_bpp))
890 return 4;
891
892 DPRINTK("No valid mode found\n");
893 return 0;
894}

897 * fb_var_to_videomode - convert fb_var_screeninfo to fb_videomode
898 * @mode: pointer to struct fb_videomode
899 * @var: pointer to struct fb_var_screeninfo
900 */
901void fb_var_to_videomode(struct fb_videomode *mode,
902 const struct fb_var_screeninfo *var)
903{
904 u32 pixclock, hfreq, htotal, vtotal;
905
906 mode->name = NULL;
907 mode->xres = var->xres;
908 mode->yres = var->yres;
909 mode->pixclock = var->pixclock;
910 mode->hsync_len = var->hsync_len;
911 mode->vsync_len = var->vsync_len;
912 mode->left_margin = var->left_margin;
913 mode->right_margin = var->right_margin;
914 mode->upper_margin = var->upper_margin;
915 mode->lower_margin = var->lower_margin;
916 mode->sync = var->sync;
917 mode->vmode = var->vmode & FB_VMODE_MASK;
918 mode->flag = FB_MODE_IS_FROM_VAR;
919 mode->refresh = 0;
920
921 if (!var->pixclock)
922 return;
923
924 pixclock = PICOS2KHZ(var->pixclock) * 1000;
925
926 htotal = var->xres + var->right_margin + var->hsync_len +
927 var->left_margin;
928 vtotal = var->yres + var->lower_margin + var->vsync_len +
929 var->upper_margin;
930
931 if (var->vmode & FB_VMODE_INTERLACED)
932 vtotal /= 2;
933 if (var->vmode & FB_VMODE_DOUBLE)
934 vtotal *= 2;
935
936 hfreq = pixclock/htotal;
937 mode->refresh = hfreq/vtotal;
938}

989/**

991 * @var: pointer to struct fb_var_screeninfo
992 * @head: pointer to struct list_head of modelist
993 *
994 * RETURNS:
995 * struct fb_videomode, NULL if none found
996 *

999 * info->modelist are valid.
1000 *
1001 * NOTES:
1002 * Finds best matching videomode which has an equal or greater dimension than
1003 * var->xres and var->yres. If more than 1 videomode is found, will return
1004 * the videomode with the highest refresh rate
1005 */
1006const struct fb_videomode *fb_find_best_mode(const struct fb_var_screeninfo *var,
1007 struct list_head *head)

1014 list_for_each(pos, head) {
1015 u32 d;
1016

1018 mode = &modelist->mode;
1019
1020 if (mode->xres >= var->xres && mode->yres >= var->yres) {
1021 d = (mode->xres - var->xres) +
1022 (mode->yres - var->yres);
1023 if (diff > d) {
1024 diff = d;
1025 best = mode;
1026 } else if (diff == d && best &&
1027 mode->refresh > best->refresh)
1028 best = mode;
1029 }
1030 }
1031 return best;

1196 struct list_head *head)
1197{
1198 struct list_head *pos;
1199 struct fb_modelist *modelist;
1200 const struct fb_videomode *m, *m1 = NULL, *md = NULL, *best = NULL;
1201 int first = 0;
1202
1203 if (!head->prev || !head->next || list_empty(head))
1204 goto finished;
1205
1206 /* get the first detailed mode and the very first mode */
1207 list_for_each(pos, head) {
1208 modelist = list_entry(pos, struct fb_modelist, list);
1209 m = &modelist->mode;
1210
1211 if (!first) {
1212 m1 = m;
1213 first = 1;
1214 }
1215
1216 if (m->flag & FB_MODE_IS_FIRST) {
1217 md = m;
1218 break;
1219 }
1220 }
1221
1222 /* first detailed timing is preferred */
1223 if (specs->misc & FB_MISC_1ST_DETAIL) {
1224 best = md;
1225 goto finished;
1226 }
1227
1228 /* find best mode based on display width and height */
1229 if (specs->max_x && specs->max_y) {
1230 struct fb_var_screeninfo var;
1231
1232 memset(&var, 0, sizeof(struct fb_var_screeninfo));
1233 var.xres = (specs->max_x * 7200)/254;
1234 var.yres = (specs->max_y * 7200)/254;
1235 m = fb_find_best_mode(&var, head);
1236 if (m) {
1237 best = m;
1238 goto finished;
1239 }
1240 }
1241
1242 /* use first detailed mode */
1243 if (md) {
1244 best = md;
1245 goto finished;
1246 }
1247
1248 /* last resort, use the very first mode */
1249 best = m1;
