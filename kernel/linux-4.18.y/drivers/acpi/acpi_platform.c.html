27static const struct acpi_device_id forbidden_id_list[] = {
28 {"PNP0000", 0}, /* PIC */
29 {"PNP0100", 0}, /* Timer */
30 {"PNP0200", 0}, /* AT DMA Controller */
31 {"ACPI0009", 0}, /* IOxAPIC */
32 {"ACPI000A", 0}, /* IOAPIC */
33 {"", 0},
34};
35
36static void acpi_platform_fill_resource(struct acpi_device *adev,
37 const struct resource *src, struct resource *dest)
38{
39 struct device *parent;
40
41 *dest = *src;
42
43 /*
44 * If the device has parent we need to take its resources into
45 * account as well because this device might consume part of those.
46 */
47 parent = acpi_get_first_physical_node(adev->parent);
48 if (parent && dev_is_pci(parent))
49 dest->parent = pci_find_resource(to_pci_dev(parent), dest);
50}

52/**
53 * acpi_create_platform_device - Create platform device for ACPI device node
54 * @adev: ACPI device node to create a platform device for.
55 * @properties: Optional collection of build-in properties.
56 *
57 * Check if the given @adev can be represented as a platform device and, if
58 * that's the case, create and register a platform device, populate its common
59 * resources and returns a pointer to it. Otherwise, return %NULL.
60 *
61 * Name of the platform device will be the same as @adev's.
62 */
63struct platform_device *acpi_create_platform_device(struct acpi_device *adev,
64 struct property_entry *properties)
65{
66 struct platform_device *pdev = NULL;
67 struct platform_device_info pdevinfo;
68 struct resource_entry *rentry;
69 struct list_head resource_list;
70 struct resource *resources = NULL;
71 int count;
72
73 /* If the ACPI node already has a physical device attached, skip it. */
74 if (adev->physical_node_count)
75 return NULL;
76
77 if (!acpi_match_device_ids(adev, forbidden_id_list))
78 return ERR_PTR(-EINVAL);
79
80 INIT_LIST_HEAD(&resource_list);
81 count = acpi_dev_get_resources(adev, &resource_list, NULL, NULL);
82 if (count < 0) {
83 return NULL;
84 } else if (count > 0) {
85 resources = kcalloc(count, sizeof(struct resource),
86 GFP_KERNEL);
87 if (!resources) {
88 dev_err(&adev->dev, "No memory for resources\n");
89 acpi_dev_free_resource_list(&resource_list);
90 return ERR_PTR(-ENOMEM);
91 }
92 count = 0;
93 list_for_each_entry(rentry, &resource_list, node)
94 acpi_platform_fill_resource(adev, rentry->res,
95 &resources[count++]);
96
97 acpi_dev_free_resource_list(&resource_list);
98 }
99
100 memset(&pdevinfo, 0, sizeof(pdevinfo));
101 /*
102 * If the ACPI node has a parent and that parent has a physical device
103 * attached to it, that physical device should be the parent of the
104 * platform device we are about to create.
105 */
106 pdevinfo.parent = adev->parent ?
107 acpi_get_first_physical_node(adev->parent) : NULL;
108 pdevinfo.name = dev_name(&adev->dev);
109 pdevinfo.id = -1;
110 pdevinfo.res = resources;
111 pdevinfo.num_res = count;
112 pdevinfo.fwnode = acpi_fwnode_handle(adev);
113 pdevinfo.properties = properties;
114
115 if (acpi_dma_supported(adev))
116 pdevinfo.dma_mask = DMA_BIT_MASK(32);
117 else
118 pdevinfo.dma_mask = 0;
119
120 pdev = platform_device_register_full(&pdevinfo);
121 if (IS_ERR(pdev))
122 dev_err(&adev->dev, "platform device creation failed: %ld\n",
123 PTR_ERR(pdev));
124 else {
125 set_dev_node(&pdev->dev, acpi_get_node(adev->handle));
126 dev_dbg(&adev->dev, "created platform device %s\n",
127 dev_name(&pdev->dev));
128 }
129
130 kfree(resources);
131
132 return pdev;
133}
134EXPORT_SYMBOL_GPL(acpi_create_platform_device);
