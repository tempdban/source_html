21#include <media/rc-map.h>
22#include <linux/module.h>
23
24/* TrekStor DVB-T USB Stick remote controller. */
25/* Imported from af9015.h.
26 Initial keytable was from Marc Schneider <macke@macke.org> */
27static struct rc_map_table trekstor[] = {
28 { 0x0084, KEY_0 },
29 { 0x0085, KEY_MUTE }, /* Mute */
30 { 0x0086, KEY_HOMEPAGE }, /* Home */
31 { 0x0087, KEY_UP }, /* Up */
32 { 0x0088, KEY_OK }, /* OK */
33 { 0x0089, KEY_RIGHT }, /* Right */
34 { 0x008a, KEY_FASTFORWARD }, /* Fast forward */
35 { 0x008b, KEY_VOLUMEUP }, /* Volume + */
36 { 0x008c, KEY_DOWN }, /* Down */
37 { 0x008d, KEY_PLAY }, /* Play/Pause */
38 { 0x008e, KEY_STOP }, /* Stop */
39 { 0x008f, KEY_EPG }, /* Info/EPG */
40 { 0x0090, KEY_7 },
41 { 0x0091, KEY_4 },
42 { 0x0092, KEY_1 },
43 { 0x0093, KEY_CHANNELDOWN }, /* Channel - */
44 { 0x0094, KEY_8 },
45 { 0x0095, KEY_5 },
46 { 0x0096, KEY_2 },
47 { 0x0097, KEY_CHANNELUP }, /* Channel + */
48 { 0x0098, KEY_9 },
49 { 0x0099, KEY_6 },
50 { 0x009a, KEY_3 },
51 { 0x009b, KEY_VOLUMEDOWN }, /* Volume - */
52 { 0x009c, KEY_TV }, /* TV */
53 { 0x009d, KEY_RECORD }, /* Record */
54 { 0x009e, KEY_REWIND }, /* Rewind */
55 { 0x009f, KEY_LEFT }, /* Left */
56};
57
58static struct rc_map_list trekstor_map = {
59 .map = {
60 .scan = trekstor,
61 .size = ARRAY_SIZE(trekstor),
62 .rc_proto = RC_PROTO_NEC,
63 .name = RC_MAP_TREKSTOR,
64 }
65};
66
67static int __init init_rc_map_trekstor(void)
68{
69 return rc_map_register(&trekstor_map);
70}
71
72static void __exit exit_rc_map_trekstor(void)
73{
74 rc_map_unregister(&trekstor_map);
75}
76
77module_init(init_rc_map_trekstor)
78module_exit(exit_rc_map_trekstor)
