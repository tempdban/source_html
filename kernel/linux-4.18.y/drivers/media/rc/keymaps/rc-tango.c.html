69static struct rc_map_list tango_map = {
70 .map = {

73 .rc_proto = RC_PROTO_NECX,
74 .name = RC_MAP_TANGO,

78static int __init init_rc_map_tango(void)
79{
80 return rc_map_register(&tango_map);

83static void __exit exit_rc_map_tango(void)
