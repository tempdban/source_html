29#define IPSEC_MAX_RX_IP_COUNT 128
30#define IPSEC_MAX_SA_COUNT 1024

32#define ESP_ICV_SIZE 16

35enum ixgbe_operation {
36 IXGBE_OP_AUTHENTICATED_ENCRYPTION,
37 IXGBE_OP_AUTHENTICATED_DECRYPTION
38};
39
40enum ixgbe_gcm_key {
41 IXGBE_GCM_KEY_128,
42 IXGBE_GCM_KEY_256

45/**
46 * Generic IP address structure
47 * TODO: Find better location for this rte_net.h possibly.
48 **/
49struct ipaddr {
50 enum ipaddr_type {
51 IPv4,
52 IPv6
53 } type;
54 /**< IP Address Type - IPv4/IPv6 */
55
56 union {
57 uint32_t ipv4;
58 uint32_t ipv6[4];
59 };
60};
61
62/** inline crypto crypto private session structure */
63struct ixgbe_crypto_session {
64 enum ixgbe_operation op;
65 uint8_t *key;
66 uint32_t salt;
67 uint32_t sa_index;
68 uint32_t spi;
69 struct ipaddr src_ip;
70 struct ipaddr dst_ip;
71 struct rte_eth_dev *dev;
72} __rte_cache_aligned;
73
74struct ixgbe_crypto_rx_ip_table {
75 struct ipaddr ip;
76 uint16_t ref_count;
77};
78struct ixgbe_crypto_rx_sa_table {
79 uint32_t spi;
80 uint32_t ip_index;
81 uint8_t mode;
82 uint8_t used;
83};
84
85struct ixgbe_crypto_tx_sa_table {
86 uint32_t spi;
87 uint8_t used;
88};
89
90union ixgbe_crypto_tx_desc_md {
91 uint64_t data;
92 struct {
93 /**< SA table index */
94 uint32_t sa_idx;
95 /**< ICV and ESP trailer length */
96 uint8_t pad_len;
97 /**< enable encryption */
98 uint8_t enc;
99 };
100};
101
102struct ixgbe_ipsec {
103 struct ixgbe_crypto_rx_ip_table rx_ip_tbl[IPSEC_MAX_RX_IP_COUNT];
104 struct ixgbe_crypto_rx_sa_table rx_sa_tbl[IPSEC_MAX_SA_COUNT];
105 struct ixgbe_crypto_tx_sa_table tx_sa_tbl[IPSEC_MAX_SA_COUNT];
106};
