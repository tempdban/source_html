508/**
509 * i40e_virtchnl_link_speed - Convert AdminQ link_speed to virtchnl definition
510 * @link_speed: the speed to convert
511 *
512 * Returns the link_speed in terms of the virtchnl interface, for use in
513 * converting link_speed as reported by the AdminQ into the format used for
514 * talking to virtchnl devices. If we can't represent the link speed properly,
515 * report LINK_SPEED_UNKNOWN.
516 **/
517STATIC INLINE enum virtchnl_link_speed
518i40e_virtchnl_link_speed(enum i40e_aq_link_speed link_speed)
519{
520 switch (link_speed) {
521 case I40E_LINK_SPEED_100MB:
522 return VIRTCHNL_LINK_SPEED_100MB;
523 case I40E_LINK_SPEED_1GB:
524 return VIRTCHNL_LINK_SPEED_1GB;
525 case I40E_LINK_SPEED_10GB:
526 return VIRTCHNL_LINK_SPEED_10GB;
527 case I40E_LINK_SPEED_40GB:
528 return VIRTCHNL_LINK_SPEED_40GB;
529 case I40E_LINK_SPEED_20GB:
530 return VIRTCHNL_LINK_SPEED_20GB;
531 case I40E_LINK_SPEED_25GB:
532 return VIRTCHNL_LINK_SPEED_25GB;
533 case I40E_LINK_SPEED_UNKNOWN:
534 default:
535 return VIRTCHNL_LINK_SPEED_UNKNOWN;
536 }
537}
