149struct eth_tx_1st_bd_flags {
150 u8 bitfields;
151/* Set to 1 in the first BD. (for debug) */
152#define ETH_TX_1ST_BD_FLAGS_START_BD_MASK 0x1
153#define ETH_TX_1ST_BD_FLAGS_START_BD_SHIFT 0
154/* Do not allow additional VLAN manipulations on this packet. */
155#define ETH_TX_1ST_BD_FLAGS_FORCE_VLAN_MODE_MASK 0x1
156#define ETH_TX_1ST_BD_FLAGS_FORCE_VLAN_MODE_SHIFT 1
157/* Recalculate IP checksum. For tunneled packet - relevant to inner header. */
158#define ETH_TX_1ST_BD_FLAGS_IP_CSUM_MASK 0x1
159#define ETH_TX_1ST_BD_FLAGS_IP_CSUM_SHIFT 2
160/* Recalculate TCP/UDP checksum.
161 * For tunneled packet - relevant to inner header.
162 */
163#define ETH_TX_1ST_BD_FLAGS_L4_CSUM_MASK 0x1
164#define ETH_TX_1ST_BD_FLAGS_L4_CSUM_SHIFT 3
165/* If set, insert VLAN tag from vlan field to the packet.
166 * For tunneled packet - relevant to outer header.
167 */
168#define ETH_TX_1ST_BD_FLAGS_VLAN_INSERTION_MASK 0x1
169#define ETH_TX_1ST_BD_FLAGS_VLAN_INSERTION_SHIFT 4
170/* If set, this is an LSO packet. Note: For Tunneled LSO packets, the offset of
171 * the inner IPV4 (and IPV6) header is limited to 253 (and 251 respectively)
172 * bytes, inclusive.
173 */
174#define ETH_TX_1ST_BD_FLAGS_LSO_MASK 0x1
175#define ETH_TX_1ST_BD_FLAGS_LSO_SHIFT 5
176/* Recalculate Tunnel IP Checksum (if Tunnel IP Header is IPv4) */
177#define ETH_TX_1ST_BD_FLAGS_TUNN_IP_CSUM_MASK 0x1
178#define ETH_TX_1ST_BD_FLAGS_TUNN_IP_CSUM_SHIFT 6
179/* Recalculate Tunnel UDP/GRE Checksum (Depending on Tunnel Type) */
180#define ETH_TX_1ST_BD_FLAGS_TUNN_L4_CSUM_MASK 0x1
181#define ETH_TX_1ST_BD_FLAGS_TUNN_L4_CSUM_SHIFT 7

199#define ETH_TX_DATA_1ST_BD_RESERVED0_MASK 0x1
200#define ETH_TX_DATA_1ST_BD_RESERVED0_SHIFT 1
201/* Total packet length - must be filled for non-LSO packets. */
202#define ETH_TX_DATA_1ST_BD_PKT_LEN_MASK 0x3FFF
203#define ETH_TX_DATA_1ST_BD_PKT_LEN_SHIFT 2

206/*
207 * The parsing information data for the second tx bd of a given packet.
208 */
209struct eth_tx_data_2nd_bd {
210/* For tunnel with IPv6+ext - Tunnel header IP datagram length (in BYTEs) */
211 __le16 tunn_ip_size;
212 __le16 bitfields1;
213/* For Tunnel header with IPv6 ext. - Inner L2 Header Size (in 2-byte WORDs) */
214#define ETH_TX_DATA_2ND_BD_TUNN_INNER_L2_HDR_SIZE_W_MASK 0xF
215#define ETH_TX_DATA_2ND_BD_TUNN_INNER_L2_HDR_SIZE_W_SHIFT 0
216/* For Tunnel header with IPv6 ext. - Inner L2 Header MAC DA Type
217 * (use enum eth_addr_type)
218 */
219#define ETH_TX_DATA_2ND_BD_TUNN_INNER_ETH_TYPE_MASK 0x3
220#define ETH_TX_DATA_2ND_BD_TUNN_INNER_ETH_TYPE_SHIFT 4
221/* Destination port mode. (use enum dest_port_mode) */
222#define ETH_TX_DATA_2ND_BD_DEST_PORT_MODE_MASK 0x3
223#define ETH_TX_DATA_2ND_BD_DEST_PORT_MODE_SHIFT 6
224/* Should be 0 in all the BDs, except the first one. (for debug) */
225#define ETH_TX_DATA_2ND_BD_START_BD_MASK 0x1
226#define ETH_TX_DATA_2ND_BD_START_BD_SHIFT 8
227/* For Tunnel header with IPv6 ext. - Tunnel Type (use enum eth_tx_tunn_type) */
228#define ETH_TX_DATA_2ND_BD_TUNN_TYPE_MASK 0x3
229#define ETH_TX_DATA_2ND_BD_TUNN_TYPE_SHIFT 9
230/* For LSO / Tunnel header with IPv6+ext - Set if inner header is IPv6 */
231#define ETH_TX_DATA_2ND_BD_TUNN_INNER_IPV6_MASK 0x1
232#define ETH_TX_DATA_2ND_BD_TUNN_INNER_IPV6_SHIFT 11
233/* In tunneling mode - Set to 1 when the Inner header is IPv6 with extension.
234 * Otherwise set to 1 if the header is IPv6 with extension.
235 */
236#define ETH_TX_DATA_2ND_BD_IPV6_EXT_MASK 0x1
237#define ETH_TX_DATA_2ND_BD_IPV6_EXT_SHIFT 12
238/* Set to 1 if Tunnel (outer = encapsulating) header has IPv6 ext. (Note: 3rd BD
239 * is required, hence EDPM does not support Tunnel [outer] header with Ipv6Ext)
240 */
241#define ETH_TX_DATA_2ND_BD_TUNN_IPV6_EXT_MASK 0x1
242#define ETH_TX_DATA_2ND_BD_TUNN_IPV6_EXT_SHIFT 13
243/* Set if (inner) L4 protocol is UDP. (Required when IPv6+ext (or tunnel with
244 * inner or outer Ipv6+ext) and l4_csum is set)
245 */
246#define ETH_TX_DATA_2ND_BD_L4_UDP_MASK 0x1
247#define ETH_TX_DATA_2ND_BD_L4_UDP_SHIFT 14
248/* The pseudo header checksum type in the L4 checksum field. Required when
249 * IPv6+ext and l4_csum is set. (use enum eth_l4_pseudo_checksum_mode)
250 */
251#define ETH_TX_DATA_2ND_BD_L4_PSEUDO_CSUM_MODE_MASK 0x1
252#define ETH_TX_DATA_2ND_BD_L4_PSEUDO_CSUM_MODE_SHIFT 15
253 __le16 bitfields2;
254/* For inner/outer header IPv6+ext - (inner) L4 header offset (in 2-byte WORDs).
255 * For regular packet - offset from the beginning of the packet. For tunneled
256 * packet - offset from the beginning of the inner header
257 */
258#define ETH_TX_DATA_2ND_BD_L4_HDR_START_OFFSET_W_MASK 0x1FFF
259#define ETH_TX_DATA_2ND_BD_L4_HDR_START_OFFSET_W_SHIFT 0
260#define ETH_TX_DATA_2ND_BD_RESERVED0_MASK 0x7
261#define ETH_TX_DATA_2ND_BD_RESERVED0_SHIFT 13
262};

519 ETH_AGG_END_SP_UPDATE /* SP configuration update */,
520/* Maximum aggregation length or maximum buffer number used. */
521 ETH_AGG_END_MAX_LEN,
522/* TCP PSH flag or TCP payload length below continue threshold. */
523 ETH_AGG_END_LAST_SEG,
524 ETH_AGG_END_TIMEOUT /* Timeout expiration. */,
525/* Packet header not consistency: different IPv4 TOS, TTL or flags, IPv6 TC,
526 * Hop limit or Flow label, TCP header length or TS options. In GRO different
527 * TS value, SMAC, DMAC, ackNum, windowSize or VLAN
528 */
529 ETH_AGG_END_NOT_CONSISTENT,
530/* Out of order or retransmission packet: sequence, ack or timestamp not
531 * consistent with previous segment.
