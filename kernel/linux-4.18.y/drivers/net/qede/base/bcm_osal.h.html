190typedef struct _osal_list_entry_t {
191 struct _osal_list_entry_t *next, *prev;
192} osal_list_entry_t;
193
194typedef struct osal_list_t {
195 osal_list_entry_t *head, *tail;
196 unsigned long cnt;
197} osal_list_t;
198
199#define OSAL_LIST_INIT(list) \
200 do { \
201 (list)->head = NULL; \
202 (list)->tail = NULL; \
203 (list)->cnt = 0; \
204 } while (0)
205
206#define OSAL_LIST_PUSH_HEAD(entry, list) \
207 do { \
208 (entry)->prev = (osal_list_entry_t *)0; \
209 (entry)->next = (list)->head; \
210 if ((list)->tail == (osal_list_entry_t *)0) { \
211 (list)->tail = (entry); \
212 } else { \
213 (list)->head->prev = (entry); \
214 } \
215 (list)->head = (entry); \
216 (list)->cnt++; \
217 } while (0)
218
219#define OSAL_LIST_PUSH_TAIL(entry, list) \
220 do { \
221 (entry)->next = (osal_list_entry_t *)0; \
222 (entry)->prev = (list)->tail; \
223 if ((list)->tail) { \
224 (list)->tail->next = (entry); \
225 } else { \
226 (list)->head = (entry); \
227 } \
228 (list)->tail = (entry); \
229 (list)->cnt++; \
230 } while (0)
231
232#define OSAL_LIST_FIRST_ENTRY(list, type, field) \
233 (type *)((list)->head)
234
235#define OSAL_LIST_REMOVE_ENTRY(entry, list) \
236 do { \
237 if ((list)->head == (entry)) { \
238 if ((list)->head) { \
239 (list)->head = (list)->head->next; \
240 if ((list)->head) { \
241 (list)->head->prev = (osal_list_entry_t *)0;\
242 } else { \
243 (list)->tail = (osal_list_entry_t *)0; \
244 } \
245 (list)->cnt--; \
246 } \
247 } else if ((list)->tail == (entry)) { \
248 if ((list)->tail) { \
249 (list)->tail = (list)->tail->prev; \
250 if ((list)->tail) { \
251 (list)->tail->next = (osal_list_entry_t *)0;\
252 } else { \
253 (list)->head = (osal_list_entry_t *)0; \
254 } \
255 (list)->cnt--; \
256 } \
257 } else { \
258 (entry)->prev->next = (entry)->next; \
259 (entry)->next->prev = (entry)->prev; \
260 (list)->cnt--; \
261 } \
262 } while (0)
263
264#define OSAL_LIST_IS_EMPTY(list) \
265 ((list)->cnt == 0)
266
267#define OSAL_LIST_NEXT(entry, field, type) \
268 (type *)((&((entry)->field))->next)
269
270/* TODO: Check field, type order */

272#define OSAL_LIST_FOR_EACH_ENTRY(entry, list, field, type) \
273 for (entry = OSAL_LIST_FIRST_ENTRY(list, type, field); \
274 entry; \
275 entry = OSAL_LIST_NEXT(entry, field, type))
276
277#define OSAL_LIST_FOR_EACH_ENTRY_SAFE(entry, tmp_entry, list, field, type) \
278 for (entry = OSAL_LIST_FIRST_ENTRY(list, type, field), \
279 tmp_entry = (entry) ? OSAL_LIST_NEXT(entry, field, type) : NULL; \
280 entry != NULL; \
281 entry = (type *)tmp_entry, \
282 tmp_entry = (entry) ? OSAL_LIST_NEXT(entry, field, type) : NULL)
