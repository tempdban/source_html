5#ifndef _AVF_LOG_H_
6#define _AVF_LOG_H_
7
8extern int avf_logtype_init;
9#define PMD_INIT_LOG(level, fmt, args...) \
10 rte_log(RTE_LOG_ ## level, avf_logtype_init, "%s(): " fmt "\n", \
11 __func__, ## args)
12#define PMD_INIT_FUNC_TRACE() PMD_INIT_LOG(DEBUG, " >>")
13
14extern int avf_logtype_driver;
15#define PMD_DRV_LOG_RAW(level, fmt, args...) \
16 rte_log(RTE_LOG_ ## level, avf_logtype_driver, "%s(): " fmt, \
17 __func__, ## args)
18
19#define PMD_DRV_LOG(level, fmt, args...) \
20 PMD_DRV_LOG_RAW(level, fmt "\n", ## args)
21#define PMD_DRV_FUNC_TRACE() PMD_DRV_LOG(DEBUG, " >>")
22
23#ifdef RTE_LIBRTE_AVF_DEBUG_RX
24#define PMD_RX_LOG(level, fmt, args...) \
25 RTE_LOG_DP(level, PMD, "%s(): " fmt "\n", __func__, ## args)
26#else
27#define PMD_RX_LOG(level, fmt, args...) do { } while (0)
28#endif
29
30#ifdef RTE_LIBRTE_AVF_DEBUG_TX
31#define PMD_TX_LOG(level, fmt, args...) \

35#endif

37#ifdef RTE_LIBRTE_AVF_DEBUG_TX_FREE
38#define PMD_TX_FREE_LOG(level, fmt, args...) \
39 RTE_LOG_DP(level, PMD, "%s(): " fmt "\n", __func__, ## args)
40#else
41#define PMD_TX_FREE_LOG(level, fmt, args...) do { } while (0)
42#endif
43
44#endif /* _AVF_LOG_H_ */
