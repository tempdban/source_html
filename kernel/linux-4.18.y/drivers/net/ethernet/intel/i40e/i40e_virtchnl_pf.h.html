49/* In ADq, max 4 VSI's can be allocated per VF including primary VF VSI.
50 * These variables are used to store indices, id's and number of queues
51 * for each VSI including that of primary VF VSI. Each Traffic class is
52 * termed as channel and each channel can in-turn have 4 queues which
53 * means max 16 queues overall per VF.
54 */
55struct i40evf_channel {
56 u16 vsi_idx; /* index in PF struct for all channel VSIs */
57 u16 vsi_id; /* VSI ID used by firmware */
58 u16 num_qps; /* number of queue pairs requested by user */
59 u64 max_tx_rate; /* bandwidth rate allocation for VSIs */
60};
61
62/* VF information structure */
63struct i40e_vf {
64 struct i40e_pf *pf;
65
66 /* VF id in the PF space */
67 s16 vf_id;
68 /* all VF vsis connect to the same parent */
69 enum i40e_switch_element_types parent_type;
70 struct virtchnl_version_info vf_ver;
71 u32 driver_caps; /* reported by VF driver */
72
73 /* VF Port Extender (PE) stag if used */
74 u16 stag;
75
76 struct virtchnl_ether_addr default_lan_addr;
77 u16 port_vlan_id;
78 bool pf_set_mac; /* The VMM admin set the VF MAC address */
79 bool trusted;
80
81 /* VSI indices - actual VSI pointers are maintained in the PF structure
82 * When assigned, these will be non-zero, because VSI 0 is always
83 * the main LAN VSI for the PF.
84 */
85 u16 lan_vsi_idx; /* index into PF struct */
86 u16 lan_vsi_id; /* ID as used by firmware */
87
88 u8 num_queue_pairs; /* num of qps assigned to VF vsis */
89 u8 num_req_queues; /* num of requested qps */
90 u64 num_mdd_events; /* num of mdd events detected */
91 /* num of continuous malformed or invalid msgs detected */
92 u64 num_invalid_msgs;
93 u64 num_valid_msgs; /* num of valid msgs detected */
94
95 unsigned long vf_caps; /* vf's adv. capabilities */
96 unsigned long vf_states; /* vf's runtime states */
97 unsigned int tx_rate; /* Tx bandwidth limit in Mbps */
98 bool link_forced;
99 bool link_up; /* only valid if VF link is forced */
100 bool spoofchk;
101 u16 num_mac;
102 u16 num_vlan;
103
104 /* ADq related variables */
105 bool adq_enabled; /* flag to enable adq */
106 u8 num_tc;
107 struct i40evf_channel ch[I40E_MAX_VF_VSI];
108 struct hlist_head cloud_filter_list;
109 u16 num_cloud_filters;
110
111 /* RDMA Client */
112 struct virtchnl_iwarp_qvlist_info *qvlist_info;
113};
