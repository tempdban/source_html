7#define IXGBE_IPSEC_MAX_SA_COUNT 1024
8#define IXGBE_IPSEC_MAX_RX_IP_COUNT 128
9#define IXGBE_IPSEC_BASE_RX_INDEX 0

11#define IXGBE_IPSEC_AUTH_BITS 128

13#define IXGBE_RXTXIDX_IPS_EN 0x00000001
14#define IXGBE_RXIDX_TBL_SHIFT 1
15enum ixgbe_ipsec_tbl_sel {

17 ips_rx_spi_tbl = 0x02,
18 ips_rx_key_tbl = 0x03,
19};

21#define IXGBE_RXTXIDX_IDX_SHIFT 3

30struct rx_sa {
31 struct hlist_node hlist;
32 struct xfrm_state *xs;
33 __be32 ipaddr[4];
34 u32 key[4];
35 u32 salt;
36 u32 mode;
37 u8 iptbl_ind;
38 bool used;
39 bool decrypt;
40};

42struct rx_ip_sa {
43 __be32 ipaddr[4];
44 u32 ref_cnt;
45 bool used;
46};
47
48struct tx_sa {
49 struct xfrm_state *xs;
50 u32 key[4];
51 u32 salt;
52 bool encrypt;
53 bool used;
54};

56struct ixgbe_ipsec_tx_data {
57 u32 flags;

59 u16 sa_idx;
60};
