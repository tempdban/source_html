49struct hn_tx_queue {
50 struct hn_data *hv;
51 struct vmbus_channel *chan;
52 uint16_t port_id;
53 uint16_t queue_id;
54 uint32_t free_thresh;
55
56 /* Applied packet transmission aggregation limits. */
57 uint32_t agg_szmax;
58 uint32_t agg_pktmax;
59 uint32_t agg_align;

66
67 struct hn_stats stats;
68};

94struct hn_data {
95 struct rte_vmbus_device *vmbus;
96 struct hn_rx_queue *primary;
97 struct rte_eth_dev *vf_dev; /* Subordinate device */
98 rte_spinlock_t vf_lock;
99 uint16_t port_id;
100 bool closed;
101 bool vf_present;
102 uint32_t link_status;
103 uint32_t link_speed;
104
105 struct rte_mem_resource *rxbuf_res; /* UIO resource for Rx */
106 struct hn_rx_bufinfo *rxbuf_info;
107 uint32_t rxbuf_section_cnt; /* # of Rx sections */
108 volatile uint32_t rxbuf_outstanding;
109 uint16_t max_queues; /* Max available queues */
110 uint16_t num_queues;
111 uint64_t rss_offloads;
112
113 struct rte_mem_resource *chim_res; /* UIO resource for Tx */
114 struct rte_mempool *tx_pool; /* Tx descriptors */
115 uint32_t chim_szmax; /* Max size per buffer */
116 uint32_t chim_cnt; /* Max packets per buffer */
117
118 uint32_t latency;
119 uint32_t nvs_ver;
120 uint32_t ndis_ver;
121 uint32_t rndis_agg_size;
122 uint32_t rndis_agg_pkts;
123 uint32_t rndis_agg_align;
124
125 volatile uint32_t rndis_pending;
126 rte_atomic32_t rndis_req_id;
127 uint8_t rndis_resp[256];
128
129 struct ether_addr mac_addr;
130
131 struct rte_eth_dev_owner owner;
132 struct rte_intr_handle vf_intr;
133
134 struct vmbus_channel *channels[HN_MAX_CHANNELS];
135};
