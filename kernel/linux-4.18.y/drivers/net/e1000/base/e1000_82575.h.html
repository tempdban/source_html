194 struct {

197 } read;
198 struct {
199 struct {

201 __le32 data;

206 } hs_rss;
207 } lo_dword;

209 __le32 rss; /* RSS Hash */
210 struct {
211 __le16 ip_id; /* IP id */

214 } hi_dword;
215 } lower;
216 struct {
217 __le32 status_error; /* ext status/error */
218 __le16 length; /* Packet length */
219 __le16 vlan; /* VLAN tag */
220 } upper;
221 } wb; /* writeback */
222};
