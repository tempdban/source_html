22#ifndef readq
23static inline u64 readq(void __iomem *reg)
24{
25 return ((u64)readl((char *)reg + 0x4UL) << 32) |
26 (u64)readl(reg);
27}
28
29static inline void writeq(u64 val, void __iomem *reg)
30{
31 writel(val & 0xffffffff, reg);
32 writel((u32)(val >> 32), (char *)reg + 0x4UL);
33}
34#endif
