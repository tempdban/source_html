30int __attribute__((cold))
31virtio_rxq_vec_setup(struct virtnet_rx *rxq)
32{
33 uintptr_t p;
34 struct rte_mbuf mb_def = { .buf_addr = 0 }; /* zeroed mbuf */
35
36 mb_def.nb_segs = 1;
37 mb_def.data_off = RTE_PKTMBUF_HEADROOM;
38 mb_def.port = rxq->port_id;
39 rte_mbuf_refcnt_set(&mb_def, 1);
40
41 /* prevent compiler reordering: rearm_data covers previous fields */
42 rte_compiler_barrier();
43 p = (uintptr_t)&mb_def.rearm_data;
44 rxq->mbuf_initializer = *(uint64_t *)p;
45
46 return 0;
47}
