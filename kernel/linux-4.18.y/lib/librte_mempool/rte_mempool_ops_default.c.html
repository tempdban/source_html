9ssize_t
10rte_mempool_op_calc_mem_size_default(const struct rte_mempool *mp,
11 uint32_t obj_num, uint32_t pg_shift,
12 size_t *min_chunk_size, size_t *align)
13{
14 size_t total_elt_sz;
15 size_t obj_per_page, pg_num, pg_sz;
16 size_t mem_size;
17
18 total_elt_sz = mp->header_size + mp->elt_size + mp->trailer_size;
19 if (total_elt_sz == 0) {
20 mem_size = 0;
21 } else if (pg_shift == 0) {
22 mem_size = total_elt_sz * obj_num;
23 } else {
24 pg_sz = (size_t)1 << pg_shift;
25 obj_per_page = pg_sz / total_elt_sz;
26 if (obj_per_page == 0) {
27 /*
28 * Note that if object size is bigger than page size,
29 * then it is assumed that pages are grouped in subsets
30 * of physically continuous pages big enough to store
31 * at least one object.
32 */
33 mem_size =
34 RTE_ALIGN_CEIL(total_elt_sz, pg_sz) * obj_num;
35 } else {
36 pg_num = (obj_num + obj_per_page - 1) / obj_per_page;
37 mem_size = pg_num << pg_shift;
38 }
39 }
40
41 *min_chunk_size = RTE_MAX((size_t)1 << pg_shift, total_elt_sz);
42
43 *align = RTE_MAX((size_t)RTE_CACHE_LINE_SIZE, (size_t)1 << pg_shift);
44
45 return mem_size;
46}

48int
49rte_mempool_op_populate_default(struct rte_mempool *mp, unsigned int max_objs,
50 void *vaddr, rte_iova_t iova, size_t len,
51 rte_mempool_populate_obj_cb_t *obj_cb, void *obj_cb_arg)
52{
53 size_t total_elt_sz;
54 size_t off;
55 unsigned int i;
56 void *obj;
57
58 total_elt_sz = mp->header_size + mp->elt_size + mp->trailer_size;
59
60 for (off = 0, i = 0; off + total_elt_sz <= len && i < max_objs; i++) {
61 off += mp->header_size;
62 obj = (char *)vaddr + off;
63 obj_cb(mp, obj_cb_arg, obj,
64 (iova == RTE_BAD_IOVA) ? RTE_BAD_IOVA : (iova + off));
65 rte_mempool_ops_enqueue_bulk(mp, &obj, 1);
66 off += mp->elt_size + mp->trailer_size;
67 }
68
69 return i;
70}
