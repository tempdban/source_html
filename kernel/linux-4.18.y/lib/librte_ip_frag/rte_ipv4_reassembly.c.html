11/*
12 * Reassemble fragments into one packet.
13 */
14struct rte_mbuf *
15ipv4_frag_reassemble(struct ip_frag_pkt *fp)
16{
17 struct ipv4_hdr *ip_hdr;
18 struct rte_mbuf *m, *prev;
19 uint32_t i, n, ofs, first_len;
20 uint32_t curr_idx = 0;
21
22 first_len = fp->frags[IP_FIRST_FRAG_IDX].len;
23 n = fp->last_idx - 1;
24
25 /*start from the last fragment. */
26 m = fp->frags[IP_LAST_FRAG_IDX].mb;
27 ofs = fp->frags[IP_LAST_FRAG_IDX].ofs;
28 curr_idx = IP_LAST_FRAG_IDX;
29
30 while (ofs != first_len) {
31
32 prev = m;
33
34 for (i = n; i != IP_FIRST_FRAG_IDX && ofs != first_len; i--) {
35
36 /* previous fragment found. */
37 if(fp->frags[i].ofs + fp->frags[i].len == ofs) {
38
39 /* adjust start of the last fragment data. */
40 rte_pktmbuf_adj(m, (uint16_t)(m->l2_len + m->l3_len));
41 rte_pktmbuf_chain(fp->frags[i].mb, m);
42
43 /* this mbuf should not be accessed directly */
44 fp->frags[curr_idx].mb = NULL;
45 curr_idx = i;
46
47 /* update our last fragment and offset. */
48 m = fp->frags[i].mb;
49 ofs = fp->frags[i].ofs;
50 }
51 }
52
53 /* error - hole in the packet. */
54 if (m == prev) {
55 return NULL;
56 }
57 }
58
59 /* chain with the first fragment. */
60 rte_pktmbuf_adj(m, (uint16_t)(m->l2_len + m->l3_len));
61 rte_pktmbuf_chain(fp->frags[IP_FIRST_FRAG_IDX].mb, m);
62 fp->frags[curr_idx].mb = NULL;
63 m = fp->frags[IP_FIRST_FRAG_IDX].mb;
64 fp->frags[IP_FIRST_FRAG_IDX].mb = NULL;
65
66 /* update mbuf fields for reassembled packet. */
67 m->ol_flags |= PKT_TX_IP_CKSUM;
68
69 /* update ipv4 header for the reassembled packet */
70 ip_hdr = rte_pktmbuf_mtod_offset(m, struct ipv4_hdr *, m->l2_len);
71
72 ip_hdr->total_length = rte_cpu_to_be_16((uint16_t)(fp->total_size +
73 m->l3_len));
74 ip_hdr->fragment_offset = (uint16_t)(ip_hdr->fragment_offset &
75 rte_cpu_to_be_16(IPV4_HDR_DF_FLAG));
76 ip_hdr->hdr_checksum = 0;
77
78 return m;
79}

81/*
82 * Process new mbuf with fragment of IPV4 packet.
83 * Incoming mbuf should have it's l2_len/l3_len fields setuped correclty.
84 * @param tbl
85 * Table where to lookup/add the fragmented packet.
86 * @param mb
87 * Incoming mbuf with IPV4 fragment.
88 * @param tms
89 * Fragment arrival timestamp.
90 * @param ip_hdr
91 * Pointer to the IPV4 header inside the fragment.
92 * @return
93 * Pointer to mbuf for reassembled packet, or NULL if:
94 * - an error occurred.
95 * - not all fragments of the packet are collected yet.
96 */
97struct rte_mbuf *
98rte_ipv4_frag_reassemble_packet(struct rte_ip_frag_tbl *tbl,
99 struct rte_ip_frag_death_row *dr, struct rte_mbuf *mb, uint64_t tms,
100 struct ipv4_hdr *ip_hdr)
101{
102 struct ip_frag_pkt *fp;
103 struct ip_frag_key key;
104 const unaligned_uint64_t *psd;
105 uint16_t ip_len;
106 uint16_t flag_offset, ip_ofs, ip_flag;
107
108 flag_offset = rte_be_to_cpu_16(ip_hdr->fragment_offset);
109 ip_ofs = (uint16_t)(flag_offset & IPV4_HDR_OFFSET_MASK);
110 ip_flag = (uint16_t)(flag_offset & IPV4_HDR_MF_FLAG);
111
112 psd = (unaligned_uint64_t *)&ip_hdr->src_addr;
113 /* use first 8 bytes only */
114 key.src_dst[0] = psd[0];
115 key.id = ip_hdr->packet_id;
116 key.key_len = IPV4_KEYLEN;
117
118 ip_ofs *= IPV4_HDR_OFFSET_UNITS;
119 ip_len = (uint16_t)(rte_be_to_cpu_16(ip_hdr->total_length) -
120 mb->l3_len);
121
122 IP_FRAG_LOG(DEBUG, "%s:%d:\n"
123 "mbuf: %p, tms: %" PRIu64
124 ", key: <%" PRIx64 ", %#x>, ofs: %u, len: %u, flags: %#x\n"
125 "tbl: %p, max_cycles: %" PRIu64 ", entry_mask: %#x, "
126 "max_entries: %u, use_entries: %u\n\n",
127 __func__, __LINE__,
128 mb, tms, key.src_dst[0], key.id, ip_ofs, ip_len, ip_flag,
129 tbl, tbl->max_cycles, tbl->entry_mask, tbl->max_entries,
130 tbl->use_entries);
131
132 /* try to find/add entry into the fragment's table. */
133 if ((fp = ip_frag_find(tbl, dr, &key, tms)) == NULL) {
134 IP_FRAG_MBUF2DR(dr, mb);
135 return NULL;
136 }
137
138 IP_FRAG_LOG(DEBUG, "%s:%d:\n"
139 "tbl: %p, max_entries: %u, use_entries: %u\n"
140 "ipv4_frag_pkt: %p, key: <%" PRIx64 ", %#x>, start: %" PRIu64
141 ", total_size: %u, frag_size: %u, last_idx: %u\n\n",
142 __func__, __LINE__,
143 tbl, tbl->max_entries, tbl->use_entries,
144 fp, fp->key.src_dst[0], fp->key.id, fp->start,
145 fp->total_size, fp->frag_size, fp->last_idx);
146
147
148 /* process the fragmented packet. */
149 mb = ip_frag_process(fp, dr, mb, ip_ofs, ip_len, ip_flag);
150 ip_frag_inuse(tbl, fp);
151
152 IP_FRAG_LOG(DEBUG, "%s:%d:\n"
153 "mbuf: %p\n"
154 "tbl: %p, max_entries: %u, use_entries: %u\n"
155 "ipv4_frag_pkt: %p, key: <%" PRIx64 ", %#x>, start: %" PRIu64
156 ", total_size: %u, frag_size: %u, last_idx: %u\n\n",
157 __func__, __LINE__, mb,
158 tbl, tbl->max_entries, tbl->use_entries,
159 fp, fp->key.src_dst[0], fp->key.id, fp->start,
160 fp->total_size, fp->frag_size, fp->last_idx);
161
162 return mb;
163}
