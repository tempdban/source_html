8/*
9 * Note, that to be able to use AVX2 classify method,
10 * both compiler and target cpu have to support AVX2 instructions.
11 */
12int
13rte_acl_classify_avx2(const struct rte_acl_ctx *ctx, const uint8_t **data,
14 uint32_t *results, uint32_t num, uint32_t categories)
15{
16 if (likely(num >= MAX_SEARCHES_AVX16))
17 return search_avx2x16(ctx, data, results, num, categories);
18 else if (num >= MAX_SEARCHES_SSE8)
19 return search_sse_8(ctx, data, results, num, categories);
20 else if (num >= MAX_SEARCHES_SSE4)
21 return search_sse_4(ctx, data, results, num, categories);
22 else
23 return rte_acl_classify_scalar(ctx, data, results, num,
24 categories);
25}
