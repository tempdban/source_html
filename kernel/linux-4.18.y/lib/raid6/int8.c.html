27/* Change this from BITS_PER_LONG if there is something better... */
28#if BITS_PER_LONG == 64
29# define NBYTES(x) ((x) * 0x0101010101010101UL)
30# define NSIZE 8
31# define NSHIFT 3
32# define NSTRING "64"
33typedef u64 unative_t;
34#else
35# define NBYTES(x) ((x) * 0x01010101U)
36# define NSIZE 4
37# define NSHIFT 2
38# define NSTRING "32"
39typedef u32 unative_t;
40#endif

45 * IA-64 wants insane amounts of unrolling. On other architectures that
46 * is just a waste of space.

51/*
52 * These sub-operations are separate inlines since they can sometimes be
53 * specially optimized using architecture-specific hacks.
54 */

56/*
57 * The SHLBYTE() operation shifts each byte left by 1, *not*
58 * rolling over into the next byte
59 */
60static inline __attribute_const__ unative_t SHLBYTE(unative_t v)
61{
62 unative_t vv;
63
64 vv = (v << 1) & NBYTES(0xfe);
65 return vv;
66}
67
68/*
69 * The MASK() operation returns 0xFF in any byte for which the high
70 * bit is 1, 0x00 for any byte for which the high bit is 0.
71 */
72static inline __attribute_const__ unative_t MASK(unative_t v)
73{
74 unative_t vv;
75
76 vv = v & NBYTES(0x80);
77 vv = (vv << 1) - (vv >> 7); /* Overflow on the top bit is OK */
78 return vv;
79}

82static void raid6_int8_gen_syndrome(int disks, size_t bytes, void **ptrs)
83{
84 u8 **dptr = (u8 **)ptrs;
85 u8 *p, *q;
86 int d, z, z0;
87
88 unative_t wd0, wq0, wp0, w10, w20;
89 unative_t wd1, wq1, wp1, w11, w21;
90 unative_t wd2, wq2, wp2, w12, w22;
91 unative_t wd3, wq3, wp3, w13, w23;
92 unative_t wd4, wq4, wp4, w14, w24;
93 unative_t wd5, wq5, wp5, w15, w25;
94 unative_t wd6, wq6, wp6, w16, w26;
95 unative_t wd7, wq7, wp7, w17, w27;
96
97 z0 = disks - 3; /* Highest data disk */
98 p = dptr[z0+1]; /* XOR parity */
99 q = dptr[z0+2]; /* RS syndrome */
100
101 for ( d = 0 ; d < bytes ; d += NSIZE*8 ) {
102 wq0 = wp0 = *(unative_t *)&dptr[z0][d+0*NSIZE];
103 wq1 = wp1 = *(unative_t *)&dptr[z0][d+1*NSIZE];
104 wq2 = wp2 = *(unative_t *)&dptr[z0][d+2*NSIZE];
105 wq3 = wp3 = *(unative_t *)&dptr[z0][d+3*NSIZE];
106 wq4 = wp4 = *(unative_t *)&dptr[z0][d+4*NSIZE];
107 wq5 = wp5 = *(unative_t *)&dptr[z0][d+5*NSIZE];
108 wq6 = wp6 = *(unative_t *)&dptr[z0][d+6*NSIZE];
109 wq7 = wp7 = *(unative_t *)&dptr[z0][d+7*NSIZE];
110 for ( z = z0-1 ; z >= 0 ; z-- ) {
111 wd0 = *(unative_t *)&dptr[z][d+0*NSIZE];
112 wd1 = *(unative_t *)&dptr[z][d+1*NSIZE];
113 wd2 = *(unative_t *)&dptr[z][d+2*NSIZE];
114 wd3 = *(unative_t *)&dptr[z][d+3*NSIZE];
115 wd4 = *(unative_t *)&dptr[z][d+4*NSIZE];
116 wd5 = *(unative_t *)&dptr[z][d+5*NSIZE];
117 wd6 = *(unative_t *)&dptr[z][d+6*NSIZE];
118 wd7 = *(unative_t *)&dptr[z][d+7*NSIZE];
119 wp0 ^= wd0;
120 wp1 ^= wd1;
121 wp2 ^= wd2;
122 wp3 ^= wd3;
123 wp4 ^= wd4;
124 wp5 ^= wd5;
125 wp6 ^= wd6;
126 wp7 ^= wd7;
127 w20 = MASK(wq0);
128 w21 = MASK(wq1);
129 w22 = MASK(wq2);
130 w23 = MASK(wq3);
131 w24 = MASK(wq4);
132 w25 = MASK(wq5);
133 w26 = MASK(wq6);
134 w27 = MASK(wq7);
135 w10 = SHLBYTE(wq0);
136 w11 = SHLBYTE(wq1);
137 w12 = SHLBYTE(wq2);
138 w13 = SHLBYTE(wq3);
139 w14 = SHLBYTE(wq4);
140 w15 = SHLBYTE(wq5);
141 w16 = SHLBYTE(wq6);
142 w17 = SHLBYTE(wq7);
143 w20 &= NBYTES(0x1d);
144 w21 &= NBYTES(0x1d);
145 w22 &= NBYTES(0x1d);
146 w23 &= NBYTES(0x1d);
147 w24 &= NBYTES(0x1d);
148 w25 &= NBYTES(0x1d);
149 w26 &= NBYTES(0x1d);
150 w27 &= NBYTES(0x1d);
151 w10 ^= w20;
152 w11 ^= w21;
153 w12 ^= w22;
154 w13 ^= w23;
155 w14 ^= w24;
156 w15 ^= w25;
157 w16 ^= w26;
158 w17 ^= w27;
159 wq0 = w10 ^ wd0;
160 wq1 = w11 ^ wd1;
161 wq2 = w12 ^ wd2;
162 wq3 = w13 ^ wd3;
163 wq4 = w14 ^ wd4;
164 wq5 = w15 ^ wd5;
165 wq6 = w16 ^ wd6;
166 wq7 = w17 ^ wd7;
167 }
168 *(unative_t *)&p[d+NSIZE*0] = wp0;
169 *(unative_t *)&p[d+NSIZE*1] = wp1;
170 *(unative_t *)&p[d+NSIZE*2] = wp2;
171 *(unative_t *)&p[d+NSIZE*3] = wp3;
172 *(unative_t *)&p[d+NSIZE*4] = wp4;
173 *(unative_t *)&p[d+NSIZE*5] = wp5;
174 *(unative_t *)&p[d+NSIZE*6] = wp6;
175 *(unative_t *)&p[d+NSIZE*7] = wp7;
176 *(unative_t *)&q[d+NSIZE*0] = wq0;
177 *(unative_t *)&q[d+NSIZE*1] = wq1;
178 *(unative_t *)&q[d+NSIZE*2] = wq2;
179 *(unative_t *)&q[d+NSIZE*3] = wq3;
180 *(unative_t *)&q[d+NSIZE*4] = wq4;
181 *(unative_t *)&q[d+NSIZE*5] = wq5;
182 *(unative_t *)&q[d+NSIZE*6] = wq6;
183 *(unative_t *)&q[d+NSIZE*7] = wq7;
184 }
185}

187static void raid6_int8_xor_syndrome(int disks, int start, int stop,
188 size_t bytes, void **ptrs)
189{
190 u8 **dptr = (u8 **)ptrs;
191 u8 *p, *q;
192 int d, z, z0;
193
194 unative_t wd0, wq0, wp0, w10, w20;
195 unative_t wd1, wq1, wp1, w11, w21;
196 unative_t wd2, wq2, wp2, w12, w22;
197 unative_t wd3, wq3, wp3, w13, w23;
198 unative_t wd4, wq4, wp4, w14, w24;
199 unative_t wd5, wq5, wp5, w15, w25;
200 unative_t wd6, wq6, wp6, w16, w26;
201 unative_t wd7, wq7, wp7, w17, w27;
202
203 z0 = stop; /* P/Q right side optimization */
204 p = dptr[disks-2]; /* XOR parity */
205 q = dptr[disks-1]; /* RS syndrome */
206
207 for ( d = 0 ; d < bytes ; d += NSIZE*8 ) {
208 /* P/Q data pages */
209 wq0 = wp0 = *(unative_t *)&dptr[z0][d+0*NSIZE];
210 wq1 = wp1 = *(unative_t *)&dptr[z0][d+1*NSIZE];
211 wq2 = wp2 = *(unative_t *)&dptr[z0][d+2*NSIZE];
212 wq3 = wp3 = *(unative_t *)&dptr[z0][d+3*NSIZE];
213 wq4 = wp4 = *(unative_t *)&dptr[z0][d+4*NSIZE];
214 wq5 = wp5 = *(unative_t *)&dptr[z0][d+5*NSIZE];
215 wq6 = wp6 = *(unative_t *)&dptr[z0][d+6*NSIZE];
216 wq7 = wp7 = *(unative_t *)&dptr[z0][d+7*NSIZE];
217 for ( z = z0-1 ; z >= start ; z-- ) {
218 wd0 = *(unative_t *)&dptr[z][d+0*NSIZE];
219 wd1 = *(unative_t *)&dptr[z][d+1*NSIZE];
220 wd2 = *(unative_t *)&dptr[z][d+2*NSIZE];
221 wd3 = *(unative_t *)&dptr[z][d+3*NSIZE];
222 wd4 = *(unative_t *)&dptr[z][d+4*NSIZE];
223 wd5 = *(unative_t *)&dptr[z][d+5*NSIZE];
224 wd6 = *(unative_t *)&dptr[z][d+6*NSIZE];
225 wd7 = *(unative_t *)&dptr[z][d+7*NSIZE];
226 wp0 ^= wd0;
227 wp1 ^= wd1;
228 wp2 ^= wd2;
229 wp3 ^= wd3;
230 wp4 ^= wd4;
231 wp5 ^= wd5;
232 wp6 ^= wd6;
233 wp7 ^= wd7;
234 w20 = MASK(wq0);
235 w21 = MASK(wq1);
236 w22 = MASK(wq2);
237 w23 = MASK(wq3);
238 w24 = MASK(wq4);
239 w25 = MASK(wq5);
240 w26 = MASK(wq6);
241 w27 = MASK(wq7);
242 w10 = SHLBYTE(wq0);
243 w11 = SHLBYTE(wq1);
244 w12 = SHLBYTE(wq2);
245 w13 = SHLBYTE(wq3);
246 w14 = SHLBYTE(wq4);
247 w15 = SHLBYTE(wq5);
248 w16 = SHLBYTE(wq6);
249 w17 = SHLBYTE(wq7);
250 w20 &= NBYTES(0x1d);
251 w21 &= NBYTES(0x1d);
252 w22 &= NBYTES(0x1d);
253 w23 &= NBYTES(0x1d);
254 w24 &= NBYTES(0x1d);
255 w25 &= NBYTES(0x1d);
256 w26 &= NBYTES(0x1d);
257 w27 &= NBYTES(0x1d);
258 w10 ^= w20;
259 w11 ^= w21;
260 w12 ^= w22;
261 w13 ^= w23;
262 w14 ^= w24;
263 w15 ^= w25;
264 w16 ^= w26;
265 w17 ^= w27;
266 wq0 = w10 ^ wd0;
267 wq1 = w11 ^ wd1;
268 wq2 = w12 ^ wd2;
269 wq3 = w13 ^ wd3;
270 wq4 = w14 ^ wd4;
271 wq5 = w15 ^ wd5;
272 wq6 = w16 ^ wd6;
273 wq7 = w17 ^ wd7;
274 }
275 /* P/Q left side optimization */
276 for ( z = start-1 ; z >= 0 ; z-- ) {
277 w20 = MASK(wq0);
278 w21 = MASK(wq1);
279 w22 = MASK(wq2);
280 w23 = MASK(wq3);
281 w24 = MASK(wq4);
282 w25 = MASK(wq5);
283 w26 = MASK(wq6);
284 w27 = MASK(wq7);
285 w10 = SHLBYTE(wq0);
286 w11 = SHLBYTE(wq1);
287 w12 = SHLBYTE(wq2);
288 w13 = SHLBYTE(wq3);
289 w14 = SHLBYTE(wq4);
290 w15 = SHLBYTE(wq5);
291 w16 = SHLBYTE(wq6);
292 w17 = SHLBYTE(wq7);
293 w20 &= NBYTES(0x1d);
294 w21 &= NBYTES(0x1d);
295 w22 &= NBYTES(0x1d);
296 w23 &= NBYTES(0x1d);
297 w24 &= NBYTES(0x1d);
298 w25 &= NBYTES(0x1d);
299 w26 &= NBYTES(0x1d);
300 w27 &= NBYTES(0x1d);
301 wq0 = w10 ^ w20;
302 wq1 = w11 ^ w21;
303 wq2 = w12 ^ w22;
304 wq3 = w13 ^ w23;
305 wq4 = w14 ^ w24;
306 wq5 = w15 ^ w25;
307 wq6 = w16 ^ w26;
308 wq7 = w17 ^ w27;
309 }
310 *(unative_t *)&p[d+NSIZE*0] ^= wp0;
311 *(unative_t *)&p[d+NSIZE*1] ^= wp1;
312 *(unative_t *)&p[d+NSIZE*2] ^= wp2;
313 *(unative_t *)&p[d+NSIZE*3] ^= wp3;
314 *(unative_t *)&p[d+NSIZE*4] ^= wp4;
315 *(unative_t *)&p[d+NSIZE*5] ^= wp5;
316 *(unative_t *)&p[d+NSIZE*6] ^= wp6;
317 *(unative_t *)&p[d+NSIZE*7] ^= wp7;
318 *(unative_t *)&q[d+NSIZE*0] ^= wq0;
319 *(unative_t *)&q[d+NSIZE*1] ^= wq1;
320 *(unative_t *)&q[d+NSIZE*2] ^= wq2;
321 *(unative_t *)&q[d+NSIZE*3] ^= wq3;
322 *(unative_t *)&q[d+NSIZE*4] ^= wq4;
323 *(unative_t *)&q[d+NSIZE*5] ^= wq5;
324 *(unative_t *)&q[d+NSIZE*6] ^= wq6;
325 *(unative_t *)&q[d+NSIZE*7] ^= wq7;
326 }
327
