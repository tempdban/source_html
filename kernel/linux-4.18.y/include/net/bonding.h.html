48/* IMPORTANT: bond_first/last_slave can return NULL in case of an empty list */
49#define bond_first_slave(bond) \
50 (bond_has_slaves(bond) ? \
51 netdev_adjacent_get_private(bond_slave_list(bond)->next) : \
52 NULL)
53#define bond_last_slave(bond) \
54 (bond_has_slaves(bond) ? \
55 netdev_adjacent_get_private(bond_slave_list(bond)->prev) : \
56 NULL)

344static inline void bond_set_backup_slave(struct slave *slave)
345{
346 if (!slave->backup) {
347 slave->backup = 1;
348 bond_queue_slave_event(slave);
349 bond_lower_state_changed(slave);
350 }

353static inline void bond_set_slave_state(struct slave *slave,
354 int slave_state, bool notify)
355{
356 if (slave->backup == slave_state)
357 return;
358
359 slave->backup = slave_state;
360 if (notify) {
361 bond_lower_state_changed(slave);
362 bond_queue_slave_event(slave);
363 slave->should_notify = 0;
364 } else {
365 if (slave->should_notify)
366 slave->should_notify = 0;
367 else
368 slave->should_notify = 1;
369 }
370}

372static inline void bond_slave_state_change(struct bonding *bond)
373{
374 struct list_head *iter;
375 struct slave *tmp;
376
377 bond_for_each_slave(bond, tmp, iter) {
378 if (tmp->link == BOND_LINK_UP)
379 bond_set_active_slave(tmp);
380 else if (tmp->link == BOND_LINK_DOWN)
381 bond_set_backup_slave(tmp);
382 }
383}
384
385static inline void bond_slave_state_notify(struct bonding *bond)
386{
387 struct list_head *iter;
388 struct slave *tmp;
389
390 bond_for_each_slave(bond, tmp, iter) {
391 if (tmp->should_notify) {
392 bond_lower_state_changed(tmp);
393 tmp->should_notify = 0;
394 }
395 }
396}

465/* Get the oldest arp which we've received on this slave for bond's
466 * arp_targets.
467 */
468static inline unsigned long slave_oldest_target_arp_rx(struct bonding *bond,
469 struct slave *slave)
470{
471 int i = 1;
472 unsigned long ret = slave->target_last_arp_rx[0];
473
474 for (; (i < BOND_MAX_ARP_TARGETS) && bond->params.arp_targets[i]; i++)
475 if (time_before(slave->target_last_arp_rx[i], ret))
476 ret = slave->target_last_arp_rx[i];
477
478 return ret;
479}

481static inline unsigned long slave_last_rx(struct bonding *bond,
482 struct slave *slave)
483{
484 if (bond->params.arp_all_targets == BOND_ARP_TARGETS_ALL)
485 return slave_oldest_target_arp_rx(bond, slave);
486
487 return slave->last_rx;
488}

532static inline void bond_commit_link_state(struct slave *slave, bool notify)
533{
534 if (slave->link == slave->link_new_state)
535 return;
536
537 slave->link = slave->link_new_state;
538 if (notify) {
539 bond_queue_slave_event(slave);
540 bond_lower_state_changed(slave);
541 slave->should_notify_link = 0;
542 } else {
543 if (slave->should_notify_link)
544 slave->should_notify_link = 0;
545 else
546 slave->should_notify_link = 1;
547 }
548}

551 bool notify)
552{
553 bond_propose_link_state(slave, state);
554 bond_commit_link_state(slave, notify);
555}

571static inline __be32 bond_confirm_addr(struct net_device *dev, __be32 dst, __be32 local)
572{
573 struct in_device *in_dev;
574 __be32 addr = 0;
575
576 rcu_read_lock();
577 in_dev = __in_dev_get_rcu(dev);
578
579 if (in_dev)
580 addr = inet_confirm_addr(dev_net(dev), in_dev, dst, local,
581 RT_SCOPE_HOST);
582 rcu_read_unlock();
583 return addr;
584}

652static inline struct slave *bond_slave_has_mac(struct bonding *bond,

666static inline struct slave *bond_slave_has_mac_rcu(struct bonding *bond,
667 const u8 *mac)
668{
669 struct list_head *iter;
670 struct slave *tmp;
671

678
679/* Caller must hold rcu_read_lock() for read */
680static inline bool bond_slave_has_mac_rx(struct bonding *bond, const u8 *mac)
681{
682 struct list_head *iter;
683 struct slave *tmp;
684 struct netdev_hw_addr *ha;
685
686 bond_for_each_slave_rcu(bond, tmp, iter)
687 if (ether_addr_equal_64bits(mac, tmp->dev->dev_addr))
688 return true;
689
690 if (netdev_uc_empty(bond->dev))
691 return false;
692
693 netdev_for_each_uc_addr(ha, bond->dev)
694 if (ether_addr_equal_64bits(mac, ha->addr))
695 return true;
696
697 return false;
698}

700/* Check if the ip is present in arp ip list, or first free slot if ip == 0
701 * Returns -1 if not found, index if found
702 */
703static inline int bond_get_targets_ip(__be32 *targets, __be32 ip)
704{
705 int i;
706
707 for (i = 0; i < BOND_MAX_ARP_TARGETS; i++)
708 if (targets[i] == ip)
709 return i;
710 else if (targets[i] == 0)
711 break;
712
713 return -1;
714}
