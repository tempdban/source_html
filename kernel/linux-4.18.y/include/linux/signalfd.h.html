16/*
17 * Deliver the signal to listening signalfd.
18 */
19static inline void signalfd_notify(struct task_struct *tsk, int sig)
20{
21 if (unlikely(waitqueue_active(&tsk->sighand->signalfd_wqh)))
22 wake_up(&tsk->sighand->signalfd_wqh);
23}
24
25extern void signalfd_cleanup(struct sighand_struct *sighand);
