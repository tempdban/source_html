97/*
98 * __MAP - apply a macro to syscall arguments
99 * __MAP(n, m, t1, a1, t2, a2, ..., tn, an) will expand to
100 * m(t1, a1), m(t2, a2), ..., m(tn, an)
101 * The first argument must be equal to the amount of type/name
102 * pairs given. Note that this list of pairs (i.e. the arguments
103 * of __MAP starting at the third one) is in the same format as
104 * for SYSCALL_DEFINE<n>/COMPAT_SYSCALL_DEFINE<n>
105 */
106#define __MAP0(m,...)
107#define __MAP1(m,t,a,...) m(t,a)
108#define __MAP2(m,t,a,...) m(t,a), __MAP1(m,__VA_ARGS__)
109#define __MAP3(m,t,a,...) m(t,a), __MAP2(m,__VA_ARGS__)
110#define __MAP4(m,t,a,...) m(t,a), __MAP3(m,__VA_ARGS__)
111#define __MAP5(m,t,a,...) m(t,a), __MAP4(m,__VA_ARGS__)
112#define __MAP6(m,t,a,...) m(t,a), __MAP5(m,__VA_ARGS__)
113#define __MAP(n,...) __MAP##n(__VA_ARGS__)
114
115#define __SC_DECL(t, a) t a
116#define __TYPE_AS(t, v) __same_type((__force t)0, v)
117#define __TYPE_IS_L(t) (__TYPE_AS(t, 0L))
118#define __TYPE_IS_UL(t) (__TYPE_AS(t, 0UL))
119#define __TYPE_IS_LL(t) (__TYPE_AS(t, 0LL) || __TYPE_AS(t, 0ULL))
120#define __SC_LONG(t, a) __typeof(__builtin_choose_expr(__TYPE_IS_LL(t), 0LL, 0L)) a
121#define __SC_CAST(t, a) (__force t) a
122#define __SC_ARGS(t, a) a
123#define __SC_TEST(t, a) (void)BUILD_BUG_ON_ZERO(!__TYPE_IS_LL(t) && sizeof(t) > sizeof(long))

134#define SYSCALL_TRACE_ENTER_EVENT(sname) \
135 static struct syscall_metadata __syscall_meta_##sname; \
136 static struct trace_event_call __used \
137 event_enter_##sname = { \
138 .class = &event_class_syscall_enter, \
139 { \
140 .name = "sys_enter"#sname, \
141 }, \
142 .event.funcs = &enter_syscall_print_funcs, \
143 .data = (void *)&__syscall_meta_##sname,\
144 .flags = TRACE_EVENT_FL_CAP_ANY, \
145 }; \
146 static struct trace_event_call __used \
147 __attribute__((section("_ftrace_events"))) \
148 *__event_enter_##sname = &event_enter_##sname;
149
150#define SYSCALL_TRACE_EXIT_EVENT(sname) \
151 static struct syscall_metadata __syscall_meta_##sname; \
152 static struct trace_event_call __used \
153 event_exit_##sname = { \
154 .class = &event_class_syscall_exit, \
155 { \
156 .name = "sys_exit"#sname, \
157 }, \
158 .event.funcs = &exit_syscall_print_funcs, \
159 .data = (void *)&__syscall_meta_##sname,\
160 .flags = TRACE_EVENT_FL_CAP_ANY, \
161 }; \
162 static struct trace_event_call __used \
163 __attribute__((section("_ftrace_events"))) \
164 *__event_exit_##sname = &event_exit_##sname;
165
166#define SYSCALL_METADATA(sname, nb, ...) \
167 static const char *types_##sname[] = { \
168 __MAP(nb,__SC_STR_TDECL,__VA_ARGS__) \
169 }; \
170 static const char *args_##sname[] = { \
171 __MAP(nb,__SC_STR_ADECL,__VA_ARGS__) \
172 }; \
173 SYSCALL_TRACE_ENTER_EVENT(sname); \
174 SYSCALL_TRACE_EXIT_EVENT(sname); \
175 static struct syscall_metadata __used \
176 __syscall_meta_##sname = { \
177 .name = "sys"#sname, \
178 .syscall_nr = -1, /* Filled in at boot */ \
179 .nb_args = nb, \
180 .types = nb ? types_##sname : NULL, \
181 .args = nb ? args_##sname : NULL, \
182 .enter_event = &event_enter_##sname, \
183 .exit_event = &event_exit_##sname, \
184 .enter_fields = LIST_HEAD_INIT(__syscall_meta_##sname.enter_fields), \
185 }; \
186 static struct syscall_metadata __used \
187 __attribute__((section("__syscalls_metadata"))) \
188 *__p_syscall_meta_##sname = &__syscall_meta_##sname;
189
190static inline int is_syscall_trace_event(struct trace_event_call *tp_event)
191{
192 return tp_event->class == &event_class_syscall_enter ||
193 tp_event->class == &event_class_syscall_exit;
194}

228/*
229 * The asmlinkage stub is aliased to a function named __se_sys_*() which
230 * sign-extends 32-bit ints to longs whenever needed. The actual work is
231 * done within __do_sys_*().
232 */
233#ifndef __SYSCALL_DEFINEx
234#define __SYSCALL_DEFINEx(x, name, ...) \
235 __diag_push(); \
236 __diag_ignore(GCC, 8, "-Wattribute-alias", \
237 "Type aliasing is used to sanitize syscall arguments");\
238 asmlinkage long sys##name(__MAP(x,__SC_DECL,__VA_ARGS__)) \
239 __attribute__((alias(__stringify(__se_sys##name)))); \
240 ALLOW_ERROR_INJECTION(sys##name, ERRNO); \
241 static inline long __do_sys##name(__MAP(x,__SC_DECL,__VA_ARGS__));\
242 asmlinkage long __se_sys##name(__MAP(x,__SC_LONG,__VA_ARGS__)); \
243 asmlinkage long __se_sys##name(__MAP(x,__SC_LONG,__VA_ARGS__)) \
244 { \
245 long ret = __do_sys##name(__MAP(x,__SC_CAST,__VA_ARGS__));\
246 __MAP(x,__SC_TEST,__VA_ARGS__); \
247 __PROTECT(x, ret,__MAP(x,__SC_ARGS,__VA_ARGS__)); \
248 return ret; \
249 } \
250 __diag_pop(); \
251 static inline long __do_sys##name(__MAP(x,__SC_DECL,__VA_ARGS__))
252#endif /* __SYSCALL_DEFINEx */

254/*
255 * Called before coming back to user-mode. Returning to user-mode with an
256 * address limit different than USER_DS can allow to overwrite kernel memory.
257 */
258static inline void addr_limit_user_check(void)
259{
260#ifdef TIF_FSCHECK
261 if (!test_thread_flag(TIF_FSCHECK))
262 return;
263#endif
264
265 if (CHECK_DATA_CORRUPTION(!segment_eq(get_fs(), USER_DS),
266 "Invalid address limit on user-mode return"))
267 force_sig(SIGKILL, current);
268
269#ifdef TIF_FSCHECK
270 clear_thread_flag(TIF_FSCHECK);
271#endif
272}
