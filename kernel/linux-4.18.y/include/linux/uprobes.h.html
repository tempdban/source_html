96struct return_instance {
97 struct uprobe *uprobe;
98 unsigned long func;
99 unsigned long stack; /* stack pointer */
100 unsigned long orig_ret_vaddr; /* original return address */
101 bool chained; /* true, if instance is nested */
102
103 struct return_instance *next; /* keep as stack */
104};
