16static inline void *
17pci_alloc_consistent(struct pci_dev *hwdev, size_t size,
18 dma_addr_t *dma_handle)
19{
20 return dma_alloc_coherent(&hwdev->dev, size, dma_handle, GFP_ATOMIC);
21}
