23#define BDO_MODE_CARRIER1 (4 << 28)
24#define BDO_MODE_CARRIER2 (5 << 28)
25#define BDO_MODE_CARRIER3 (6 << 28)
26#define BDO_MODE_EYE (7 << 28)
27#define BDO_MODE_TESTDATA (8 << 28)
