51/**
52 * rcu_dereference_rtnl - rcu_dereference with debug checking
53 * @p: The pointer to read, prior to dereferencing
54 *
55 * Do an rcu_dereference(p), but check caller either holds rcu_read_lock()
56 * or RTNL. Note : Please prefer rtnl_dereference() or rcu_dereference()
57 */
58#define rcu_dereference_rtnl(p) \
59 rcu_dereference_check(p, lockdep_rtnl_is_held())
60
61/**
62 * rcu_dereference_bh_rtnl - rcu_dereference_bh with debug checking
63 * @p: The pointer to read, prior to dereference
64 *
65 * Do an rcu_dereference_bh(p), but check caller either holds rcu_read_lock_bh()
66 * or RTNL. Note : Please prefer rtnl_dereference() or rcu_dereference_bh()
67 */
68#define rcu_dereference_bh_rtnl(p) \
69 rcu_dereference_bh_check(p, lockdep_rtnl_is_held())
70
71/**
72 * rtnl_dereference - fetch RCU pointer when updates are prevented by RTNL
73 * @p: The pointer to read, prior to dereferencing
74 *
75 * Return the value of the specified RCU-protected pointer, but omit
76 * the READ_ONCE(), because caller holds RTNL.
77 */
78#define rtnl_dereference(p) \
79 rcu_dereference_protected(p, lockdep_rtnl_is_held())
