290/**
291 * struct hwmon_ops - hwmon device operations
292 * @is_visible: Callback to return attribute visibility. Mandatory.
293 * Parameters are:
294 * @const void *drvdata:
295 * Pointer to driver-private data structure passed
296 * as argument to hwmon_device_register_with_info().
297 * @type: Sensor type
298 * @attr: Sensor attribute
299 * @channel:
300 * Channel number
301 * The function returns the file permissions.
302 * If the return value is 0, no attribute will be created.
303 * @read: Read callback for data attributes. Mandatory if readable
304 * data attributes are present.
305 * Parameters are:
306 * @dev: Pointer to hardware monitoring device
307 * @type: Sensor type
308 * @attr: Sensor attribute
309 * @channel:
310 * Channel number
311 * @val: Pointer to returned value
312 * The function returns 0 on success or a negative error number.
313 * @read_string:
314 * Read callback for string attributes. Mandatory if string
315 * attributes are present.
316 * Parameters are:
317 * @dev: Pointer to hardware monitoring device
318 * @type: Sensor type
319 * @attr: Sensor attribute
320 * @channel:
321 * Channel number
322 * @str: Pointer to returned string
323 * The function returns 0 on success or a negative error number.
324 * @write: Write callback for data attributes. Mandatory if writeable
325 * data attributes are present.
326 * Parameters are:
327 * @dev: Pointer to hardware monitoring device
328 * @type: Sensor type
329 * @attr: Sensor attribute
330 * @channel:
331 * Channel number
332 * @val: Value to write
333 * The function returns 0 on success or a negative error number.
334 */
335struct hwmon_ops {
336 umode_t (*is_visible)(const void *drvdata, enum hwmon_sensor_types type,
337 u32 attr, int channel);
338 int (*read)(struct device *dev, enum hwmon_sensor_types type,
339 u32 attr, int channel, long *val);
340 int (*read_string)(struct device *dev, enum hwmon_sensor_types type,
341 u32 attr, int channel, const char **str);
342 int (*write)(struct device *dev, enum hwmon_sensor_types type,
343 u32 attr, int channel, long val);
344};
