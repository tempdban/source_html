15/*
16 * Types of block ranges for iomap mappings:
17 */
18#define IOMAP_HOLE 0x01 /* no blocks allocated, need allocation */
19#define IOMAP_DELALLOC 0x02 /* delayed allocation blocks */
20#define IOMAP_MAPPED 0x03 /* blocks allocated at @addr */
21#define IOMAP_UNWRITTEN 0x04 /* blocks allocated at @addr in unwritten state */
22#define IOMAP_INLINE 0x05 /* data inline in the inode */
23
24/*
25 * Flags for all iomap mappings:
26 *
27 * IOMAP_F_DIRTY indicates the inode has uncommitted metadata needed to access
28 * written data and requires fdatasync to commit them to persistent storage.
29 */
30#define IOMAP_F_NEW 0x01 /* blocks have been newly allocated */
31#define IOMAP_F_DIRTY 0x02 /* uncommitted metadata */
32
33/*
34 * Flags that only need to be reported for IOMAP_REPORT requests:
35 */
36#define IOMAP_F_MERGED 0x10 /* contains multiple blocks/extents */
37#define IOMAP_F_SHARED 0x20 /* block shared with another file */
38
39/*
40 * Flags from 0x1000 up are for file system specific usage:
41 */
42#define IOMAP_F_PRIVATE 0x1000
43
44
45/*
46 * Magic value for addr:
47 */
48#define IOMAP_NULL_ADDR -1ULL /* addr is not valid */
49
50struct iomap {
51 u64 addr; /* disk offset of mapping, bytes */
52 loff_t offset; /* file offset of mapping, bytes */
53 u64 length; /* length of mapping, bytes */
54 u16 type; /* type of mapping */
55 u16 flags; /* flags for mapping */
56 struct block_device *bdev; /* block device for I/O */
57 struct dax_device *dax_dev; /* dax_dev for dax operations */
58};
59
60/*
61 * Flags for iomap_begin / iomap_end. No flag implies a read.
62 */
63#define IOMAP_WRITE (1 << 0) /* writing, must allocate blocks */
64#define IOMAP_ZERO (1 << 1) /* zeroing operation, may skip holes */
65#define IOMAP_REPORT (1 << 2) /* report extent status, e.g. FIEMAP */
66#define IOMAP_FAULT (1 << 3) /* mapping for page fault */
67#define IOMAP_DIRECT (1 << 4) /* direct I/O */
68#define IOMAP_NOWAIT (1 << 5) /* do not block */
69
70struct iomap_ops {
71 /*
72 * Return the existing mapping at pos, or reserve space starting at
73 * pos for up to length, as long as we can do it as a single mapping.
74 * The actual length is returned in iomap->length.
75 */
76 int (*iomap_begin)(struct inode *inode, loff_t pos, loff_t length,
77 unsigned flags, struct iomap *iomap);
78
79 /*
80 * Commit and/or unreserve space previous allocated using iomap_begin.
81 * Written indicates the length of the successful write operation which
82 * needs to be commited, while the rest needs to be unreserved.
83 * Written might be zero if no data was written.
84 */
85 int (*iomap_end)(struct inode *inode, loff_t pos, loff_t length,
86 ssize_t written, unsigned flags, struct iomap *iomap);
87};
