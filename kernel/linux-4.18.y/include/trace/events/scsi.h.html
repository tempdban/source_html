200TRACE_EVENT(scsi_dispatch_cmd_start,
201
202 TP_PROTO(struct scsi_cmnd *cmd),
203
204 TP_ARGS(cmd),
205
206 TP_STRUCT__entry(
207 __field( unsigned int, host_no )
208 __field( unsigned int, channel )
209 __field( unsigned int, id )
210 __field( unsigned int, lun )
211 __field( unsigned int, opcode )
212 __field( unsigned int, cmd_len )
213 __field( unsigned int, data_sglen )
214 __field( unsigned int, prot_sglen )
215 __field( unsigned char, prot_op )
216 __dynamic_array(unsigned char, cmnd, cmd->cmd_len)
217 ),
218
219 TP_fast_assign(
220 __entry->host_no = cmd->device->host->host_no;
221 __entry->channel = cmd->device->channel;
222 __entry->id = cmd->device->id;
223 __entry->lun = cmd->device->lun;
224 __entry->opcode = cmd->cmnd[0];
225 __entry->cmd_len = cmd->cmd_len;
226 __entry->data_sglen = scsi_sg_count(cmd);
227 __entry->prot_sglen = scsi_prot_sg_count(cmd);
228 __entry->prot_op = scsi_get_prot_op(cmd);
229 memcpy(__get_dynamic_array(cmnd), cmd->cmnd, cmd->cmd_len);
230 ),
231
232 TP_printk("host_no=%u channel=%u id=%u lun=%u data_sgl=%u prot_sgl=%u" \
233 " prot_op=%s cmnd=(%s %s raw=%s)",
234 __entry->host_no, __entry->channel, __entry->id,
235 __entry->lun, __entry->data_sglen, __entry->prot_sglen,
236 show_prot_op_name(__entry->prot_op),
237 show_opcode_name(__entry->opcode),
238 __parse_cdb(__get_dynamic_array(cmnd), __entry->cmd_len),
239 __print_hex(__get_dynamic_array(cmnd), __entry->cmd_len))
240);

287DECLARE_EVENT_CLASS(scsi_cmd_done_timeout_template,
288
289 TP_PROTO(struct scsi_cmnd *cmd),
290
291 TP_ARGS(cmd),
292
293 TP_STRUCT__entry(
294 __field( unsigned int, host_no )
295 __field( unsigned int, channel )
296 __field( unsigned int, id )
297 __field( unsigned int, lun )
298 __field( int, result )
299 __field( unsigned int, opcode )
300 __field( unsigned int, cmd_len )
301 __field( unsigned int, data_sglen )
302 __field( unsigned int, prot_sglen )
303 __field( unsigned char, prot_op )
304 __dynamic_array(unsigned char, cmnd, cmd->cmd_len)
305 ),
306
307 TP_fast_assign(
308 __entry->host_no = cmd->device->host->host_no;
309 __entry->channel = cmd->device->channel;
310 __entry->id = cmd->device->id;
311 __entry->lun = cmd->device->lun;
312 __entry->result = cmd->result;
313 __entry->opcode = cmd->cmnd[0];
314 __entry->cmd_len = cmd->cmd_len;
315 __entry->data_sglen = scsi_sg_count(cmd);
316 __entry->prot_sglen = scsi_prot_sg_count(cmd);
317 __entry->prot_op = scsi_get_prot_op(cmd);
318 memcpy(__get_dynamic_array(cmnd), cmd->cmnd, cmd->cmd_len);
319 ),
320
321 TP_printk("host_no=%u channel=%u id=%u lun=%u data_sgl=%u " \
322 "prot_sgl=%u prot_op=%s cmnd=(%s %s raw=%s) result=(driver=" \
323 "%s host=%s message=%s status=%s)",
324 __entry->host_no, __entry->channel, __entry->id,
325 __entry->lun, __entry->data_sglen, __entry->prot_sglen,
326 show_prot_op_name(__entry->prot_op),
327 show_opcode_name(__entry->opcode),
328 __parse_cdb(__get_dynamic_array(cmnd), __entry->cmd_len),
329 __print_hex(__get_dynamic_array(cmnd), __entry->cmd_len),
330 show_driverbyte_name(((__entry->result) >> 24) & 0xff),
331 show_hostbyte_name(((__entry->result) >> 16) & 0xff),
332 show_msgbyte_name(((__entry->result) >> 8) & 0xff),
333 show_statusbyte_name(__entry->result & 0xff))
334);
