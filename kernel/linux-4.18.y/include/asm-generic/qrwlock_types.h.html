9/*
10 * The queue read/write lock data structure
11 */

13typedef struct qrwlock {
14 union {
15 atomic_t cnts;
16 struct {
17#ifdef __LITTLE_ENDIAN
18 u8 wlocked; /* Locked for write? */

20#else
21 u8 __lstate[3];

23#endif
24 };
25 };
26 arch_spinlock_t wait_lock;
27} arch_rwlock_t;

29#define __ARCH_RW_LOCK_UNLOCKED { \

32}
