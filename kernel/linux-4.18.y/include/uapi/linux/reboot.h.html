16/*
17 * Commands accepted by the _reboot() system call.
18 *
19 * RESTART Restart system using default command and mode.
20 * HALT Stop OS and give system control to ROM monitor, if any.
21 * CAD_ON Ctrl-Alt-Del sequence causes RESTART command.
22 * CAD_OFF Ctrl-Alt-Del sequence sends SIGINT to init task.
23 * POWER_OFF Stop OS and remove all power from system, if possible.
24 * RESTART2 Restart system using given command string.
25 * SW_SUSPEND Suspend system using software suspend if compiled in.
26 * KEXEC Restart system using a previously loaded Linux kernel
27 */
28
29#define LINUX_REBOOT_CMD_RESTART 0x01234567
30#define LINUX_REBOOT_CMD_HALT 0xCDEF0123
31#define LINUX_REBOOT_CMD_CAD_ON 0x89ABCDEF
32#define LINUX_REBOOT_CMD_CAD_OFF 0x00000000
33#define LINUX_REBOOT_CMD_POWER_OFF 0x4321FEDC
34#define LINUX_REBOOT_CMD_RESTART2 0xA1B2C3D4
35#define LINUX_REBOOT_CMD_SW_SUSPEND 0xD000FCE2
36#define LINUX_REBOOT_CMD_KEXEC 0x45584543
