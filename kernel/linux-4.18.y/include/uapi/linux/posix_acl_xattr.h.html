18#ifndef __UAPI_POSIX_ACL_XATTR_H
19#define __UAPI_POSIX_ACL_XATTR_H
20
21#include <linux/types.h>
22
23/* Supported ACL a_version fields */
24#define POSIX_ACL_XATTR_VERSION 0x0002
25
26/* An undefined entry e_id value */
27#define ACL_UNDEFINED_ID (-1)
28
29struct posix_acl_xattr_entry {
30 __le16 e_tag;
31 __le16 e_perm;
32 __le32 e_id;
33};
34
35struct posix_acl_xattr_header {
36 __le32 a_version;
37};
38
39#endif /* __UAPI_POSIX_ACL_XATTR_H */
