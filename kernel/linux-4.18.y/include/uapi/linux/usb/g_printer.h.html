22#ifndef __LINUX_USB_G_PRINTER_H
23#define __LINUX_USB_G_PRINTER_H
24
25#define PRINTER_NOT_ERROR 0x08
26#define PRINTER_SELECTED 0x10
27#define PRINTER_PAPER_EMPTY 0x20
28
29/* The 'g' code is also used by gadgetfs ioctl requests.
30 * Don't add any colliding codes to either driver, and keep
31 * them in unique ranges (size 0x20 for now).
32 */
33#define GADGET_GET_PRINTER_STATUS _IOR('g', 0x21, unsigned char)
34#define GADGET_SET_PRINTER_STATUS _IOWR('g', 0x22, unsigned char)
35
36#endif /* __LINUX_USB_G_PRINTER_H */
