39enum op_dest_type {
40 OP_DEST_REG,
41 OP_DEST_REG_INDIRECT,
42 OP_DEST_MEM,
43 OP_DEST_PUSH,
44 OP_DEST_LEAVE,
45};
46
47struct op_dest {
48 enum op_dest_type type;
49 unsigned char reg;
50 int offset;
51};
52
53enum op_src_type {
54 OP_SRC_REG,
55 OP_SRC_REG_INDIRECT,
56 OP_SRC_CONST,
57 OP_SRC_POP,
58 OP_SRC_ADD,
59 OP_SRC_AND,
60};
61
62struct op_src {
63 enum op_src_type type;
64 unsigned char reg;
65 int offset;
66};
67
68struct stack_op {
69 struct op_dest dest;
70 struct op_src src;
71};
