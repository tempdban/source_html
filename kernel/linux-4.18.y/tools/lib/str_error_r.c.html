7/*
8 * The tools so far have been using the strerror_r() GNU variant, that returns
9 * a string, be it the buffer passed or something else.
10 *
11 * But that, besides being tricky in cases where we expect that the function
12 * using strerror_r() returns the error formatted in a provided buffer (we have
13 * to check if it returned something else and copy that instead), breaks the
14 * build on systems not using glibc, like Alpine Linux, where musl libc is
15 * used.
16 *
17 * So, introduce yet another wrapper, str_error_r(), that has the GNU
18 * interface, but uses the portable XSI variant of strerror_r(), so that users
19 * rest asured that the provided buffer is used and it is what is returned.
20 */
21char *str_error_r(int errnum, char *buf, size_t buflen)
22{
23 int err = strerror_r(errnum, buf, buflen);
24 if (err)
25 snprintf(buf, buflen, "INTERNAL ERROR: strerror_r(%d, [buf], %zd)=%d", errnum, buflen, err);
26 return buf;
27}
