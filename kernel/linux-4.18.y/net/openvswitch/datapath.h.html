39/**
40 * struct dp_stats_percpu - per-cpu packet processing statistics for a given
41 * datapath.
42 * @n_hit: Number of received packets for which a matching flow was found in
43 * the flow table.
44 * @n_miss: Number of received packets that had no matching flow in the flow
45 * table. The sum of @n_hit and @n_miss is the number of packets that have
46 * been received by the datapath.
47 * @n_lost: Number of received packets that had no matching flow in the flow
48 * table that could not be sent to userspace (normally due to an overflow in
49 * one of the datapath's queues).
50 * @n_mask_hit: Number of masks looked up for flow match.
51 * @n_mask_hit / (@n_hit + @n_missed) will be the average masks looked
52 * up per packet.
53 */
54struct dp_stats_percpu {
55 u64 n_hit;
56 u64 n_missed;
57 u64 n_lost;
58 u64 n_mask_hit;
59 struct u64_stats_sync syncp;
60};

62/**
63 * struct datapath - datapath for flow-based packet switching
64 * @rcu: RCU callback head for deferred destruction.
65 * @list_node: Element in global 'dps' list.
66 * @table: flow table.
67 * @ports: Hash table for ports. %OVSP_LOCAL port always exists. Protected by
68 * ovs_mutex and RCU.
69 * @stats_percpu: Per-CPU datapath statistics.
70 * @net: Reference to net namespace.
71 * @max_headroom: the maximum headroom of all vports in this datapath; it will
72 * be used by all the internal vports in this dp.
73 *
74 * Context: See the comment on locking at the top of datapath.c for additional
75 * locking information.
76 */
77struct datapath {
78 struct rcu_head rcu;
79 struct list_head list_node;
80
81 /* Flow table. */
82 struct flow_table table;
83
84 /* Switch ports. */
85 struct hlist_head *ports;
86
87 /* Stats. */
88 struct dp_stats_percpu __percpu *stats_percpu;
89
90 /* Network namespace ref. */
91 possible_net_t net;
92
93 u32 user_features;
94
95 u32 max_headroom;
96
97 /* Switch meters. */
98 struct hlist_head *meters;
99};

101/**
102 * struct ovs_skb_cb - OVS data in skb CB
103 * @input_vport: The original vport packet came in on. This value is cached
104 * when a packet is received by OVS.
105 * @mru: The maximum received fragement size; 0 if the packet is not
106 * fragmented.
107 * @acts_origlen: The netlink size of the flow actions applied to this skb.
108 * @cutlen: The number of bytes from the packet end to be removed.
109 */
110struct ovs_skb_cb {
111 struct vport *input_vport;
112 u16 mru;
113 u16 acts_origlen;
114 u32 cutlen;
115};
116#define OVS_CB(skb) ((struct ovs_skb_cb *)(skb)->cb)

118/**
119 * struct dp_upcall - metadata to include with a packet to send to userspace
120 * @cmd: One of %OVS_PACKET_CMD_*.
121 * @userdata: If nonnull, its variable-length value is passed to userspace as
122 * %OVS_PACKET_ATTR_USERDATA.
123 * @portid: Netlink portid to which packet should be sent. If @portid is 0
124 * then no packet is sent and the packet is accounted in the datapath's @n_lost
125 * counter.
126 * @egress_tun_info: If nonnull, becomes %OVS_PACKET_ATTR_EGRESS_TUN_KEY.
127 * @mru: If not zero, Maximum received IP fragment size.
128 */
129struct dp_upcall_info {
130 struct ip_tunnel_info *egress_tun_info;
131 const struct nlattr *userdata;
132 const struct nlattr *actions;
133 int actions_len;
134 u32 portid;
135 u8 cmd;
136 u16 mru;
137};

201/* Must be called with rcu_read_lock. */
202static inline struct datapath *get_dp_rcu(struct net *net, int dp_ifindex)
203{
204 struct net_device *dev = dev_get_by_index_rcu(net, dp_ifindex);
205
206 if (dev) {
207 struct vport *vport = ovs_internal_dev_get_vport(dev);
208
209 if (vport)
210 return vport->dp;
211 }
212
213 return NULL;
214}
215
216/* The caller must hold either ovs_mutex or rcu_read_lock to keep the
217 * returned dp pointer valid.
218 */
219static inline struct datapath *get_dp(struct net *net, int dp_ifindex)
220{
221 struct datapath *dp;
222
223 WARN_ON_ONCE(!rcu_read_lock_held() && !lockdep_ovsl_is_held());
224 rcu_read_lock();
225 dp = get_dp_rcu(net, dp_ifindex);
226 rcu_read_unlock();
227
228 return dp;
229}
