20static DEFINE_IDA(mem_id_pool);
21static DEFINE_MUTEX(mem_id_lock);
22#define MEM_ID_MAX 0xFFFE
23#define MEM_ID_MIN 1
24static int mem_id_next = MEM_ID_MIN;

26static bool mem_id_init; /* false */
27static struct rhashtable *mem_id_ht;

29struct xdp_mem_allocator {
30 struct xdp_mem_info mem;
31 union {
32 void *allocator;
33 struct page_pool *page_pool;
34 struct zero_copy_allocator *zc_alloc;
35 };
36 struct rhash_head node;
37 struct rcu_head rcu;
38};

40static u32 xdp_mem_id_hashfn(const void *data, u32 len, u32 seed)
41{
42 const u32 *k = data;
43 const u32 key = *k;
44
45 BUILD_BUG_ON(FIELD_SIZEOF(struct xdp_mem_allocator, mem.id)
46 != sizeof(u32));
47
48 /* Use cyclic increasing ID as direct hash key, see rht_bucket_index */
49 return key << RHT_HASH_RESERVED_SPACE;
50}
51
52static int xdp_mem_id_cmp(struct rhashtable_compare_arg *arg,
53 const void *ptr)
54{
55 const struct xdp_mem_allocator *xa = ptr;
56 u32 mem_id = *(u32 *)arg->key;
57
58 return xa->mem.id != mem_id;
59}

61static const struct rhashtable_params mem_id_rht_params = {
62 .nelem_hint = 64,
63 .head_offset = offsetof(struct xdp_mem_allocator, node),
64 .key_offset = offsetof(struct xdp_mem_allocator, mem.id),
65 .key_len = FIELD_SIZEOF(struct xdp_mem_allocator, mem.id),
66 .max_size = MEM_ID_MAX,
67 .min_size = 8,
68 .automatic_shrinking = true,
69 .hashfn = xdp_mem_id_hashfn,
70 .obj_cmpfn = xdp_mem_id_cmp,
71};

73static void __xdp_mem_allocator_rcu_free(struct rcu_head *rcu)
74{
75 struct xdp_mem_allocator *xa;
76
77 xa = container_of(rcu, struct xdp_mem_allocator, rcu);
78
79 /* Allow this ID to be reused */
80 ida_simple_remove(&mem_id_pool, xa->mem.id);
81
82 /* Notice, driver is expected to free the *allocator,
83 * e.g. page_pool, and MUST also use RCU free.
84 */
85
86 /* Poison memory */
87 xa->mem.id = 0xFFFF;
88 xa->mem.type = 0xF0F0;
89 xa->allocator = (void *)0xDEAD9001;
90
91 kfree(xa);
92}

94static void __xdp_rxq_info_unreg_mem_model(struct xdp_rxq_info *xdp_rxq)
95{
96 struct xdp_mem_allocator *xa;
97 int id = xdp_rxq->mem.id;
98
99 if (id == 0)
100 return;
101
102 mutex_lock(&mem_id_lock);
103
104 xa = rhashtable_lookup_fast(mem_id_ht, &id, mem_id_rht_params);
105 if (xa && !rhashtable_remove_fast(mem_id_ht, &xa->node, mem_id_rht_params))
106 call_rcu(&xa->rcu, __xdp_mem_allocator_rcu_free);
107
108 mutex_unlock(&mem_id_lock);
109}

111void xdp_rxq_info_unreg(struct xdp_rxq_info *xdp_rxq)
112{
113 /* Simplify driver cleanup code paths, allow unreg "unused" */
114 if (xdp_rxq->reg_state == REG_STATE_UNUSED)
115 return;
116
117 WARN(!(xdp_rxq->reg_state == REG_STATE_REGISTERED), "Driver BUG");
118
119 __xdp_rxq_info_unreg_mem_model(xdp_rxq);
120
121 xdp_rxq->reg_state = REG_STATE_UNREGISTERED;
122 xdp_rxq->dev = NULL;
123
124 /* Reset mem info to defaults */
125 xdp_rxq->mem.id = 0;
126 xdp_rxq->mem.type = 0;
127}
128EXPORT_SYMBOL_GPL(xdp_rxq_info_unreg);

132 memset(xdp_rxq, 0, sizeof(*xdp_rxq));

135/* Returns 0 on success, negative on failure */
136int xdp_rxq_info_reg(struct xdp_rxq_info *xdp_rxq,
137 struct net_device *dev, u32 queue_index)
138{
139 if (xdp_rxq->reg_state == REG_STATE_UNUSED) {
140 WARN(1, "Driver promised not to register this");
141 return -EINVAL;
142 }
143
144 if (xdp_rxq->reg_state == REG_STATE_REGISTERED) {
145 WARN(1, "Missing unregister, handled but fix driver");
146 xdp_rxq_info_unreg(xdp_rxq);
147 }
148
149 if (!dev) {
150 WARN(1, "Missing net_device from driver");
151 return -ENODEV;
152 }
153
154 /* State either UNREGISTERED or NEW */
155 xdp_rxq_info_init(xdp_rxq);
156 xdp_rxq->dev = dev;
157 xdp_rxq->queue_index = queue_index;
158
159 xdp_rxq->reg_state = REG_STATE_REGISTERED;
160 return 0;
161}

176static int __mem_id_init_hash_table(void)
177{
178 struct rhashtable *rht;
179 int ret;
180
181 if (unlikely(mem_id_init))
182 return 0;
183
184 rht = kzalloc(sizeof(*rht), GFP_KERNEL);
185 if (!rht)
186 return -ENOMEM;
187
188 ret = rhashtable_init(rht, &mem_id_rht_params);
189 if (ret < 0) {
190 kfree(rht);
191 return ret;
192 }
193 mem_id_ht = rht;
194 smp_mb(); /* mutex lock should provide enough pairing */
195 mem_id_init = true;
196
197 return 0;
198}
199
200/* Allocate a cyclic ID that maps to allocator pointer.
201 * See: https://www.kernel.org/doc/html/latest/core-api/idr.html
202 *
203 * Caller must lock mem_id_lock.
204 */
205static int __mem_id_cyclic_get(gfp_t gfp)
206{
207 int retries = 1;
208 int id;
209
210again:
211 id = ida_simple_get(&mem_id_pool, mem_id_next, MEM_ID_MAX, gfp);
212 if (id < 0) {
213 if (id == -ENOSPC) {
214 /* Cyclic allocator, reset next id */
215 if (retries--) {
216 mem_id_next = MEM_ID_MIN;
217 goto again;
218 }
219 }
220 return id; /* errno */
221 }
222 mem_id_next = id + 1;
223
224 return id;
225}

227static bool __is_supported_mem_type(enum xdp_mem_type type)
228{
229 if (type == MEM_TYPE_PAGE_POOL)
230 return is_page_pool_compiled_in();
231
232 if (type >= MEM_TYPE_MAX)
233 return false;
234
235 return true;
236}

238int xdp_rxq_info_reg_mem_model(struct xdp_rxq_info *xdp_rxq,
239 enum xdp_mem_type type, void *allocator)
240{
241 struct xdp_mem_allocator *xdp_alloc;
242 gfp_t gfp = GFP_KERNEL;
243 int id, errno, ret;
244 void *ptr;
245
246 if (xdp_rxq->reg_state != REG_STATE_REGISTERED) {
247 WARN(1, "Missing register, driver bug");
248 return -EFAULT;
249 }
250
251 if (!__is_supported_mem_type(type))
252 return -EOPNOTSUPP;
253
254 xdp_rxq->mem.type = type;
255
256 if (!allocator) {
257 if (type == MEM_TYPE_PAGE_POOL || type == MEM_TYPE_ZERO_COPY)
258 return -EINVAL; /* Setup time check page_pool req */
259 return 0;
260 }
261
262 /* Delay init of rhashtable to save memory if feature isn't used */
263 if (!mem_id_init) {
264 mutex_lock(&mem_id_lock);
265 ret = __mem_id_init_hash_table();
266 mutex_unlock(&mem_id_lock);
267 if (ret < 0) {
268 WARN_ON(1);
269 return ret;
270 }
271 }
272
273 xdp_alloc = kzalloc(sizeof(*xdp_alloc), gfp);
274 if (!xdp_alloc)
275 return -ENOMEM;
276
277 mutex_lock(&mem_id_lock);
278 id = __mem_id_cyclic_get(gfp);
279 if (id < 0) {
280 errno = id;
281 goto err;
282 }
283 xdp_rxq->mem.id = id;
284 xdp_alloc->mem = xdp_rxq->mem;
285 xdp_alloc->allocator = allocator;
286
287 /* Insert allocator into ID lookup table */
288 ptr = rhashtable_insert_slow(mem_id_ht, &id, &xdp_alloc->node);
289 if (IS_ERR(ptr)) {
290 errno = PTR_ERR(ptr);
291 goto err;
292 }
293
294 mutex_unlock(&mem_id_lock);
295
296 return 0;
297err:
298 mutex_unlock(&mem_id_lock);
299 kfree(xdp_alloc);
300 return errno;
301}
302EXPORT_SYMBOL_GPL(xdp_rxq_info_reg_mem_model);

304/* XDP RX runs under NAPI protection, and in different delivery error
305 * scenarios (e.g. queue full), it is possible to return the xdp_frame
306 * while still leveraging this protection. The @napi_direct boolian
307 * is used for those calls sites. Thus, allowing for faster recycling
308 * of xdp_frames/pages in those cases.
309 */
310static void __xdp_return(void *data, struct xdp_mem_info *mem, bool napi_direct,
311 unsigned long handle)
312{
313 struct xdp_mem_allocator *xa;
314 struct page *page;
315
316 switch (mem->type) {
317 case MEM_TYPE_PAGE_POOL:
318 rcu_read_lock();
319 /* mem->id is valid, checked in xdp_rxq_info_reg_mem_model() */
320 xa = rhashtable_lookup(mem_id_ht, &mem->id, mem_id_rht_params);
321 page = virt_to_head_page(data);
322 if (xa)
323 page_pool_put_page(xa->page_pool, page, napi_direct);
324 else
325 put_page(page);
326 rcu_read_unlock();
327 break;
328 case MEM_TYPE_PAGE_SHARED:
329 page_frag_free(data);
330 break;
331 case MEM_TYPE_PAGE_ORDER0:
332 page = virt_to_page(data); /* Assumes order0 page*/
333 put_page(page);
334 break;
335 case MEM_TYPE_ZERO_COPY:
336 /* NB! Only valid from an xdp_buff! */
337 rcu_read_lock();
338 /* mem->id is valid, checked in xdp_rxq_info_reg_mem_model() */
339 xa = rhashtable_lookup(mem_id_ht, &mem->id, mem_id_rht_params);
340 if (!WARN_ON_ONCE(!xa))
341 xa->zc_alloc->free(xa->zc_alloc, handle);
342 rcu_read_unlock();
343 default:
344 /* Not possible, checked in xdp_rxq_info_reg_mem_model() */
345 break;
346 }
347}
