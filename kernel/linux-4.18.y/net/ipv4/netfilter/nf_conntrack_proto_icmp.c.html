32static bool icmp_pkt_to_tuple(const struct sk_buff *skb, unsigned int dataoff,
33 struct net *net, struct nf_conntrack_tuple *tuple)
34{
35 const struct icmphdr *hp;
36 struct icmphdr _hdr;
37
38 hp = skb_header_pointer(skb, dataoff, sizeof(_hdr), &_hdr);
39 if (hp == NULL)
40 return false;
41
42 tuple->dst.u.icmp.type = hp->type;
43 tuple->src.u.icmp.id = hp->un.echo.id;
44 tuple->dst.u.icmp.code = hp->code;
45
46 return true;
47}
48
49/* Add 1; spaces filled with 0. */
50static const u_int8_t invmap[] = {
51 [ICMP_ECHO] = ICMP_ECHOREPLY + 1,
52 [ICMP_ECHOREPLY] = ICMP_ECHO + 1,
53 [ICMP_TIMESTAMP] = ICMP_TIMESTAMPREPLY + 1,
54 [ICMP_TIMESTAMPREPLY] = ICMP_TIMESTAMP + 1,
55 [ICMP_INFO_REQUEST] = ICMP_INFO_REPLY + 1,
56 [ICMP_INFO_REPLY] = ICMP_INFO_REQUEST + 1,
57 [ICMP_ADDRESS] = ICMP_ADDRESSREPLY + 1,
58 [ICMP_ADDRESSREPLY] = ICMP_ADDRESS + 1
59};
60
61static bool icmp_invert_tuple(struct nf_conntrack_tuple *tuple,
62 const struct nf_conntrack_tuple *orig)
63{
64 if (orig->dst.u.icmp.type >= sizeof(invmap) ||
65 !invmap[orig->dst.u.icmp.type])
66 return false;
67
68 tuple->src.u.icmp.id = orig->src.u.icmp.id;
69 tuple->dst.u.icmp.type = invmap[orig->dst.u.icmp.type] - 1;
70 tuple->dst.u.icmp.code = orig->dst.u.icmp.code;
71 return true;
72}

79/* Returns verdict for packet, or -1 for invalid. */
80static int icmp_packet(struct nf_conn *ct,
81 const struct sk_buff *skb,
82 unsigned int dataoff,
83 enum ip_conntrack_info ctinfo,
84 unsigned int *timeout)
85{
86 /* Do not immediately delete the connection after the first
87 successful reply to avoid excessive conntrackd traffic
88 and also to handle correctly ICMP echo reply duplicates. */
89 nf_ct_refresh_acct(ct, ctinfo, skb, *timeout);
90
91 return NF_ACCEPT;
92}

94/* Called when a new connection for this protocol found. */
95static bool icmp_new(struct nf_conn *ct, const struct sk_buff *skb,
96 unsigned int dataoff, unsigned int *timeouts)
97{
98 static const u_int8_t valid_new[] = {
99 [ICMP_ECHO] = 1,
100 [ICMP_TIMESTAMP] = 1,
101 [ICMP_INFO_REQUEST] = 1,
102 [ICMP_ADDRESS] = 1
103 };
104
105 if (ct->tuplehash[0].tuple.dst.u.icmp.type >= sizeof(valid_new) ||
106 !valid_new[ct->tuplehash[0].tuple.dst.u.icmp.type]) {
107 /* Can't create a new ICMP `conn' with this. */
108 pr_debug("icmp: can't create new conn with type %u\n",
109 ct->tuplehash[0].tuple.dst.u.icmp.type);
110 nf_ct_dump_tuple_ip(&ct->tuplehash[0].tuple);
111 return false;
112 }
113 return true;
114}

116/* Returns conntrack if it dealt with ICMP, and filled in skb fields */
117static int
118icmp_error_message(struct net *net, struct nf_conn *tmpl, struct sk_buff *skb,
119 unsigned int hooknum)
120{
121 struct nf_conntrack_tuple innertuple, origtuple;
122 const struct nf_conntrack_l4proto *innerproto;
123 const struct nf_conntrack_tuple_hash *h;
124 const struct nf_conntrack_zone *zone;
125 enum ip_conntrack_info ctinfo;
126 struct nf_conntrack_zone tmp;
127
128 WARN_ON(skb_nfct(skb));
129 zone = nf_ct_zone_tmpl(tmpl, skb, &tmp);
130
131 /* Are they talking about one of our connections? */
132 if (!nf_ct_get_tuplepr(skb,
133 skb_network_offset(skb) + ip_hdrlen(skb)
134 + sizeof(struct icmphdr),
135 PF_INET, net, &origtuple)) {
136 pr_debug("icmp_error_message: failed to get tuple\n");
137 return -NF_ACCEPT;
138 }
139
140 /* rcu_read_lock()ed by nf_hook_thresh */
141 innerproto = __nf_ct_l4proto_find(PF_INET, origtuple.dst.protonum);
142
143 /* Ordinarily, we'd expect the inverted tupleproto, but it's
144 been preserved inside the ICMP. */
145 if (!nf_ct_invert_tuple(&innertuple, &origtuple,
146 &nf_conntrack_l3proto_ipv4, innerproto)) {
147 pr_debug("icmp_error_message: no match\n");
148 return -NF_ACCEPT;
149 }
150
151 ctinfo = IP_CT_RELATED;
152
153 h = nf_conntrack_find_get(net, zone, &innertuple);
154 if (!h) {
155 pr_debug("icmp_error_message: no match\n");
156 return -NF_ACCEPT;
157 }
158
159 if (NF_CT_DIRECTION(h) == IP_CT_DIR_REPLY)
160 ctinfo += IP_CT_IS_REPLY;
161
162 /* Update skb to refer to this connection */
163 nf_ct_set(skb, nf_ct_tuplehash_to_ctrack(h), ctinfo);
164 return NF_ACCEPT;
165}

170 nf_l4proto_log_invalid(skb, net, pf, IPPROTO_ICMP, "%s", msg);

173/* Small and modified version of icmp_rcv */
174static int
175icmp_error(struct net *net, struct nf_conn *tmpl,
176 struct sk_buff *skb, unsigned int dataoff,
177 u8 pf, unsigned int hooknum)
178{
179 const struct icmphdr *icmph;
180 struct icmphdr _ih;
181
182 /* Not enough header? */
183 icmph = skb_header_pointer(skb, ip_hdrlen(skb), sizeof(_ih), &_ih);
184 if (icmph == NULL) {
185 icmp_error_log(skb, net, pf, "short packet");
186 return -NF_ACCEPT;
187 }
188
189 /* See ip_conntrack_proto_tcp.c */
190 if (net->ct.sysctl_checksum && hooknum == NF_INET_PRE_ROUTING &&
191 nf_ip_checksum(skb, hooknum, dataoff, 0)) {
192 icmp_error_log(skb, net, pf, "bad hw icmp checksum");
193 return -NF_ACCEPT;
194 }
195
196 /*
197 * 18 is the highest 'known' ICMP type. Anything else is a mystery
198 *
199 * RFC 1122: 3.2.2 Unknown ICMP messages types MUST be silently
200 * discarded.
201 */
202 if (icmph->type > NR_ICMP_TYPES) {
203 icmp_error_log(skb, net, pf, "invalid icmp type");
204 return -NF_ACCEPT;
205 }
206
207 /* Need to track icmp error message? */
208 if (icmph->type != ICMP_DEST_UNREACH &&
209 icmph->type != ICMP_SOURCE_QUENCH &&
210 icmph->type != ICMP_TIME_EXCEEDED &&
211 icmph->type != ICMP_PARAMETERPROB &&
212 icmph->type != ICMP_REDIRECT)
213 return NF_ACCEPT;
214
215 return icmp_error_message(net, tmpl, skb, hooknum);
216}

223static int icmp_tuple_to_nlattr(struct sk_buff *skb,
224 const struct nf_conntrack_tuple *t)
225{
226 if (nla_put_be16(skb, CTA_PROTO_ICMP_ID, t->src.u.icmp.id) ||
227 nla_put_u8(skb, CTA_PROTO_ICMP_TYPE, t->dst.u.icmp.type) ||
228 nla_put_u8(skb, CTA_PROTO_ICMP_CODE, t->dst.u.icmp.code))
229 goto nla_put_failure;
230 return 0;
231
232nla_put_failure:
233 return -1;
234}
235
236static const struct nla_policy icmp_nla_policy[CTA_PROTO_MAX+1] = {
237 [CTA_PROTO_ICMP_TYPE] = { .type = NLA_U8 },
238 [CTA_PROTO_ICMP_CODE] = { .type = NLA_U8 },
239 [CTA_PROTO_ICMP_ID] = { .type = NLA_U16 },
240};
241
242static int icmp_nlattr_to_tuple(struct nlattr *tb[],
243 struct nf_conntrack_tuple *tuple)
244{
245 if (!tb[CTA_PROTO_ICMP_TYPE] ||
246 !tb[CTA_PROTO_ICMP_CODE] ||
247 !tb[CTA_PROTO_ICMP_ID])
248 return -EINVAL;
249
250 tuple->dst.u.icmp.type = nla_get_u8(tb[CTA_PROTO_ICMP_TYPE]);
251 tuple->dst.u.icmp.code = nla_get_u8(tb[CTA_PROTO_ICMP_CODE]);
252 tuple->src.u.icmp.id = nla_get_be16(tb[CTA_PROTO_ICMP_ID]);
253
254 if (tuple->dst.u.icmp.type >= sizeof(invmap) ||
255 !invmap[tuple->dst.u.icmp.type])
256 return -EINVAL;
257
258 return 0;
259}
260
261static unsigned int icmp_nlattr_tuple_size(void)
262{
263 static unsigned int size __read_mostly;
264
265 if (!size)
266 size = nla_policy_len(icmp_nla_policy, CTA_PROTO_MAX + 1);
267
268 return size;
269}

277static int icmp_timeout_nlattr_to_obj(struct nlattr *tb[],
278 struct net *net, void *data)
279{
280 unsigned int *timeout = data;
281 struct nf_icmp_net *in = icmp_pernet(net);
282
283 if (tb[CTA_TIMEOUT_ICMP_TIMEOUT]) {
284 *timeout =
285 ntohl(nla_get_be32(tb[CTA_TIMEOUT_ICMP_TIMEOUT])) * HZ;
286 } else {
287 /* Set default ICMP timeout. */
288 *timeout = in->timeout;
289 }
290 return 0;
291}
292
293static int
294icmp_timeout_obj_to_nlattr(struct sk_buff *skb, const void *data)
295{
296 const unsigned int *timeout = data;
297
298 if (nla_put_be32(skb, CTA_TIMEOUT_ICMP_TIMEOUT, htonl(*timeout / HZ)))
299 goto nla_put_failure;
300 return 0;
301
302nla_put_failure:
303 return -ENOSPC;
304}

306static const struct nla_policy
307icmp_timeout_nla_policy[CTA_TIMEOUT_ICMP_MAX+1] = {
308 [CTA_TIMEOUT_ICMP_TIMEOUT] = { .type = NLA_U32 },
309};
310#endif /* CONFIG_NF_CT_NETLINK_TIMEOUT */

339static int icmp_init_net(struct net *net, u_int16_t proto)
340{
341 struct nf_icmp_net *in = icmp_pernet(net);
342 struct nf_proto_net *pn = &in->pn;
343
344 in->timeout = nf_ct_icmp_timeout;
345
346 return icmp_kmemdup_sysctl_table(pn, in);
347}
