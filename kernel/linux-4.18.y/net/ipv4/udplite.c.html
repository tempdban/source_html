14#define pr_fmt(fmt) "UDPLite: " fmt

16#include <linux/export.h>
17#include <linux/proc_fs.h>
18#include "udp_impl.h"
19
20struct udp_table udplite_table __read_mostly;
21EXPORT_SYMBOL(udplite_table);
22
23static int udplite_rcv(struct sk_buff *skb)
24{
25 return __udp4_lib_rcv(skb, &udplite_table, IPPROTO_UDPLITE);
26}
27
28static void udplite_err(struct sk_buff *skb, u32 info)
29{
30 __udp4_lib_err(skb, info, &udplite_table);
31}
32
33static const struct net_protocol udplite_protocol = {
34 .handler = udplite_rcv,
35 .err_handler = udplite_err,
36 .no_policy = 1,
37 .netns_ok = 1,
38};
39
40struct proto udplite_prot = {
41 .name = "UDP-Lite",
42 .owner = THIS_MODULE,
43 .close = udp_lib_close,
44 .connect = ip4_datagram_connect,
45 .disconnect = udp_disconnect,
46 .ioctl = udp_ioctl,
47 .init = udplite_sk_init,
48 .destroy = udp_destroy_sock,
49 .setsockopt = udp_setsockopt,
50 .getsockopt = udp_getsockopt,
51 .sendmsg = udp_sendmsg,
52 .recvmsg = udp_recvmsg,
53 .sendpage = udp_sendpage,
54 .hash = udp_lib_hash,
55 .unhash = udp_lib_unhash,
56 .get_port = udp_v4_get_port,
57 .memory_allocated = &udp_memory_allocated,
58 .sysctl_mem = sysctl_udp_mem,
59 .obj_size = sizeof(struct udp_sock),
60 .h.udp_table = &udplite_table,
61#ifdef CONFIG_COMPAT
62 .compat_setsockopt = compat_udp_setsockopt,
63 .compat_getsockopt = compat_udp_getsockopt,
64#endif
65};
66EXPORT_SYMBOL(udplite_prot);
67
68static struct inet_protosw udplite4_protosw = {
69 .type = SOCK_DGRAM,
70 .protocol = IPPROTO_UDPLITE,
71 .prot = &udplite_prot,
72 .ops = &inet_dgram_ops,
73 .flags = INET_PROTOSW_PERMANENT,
74};
75
76#ifdef CONFIG_PROC_FS
77static struct udp_seq_afinfo udplite4_seq_afinfo = {
78 .family = AF_INET,
79 .udp_table = &udplite_table,
80};
81
82static int __net_init udplite4_proc_init_net(struct net *net)
83{
84 if (!proc_create_net_data("udplite", 0444, net->proc_net, &udp_seq_ops,
85 sizeof(struct udp_iter_state), &udplite4_seq_afinfo))
86 return -ENOMEM;
87 return 0;
88}
89
90static void __net_exit udplite4_proc_exit_net(struct net *net)
91{
92 remove_proc_entry("udplite", net->proc_net);
93}
94
95static struct pernet_operations udplite4_net_ops = {
96 .init = udplite4_proc_init_net,
97 .exit = udplite4_proc_exit_net,
98};
99
100static __init int udplite4_proc_init(void)
101{
102 return register_pernet_subsys(&udplite4_net_ops);
103}
104#else
105static inline int udplite4_proc_init(void)
106{
107 return 0;
108}
109#endif
110
111void __init udplite4_register(void)
112{
113 udp_table_init(&udplite_table, "UDP-Lite");
114 if (proto_register(&udplite_prot, 1))
115 goto out_register_err;
116
117 if (inet_add_protocol(&udplite_protocol, IPPROTO_UDPLITE) < 0)
118 goto out_unregister_proto;
119
120 inet_register_protosw(&udplite4_protosw);
121
122 if (udplite4_proc_init())
123 pr_err("%s: Cannot register /proc!\n", __func__);
124 return;
125
126out_unregister_proto:
127 proto_unregister(&udplite_prot);
128out_register_err:
129 pr_crit("%s: Cannot add UDP-Lite protocol\n", __func__);
130}
