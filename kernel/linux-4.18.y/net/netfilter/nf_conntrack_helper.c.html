58static int nf_conntrack_helper_init_sysctl(struct net *net)
59{
60 struct ctl_table *table;
61
62 table = kmemdup(helper_sysctl_table, sizeof(helper_sysctl_table),
63 GFP_KERNEL);
64 if (!table)
65 goto out;
66
67 table[0].data = &net->ct.sysctl_auto_assign_helper;
68
69 /* Don't export sysctls to unprivileged users */
70 if (net->user_ns != &init_user_ns)
71 table[0].procname = NULL;
72
73 net->ct.helper_sysctl_header =
74 register_net_sysctl(net, "net/netfilter", table);
75
76 if (!net->ct.helper_sysctl_header) {
77 pr_err("nf_conntrack_helper: can't register to sysctl.\n");
78 goto out_register;
79 }
80 return 0;
81
82out_register:
83 kfree(table);
84out:
85 return -ENOMEM;
86}

107/* Stupid hash, but collision free for the default registrations of the
108 * helpers currently in the kernel. */
109static unsigned int helper_hash(const struct nf_conntrack_tuple *tuple)
110{
111 return (((tuple->src.l3num << 8) | tuple->dst.protonum) ^
112 (__force __u16)tuple->src.u.all) % nf_ct_helper_hsize;
113}

115static struct nf_conntrack_helper *
116__nf_ct_helper_find(const struct nf_conntrack_tuple *tuple)
117{
118 struct nf_conntrack_helper *helper;
119 struct nf_conntrack_tuple_mask mask = { .src.u.all = htons(0xFFFF) };
120 unsigned int h;
121
122 if (!nf_ct_helper_count)
123 return NULL;
124
125 h = helper_hash(tuple);
126 hlist_for_each_entry_rcu(helper, &nf_ct_helper_hash[h], hnode) {
127 if (nf_ct_tuple_src_mask_cmp(tuple, &helper->tuple, &mask))
128 return helper;
129 }
130 return NULL;
131}

133struct nf_conntrack_helper *
134__nf_conntrack_helper_find(const char *name, u16 l3num, u8 protonum)
135{
136 struct nf_conntrack_helper *h;
137 unsigned int i;
138
139 for (i = 0; i < nf_ct_helper_hsize; i++) {
140 hlist_for_each_entry_rcu(h, &nf_ct_helper_hash[i], hnode) {
141 if (strcmp(h->name, name))
142 continue;
143
144 if (h->tuple.src.l3num != NFPROTO_UNSPEC &&
145 h->tuple.src.l3num != l3num)
146 continue;
147
148 if (h->tuple.dst.protonum == protonum)
149 return h;
150 }
151 }
152 return NULL;
153}
154EXPORT_SYMBOL_GPL(__nf_conntrack_helper_find);

156struct nf_conntrack_helper *
157nf_conntrack_helper_try_module_get(const char *name, u16 l3num, u8 protonum)
158{
159 struct nf_conntrack_helper *h;
160
161 rcu_read_lock();
162
163 h = __nf_conntrack_helper_find(name, l3num, protonum);
164#ifdef CONFIG_MODULES
165 if (h == NULL) {
166 rcu_read_unlock();
167 if (request_module("nfct-helper-%s", name) == 0) {
168 rcu_read_lock();
169 h = __nf_conntrack_helper_find(name, l3num, protonum);
170 } else {
171 return h;
172 }
173 }
174#endif
175 if (h != NULL && !try_module_get(h->me))
176 h = NULL;
177 if (h != NULL && !refcount_inc_not_zero(&h->refcnt)) {
178 module_put(h->me);
179 h = NULL;
180 }
181
182 rcu_read_unlock();
183
184 return h;
185}
186EXPORT_SYMBOL_GPL(nf_conntrack_helper_try_module_get);

210static struct nf_conntrack_helper *
211nf_ct_lookup_helper(struct nf_conn *ct, struct net *net)
212{
213 if (!net->ct.sysctl_auto_assign_helper) {
214 if (net->ct.auto_assign_helper_warned)
215 return NULL;
216 if (!__nf_ct_helper_find(&ct->tuplehash[IP_CT_DIR_REPLY].tuple))
217 return NULL;
218 pr_info("nf_conntrack: default automatic helper assignment "
219 "has been turned off for security reasons and CT-based "
220 " firewall rule not found. Use the iptables CT target "
221 "to attach helpers instead.\n");
222 net->ct.auto_assign_helper_warned = 1;
223 return NULL;
224 }
225
226 return __nf_ct_helper_find(&ct->tuplehash[IP_CT_DIR_REPLY].tuple);
227}

230int __nf_ct_try_assign_helper(struct nf_conn *ct, struct nf_conn *tmpl,
231 gfp_t flags)
232{
233 struct nf_conntrack_helper *helper = NULL;
234 struct nf_conn_help *help;
235 struct net *net = nf_ct_net(ct);
236
237 /* We already got a helper explicitly attached. The function
238 * nf_conntrack_alter_reply - in case NAT is in use - asks for looking
239 * the helper up again. Since now the user is in full control of
240 * making consistent helper configurations, skip this automatic
241 * re-lookup, otherwise we'll lose the helper.
242 */
243 if (test_bit(IPS_HELPER_BIT, &ct->status))
244 return 0;
245
246 if (tmpl != NULL) {
247 help = nfct_help(tmpl);
248 if (help != NULL) {
249 helper = help->helper;
250 set_bit(IPS_HELPER_BIT, &ct->status);
251 }
252 }
253
254 help = nfct_help(ct);
255
256 if (helper == NULL) {
257 helper = nf_ct_lookup_helper(ct, net);
258 if (helper == NULL) {
259 if (help)
260 RCU_INIT_POINTER(help->helper, NULL);
261 return 0;
262 }
263 }
264
265 if (help == NULL) {
266 help = nf_ct_helper_ext_add(ct, helper, flags);
267 if (help == NULL)
268 return -ENOMEM;
269 } else {
270 /* We only allow helper re-assignment of the same sort since
271 * we cannot reallocate the helper extension area.
272 */
273 struct nf_conntrack_helper *tmp = rcu_dereference(help->helper);
274
275 if (tmp && tmp->help != helper->help) {
276 RCU_INIT_POINTER(help->helper, NULL);
277 return 0;
278 }
279 }
280
281 rcu_assign_pointer(help->helper, helper);
282
283 return 0;
284}
285EXPORT_SYMBOL_GPL(__nf_ct_try_assign_helper);

287/* appropriate ct lock protecting must be taken by caller */
288static int unhelp(struct nf_conn *ct, void *me)
289{
290 struct nf_conn_help *help = nfct_help(ct);
291
292 if (help && rcu_dereference_raw(help->helper) == me) {
293 nf_conntrack_event(IPCT_HELPER, ct);
294 RCU_INIT_POINTER(help->helper, NULL);
295 }
296
297 /* We are not intended to delete this conntrack. */
298 return 0;
299}

301void nf_ct_helper_destroy(struct nf_conn *ct)
302{
303 struct nf_conn_help *help = nfct_help(ct);
304 struct nf_conntrack_helper *helper;
305
306 if (help) {
307 rcu_read_lock();
308 helper = rcu_dereference(help->helper);
309 if (helper && helper->destroy)
310 helper->destroy(ct);
311 rcu_read_unlock();
312 }
313}

367__printf(3, 4)
368void nf_ct_helper_log(struct sk_buff *skb, const struct nf_conn *ct,
369 const char *fmt, ...)
370{
371 const struct nf_conn_help *help;
372 const struct nf_conntrack_helper *helper;
373 struct va_format vaf;
374 va_list args;
375
376 va_start(args, fmt);
377
378 vaf.fmt = fmt;
379 vaf.va = &args;
380
381 /* Called from the helper function, this call never fails */
382 help = nfct_help(ct);
383
384 /* rcu_read_lock()ed by nf_hook_thresh */
385 helper = rcu_dereference(help->helper);
386
387 nf_log_packet(nf_ct_net(ct), nf_ct_l3num(ct), 0, skb, NULL, NULL, NULL,
388 "nf_ct_%s: dropping packet: %pV ", helper->name, &vaf);
389
390 va_end(args);
391}

394int nf_conntrack_helper_register(struct nf_conntrack_helper *me)
395{
396 struct nf_conntrack_tuple_mask mask = { .src.u.all = htons(0xFFFF) };
397 unsigned int h = helper_hash(&me->tuple);
398 struct nf_conntrack_helper *cur;
399 int ret = 0, i;
400
401 BUG_ON(me->expect_policy == NULL);
402 BUG_ON(me->expect_class_max >= NF_CT_MAX_EXPECT_CLASSES);
403 BUG_ON(strlen(me->name) > NF_CT_HELPER_NAME_LEN - 1);
404
405 if (me->expect_policy->max_expected > NF_CT_EXPECT_MAX_CNT)
406 return -EINVAL;
407
408 mutex_lock(&nf_ct_helper_mutex);
409 for (i = 0; i < nf_ct_helper_hsize; i++) {
410 hlist_for_each_entry(cur, &nf_ct_helper_hash[i], hnode) {
411 if (!strcmp(cur->name, me->name) &&
412 (cur->tuple.src.l3num == NFPROTO_UNSPEC ||
413 cur->tuple.src.l3num == me->tuple.src.l3num) &&
414 cur->tuple.dst.protonum == me->tuple.dst.protonum) {
415 ret = -EEXIST;
416 goto out;
417 }
418 }
419 }
420
421 /* avoid unpredictable behaviour for auto_assign_helper */
422 if (!(me->flags & NF_CT_HELPER_F_USERSPACE)) {
423 hlist_for_each_entry(cur, &nf_ct_helper_hash[h], hnode) {
424 if (nf_ct_tuple_src_mask_cmp(&cur->tuple, &me->tuple,
425 &mask)) {
426 ret = -EEXIST;
427 goto out;
428 }
429 }
430 }
431 refcount_set(&me->refcnt, 1);
432 hlist_add_head_rcu(&me->hnode, &nf_ct_helper_hash[h]);
433 nf_ct_helper_count++;
434out:
435 mutex_unlock(&nf_ct_helper_mutex);
436 return ret;
437}
438EXPORT_SYMBOL_GPL(nf_conntrack_helper_register);

440static bool expect_iter_me(struct nf_conntrack_expect *exp, void *data)
441{
442 struct nf_conn_help *help = nfct_help(exp->master);
443 const struct nf_conntrack_helper *me = data;
444 const struct nf_conntrack_helper *this;
445
446 if (exp->helper == me)
447 return true;
448
449 this = rcu_dereference_protected(help->helper,
450 lockdep_is_held(&nf_conntrack_expect_lock));
451 return this == me;
452}

454void nf_conntrack_helper_unregister(struct nf_conntrack_helper *me)
455{
456 mutex_lock(&nf_ct_helper_mutex);
457 hlist_del_rcu(&me->hnode);
458 nf_ct_helper_count--;
459 mutex_unlock(&nf_ct_helper_mutex);
460
461 /* Make sure every nothing is still using the helper unless its a
462 * connection in the hash.
463 */
464 synchronize_rcu();
465
466 nf_ct_expect_iterate_destroy(expect_iter_me, NULL);
467 nf_ct_iterate_destroy(unhelp, me);
468
469 /* Maybe someone has gotten the helper already when unhelp above.
470 * So need to wait it.
471 */
472 synchronize_rcu();
473}
474EXPORT_SYMBOL_GPL(nf_conntrack_helper_unregister);

476void nf_ct_helper_init(struct nf_conntrack_helper *helper,
477 u16 l3num, u16 protonum, const char *name,
478 u16 default_port, u16 spec_port, u32 id,
479 const struct nf_conntrack_expect_policy *exp_pol,
480 u32 expect_class_max,
481 int (*help)(struct sk_buff *skb, unsigned int protoff,
482 struct nf_conn *ct,
483 enum ip_conntrack_info ctinfo),
484 int (*from_nlattr)(struct nlattr *attr,
485 struct nf_conn *ct),
486 struct module *module)
487{
488 helper->tuple.src.l3num = l3num;
489 helper->tuple.dst.protonum = protonum;
490 helper->tuple.src.u.all = htons(spec_port);
491 helper->expect_policy = exp_pol;
492 helper->expect_class_max = expect_class_max;
493 helper->help = help;
494 helper->from_nlattr = from_nlattr;
495 helper->me = module;
496
497 if (spec_port == default_port)
498 snprintf(helper->name, sizeof(helper->name), "%s", name);
499 else
500 snprintf(helper->name, sizeof(helper->name), "%s-%u", name, id);
501}

504int nf_conntrack_helpers_register(struct nf_conntrack_helper *helper,
505 unsigned int n)
506{
507 unsigned int i;
508 int err = 0;
509
510 for (i = 0; i < n; i++) {
511 err = nf_conntrack_helper_register(&helper[i]);
512 if (err < 0)
513 goto err;
514 }
515
516 return err;
517err:
518 if (i > 0)
519 nf_conntrack_helpers_unregister(helper, i);
520 return err;
521}

550int nf_conntrack_helper_init(void)
551{
552 int ret;
553 nf_ct_helper_hsize = 1; /* gets rounded up to use one page */
554 nf_ct_helper_hash =
555 nf_ct_alloc_hashtable(&nf_ct_helper_hsize, 0);
556 if (!nf_ct_helper_hash)
557 return -ENOMEM;
558
559 ret = nf_ct_extend_register(&helper_extend);
560 if (ret < 0) {
561 pr_err("nf_ct_helper: Unable to register helper extension.\n");
562 goto out_extend;
563 }
564
565 return 0;
566out_extend:
567 nf_ct_free_hashtable(nf_ct_helper_hash, nf_ct_helper_hsize);
568 return ret;
569}

571void nf_conntrack_helper_fini(void)
572{
573 nf_ct_extend_unregister(&helper_extend);
574 nf_ct_free_hashtable(nf_ct_helper_hash, nf_ct_helper_hsize);
575}
