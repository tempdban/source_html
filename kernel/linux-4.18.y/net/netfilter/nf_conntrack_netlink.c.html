60static int ctnetlink_dump_tuples_proto(struct sk_buff *skb,
61 const struct nf_conntrack_tuple *tuple,
62 const struct nf_conntrack_l4proto *l4proto)
63{
64 int ret = 0;
65 struct nlattr *nest_parms;
66
67 nest_parms = nla_nest_start(skb, CTA_TUPLE_PROTO | NLA_F_NESTED);
68 if (!nest_parms)
69 goto nla_put_failure;
70 if (nla_put_u8(skb, CTA_PROTO_NUM, tuple->dst.protonum))
71 goto nla_put_failure;
72
73 if (likely(l4proto->tuple_to_nlattr))
74 ret = l4proto->tuple_to_nlattr(skb, tuple);
75
76 nla_nest_end(skb, nest_parms);
77
78 return ret;
79
80nla_put_failure:
81 return -1;
82}

106static int ctnetlink_dump_tuples(struct sk_buff *skb,
107 const struct nf_conntrack_tuple *tuple)
108{
109 const struct nf_conntrack_l3proto *l3proto;
110 const struct nf_conntrack_l4proto *l4proto;
111 int ret;
112
113 rcu_read_lock();
114 l3proto = __nf_ct_l3proto_find(tuple->src.l3num);
115 ret = ctnetlink_dump_tuples_ip(skb, tuple, l3proto);
116
117 if (ret >= 0) {
118 l4proto = __nf_ct_l4proto_find(tuple->src.l3num,
119 tuple->dst.protonum);
120 ret = ctnetlink_dump_tuples_proto(skb, tuple, l4proto);
121 }
122 rcu_read_unlock();
123 return ret;
124}

185static int ctnetlink_dump_helpinfo(struct sk_buff *skb,
186 const struct nf_conn *ct)
187{
188 struct nlattr *nest_helper;
189 const struct nf_conn_help *help = nfct_help(ct);
190 struct nf_conntrack_helper *helper;
191
192 if (!help)
193 return 0;
194
195 helper = rcu_dereference(help->helper);
196 if (!helper)
197 goto out;
198
199 nest_helper = nla_nest_start(skb, CTA_HELP | NLA_F_NESTED);
200 if (!nest_helper)
201 goto nla_put_failure;
202 if (nla_put_string(skb, CTA_HELP_NAME, helper->name))
203 goto nla_put_failure;
204
205 if (helper->to_nlattr)
206 helper->to_nlattr(skb, ct);
207
208 nla_nest_end(skb, nest_helper);
209out:
210 return 0;
211
212nla_put_failure:
213 return -1;
214}

216static int
217dump_counters(struct sk_buff *skb, struct nf_conn_acct *acct,
218 enum ip_conntrack_dir dir, int type)
219{
220 enum ctattr_type attr = dir ? CTA_COUNTERS_REPLY: CTA_COUNTERS_ORIG;
221 struct nf_conn_counter *counter = acct->counter;
222 struct nlattr *nest_count;
223 u64 pkts, bytes;
224
225 if (type == IPCTNL_MSG_CT_GET_CTRZERO) {
226 pkts = atomic64_xchg(&counter[dir].packets, 0);
227 bytes = atomic64_xchg(&counter[dir].bytes, 0);
228 } else {
229 pkts = atomic64_read(&counter[dir].packets);
230 bytes = atomic64_read(&counter[dir].bytes);
231 }
232
233 nest_count = nla_nest_start(skb, attr | NLA_F_NESTED);
234 if (!nest_count)
235 goto nla_put_failure;
236
237 if (nla_put_be64(skb, CTA_COUNTERS_PACKETS, cpu_to_be64(pkts),
238 CTA_COUNTERS_PAD) ||
239 nla_put_be64(skb, CTA_COUNTERS_BYTES, cpu_to_be64(bytes),
240 CTA_COUNTERS_PAD))
241 goto nla_put_failure;
242
243 nla_nest_end(skb, nest_count);
244
245 return 0;
246
247nla_put_failure:
248 return -1;
249}

287 nla_nest_end(skb, nest_count);

419static int ctnetlink_dump_ct_seq_adj(struct sk_buff *skb, struct nf_conn *ct)
420{
421 struct nf_conn_seqadj *seqadj = nfct_seqadj(ct);
422 struct nf_ct_seqadj *seq;
423
424 if (!(ct->status & IPS_SEQ_ADJUST) || !seqadj)
425 return 0;
426
427 spin_lock_bh(&ct->lock);
428 seq = &seqadj->seq[IP_CT_DIR_ORIGINAL];
429 if (dump_ct_seq_adj(skb, seq, CTA_SEQ_ADJ_ORIG) == -1)
430 goto err;
431
432 seq = &seqadj->seq[IP_CT_DIR_REPLY];
433 if (dump_ct_seq_adj(skb, seq, CTA_SEQ_ADJ_REPLY) == -1)
434 goto err;
435
436 spin_unlock_bh(&ct->lock);
437 return 0;
438err:
439 spin_unlock_bh(&ct->lock);
440 return -1;
441}

488static int
489ctnetlink_fill_info(struct sk_buff *skb, u32 portid, u32 seq, u32 type,
490 struct nf_conn *ct)
491{
492 const struct nf_conntrack_zone *zone;
493 struct nlmsghdr *nlh;
494 struct nfgenmsg *nfmsg;
495 struct nlattr *nest_parms;
496 unsigned int flags = portid ? NLM_F_MULTI : 0, event;
497
498 event = nfnl_msg_type(NFNL_SUBSYS_CTNETLINK, IPCTNL_MSG_CT_NEW);
499 nlh = nlmsg_put(skb, portid, seq, event, sizeof(*nfmsg), flags);
500 if (nlh == NULL)
501 goto nlmsg_failure;
502
503 nfmsg = nlmsg_data(nlh);
504 nfmsg->nfgen_family = nf_ct_l3num(ct);
505 nfmsg->version = NFNETLINK_V0;
506 nfmsg->res_id = 0;
507
508 zone = nf_ct_zone(ct);
509
510 nest_parms = nla_nest_start(skb, CTA_TUPLE_ORIG | NLA_F_NESTED);
511 if (!nest_parms)
512 goto nla_put_failure;
513 if (ctnetlink_dump_tuples(skb, nf_ct_tuple(ct, IP_CT_DIR_ORIGINAL)) < 0)
514 goto nla_put_failure;
515 if (ctnetlink_dump_zone_id(skb, CTA_TUPLE_ZONE, zone,
516 NF_CT_ZONE_DIR_ORIG) < 0)
517 goto nla_put_failure;
518 nla_nest_end(skb, nest_parms);
519
520 nest_parms = nla_nest_start(skb, CTA_TUPLE_REPLY | NLA_F_NESTED);
521 if (!nest_parms)
522 goto nla_put_failure;
523 if (ctnetlink_dump_tuples(skb, nf_ct_tuple(ct, IP_CT_DIR_REPLY)) < 0)
524 goto nla_put_failure;
525 if (ctnetlink_dump_zone_id(skb, CTA_TUPLE_ZONE, zone,
526 NF_CT_ZONE_DIR_REPL) < 0)
527 goto nla_put_failure;
528 nla_nest_end(skb, nest_parms);
529
530 if (ctnetlink_dump_zone_id(skb, CTA_ZONE, zone,
531 NF_CT_DEFAULT_ZONE_DIR) < 0)
532 goto nla_put_failure;
533
534 if (ctnetlink_dump_status(skb, ct) < 0 ||
535 ctnetlink_dump_timeout(skb, ct) < 0 ||
536 ctnetlink_dump_acct(skb, ct, type) < 0 ||
537 ctnetlink_dump_timestamp(skb, ct) < 0 ||
538 ctnetlink_dump_protoinfo(skb, ct) < 0 ||
539 ctnetlink_dump_helpinfo(skb, ct) < 0 ||
540 ctnetlink_dump_mark(skb, ct) < 0 ||
541 ctnetlink_dump_secctx(skb, ct) < 0 ||
542 ctnetlink_dump_labels(skb, ct) < 0 ||
543 ctnetlink_dump_id(skb, ct) < 0 ||
544 ctnetlink_dump_use(skb, ct) < 0 ||
545 ctnetlink_dump_master(skb, ct) < 0 ||
546 ctnetlink_dump_ct_seq_adj(skb, ct) < 0 ||
547 ctnetlink_dump_ct_synproxy(skb, ct) < 0)
548 goto nla_put_failure;
549
550 nlmsg_end(skb, nlh);
551 return skb->len;
552
553nlmsg_failure:
554nla_put_failure:
555 nlmsg_cancel(skb, nlh);
556 return -1;
557}

560static size_t ctnetlink_proto_size(const struct nf_conn *ct)
561{
562 const struct nf_conntrack_l3proto *l3proto;
563 const struct nf_conntrack_l4proto *l4proto;
564 size_t len, len4 = 0;
565
566 l3proto = __nf_ct_l3proto_find(nf_ct_l3num(ct));
567 len = l3proto->nla_size;
568 len *= 3u; /* ORIG, REPLY, MASTER */
569
570 l4proto = __nf_ct_l4proto_find(nf_ct_l3num(ct), nf_ct_protonum(ct));
571 len += l4proto->nlattr_size;
572 if (l4proto->nlattr_tuple_size) {
573 len4 = l4proto->nlattr_tuple_size();
574 len4 *= 3u; /* ORIG, REPLY, MASTER */
575 }
576
577 return len + len4;
578}

650static int
651ctnetlink_conntrack_event(unsigned int events, struct nf_ct_event *item)
652{
653 const struct nf_conntrack_zone *zone;
654 struct net *net;
655 struct nlmsghdr *nlh;
656 struct nfgenmsg *nfmsg;
657 struct nlattr *nest_parms;
658 struct nf_conn *ct = item->ct;
659 struct sk_buff *skb;
660 unsigned int type;
661 unsigned int flags = 0, group;
662 int err;
663
664 if (events & (1 << IPCT_DESTROY)) {
665 type = IPCTNL_MSG_CT_DELETE;
666 group = NFNLGRP_CONNTRACK_DESTROY;
667 } else if (events & ((1 << IPCT_NEW) | (1 << IPCT_RELATED))) {
668 type = IPCTNL_MSG_CT_NEW;
669 flags = NLM_F_CREATE|NLM_F_EXCL;
670 group = NFNLGRP_CONNTRACK_NEW;
671 } else if (events) {
672 type = IPCTNL_MSG_CT_NEW;
673 group = NFNLGRP_CONNTRACK_UPDATE;
674 } else
675 return 0;
676
677 net = nf_ct_net(ct);
678 if (!item->report && !nfnetlink_has_listeners(net, group))
679 return 0;
680
681 skb = nlmsg_new(ctnetlink_nlmsg_size(ct), GFP_ATOMIC);
682 if (skb == NULL)
683 goto errout;
684
685 type = nfnl_msg_type(NFNL_SUBSYS_CTNETLINK, type);
686 nlh = nlmsg_put(skb, item->portid, 0, type, sizeof(*nfmsg), flags);
687 if (nlh == NULL)
688 goto nlmsg_failure;
689
690 nfmsg = nlmsg_data(nlh);
691 nfmsg->nfgen_family = nf_ct_l3num(ct);
692 nfmsg->version = NFNETLINK_V0;
693 nfmsg->res_id = 0;
694
695 zone = nf_ct_zone(ct);
696
697 nest_parms = nla_nest_start(skb, CTA_TUPLE_ORIG | NLA_F_NESTED);
698 if (!nest_parms)
699 goto nla_put_failure;
700 if (ctnetlink_dump_tuples(skb, nf_ct_tuple(ct, IP_CT_DIR_ORIGINAL)) < 0)
701 goto nla_put_failure;
702 if (ctnetlink_dump_zone_id(skb, CTA_TUPLE_ZONE, zone,
703 NF_CT_ZONE_DIR_ORIG) < 0)
704 goto nla_put_failure;
705 nla_nest_end(skb, nest_parms);
706
707 nest_parms = nla_nest_start(skb, CTA_TUPLE_REPLY | NLA_F_NESTED);
708 if (!nest_parms)
709 goto nla_put_failure;
710 if (ctnetlink_dump_tuples(skb, nf_ct_tuple(ct, IP_CT_DIR_REPLY)) < 0)
711 goto nla_put_failure;
712 if (ctnetlink_dump_zone_id(skb, CTA_TUPLE_ZONE, zone,
713 NF_CT_ZONE_DIR_REPL) < 0)
714 goto nla_put_failure;
715 nla_nest_end(skb, nest_parms);
716
717 if (ctnetlink_dump_zone_id(skb, CTA_ZONE, zone,
718 NF_CT_DEFAULT_ZONE_DIR) < 0)
719 goto nla_put_failure;
720
721 if (ctnetlink_dump_id(skb, ct) < 0)
722 goto nla_put_failure;
723
724 if (ctnetlink_dump_status(skb, ct) < 0)
725 goto nla_put_failure;
726
727 if (events & (1 << IPCT_DESTROY)) {
728 if (ctnetlink_dump_acct(skb, ct, type) < 0 ||
729 ctnetlink_dump_timestamp(skb, ct) < 0)
730 goto nla_put_failure;
731 } else {
732 if (ctnetlink_dump_timeout(skb, ct) < 0)
733 goto nla_put_failure;
734
735 if (events & (1 << IPCT_PROTOINFO)
736 && ctnetlink_dump_protoinfo(skb, ct) < 0)
737 goto nla_put_failure;
738
739 if ((events & (1 << IPCT_HELPER) || nfct_help(ct))
740 && ctnetlink_dump_helpinfo(skb, ct) < 0)
741 goto nla_put_failure;
742
743#ifdef CONFIG_NF_CONNTRACK_SECMARK
744 if ((events & (1 << IPCT_SECMARK) || ct->secmark)
745 && ctnetlink_dump_secctx(skb, ct) < 0)
746 goto nla_put_failure;
747#endif
748 if (events & (1 << IPCT_LABEL) &&
749 ctnetlink_dump_labels(skb, ct) < 0)
750 goto nla_put_failure;
751
752 if (events & (1 << IPCT_RELATED) &&
753 ctnetlink_dump_master(skb, ct) < 0)
754 goto nla_put_failure;
755
756 if (events & (1 << IPCT_SEQADJ) &&
757 ctnetlink_dump_ct_seq_adj(skb, ct) < 0)
758 goto nla_put_failure;
759
760 if (events & (1 << IPCT_SYNPROXY) &&
761 ctnetlink_dump_ct_synproxy(skb, ct) < 0)
762 goto nla_put_failure;
763 }
764
765#ifdef CONFIG_NF_CONNTRACK_MARK
766 if ((events & (1 << IPCT_MARK) || ct->mark)
767 && ctnetlink_dump_mark(skb, ct) < 0)
768 goto nla_put_failure;
769#endif
770 nlmsg_end(skb, nlh);
771 err = nfnetlink_send(skb, net, item->portid, group, item->report,
772 GFP_ATOMIC);
773 if (err == -ENOBUFS || err == -EAGAIN)
774 return -ENOBUFS;
775
776 return 0;
777
778nla_put_failure:
779 nlmsg_cancel(skb, nlh);
780nlmsg_failure:
781 kfree_skb(skb);
782errout:
783 if (nfnetlink_set_err(net, 0, group, -ENOBUFS) > 0)
784 return -ENOBUFS;
785
786 return 0;
787}
788#endif /* CONFIG_NF_CONNTRACK_EVENTS */

854static int
855ctnetlink_dump_table(struct sk_buff *skb, struct netlink_callback *cb)
856{
857 struct net *net = sock_net(skb->sk);
858 struct nf_conn *ct, *last;
859 struct nf_conntrack_tuple_hash *h;
860 struct hlist_nulls_node *n;
861 struct nfgenmsg *nfmsg = nlmsg_data(cb->nlh);
862 u_int8_t l3proto = nfmsg->nfgen_family;
863 struct nf_conn *nf_ct_evict[8];
864 int res, i;
865 spinlock_t *lockp;
866
867 last = (struct nf_conn *)cb->args[1];
868 i = 0;
869
870 local_bh_disable();
871 for (; cb->args[0] < nf_conntrack_htable_size; cb->args[0]++) {
872restart:
873 while (i) {
874 i--;
875 if (nf_ct_should_gc(nf_ct_evict[i]))
876 nf_ct_kill(nf_ct_evict[i]);
877 nf_ct_put(nf_ct_evict[i]);
878 }
879
880 lockp = &nf_conntrack_locks[cb->args[0] % CONNTRACK_LOCKS];
881 nf_conntrack_lock(lockp);
882 if (cb->args[0] >= nf_conntrack_htable_size) {
883 spin_unlock(lockp);
884 goto out;
885 }
886 hlist_nulls_for_each_entry(h, n, &nf_conntrack_hash[cb->args[0]],
887 hnnode) {
888 if (NF_CT_DIRECTION(h) != IP_CT_DIR_ORIGINAL)
889 continue;
890 ct = nf_ct_tuplehash_to_ctrack(h);
891 if (nf_ct_is_expired(ct)) {
892 if (i < ARRAY_SIZE(nf_ct_evict) &&
893 atomic_inc_not_zero(&ct->ct_general.use))
894 nf_ct_evict[i++] = ct;
895 continue;
896 }
897
898 if (!net_eq(net, nf_ct_net(ct)))
899 continue;
900
901 /* Dump entries of a given L3 protocol number.
902 * If it is not specified, ie. l3proto == 0,
903 * then dump everything. */
904 if (l3proto && nf_ct_l3num(ct) != l3proto)
905 continue;
906 if (cb->args[1]) {
907 if (ct != last)
908 continue;
909 cb->args[1] = 0;
910 }
911 if (!ctnetlink_filter_match(ct, cb->data))
912 continue;
913
914 rcu_read_lock();
915 res =
916 ctnetlink_fill_info(skb, NETLINK_CB(cb->skb).portid,
917 cb->nlh->nlmsg_seq,
918 NFNL_MSG_TYPE(cb->nlh->nlmsg_type),
919 ct);
920 rcu_read_unlock();
921 if (res < 0) {
922 nf_conntrack_get(&ct->ct_general);
923 cb->args[1] = (unsigned long)ct;
924 spin_unlock(lockp);
925 goto out;
926 }
927 }
928 spin_unlock(lockp);
929 if (cb->args[1]) {
930 cb->args[1] = 0;
931 goto restart;
932 }
933 }
934out:
935 local_bh_enable();
936 if (last) {
937 /* nf ct hash resize happened, now clear the leftover. */
938 if ((struct nf_conn *)cb->args[1] == last)
939 cb->args[1] = 0;
940
941 nf_ct_put(last);
942 }
943
944 while (i) {
945 i--;
946 if (nf_ct_should_gc(nf_ct_evict[i]))
947 nf_ct_kill(nf_ct_evict[i]);
948 nf_ct_put(nf_ct_evict[i]);
949 }
950
951 return skb->len;
952}

1183static int ctnetlink_del_conntrack(struct net *net, struct sock *ctnl,

1290 skb2 = nlmsg_new(NLMSG_DEFAULT_SIZE, GFP_KERNEL);
1291 if (skb2 == NULL) {
1292 nf_ct_put(ct);
1293 return -ENOMEM;
1294 }
1295
1296 rcu_read_lock();
1297 err = ctnetlink_fill_info(skb2, NETLINK_CB(skb).portid, nlh->nlmsg_seq,
1298 NFNL_MSG_TYPE(nlh->nlmsg_type), ct);
1299 rcu_read_unlock();
1300 nf_ct_put(ct);
1301 if (err <= 0)
1302 goto free;
1303
1304 err = netlink_unicast(ctnl, skb2, NETLINK_CB(skb).portid, MSG_DONTWAIT);
1305 if (err < 0)
1306 goto out;
1307
1308 return 0;
1309
1310free:
1311 kfree_skb(skb2);
1312out:
1313 /* this avoids a loop in nfnetlink. */
1314 return err == -EAGAIN ? -ENOBUFS : err;
1315}

1324static int
1325ctnetlink_dump_list(struct sk_buff *skb, struct netlink_callback *cb, bool dying)
1326{
1327 struct nf_conn *ct, *last;
1328 struct nf_conntrack_tuple_hash *h;
1329 struct hlist_nulls_node *n;
1330 struct nfgenmsg *nfmsg = nlmsg_data(cb->nlh);
1331 u_int8_t l3proto = nfmsg->nfgen_family;
1332 int res;
1333 int cpu;
1334 struct hlist_nulls_head *list;
1335 struct net *net = sock_net(skb->sk);
1336
1337 if (cb->args[2])
1338 return 0;
1339
1340 last = (struct nf_conn *)cb->args[1];
1341
1342 for (cpu = cb->args[0]; cpu < nr_cpu_ids; cpu++) {
1343 struct ct_pcpu *pcpu;
1344
1345 if (!cpu_possible(cpu))
1346 continue;
1347
1348 pcpu = per_cpu_ptr(net->ct.pcpu_lists, cpu);
1349 spin_lock_bh(&pcpu->lock);
1350 list = dying ? &pcpu->dying : &pcpu->unconfirmed;
1351restart:
1352 hlist_nulls_for_each_entry(h, n, list, hnnode) {
1353 ct = nf_ct_tuplehash_to_ctrack(h);
1354 if (l3proto && nf_ct_l3num(ct) != l3proto)
1355 continue;
1356 if (cb->args[1]) {
1357 if (ct != last)
1358 continue;
1359 cb->args[1] = 0;
1360 }
1361 rcu_read_lock();
1362 res = ctnetlink_fill_info(skb, NETLINK_CB(cb->skb).portid,
1363 cb->nlh->nlmsg_seq,
1364 NFNL_MSG_TYPE(cb->nlh->nlmsg_type),
1365 ct);
1366 rcu_read_unlock();
1367 if (res < 0) {
1368 if (!atomic_inc_not_zero(&ct->ct_general.use))
1369 continue;
1370 cb->args[0] = cpu;
1371 cb->args[1] = (unsigned long)ct;
1372 spin_unlock_bh(&pcpu->lock);
1373 goto out;
1374 }
1375 }
1376 if (cb->args[1]) {
1377 cb->args[1] = 0;
1378 goto restart;
1379 }
1380 spin_unlock_bh(&pcpu->lock);
1381 }
1382 cb->args[2] = 1;
1383out:
1384 if (last)
1385 nf_ct_put(last);
1386
1387 return skb->len;
1388}

1484static void
1485__ctnetlink_change_status(struct nf_conn *ct, unsigned long on,
1486 unsigned long off)
1487{
1488 unsigned int bit;
1489
1490 /* Ignore these unchangable bits */
1491 on &= ~IPS_UNCHANGEABLE_MASK;
1492 off &= ~IPS_UNCHANGEABLE_MASK;
1493
1494 for (bit = 0; bit < __IPS_MAX_BIT; bit++) {
1495 if (on & (1 << bit))
1496 set_bit(bit, &ct->status);
1497 else if (off & (1 << bit))
1498 clear_bit(bit, &ct->status);
1499 }
1500}

1502static int
1503ctnetlink_change_status(struct nf_conn *ct, const struct nlattr * const cda[])
1504{
1505 unsigned long d;
1506 unsigned int status = ntohl(nla_get_be32(cda[CTA_STATUS]));
1507 d = ct->status ^ status;
1508
1509 if (d & (IPS_EXPECTED|IPS_CONFIRMED|IPS_DYING))
1510 /* unchangeable */
1511 return -EBUSY;
1512
1513 if (d & IPS_SEEN_REPLY && !(status & IPS_SEEN_REPLY))
1514 /* SEEN_REPLY bit can only be set */
1515 return -EBUSY;
1516
1517 if (d & IPS_ASSURED && !(status & IPS_ASSURED))
1518 /* ASSURED bit can only be set */
1519 return -EBUSY;
1520
1521 __ctnetlink_change_status(ct, status, 0);
1522 return 0;
1523}

1548static int ctnetlink_change_helper(struct nf_conn *ct,
1549 const struct nlattr * const cda[])
1550{
1551 struct nf_conntrack_helper *helper;
1552 struct nf_conn_help *help = nfct_help(ct);
1553 char *helpname = NULL;
1554 struct nlattr *helpinfo = NULL;
1555 int err;
1556
1557 err = ctnetlink_parse_help(cda[CTA_HELP], &helpname, &helpinfo);
1558 if (err < 0)
1559 return err;
1560
1561 /* don't change helper of sibling connections */
1562 if (ct->master) {
1563 /* If we try to change the helper to the same thing twice,
1564 * treat the second attempt as a no-op instead of returning
1565 * an error.
1566 */
1567 err = -EBUSY;
1568 if (help) {
1569 rcu_read_lock();
1570 helper = rcu_dereference(help->helper);
1571 if (helper && !strcmp(helper->name, helpname))
1572 err = 0;
1573 rcu_read_unlock();
1574 }
1575
1576 return err;
1577 }
1578
1579 if (!strcmp(helpname, "")) {
1580 if (help && help->helper) {
1581 /* we had a helper before ... */
1582 nf_ct_remove_expectations(ct);
1583 RCU_INIT_POINTER(help->helper, NULL);
1584 }
1585
1586 return 0;
1587 }
1588
1589 rcu_read_lock();
1590 helper = __nf_conntrack_helper_find(helpname, nf_ct_l3num(ct),
1591 nf_ct_protonum(ct));
1592 if (helper == NULL) {
1593 rcu_read_unlock();
1594 return -EOPNOTSUPP;
1595 }
1596
1597 if (help) {
1598 if (help->helper == helper) {
1599 /* update private helper data if allowed. */
1600 if (helper->from_nlattr)
1601 helper->from_nlattr(helpinfo, ct);
1602 err = 0;
1603 } else
1604 err = -EBUSY;
1605 } else {
1606 /* we cannot set a helper for an existing conntrack */
1607 err = -EOPNOTSUPP;
1608 }
1609
1610 rcu_read_unlock();
1611 return err;
1612}

1848static struct nf_conn *
1849ctnetlink_create_conntrack(struct net *net,
1850 const struct nf_conntrack_zone *zone,
1851 const struct nlattr * const cda[],
1852 struct nf_conntrack_tuple *otuple,
1853 struct nf_conntrack_tuple *rtuple,
1854 u8 u3)
1855{
1856 struct nf_conn *ct;
1857 int err = -EINVAL;
1858 struct nf_conntrack_helper *helper;
1859 struct nf_conn_tstamp *tstamp;
1860 u64 timeout;
1861
1862 ct = nf_conntrack_alloc(net, zone, otuple, rtuple, GFP_ATOMIC);
1863 if (IS_ERR(ct))
1864 return ERR_PTR(-ENOMEM);
1865
1866 if (!cda[CTA_TIMEOUT])
1867 goto err1;
1868
1869 timeout = (u64)ntohl(nla_get_be32(cda[CTA_TIMEOUT])) * HZ;
1870 if (timeout > INT_MAX)
1871 timeout = INT_MAX;
1872 ct->timeout = (u32)timeout + nfct_time_stamp;
1873
1874 rcu_read_lock();
1875 if (cda[CTA_HELP]) {
1876 char *helpname = NULL;
1877 struct nlattr *helpinfo = NULL;
1878
1879 err = ctnetlink_parse_help(cda[CTA_HELP], &helpname, &helpinfo);
1880 if (err < 0)
1881 goto err2;
1882
1883 helper = __nf_conntrack_helper_find(helpname, nf_ct_l3num(ct),
1884 nf_ct_protonum(ct));
1885 if (helper == NULL) {
1886 rcu_read_unlock();
1887#ifdef CONFIG_MODULES
1888 if (request_module("nfct-helper-%s", helpname) < 0) {
1889 err = -EOPNOTSUPP;
1890 goto err1;
1891 }
1892
1893 rcu_read_lock();
1894 helper = __nf_conntrack_helper_find(helpname,
1895 nf_ct_l3num(ct),
1896 nf_ct_protonum(ct));
1897 if (helper) {
1898 err = -EAGAIN;
1899 goto err2;
1900 }
1901 rcu_read_unlock();
1902#endif
1903 err = -EOPNOTSUPP;
1904 goto err1;
1905 } else {
1906 struct nf_conn_help *help;
1907
1908 help = nf_ct_helper_ext_add(ct, helper, GFP_ATOMIC);
1909 if (help == NULL) {
1910 err = -ENOMEM;
1911 goto err2;
1912 }
1913 /* set private helper data if allowed. */
1914 if (helper->from_nlattr)
1915 helper->from_nlattr(helpinfo, ct);
1916
1917 /* not in hash table yet so not strictly necessary */
1918 RCU_INIT_POINTER(help->helper, helper);
1919 }
1920 } else {
1921 /* try an implicit helper assignation */
1922 err = __nf_ct_try_assign_helper(ct, NULL, GFP_ATOMIC);
1923 if (err < 0)
1924 goto err2;
1925 }
1926
1927 err = ctnetlink_setup_nat(ct, cda);
1928 if (err < 0)
1929 goto err2;
1930
1931 nf_ct_acct_ext_add(ct, GFP_ATOMIC);
1932 nf_ct_tstamp_ext_add(ct, GFP_ATOMIC);
1933 nf_ct_ecache_ext_add(ct, 0, 0, GFP_ATOMIC);
1934 nf_ct_labels_ext_add(ct);
1935 nfct_seqadj_ext_add(ct);
1936 nfct_synproxy_ext_add(ct);
1937
1938 /* we must add conntrack extensions before confirmation. */
1939 ct->status |= IPS_CONFIRMED;
1940
1941 if (cda[CTA_STATUS]) {
1942 err = ctnetlink_change_status(ct, cda);
1943 if (err < 0)
1944 goto err2;
1945 }
1946
1947 if (cda[CTA_SEQ_ADJ_ORIG] || cda[CTA_SEQ_ADJ_REPLY]) {
1948 err = ctnetlink_change_seq_adj(ct, cda);
1949 if (err < 0)
1950 goto err2;
1951 }
1952
1953 memset(&ct->proto, 0, sizeof(ct->proto));
1954 if (cda[CTA_PROTOINFO]) {

2414static int
2415ctnetlink_update_status(struct nf_conn *ct, const struct nlattr * const cda[])
2416{
2417 unsigned int status = ntohl(nla_get_be32(cda[CTA_STATUS]));
2418 unsigned long d = ct->status ^ status;
2419
2420 if (d & IPS_SEEN_REPLY && !(status & IPS_SEEN_REPLY))
2421 /* SEEN_REPLY bit can only be set */
2422 return -EBUSY;
2423
2424 if (d & IPS_ASSURED && !(status & IPS_ASSURED))
2425 /* ASSURED bit can only be set */
2426 return -EBUSY;
2427
2428 /* This check is less strict than ctnetlink_change_status()
2429 * because callers often flip IPS_EXPECTED bits when sending
2430 * an NFQA_CT attribute to the kernel. So ignore the
2431 * unchangeable bits but do not error out. Also user programs
2432 * are allowed to clear the bits that they are allowed to change.
2433 */
2434 __ctnetlink_change_status(ct, status, ~status);
2435 return 0;
2436}
