20/*
21 * This code uses an extended set of video mode numbers. These include:
22 * Aliases for standard modes
23 * NORMAL_VGA (-1)
24 * EXTENDED_VGA (-2)
25 * ASK_VGA (-3)
26 * Video modes numbered by menu position -- NOT RECOMMENDED because of lack
27 * of compatibility when extending the table. These are between 0x00 and 0xff.
28 */
29#define VIDEO_FIRST_MENU 0x0000
30
31/* Standard BIOS video modes (BIOS number + 0x0100) */
32#define VIDEO_FIRST_BIOS 0x0100
33
34/* VESA BIOS video modes (VESA number + 0x0200) */
35#define VIDEO_FIRST_VESA 0x0200
36
37/* Video7 special modes (BIOS number + 0x0900) */
38#define VIDEO_FIRST_V7 0x0900

40/* Special video modes */

98/* Accessing VGA indexed registers */
99static inline u8 in_idx(u16 port, u8 index)
100{
101 outb(index, port);
102 return inb(port+1);
103}
104
105static inline void out_idx(u8 v, u16 port, u8 index)
106{
107 outw(index+(v << 8), port);
108}
109
110/* Writes a value to an indexed port and then reads the port again */
111static inline u8 tst_idx(u8 v, u16 port, u8 index)
112{
113 out_idx(port, index, v);
114 return in_idx(port, index);
115}
