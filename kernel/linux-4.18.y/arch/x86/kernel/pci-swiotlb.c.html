18int swiotlb __read_mostly;
19
20/*
21 * pci_swiotlb_detect_override - set swiotlb to 1 if necessary
22 *
23 * This returns non-zero if we are forced to use swiotlb (by the boot
24 * option).
25 */
26int __init pci_swiotlb_detect_override(void)
27{
28 if (swiotlb_force == SWIOTLB_FORCE)
29 swiotlb = 1;
30
31 return swiotlb;
32}
33IOMMU_INIT_FINISH(pci_swiotlb_detect_override,
34 pci_xen_swiotlb_detect,
35 pci_swiotlb_init,
36 pci_swiotlb_late_init);
37
38/*
39 * If 4GB or more detected (and iommu=off not set) or if SME is active
40 * then set swiotlb to 1 and return 1.
41 */
42int __init pci_swiotlb_detect_4gb(void)
43{
44 /* don't initialize swiotlb if iommu=off (no_iommu=1) */
45#ifdef CONFIG_X86_64
46 if (!no_iommu && max_possible_pfn > MAX_DMA32_PFN)
47 swiotlb = 1;
48#endif
49
50 /*
51 * If SME is active then swiotlb will be set to 1 so that bounce
52 * buffers are allocated and used for devices that do not support
53 * the addressing range required for the encryption mask.
54 */
55 if (sme_active())
56 swiotlb = 1;
57
58 return swiotlb;
59}
60IOMMU_INIT(pci_swiotlb_detect_4gb,
61 pci_swiotlb_detect_override,
62 pci_swiotlb_init,
63 pci_swiotlb_late_init);
64
65void __init pci_swiotlb_init(void)
66{
67 if (swiotlb) {
68 swiotlb_init(0);
69 dma_ops = &swiotlb_dma_ops;
70 }
71}
72
73void __init pci_swiotlb_late_init(void)
74{
75 /* An IOMMU turned us off. */
76 if (!swiotlb)
77 swiotlb_exit();
78 else {
79 printk(KERN_INFO "PCI-DMA: "
80 "Using software bounce buffering for IO (SWIOTLB)\n");
81 swiotlb_print_info();
82 }
83}
