104static __rte_always_inline void
105pipeline_fwd_event(struct rte_event *ev, uint8_t sched)
106{
107 ev->event_type = RTE_EVENT_TYPE_CPU;
108 ev->op = RTE_EVENT_OP_FORWARD;
109 ev->sched_type = sched;
110}
111
112static __rte_always_inline void
113pipeline_event_enqueue(const uint8_t dev, const uint8_t port,
114 struct rte_event *ev)
115{
116 while (rte_event_enqueue_burst(dev, port, ev, 1) != 1)
117 rte_pause();
118}
119
120static __rte_always_inline void
121pipeline_event_enqueue_burst(const uint8_t dev, const uint8_t port,
122 struct rte_event *ev, const uint16_t nb_rx)
123{
124 uint16_t enq;
125
126 enq = rte_event_enqueue_burst(dev, port, ev, nb_rx);
127 while (enq < nb_rx) {
128 enq += rte_event_enqueue_burst(dev, port,
129 ev + enq, nb_rx - enq);
130 }
131}
132
133static __rte_always_inline void
134pipeline_tx_pkt(struct rte_mbuf *mbuf)
135{
136 while (rte_eth_tx_burst(mbuf->port, 0, &mbuf, 1) != 1)
137 rte_pause();
138}
