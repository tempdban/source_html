39/**
40 * clocks_calc_mult_shift - calculate mult/shift factors for scaled math of clocks
41 * @mult: pointer to mult variable
42 * @shift: pointer to shift variable
43 * @from: frequency to convert from
44 * @to: frequency to convert to
45 * @maxsec: guaranteed runtime conversion range in seconds
46 *
47 * The function evaluates the shift/mult pair for the scaled math
48 * operations of clocksources and clockevents.
49 *
50 * @to and @from are frequency values in HZ. For clock sources @to is
51 * NSEC_PER_SEC == 1GHz and @from is the counter frequency. For clock
52 * event @to is the counter frequency and @from is NSEC_PER_SEC.
53 *
54 * The @maxsec conversion range argument controls the time frame in
55 * seconds which must be covered by the runtime conversion with the
56 * calculated mult and shift factors. This guarantees that no 64bit
57 * overflow happens when the input value of the conversion is
58 * multiplied with the calculated mult factor. Larger ranges may
59 * reduce the conversion accuracy by chosing smaller mult and shift
60 * factors.
61 */
62void
63clocks_calc_mult_shift(u32 *mult, u32 *shift, u32 from, u32 to, u32 maxsec)
64{
65 u64 tmp;
66 u32 sft, sftacc= 32;
67
68 /*
69 * Calculate the shift factor which is limiting the conversion
70 * range:
71 */
72 tmp = ((u64)maxsec * from) >> 32;
73 while (tmp) {
74 tmp >>=1;
75 sftacc--;
76 }
77
78 /*
79 * Find the conversion shift/mult pair which has the best
80 * accuracy and fits the maxsec conversion range:
81 */
82 for (sft = 32; sft > 0; sft--) {
83 tmp = (u64) to << sft;
84 tmp += from / 2;
85 do_div(tmp, from);
86 if ((tmp >> sftacc) == 0)
87 break;
88 }
89 *mult = tmp;
90 *shift = sft;
91}
92EXPORT_SYMBOL_GPL(clocks_calc_mult_shift);

136 * Interval: 0.5sec Threshold: 0.0625s

138#define WATCHDOG_INTERVAL (HZ >> 1)
139#define WATCHDOG_THRESHOLD (NSEC_PER_SEC >> 4)

141static void clocksource_watchdog_work(struct work_struct *work)
142{
143 /*
144 * We cannot directly run clocksource_watchdog_kthread() here, because
145 * clocksource_select() calls timekeeping_notify() which uses
146 * stop_machine(). One cannot use stop_machine() from a workqueue() due
147 * lock inversions wrt CPU hotplug.
148 *
149 * Also, we only ever run this work once or twice during the lifetime
150 * of the kernel, so there is no point in creating a more permanent
151 * kthread for this.
152 *
153 * If kthread_run fails the next watchdog scan over the
154 * watchdog_list will find the unstable clock again.
155 */
156 kthread_run(clocksource_watchdog_kthread, NULL, "kwatchdog");
157}

159static void __clocksource_unstable(struct clocksource *cs)
160{
161 cs->flags &= ~(CLOCK_SOURCE_VALID_FOR_HRES | CLOCK_SOURCE_WATCHDOG);
162 cs->flags |= CLOCK_SOURCE_UNSTABLE;
163
164 /*
165 * If the clocksource is registered clocksource_watchdog_kthread() will
166 * re-rate and re-select.
167 */
168 if (list_empty(&cs->list)) {
169 cs->rating = 0;
170 return;
171 }
172
173 if (cs->mark_unstable)
174 cs->mark_unstable(cs);
175
176 /* kick clocksource_watchdog_kthread() */
177 if (finished_booting)
178 schedule_work(&watchdog_work);
179}

181/**
182 * clocksource_mark_unstable - mark clocksource unstable via watchdog
183 * @cs: clocksource to be marked unstable
184 *
185 * This function is called by the x86 TSC code to mark clocksources as unstable;
186 * it defers demotion and re-selection to a kthread.
187 */
188void clocksource_mark_unstable(struct clocksource *cs)
189{
190 unsigned long flags;
191
192 spin_lock_irqsave(&watchdog_lock, flags);
193 if (!(cs->flags & CLOCK_SOURCE_UNSTABLE)) {
194 if (!list_empty(&cs->list) && list_empty(&cs->wd_list))
195 list_add(&cs->wd_list, &watchdog_list);
196 __clocksource_unstable(cs);
197 }
198 spin_unlock_irqrestore(&watchdog_lock, flags);
199}

201static void clocksource_watchdog(struct timer_list *unused)
202{
203 struct clocksource *cs;
204 u64 csnow, wdnow, cslast, wdlast, delta;
205 int64_t wd_nsec, cs_nsec;
206 int next_cpu, reset_pending;
207
208 spin_lock(&watchdog_lock);
209 if (!watchdog_running)
210 goto out;
211
212 reset_pending = atomic_read(&watchdog_reset_pending);
213
214 list_for_each_entry(cs, &watchdog_list, wd_list) {
215
216 /* Clocksource already marked unstable? */
217 if (cs->flags & CLOCK_SOURCE_UNSTABLE) {
218 if (finished_booting)
219 schedule_work(&watchdog_work);
220 continue;
221 }
222
223 local_irq_disable();
224 csnow = cs->read(cs);
225 wdnow = watchdog->read(watchdog);
226 local_irq_enable();
227
228 /* Clocksource initialized ? */
229 if (!(cs->flags & CLOCK_SOURCE_WATCHDOG) ||
230 atomic_read(&watchdog_reset_pending)) {
231 cs->flags |= CLOCK_SOURCE_WATCHDOG;
232 cs->wd_last = wdnow;
233 cs->cs_last = csnow;
234 continue;
235 }
236
237 delta = clocksource_delta(wdnow, cs->wd_last, watchdog->mask);
238 wd_nsec = clocksource_cyc2ns(delta, watchdog->mult,
239 watchdog->shift);
240
241 delta = clocksource_delta(csnow, cs->cs_last, cs->mask);
242 cs_nsec = clocksource_cyc2ns(delta, cs->mult, cs->shift);
243 wdlast = cs->wd_last; /* save these in case we print them */
244 cslast = cs->cs_last;
245 cs->cs_last = csnow;
246 cs->wd_last = wdnow;
247
248 if (atomic_read(&watchdog_reset_pending))
249 continue;
250
251 /* Check the deviation from the watchdog clocksource. */
252 if (abs(cs_nsec - wd_nsec) > WATCHDOG_THRESHOLD) {
253 pr_warn("timekeeping watchdog on CPU%d: Marking clocksource '%s' as unstable because the skew is too large:\n",
254 smp_processor_id(), cs->name);
255 pr_warn(" '%s' wd_now: %llx wd_last: %llx mask: %llx\n",
256 watchdog->name, wdnow, wdlast, watchdog->mask);
257 pr_warn(" '%s' cs_now: %llx cs_last: %llx mask: %llx\n",
258 cs->name, csnow, cslast, cs->mask);
259 __clocksource_unstable(cs);
260 continue;
261 }
262
263 if (cs == curr_clocksource && cs->tick_stable)
264 cs->tick_stable(cs);
265
266 if (!(cs->flags & CLOCK_SOURCE_VALID_FOR_HRES) &&
267 (cs->flags & CLOCK_SOURCE_IS_CONTINUOUS) &&
268 (watchdog->flags & CLOCK_SOURCE_IS_CONTINUOUS)) {
269 /* Mark it valid for high-res. */
270 cs->flags |= CLOCK_SOURCE_VALID_FOR_HRES;
271
272 /*
273 * clocksource_done_booting() will sort it if
274 * finished_booting is not set yet.
275 */
276 if (!finished_booting)
277 continue;
278
279 /*
280 * If this is not the current clocksource let
281 * the watchdog thread reselect it. Due to the
282 * change to high res this clocksource might
283 * be preferred now. If it is the current
284 * clocksource let the tick code know about
285 * that change.
286 */
287 if (cs != curr_clocksource) {
288 cs->flags |= CLOCK_SOURCE_RESELECT;
289 schedule_work(&watchdog_work);
290 } else {
291 tick_clock_notify();
292 }
293 }
294 }
295
296 /*
297 * We only clear the watchdog_reset_pending, when we did a
298 * full cycle through all clocksources.
299 */
300 if (reset_pending)
301 atomic_dec(&watchdog_reset_pending);
302
303 /*
304 * Cycle through CPUs to check if the CPUs stay synchronized
305 * to each other.
306 */
307 next_cpu = cpumask_next(raw_smp_processor_id(), cpu_online_mask);
308 if (next_cpu >= nr_cpu_ids)
309 next_cpu = cpumask_first(cpu_online_mask);
310 watchdog_timer.expires += WATCHDOG_INTERVAL;
311 add_timer_on(&watchdog_timer, next_cpu);
312out:
313 spin_unlock(&watchdog_lock);
314}

316static inline void clocksource_start_watchdog(void)
317{
318 if (watchdog_running || !watchdog || list_empty(&watchdog_list))
319 return;
320 timer_setup(&watchdog_timer, clocksource_watchdog, 0);
321 watchdog_timer.expires = jiffies + WATCHDOG_INTERVAL;
322 add_timer_on(&watchdog_timer, cpumask_first(cpu_online_mask));
323 watchdog_running = 1;
324}
325
326static inline void clocksource_stop_watchdog(void)
327{
328 if (!watchdog_running || (watchdog && !list_empty(&watchdog_list)))
329 return;
330 del_timer(&watchdog_timer);
331 watchdog_running = 0;
332}

334static inline void clocksource_reset_watchdog(void)
335{
336 struct clocksource *cs;

338 list_for_each_entry(cs, &watchdog_list, wd_list)
339 cs->flags &= ~CLOCK_SOURCE_WATCHDOG;

342static void clocksource_resume_watchdog(void)

347static void clocksource_enqueue_watchdog(struct clocksource *cs)
348{
349 INIT_LIST_HEAD(&cs->wd_list);
350
351 if (cs->flags & CLOCK_SOURCE_MUST_VERIFY) {
352 /* cs is a clocksource to be watched. */
353 list_add(&cs->wd_list, &watchdog_list);
354 cs->flags &= ~CLOCK_SOURCE_WATCHDOG;
355 } else {
356 /* cs is a watchdog. */
357 if (cs->flags & CLOCK_SOURCE_IS_CONTINUOUS)
358 cs->flags |= CLOCK_SOURCE_VALID_FOR_HRES;
359 }
360}

362static void clocksource_select_watchdog(bool fallback)
363{
364 struct clocksource *cs, *old_wd;
365 unsigned long flags;
366
367 spin_lock_irqsave(&watchdog_lock, flags);
368 /* save current watchdog */
369 old_wd = watchdog;
370 if (fallback)
371 watchdog = NULL;
372
373 list_for_each_entry(cs, &clocksource_list, list) {
374 /* cs is a clocksource to be watched. */
375 if (cs->flags & CLOCK_SOURCE_MUST_VERIFY)
376 continue;
377
378 /* Skip current if we were requested for a fallback. */
379 if (fallback && cs == old_wd)
380 continue;
381
382 /* Pick the best watchdog. */
383 if (!watchdog || cs->rating > watchdog->rating)
384 watchdog = cs;
385 }
386 /* If we failed to find a fallback restore the old one. */
387 if (!watchdog)
388 watchdog = old_wd;
389
390 /* If we changed the watchdog we need to reset cycles. */
391 if (watchdog != old_wd)
392 clocksource_reset_watchdog();
393
394 /* Check if the watchdog timer needs to be started. */
395 clocksource_start_watchdog();
396 spin_unlock_irqrestore(&watchdog_lock, flags);
397}
398
399static void clocksource_dequeue_watchdog(struct clocksource *cs)
400{
401 if (cs != watchdog) {
402 if (cs->flags & CLOCK_SOURCE_MUST_VERIFY) {
403 /* cs is a watched clocksource. */
404 list_del_init(&cs->wd_list);
405 /* Check if the watchdog timer needs to be stopped. */
406 clocksource_stop_watchdog();
407 }
408 }
409}

411static int __clocksource_watchdog_kthread(void)
412{
413 struct clocksource *cs, *tmp;
414 unsigned long flags;
415 int select = 0;
416
417 spin_lock_irqsave(&watchdog_lock, flags);
418 list_for_each_entry_safe(cs, tmp, &watchdog_list, wd_list) {
419 if (cs->flags & CLOCK_SOURCE_UNSTABLE) {
420 list_del_init(&cs->wd_list);
421 __clocksource_change_rating(cs, 0);
422 select = 1;
423 }
424 if (cs->flags & CLOCK_SOURCE_RESELECT) {
425 cs->flags &= ~CLOCK_SOURCE_RESELECT;
426 select = 1;
427 }
428 }
429 /* Check if the watchdog timer needs to be stopped. */
430 clocksource_stop_watchdog();
431 spin_unlock_irqrestore(&watchdog_lock, flags);
432
433 return select;
434}

436static int clocksource_watchdog_kthread(void *data)
437{
438 mutex_lock(&clocksource_mutex);
439 if (__clocksource_watchdog_kthread())
440 clocksource_select();
441 mutex_unlock(&clocksource_mutex);
442 return 0;
443}

508/**
509 * clocksource_max_adjustment- Returns max adjustment amount
510 * @cs: Pointer to clocksource
511 *
512 */
513static u32 clocksource_max_adjustment(struct clocksource *cs)
514{
515 u64 ret;
516 /*
517 * We won't try to correct for more than 11% adjustments (110,000 ppm),
518 */
519 ret = (u64)cs->mult * 11;
520 do_div(ret,100);
521 return (u32)ret;
522}

524/**
525 * clocks_calc_max_nsecs - Returns maximum nanoseconds that can be converted
526 * @mult: cycle to nanosecond multiplier
527 * @shift: cycle to nanosecond divisor (power of two)
528 * @maxadj: maximum adjustment value to mult (~11%)
529 * @mask: bitmask for two's complement subtraction of non 64 bit counters
530 * @max_cyc: maximum cycle value before potential overflow (does not include
531 * any safety margin)
532 *
533 * NOTE: This function includes a safety margin of 50%, in other words, we
534 * return half the number of nanoseconds the hardware counter can technically
535 * cover. This is done so that we can potentially detect problems caused by
536 * delayed timers or bad hardware, which might result in time intervals that
537 * are larger than what the math used can handle without overflows.
538 */
539u64 clocks_calc_max_nsecs(u32 mult, u32 shift, u32 maxadj, u64 mask, u64 *max_cyc)
540{
541 u64 max_nsecs, max_cycles;
542
543 /*
544 * Calculate the maximum number of cycles that we can pass to the
545 * cyc2ns() function without overflowing a 64-bit result.
546 */
547 max_cycles = ULLONG_MAX;
548 do_div(max_cycles, mult+maxadj);
549
550 /*
551 * The actual maximum number of cycles we can defer the clocksource is
552 * determined by the minimum of max_cycles and mask.
553 * Note: Here we subtract the maxadj to make sure we don't sleep for
554 * too long if there's a large negative adjustment.
555 */
556 max_cycles = min(max_cycles, mask);
557 max_nsecs = clocksource_cyc2ns(max_cycles, mult - maxadj, shift);
558
559 /* return the max_cycles value as well if requested */
560 if (max_cyc)
561 *max_cyc = max_cycles;
562
563 /* Return 50% of the actual maximum, so we can detect bad values */
564 max_nsecs >>= 1;
565
566 return max_nsecs;
567}

569/**
570 * clocksource_update_max_deferment - Updates the clocksource max_idle_ns & max_cycles
571 * @cs: Pointer to clocksource to be updated
572 *
573 */
574static inline void clocksource_update_max_deferment(struct clocksource *cs)
575{
576 cs->max_idle_ns = clocks_calc_max_nsecs(cs->mult, cs->shift,
577 cs->maxadj, cs->mask,
578 &cs->max_cycles);
579}

583static struct clocksource *clocksource_find_best(bool oneshot, bool skipcur)
584{
585 struct clocksource *cs;
586
587 if (!finished_booting || list_empty(&clocksource_list))
588 return NULL;
589
590 /*
591 * We pick the clocksource with the highest rating. If oneshot
592 * mode is active, we pick the highres valid clocksource with
593 * the best rating.
594 */
595 list_for_each_entry(cs, &clocksource_list, list) {
596 if (skipcur && cs == curr_clocksource)
597 continue;
598 if (oneshot && !(cs->flags & CLOCK_SOURCE_VALID_FOR_HRES))
599 continue;
600 return cs;
601 }
602 return NULL;
603}
604
605static void __clocksource_select(bool skipcur)
606{
607 bool oneshot = tick_oneshot_mode_active();
608 struct clocksource *best, *cs;
609
610 /* Find the best suitable clocksource */
611 best = clocksource_find_best(oneshot, skipcur);
612 if (!best)
613 return;
614
615 if (!strlen(override_name))
616 goto found;
617
618 /* Check for the override clocksource. */
619 list_for_each_entry(cs, &clocksource_list, list) {
620 if (skipcur && cs == curr_clocksource)
621 continue;
622 if (strcmp(cs->name, override_name) != 0)
623 continue;
624 /*
625 * Check to make sure we don't switch to a non-highres
626 * capable clocksource if the tick code is in oneshot
627 * mode (highres or nohz)
628 */
629 if (!(cs->flags & CLOCK_SOURCE_VALID_FOR_HRES) && oneshot) {
630 /* Override clocksource cannot be used. */
631 if (cs->flags & CLOCK_SOURCE_UNSTABLE) {
632 pr_warn("Override clocksource %s is unstable and not HRT compatible - cannot switch while in HRT/NOHZ mode\n",
633 cs->name);
634 override_name[0] = 0;
635 } else {
636 /*
637 * The override cannot be currently verified.
638 * Deferring to let the watchdog check.
639 */
640 pr_info("Override clocksource %s is not currently HRT compatible - deferring\n",
641 cs->name);
642 }
643 } else
644 /* Override clocksource can be used. */
645 best = cs;
646 break;
647 }
648
649found:
650 if (curr_clocksource != best && !timekeeping_notify(best)) {
651 pr_info("Switched to clocksource %s\n", best->name);
652 curr_clocksource = best;
653 }
654}

680/*
681 * clocksource_done_booting - Called near the end of core bootup
682 *
683 * Hack to avoid lots of clocksource churn at boot time.
684 * We use fs_initcall because we want this to start before
685 * device_initcall but after subsys_initcall.
686 */
687static int __init clocksource_done_booting(void)
688{
689 mutex_lock(&clocksource_mutex);
690 curr_clocksource = clocksource_default_clock();
691 finished_booting = 1;
692 /*
693 * Run the watchdog first to eliminate unstable clock sources
694 */
695 __clocksource_watchdog_kthread();
696 clocksource_select();
697 mutex_unlock(&clocksource_mutex);
698 return 0;
699}
700fs_initcall(clocksource_done_booting);

702/*
703 * Enqueue the clocksource sorted by rating
704 */
705static void clocksource_enqueue(struct clocksource *cs)
706{
707 struct list_head *entry = &clocksource_list;
708 struct clocksource *tmp;
709
710 list_for_each_entry(tmp, &clocksource_list, list) {

714 entry = &tmp->list;
715 }
716 list_add(&cs->list, entry);

719/**
720 * __clocksource_update_freq_scale - Used update clocksource with new freq
721 * @cs: clocksource to be registered
722 * @scale: Scale factor multiplied against freq to get clocksource hz
723 * @freq: clocksource frequency (cycles per second) divided by scale
724 *
725 * This should only be called from the clocksource->enable() method.
726 *
727 * This *SHOULD NOT* be called directly! Please use the
728 * __clocksource_update_freq_hz() or __clocksource_update_freq_khz() helper
729 * functions.
730 */
731void __clocksource_update_freq_scale(struct clocksource *cs, u32 scale, u32 freq)
732{
733 u64 sec;
734
735 /*
736 * Default clocksources are *special* and self-define their mult/shift.
737 * But, you're not special, so you should specify a freq value.
738 */
739 if (freq) {
740 /*
741 * Calc the maximum number of seconds which we can run before
742 * wrapping around. For clocksources which have a mask > 32-bit
743 * we need to limit the max sleep time to have a good
744 * conversion precision. 10 minutes is still a reasonable
745 * amount. That results in a shift value of 24 for a
746 * clocksource with mask >= 40-bit and f >= 4GHz. That maps to
747 * ~ 0.06ppm granularity for NTP.
748 */
749 sec = cs->mask;
750 do_div(sec, freq);
751 do_div(sec, scale);
752 if (!sec)
753 sec = 1;
754 else if (sec > 600 && cs->mask > UINT_MAX)
755 sec = 600;
756
757 clocks_calc_mult_shift(&cs->mult, &cs->shift, freq,
758 NSEC_PER_SEC / scale, sec * scale);
759 }
760 /*
761 * Ensure clocksources that have large 'mult' values don't overflow
762 * when adjusted.
763 */
764 cs->maxadj = clocksource_max_adjustment(cs);
765 while (freq && ((cs->mult + cs->maxadj < cs->mult)
766 || (cs->mult - cs->maxadj > cs->mult))) {
767 cs->mult >>= 1;
768 cs->shift--;
769 cs->maxadj = clocksource_max_adjustment(cs);
770 }
771
772 /*
773 * Only warn for *special* clocksources that self-define
774 * their mult/shift values and don't specify a freq.
775 */
776 WARN_ONCE(cs->mult + cs->maxadj < cs->mult,
777 "timekeeping: Clocksource %s might overflow on 11%% adjustment\n",
778 cs->name);
779
780 clocksource_update_max_deferment(cs);
781
782 pr_info("%s: mask: 0x%llx max_cycles: 0x%llx, max_idle_ns: %lld ns\n",
783 cs->name, cs->mask, cs->max_cycles, cs->max_idle_ns);
784}
785EXPORT_SYMBOL_GPL(__clocksource_update_freq_scale);

787/**
788 * __clocksource_register_scale - Used to install new clocksources
789 * @cs: clocksource to be registered
790 * @scale: Scale factor multiplied against freq to get clocksource hz
791 * @freq: clocksource frequency (cycles per second) divided by scale
792 *
793 * Returns -EBUSY if registration fails, zero otherwise.
794 *
795 * This *SHOULD NOT* be called directly! Please use the
796 * clocksource_register_hz() or clocksource_register_khz helper functions.
797 */
798int __clocksource_register_scale(struct clocksource *cs, u32 scale, u32 freq)
799{
800 unsigned long flags;
801
802 /* Initialize mult/shift and max_idle_ns */
803 __clocksource_update_freq_scale(cs, scale, freq);
804
805 /* Add clocksource to the clocksource list */
806 mutex_lock(&clocksource_mutex);
807
808 clocksource_watchdog_lock(&flags);
809 clocksource_enqueue(cs);
810 clocksource_enqueue_watchdog(cs);
811 clocksource_watchdog_unlock(&flags);
812
813 clocksource_select();
814 clocksource_select_watchdog(false);
815 mutex_unlock(&clocksource_mutex);
816 return 0;
817}
818EXPORT_SYMBOL_GPL(__clocksource_register_scale);

847/*
848 * Unbind clocksource @cs. Called with clocksource_mutex held
849 */
850static int clocksource_unbind(struct clocksource *cs)
851{
852 unsigned long flags;
853
854 if (clocksource_is_watchdog(cs)) {
855 /* Select and try to install a replacement watchdog. */
856 clocksource_select_watchdog(true);
857 if (clocksource_is_watchdog(cs))
858 return -EBUSY;
859 }
860
861 if (cs == curr_clocksource) {
862 /* Select and try to install a replacement clock source */
863 clocksource_select_fallback();
864 if (curr_clocksource == cs)
865 return -EBUSY;
866 }
867
868 clocksource_watchdog_lock(&flags);
869 clocksource_dequeue_watchdog(cs);
870 list_del_init(&cs->list);
871 clocksource_watchdog_unlock(&flags);
872
873 return 0;
874}

876/**
877 * clocksource_unregister - remove a registered clocksource
878 * @cs: clocksource to be unregistered
879 */
880int clocksource_unregister(struct clocksource *cs)
881{
882 int ret = 0;
883
884 mutex_lock(&clocksource_mutex);
885 if (!list_empty(&cs->list))
886 ret = clocksource_unbind(cs);
887 mutex_unlock(&clocksource_mutex);
888 return ret;
889}
