116/* Flag whether block is compressed or uncompressed, bit is set if block is
117 * uncompressed */
118#define SQUASHFS_COMPRESSED_BIT (1 << 15)
119
120#define SQUASHFS_COMPRESSED_SIZE(B) (((B) & ~SQUASHFS_COMPRESSED_BIT) ? \
121 (B) & ~SQUASHFS_COMPRESSED_BIT : SQUASHFS_COMPRESSED_BIT)
122
123#define SQUASHFS_COMPRESSED(B) (!((B) & SQUASHFS_COMPRESSED_BIT))
124
125#define SQUASHFS_COMPRESSED_BIT_BLOCK (1 << 24)
126
127#define SQUASHFS_COMPRESSED_SIZE_BLOCK(B) ((B) & \
128 ~SQUASHFS_COMPRESSED_BIT_BLOCK)
129
130#define SQUASHFS_COMPRESSED_BLOCK(B) (!((B) & SQUASHFS_COMPRESSED_BIT_BLOCK))
131
132static inline int squashfs_block_size(__le32 raw)
133{
134 u32 size = le32_to_cpu(raw);
135 return (size >> 25) ? -EIO : size;
136}

138/*
139 * Inode number ops. Inodes consist of a compressed block number, and an
140 * uncompressed offset within that block
141 */
142#define SQUASHFS_INODE_BLK(A) ((unsigned int) ((A) >> 16))
143
144#define SQUASHFS_INODE_OFFSET(A) ((unsigned int) ((A) & 0xffff))
145
146#define SQUASHFS_MKINODE(A, B) ((long long)(((long long) (A)\
147 << 16) + (B)))

149/* fragment and fragment table defines */
150#define SQUASHFS_FRAGMENT_BYTES(A) \

153#define SQUASHFS_FRAGMENT_INDEX(A) (SQUASHFS_FRAGMENT_BYTES(A) / \
154 SQUASHFS_METADATA_SIZE)

156#define SQUASHFS_FRAGMENT_INDEX_OFFSET(A) (SQUASHFS_FRAGMENT_BYTES(A) % \
157 SQUASHFS_METADATA_SIZE)

159#define SQUASHFS_FRAGMENT_INDEXES(A) ((SQUASHFS_FRAGMENT_BYTES(A) + \
160 SQUASHFS_METADATA_SIZE - 1) / \
161 SQUASHFS_METADATA_SIZE)
162
163#define SQUASHFS_FRAGMENT_INDEX_BYTES(A) (SQUASHFS_FRAGMENT_INDEXES(A) *\
164 sizeof(u64))

173 SQUASHFS_METADATA_SIZE)

175#define SQUASHFS_LOOKUP_BLOCKS(A) ((SQUASHFS_LOOKUP_BYTES(A) + \

180 sizeof(u64))

182/* uid/gid lookup table defines */
183#define SQUASHFS_ID_BYTES(A) ((A) * sizeof(unsigned int))
184
185#define SQUASHFS_ID_BLOCK(A) (SQUASHFS_ID_BYTES(A) / \

187
188#define SQUASHFS_ID_BLOCK_OFFSET(A) (SQUASHFS_ID_BYTES(A) % \
189 SQUASHFS_METADATA_SIZE)

191#define SQUASHFS_ID_BLOCKS(A) ((SQUASHFS_ID_BYTES(A) + \
192 SQUASHFS_METADATA_SIZE - 1) / \
193 SQUASHFS_METADATA_SIZE)

195#define SQUASHFS_ID_BLOCK_BYTES(A) (SQUASHFS_ID_BLOCKS(A) *\

204 SQUASHFS_METADATA_SIZE)
205
206#define SQUASHFS_XATTR_BLOCKS(A) ((SQUASHFS_XATTR_BYTES(A) + \
