17bool f2fs_may_inline_data(struct inode *inode)
18{
19 if (f2fs_is_atomic_file(inode))
20 return false;
21
22 if (!S_ISREG(inode->i_mode) && !S_ISLNK(inode->i_mode))
23 return false;
24
25 if (i_size_read(inode) > MAX_INLINE_DATA(inode))
26 return false;
27
28 if (f2fs_post_read_required(inode))
29 return false;
30
31 return true;
32}
33
34bool f2fs_may_inline_dentry(struct inode *inode)
35{
36 if (!test_opt(F2FS_I_SB(inode), INLINE_DENTRY))
37 return false;
38
39 if (!S_ISDIR(inode->i_mode))
40 return false;
41
42 return true;
43}

45void f2fs_do_read_inline_data(struct page *page, struct page *ipage)
46{
47 struct inode *inode = page->mapping->host;
48 void *src_addr, *dst_addr;
49
50 if (PageUptodate(page))
51 return;
52
53 f2fs_bug_on(F2FS_P_SB(page), page->index);
54
55 zero_user_segment(page, MAX_INLINE_DATA(inode), PAGE_SIZE);
56
57 /* Copy the whole inline data block */
58 src_addr = inline_data_addr(inode, ipage);
59 dst_addr = kmap_atomic(page);
60 memcpy(dst_addr, src_addr, MAX_INLINE_DATA(inode));
61 flush_dcache_page(page);
62 kunmap_atomic(dst_addr);
63 if (!PageUptodate(page))
64 SetPageUptodate(page);
65}

67void f2fs_truncate_inline_inode(struct inode *inode,
68 struct page *ipage, u64 from)
69{
70 void *addr;
71
72 if (from >= MAX_INLINE_DATA(inode))
73 return;
74
75 addr = inline_data_addr(inode, ipage);
76
77 f2fs_wait_on_page_writeback(ipage, NODE, true);
78 memset(addr + from, 0, MAX_INLINE_DATA(inode) - from);
79 set_page_dirty(ipage);
80
81 if (from == 0)
82 clear_inode_flag(inode, FI_DATA_EXIST);
83}

85int f2fs_read_inline_data(struct inode *inode, struct page *page)
86{
87 struct page *ipage;
88
89 ipage = f2fs_get_node_page(F2FS_I_SB(inode), inode->i_ino);
90 if (IS_ERR(ipage)) {
91 unlock_page(page);
92 return PTR_ERR(ipage);
93 }
94
95 if (!f2fs_has_inline_data(inode)) {
96 f2fs_put_page(ipage, 1);
97 return -EAGAIN;
98 }
99
100 if (page->index)
101 zero_user_segment(page, 0, PAGE_SIZE);
102 else
103 f2fs_do_read_inline_data(page, ipage);
104
105 if (!PageUptodate(page))
106 SetPageUptodate(page);
107 f2fs_put_page(ipage, 1);
108 unlock_page(page);
109 return 0;
110}

112int f2fs_convert_inline_page(struct dnode_of_data *dn, struct page *page)
113{
114 struct f2fs_io_info fio = {
115 .sbi = F2FS_I_SB(dn->inode),
116 .ino = dn->inode->i_ino,
117 .type = DATA,
118 .op = REQ_OP_WRITE,
119 .op_flags = REQ_SYNC | REQ_PRIO,
120 .page = page,
121 .encrypted_page = NULL,
122 .io_type = FS_DATA_IO,
123 };
124 int dirty, err;
125
126 if (!f2fs_exist_data(dn->inode))
127 goto clear_out;
128
129 err = f2fs_reserve_block(dn, 0);
130 if (err)
131 return err;
132
133 if (unlikely(dn->data_blkaddr != NEW_ADDR)) {
134 f2fs_put_dnode(dn);
135 set_sbi_flag(fio.sbi, SBI_NEED_FSCK);
136 f2fs_msg(fio.sbi->sb, KERN_WARNING,
137 "%s: corrupted inline inode ino=%lx, i_addr[0]:0x%x, "
138 "run fsck to fix.",
139 __func__, dn->inode->i_ino, dn->data_blkaddr);
140 return -EINVAL;
141 }
142
143 f2fs_bug_on(F2FS_P_SB(page), PageWriteback(page));
144
145 f2fs_do_read_inline_data(page, dn->inode_page);
146 set_page_dirty(page);
147
148 /* clear dirty state */
149 dirty = clear_page_dirty_for_io(page);
150
151 /* write data page to try to make data consistent */
152 set_page_writeback(page);
153 ClearPageError(page);
154 fio.old_blkaddr = dn->data_blkaddr;
155 set_inode_flag(dn->inode, FI_HOT_DATA);
156 f2fs_outplace_write_data(dn, &fio);
157 f2fs_wait_on_page_writeback(page, DATA, true);
158 if (dirty) {
159 inode_dec_dirty_pages(dn->inode);
160 f2fs_remove_dirty_inode(dn->inode);
161 }
162
163 /* this converted inline_data should be recovered. */
164 set_inode_flag(dn->inode, FI_APPEND_WRITE);
165
166 /* clear inline data and flag after data writeback */
167 f2fs_truncate_inline_inode(dn->inode, dn->inode_page, 0);
168 clear_inline_node(dn->inode_page);
169clear_out:
170 stat_dec_inline_inode(dn->inode);
171 clear_inode_flag(dn->inode, FI_INLINE_DATA);
172 f2fs_put_dnode(dn);
173 return 0;
174}

176int f2fs_convert_inline_inode(struct inode *inode)
177{
178 struct f2fs_sb_info *sbi = F2FS_I_SB(inode);
179 struct dnode_of_data dn;
180 struct page *ipage, *page;
181 int err = 0;
182
183 if (!f2fs_has_inline_data(inode))
184 return 0;
185
186 page = f2fs_grab_cache_page(inode->i_mapping, 0, false);
187 if (!page)
188 return -ENOMEM;
189
190 f2fs_lock_op(sbi);
191
192 ipage = f2fs_get_node_page(sbi, inode->i_ino);
193 if (IS_ERR(ipage)) {
194 err = PTR_ERR(ipage);
195 goto out;
196 }
197
198 set_new_dnode(&dn, inode, ipage, ipage, 0);
199
200 if (f2fs_has_inline_data(inode))
201 err = f2fs_convert_inline_page(&dn, page);
202
203 f2fs_put_dnode(&dn);
204out:
205 f2fs_unlock_op(sbi);
206
207 f2fs_put_page(page, 1);
208
209 f2fs_balance_fs(sbi, dn.node_changed);
210
211 return err;
212}

214int f2fs_write_inline_data(struct inode *inode, struct page *page)
215{
216 void *src_addr, *dst_addr;
217 struct dnode_of_data dn;
218 int err;
219
220 set_new_dnode(&dn, inode, NULL, NULL, 0);
221 err = f2fs_get_dnode_of_data(&dn, 0, LOOKUP_NODE);
222 if (err)
223 return err;
224
225 if (!f2fs_has_inline_data(inode)) {
226 f2fs_put_dnode(&dn);
227 return -EAGAIN;
228 }
229
230 f2fs_bug_on(F2FS_I_SB(inode), page->index);
231
232 f2fs_wait_on_page_writeback(dn.inode_page, NODE, true);
233 src_addr = kmap_atomic(page);
234 dst_addr = inline_data_addr(inode, dn.inode_page);
235 memcpy(dst_addr, src_addr, MAX_INLINE_DATA(inode));
236 kunmap_atomic(src_addr);
237 set_page_dirty(dn.inode_page);
238
239 f2fs_clear_radix_tree_dirty_tag(page);
240
241 set_inode_flag(inode, FI_APPEND_WRITE);
242 set_inode_flag(inode, FI_DATA_EXIST);
243
244 clear_inline_node(dn.inode_page);
245 f2fs_put_dnode(&dn);
246 return 0;
247}

249bool f2fs_recover_inline_data(struct inode *inode, struct page *npage)
250{
251 struct f2fs_sb_info *sbi = F2FS_I_SB(inode);
252 struct f2fs_inode *ri = NULL;
253 void *src_addr, *dst_addr;
254 struct page *ipage;
255
256 /*
257 * The inline_data recovery policy is as follows.
258 * [prev.] [next] of inline_data flag
259 * o o -> recover inline_data
260 * o x -> remove inline_data, and then recover data blocks
261 * x o -> remove inline_data, and then recover inline_data
262 * x x -> recover data blocks
263 */
264 if (IS_INODE(npage))
265 ri = F2FS_INODE(npage);
266
267 if (f2fs_has_inline_data(inode) &&
268 ri && (ri->i_inline & F2FS_INLINE_DATA)) {
269process_inline:
270 ipage = f2fs_get_node_page(sbi, inode->i_ino);
271 f2fs_bug_on(sbi, IS_ERR(ipage));
272
273 f2fs_wait_on_page_writeback(ipage, NODE, true);
274
275 src_addr = inline_data_addr(inode, npage);
276 dst_addr = inline_data_addr(inode, ipage);
277 memcpy(dst_addr, src_addr, MAX_INLINE_DATA(inode));
278
279 set_inode_flag(inode, FI_INLINE_DATA);
280 set_inode_flag(inode, FI_DATA_EXIST);
281
282 set_page_dirty(ipage);
283 f2fs_put_page(ipage, 1);
284 return true;
285 }
286
287 if (f2fs_has_inline_data(inode)) {
288 ipage = f2fs_get_node_page(sbi, inode->i_ino);
289 f2fs_bug_on(sbi, IS_ERR(ipage));
290 f2fs_truncate_inline_inode(inode, ipage, 0);
291 clear_inode_flag(inode, FI_INLINE_DATA);
292 f2fs_put_page(ipage, 1);
293 } else if (ri && (ri->i_inline & F2FS_INLINE_DATA)) {
294 if (f2fs_truncate_blocks(inode, 0, false))
295 return false;
296 goto process_inline;
297 }
298 return false;
299}

301struct f2fs_dir_entry *f2fs_find_in_inline_dir(struct inode *dir,
302 struct fscrypt_name *fname, struct page **res_page)
303{
304 struct f2fs_sb_info *sbi = F2FS_SB(dir->i_sb);
305 struct qstr name = FSTR_TO_QSTR(&fname->disk_name);
306 struct f2fs_dir_entry *de;
307 struct f2fs_dentry_ptr d;
308 struct page *ipage;
309 void *inline_dentry;
310 f2fs_hash_t namehash;
311
312 ipage = f2fs_get_node_page(sbi, dir->i_ino);
313 if (IS_ERR(ipage)) {
314 *res_page = ipage;
315 return NULL;
316 }
317
318 namehash = f2fs_dentry_hash(&name, fname);
319
320 inline_dentry = inline_data_addr(dir, ipage);
321
322 make_dentry_ptr_inline(dir, &d, inline_dentry);
323 de = f2fs_find_target_dentry(fname, namehash, NULL, &d);
324 unlock_page(ipage);
325 if (de)
326 *res_page = ipage;
327 else
328 f2fs_put_page(ipage, 0);
329
330 return de;
331}

333int f2fs_make_empty_inline_dir(struct inode *inode, struct inode *parent,
334 struct page *ipage)
335{
336 struct f2fs_dentry_ptr d;
337 void *inline_dentry;
338
339 inline_dentry = inline_data_addr(inode, ipage);
340
341 make_dentry_ptr_inline(inode, &d, inline_dentry);
342 f2fs_do_make_empty_dir(inode, parent, &d);
343
344 set_page_dirty(ipage);
345
346 /* update i_size to MAX_INLINE_DATA */
347 if (i_size_read(inode) < MAX_INLINE_DATA(inode))
348 f2fs_i_size_write(inode, MAX_INLINE_DATA(inode));
349 return 0;
350}

352/*
353 * NOTE: ipage is grabbed by caller, but if any error occurs, we should
354 * release ipage in this function.
355 */
356static int f2fs_move_inline_dirents(struct inode *dir, struct page *ipage,
357 void *inline_dentry)
358{
359 struct page *page;
360 struct dnode_of_data dn;
361 struct f2fs_dentry_block *dentry_blk;
362 struct f2fs_dentry_ptr src, dst;
363 int err;
364
365 page = f2fs_grab_cache_page(dir->i_mapping, 0, false);
366 if (!page) {
367 f2fs_put_page(ipage, 1);
368 return -ENOMEM;
369 }
370
371 set_new_dnode(&dn, dir, ipage, NULL, 0);
372 err = f2fs_reserve_block(&dn, 0);
373 if (err)
374 goto out;
375
376 if (unlikely(dn.data_blkaddr != NEW_ADDR)) {
377 f2fs_put_dnode(&dn);
378 set_sbi_flag(F2FS_P_SB(page), SBI_NEED_FSCK);
379 f2fs_msg(F2FS_P_SB(page)->sb, KERN_WARNING,
380 "%s: corrupted inline inode ino=%lx, i_addr[0]:0x%x, "
381 "run fsck to fix.",
382 __func__, dir->i_ino, dn.data_blkaddr);
383 err = -EINVAL;
384 goto out;
385 }
386
387 f2fs_wait_on_page_writeback(page, DATA, true);
388
389 dentry_blk = page_address(page);
390
391 make_dentry_ptr_inline(dir, &src, inline_dentry);
392 make_dentry_ptr_block(dir, &dst, dentry_blk);
393
394 /* copy data from inline dentry block to new dentry block */
395 memcpy(dst.bitmap, src.bitmap, src.nr_bitmap);
396 memset(dst.bitmap + src.nr_bitmap, 0, dst.nr_bitmap - src.nr_bitmap);
397 /*
398 * we do not need to zero out remainder part of dentry and filename
399 * field, since we have used bitmap for marking the usage status of
400 * them, besides, we can also ignore copying/zeroing reserved space
401 * of dentry block, because them haven't been used so far.
402 */
403 memcpy(dst.dentry, src.dentry, SIZE_OF_DIR_ENTRY * src.max);
404 memcpy(dst.filename, src.filename, src.max * F2FS_SLOT_LEN);
405
406 if (!PageUptodate(page))
407 SetPageUptodate(page);
408 set_page_dirty(page);
409
410 /* clear inline dir and flag after data writeback */
411 f2fs_truncate_inline_inode(dir, ipage, 0);
412
413 stat_dec_inline_dir(dir);
414 clear_inode_flag(dir, FI_INLINE_DENTRY);
415
416 f2fs_i_depth_write(dir, 1);
417 if (i_size_read(dir) < PAGE_SIZE)
418 f2fs_i_size_write(dir, PAGE_SIZE);
419out:
420 f2fs_put_page(page, 1);
421 return err;
422}

424static int f2fs_add_inline_entries(struct inode *dir, void *inline_dentry)
425{
426 struct f2fs_dentry_ptr d;
427 unsigned long bit_pos = 0;
428 int err = 0;
429
430 make_dentry_ptr_inline(dir, &d, inline_dentry);
431
432 while (bit_pos < d.max) {
433 struct f2fs_dir_entry *de;
434 struct qstr new_name;
435 nid_t ino;
436 umode_t fake_mode;
437
438 if (!test_bit_le(bit_pos, d.bitmap)) {
439 bit_pos++;
440 continue;
441 }
442
443 de = &d.dentry[bit_pos];
444
445 if (unlikely(!de->name_len)) {
446 bit_pos++;
447 continue;
448 }
449
450 new_name.name = d.filename[bit_pos];
451 new_name.len = le16_to_cpu(de->name_len);
452
453 ino = le32_to_cpu(de->ino);
454 fake_mode = f2fs_get_de_type(de) << S_SHIFT;
455
456 err = f2fs_add_regular_entry(dir, &new_name, NULL, NULL,
457 ino, fake_mode);
458 if (err)
459 goto punch_dentry_pages;
460
461 bit_pos += GET_DENTRY_SLOTS(le16_to_cpu(de->name_len));
462 }
463 return 0;
464punch_dentry_pages:
465 truncate_inode_pages(&dir->i_data, 0);
466 f2fs_truncate_blocks(dir, 0, false);
467 f2fs_remove_dirty_inode(dir);
468 return err;
469}

471static int f2fs_move_rehashed_dirents(struct inode *dir, struct page *ipage,
472 void *inline_dentry)
473{
474 void *backup_dentry;
475 int err;
476
477 backup_dentry = f2fs_kmalloc(F2FS_I_SB(dir),
478 MAX_INLINE_DATA(dir), GFP_F2FS_ZERO);
479 if (!backup_dentry) {
480 f2fs_put_page(ipage, 1);
481 return -ENOMEM;
482 }
483
484 memcpy(backup_dentry, inline_dentry, MAX_INLINE_DATA(dir));
485 f2fs_truncate_inline_inode(dir, ipage, 0);
486
487 unlock_page(ipage);
488
489 err = f2fs_add_inline_entries(dir, backup_dentry);
490 if (err)
491 goto recover;
492
493 lock_page(ipage);
494
495 stat_dec_inline_dir(dir);
496 clear_inode_flag(dir, FI_INLINE_DENTRY);
497 kfree(backup_dentry);
498 return 0;
499recover:
500 lock_page(ipage);
501 f2fs_wait_on_page_writeback(ipage, NODE, true);
502 memcpy(inline_dentry, backup_dentry, MAX_INLINE_DATA(dir));
503 f2fs_i_depth_write(dir, 0);
504 f2fs_i_size_write(dir, MAX_INLINE_DATA(dir));
505 set_page_dirty(ipage);
506 f2fs_put_page(ipage, 1);
507
508 kfree(backup_dentry);
509 return err;
510}

512static int f2fs_convert_inline_dir(struct inode *dir, struct page *ipage,
513 void *inline_dentry)
514{
515 if (!F2FS_I(dir)->i_dir_level)
516 return f2fs_move_inline_dirents(dir, ipage, inline_dentry);
517 else
518 return f2fs_move_rehashed_dirents(dir, ipage, inline_dentry);
519}

521int f2fs_add_inline_entry(struct inode *dir, const struct qstr *new_name,
522 const struct qstr *orig_name,
523 struct inode *inode, nid_t ino, umode_t mode)
524{
525 struct f2fs_sb_info *sbi = F2FS_I_SB(dir);
526 struct page *ipage;
527 unsigned int bit_pos;
528 f2fs_hash_t name_hash;
529 void *inline_dentry = NULL;
530 struct f2fs_dentry_ptr d;
531 int slots = GET_DENTRY_SLOTS(new_name->len);
532 struct page *page = NULL;
533 int err = 0;
534
535 ipage = f2fs_get_node_page(sbi, dir->i_ino);
536 if (IS_ERR(ipage))
537 return PTR_ERR(ipage);
538
539 inline_dentry = inline_data_addr(dir, ipage);
540 make_dentry_ptr_inline(dir, &d, inline_dentry);
541
542 bit_pos = f2fs_room_for_filename(d.bitmap, slots, d.max);
543 if (bit_pos >= d.max) {
544 err = f2fs_convert_inline_dir(dir, ipage, inline_dentry);
545 if (err)
546 return err;
547 err = -EAGAIN;
548 goto out;
549 }
550
551 if (inode) {
552 down_write(&F2FS_I(inode)->i_sem);
553 page = f2fs_init_inode_metadata(inode, dir, new_name,
554 orig_name, ipage);
555 if (IS_ERR(page)) {
556 err = PTR_ERR(page);
557 goto fail;
558 }
559 }
560
561 f2fs_wait_on_page_writeback(ipage, NODE, true);
562
563 name_hash = f2fs_dentry_hash(new_name, NULL);
564 f2fs_update_dentry(ino, mode, &d, new_name, name_hash, bit_pos);
565
566 set_page_dirty(ipage);
567
568 /* we don't need to mark_inode_dirty now */
569 if (inode) {
570 f2fs_i_pino_write(inode, dir->i_ino);
571 f2fs_put_page(page, 1);
572 }
573
574 f2fs_update_parent_metadata(dir, inode, 0);
575fail:
576 if (inode)
577 up_write(&F2FS_I(inode)->i_sem);
578out:
579 f2fs_put_page(ipage, 1);
580 return err;
581}

583void f2fs_delete_inline_entry(struct f2fs_dir_entry *dentry, struct page *page,
584 struct inode *dir, struct inode *inode)
585{
586 struct f2fs_dentry_ptr d;
587 void *inline_dentry;
588 int slots = GET_DENTRY_SLOTS(le16_to_cpu(dentry->name_len));
589 unsigned int bit_pos;
590 int i;
591
592 lock_page(page);
593 f2fs_wait_on_page_writeback(page, NODE, true);
594
595 inline_dentry = inline_data_addr(dir, page);
596 make_dentry_ptr_inline(dir, &d, inline_dentry);
597
598 bit_pos = dentry - d.dentry;
599 for (i = 0; i < slots; i++)
600 __clear_bit_le(bit_pos + i, d.bitmap);
601
602 set_page_dirty(page);
603 f2fs_put_page(page, 1);
604
605 dir->i_ctime = dir->i_mtime = current_time(dir);
606 f2fs_mark_inode_dirty_sync(dir, false);
607
608 if (inode)
609 f2fs_drop_nlink(dir, inode);
610}

612bool f2fs_empty_inline_dir(struct inode *dir)
613{
614 struct f2fs_sb_info *sbi = F2FS_I_SB(dir);
615 struct page *ipage;
616 unsigned int bit_pos = 2;
617 void *inline_dentry;
618 struct f2fs_dentry_ptr d;
619
620 ipage = f2fs_get_node_page(sbi, dir->i_ino);
621 if (IS_ERR(ipage))
622 return false;
623
624 inline_dentry = inline_data_addr(dir, ipage);
625 make_dentry_ptr_inline(dir, &d, inline_dentry);
626
627 bit_pos = find_next_bit_le(d.bitmap, d.max, bit_pos);
628
629 f2fs_put_page(ipage, 1);
630
631 if (bit_pos < d.max)
632 return false;
633
634 return true;
635}

637int f2fs_read_inline_dir(struct file *file, struct dir_context *ctx,
638 struct fscrypt_str *fstr)
639{
640 struct inode *inode = file_inode(file);
641 struct page *ipage = NULL;
642 struct f2fs_dentry_ptr d;
643 void *inline_dentry = NULL;
644 int err;
645
646 make_dentry_ptr_inline(inode, &d, inline_dentry);
647
648 if (ctx->pos == d.max)
649 return 0;
650
651 ipage = f2fs_get_node_page(F2FS_I_SB(inode), inode->i_ino);
652 if (IS_ERR(ipage))
653 return PTR_ERR(ipage);
654
655 inline_dentry = inline_data_addr(inode, ipage);
656
657 make_dentry_ptr_inline(inode, &d, inline_dentry);
658
659 err = f2fs_fill_dentries(ctx, &d, 0, fstr);
660 if (!err)
661 ctx->pos = d.max;
662
663 f2fs_put_page(ipage, 1);
664 return err < 0 ? err : 0;
665}

667int f2fs_inline_data_fiemap(struct inode *inode,
668 struct fiemap_extent_info *fieinfo, __u64 start, __u64 len)
669{
670 __u64 byteaddr, ilen;
671 __u32 flags = FIEMAP_EXTENT_DATA_INLINE | FIEMAP_EXTENT_NOT_ALIGNED |
672 FIEMAP_EXTENT_LAST;
673 struct node_info ni;
674 struct page *ipage;
675 int err = 0;
676
677 ipage = f2fs_get_node_page(F2FS_I_SB(inode), inode->i_ino);
678 if (IS_ERR(ipage))
679 return PTR_ERR(ipage);
680
681 if (!f2fs_has_inline_data(inode)) {
682 err = -EAGAIN;
683 goto out;
684 }
685
686 ilen = min_t(size_t, MAX_INLINE_DATA(inode), i_size_read(inode));
687 if (start >= ilen)
688 goto out;
689 if (start + len < ilen)
690 ilen = start + len;
691 ilen -= start;
692
693 f2fs_get_node_info(F2FS_I_SB(inode), inode->i_ino, &ni);
