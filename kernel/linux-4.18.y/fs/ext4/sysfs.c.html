52static ssize_t session_write_kbytes_show(struct ext4_sb_info *sbi, char *buf)
53{
54 struct super_block *sb = sbi->s_buddy_cache->i_sb;
55
56 if (!sb->s_bdev->bd_part)
57 return snprintf(buf, PAGE_SIZE, "0\n");
58 return snprintf(buf, PAGE_SIZE, "%lu\n",
59 (part_stat_read(sb->s_bdev->bd_part, sectors[1]) -
60 sbi->s_sectors_written_start) >> 1);
61}
62
63static ssize_t lifetime_write_kbytes_show(struct ext4_sb_info *sbi, char *buf)
64{
65 struct super_block *sb = sbi->s_buddy_cache->i_sb;
66
67 if (!sb->s_bdev->bd_part)
68 return snprintf(buf, PAGE_SIZE, "0\n");
69 return snprintf(buf, PAGE_SIZE, "%llu\n",
70 (unsigned long long)(sbi->s_kbytes_written +
71 ((part_stat_read(sb->s_bdev->bd_part, sectors[1]) -
72 EXT4_SB(sb)->s_sectors_written_start) >> 1)));
73}

75static ssize_t inode_readahead_blks_store(struct ext4_sb_info *sbi,
76 const char *buf, size_t count)
77{
78 unsigned long t;
79 int ret;
80
81 ret = kstrtoul(skip_spaces(buf), 0, &t);
82 if (ret)
83 return ret;
84
85 if (t && (!is_power_of_2(t) || t > 0x40000000))
86 return -EINVAL;
87
88 sbi->s_inode_readahead_blks = t;
89 return count;
90}

92static ssize_t reserved_clusters_store(struct ext4_sb_info *sbi,
93 const char *buf, size_t count)
94{
95 unsigned long long val;
96 ext4_fsblk_t clusters = (ext4_blocks_count(sbi->s_es) >>
97 sbi->s_cluster_bits);
98 int ret;
99
100 ret = kstrtoull(skip_spaces(buf), 0, &val);
101 if (ret || val >= clusters)
102 return -EINVAL;
103
104 atomic64_set(&sbi->s_resv_clusters, val);
105 return count;
106}

108static ssize_t trigger_test_error(struct ext4_sb_info *sbi,
109 const char *buf, size_t count)
110{
111 int len = count;
112
113 if (!capable(CAP_SYS_ADMIN))
114 return -EPERM;
115
116 if (len && buf[len-1] == '\n')
117 len--;
118
119 if (len)
120 ext4_error(sbi->s_sb, "%.*s", len, buf);
121 return count;
122}

239static void *calc_ptr(struct ext4_attr *a, struct ext4_sb_info *sbi)
240{
241 switch (a->attr_ptr) {
242 case ptr_explicit:
243 return a->u.explicit_ptr;
244 case ptr_ext4_sb_info_offset:
245 return (void *) (((char *) sbi) + a->u.offset);
246 case ptr_ext4_super_block_offset:
247 return (void *) (((char *) sbi->s_es) + a->u.offset);
248 }
249 return NULL;
250}

252static ssize_t ext4_attr_show(struct kobject *kobj,
253 struct attribute *attr, char *buf)
254{
255 struct ext4_sb_info *sbi = container_of(kobj, struct ext4_sb_info,
256 s_kobj);
257 struct ext4_attr *a = container_of(attr, struct ext4_attr, attr);
258 void *ptr = calc_ptr(a, sbi);
259
260 switch (a->attr_id) {
261 case attr_delayed_allocation_blocks:
262 return snprintf(buf, PAGE_SIZE, "%llu\n",
263 (s64) EXT4_C2B(sbi,
264 percpu_counter_sum(&sbi->s_dirtyclusters_counter)));
265 case attr_session_write_kbytes:
266 return session_write_kbytes_show(sbi, buf);
267 case attr_lifetime_write_kbytes:
268 return lifetime_write_kbytes_show(sbi, buf);
269 case attr_reserved_clusters:
270 return snprintf(buf, PAGE_SIZE, "%llu\n",
271 (unsigned long long)
272 atomic64_read(&sbi->s_resv_clusters));
273 case attr_inode_readahead:
274 case attr_pointer_ui:
275 if (!ptr)
276 return 0;
277 if (a->attr_ptr == ptr_ext4_super_block_offset)
278 return snprintf(buf, PAGE_SIZE, "%u\n",
279 le32_to_cpup(ptr));
280 else
281 return snprintf(buf, PAGE_SIZE, "%u\n",
282 *((unsigned int *) ptr));
283 case attr_pointer_atomic:
284 if (!ptr)
285 return 0;
286 return snprintf(buf, PAGE_SIZE, "%d\n",
287 atomic_read((atomic_t *) ptr));
288 case attr_feature:
289 return snprintf(buf, PAGE_SIZE, "supported\n");
290 }
291
292 return 0;
293}

295static ssize_t ext4_attr_store(struct kobject *kobj,
296 struct attribute *attr,
297 const char *buf, size_t len)
298{
299 struct ext4_sb_info *sbi = container_of(kobj, struct ext4_sb_info,
300 s_kobj);
301 struct ext4_attr *a = container_of(attr, struct ext4_attr, attr);
302 void *ptr = calc_ptr(a, sbi);
303 unsigned long t;
304 int ret;
305
306 switch (a->attr_id) {
307 case attr_reserved_clusters:
308 return reserved_clusters_store(sbi, buf, len);
309 case attr_pointer_ui:
310 if (!ptr)
311 return 0;
312 ret = kstrtoul(skip_spaces(buf), 0, &t);
313 if (ret)
314 return ret;
315 if (a->attr_ptr == ptr_ext4_super_block_offset)
316 *((__le32 *) ptr) = cpu_to_le32(t);
317 else
318 *((unsigned int *) ptr) = t;
319 return len;
320 case attr_inode_readahead:
321 return inode_readahead_blks_store(sbi, buf, len);
322 case attr_trigger_test_error:
323 return trigger_test_error(sbi, buf, len);
324 }
325 return 0;
326}

356int ext4_register_sysfs(struct super_block *sb)
357{
358 struct ext4_sb_info *sbi = EXT4_SB(sb);
359 int err;
360
361 init_completion(&sbi->s_kobj_unregister);
362 err = kobject_init_and_add(&sbi->s_kobj, &ext4_sb_ktype, ext4_root,
363 "%s", sb->s_id);
364 if (err) {
365 kobject_put(&sbi->s_kobj);
366 wait_for_completion(&sbi->s_kobj_unregister);
367 return err;
368 }
369
370 if (ext4_proc_root)
371 sbi->s_proc = proc_mkdir(sb->s_id, ext4_proc_root);
372 if (sbi->s_proc) {
373 proc_create_single_data("options", S_IRUGO, sbi->s_proc,
374 ext4_seq_options_show, sb);
375 proc_create_single_data("es_shrinker_info", S_IRUGO,
376 sbi->s_proc, ext4_seq_es_shrinker_info_show,
377 sb);
378 proc_create_seq_data("mb_groups", S_IRUGO, sbi->s_proc,
379 &ext4_mb_seq_groups_ops, sb);
380 }
381 return 0;
382}

384void ext4_unregister_sysfs(struct super_block *sb)
385{
386 struct ext4_sb_info *sbi = EXT4_SB(sb);
387
388 if (sbi->s_proc)
389 remove_proc_subtree(sb->s_id, ext4_proc_root);
390 kobject_del(&sbi->s_kobj);
391}

393int __init ext4_init_sysfs(void)
394{
395 int ret;
396
397 ext4_root = kobject_create_and_add("ext4", fs_kobj);
398 if (!ext4_root)
399 return -ENOMEM;
400
401 ext4_feat = kzalloc(sizeof(*ext4_feat), GFP_KERNEL);
402 if (!ext4_feat) {
403 ret = -ENOMEM;
404 goto root_err;
405 }
406
407 ret = kobject_init_and_add(ext4_feat, &ext4_feat_ktype,
408 ext4_root, "features");
409 if (ret)
410 goto feat_err;
411
412 ext4_proc_root = proc_mkdir(proc_dirname, NULL);
413 return ret;
414
415feat_err:
416 kobject_put(ext4_feat);
417 ext4_feat = NULL;
418root_err:
419 kobject_put(ext4_root);
420 ext4_root = NULL;
421 return ret;
422}
423
424void ext4_exit_sysfs(void)
425{
426 kobject_put(ext4_feat);
427 ext4_feat = NULL;
428 kobject_put(ext4_root);
429 ext4_root = NULL;
430 remove_proc_entry(proc_dirname, NULL);
431 ext4_proc_root = NULL;
432}
