176/*Macros to define operations on FD*/
177#define DPAA2_SET_FD_ADDR(fd, addr) do { \
178 (fd)->simple.addr_lo = lower_32_bits((size_t)(addr)); \
179 (fd)->simple.addr_hi = upper_32_bits((uint64_t)(addr)); \
180} while (0)
181#define DPAA2_SET_FD_LEN(fd, length) ((fd)->simple.len = length)
182#define DPAA2_SET_FD_BPID(fd, bpid) ((fd)->simple.bpid_offset |= bpid)
183#define DPAA2_SET_ONLY_FD_BPID(fd, bpid) \
184 ((fd)->simple.bpid_offset = bpid)
185#define DPAA2_SET_FD_IVP(fd) (((fd)->simple.bpid_offset |= 0x00004000))
186#define DPAA2_SET_FD_OFFSET(fd, offset) \
187 (((fd)->simple.bpid_offset |= (uint32_t)(offset) << 16))
188#define DPAA2_SET_FD_INTERNAL_JD(fd, len) \
189 ((fd)->simple.frc = (0x80000000 | (len)))
190#define DPAA2_GET_FD_FRC_PARSE_SUM(fd) \
191 ((uint16_t)(((fd)->simple.frc & 0xffff0000) >> 16))
192#define DPAA2_SET_FD_FRC(fd, _frc) ((fd)->simple.frc = _frc)
193#define DPAA2_RESET_FD_CTRL(fd) ((fd)->simple.ctrl = 0)
194
195#define DPAA2_SET_FD_ASAL(fd, asal) ((fd)->simple.ctrl |= (asal << 16))
196#define DPAA2_SET_FD_FLC(fd, addr) do { \
197 (fd)->simple.flc_lo = lower_32_bits((size_t)(addr)); \
198 (fd)->simple.flc_hi = upper_32_bits((uint64_t)(addr)); \
199} while (0)
200#define DPAA2_SET_FLE_INTERNAL_JD(fle, len) ((fle)->frc = (0x80000000 | (len)))
201#define DPAA2_GET_FLE_ADDR(fle) \
202 (size_t)((((uint64_t)((fle)->addr_hi)) << 32) + (fle)->addr_lo)
203#define DPAA2_SET_FLE_ADDR(fle, addr) do { \
204 (fle)->addr_lo = lower_32_bits((size_t)addr); \
205 (fle)->addr_hi = upper_32_bits((uint64_t)addr); \
206} while (0)
207#define DPAA2_GET_FLE_CTXT(fle) \
208 ((((uint64_t)((fle)->reserved[1])) << 32) + (fle)->reserved[0])
209#define DPAA2_FLE_SAVE_CTXT(fle, addr) do { \
210 (fle)->reserved[0] = lower_32_bits((size_t)addr); \
211 (fle)->reserved[1] = upper_32_bits((uint64_t)addr); \
212} while (0)
213#define DPAA2_SET_FLE_OFFSET(fle, offset) \
214 ((fle)->fin_bpid_offset |= (uint32_t)(offset) << 16)
215#define DPAA2_SET_FLE_LEN(fle, len) ((fle)->length = len)
216#define DPAA2_SET_FLE_BPID(fle, bpid) ((fle)->fin_bpid_offset |= (size_t)bpid)
217#define DPAA2_GET_FLE_BPID(fle) ((fle)->fin_bpid_offset & 0x000000ff)
218#define DPAA2_SET_FLE_FIN(fle) ((fle)->fin_bpid_offset |= 1 << 31)
219#define DPAA2_SET_FLE_IVP(fle) (((fle)->fin_bpid_offset |= 0x00004000))
220#define DPAA2_SET_FLE_BMT(fle) (((fle)->fin_bpid_offset |= 0x00008000))
221#define DPAA2_SET_FD_COMPOUND_FMT(fd) \
222 ((fd)->simple.bpid_offset |= (uint32_t)1 << 28)
223#define DPAA2_GET_FD_ADDR(fd) \
224(((((uint64_t)((fd)->simple.addr_hi)) << 32) + (fd)->simple.addr_lo))
225
226#define DPAA2_GET_FD_LEN(fd) ((fd)->simple.len)
227#define DPAA2_GET_FD_BPID(fd) (((fd)->simple.bpid_offset & 0x00003FFF))
228#define DPAA2_GET_FD_IVP(fd) (((fd)->simple.bpid_offset & 0x00004000) >> 14)
229#define DPAA2_GET_FD_OFFSET(fd) (((fd)->simple.bpid_offset & 0x0FFF0000) >> 16)
230#define DPAA2_GET_FD_FRC(fd) ((fd)->simple.frc)
231#define DPAA2_GET_FD_FLC(fd) \
232 (((uint64_t)((fd)->simple.flc_hi) << 32) + (fd)->simple.flc_lo)
233#define DPAA2_GET_FD_ERR(fd) ((fd)->simple.bpid_offset & 0x000000FF)
234#define DPAA2_GET_FLE_OFFSET(fle) (((fle)->fin_bpid_offset & 0x0FFF0000) >> 16)
235#define DPAA2_SET_FLE_SG_EXT(fle) ((fle)->fin_bpid_offset |= (uint64_t)1 << 29)
236#define DPAA2_IS_SET_FLE_SG_EXT(fle) \
237 (((fle)->fin_bpid_offset & ((uint64_t)1 << 29)) ? 1 : 0)

276extern uint8_t dpaa2_virt_mode;
277static void *dpaa2_mem_ptov(phys_addr_t paddr) __attribute__((unused));
278/* todo - this is costly, need to write a fast coversion routine */
279static void *dpaa2_mem_ptov(phys_addr_t paddr)
280{
281 struct dpaa2_memseg *ms;
282
283 if (dpaa2_virt_mode)
284 return (void *)(size_t)paddr;
285
286 /* Check if the address is already part of the memseg list internally
287 * maintained by the dpaa2 driver.
288 */
289 TAILQ_FOREACH(ms, &rte_dpaa2_memsegs, next) {
290 if (paddr >= ms->iova && paddr <
291 ms->iova + ms->len)
292 return RTE_PTR_ADD(ms->vaddr, (uintptr_t)(paddr - ms->iova));
293 }
294
295 /* If not, Fallback to full memseg list searching */
296 return rte_mem_iova2virt(paddr);
297}

299static phys_addr_t dpaa2_mem_vtop(uint64_t vaddr) __attribute__((unused));
300static phys_addr_t dpaa2_mem_vtop(uint64_t vaddr)
301{
302 const struct rte_memseg *memseg;
303
304 if (dpaa2_virt_mode)
305 return vaddr;
306
307 memseg = rte_mem_virt2memseg((void *)(uintptr_t)vaddr, NULL);
308 if (memseg)
309 return memseg->phys_addr + RTE_PTR_DIFF(vaddr, memseg->addr);
310 return (size_t)NULL;
311}
