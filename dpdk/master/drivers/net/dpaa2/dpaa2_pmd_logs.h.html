6#ifndef _DPAA2_PMD_LOGS_H_
7#define _DPAA2_PMD_LOGS_H_
8
9extern int dpaa2_logtype_pmd;
10
11#define DPAA2_PMD_LOG(level, fmt, args...) \
12 rte_log(RTE_LOG_ ## level, dpaa2_logtype_pmd, "dpaa2_net: " \
13 fmt "\n", ##args)
14
15#define DPAA2_PMD_DEBUG(fmt, args...) \
16 rte_log(RTE_LOG_DEBUG, dpaa2_logtype_pmd, "dpaa2_net: %s(): "\
17 fmt "\n", __func__, ##args)
18
19#define PMD_INIT_FUNC_TRACE() DPAA2_PMD_DEBUG(">>")
20
21#define DPAA2_PMD_CRIT(fmt, args...) \
22 DPAA2_PMD_LOG(CRIT, fmt, ## args)
23#define DPAA2_PMD_INFO(fmt, args...) \
24 DPAA2_PMD_LOG(INFO, fmt, ## args)
25#define DPAA2_PMD_ERR(fmt, args...) \
26 DPAA2_PMD_LOG(ERR, fmt, ## args)
27#define DPAA2_PMD_WARN(fmt, args...) \
28 DPAA2_PMD_LOG(WARNING, fmt, ## args)
29
30/* DP Logs, toggled out at compile time if level lower than current level */
31#define DPAA2_PMD_DP_LOG(level, fmt, args...) \
32 RTE_LOG_DP(level, PMD, fmt, ## args)
33
34#define DPAA2_PMD_DP_DEBUG(fmt, args...) \
35 DPAA2_PMD_DP_LOG(DEBUG, fmt, ## args)
36#define DPAA2_PMD_DP_INFO(fmt, args...) \
37 DPAA2_PMD_DP_LOG(INFO, fmt, ## args)
38#define DPAA2_PMD_DP_WARN(fmt, args...) \
39 DPAA2_PMD_DP_LOG(WARNING, fmt, ## args)
40
41#endif /* _DPAA2_PMD_LOGS_H_ */
