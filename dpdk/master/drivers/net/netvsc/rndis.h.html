137/*
138 * RNDIS data message
139 */
140#define RNDIS_PACKET_MSG 0x00000001
141
142struct rndis_packet_msg {
143 uint32_t type;
144 uint32_t len;
145 uint32_t dataoffset;
146 uint32_t datalen;
147 uint32_t oobdataoffset;
148 uint32_t oobdatalen;
149 uint32_t oobdataelements;
150 uint32_t pktinfooffset;
151 uint32_t pktinfolen;
152 uint32_t vchandle;
153 uint32_t reserved;
154};
155
156/*
157 * Minimum value for dataoffset, oobdataoffset, and
158 * pktinfooffset.
159 */
160#define RNDIS_PACKET_MSG_OFFSET_MIN \
161 (sizeof(struct rndis_packet_msg) - \
162 offsetof(struct rndis_packet_msg, dataoffset))
163
164/* Offset from the beginning of rndis_packet_msg. */
165#define RNDIS_PACKET_MSG_OFFSET_ABS(ofs) \
166 ((ofs) + offsetof(struct rndis_packet_msg, dataoffset))
167
168#define RNDIS_PACKET_MSG_OFFSET_ALIGN 4
169#define RNDIS_PACKET_MSG_OFFSET_ALIGNMASK \
170 (RNDIS_PACKET_MSG_OFFSET_ALIGN - 1)
171
172/* Per-packet-info for RNDIS data message */
173struct rndis_pktinfo {
174 uint32_t size;
175 uint32_t type; /* NDIS_PKTINFO_TYPE_ */
176 uint32_t offset;
177 uint8_t data[];
178};
179
180#define RNDIS_PKTINFO_OFFSET \
181 offsetof(struct rndis_pktinfo, data[0])
182#define RNDIS_PKTINFO_SIZE_ALIGN 4
183#define RNDIS_PKTINFO_SIZE_ALIGNMASK (RNDIS_PKTINFO_SIZE_ALIGN - 1)
184
185#define NDIS_PKTINFO_TYPE_CSUM 0
186#define NDIS_PKTINFO_TYPE_IPSEC 1
187#define NDIS_PKTINFO_TYPE_LSO 2
188#define NDIS_PKTINFO_TYPE_CLASSIFY 3
189/* reserved 4 */
190#define NDIS_PKTINFO_TYPE_SGLIST 5
191#define NDIS_PKTINFO_TYPE_VLAN 6
192#define NDIS_PKTINFO_TYPE_ORIG 7
193#define NDIS_PKTINFO_TYPE_PKT_CANCELID 8
194#define NDIS_PKTINFO_TYPE_ORIG_NBLIST 9
195#define NDIS_PKTINFO_TYPE_CACHE_NBLIST 10
196#define NDIS_PKTINFO_TYPE_PKT_PAD 11
197
198/* RNDIS extension */
199
200/* Per-packet hash info */
201#define NDIS_HASH_INFO_SIZE sizeof(uint32_t)
202#define NDIS_PKTINFO_TYPE_HASHINF NDIS_PKTINFO_TYPE_ORIG_NBLIST
203/* NDIS_HASH_ */
204
205/* Per-packet hash value */
206#define NDIS_HASH_VALUE_SIZE sizeof(uint32_t)
207#define NDIS_PKTINFO_TYPE_HASHVAL NDIS_PKTINFO_TYPE_PKT_CANCELID
208
209/* Per-packet-info size */
210#define RNDIS_PKTINFO_SIZE(dlen) offsetof(struct rndis_pktinfo, data[dlen])
