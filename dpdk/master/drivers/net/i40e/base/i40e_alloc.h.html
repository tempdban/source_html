34#ifndef _I40E_ALLOC_H_
35#define _I40E_ALLOC_H_
36
37struct i40e_hw;
38
39/* Memory allocation types */
40enum i40e_memory_type {
41 i40e_mem_arq_buf = 0, /* ARQ indirect command buffer */
42 i40e_mem_asq_buf = 1,
43 i40e_mem_atq_buf = 2, /* ATQ indirect command buffer */
44 i40e_mem_arq_ring = 3, /* ARQ descriptor ring */
45 i40e_mem_atq_ring = 4, /* ATQ descriptor ring */
46 i40e_mem_pd = 5, /* Page Descriptor */
47 i40e_mem_bp = 6, /* Backing Page - 4KB */
48 i40e_mem_bp_jumbo = 7, /* Backing Page - > 4KB */
49 i40e_mem_reserved
50};
51
52/* prototype for functions used for dynamic memory allocation */
53enum i40e_status_code i40e_allocate_dma_mem(struct i40e_hw *hw,
54 struct i40e_dma_mem *mem,
55 enum i40e_memory_type type,
56 u64 size, u32 alignment);
57enum i40e_status_code i40e_free_dma_mem(struct i40e_hw *hw,
58 struct i40e_dma_mem *mem);
59enum i40e_status_code i40e_allocate_virt_mem(struct i40e_hw *hw,
60 struct i40e_virt_mem *mem,
61 u32 size);
62enum i40e_status_code i40e_free_virt_mem(struct i40e_hw *hw,
63 struct i40e_virt_mem *mem);
64
65#endif /* _I40E_ALLOC_H_ */
