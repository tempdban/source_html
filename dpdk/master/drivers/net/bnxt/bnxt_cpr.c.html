14/*
15 * Async event handling
16 */
17void bnxt_handle_async_event(struct bnxt *bp,
18 struct cmpl_base *cmp)
19{
20 struct hwrm_async_event_cmpl *async_cmp =
21 (struct hwrm_async_event_cmpl *)cmp;
22 uint16_t event_id = rte_le_to_cpu_16(async_cmp->event_id);
23
24 /* TODO: HWRM async events are not defined yet */
25 /* Needs to handle: link events, error events, etc. */
26 switch (event_id) {
27 case HWRM_ASYNC_EVENT_CMPL_EVENT_ID_LINK_STATUS_CHANGE:
28 case HWRM_ASYNC_EVENT_CMPL_EVENT_ID_LINK_SPEED_CHANGE:
29 case HWRM_ASYNC_EVENT_CMPL_EVENT_ID_LINK_SPEED_CFG_CHANGE:
30 /* FALLTHROUGH */
31 bnxt_link_update_op(bp->eth_dev, 1);
32 break;
33 case HWRM_ASYNC_EVENT_CMPL_EVENT_ID_PF_DRVR_UNLOAD:
34 PMD_DRV_LOG(INFO, "Async event: PF driver unloaded\n");
35 break;
36 case HWRM_ASYNC_EVENT_CMPL_EVENT_ID_VF_CFG_CHANGE:
37 PMD_DRV_LOG(INFO, "Async event: VF config changed\n");
38 break;
39 case HWRM_ASYNC_EVENT_CMPL_EVENT_ID_PORT_CONN_NOT_ALLOWED:
40 PMD_DRV_LOG(INFO, "Port conn async event\n");
41 break;
42 default:
43 PMD_DRV_LOG(INFO, "handle_async_event id = 0x%x\n", event_id);
44 break;
45 }
46}

48void bnxt_handle_fwd_req(struct bnxt *bp, struct cmpl_base *cmpl)
49{
50 struct hwrm_exec_fwd_resp_input *fwreq;
51 struct hwrm_fwd_req_cmpl *fwd_cmpl = (struct hwrm_fwd_req_cmpl *)cmpl;
52 struct input *fwd_cmd;
53 uint16_t fw_vf_id;
54 uint16_t vf_id;
55 uint16_t req_len;
56 int rc;
57
58 if (bp->pf.active_vfs <= 0) {
59 PMD_DRV_LOG(ERR, "Forwarded VF with no active VFs\n");
60 return;
61 }
62
63 /* Qualify the fwd request */
64 fw_vf_id = rte_le_to_cpu_16(fwd_cmpl->source_id);
65 vf_id = fw_vf_id - bp->pf.first_vf_id;
66
67 req_len = (rte_le_to_cpu_16(fwd_cmpl->req_len_type) &
68 HWRM_FWD_REQ_CMPL_REQ_LEN_MASK) >>
69 HWRM_FWD_REQ_CMPL_REQ_LEN_SFT;
70 if (req_len > sizeof(fwreq->encap_request))
71 req_len = sizeof(fwreq->encap_request);
72
73 /* Locate VF's forwarded command */
74 fwd_cmd = (struct input *)bp->pf.vf_info[vf_id].req_buf;
75
76 if (fw_vf_id < bp->pf.first_vf_id ||
77 fw_vf_id >= (bp->pf.first_vf_id) + bp->pf.active_vfs) {
78 PMD_DRV_LOG(ERR,
79 "FWD req's source_id 0x%x out of range 0x%x - 0x%x (%d %d)\n",
80 fw_vf_id, bp->pf.first_vf_id,
81 (bp->pf.first_vf_id) + bp->pf.active_vfs - 1,
82 bp->pf.first_vf_id, bp->pf.active_vfs);
83 goto reject;
84 }
85
86 if (bnxt_rcv_msg_from_vf(bp, vf_id, fwd_cmd) == true) {
87 /*
88 * In older firmware versions, the MAC had to be all zeros for
89 * the VF to set it's MAC via hwrm_func_vf_cfg. Set to all
90 * zeros if it's being configured and has been ok'd by caller.
91 */
92 if (fwd_cmd->req_type == HWRM_FUNC_VF_CFG) {
93 struct hwrm_func_vf_cfg_input *vfc = (void *)fwd_cmd;
94
95 if (vfc->enables &
96 HWRM_FUNC_VF_CFG_INPUT_ENABLES_DFLT_MAC_ADDR) {
97 bnxt_hwrm_func_vf_mac(bp, vf_id,
98 (const uint8_t *)"\x00\x00\x00\x00\x00");
99 }
100 }
101 if (fwd_cmd->req_type == HWRM_CFA_L2_SET_RX_MASK) {
102 struct hwrm_cfa_l2_set_rx_mask_input *srm =
103 (void *)fwd_cmd;
104
105 srm->vlan_tag_tbl_addr = rte_cpu_to_le_64(0);
106 srm->num_vlan_tags = rte_cpu_to_le_32(0);
107 srm->mask &= ~rte_cpu_to_le_32(
108 HWRM_CFA_L2_SET_RX_MASK_INPUT_MASK_VLANONLY |
109 HWRM_CFA_L2_SET_RX_MASK_INPUT_MASK_VLAN_NONVLAN |
110 HWRM_CFA_L2_SET_RX_MASK_INPUT_MASK_ANYVLAN_NONVLAN);
111 }
112 /* Forward */
113 rc = bnxt_hwrm_exec_fwd_resp(bp, fw_vf_id, fwd_cmd, req_len);
114 if (rc) {
115 PMD_DRV_LOG(ERR,
116 "Failed to send FWD req VF 0x%x, type 0x%x.\n",
117 fw_vf_id - bp->pf.first_vf_id,
118 rte_le_to_cpu_16(fwd_cmd->req_type));
119 }
120 return;
121 }
122
123reject:
124 rc = bnxt_hwrm_reject_fwd_resp(bp, fw_vf_id, fwd_cmd, req_len);
125 if (rc) {
126 PMD_DRV_LOG(ERR,
127 "Failed to send REJECT req VF 0x%x, type 0x%x.\n",
128 fw_vf_id - bp->pf.first_vf_id,
129 rte_le_to_cpu_16(fwd_cmd->req_type));
130 }
131
132 return;
133}

135int bnxt_event_hwrm_resp_handler(struct bnxt *bp, struct cmpl_base *cmp)
136{
137 bool evt = 0;
138
139 if (bp == NULL || cmp == NULL) {
140 PMD_DRV_LOG(ERR, "invalid NULL argument\n");
141 return evt;
142 }
143
144 switch (CMP_TYPE(cmp)) {
145 case CMPL_BASE_TYPE_HWRM_ASYNC_EVENT:
146 /* Handle any async event */
147 bnxt_handle_async_event(bp, cmp);
148 evt = 1;
149 break;
150 case CMPL_BASE_TYPE_HWRM_FWD_RESP:
151 /* Handle HWRM forwarded responses */
152 bnxt_handle_fwd_req(bp, cmp);
153 evt = 1;
154 break;
155 default:
156 /* Ignore any other events */
157 PMD_DRV_LOG(INFO, "Ignoring %02x completion\n", CMP_TYPE(cmp));
158 break;
159 }
160
161 return evt;
162}
