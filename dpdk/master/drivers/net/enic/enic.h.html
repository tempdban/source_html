190/* Compute ethdev's max packet size from MTU */
191static inline uint32_t enic_mtu_to_max_rx_pktlen(uint32_t mtu)
192{
193 /* ethdev max size includes eth and crc whereas NIC MTU does not */
194 return mtu + ETHER_HDR_LEN + ETHER_CRC_LEN;
195}
196
197/* Get the CQ index from a Start of Packet(SOP) RQ index */
198static inline unsigned int enic_sop_rq_idx_to_cq_idx(unsigned int sop_idx)
199{
200 return sop_idx / 2;
201}
202
203/* Get the RTE RQ index from a Start of Packet(SOP) RQ index */
204static inline unsigned int enic_sop_rq_idx_to_rte_idx(unsigned int sop_idx)
205{
206 return sop_idx / 2;
207}
208
209/* Get the Start of Packet(SOP) RQ index from a RTE RQ index */
210static inline unsigned int enic_rte_rq_idx_to_sop_idx(unsigned int rte_idx)
211{
212 return rte_idx * 2;
213}
214
215/* Get the Data RQ index from a RTE RQ index */
216static inline unsigned int enic_rte_rq_idx_to_data_idx(unsigned int rte_idx)
217{
218 return rte_idx * 2 + 1;
219}
220
221static inline unsigned int enic_vnic_rq_count(struct enic *enic)
222{
223 return enic->rq_count * 2;
224}
225
226static inline unsigned int enic_cq_rq(__rte_unused struct enic *enic, unsigned int rq)
227{
228 /* Scatter rx uses two receive queues together with one
229 * completion queue, so the completion queue number is no
230 * longer the same as the rq number.
231 */
232 return rq / 2;
233}
234
235static inline unsigned int enic_cq_wq(struct enic *enic, unsigned int wq)
236{
237 return enic->rq_count + wq;
238}

245static inline uint32_t
246enic_ring_add(uint32_t n_descriptors, uint32_t i0, uint32_t i1)
247{
248 uint32_t d = i0 + i1;
249 d -= (d >= n_descriptors) ? n_descriptors : 0;
250 return d;
251}
252
253static inline uint32_t
254enic_ring_sub(uint32_t n_descriptors, uint32_t i0, uint32_t i1)
255{
256 int32_t d = i1 - i0;
257 return (uint32_t)((d < 0) ? ((int32_t)n_descriptors + d) : d);
258}
259
260static inline uint32_t
261enic_ring_incr(uint32_t n_descriptors, uint32_t idx)
262{
263 idx++;
264 if (unlikely(idx == n_descriptors))
265 idx = 0;
266 return idx;
267}
