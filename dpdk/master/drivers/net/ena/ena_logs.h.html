34#ifndef _ENA_LOGS_H_
35#define _ENA_LOGS_H_
36
37extern int ena_logtype_init;
38#define PMD_INIT_LOG(level, fmt, args...) \
39 rte_log(RTE_LOG_ ## level, ena_logtype_init, \
40 "%s(): " fmt "\n", __func__, ## args)
41
42#ifdef RTE_LIBRTE_ENA_DEBUG_RX
43#define PMD_RX_LOG(level, fmt, args...) \
44 RTE_LOG(level, PMD, "%s(): " fmt "\n", __func__, ## args)
45#else
46#define PMD_RX_LOG(level, fmt, args...) do { } while (0)
47#endif
48
49#ifdef RTE_LIBRTE_ENA_DEBUG_TX
50#define PMD_TX_LOG(level, fmt, args...) \
51 RTE_LOG(level, PMD, "%s(): " fmt "\n", __func__, ## args)
52#else
53#define PMD_TX_LOG(level, fmt, args...) do { } while (0)
54#endif
55
56#ifdef RTE_LIBRTE_ENA_DEBUG_TX_FREE
57#define PMD_TX_FREE_LOG(level, fmt, args...) \
58 RTE_LOG(level, PMD, "%s(): " fmt "\n", __func__, ## args)
59#else
60#define PMD_TX_FREE_LOG(level, fmt, args...) do { } while (0)
61#endif
62
63extern int ena_logtype_driver;
64#define PMD_DRV_LOG(level, fmt, args...) \
65 rte_log(RTE_LOG_ ## level, ena_logtype_driver, \
66 "%s(): " fmt "\n", __func__, ## args)
67
68#endif /* _ENA_LOGS_H_ */
