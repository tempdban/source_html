18/** private data structure for a QAT compression device.
19 * This QAT device is a device offering only a compression service,
20 * there can be one of these on each qat_pci_device (VF).
21 */
22struct qat_comp_dev_private {
23 struct qat_pci_device *qat_dev;
24 /**< The qat pci device hosting the service */
25 struct rte_compressdev *compressdev;
26 /**< The pointer to this compression device structure */
27 const struct rte_compressdev_capabilities *qat_dev_capabilities;
28 /* QAT device compression capabilities */
29 const struct rte_memzone *interm_buff_mz;
30 /**< The device's memory for intermediate buffers */
31 struct rte_mempool *xformpool;
32 /**< The device's pool for qat_comp_xforms */
33};
