19static inline void
20ip_frag_memmove(char *dst, char *src, int len)
21{
22 int i;
23
24 /* go backwards to make sure we don't overwrite anything important */
25 for (i = len - 1; i >= 0; i--)
26 dst[i] = src[i];
27}

29/*
30 * Reassemble fragments into one packet.
31 */
32struct rte_mbuf *
33ipv6_frag_reassemble(struct ip_frag_pkt *fp)
34{
35 struct ipv6_hdr *ip_hdr;
36 struct ipv6_extension_fragment *frag_hdr;
37 struct rte_mbuf *m, *prev;
38 uint32_t i, n, ofs, first_len;
39 uint32_t last_len, move_len, payload_len;
40 uint32_t curr_idx = 0;
41
42 first_len = fp->frags[IP_FIRST_FRAG_IDX].len;
43 n = fp->last_idx - 1;
44
45 /*start from the last fragment. */
46 m = fp->frags[IP_LAST_FRAG_IDX].mb;
47 ofs = fp->frags[IP_LAST_FRAG_IDX].ofs;
48 last_len = fp->frags[IP_LAST_FRAG_IDX].len;
49 curr_idx = IP_LAST_FRAG_IDX;
50
51 payload_len = ofs + last_len;
52
53 while (ofs != first_len) {
54
55 prev = m;
56
57 for (i = n; i != IP_FIRST_FRAG_IDX && ofs != first_len; i--) {
58
59 /* previous fragment found. */
60 if (fp->frags[i].ofs + fp->frags[i].len == ofs) {
61
62 /* adjust start of the last fragment data. */
63 rte_pktmbuf_adj(m, (uint16_t)(m->l2_len + m->l3_len));
64 rte_pktmbuf_chain(fp->frags[i].mb, m);
65
66 /* this mbuf should not be accessed directly */
67 fp->frags[curr_idx].mb = NULL;
68 curr_idx = i;
69
70 /* update our last fragment and offset. */
71 m = fp->frags[i].mb;
72 ofs = fp->frags[i].ofs;
73 }
74 }
75
76 /* error - hole in the packet. */
77 if (m == prev) {
78 return NULL;
79 }
80 }
81
82 /* chain with the first fragment. */
83 rte_pktmbuf_adj(m, (uint16_t)(m->l2_len + m->l3_len));
84 rte_pktmbuf_chain(fp->frags[IP_FIRST_FRAG_IDX].mb, m);
85 fp->frags[curr_idx].mb = NULL;
86 m = fp->frags[IP_FIRST_FRAG_IDX].mb;
87 fp->frags[IP_FIRST_FRAG_IDX].mb = NULL;
88
89 /* update mbuf fields for reassembled packet. */
90 m->ol_flags |= PKT_TX_IP_CKSUM;
91
92 /* update ipv6 header for the reassembled datagram */
93 ip_hdr = rte_pktmbuf_mtod_offset(m, struct ipv6_hdr *, m->l2_len);
94
95 ip_hdr->payload_len = rte_cpu_to_be_16(payload_len);
96
97 /*
98 * remove fragmentation header. note that per RFC2460, we need to update
99 * the last non-fragmentable header with the "next header" field to contain
100 * type of the first fragmentable header, but we currently don't support
101 * other headers, so we assume there are no other headers and thus update
102 * the main IPv6 header instead.
103 */
104 move_len = m->l2_len + m->l3_len - sizeof(*frag_hdr);
105 frag_hdr = (struct ipv6_extension_fragment *) (ip_hdr + 1);
106 ip_hdr->proto = frag_hdr->next_header;
107
108 ip_frag_memmove(rte_pktmbuf_mtod_offset(m, char *, sizeof(*frag_hdr)),
109 rte_pktmbuf_mtod(m, char*), move_len);
110
111 rte_pktmbuf_adj(m, sizeof(*frag_hdr));
112
113 return m;
114}

116/*
117 * Process new mbuf with fragment of IPV6 datagram.
118 * Incoming mbuf should have its l2_len/l3_len fields setup correctly.
119 * @param tbl
120 * Table where to lookup/add the fragmented packet.
121 * @param mb
122 * Incoming mbuf with IPV6 fragment.
123 * @param tms
124 * Fragment arrival timestamp.
125 * @param ip_hdr
126 * Pointer to the IPV6 header.
127 * @param frag_hdr
128 * Pointer to the IPV6 fragment extension header.
129 * @return
130 * Pointer to mbuf for reassembled packet, or NULL if:
131 * - an error occurred.
132 * - not all fragments of the packet are collected yet.
133 */
134#define MORE_FRAGS(x) (((x) & 0x100) >> 8)
135#define FRAG_OFFSET(x) (rte_cpu_to_be_16(x) >> 3)
136struct rte_mbuf *
137rte_ipv6_frag_reassemble_packet(struct rte_ip_frag_tbl *tbl,
138 struct rte_ip_frag_death_row *dr, struct rte_mbuf *mb, uint64_t tms,
139 struct ipv6_hdr *ip_hdr, struct ipv6_extension_fragment *frag_hdr)
140{
141 struct ip_frag_pkt *fp;
142 struct ip_frag_key key;
143 uint16_t ip_len, ip_ofs;
144
145 rte_memcpy(&key.src_dst[0], ip_hdr->src_addr, 16);
146 rte_memcpy(&key.src_dst[2], ip_hdr->dst_addr, 16);
147
148 key.id = frag_hdr->id;
149 key.key_len = IPV6_KEYLEN;
150
151 ip_ofs = FRAG_OFFSET(frag_hdr->frag_data) * 8;
152
153 /*
154 * as per RFC2460, payload length contains all extension headers as well.
155 * since we don't support anything but frag headers, this is what we remove
156 * from the payload len.
157 */
158 ip_len = rte_be_to_cpu_16(ip_hdr->payload_len) - sizeof(*frag_hdr);
159
160 IP_FRAG_LOG(DEBUG, "%s:%d:\n"
161 "mbuf: %p, tms: %" PRIu64
162 ", key: <" IPv6_KEY_BYTES_FMT ", %#x>, ofs: %u, len: %u, flags: %#x\n"
163 "tbl: %p, max_cycles: %" PRIu64 ", entry_mask: %#x, "
164 "max_entries: %u, use_entries: %u\n\n",
165 __func__, __LINE__,
166 mb, tms, IPv6_KEY_BYTES(key.src_dst), key.id, ip_ofs, ip_len,
167 RTE_IPV6_GET_MF(frag_hdr->frag_data),
168 tbl, tbl->max_cycles, tbl->entry_mask, tbl->max_entries,
169 tbl->use_entries);
170
171 /* try to find/add entry into the fragment's table. */
172 fp = ip_frag_find(tbl, dr, &key, tms);
173 if (fp == NULL) {
174 IP_FRAG_MBUF2DR(dr, mb);
175 return NULL;
176 }
177
178 IP_FRAG_LOG(DEBUG, "%s:%d:\n"
179 "tbl: %p, max_entries: %u, use_entries: %u\n"
180 "ipv6_frag_pkt: %p, key: <" IPv6_KEY_BYTES_FMT ", %#x>, start: %" PRIu64
181 ", total_size: %u, frag_size: %u, last_idx: %u\n\n",
182 __func__, __LINE__,
183 tbl, tbl->max_entries, tbl->use_entries,
184 fp, IPv6_KEY_BYTES(fp->key.src_dst), fp->key.id, fp->start,
185 fp->total_size, fp->frag_size, fp->last_idx);
186
187
188 /* process the fragmented packet. */
189 mb = ip_frag_process(fp, dr, mb, ip_ofs, ip_len,
190 MORE_FRAGS(frag_hdr->frag_data));
191 ip_frag_inuse(tbl, fp);
192
193 IP_FRAG_LOG(DEBUG, "%s:%d:\n"
194 "mbuf: %p\n"
195 "tbl: %p, max_entries: %u, use_entries: %u\n"
196 "ipv6_frag_pkt: %p, key: <" IPv6_KEY_BYTES_FMT ", %#x>, start: %" PRIu64
197 ", total_size: %u, frag_size: %u, last_idx: %u\n\n",
198 __func__, __LINE__, mb,
199 tbl, tbl->max_entries, tbl->use_entries,
200 fp, IPv6_KEY_BYTES(fp->key.src_dst), fp->key.id, fp->start,
201 fp->total_size, fp->frag_size, fp->last_idx);
202
203 return mb;
204}
