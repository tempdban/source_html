51/**
52 * @file
53 * Huge page mapping under linux
54 *
55 * To reserve a big contiguous amount of memory, we use the hugepage
56 * feature of linux. For that, we need to have hugetlbfs mounted. This
57 * code will create many files in this directory (one per page) and
58 * map them in virtual memory. For each page, we will retrieve its
59 * physical address and remap it in order to have a virtual contiguous
60 * zone as well as a physical contiguous zone.
61 */

65#define RANDOMIZE_VA_SPACE_FILE "/proc/sys/kernel/randomize_va_space"

67static void
68test_phys_addrs_available(void)
69{
70 uint64_t tmp = 0;
71 phys_addr_t physaddr;
72
73 if (!rte_eal_has_hugepages()) {
74 RTE_LOG(ERR, EAL,
75 "Started without hugepages support, physical addresses not available\n");
76 phys_addrs_available = false;
77 return;
78 }
79
80 physaddr = rte_mem_virt2phy(&tmp);
81 if (physaddr == RTE_BAD_PHYS_ADDR) {
82 if (rte_eal_iova_mode() == RTE_IOVA_PA)
83 RTE_LOG(ERR, EAL,
84 "Cannot obtain physical addresses: %s. "
85 "Only vfio will function.\n",
86 strerror(errno));
87 phys_addrs_available = false;
88 }
89}

91/*
92 * Get physical address of any mapped virtual address in the current process.
93 */
94phys_addr_t
95rte_mem_virt2phy(const void *virtaddr)
96{
97 int fd, retval;
98 uint64_t page, physaddr;
99 unsigned long virt_pfn;
100 int page_size;
101 off_t offset;
102
103 /* Cannot parse /proc/self/pagemap, no need to log errors everywhere */
104 if (!phys_addrs_available)
105 return RTE_BAD_IOVA;
106
107 /* standard page size */
108 page_size = getpagesize();
109
110 fd = open("/proc/self/pagemap", O_RDONLY);
111 if (fd < 0) {
112 RTE_LOG(ERR, EAL, "%s(): cannot open /proc/self/pagemap: %s\n",
113 __func__, strerror(errno));
114 return RTE_BAD_IOVA;
115 }
116
117 virt_pfn = (unsigned long)virtaddr / page_size;
118 offset = sizeof(uint64_t) * virt_pfn;
119 if (lseek(fd, offset, SEEK_SET) == (off_t) -1) {
120 RTE_LOG(ERR, EAL, "%s(): seek error in /proc/self/pagemap: %s\n",
121 __func__, strerror(errno));
122 close(fd);
123 return RTE_BAD_IOVA;
124 }
125
126 retval = read(fd, &page, PFN_MASK_SIZE);
127 close(fd);
128 if (retval < 0) {
129 RTE_LOG(ERR, EAL, "%s(): cannot read /proc/self/pagemap: %s\n",
130 __func__, strerror(errno));
131 return RTE_BAD_IOVA;
132 } else if (retval != PFN_MASK_SIZE) {
133 RTE_LOG(ERR, EAL, "%s(): read %d bytes from /proc/self/pagemap "
134 "but expected %d:\n",
135 __func__, retval, PFN_MASK_SIZE);
136 return RTE_BAD_IOVA;
137 }
138
139 /*
140 * the pfn (page frame number) are bits 0-54 (see
141 * pagemap.txt in linux Documentation)
142 */
143 if ((page & 0x7fffffffffffffULL) == 0)
144 return RTE_BAD_IOVA;
145
146 physaddr = ((page & 0x7fffffffffffffULL) * page_size)
147 + ((unsigned long)virtaddr % page_size);
148
149 return physaddr;
150}

160/*
161 * For each hugepage in hugepg_tbl, fill the physaddr value. We find
162 * it by browsing the /proc/self/pagemap special file.
163 */
164static int
165find_physaddrs(struct hugepage_file *hugepg_tbl, struct hugepage_info *hpi)
166{
167 unsigned int i;
168 phys_addr_t addr;
169
170 for (i = 0; i < hpi->num_pages[0]; i++) {
171 addr = rte_mem_virt2phy(hugepg_tbl[i].orig_va);
172 if (addr == RTE_BAD_PHYS_ADDR)
173 return -1;
174 hugepg_tbl[i].physaddr = addr;
175 }
176 return 0;
177}

195/*
196 * Check whether address-space layout randomization is enabled in
197 * the kernel. This is important for multi-process as it can prevent
198 * two processes mapping data to the same virtual address
199 * Returns:
200 * 0 - address space randomization disabled
201 * 1/2 - address space randomization enabled
202 * negative error code on error
203 */
204static int
205aslr_enabled(void)
206{
207 char c;
208 int retval, fd = open(RANDOMIZE_VA_SPACE_FILE, O_RDONLY);
209 if (fd < 0)
210 return -errno;
211 retval = read(fd, &c, 1);
212 close(fd);
213 if (retval < 0)
214 return -errno;
215 if (retval == 0)
216 return -EIO;
217 switch (c) {
218 case '0' : return 0;
219 case '1' : return 1;
220 case '2' : return 2;
221 default: return -EINVAL;
222 }
223}

249/*
250 * Mmap all hugepages of hugepage table: it first open a file in
251 * hugetlbfs, then mmap() hugepage_sz data in it. If orig is set, the
252 * virtual address is stored in hugepg_tbl[i].orig_va, else it is stored
253 * in hugepg_tbl[i].final_va. The second mapping (when orig is 0) tries to
254 * map contiguous physical blocks in contiguous virtual blocks.
255 */
256static unsigned
257map_all_hugepages(struct hugepage_file *hugepg_tbl, struct hugepage_info *hpi,
258 uint64_t *essential_memory __rte_unused)
259{
260 int fd;
261 unsigned i;
262 void *virtaddr;
263#ifdef RTE_EAL_NUMA_AWARE_HUGEPAGES
264 int node_id = -1;
265 int essential_prev = 0;
266 int oldpolicy;
267 struct bitmask *oldmask = numa_allocate_nodemask();
268 bool have_numa = true;
269 unsigned long maxnode = 0;
270
271 /* Check if kernel supports NUMA. */
272 if (numa_available() != 0) {
273 RTE_LOG(DEBUG, EAL, "NUMA is not supported.\n");
274 have_numa = false;
275 }
276
277 if (have_numa) {
278 RTE_LOG(DEBUG, EAL, "Trying to obtain current memory policy.\n");
279 if (get_mempolicy(&oldpolicy, oldmask->maskp,
280 oldmask->size + 1, 0, 0) < 0) {
281 RTE_LOG(ERR, EAL,
282 "Failed to get current mempolicy: %s. "
283 "Assuming MPOL_DEFAULT.\n", strerror(errno));
284 oldpolicy = MPOL_DEFAULT;
285 }
286 for (i = 0; i < RTE_MAX_NUMA_NODES; i++)
287 if (internal_config.socket_mem[i])
288 maxnode = i + 1;
289 }
290#endif
291
292 for (i = 0; i < hpi->num_pages[0]; i++) {
293 struct hugepage_file *hf = &hugepg_tbl[i];
294 uint64_t hugepage_sz = hpi->hugepage_sz;
295
296#ifdef RTE_EAL_NUMA_AWARE_HUGEPAGES
297 if (maxnode) {
298 unsigned int j;
299
300 for (j = 0; j < maxnode; j++)
301 if (essential_memory[j])
302 break;
303
304 if (j == maxnode) {
305 node_id = (node_id + 1) % maxnode;
306 while (!internal_config.socket_mem[node_id]) {
307 node_id++;
308 node_id %= maxnode;
309 }
310 essential_prev = 0;
311 } else {
312 node_id = j;
313 essential_prev = essential_memory[j];
314
315 if (essential_memory[j] < hugepage_sz)
316 essential_memory[j] = 0;
317 else
318 essential_memory[j] -= hugepage_sz;
319 }
320
321 RTE_LOG(DEBUG, EAL,
322 "Setting policy MPOL_PREFERRED for socket %d\n",
323 node_id);
324 numa_set_preferred(node_id);
325 }
326#endif
327
328 hf->file_id = i;
329 hf->size = hugepage_sz;
330 eal_get_hugefile_path(hf->filepath, sizeof(hf->filepath),
331 hpi->hugedir, hf->file_id);
332 hf->filepath[sizeof(hf->filepath) - 1] = '\0';
333
334 /* try to create hugepage file */
335 fd = open(hf->filepath, O_CREAT | O_RDWR, 0600);
336 if (fd < 0) {
337 RTE_LOG(DEBUG, EAL, "%s(): open failed: %s\n", __func__,
338 strerror(errno));
339 goto out;
340 }
341
342 /* map the segment, and populate page tables,
343 * the kernel fills this segment with zeros. we don't care where
344 * this gets mapped - we already have contiguous memory areas
345 * ready for us to map into.
346 */
347 virtaddr = mmap(NULL, hugepage_sz, PROT_READ | PROT_WRITE,
348 MAP_SHARED | MAP_POPULATE, fd, 0);
349 if (virtaddr == MAP_FAILED) {
350 RTE_LOG(DEBUG, EAL, "%s(): mmap failed: %s\n", __func__,
351 strerror(errno));
352 close(fd);
353 goto out;
354 }
355
356 hf->orig_va = virtaddr;
357
358 /* In linux, hugetlb limitations, like cgroup, are
359 * enforced at fault time instead of mmap(), even
360 * with the option of MAP_POPULATE. Kernel will send
361 * a SIGBUS signal. To avoid to be killed, save stack
362 * environment here, if SIGBUS happens, we can jump
363 * back here.
364 */
365 if (huge_wrap_sigsetjmp()) {
366 RTE_LOG(DEBUG, EAL, "SIGBUS: Cannot mmap more "
367 "hugepages of size %u MB\n",
368 (unsigned int)(hugepage_sz / 0x100000));
369 munmap(virtaddr, hugepage_sz);
370 close(fd);
371 unlink(hugepg_tbl[i].filepath);
372#ifdef RTE_EAL_NUMA_AWARE_HUGEPAGES
373 if (maxnode)
374 essential_memory[node_id] =
375 essential_prev;
376#endif
377 goto out;
378 }
379 *(int *)virtaddr = 0;
380
381 /* set shared lock on the file. */
382 if (flock(fd, LOCK_SH) < 0) {
383 RTE_LOG(DEBUG, EAL, "%s(): Locking file failed:%s \n",
384 __func__, strerror(errno));
385 close(fd);
386 goto out;
387 }
388
389 close(fd);
390 }
391
392out:
393#ifdef RTE_EAL_NUMA_AWARE_HUGEPAGES
394 if (maxnode) {
395 RTE_LOG(DEBUG, EAL,
396 "Restoring previous memory policy: %d\n", oldpolicy);
397 if (oldpolicy == MPOL_DEFAULT) {
398 numa_set_localalloc();
399 } else if (set_mempolicy(oldpolicy, oldmask->maskp,
400 oldmask->size + 1) < 0) {
401 RTE_LOG(ERR, EAL, "Failed to restore mempolicy: %s\n",
402 strerror(errno));
403 numa_set_localalloc();
404 }
405 }
406 numa_free_cpumask(oldmask);
407#endif
408 return i;
409}

411/*
412 * Parse /proc/self/numa_maps to get the NUMA socket ID for each huge
413 * page.
414 */
415static int
416find_numasocket(struct hugepage_file *hugepg_tbl, struct hugepage_info *hpi)
417{
418 int socket_id;
419 char *end, *nodestr;
420 unsigned i, hp_count = 0;
421 uint64_t virt_addr;
422 char buf[BUFSIZ];
423 char hugedir_str[PATH_MAX];
424 FILE *f;
425
426 f = fopen("/proc/self/numa_maps", "r");
427 if (f == NULL) {
428 RTE_LOG(NOTICE, EAL, "NUMA support not available"
429 " consider that all memory is in socket_id 0\n");
430 return 0;
431 }
432
433 snprintf(hugedir_str, sizeof(hugedir_str),
434 "%s/%s", hpi->hugedir, internal_config.hugefile_prefix);
435
436 /* parse numa map */
437 while (fgets(buf, sizeof(buf), f) != NULL) {
438
439 /* ignore non huge page */
440 if (strstr(buf, " huge ") == NULL &&
441 strstr(buf, hugedir_str) == NULL)
442 continue;
443
444 /* get zone addr */
445 virt_addr = strtoull(buf, &end, 16);
446 if (virt_addr == 0 || end == buf) {
447 RTE_LOG(ERR, EAL, "%s(): error in numa_maps parsing\n", __func__);
448 goto error;
449 }
450
451 /* get node id (socket id) */
452 nodestr = strstr(buf, " N");
453 if (nodestr == NULL) {
454 RTE_LOG(ERR, EAL, "%s(): error in numa_maps parsing\n", __func__);
455 goto error;
456 }
457 nodestr += 2;
458 end = strstr(nodestr, "=");
459 if (end == NULL) {
460 RTE_LOG(ERR, EAL, "%s(): error in numa_maps parsing\n", __func__);
461 goto error;
462 }
463 end[0] = '\0';
464 end = NULL;
465
466 socket_id = strtoul(nodestr, &end, 0);
467 if ((nodestr[0] == '\0') || (end == NULL) || (*end != '\0')) {
468 RTE_LOG(ERR, EAL, "%s(): error in numa_maps parsing\n", __func__);
469 goto error;
470 }
471
472 /* if we find this page in our mappings, set socket_id */
473 for (i = 0; i < hpi->num_pages[0]; i++) {
474 void *va = (void *)(unsigned long)virt_addr;
475 if (hugepg_tbl[i].orig_va == va) {
476 hugepg_tbl[i].socket_id = socket_id;
477 hp_count++;
478#ifdef RTE_EAL_NUMA_AWARE_HUGEPAGES
479 RTE_LOG(DEBUG, EAL,
480 "Hugepage %s is on socket %d\n",
481 hugepg_tbl[i].filepath, socket_id);
482#endif
483 }
484 }
485 }
486
487 if (hp_count < hpi->num_pages[0])
488 goto error;
489
490 fclose(f);
491 return 0;
492
493error:
494 fclose(f);
495 return -1;
496}

498static int
499cmp_physaddr(const void *a, const void *b)
500{
501#ifndef RTE_ARCH_PPC_64
502 const struct hugepage_file *p1 = a;
503 const struct hugepage_file *p2 = b;
504#else
505 /* PowerPC needs memory sorted in reverse order from x86 */
506 const struct hugepage_file *p1 = b;
507 const struct hugepage_file *p2 = a;
508#endif
509 if (p1->physaddr < p2->physaddr)
510 return -1;
511 else if (p1->physaddr > p2->physaddr)
512 return 1;
513 else
514 return 0;
515}

517/*
518 * Uses mmap to create a shared memory area for storage of data
519 * Used in this file to store the hugepage file map on disk
520 */
521static void *
522create_shared_memory(const char *filename, const size_t mem_size)
523{
524 void *retval;
525 int fd;
526
527 /* if no shared files mode is used, create anonymous memory instead */
528 if (internal_config.no_shconf) {
529 retval = mmap(NULL, mem_size, PROT_READ | PROT_WRITE,
530 MAP_PRIVATE | MAP_ANONYMOUS, -1, 0);
531 if (retval == MAP_FAILED)
532 return NULL;
533 return retval;
534 }
535
536 fd = open(filename, O_CREAT | O_RDWR, 0666);
537 if (fd < 0)
538 return NULL;
539 if (ftruncate(fd, mem_size) < 0) {
540 close(fd);
541 return NULL;
542 }
543 retval = mmap(NULL, mem_size, PROT_READ | PROT_WRITE, MAP_SHARED, fd, 0);
544 close(fd);
545 if (retval == MAP_FAILED)
546 return NULL;
547 return retval;
548}

550/*
551 * this copies *active* hugepages from one hugepage table to another.
552 * destination is typically the shared memory.
553 */
554static int
555copy_hugepages_to_shared_mem(struct hugepage_file * dst, int dest_size,
556 const struct hugepage_file * src, int src_size)
557{
558 int src_pos, dst_pos = 0;
559
560 for (src_pos = 0; src_pos < src_size; src_pos++) {
561 if (src[src_pos].orig_va != NULL) {
562 /* error on overflow attempt */
563 if (dst_pos == dest_size)
564 return -1;
565 memcpy(&dst[dst_pos], &src[src_pos], sizeof(struct hugepage_file));
566 dst_pos++;
567 }
568 }
569 return 0;
570}

572static int
573unlink_hugepage_files(struct hugepage_file *hugepg_tbl,
574 unsigned num_hp_info)
575{
576 unsigned socket, size;
577 int page, nrpages = 0;
578
579 /* get total number of hugepages */
580 for (size = 0; size < num_hp_info; size++)
581 for (socket = 0; socket < RTE_MAX_NUMA_NODES; socket++)
582 nrpages +=
583 internal_config.hugepage_info[size].num_pages[socket];
584
585 for (page = 0; page < nrpages; page++) {
586 struct hugepage_file *hp = &hugepg_tbl[page];
587
588 if (hp->final_va != NULL && unlink(hp->filepath)) {
589 RTE_LOG(WARNING, EAL, "%s(): Removing %s failed: %s\n",
590 __func__, hp->filepath, strerror(errno));
591 }
592 }
593 return 0;
594}

596/*
597 * unmaps hugepages that are not going to be used. since we originally allocate
598 * ALL hugepages (not just those we need), additional unmapping needs to be done.
599 */
600static int
601unmap_unneeded_hugepages(struct hugepage_file *hugepg_tbl,
602 struct hugepage_info *hpi,
603 unsigned num_hp_info)
604{
605 unsigned socket, size;
606 int page, nrpages = 0;
607
608 /* get total number of hugepages */
609 for (size = 0; size < num_hp_info; size++)
610 for (socket = 0; socket < RTE_MAX_NUMA_NODES; socket++)
611 nrpages += internal_config.hugepage_info[size].num_pages[socket];
612
613 for (size = 0; size < num_hp_info; size++) {
614 for (socket = 0; socket < RTE_MAX_NUMA_NODES; socket++) {
615 unsigned pages_found = 0;
616
617 /* traverse until we have unmapped all the unused pages */
618 for (page = 0; page < nrpages; page++) {
619 struct hugepage_file *hp = &hugepg_tbl[page];
620
621 /* find a page that matches the criteria */
622 if ((hp->size == hpi[size].hugepage_sz) &&
623 (hp->socket_id == (int) socket)) {
624
625 /* if we skipped enough pages, unmap the rest */
626 if (pages_found == hpi[size].num_pages[socket]) {
627 uint64_t unmap_len;
628
629 unmap_len = hp->size;
630
631 /* get start addr and len of the remaining segment */
632 munmap(hp->orig_va,
633 (size_t)unmap_len);
634
635 hp->orig_va = NULL;
636 if (unlink(hp->filepath) == -1) {
637 RTE_LOG(ERR, EAL, "%s(): Removing %s failed: %s\n",
638 __func__, hp->filepath, strerror(errno));
639 return -1;
640 }
641 } else {
642 /* lock the page and skip */
643 pages_found++;
644 }
645
646 } /* match page */
647 } /* foreach page */
648 } /* foreach socket */
649 } /* foreach pagesize */
650
651 return 0;
652}

654static int
655remap_segment(struct hugepage_file *hugepages, int seg_start, int seg_end)
656{
657 struct rte_mem_config *mcfg = rte_eal_get_configuration()->mem_config;
658 struct rte_memseg_list *msl;
659 struct rte_fbarray *arr;
660 int cur_page, seg_len;
661 unsigned int msl_idx;
662 int ms_idx;
663 uint64_t page_sz;
664 size_t memseg_len;
665 int socket_id;
666
667 page_sz = hugepages[seg_start].size;
668 socket_id = hugepages[seg_start].socket_id;
669 seg_len = seg_end - seg_start;
670
671 RTE_LOG(DEBUG, EAL, "Attempting to map %" PRIu64 "M on socket %i\n",
672 (seg_len * page_sz) >> 20ULL, socket_id);
673
674 /* find free space in memseg lists */
675 for (msl_idx = 0; msl_idx < RTE_MAX_MEMSEG_LISTS; msl_idx++) {
676 bool empty;
677 msl = &mcfg->memsegs[msl_idx];
678 arr = &msl->memseg_arr;
679
680 if (msl->page_sz != page_sz)
681 continue;
682 if (msl->socket_id != socket_id)
683 continue;
684
685 /* leave space for a hole if array is not empty */
686 empty = arr->count == 0;
687 ms_idx = rte_fbarray_find_next_n_free(arr, 0,
688 seg_len + (empty ? 0 : 1));
689
690 /* memseg list is full? */
691 if (ms_idx < 0)
692 continue;
693
694 /* leave some space between memsegs, they are not IOVA
695 * contiguous, so they shouldn't be VA contiguous either.
696 */
697 if (!empty)
698 ms_idx++;
699 break;
700 }
701 if (msl_idx == RTE_MAX_MEMSEG_LISTS) {
702 RTE_LOG(ERR, EAL, "Could not find space for memseg. Please increase %s and/or %s in configuration.\n",
703 RTE_STR(CONFIG_RTE_MAX_MEMSEG_PER_TYPE),
704 RTE_STR(CONFIG_RTE_MAX_MEM_PER_TYPE));
705 return -1;
706 }
707
708#ifdef RTE_ARCH_PPC64
709 /* for PPC64 we go through the list backwards */
710 for (cur_page = seg_end - 1; cur_page >= seg_start;
711 cur_page--, ms_idx++) {
712#else
713 for (cur_page = seg_start; cur_page < seg_end; cur_page++, ms_idx++) {
714#endif
715 struct hugepage_file *hfile = &hugepages[cur_page];
716 struct rte_memseg *ms = rte_fbarray_get(arr, ms_idx);
717 void *addr;
718 int fd;
719
720 fd = open(hfile->filepath, O_RDWR);
721 if (fd < 0) {
722 RTE_LOG(ERR, EAL, "Could not open '%s': %s\n",
723 hfile->filepath, strerror(errno));
724 return -1;
725 }
726 /* set shared lock on the file. */
727 if (flock(fd, LOCK_SH) < 0) {
728 RTE_LOG(DEBUG, EAL, "Could not lock '%s': %s\n",
729 hfile->filepath, strerror(errno));
730 close(fd);
731 return -1;
732 }
733 memseg_len = (size_t)page_sz;
734 addr = RTE_PTR_ADD(msl->base_va, ms_idx * memseg_len);
735
736 /* we know this address is already mmapped by memseg list, so
737 * using MAP_FIXED here is safe
738 */
739 addr = mmap(addr, page_sz, PROT_READ | PROT_WRITE,
740 MAP_SHARED | MAP_POPULATE | MAP_FIXED, fd, 0);
741 if (addr == MAP_FAILED) {
742 RTE_LOG(ERR, EAL, "Couldn't remap '%s': %s\n",
743 hfile->filepath, strerror(errno));
744 close(fd);
745 return -1;
746 }
747
748 /* we have a new address, so unmap previous one */
749#ifndef RTE_ARCH_64
750 /* in 32-bit legacy mode, we have already unmapped the page */
751 if (!internal_config.legacy_mem)
752 munmap(hfile->orig_va, page_sz);
753#else
754 munmap(hfile->orig_va, page_sz);
755#endif
756
757 hfile->orig_va = NULL;
758 hfile->final_va = addr;
759
760 /* rewrite physical addresses in IOVA as VA mode */
761 if (rte_eal_iova_mode() == RTE_IOVA_VA)
762 hfile->physaddr = (uintptr_t)addr;
763
764 /* set up memseg data */
765 ms->addr = addr;
766 ms->hugepage_sz = page_sz;
767 ms->len = memseg_len;
768 ms->iova = hfile->physaddr;
769 ms->socket_id = hfile->socket_id;
770 ms->nchannel = rte_memory_get_nchannel();
771 ms->nrank = rte_memory_get_nrank();
772
773 rte_fbarray_set_used(arr, ms_idx);
774
775 /* store segment fd internally */
776 if (eal_memalloc_set_seg_fd(msl_idx, ms_idx, fd) < 0)
777 RTE_LOG(ERR, EAL, "Could not store segment fd: %s\n",
778 rte_strerror(rte_errno));
779 }
780 RTE_LOG(DEBUG, EAL, "Allocated %" PRIu64 "M on socket %i\n",
781 (seg_len * page_sz) >> 20, socket_id);
782 return 0;
783}

839static int
840alloc_va_space(struct rte_memseg_list *msl)
841{
842 uint64_t page_sz;
843 size_t mem_sz;
844 void *addr;
845 int flags = 0;
846
847#ifdef RTE_ARCH_PPC_64
848 flags |= MAP_HUGETLB;
849#endif
850
851 page_sz = msl->page_sz;
852 mem_sz = page_sz * msl->memseg_arr.len;
853
854 addr = eal_get_virtual_area(msl->base_va, &mem_sz, page_sz, 0, flags);
855 if (addr == NULL) {
856 if (rte_errno == EADDRNOTAVAIL)
857 RTE_LOG(ERR, EAL, "Could not mmap %llu bytes at [%p] - please use '--base-virtaddr' option\n",
858 (unsigned long long)mem_sz, msl->base_va);
859 else
860 RTE_LOG(ERR, EAL, "Cannot reserve memory\n");
861 return -1;
862 }
863 msl->base_va = addr;
864
865 return 0;
866}

868/*
869 * Our VA space is not preallocated yet, so preallocate it here. We need to know
870 * how many segments there are in order to map all pages into one address space,
871 * and leave appropriate holes between segments so that rte_malloc does not
872 * concatenate them into one big segment.
873 *
874 * we also need to unmap original pages to free up address space.
875 */
876static int __rte_unused
877prealloc_segments(struct hugepage_file *hugepages, int n_pages)
878{
879 struct rte_mem_config *mcfg = rte_eal_get_configuration()->mem_config;
880 int cur_page, seg_start_page, end_seg, new_memseg;
881 unsigned int hpi_idx, socket, i;
882 int n_contig_segs, n_segs;
883 int msl_idx;
884
885 /* before we preallocate segments, we need to free up our VA space.
886 * we're not removing files, and we already have information about
887 * PA-contiguousness, so it is safe to unmap everything.
888 */
889 for (cur_page = 0; cur_page < n_pages; cur_page++) {
890 struct hugepage_file *hpi = &hugepages[cur_page];
891 munmap(hpi->orig_va, hpi->size);
892 hpi->orig_va = NULL;
893 }
894
895 /* we cannot know how many page sizes and sockets we have discovered, so
896 * loop over all of them
897 */
898 for (hpi_idx = 0; hpi_idx < internal_config.num_hugepage_sizes;
899 hpi_idx++) {
900 uint64_t page_sz =
901 internal_config.hugepage_info[hpi_idx].hugepage_sz;
902
903 for (i = 0; i < rte_socket_count(); i++) {
904 struct rte_memseg_list *msl;
905
906 socket = rte_socket_id_by_idx(i);
907 n_contig_segs = 0;
908 n_segs = 0;
909 seg_start_page = -1;
910
911 for (cur_page = 0; cur_page < n_pages; cur_page++) {
912 struct hugepage_file *prev, *cur;
913 int prev_seg_start_page = -1;
914
915 cur = &hugepages[cur_page];
916 prev = cur_page == 0 ? NULL :
917 &hugepages[cur_page - 1];
918
919 new_memseg = 0;
920 end_seg = 0;
921
922 if (cur->size == 0)
923 end_seg = 1;
924 else if (cur->socket_id != (int) socket)
925 end_seg = 1;
926 else if (cur->size != page_sz)
927 end_seg = 1;
928 else if (cur_page == 0)
929 new_memseg = 1;
930#ifdef RTE_ARCH_PPC_64
931 /* On PPC64 architecture, the mmap always start
932 * from higher address to lower address. Here,
933 * physical addresses are in descending order.
934 */
935 else if ((prev->physaddr - cur->physaddr) !=
936 cur->size)
937 new_memseg = 1;
938#else
939 else if ((cur->physaddr - prev->physaddr) !=
940 cur->size)
941 new_memseg = 1;
942#endif
943 if (new_memseg) {
944 /* if we're already inside a segment,
945 * new segment means end of current one
946 */
947 if (seg_start_page != -1) {
948 end_seg = 1;
949 prev_seg_start_page =
950 seg_start_page;
951 }
952 seg_start_page = cur_page;
953 }
954
955 if (end_seg) {
956 if (prev_seg_start_page != -1) {
957 /* we've found a new segment */
958 n_contig_segs++;
959 n_segs += cur_page -
960 prev_seg_start_page;
961 } else if (seg_start_page != -1) {
962 /* we didn't find new segment,
963 * but did end current one
964 */
965 n_contig_segs++;
966 n_segs += cur_page -
967 seg_start_page;
968 seg_start_page = -1;
969 continue;
970 } else {
971 /* we're skipping this page */
972 continue;
973 }
974 }
975 /* segment continues */
976 }
977 /* check if we missed last segment */
978 if (seg_start_page != -1) {
979 n_contig_segs++;
980 n_segs += cur_page - seg_start_page;
981 }
982
983 /* if no segments were found, do not preallocate */
984 if (n_segs == 0)
985 continue;
986
987 /* we now have total number of pages that we will
988 * allocate for this segment list. add separator pages
989 * to the total count, and preallocate VA space.
990 */
991 n_segs += n_contig_segs - 1;
992
993 /* now, preallocate VA space for these segments */
994
995 /* first, find suitable memseg list for this */
996 for (msl_idx = 0; msl_idx < RTE_MAX_MEMSEG_LISTS;
997 msl_idx++) {
998 msl = &mcfg->memsegs[msl_idx];
999
1000 if (msl->base_va != NULL)
1001 continue;
1002 break;
1003 }
1004 if (msl_idx == RTE_MAX_MEMSEG_LISTS) {
1005 RTE_LOG(ERR, EAL, "Not enough space in memseg lists, please increase %s\n",
1006 RTE_STR(CONFIG_RTE_MAX_MEMSEG_LISTS));
1007 return -1;
1008 }
1009
1010 /* now, allocate fbarray itself */
1011 if (alloc_memseg_list(msl, page_sz, n_segs, socket,
1012 msl_idx) < 0)
1013 return -1;
1014
1015 /* finally, allocate VA space */
1016 if (alloc_va_space(msl) < 0)
1017 return -1;
1018 }
1019 }
1020 return 0;
1021}

1023/*
1024 * We cannot reallocate memseg lists on the fly because PPC64 stores pages
1025 * backwards, therefore we have to process the entire memseg first before
1026 * remapping it into memseg list VA space.
1027 */
1028static int
1029remap_needed_hugepages(struct hugepage_file *hugepages, int n_pages)
1030{
1031 int cur_page, seg_start_page, new_memseg, ret;
1032
1033 seg_start_page = 0;
1034 for (cur_page = 0; cur_page < n_pages; cur_page++) {
1035 struct hugepage_file *prev, *cur;
1036
1037 new_memseg = 0;
1038
1039 cur = &hugepages[cur_page];
1040 prev = cur_page == 0 ? NULL : &hugepages[cur_page - 1];
1041
1042 /* if size is zero, no more pages left */
1043 if (cur->size == 0)
1044 break;
1045
1046 if (cur_page == 0)
1047 new_memseg = 1;
1048 else if (cur->socket_id != prev->socket_id)
1049 new_memseg = 1;
1050 else if (cur->size != prev->size)
1051 new_memseg = 1;
1052#ifdef RTE_ARCH_PPC_64
1053 /* On PPC64 architecture, the mmap always start from higher
1054 * address to lower address. Here, physical addresses are in
1055 * descending order.
1056 */
1057 else if ((prev->physaddr - cur->physaddr) != cur->size)
1058 new_memseg = 1;
1059#else
1060 else if ((cur->physaddr - prev->physaddr) != cur->size)
1061 new_memseg = 1;
1062#endif
1063
1064 if (new_memseg) {
1065 /* if this isn't the first time, remap segment */
1066 if (cur_page != 0) {
1067 ret = remap_segment(hugepages, seg_start_page,
1068 cur_page);
1069 if (ret != 0)
1070 return -1;
1071 }
1072 /* remember where we started */
1073 seg_start_page = cur_page;
1074 }
1075 /* continuation of previous memseg */
1076 }
1077 /* we were stopped, but we didn't remap the last segment, do it now */
1078 if (cur_page != 0) {
1079 ret = remap_segment(hugepages, seg_start_page,
1080 cur_page);
1081 if (ret != 0)
1082 return -1;
1083 }
1084 return 0;
1085}

1101/*
1102 * This function is a NUMA-aware equivalent of calc_num_pages.
1103 * It takes in the list of hugepage sizes and the
1104 * number of pages thereof, and calculates the best number of
1105 * pages of each size to fulfill the request for <memory> ram
1106 */
1107static int
1108calc_num_pages_per_socket(uint64_t * memory,
1109 struct hugepage_info *hp_info,
1110 struct hugepage_info *hp_used,
1111 unsigned num_hp_info)
1112{
1113 unsigned socket, j, i = 0;
1114 unsigned requested, available;
1115 int total_num_pages = 0;
1116 uint64_t remaining_mem, cur_mem;
1117 uint64_t total_mem = internal_config.memory;
1118
1119 if (num_hp_info == 0)
1120 return -1;
1121
1122 /* if specific memory amounts per socket weren't requested */
1123 if (internal_config.force_sockets == 0) {
1124 size_t total_size;
1125#ifdef RTE_ARCH_64
1126 int cpu_per_socket[RTE_MAX_NUMA_NODES];
1127 size_t default_size;
1128 unsigned lcore_id;
1129
1130 /* Compute number of cores per socket */
1131 memset(cpu_per_socket, 0, sizeof(cpu_per_socket));
1132 RTE_LCORE_FOREACH(lcore_id) {
1133 cpu_per_socket[rte_lcore_to_socket_id(lcore_id)]++;
1134 }
1135
1136 /*
1137 * Automatically spread requested memory amongst detected sockets according
1138 * to number of cores from cpu mask present on each socket
1139 */
1140 total_size = internal_config.memory;
1141 for (socket = 0; socket < RTE_MAX_NUMA_NODES && total_size != 0; socket++) {
1142
1143 /* Set memory amount per socket */
1144 default_size = (internal_config.memory * cpu_per_socket[socket])
1145 / rte_lcore_count();
1146
1147 /* Limit to maximum available memory on socket */
1148 default_size = RTE_MIN(default_size, get_socket_mem_size(socket));
1149
1150 /* Update sizes */
1151 memory[socket] = default_size;
1152 total_size -= default_size;
1153 }
1154
1155 /*
1156 * If some memory is remaining, try to allocate it by getting all
1157 * available memory from sockets, one after the other
1158 */
1159 for (socket = 0; socket < RTE_MAX_NUMA_NODES && total_size != 0; socket++) {
1160 /* take whatever is available */
1161 default_size = RTE_MIN(get_socket_mem_size(socket) - memory[socket],
1162 total_size);
1163
1164 /* Update sizes */
1165 memory[socket] += default_size;
1166 total_size -= default_size;
1167 }
1168#else
1169 /* in 32-bit mode, allocate all of the memory only on master
1170 * lcore socket
1171 */
1172 total_size = internal_config.memory;
1173 for (socket = 0; socket < RTE_MAX_NUMA_NODES && total_size != 0;
1174 socket++) {
1175 struct rte_config *cfg = rte_eal_get_configuration();
1176 unsigned int master_lcore_socket;
1177
1178 master_lcore_socket =
1179 rte_lcore_to_socket_id(cfg->master_lcore);
1180
1181 if (master_lcore_socket != socket)
1182 continue;
1183
1184 /* Update sizes */
1185 memory[socket] = total_size;
1186 break;
1187 }
1188#endif
1189 }
1190
1191 for (socket = 0; socket < RTE_MAX_NUMA_NODES && total_mem != 0; socket++) {
1192 /* skips if the memory on specific socket wasn't requested */
1193 for (i = 0; i < num_hp_info && memory[socket] != 0; i++){
1194 strlcpy(hp_used[i].hugedir, hp_info[i].hugedir,
1195 sizeof(hp_used[i].hugedir));
1196 hp_used[i].num_pages[socket] = RTE_MIN(
1197 memory[socket] / hp_info[i].hugepage_sz,
1198 hp_info[i].num_pages[socket]);
1199
1200 cur_mem = hp_used[i].num_pages[socket] *
1201 hp_used[i].hugepage_sz;
1202
1203 memory[socket] -= cur_mem;
1204 total_mem -= cur_mem;
1205
1206 total_num_pages += hp_used[i].num_pages[socket];
1207
1208 /* check if we have met all memory requests */
1209 if (memory[socket] == 0)
1210 break;
1211
1212 /* check if we have any more pages left at this size, if so
1213 * move on to next size */
1214 if (hp_used[i].num_pages[socket] == hp_info[i].num_pages[socket])
1215 continue;
1216 /* At this point we know that there are more pages available that are
1217 * bigger than the memory we want, so lets see if we can get enough
1218 * from other page sizes.
1219 */
1220 remaining_mem = 0;
1221 for (j = i+1; j < num_hp_info; j++)
1222 remaining_mem += hp_info[j].hugepage_sz *
1223 hp_info[j].num_pages[socket];
1224
1225 /* is there enough other memory, if not allocate another page and quit */
1226 if (remaining_mem < memory[socket]){
1227 cur_mem = RTE_MIN(memory[socket],
1228 hp_info[i].hugepage_sz);
1229 memory[socket] -= cur_mem;
1230 total_mem -= cur_mem;
1231 hp_used[i].num_pages[socket]++;
1232 total_num_pages++;
1233 break; /* we are done with this socket*/
1234 }
1235 }
1236 /* if we didn't satisfy all memory requirements per socket */
1237 if (memory[socket] > 0 &&
1238 internal_config.socket_mem[socket] != 0) {
1239 /* to prevent icc errors */
1240 requested = (unsigned) (internal_config.socket_mem[socket] /
1241 0x100000);
1242 available = requested -
1243 ((unsigned) (memory[socket] / 0x100000));
1244 RTE_LOG(ERR, EAL, "Not enough memory available on socket %u! "
1245 "Requested: %uMB, available: %uMB\n", socket,
1246 requested, available);
1247 return -1;
1248 }
1249 }
1250
1251 /* if we didn't satisfy total memory requirements */
1252 if (total_mem > 0) {
1253 requested = (unsigned) (internal_config.memory / 0x100000);
1254 available = requested - (unsigned) (total_mem / 0x100000);
1255 RTE_LOG(ERR, EAL, "Not enough memory available! Requested: %uMB,"
1256 " available: %uMB\n", requested, available);
1257 return -1;
1258 }
1259 return total_num_pages;
1260}

1307/*
1308 * Prepare physical memory mapping: fill configuration structure with
1309 * these infos, return 0 on success.
1310 * 1. map N huge pages in separate files in hugetlbfs
1311 * 2. find associated physical addr
1312 * 3. find associated NUMA socket ID
1313 * 4. sort all huge pages by physical address
1314 * 5. remap these N huge pages in the correct order
1315 * 6. unmap the first mapping
1316 * 7. fill memsegs in configuration with contiguous zones
1317 */
1318static int
1319eal_legacy_hugepage_init(void)
1320{
1321 struct rte_mem_config *mcfg;
1322 struct hugepage_file *hugepage = NULL, *tmp_hp = NULL;
1323 struct hugepage_info used_hp[MAX_HUGEPAGE_SIZES];
1324 struct rte_fbarray *arr;
1325 struct rte_memseg *ms;
1326
1327 uint64_t memory[RTE_MAX_NUMA_NODES];
1328
1329 unsigned hp_offset;
1330 int i, j;
1331 int nr_hugefiles, nr_hugepages = 0;
1332 void *addr;
1333
1334 test_phys_addrs_available();
1335
1336 memset(used_hp, 0, sizeof(used_hp));
1337
1338 /* get pointer to global configuration */
1339 mcfg = rte_eal_get_configuration()->mem_config;
1340
1341 /* hugetlbfs can be disabled */
1342 if (internal_config.no_hugetlbfs) {
1343 struct rte_memseg_list *msl;
1344 uint64_t page_sz;
1345 int n_segs, cur_seg;
1346
1347 /* nohuge mode is legacy mode */
1348 internal_config.legacy_mem = 1;
1349
1350 /* create a memseg list */
1351 msl = &mcfg->memsegs[0];
1352
1353 page_sz = RTE_PGSIZE_4K;
1354 n_segs = internal_config.memory / page_sz;
1355
1356 if (rte_fbarray_init(&msl->memseg_arr, "nohugemem", n_segs,
1357 sizeof(struct rte_memseg))) {
1358 RTE_LOG(ERR, EAL, "Cannot allocate memseg list\n");
1359 return -1;
1360 }
1361
1362 addr = mmap(NULL, internal_config.memory, PROT_READ | PROT_WRITE,
1363 MAP_PRIVATE | MAP_ANONYMOUS, -1, 0);
1364 if (addr == MAP_FAILED) {
1365 RTE_LOG(ERR, EAL, "%s: mmap() failed: %s\n", __func__,
1366 strerror(errno));
1367 return -1;
1368 }
1369 msl->base_va = addr;
1370 msl->page_sz = page_sz;
1371 msl->socket_id = 0;
1372
1373 /* populate memsegs. each memseg is one page long */
1374 for (cur_seg = 0; cur_seg < n_segs; cur_seg++) {
1375 arr = &msl->memseg_arr;
1376
1377 ms = rte_fbarray_get(arr, cur_seg);
1378 if (rte_eal_iova_mode() == RTE_IOVA_VA)
1379 ms->iova = (uintptr_t)addr;
1380 else
1381 ms->iova = RTE_BAD_IOVA;
1382 ms->addr = addr;
1383 ms->hugepage_sz = page_sz;
1384 ms->socket_id = 0;
1385 ms->len = page_sz;
1386
1387 rte_fbarray_set_used(arr, cur_seg);
1388
1389 addr = RTE_PTR_ADD(addr, (size_t)page_sz);
1390 }
1391 return 0;
1392 }
1393
1394 /* calculate total number of hugepages available. at this point we haven't
1395 * yet started sorting them so they all are on socket 0 */
1396 for (i = 0; i < (int) internal_config.num_hugepage_sizes; i++) {
1397 /* meanwhile, also initialize used_hp hugepage sizes in used_hp */
1398 used_hp[i].hugepage_sz = internal_config.hugepage_info[i].hugepage_sz;
1399
1400 nr_hugepages += internal_config.hugepage_info[i].num_pages[0];
1401 }
1402
1403 /*
1404 * allocate a memory area for hugepage table.
1405 * this isn't shared memory yet. due to the fact that we need some
1406 * processing done on these pages, shared memory will be created
1407 * at a later stage.
1408 */
1409 tmp_hp = malloc(nr_hugepages * sizeof(struct hugepage_file));
1410 if (tmp_hp == NULL)
1411 goto fail;
1412
1413 memset(tmp_hp, 0, nr_hugepages * sizeof(struct hugepage_file));
1414
1415 hp_offset = 0; /* where we start the current page size entries */
1416
1417 huge_register_sigbus();
1418
1419 /* make a copy of socket_mem, needed for balanced allocation. */
1420 for (i = 0; i < RTE_MAX_NUMA_NODES; i++)
1421 memory[i] = internal_config.socket_mem[i];
1422
1423 /* map all hugepages and sort them */
1424 for (i = 0; i < (int)internal_config.num_hugepage_sizes; i ++){
1425 unsigned pages_old, pages_new;
1426 struct hugepage_info *hpi;
1427
1428 /*
1429 * we don't yet mark hugepages as used at this stage, so
1430 * we just map all hugepages available to the system
1431 * all hugepages are still located on socket 0
1432 */
1433 hpi = &internal_config.hugepage_info[i];
1434
1435 if (hpi->num_pages[0] == 0)
1436 continue;
1437
1438 /* map all hugepages available */
1439 pages_old = hpi->num_pages[0];
1440 pages_new = map_all_hugepages(&tmp_hp[hp_offset], hpi, memory);
1441 if (pages_new < pages_old) {
1442 RTE_LOG(DEBUG, EAL,
1443 "%d not %d hugepages of size %u MB allocated\n",
1444 pages_new, pages_old,
1445 (unsigned)(hpi->hugepage_sz / 0x100000));
1446
1447 int pages = pages_old - pages_new;
1448
1449 nr_hugepages -= pages;
1450 hpi->num_pages[0] = pages_new;
1451 if (pages_new == 0)
1452 continue;
1453 }
1454
1455 if (phys_addrs_available &&
1456 rte_eal_iova_mode() != RTE_IOVA_VA) {
1457 /* find physical addresses for each hugepage */
1458 if (find_physaddrs(&tmp_hp[hp_offset], hpi) < 0) {
1459 RTE_LOG(DEBUG, EAL, "Failed to find phys addr "
1460 "for %u MB pages\n",
1461 (unsigned int)(hpi->hugepage_sz / 0x100000));
1462 goto fail;
1463 }
1464 } else {
1465 /* set physical addresses for each hugepage */
1466 if (set_physaddrs(&tmp_hp[hp_offset], hpi) < 0) {
1467 RTE_LOG(DEBUG, EAL, "Failed to set phys addr "
1468 "for %u MB pages\n",
1469 (unsigned int)(hpi->hugepage_sz / 0x100000));
1470 goto fail;
1471 }
1472 }
1473
1474 if (find_numasocket(&tmp_hp[hp_offset], hpi) < 0){
1475 RTE_LOG(DEBUG, EAL, "Failed to find NUMA socket for %u MB pages\n",
1476 (unsigned)(hpi->hugepage_sz / 0x100000));
1477 goto fail;
1478 }
1479
1480 qsort(&tmp_hp[hp_offset], hpi->num_pages[0],
1481 sizeof(struct hugepage_file), cmp_physaddr);
1482
1483 /* we have processed a num of hugepages of this size, so inc offset */
1484 hp_offset += hpi->num_pages[0];
1485 }
1486
1487 huge_recover_sigbus();
1488
1489 if (internal_config.memory == 0 && internal_config.force_sockets == 0)
1490 internal_config.memory = eal_get_hugepage_mem_size();
1491
1492 nr_hugefiles = nr_hugepages;
1493
1494
1495 /* clean out the numbers of pages */
1496 for (i = 0; i < (int) internal_config.num_hugepage_sizes; i++)
1497 for (j = 0; j < RTE_MAX_NUMA_NODES; j++)
1498 internal_config.hugepage_info[i].num_pages[j] = 0;
1499
1500 /* get hugepages for each socket */
1501 for (i = 0; i < nr_hugefiles; i++) {
1502 int socket = tmp_hp[i].socket_id;
1503
1504 /* find a hugepage info with right size and increment num_pages */
1505 const int nb_hpsizes = RTE_MIN(MAX_HUGEPAGE_SIZES,
1506 (int)internal_config.num_hugepage_sizes);
1507 for (j = 0; j < nb_hpsizes; j++) {
1508 if (tmp_hp[i].size ==
1509 internal_config.hugepage_info[j].hugepage_sz) {
1510 internal_config.hugepage_info[j].num_pages[socket]++;
1511 }
1512 }
1513 }
1514
1515 /* make a copy of socket_mem, needed for number of pages calculation */
1516 for (i = 0; i < RTE_MAX_NUMA_NODES; i++)
1517 memory[i] = internal_config.socket_mem[i];
1518
1519 /* calculate final number of pages */
1520 nr_hugepages = calc_num_pages_per_socket(memory,
1521 internal_config.hugepage_info, used_hp,
1522 internal_config.num_hugepage_sizes);
1523
1524 /* error if not enough memory available */
1525 if (nr_hugepages < 0)
1526 goto fail;
1527
1528 /* reporting in! */
1529 for (i = 0; i < (int) internal_config.num_hugepage_sizes; i++) {
1530 for (j = 0; j < RTE_MAX_NUMA_NODES; j++) {
1531 if (used_hp[i].num_pages[j] > 0) {
1532 RTE_LOG(DEBUG, EAL,
1533 "Requesting %u pages of size %uMB"
1534 " from socket %i\n",
1535 used_hp[i].num_pages[j],
1536 (unsigned)
1537 (used_hp[i].hugepage_sz / 0x100000),
1538 j);
1539 }
1540 }
1541 }
1542
1543 /* create shared memory */
1544 hugepage = create_shared_memory(eal_hugepage_data_path(),
1545 nr_hugefiles * sizeof(struct hugepage_file));
1546
1547 if (hugepage == NULL) {
1548 RTE_LOG(ERR, EAL, "Failed to create shared memory!\n");
1549 goto fail;
1550 }
1551 memset(hugepage, 0, nr_hugefiles * sizeof(struct hugepage_file));
1552
1553 /*
1554 * unmap pages that we won't need (looks at used_hp).
1555 * also, sets final_va to NULL on pages that were unmapped.
1556 */
1557 if (unmap_unneeded_hugepages(tmp_hp, used_hp,
1558 internal_config.num_hugepage_sizes) < 0) {
1559 RTE_LOG(ERR, EAL, "Unmapping and locking hugepages failed!\n");
1560 goto fail;
1561 }
1562
1563 /*
1564 * copy stuff from malloc'd hugepage* to the actual shared memory.
1565 * this procedure only copies those hugepages that have orig_va
1566 * not NULL. has overflow protection.
1567 */
1568 if (copy_hugepages_to_shared_mem(hugepage, nr_hugefiles,
1569 tmp_hp, nr_hugefiles) < 0) {
1570 RTE_LOG(ERR, EAL, "Copying tables to shared memory failed!\n");
1571 goto fail;
1572 }
1573
1574#ifndef RTE_ARCH_64
1575 /* for legacy 32-bit mode, we did not preallocate VA space, so do it */
1576 if (internal_config.legacy_mem &&
1577 prealloc_segments(hugepage, nr_hugefiles)) {
1578 RTE_LOG(ERR, EAL, "Could not preallocate VA space for hugepages\n");
1579 goto fail;
1580 }
1581#endif
1582
1583 /* remap all pages we do need into memseg list VA space, so that those
1584 * pages become first-class citizens in DPDK memory subsystem
1585 */
1586 if (remap_needed_hugepages(hugepage, nr_hugefiles)) {
1587 RTE_LOG(ERR, EAL, "Couldn't remap hugepage files into memseg lists\n");
1588 goto fail;
1589 }
1590
1591 /* free the hugepage backing files */
1592 if (internal_config.hugepage_unlink &&
1593 unlink_hugepage_files(tmp_hp, internal_config.num_hugepage_sizes) < 0) {
1594 RTE_LOG(ERR, EAL, "Unlinking hugepage files failed!\n");
1595 goto fail;
1596 }
1597
1598 /* free the temporary hugepage table */
1599 free(tmp_hp);
1600 tmp_hp = NULL;
1601
1602 munmap(hugepage, nr_hugefiles * sizeof(struct hugepage_file));
1603
1604 /* we're not going to allocate more pages, so release VA space for
1605 * unused memseg lists
1606 */
1607 for (i = 0; i < RTE_MAX_MEMSEG_LISTS; i++) {
1608 struct rte_memseg_list *msl = &mcfg->memsegs[i];
1609 size_t mem_sz;
1610
1611 /* skip inactive lists */
1612 if (msl->base_va == NULL)
1613 continue;
1614 /* skip lists where there is at least one page allocated */
1615 if (msl->memseg_arr.count > 0)
1616 continue;
1617 /* this is an unused list, deallocate it */
1618 mem_sz = (size_t)msl->page_sz * msl->memseg_arr.len;
1619 munmap(msl->base_va, mem_sz);
1620 msl->base_va = NULL;
1621
1622 /* destroy backing fbarray */
1623 rte_fbarray_destroy(&msl->memseg_arr);
1624 }
1625
1626 return 0;
1627
1628fail:
1629 huge_recover_sigbus();
1630 free(tmp_hp);
1631 if (hugepage != NULL)
1632 munmap(hugepage, nr_hugefiles * sizeof(struct hugepage_file));
1633
1634 return -1;
1635}

1658static int
1659eal_hugepage_init(void)
1660{
1661 struct hugepage_info used_hp[MAX_HUGEPAGE_SIZES];
1662 uint64_t memory[RTE_MAX_NUMA_NODES];
1663 int hp_sz_idx, socket_id;
1664
1665 test_phys_addrs_available();
1666
1667 memset(used_hp, 0, sizeof(used_hp));
1668
1669 for (hp_sz_idx = 0;
1670 hp_sz_idx < (int) internal_config.num_hugepage_sizes;
1671 hp_sz_idx++) {
1672#ifndef RTE_ARCH_64
1673 struct hugepage_info dummy;
1674 unsigned int i;
1675#endif
1676 /* also initialize used_hp hugepage sizes in used_hp */
1677 struct hugepage_info *hpi;
1678 hpi = &internal_config.hugepage_info[hp_sz_idx];
1679 used_hp[hp_sz_idx].hugepage_sz = hpi->hugepage_sz;
1680
1681#ifndef RTE_ARCH_64
1682 /* for 32-bit, limit number of pages on socket to whatever we've
1683 * preallocated, as we cannot allocate more.
1684 */
1685 memset(&dummy, 0, sizeof(dummy));
1686 dummy.hugepage_sz = hpi->hugepage_sz;
1687 if (rte_memseg_list_walk(hugepage_count_walk, &dummy) < 0)
1688 return -1;
1689
1690 for (i = 0; i < RTE_DIM(dummy.num_pages); i++) {
1691 hpi->num_pages[i] = RTE_MIN(hpi->num_pages[i],
1692 dummy.num_pages[i]);
1693 }
1694#endif
1695 }
1696
1697 /* make a copy of socket_mem, needed for balanced allocation. */
1698 for (hp_sz_idx = 0; hp_sz_idx < RTE_MAX_NUMA_NODES; hp_sz_idx++)
1699 memory[hp_sz_idx] = internal_config.socket_mem[hp_sz_idx];
1700
1701 /* calculate final number of pages */
1702 if (calc_num_pages_per_socket(memory,
1703 internal_config.hugepage_info, used_hp,
1704 internal_config.num_hugepage_sizes) < 0)
1705 return -1;
1706
1707 for (hp_sz_idx = 0;
1708 hp_sz_idx < (int)internal_config.num_hugepage_sizes;
1709 hp_sz_idx++) {
1710 for (socket_id = 0; socket_id < RTE_MAX_NUMA_NODES;
1711 socket_id++) {
1712 struct rte_memseg **pages;
1713 struct hugepage_info *hpi = &used_hp[hp_sz_idx];
1714 unsigned int num_pages = hpi->num_pages[socket_id];
1715 int num_pages_alloc, i;
1716
1717 if (num_pages == 0)
1718 continue;
1719
1720 pages = malloc(sizeof(*pages) * num_pages);
1721
1722 RTE_LOG(DEBUG, EAL, "Allocating %u pages of size %" PRIu64 "M on socket %i\n",
1723 num_pages, hpi->hugepage_sz >> 20, socket_id);
1724
1725 num_pages_alloc = eal_memalloc_alloc_seg_bulk(pages,
1726 num_pages, hpi->hugepage_sz,
1727 socket_id, true);
1728 if (num_pages_alloc < 0) {
1729 free(pages);
1730 return -1;
1731 }
1732
1733 /* mark preallocated pages as unfreeable */
1734 for (i = 0; i < num_pages_alloc; i++) {
1735 struct rte_memseg *ms = pages[i];
1736 ms->flags |= RTE_MEMSEG_FLAG_DO_NOT_FREE;
1737 }
1738 free(pages);
1739 }
1740 }
1741 /* if socket limits were specified, set them */
1742 if (internal_config.force_socket_limits) {
1743 unsigned int i;
1744 for (i = 0; i < RTE_MAX_NUMA_NODES; i++) {
1745 uint64_t limit = internal_config.socket_limit[i];
1746 if (limit == 0)
1747 continue;
1748 if (rte_mem_alloc_validator_register("socket-limit",
1749 limits_callback, i, limit))
1750 RTE_LOG(ERR, EAL, "Failed to register socket limits validator callback\n");
1751 }
1752 }
1753 return 0;
1754}

1768/*
1769 * This creates the memory mappings in the secondary process to match that of
1770 * the server process. It goes through each memory segment in the DPDK runtime
1771 * configuration and finds the hugepages which form that segment, mapping them
1772 * in order to form a contiguous block in the virtual memory space
1773 */
1774static int
1775eal_legacy_hugepage_attach(void)
1776{
1777 struct rte_mem_config *mcfg = rte_eal_get_configuration()->mem_config;
1778 struct hugepage_file *hp = NULL;
1779 unsigned int num_hp = 0;
1780 unsigned int i = 0;
1781 unsigned int cur_seg;
1782 off_t size = 0;
1783 int fd, fd_hugepage = -1;
1784
1785 if (aslr_enabled() > 0) {
1786 RTE_LOG(WARNING, EAL, "WARNING: Address Space Layout Randomization "
1787 "(ASLR) is enabled in the kernel.\n");
1788 RTE_LOG(WARNING, EAL, " This may cause issues with mapping memory "
1789 "into secondary processes\n");
1790 }
1791
1792 test_phys_addrs_available();
1793
1794 fd_hugepage = open(eal_hugepage_data_path(), O_RDONLY);
1795 if (fd_hugepage < 0) {
1796 RTE_LOG(ERR, EAL, "Could not open %s\n",
1797 eal_hugepage_data_path());
1798 goto error;
1799 }
1800
1801 size = getFileSize(fd_hugepage);
1802 hp = mmap(NULL, size, PROT_READ, MAP_PRIVATE, fd_hugepage, 0);
1803 if (hp == MAP_FAILED) {
1804 RTE_LOG(ERR, EAL, "Could not mmap %s\n",
1805 eal_hugepage_data_path());
1806 goto error;
1807 }
1808
1809 num_hp = size / sizeof(struct hugepage_file);
1810 RTE_LOG(DEBUG, EAL, "Analysing %u files\n", num_hp);
1811
1812 /* map all segments into memory to make sure we get the addrs. the
1813 * segments themselves are already in memseg list (which is shared and
1814 * has its VA space already preallocated), so we just need to map
1815 * everything into correct addresses.
1816 */
1817 for (i = 0; i < num_hp; i++) {
1818 struct hugepage_file *hf = &hp[i];
1819 size_t map_sz = hf->size;
1820 void *map_addr = hf->final_va;
1821 int msl_idx, ms_idx;
1822 struct rte_memseg_list *msl;
1823 struct rte_memseg *ms;
1824
1825 /* if size is zero, no more pages left */
1826 if (map_sz == 0)
1827 break;
1828
1829 fd = open(hf->filepath, O_RDWR);
1830 if (fd < 0) {
1831 RTE_LOG(ERR, EAL, "Could not open %s: %s\n",
1832 hf->filepath, strerror(errno));
1833 goto error;
1834 }
1835
1836 map_addr = mmap(map_addr, map_sz, PROT_READ | PROT_WRITE,
1837 MAP_SHARED | MAP_FIXED, fd, 0);
1838 if (map_addr == MAP_FAILED) {
1839 RTE_LOG(ERR, EAL, "Could not map %s: %s\n",
1840 hf->filepath, strerror(errno));
1841 goto fd_error;
1842 }
1843
1844 /* set shared lock on the file. */
1845 if (flock(fd, LOCK_SH) < 0) {
1846 RTE_LOG(DEBUG, EAL, "%s(): Locking file failed: %s\n",
1847 __func__, strerror(errno));
1848 goto fd_error;
1849 }
1850
1851 /* find segment data */
1852 msl = rte_mem_virt2memseg_list(map_addr);
1853 if (msl == NULL) {
1854 RTE_LOG(DEBUG, EAL, "%s(): Cannot find memseg list\n",
1855 __func__);
1856 goto fd_error;
1857 }
1858 ms = rte_mem_virt2memseg(map_addr, msl);
1859 if (ms == NULL) {
1860 RTE_LOG(DEBUG, EAL, "%s(): Cannot find memseg\n",
1861 __func__);
1862 goto fd_error;
1863 }
1864
1865 msl_idx = msl - mcfg->memsegs;
1866 ms_idx = rte_fbarray_find_idx(&msl->memseg_arr, ms);
1867 if (ms_idx < 0) {
1868 RTE_LOG(DEBUG, EAL, "%s(): Cannot find memseg idx\n",
1869 __func__);
1870 goto fd_error;
1871 }
1872
1873 /* store segment fd internally */
1874 if (eal_memalloc_set_seg_fd(msl_idx, ms_idx, fd) < 0)
1875 RTE_LOG(ERR, EAL, "Could not store segment fd: %s\n",
1876 rte_strerror(rte_errno));
1877 }
1878 /* unmap the hugepage config file, since we are done using it */
1879 munmap(hp, size);
1880 close(fd_hugepage);
1881 return 0;
1882
1883fd_error:
1884 close(fd);
1885error:
1886 /* map all segments into memory to make sure we get the addrs */
1887 cur_seg = 0;
1888 for (cur_seg = 0; cur_seg < i; cur_seg++) {
1889 struct hugepage_file *hf = &hp[i];
1890 size_t map_sz = hf->size;
1891 void *map_addr = hf->final_va;
1892
1893 munmap(map_addr, map_sz);
1894 }
1895 if (hp != NULL && hp != MAP_FAILED)
1896 munmap(hp, size);
1897 if (fd_hugepage >= 0)
1898 close(fd_hugepage);
1899 return -1;
1900}

1936static int __rte_unused
1937memseg_primary_init_32(void)
1938{
1939 struct rte_mem_config *mcfg = rte_eal_get_configuration()->mem_config;
1940 int active_sockets, hpi_idx, msl_idx = 0;
1941 unsigned int socket_id, i;
1942 struct rte_memseg_list *msl;
1943 uint64_t extra_mem_per_socket, total_extra_mem, total_requested_mem;
1944 uint64_t max_mem;
1945
1946 /* no-huge does not need this at all */
1947 if (internal_config.no_hugetlbfs)
1948 return 0;
1949
1950 /* this is a giant hack, but desperate times call for desperate
1951 * measures. in legacy 32-bit mode, we cannot preallocate VA space,
1952 * because having upwards of 2 gigabytes of VA space already mapped will
1953 * interfere with our ability to map and sort hugepages.
1954 *
1955 * therefore, in legacy 32-bit mode, we will be initializing memseg
1956 * lists much later - in eal_memory.c, right after we unmap all the
1957 * unneeded pages. this will not affect secondary processes, as those
1958 * should be able to mmap the space without (too many) problems.
1959 */
1960 if (internal_config.legacy_mem)
1961 return 0;
1962
1963 /* 32-bit mode is a very special case. we cannot know in advance where
1964 * the user will want to allocate their memory, so we have to do some
1965 * heuristics.
1966 */
1967 active_sockets = 0;
1968 total_requested_mem = 0;
1969 if (internal_config.force_sockets)
1970 for (i = 0; i < rte_socket_count(); i++) {
1971 uint64_t mem;
1972
1973 socket_id = rte_socket_id_by_idx(i);
1974 mem = internal_config.socket_mem[socket_id];
1975
1976 if (mem == 0)
1977 continue;
1978
1979 active_sockets++;
1980 total_requested_mem += mem;
1981 }
1982 else
1983 total_requested_mem = internal_config.memory;
1984
1985 max_mem = (uint64_t)RTE_MAX_MEM_MB << 20;
1986 if (total_requested_mem > max_mem) {
1987 RTE_LOG(ERR, EAL, "Invalid parameters: 32-bit process can at most use %uM of memory\n",
1988 (unsigned int)(max_mem >> 20));
1989 return -1;
1990 }
1991 total_extra_mem = max_mem - total_requested_mem;
1992 extra_mem_per_socket = active_sockets == 0 ? total_extra_mem :
1993 total_extra_mem / active_sockets;
1994
1995 /* the allocation logic is a little bit convoluted, but here's how it
1996 * works, in a nutshell:
1997 * - if user hasn't specified on which sockets to allocate memory via
1998 * --socket-mem, we allocate all of our memory on master core socket.
1999 * - if user has specified sockets to allocate memory on, there may be
2000 * some "unused" memory left (e.g. if user has specified --socket-mem
2001 * such that not all memory adds up to 2 gigabytes), so add it to all
2002 * sockets that are in use equally.
2003 *
2004 * page sizes are sorted by size in descending order, so we can safely
2005 * assume that we dispense with bigger page sizes first.
2006 */
2007
2008 /* create memseg lists */
2009 for (i = 0; i < rte_socket_count(); i++) {
2010 int hp_sizes = (int) internal_config.num_hugepage_sizes;
2011 uint64_t max_socket_mem, cur_socket_mem;
2012 unsigned int master_lcore_socket;
2013 struct rte_config *cfg = rte_eal_get_configuration();
2014 bool skip;
2015
2016 socket_id = rte_socket_id_by_idx(i);
2017
2018#ifndef RTE_EAL_NUMA_AWARE_HUGEPAGES
2019 if (socket_id > 0)
2020 break;
2021#endif
2022
2023 /* if we didn't specifically request memory on this socket */
2024 skip = active_sockets != 0 &&
2025 internal_config.socket_mem[socket_id] == 0;
2026 /* ...or if we didn't specifically request memory on *any*
2027 * socket, and this is not master lcore
2028 */
2029 master_lcore_socket = rte_lcore_to_socket_id(cfg->master_lcore);
2030 skip |= active_sockets == 0 && socket_id != master_lcore_socket;
2031
2032 if (skip) {
2033 RTE_LOG(DEBUG, EAL, "Will not preallocate memory on socket %u\n",
2034 socket_id);
2035 continue;
2036 }
2037
2038 /* max amount of memory on this socket */
2039 max_socket_mem = (active_sockets != 0 ?
2040 internal_config.socket_mem[socket_id] :
2041 internal_config.memory) +
2042 extra_mem_per_socket;
2043 cur_socket_mem = 0;
2044
2045 for (hpi_idx = 0; hpi_idx < hp_sizes; hpi_idx++) {
2046 uint64_t max_pagesz_mem, cur_pagesz_mem = 0;
2047 uint64_t hugepage_sz;
2048 struct hugepage_info *hpi;
2049 int type_msl_idx, max_segs, total_segs = 0;
2050
2051 hpi = &internal_config.hugepage_info[hpi_idx];
2052 hugepage_sz = hpi->hugepage_sz;
2053
2054 /* check if pages are actually available */
2055 if (hpi->num_pages[socket_id] == 0)
2056 continue;
2057
2058 max_segs = RTE_MAX_MEMSEG_PER_TYPE;
2059 max_pagesz_mem = max_socket_mem - cur_socket_mem;
2060
2061 /* make it multiple of page size */
2062 max_pagesz_mem = RTE_ALIGN_FLOOR(max_pagesz_mem,
2063 hugepage_sz);
2064
2065 RTE_LOG(DEBUG, EAL, "Attempting to preallocate "
2066 "%" PRIu64 "M on socket %i\n",
2067 max_pagesz_mem >> 20, socket_id);
2068
2069 type_msl_idx = 0;
2070 while (cur_pagesz_mem < max_pagesz_mem &&
2071 total_segs < max_segs) {
2072 uint64_t cur_mem;
2073 unsigned int n_segs;
2074
2075 if (msl_idx >= RTE_MAX_MEMSEG_LISTS) {
2076 RTE_LOG(ERR, EAL,
2077 "No more space in memseg lists, please increase %s\n",
2078 RTE_STR(CONFIG_RTE_MAX_MEMSEG_LISTS));
2079 return -1;
2080 }
2081
2082 msl = &mcfg->memsegs[msl_idx];
2083
2084 cur_mem = get_mem_amount(hugepage_sz,
2085 max_pagesz_mem);
2086 n_segs = cur_mem / hugepage_sz;
2087
2088 if (alloc_memseg_list(msl, hugepage_sz, n_segs,
2089 socket_id, type_msl_idx)) {
2090 /* failing to allocate a memseg list is
2091 * a serious error.
2092 */
2093 RTE_LOG(ERR, EAL, "Cannot allocate memseg list\n");
2094 return -1;
2095 }
2096
2097 if (alloc_va_space(msl)) {
2098 /* if we couldn't allocate VA space, we
2099 * can try with smaller page sizes.
2100 */
2101 RTE_LOG(ERR, EAL, "Cannot allocate VA space for memseg list, retrying with different page size\n");
2102 /* deallocate memseg list */
2103 if (free_memseg_list(msl))
2104 return -1;
2105 break;
2106 }
2107
2108 total_segs += msl->memseg_arr.len;
2109 cur_pagesz_mem = total_segs * hugepage_sz;
2110 type_msl_idx++;
2111 msl_idx++;
2112 }
2113 cur_socket_mem += cur_pagesz_mem;
2114 }
2115 if (cur_socket_mem == 0) {
2116 RTE_LOG(ERR, EAL, "Cannot allocate VA space on socket %u\n",
2117 socket_id);
2118 return -1;
2119 }
2120 }
2121
2122 return 0;
2123}
