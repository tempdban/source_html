37/* DCB register definitions */
38
39#define IXGBE_DPMCS_MTSOS_SHIFT 16
40#define IXGBE_DPMCS_TDPAC 0x00000001 /* 0 Round Robin,
41 * 1 DFP - Deficit Fixed Priority */
42#define IXGBE_DPMCS_TRM 0x00000010 /* Transmit Recycle Mode */
43#define IXGBE_DPMCS_ARBDIS 0x00000040 /* DCB arbiter disable */
44#define IXGBE_DPMCS_TSOEF 0x00080000 /* TSO Expand Factor: 0=x4, 1=x2 */
45
46#define IXGBE_RUPPBMR_MQA 0x80000000 /* Enable UP to queue mapping */
47
48#define IXGBE_RT2CR_MCL_SHIFT 12 /* Offset to Max Credit Limit setting */
49#define IXGBE_RT2CR_LSP 0x80000000 /* LSP enable bit */
50
51#define IXGBE_RDRXCTL_MPBEN 0x00000010 /* DMA config for multiple packet
52 * buffers enable */
53#define IXGBE_RDRXCTL_MCEN 0x00000040 /* DMA config for multiple cores
54 * (RSS) enable */
55
56#define IXGBE_TDTQ2TCCR_MCL_SHIFT 12
57#define IXGBE_TDTQ2TCCR_BWG_SHIFT 9
58#define IXGBE_TDTQ2TCCR_GSP 0x40000000
59#define IXGBE_TDTQ2TCCR_LSP 0x80000000
60
61#define IXGBE_TDPT2TCCR_MCL_SHIFT 12
62#define IXGBE_TDPT2TCCR_BWG_SHIFT 9
63#define IXGBE_TDPT2TCCR_GSP 0x40000000
64#define IXGBE_TDPT2TCCR_LSP 0x80000000
65
66#define IXGBE_PDPMCS_TPPAC 0x00000020 /* 0 Round Robin,
67 * 1 DFP - Deficit Fixed Priority */
68#define IXGBE_PDPMCS_ARBDIS 0x00000040 /* Arbiter disable */
69#define IXGBE_PDPMCS_TRM 0x00000100 /* Transmit Recycle Mode enable */
70
71#define IXGBE_DTXCTL_ENDBUBD 0x00000004 /* Enable DBU buffer division */
72
73#define IXGBE_TXPBSIZE_40KB 0x0000A000 /* 40KB Packet Buffer */
74#define IXGBE_RXPBSIZE_48KB 0x0000C000 /* 48KB Packet Buffer */
75#define IXGBE_RXPBSIZE_64KB 0x00010000 /* 64KB Packet Buffer */
76#define IXGBE_RXPBSIZE_80KB 0x00014000 /* 80KB Packet Buffer */
