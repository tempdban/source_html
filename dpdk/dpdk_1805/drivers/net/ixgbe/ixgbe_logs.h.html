5#ifndef _IXGBE_LOGS_H_
6#define _IXGBE_LOGS_H_
7
8extern int ixgbe_logtype_init;
9#define PMD_INIT_LOG(level, fmt, args...) \
10 rte_log(RTE_LOG_ ## level, ixgbe_logtype_init, \
11 "%s(): " fmt "\n", __func__, ##args)
12
13#define PMD_INIT_FUNC_TRACE() PMD_INIT_LOG(DEBUG, " >>")
14
15#ifdef RTE_LIBRTE_IXGBE_DEBUG_RX
16#define PMD_RX_LOG(level, fmt, args...) \
17 RTE_LOG(level, PMD, "%s(): " fmt "\n", __func__, ## args)
18#else
19#define PMD_RX_LOG(level, fmt, args...) do { } while(0)
20#endif
21
22#ifdef RTE_LIBRTE_IXGBE_DEBUG_TX
23#define PMD_TX_LOG(level, fmt, args...) \
24 RTE_LOG(level, PMD, "%s(): " fmt "\n", __func__, ## args)
25#else
26#define PMD_TX_LOG(level, fmt, args...) do { } while(0)
27#endif
28
29#ifdef RTE_LIBRTE_IXGBE_DEBUG_TX_FREE
30#define PMD_TX_FREE_LOG(level, fmt, args...) \
31 RTE_LOG(level, PMD, "%s(): " fmt "\n", __func__, ## args)
32#else
33#define PMD_TX_FREE_LOG(level, fmt, args...) do { } while(0)
34#endif
35
36extern int ixgbe_logtype_driver;
37#define PMD_DRV_LOG_RAW(level, fmt, args...) \
38 rte_log(RTE_LOG_ ## level, ixgbe_logtype_driver, "%s(): " fmt, \
39 __func__, ## args)
40
41#define PMD_DRV_LOG(level, fmt, args...) \
42 PMD_DRV_LOG_RAW(level, fmt "\n", ## args)
43
44#endif /* _IXGBE_LOGS_H_ */
