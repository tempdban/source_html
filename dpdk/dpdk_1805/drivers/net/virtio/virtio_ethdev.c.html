144static int
145virtio_send_command(struct virtnet_ctl *cvq, struct virtio_pmd_ctrl *ctrl,
146 int *dlen, int pkt_num)
147{
148 uint32_t head, i;
149 int k, sum = 0;
150 virtio_net_ctrl_ack status = ~0;
151 struct virtio_pmd_ctrl *result;
152 struct virtqueue *vq;
153
154 ctrl->status = status;
155
156 if (!cvq || !cvq->vq) {
157 PMD_INIT_LOG(ERR, "Control queue is not supported.");
158 return -1;
159 }
160
161 rte_spinlock_lock(&cvq->lock);
162 vq = cvq->vq;
163 head = vq->vq_desc_head_idx;
164
165 PMD_INIT_LOG(DEBUG, "vq->vq_desc_head_idx = %d, status = %d, "
166 "vq->hw->cvq = %p vq = %p",
167 vq->vq_desc_head_idx, status, vq->hw->cvq, vq);
168
169 if (vq->vq_free_cnt < pkt_num + 2 || pkt_num < 1) {
170 rte_spinlock_unlock(&cvq->lock);
171 return -1;
172 }
173
174 memcpy(cvq->virtio_net_hdr_mz->addr, ctrl,
175 sizeof(struct virtio_pmd_ctrl));
176
177 /*
178 * Format is enforced in qemu code:
179 * One TX packet for header;
180 * At least one TX packet per argument;
181 * One RX packet for ACK.
182 */
183 vq->vq_ring.desc[head].flags = VRING_DESC_F_NEXT;
184 vq->vq_ring.desc[head].addr = cvq->virtio_net_hdr_mem;
185 vq->vq_ring.desc[head].len = sizeof(struct virtio_net_ctrl_hdr);
186 vq->vq_free_cnt--;
187 i = vq->vq_ring.desc[head].next;
188
189 for (k = 0; k < pkt_num; k++) {
190 vq->vq_ring.desc[i].flags = VRING_DESC_F_NEXT;
191 vq->vq_ring.desc[i].addr = cvq->virtio_net_hdr_mem
192 + sizeof(struct virtio_net_ctrl_hdr)
193 + sizeof(ctrl->status) + sizeof(uint8_t)*sum;
194 vq->vq_ring.desc[i].len = dlen[k];
195 sum += dlen[k];
196 vq->vq_free_cnt--;
197 i = vq->vq_ring.desc[i].next;
198 }
199
200 vq->vq_ring.desc[i].flags = VRING_DESC_F_WRITE;
201 vq->vq_ring.desc[i].addr = cvq->virtio_net_hdr_mem
202 + sizeof(struct virtio_net_ctrl_hdr);
203 vq->vq_ring.desc[i].len = sizeof(ctrl->status);
204 vq->vq_free_cnt--;
205
206 vq->vq_desc_head_idx = vq->vq_ring.desc[i].next;
207
208 vq_update_avail_ring(vq, head);
209 vq_update_avail_idx(vq);
210
211 PMD_INIT_LOG(DEBUG, "vq->vq_queue_index = %d", vq->vq_queue_index);
212
213 virtqueue_notify(vq);
214
215 rte_rmb();
216 while (VIRTQUEUE_NUSED(vq) == 0) {
217 rte_rmb();
218 usleep(100);
219 }
220
221 while (VIRTQUEUE_NUSED(vq)) {
222 uint32_t idx, desc_idx, used_idx;
223 struct vring_used_elem *uep;
224
225 used_idx = (uint32_t)(vq->vq_used_cons_idx
226 & (vq->vq_nentries - 1));
227 uep = &vq->vq_ring.used->ring[used_idx];
228 idx = (uint32_t) uep->id;
229 desc_idx = idx;
230
231 while (vq->vq_ring.desc[desc_idx].flags & VRING_DESC_F_NEXT) {
232 desc_idx = vq->vq_ring.desc[desc_idx].next;
233 vq->vq_free_cnt++;
234 }
235
236 vq->vq_ring.desc[desc_idx].next = vq->vq_desc_head_idx;
237 vq->vq_desc_head_idx = idx;
238
239 vq->vq_used_cons_idx++;
240 vq->vq_free_cnt++;
241 }
242
243 PMD_INIT_LOG(DEBUG, "vq->vq_free_cnt=%d\nvq->vq_desc_head_idx=%d",
244 vq->vq_free_cnt, vq->vq_desc_head_idx);
245
246 result = cvq->virtio_net_hdr_mz->addr;
247
248 rte_spinlock_unlock(&cvq->lock);
249 return result->status;
250}

252static int
253virtio_set_multiple_queues(struct rte_eth_dev *dev, uint16_t nb_queues)
254{
255 struct virtio_hw *hw = dev->data->dev_private;
256 struct virtio_pmd_ctrl ctrl;
257 int dlen[1];
258 int ret;
259
260 ctrl.hdr.class = VIRTIO_NET_CTRL_MQ;
261 ctrl.hdr.cmd = VIRTIO_NET_CTRL_MQ_VQ_PAIRS_SET;
262 memcpy(ctrl.data, &nb_queues, sizeof(uint16_t));
263
264 dlen[0] = sizeof(uint16_t);
265
266 ret = virtio_send_command(hw->cvq, &ctrl, dlen, 1);
267 if (ret) {
268 PMD_INIT_LOG(ERR, "Multiqueue configured but send command "
269 "failed, this is too late now...");
270 return -EINVAL;
271 }
272
273 return 0;
274}

293static void
294virtio_init_vring(struct virtqueue *vq)
295{
296 int size = vq->vq_nentries;
297 struct vring *vr = &vq->vq_ring;
298 uint8_t *ring_mem = vq->vq_ring_virt_mem;
299
300 PMD_INIT_FUNC_TRACE();
301
302 /*
303 * Reinitialise since virtio port might have been stopped and restarted
304 */
305 memset(ring_mem, 0, vq->vq_ring_size);
306 vring_init(vr, size, ring_mem, VIRTIO_PCI_VRING_ALIGN);
307 vq->vq_used_cons_idx = 0;
308 vq->vq_desc_head_idx = 0;
309 vq->vq_avail_idx = 0;
310 vq->vq_desc_tail_idx = (uint16_t)(vq->vq_nentries - 1);
311 vq->vq_free_cnt = vq->vq_nentries;
312 memset(vq->vq_descx, 0, sizeof(struct vq_desc_extra) * vq->vq_nentries);
313
314 vring_desc_init(vr->desc, size);
315
316 /*
317 * Disable device(host) interrupting guest
318 */
319 virtqueue_disable_intr(vq);
320}

322static int
323virtio_init_queue(struct rte_eth_dev *dev, uint16_t vtpci_queue_idx)
324{
325 char vq_name[VIRTQUEUE_MAX_NAME_SZ];
326 char vq_hdr_name[VIRTQUEUE_MAX_NAME_SZ];
327 const struct rte_memzone *mz = NULL, *hdr_mz = NULL;
328 unsigned int vq_size, size;
329 struct virtio_hw *hw = dev->data->dev_private;
330 struct virtnet_rx *rxvq = NULL;
331 struct virtnet_tx *txvq = NULL;
332 struct virtnet_ctl *cvq = NULL;
333 struct virtqueue *vq;
334 size_t sz_hdr_mz = 0;
335 void *sw_ring = NULL;
336 int queue_type = virtio_get_queue_type(hw, vtpci_queue_idx);
337 int ret;
338
339 PMD_INIT_LOG(DEBUG, "setting up queue: %u", vtpci_queue_idx);
340
341 /*
342 * Read the virtqueue size from the Queue Size field
343 * Always power of 2 and if 0 virtqueue does not exist
344 */
345 vq_size = VTPCI_OPS(hw)->get_queue_num(hw, vtpci_queue_idx);
346 PMD_INIT_LOG(DEBUG, "vq_size: %u", vq_size);
347 if (vq_size == 0) {
348 PMD_INIT_LOG(ERR, "virtqueue does not exist");
349 return -EINVAL;
350 }
351
352 if (!rte_is_power_of_2(vq_size)) {
353 PMD_INIT_LOG(ERR, "virtqueue size is not powerof 2");
354 return -EINVAL;
355 }
356
357 snprintf(vq_name, sizeof(vq_name), "port%d_vq%d",
358 dev->data->port_id, vtpci_queue_idx);
359
360 size = RTE_ALIGN_CEIL(sizeof(*vq) +
361 vq_size * sizeof(struct vq_desc_extra),
362 RTE_CACHE_LINE_SIZE);
363 if (queue_type == VTNET_TQ) {
364 /*
365 * For each xmit packet, allocate a virtio_net_hdr
366 * and indirect ring elements
367 */
368 sz_hdr_mz = vq_size * sizeof(struct virtio_tx_region);
369 } else if (queue_type == VTNET_CQ) {
370 /* Allocate a page for control vq command, data and status */
371 sz_hdr_mz = PAGE_SIZE;
372 }
373
374 vq = rte_zmalloc_socket(vq_name, size, RTE_CACHE_LINE_SIZE,
375 SOCKET_ID_ANY);
376 if (vq == NULL) {
377 PMD_INIT_LOG(ERR, "can not allocate vq");
378 return -ENOMEM;
379 }
380 hw->vqs[vtpci_queue_idx] = vq;
381
382 vq->hw = hw;
383 vq->vq_queue_index = vtpci_queue_idx;
384 vq->vq_nentries = vq_size;
385
386 /*
387 * Reserve a memzone for vring elements
388 */
389 size = vring_size(vq_size, VIRTIO_PCI_VRING_ALIGN);
390 vq->vq_ring_size = RTE_ALIGN_CEIL(size, VIRTIO_PCI_VRING_ALIGN);
391 PMD_INIT_LOG(DEBUG, "vring_size: %d, rounded_vring_size: %d",
392 size, vq->vq_ring_size);
393
394 mz = rte_memzone_reserve_aligned(vq_name, vq->vq_ring_size,
395 SOCKET_ID_ANY, RTE_MEMZONE_IOVA_CONTIG,
396 VIRTIO_PCI_VRING_ALIGN);
397 if (mz == NULL) {
398 if (rte_errno == EEXIST)
399 mz = rte_memzone_lookup(vq_name);
400 if (mz == NULL) {
401 ret = -ENOMEM;
402 goto fail_q_alloc;
403 }
404 }
405
406 memset(mz->addr, 0, mz->len);
407
408 vq->vq_ring_mem = mz->iova;
409 vq->vq_ring_virt_mem = mz->addr;
410 PMD_INIT_LOG(DEBUG, "vq->vq_ring_mem: 0x%" PRIx64,
411 (uint64_t)mz->iova);
412 PMD_INIT_LOG(DEBUG, "vq->vq_ring_virt_mem: 0x%" PRIx64,
413 (uint64_t)(uintptr_t)mz->addr);
414
415 virtio_init_vring(vq);
416
417 if (sz_hdr_mz) {
418 snprintf(vq_hdr_name, sizeof(vq_hdr_name), "port%d_vq%d_hdr",
419 dev->data->port_id, vtpci_queue_idx);
420 hdr_mz = rte_memzone_reserve_aligned(vq_hdr_name, sz_hdr_mz,
421 SOCKET_ID_ANY, RTE_MEMZONE_IOVA_CONTIG,
422 RTE_CACHE_LINE_SIZE);
423 if (hdr_mz == NULL) {
424 if (rte_errno == EEXIST)
425 hdr_mz = rte_memzone_lookup(vq_hdr_name);
426 if (hdr_mz == NULL) {
427 ret = -ENOMEM;
428 goto fail_q_alloc;
429 }
430 }
431 }
432
433 if (queue_type == VTNET_RQ) {
434 size_t sz_sw = (RTE_PMD_VIRTIO_RX_MAX_BURST + vq_size) *
435 sizeof(vq->sw_ring[0]);
436
437 sw_ring = rte_zmalloc_socket("sw_ring", sz_sw,
438 RTE_CACHE_LINE_SIZE, SOCKET_ID_ANY);
439 if (!sw_ring) {
440 PMD_INIT_LOG(ERR, "can not allocate RX soft ring");
441 ret = -ENOMEM;
442 goto fail_q_alloc;
443 }
444
445 vq->sw_ring = sw_ring;
446 rxvq = &vq->rxq;
447 rxvq->vq = vq;
448 rxvq->port_id = dev->data->port_id;
449 rxvq->mz = mz;
450 } else if (queue_type == VTNET_TQ) {
451 txvq = &vq->txq;
452 txvq->vq = vq;
453 txvq->port_id = dev->data->port_id;
454 txvq->mz = mz;
455 txvq->virtio_net_hdr_mz = hdr_mz;
456 txvq->virtio_net_hdr_mem = hdr_mz->iova;
457 } else if (queue_type == VTNET_CQ) {
458 cvq = &vq->cq;
459 cvq->vq = vq;
460 cvq->mz = mz;
461 cvq->virtio_net_hdr_mz = hdr_mz;
462 cvq->virtio_net_hdr_mem = hdr_mz->iova;
463 memset(cvq->virtio_net_hdr_mz->addr, 0, PAGE_SIZE);
464
465 hw->cvq = cvq;
466 }
467
468 /* For virtio_user case (that is when hw->dev is NULL), we use
469 * virtual address. And we need properly set _offset_, please see
470 * VIRTIO_MBUF_DATA_DMA_ADDR in virtqueue.h for more information.
471 */
472 if (!hw->virtio_user_dev)
473 vq->offset = offsetof(struct rte_mbuf, buf_iova);
474 else {
475 vq->vq_ring_mem = (uintptr_t)mz->addr;
476 vq->offset = offsetof(struct rte_mbuf, buf_addr);
477 if (queue_type == VTNET_TQ)
478 txvq->virtio_net_hdr_mem = (uintptr_t)hdr_mz->addr;
479 else if (queue_type == VTNET_CQ)
480 cvq->virtio_net_hdr_mem = (uintptr_t)hdr_mz->addr;
481 }
482
483 if (queue_type == VTNET_TQ) {
484 struct virtio_tx_region *txr;
485 unsigned int i;
486
487 txr = hdr_mz->addr;
488 memset(txr, 0, vq_size * sizeof(*txr));
489 for (i = 0; i < vq_size; i++) {
490 struct vring_desc *start_dp = txr[i].tx_indir;
491
492 vring_desc_init(start_dp, RTE_DIM(txr[i].tx_indir));
493
494 /* first indirect descriptor is always the tx header */
495 start_dp->addr = txvq->virtio_net_hdr_mem
496 + i * sizeof(*txr)
497 + offsetof(struct virtio_tx_region, tx_hdr);
498
499 start_dp->len = hw->vtnet_hdr_size;
500 start_dp->flags = VRING_DESC_F_NEXT;
501 }
502 }
503
504 if (VTPCI_OPS(hw)->setup_queue(hw, vq) < 0) {
505 PMD_INIT_LOG(ERR, "setup_queue failed");
506 return -EINVAL;
507 }
508
509 return 0;
510
511fail_q_alloc:
512 rte_free(sw_ring);
513 rte_memzone_free(hdr_mz);
514 rte_memzone_free(mz);
515 rte_free(vq);
516
517 return ret;
518}

520static void
521virtio_free_queues(struct virtio_hw *hw)
522{
523 uint16_t nr_vq = virtio_get_nr_vq(hw);
524 struct virtqueue *vq;
525 int queue_type;
526 uint16_t i;
527
528 if (hw->vqs == NULL)
529 return;
530
531 for (i = 0; i < nr_vq; i++) {
532 vq = hw->vqs[i];
533 if (!vq)
534 continue;
535
536 queue_type = virtio_get_queue_type(hw, i);
537 if (queue_type == VTNET_RQ) {
538 rte_free(vq->sw_ring);
539 rte_memzone_free(vq->rxq.mz);
540 } else if (queue_type == VTNET_TQ) {
541 rte_memzone_free(vq->txq.mz);
542 rte_memzone_free(vq->txq.virtio_net_hdr_mz);
543 } else {
544 rte_memzone_free(vq->cq.mz);
545 rte_memzone_free(vq->cq.virtio_net_hdr_mz);
546 }
547
548 rte_free(vq);
549 hw->vqs[i] = NULL;
550 }
551
552 rte_free(hw->vqs);
553 hw->vqs = NULL;
554}

556static int
557virtio_alloc_queues(struct rte_eth_dev *dev)
558{
559 struct virtio_hw *hw = dev->data->dev_private;
560 uint16_t nr_vq = virtio_get_nr_vq(hw);
561 uint16_t i;
562 int ret;
563
564 hw->vqs = rte_zmalloc(NULL, sizeof(struct virtqueue *) * nr_vq, 0);
565 if (!hw->vqs) {
566 PMD_INIT_LOG(ERR, "failed to allocate vqs");
567 return -ENOMEM;
568 }
569
570 for (i = 0; i < nr_vq; i++) {

1132static int
1133virtio_negotiate_features(struct virtio_hw *hw, uint64_t req_features)
1134{
1135 uint64_t host_features;
1136
1137 /* Prepare guest_features: feature that driver wants to support */
1138 PMD_INIT_LOG(DEBUG, "guest_features before negotiate = %" PRIx64,
1139 req_features);
1140
1141 /* Read device(host) feature bits */
1142 host_features = VTPCI_OPS(hw)->get_features(hw);
1143 PMD_INIT_LOG(DEBUG, "host_features before negotiate = %" PRIx64,
1144 host_features);
1145
1146 /* If supported, ensure MTU value is valid before acknowledging it. */
1147 if (host_features & req_features & (1ULL << VIRTIO_NET_F_MTU)) {
1148 struct virtio_net_config config;
1149
1150 vtpci_read_dev_config(hw,
1151 offsetof(struct virtio_net_config, mtu),
1152 &config.mtu, sizeof(config.mtu));
1153
1154 if (config.mtu < ETHER_MIN_MTU)
1155 req_features &= ~(1ULL << VIRTIO_NET_F_MTU);
1156 }
1157
1158 /*
1159 * Negotiate features: Subset of device feature bits are written back
1160 * guest feature bits.
1161 */
1162 hw->guest_features = req_features;
1163 hw->guest_features = vtpci_negotiate_features(hw, host_features);
1164 PMD_INIT_LOG(DEBUG, "features after negotiate = %" PRIx64,
1165 hw->guest_features);
1166
1167 if (hw->modern) {
1168 if (!vtpci_with_feature(hw, VIRTIO_F_VERSION_1)) {
1169 PMD_INIT_LOG(ERR,
1170 "VIRTIO_F_VERSION_1 features is not enabled.");
1171 return -1;
1172 }
1173 vtpci_set_status(hw, VIRTIO_CONFIG_STATUS_FEATURES_OK);
1174 if (!(vtpci_get_status(hw) & VIRTIO_CONFIG_STATUS_FEATURES_OK)) {
1175 PMD_INIT_LOG(ERR,
1176 "failed to set FEATURES_OK status!");
1177 return -1;
1178 }
1179 }
1180
1181 hw->req_guest_features = req_features;
1182
1183 return 0;
1184}

1186int
1187virtio_dev_pause(struct rte_eth_dev *dev)
1188{
1189 struct virtio_hw *hw = dev->data->dev_private;
1190
1191 rte_spinlock_lock(&hw->state_lock);
1192
1193 if (hw->started == 0) {
1194 /* Device is just stopped. */
1195 rte_spinlock_unlock(&hw->state_lock);
1196 return -1;
1197 }
1198 hw->started = 0;
1199 /*
1200 * Prevent the worker threads from touching queues to avoid contention,
1201 * 1 ms should be enough for the ongoing Tx function to finish.
1202 */
1203 rte_delay_ms(1);
1204 return 0;
1205}
1206
1207/*
1208 * Recover hw state to let the worker threads continue.

1237static void
1238virtio_notify_peers(struct rte_eth_dev *dev)
1239{
1240 struct virtio_hw *hw = dev->data->dev_private;
1241 struct virtnet_rx *rxvq;
1242 struct rte_mbuf *rarp_mbuf;
1243
1244 if (!dev->data->rx_queues)
1245 return;
1246
1247 rxvq = dev->data->rx_queues[0];
1248 if (!rxvq)
1249 return;
1250
1251 rarp_mbuf = rte_net_make_rarp_packet(rxvq->mpool,
1252 (struct ether_addr *)hw->mac_addr);
1253 if (rarp_mbuf == NULL) {
1254 PMD_DRV_LOG(ERR, "failed to make RARP packet.");
1255 return;
1256 }
1257
1258 /* If virtio port just stopped, no need to send RARP */
1259 if (virtio_dev_pause(dev) < 0) {
1260 rte_pktmbuf_free(rarp_mbuf);
1261 return;
1262 }
1263
1264 virtio_inject_pkts(dev, &rarp_mbuf, 1);
1265 virtio_dev_resume(dev);
1266}

1280/*
1281 * Process virtio config changed interrupt. Call the callback
1282 * if link state changed, generate gratuitous RARP packet if
1283 * the status indicates an ANNOUNCE.
1284 */
1285void
1286virtio_interrupt_handler(void *param)
1287{
1288 struct rte_eth_dev *dev = param;
1289 struct virtio_hw *hw = dev->data->dev_private;
1290 uint8_t isr;
1291
1292 /* Read interrupt status which clears interrupt */
1293 isr = vtpci_isr(hw);
1294 PMD_DRV_LOG(INFO, "interrupt status = %#x", isr);
1295
1296 if (virtio_intr_enable(dev) < 0)
1297 PMD_DRV_LOG(ERR, "interrupt enable failed");
1298
1299 if (isr & VIRTIO_PCI_ISR_CONFIG) {
1300 if (virtio_dev_link_update(dev, 0) == 0)
1301 _rte_eth_dev_callback_process(dev,
1302 RTE_ETH_EVENT_INTR_LSC,
1303 NULL);
1304 }
1305
1306 if (isr & VIRTIO_NET_S_ANNOUNCE) {
1307 virtio_notify_peers(dev);
1308 if (hw->cvq)
1309 virtio_ack_link_announce(dev);
1310 }
1311}

1345/* Only support 1:1 queue/interrupt mapping so far.
1346 * TODO: support n:1 queue/interrupt mapping when there are limited number of
1347 * interrupt vectors (<N+1).
1348 */
1349static int
1350virtio_queues_bind_intr(struct rte_eth_dev *dev)
1351{
1352 uint32_t i;
1353 struct virtio_hw *hw = dev->data->dev_private;
1354
1355 PMD_INIT_LOG(INFO, "queue/interrupt binding");
1356 for (i = 0; i < dev->data->nb_rx_queues; ++i) {
1357 dev->intr_handle->intr_vec[i] = i + 1;
1358 if (VTPCI_OPS(hw)->set_queue_irq(hw, hw->vqs[i * 2], i + 1) ==
1359 VIRTIO_MSI_NO_VECTOR) {
1360 PMD_DRV_LOG(ERR, "failed to set queue vector");
1361 return -EBUSY;
1362 }
1363 }
1364
1365 return 0;
1366}

1369virtio_queues_unbind_intr(struct rte_eth_dev *dev)
1370{
1371 uint32_t i;
1372 struct virtio_hw *hw = dev->data->dev_private;
1373
1374 PMD_INIT_LOG(INFO, "queue/interrupt unbinding");
1375 for (i = 0; i < dev->data->nb_rx_queues; ++i)

1377 hw->vqs[i * VTNET_CQ],

1381static int
1382virtio_configure_intr(struct rte_eth_dev *dev)
1383{
1384 struct virtio_hw *hw = dev->data->dev_private;
1385
1386 if (!rte_intr_cap_multiple(dev->intr_handle)) {
1387 PMD_INIT_LOG(ERR, "Multiple intr vector not supported");
1388 return -ENOTSUP;
1389 }
1390
1391 if (rte_intr_efd_enable(dev->intr_handle, dev->data->nb_rx_queues)) {
1392 PMD_INIT_LOG(ERR, "Fail to create eventfd");
1393 return -1;
1394 }
1395
1396 if (!dev->intr_handle->intr_vec) {
1397 dev->intr_handle->intr_vec =
1398 rte_zmalloc("intr_vec",
1399 hw->max_queue_pairs * sizeof(int), 0);
1400 if (!dev->intr_handle->intr_vec) {
1401 PMD_INIT_LOG(ERR, "Failed to allocate %u rxq vectors",
1402 hw->max_queue_pairs);
1403 return -ENOMEM;
1404 }
1405 }
1406
1407 /* Re-register callback to update max_intr */
1408 rte_intr_callback_unregister(dev->intr_handle,
1409 virtio_interrupt_handler,
1410 dev);
1411 rte_intr_callback_register(dev->intr_handle,
1412 virtio_interrupt_handler,
1413 dev);
1414
1415 /* DO NOT try to remove this! This function will enable msix, or QEMU
1416 * will encounter SIGSEGV when DRIVER_OK is sent.
1417 * And for legacy devices, this should be done before queue/vec binding
1418 * to change the config size from 20 to 24, or VIRTIO_MSI_QUEUE_VECTOR
1419 * (22) will be ignored.
1420 */
1421 if (virtio_intr_enable(dev) < 0) {
1422 PMD_DRV_LOG(ERR, "interrupt enable failed");
1423 return -1;
1424 }
1425
1426 if (virtio_queues_bind_intr(dev) < 0) {
1427 PMD_INIT_LOG(ERR, "Failed to bind queue/interrupt");
1428 return -1;
1429 }
1430
1431 return 0;
1432}

1434/* reset device and renegotiate features if needed */
1435static int
1436virtio_init_device(struct rte_eth_dev *eth_dev, uint64_t req_features)
1437{
1438 struct virtio_hw *hw = eth_dev->data->dev_private;
1439 struct virtio_net_config *config;
1440 struct virtio_net_config local_config;
1441 struct rte_pci_device *pci_dev = NULL;
1442 int ret;
1443
1444 /* Reset the device although not necessary at startup */
1445 vtpci_reset(hw);
1446
1447 if (hw->vqs) {
1448 virtio_dev_free_mbufs(eth_dev);
1449 virtio_free_queues(hw);
1450 }
1451
1452 /* Tell the host we've noticed this device. */
1453 vtpci_set_status(hw, VIRTIO_CONFIG_STATUS_ACK);
1454
1455 /* Tell the host we've known how to drive the device. */
1456 vtpci_set_status(hw, VIRTIO_CONFIG_STATUS_DRIVER);
1457 if (virtio_negotiate_features(hw, req_features) < 0)
1458 return -1;
1459
1460 if (!hw->virtio_user_dev) {
1461 pci_dev = RTE_ETH_DEV_TO_PCI(eth_dev);
1462 rte_eth_copy_pci_info(eth_dev, pci_dev);
1463 }
1464
1465 /* If host does not support both status and MSI-X then disable LSC */
1466 if (vtpci_with_feature(hw, VIRTIO_NET_F_STATUS) &&
1467 hw->use_msix != VIRTIO_MSIX_NONE)
1468 eth_dev->data->dev_flags |= RTE_ETH_DEV_INTR_LSC;
1469 else
1470 eth_dev->data->dev_flags &= ~RTE_ETH_DEV_INTR_LSC;
1471
1472 /* Setting up rx_header size for the device */
1473 if (vtpci_with_feature(hw, VIRTIO_NET_F_MRG_RXBUF) ||
1474 vtpci_with_feature(hw, VIRTIO_F_VERSION_1))
1475 hw->vtnet_hdr_size = sizeof(struct virtio_net_hdr_mrg_rxbuf);
1476 else
1477 hw->vtnet_hdr_size = sizeof(struct virtio_net_hdr);
1478
1479 /* Copy the permanent MAC address to: virtio_hw */
1480 virtio_get_hwaddr(hw);
1481 ether_addr_copy((struct ether_addr *) hw->mac_addr,
1482 &eth_dev->data->mac_addrs[0]);
1483 PMD_INIT_LOG(DEBUG,
1484 "PORT MAC: %02X:%02X:%02X:%02X:%02X:%02X",
1485 hw->mac_addr[0], hw->mac_addr[1], hw->mac_addr[2],
1486 hw->mac_addr[3], hw->mac_addr[4], hw->mac_addr[5]);
1487
1488 if (vtpci_with_feature(hw, VIRTIO_NET_F_CTRL_VQ)) {
1489 config = &local_config;
1490
1491 vtpci_read_dev_config(hw,
1492 offsetof(struct virtio_net_config, mac),
1493 &config->mac, sizeof(config->mac));
1494
1495 if (vtpci_with_feature(hw, VIRTIO_NET_F_STATUS)) {
1496 vtpci_read_dev_config(hw,
1497 offsetof(struct virtio_net_config, status),
1498 &config->status, sizeof(config->status));
1499 } else {
1500 PMD_INIT_LOG(DEBUG,
1501 "VIRTIO_NET_F_STATUS is not supported");
1502 config->status = 0;
1503 }
1504
1505 if (vtpci_with_feature(hw, VIRTIO_NET_F_MQ)) {
1506 vtpci_read_dev_config(hw,
1507 offsetof(struct virtio_net_config, max_virtqueue_pairs),
1508 &config->max_virtqueue_pairs,
1509 sizeof(config->max_virtqueue_pairs));
1510 } else {
1511 PMD_INIT_LOG(DEBUG,
1512 "VIRTIO_NET_F_MQ is not supported");
1513 config->max_virtqueue_pairs = 1;
1514 }
1515
1516 hw->max_queue_pairs = config->max_virtqueue_pairs;
1517
1518 if (vtpci_with_feature(hw, VIRTIO_NET_F_MTU)) {
1519 vtpci_read_dev_config(hw,
1520 offsetof(struct virtio_net_config, mtu),
1521 &config->mtu,
1522 sizeof(config->mtu));
1523
1524 /*
1525 * MTU value has already been checked at negotiation
1526 * time, but check again in case it has changed since
1527 * then, which should not happen.
1528 */
1529 if (config->mtu < ETHER_MIN_MTU) {
1530 PMD_INIT_LOG(ERR, "invalid max MTU value (%u)",
1531 config->mtu);
1532 return -1;
1533 }
1534
1535 hw->max_mtu = config->mtu;
1536 /* Set initial MTU to maximum one supported by vhost */
1537 eth_dev->data->mtu = config->mtu;
1538
1539 } else {
1540 hw->max_mtu = VIRTIO_MAX_RX_PKTLEN - ETHER_HDR_LEN -
1541 VLAN_TAG_LEN - hw->vtnet_hdr_size;
1542 }
1543
1544 PMD_INIT_LOG(DEBUG, "config->max_virtqueue_pairs=%d",
1545 config->max_virtqueue_pairs);
1546 PMD_INIT_LOG(DEBUG, "config->status=%d", config->status);
1547 PMD_INIT_LOG(DEBUG,
1548 "PORT MAC: %02X:%02X:%02X:%02X:%02X:%02X",
1549 config->mac[0], config->mac[1],
1550 config->mac[2], config->mac[3],
1551 config->mac[4], config->mac[5]);
1552 } else {
1553 PMD_INIT_LOG(DEBUG, "config->max_virtqueue_pairs=1");
1554 hw->max_queue_pairs = 1;
1555 hw->max_mtu = VIRTIO_MAX_RX_PKTLEN - ETHER_HDR_LEN -
1556 VLAN_TAG_LEN - hw->vtnet_hdr_size;
1557 }
1558
1559 ret = virtio_alloc_queues(eth_dev);
1560 if (ret < 0)
1561 return ret;
1562
1563 if (eth_dev->data->dev_conf.intr_conf.rxq) {
1564 if (virtio_configure_intr(eth_dev) < 0) {
1565 PMD_INIT_LOG(ERR, "failed to configure interrupt");
1566 return -1;
1567 }
1568 }
1569
1570 vtpci_reinit_complete(hw);
1571
1572 if (pci_dev)
1573 PMD_INIT_LOG(DEBUG, "port %d vendorID=0x%x deviceID=0x%x",
1574 eth_dev->data->port_id, pci_dev->id.vendor_id,
1575 pci_dev->id.device_id);
1576
1577 return 0;
1578}

1580/*
1581 * Remap the PCI device again (IO port map for legacy device and
1582 * memory map for modern device), so that the secondary process
1583 * could have the PCI initiated correctly.
1584 */
1585static int
1586virtio_remap_pci(struct rte_pci_device *pci_dev, struct virtio_hw *hw)
1587{
1588 if (hw->modern) {
1589 /*
1590 * We don't have to re-parse the PCI config space, since
1591 * rte_pci_map_device() makes sure the mapped address
1592 * in secondary process would equal to the one mapped in
1593 * the primary process: error will be returned if that
1594 * requirement is not met.
1595 *
1596 * That said, we could simply reuse all cap pointers
1597 * (such as dev_cfg, common_cfg, etc.) parsed from the
1598 * primary process, which is stored in shared memory.
1599 */
1600 if (rte_pci_map_device(pci_dev)) {
1601 PMD_INIT_LOG(DEBUG, "failed to map pci device!");
1602 return -1;
1603 }
1604 } else {
1605 if (rte_pci_ioport_map(pci_dev, 0, VTPCI_IO(hw)) < 0)
1606 return -1;
1607 }
1608
1609 return 0;
1610}

1626/*
1627 * This function is based on probe() function in virtio_pci.c
1628 * It returns 0 on success.
1629 */
1630int
1631eth_virtio_dev_init(struct rte_eth_dev *eth_dev)
1632{
1633 struct virtio_hw *hw = eth_dev->data->dev_private;
1634 int ret;
1635
1636 RTE_BUILD_BUG_ON(RTE_PKTMBUF_HEADROOM < sizeof(struct virtio_net_hdr_mrg_rxbuf));
1637
1638 eth_dev->dev_ops = &virtio_eth_dev_ops;
1639
1640 if (rte_eal_process_type() == RTE_PROC_SECONDARY) {
1641 if (!hw->virtio_user_dev) {
1642 ret = virtio_remap_pci(RTE_ETH_DEV_TO_PCI(eth_dev), hw);
1643 if (ret)
1644 return ret;
1645 }
1646
1647 virtio_set_vtpci_ops(hw);
1648 set_rxtx_funcs(eth_dev);
1649
1650 return 0;
1651 }
1652
1653 /* Allocate memory for storing MAC addresses */
1654 eth_dev->data->mac_addrs = rte_zmalloc("virtio", VIRTIO_MAX_MAC_ADDRS * ETHER_ADDR_LEN, 0);
1655 if (eth_dev->data->mac_addrs == NULL) {
1656 PMD_INIT_LOG(ERR,
1657 "Failed to allocate %d bytes needed to store MAC addresses",
1658 VIRTIO_MAX_MAC_ADDRS * ETHER_ADDR_LEN);
1659 return -ENOMEM;
1660 }
1661
1662 hw->port_id = eth_dev->data->port_id;
1663 /* For virtio_user case the hw->virtio_user_dev is populated by
1664 * virtio_user_eth_dev_alloc() before eth_virtio_dev_init() is called.
1665 */
1666 if (!hw->virtio_user_dev) {
1667 ret = vtpci_init(RTE_ETH_DEV_TO_PCI(eth_dev), hw);
1668 if (ret)
1669 goto out;
1670 }
1671
1672 /* reset device and negotiate default features */
1673 ret = virtio_init_device(eth_dev, VIRTIO_PMD_DEFAULT_GUEST_FEATURES);
1674 if (ret < 0)
1675 goto out;
1676
1677 /* Setup interrupt callback */
1678 if (eth_dev->data->dev_flags & RTE_ETH_DEV_INTR_LSC)
1679 rte_intr_callback_register(eth_dev->intr_handle,
1680 virtio_interrupt_handler, eth_dev);
1681
1682 return 0;
1683
1684out:
1685 rte_free(eth_dev->data->mac_addrs);
1686 return ret;
1687}

1796/*
1797 * Configure virtio device
1798 * It returns 0 on success.
1799 */
1800static int
1801virtio_dev_configure(struct rte_eth_dev *dev)
1802{
1803 const struct rte_eth_rxmode *rxmode = &dev->data->dev_conf.rxmode;
1804 struct virtio_hw *hw = dev->data->dev_private;
1805 uint64_t rx_offloads = rxmode->offloads;
1806 uint64_t req_features;
1807 int ret;
1808
1809 PMD_INIT_LOG(DEBUG, "configure");
1810 req_features = VIRTIO_PMD_DEFAULT_GUEST_FEATURES;
1811
1812 if (dev->data->dev_conf.intr_conf.rxq) {
1813 ret = virtio_init_device(dev, hw->req_guest_features);
1814 if (ret < 0)
1815 return ret;
1816 }
1817
1818 if (rx_offloads & (DEV_RX_OFFLOAD_UDP_CKSUM |
1819 DEV_RX_OFFLOAD_TCP_CKSUM))
1820 req_features |= (1ULL << VIRTIO_NET_F_GUEST_CSUM);
1821
1822 if (rx_offloads & DEV_RX_OFFLOAD_TCP_LRO)
1823 req_features |=
1824 (1ULL << VIRTIO_NET_F_GUEST_TSO4) |
1825 (1ULL << VIRTIO_NET_F_GUEST_TSO6);
1826
1827 /* if request features changed, reinit the device */
1828 if (req_features != hw->req_guest_features) {
1829 ret = virtio_init_device(dev, req_features);
1830 if (ret < 0)
1831 return ret;
1832 }
1833
1834 if ((rx_offloads & (DEV_RX_OFFLOAD_UDP_CKSUM |
1835 DEV_RX_OFFLOAD_TCP_CKSUM)) &&
1836 !vtpci_with_feature(hw, VIRTIO_NET_F_GUEST_CSUM)) {
1837 PMD_DRV_LOG(ERR,
1838 "rx checksum not available on this host");
1839 return -ENOTSUP;
1840 }
1841
1842 if ((rx_offloads & DEV_RX_OFFLOAD_TCP_LRO) &&
1843 (!vtpci_with_feature(hw, VIRTIO_NET_F_GUEST_TSO4) ||
1844 !vtpci_with_feature(hw, VIRTIO_NET_F_GUEST_TSO6))) {
1845 PMD_DRV_LOG(ERR,
1846 "Large Receive Offload not available on this host");
1847 return -ENOTSUP;
1848 }
1849
1850 /* start control queue */
1851 if (vtpci_with_feature(hw, VIRTIO_NET_F_CTRL_VQ))
1852 virtio_dev_cq_start(dev);
1853
1854 if (rx_offloads & DEV_RX_OFFLOAD_VLAN_STRIP)
1855 hw->vlan_strip = 1;
1856
1857 if ((rx_offloads & DEV_RX_OFFLOAD_VLAN_FILTER)
1858 && !vtpci_with_feature(hw, VIRTIO_NET_F_CTRL_VLAN)) {
1859 PMD_DRV_LOG(ERR,
1860 "vlan filtering not available on this host");
1861 return -ENOTSUP;
1862 }
1863
1864 if (dev->data->dev_flags & RTE_ETH_DEV_INTR_LSC)
1865 /* Enable vector (0) for Link State Intrerrupt */
1866 if (VTPCI_OPS(hw)->set_config_irq(hw, 0) ==
1867 VIRTIO_MSI_NO_VECTOR) {
1868 PMD_DRV_LOG(ERR, "failed to set config vector");
1869 return -EBUSY;
1870 }
1871
1872 rte_spinlock_init(&hw->state_lock);
1873
1874 hw->use_simple_rx = 1;
1875 hw->use_simple_tx = 1;
1876
1877#if defined RTE_ARCH_ARM64 || defined RTE_ARCH_ARM
1878 if (!rte_cpu_get_flag_enabled(RTE_CPUFLAG_NEON)) {
1879 hw->use_simple_rx = 0;
1880 hw->use_simple_tx = 0;
1881 }
1882#endif
1883 if (vtpci_with_feature(hw, VIRTIO_NET_F_MRG_RXBUF)) {
1884 hw->use_simple_rx = 0;
1885 hw->use_simple_tx = 0;
1886 }
1887
1888 if (rx_offloads & (DEV_RX_OFFLOAD_UDP_CKSUM |
1889 DEV_RX_OFFLOAD_TCP_CKSUM))
1890 hw->use_simple_rx = 0;
1891
1892 return 0;
1893}

1896static int
1897virtio_dev_start(struct rte_eth_dev *dev)
1898{
1899 uint16_t nb_queues, i;
1900 struct virtnet_rx *rxvq;
1901 struct virtnet_tx *txvq __rte_unused;
1902 struct virtio_hw *hw = dev->data->dev_private;
1903 int ret;
1904
1905 /* Finish the initialization of the queues */
1906 for (i = 0; i < dev->data->nb_rx_queues; i++) {
1907 ret = virtio_dev_rx_queue_setup_finish(dev, i);
1908 if (ret < 0)
1909 return ret;
1910 }
1911 for (i = 0; i < dev->data->nb_tx_queues; i++) {
1912 ret = virtio_dev_tx_queue_setup_finish(dev, i);
1913 if (ret < 0)
1914 return ret;
1915 }
1916
1917 /* check if lsc interrupt feature is enabled */
1918 if (dev->data->dev_conf.intr_conf.lsc) {
1919 if (!(dev->data->dev_flags & RTE_ETH_DEV_INTR_LSC)) {
1920 PMD_DRV_LOG(ERR, "link status not supported by host");
1921 return -ENOTSUP;
1922 }
1923 }
1924
1925 /* Enable uio/vfio intr/eventfd mapping: althrough we already did that
1926 * in device configure, but it could be unmapped when device is
1927 * stopped.
1928 */
1929 if (dev->data->dev_conf.intr_conf.lsc ||
1930 dev->data->dev_conf.intr_conf.rxq) {
1931 virtio_intr_disable(dev);
1932
1933 if (virtio_intr_enable(dev) < 0) {
1934 PMD_DRV_LOG(ERR, "interrupt enable failed");
1935 return -EIO;
1936 }
1937 }
1938
1939 /*Notify the backend
1940 *Otherwise the tap backend might already stop its queue due to fullness.
1941 *vhost backend will have no chance to be waked up
1942 */
1943 nb_queues = RTE_MAX(dev->data->nb_rx_queues, dev->data->nb_tx_queues);
1944 if (hw->max_queue_pairs > 1) {
1945 if (virtio_set_multiple_queues(dev, nb_queues) != 0)
1946 return -EINVAL;
1947 }
1948
1949 PMD_INIT_LOG(DEBUG, "nb_queues=%d", nb_queues);
1950
1951 for (i = 0; i < dev->data->nb_rx_queues; i++) {
1952 rxvq = dev->data->rx_queues[i];
1953 /* Flush the old packets */
1954 virtqueue_rxvq_flush(rxvq->vq);
1955 virtqueue_notify(rxvq->vq);
1956 }
1957
1958 for (i = 0; i < dev->data->nb_tx_queues; i++) {
1959 txvq = dev->data->tx_queues[i];
1960 virtqueue_notify(txvq->vq);
1961 }
1962
1963 PMD_INIT_LOG(DEBUG, "Notified backend at initialization");
1964
1965 for (i = 0; i < dev->data->nb_rx_queues; i++) {
1966 rxvq = dev->data->rx_queues[i];
1967 VIRTQUEUE_DUMP(rxvq->vq);
1968 }
1969
1970 for (i = 0; i < dev->data->nb_tx_queues; i++) {
1971 txvq = dev->data->tx_queues[i];
1972 VIRTQUEUE_DUMP(txvq->vq);
1973 }
1974
1975 set_rxtx_funcs(dev);
1976 hw->started = 1;
1977
1978 /* Initialize Link state */
1979 virtio_dev_link_update(dev, 0);
1980
1981 return 0;
1982}

1984static void virtio_dev_free_mbufs(struct rte_eth_dev *dev)
1985{
1986 struct virtio_hw *hw = dev->data->dev_private;
1987 uint16_t nr_vq = virtio_get_nr_vq(hw);
1988 const char *type __rte_unused;
1989 unsigned int i, mbuf_num = 0;
1990 struct virtqueue *vq;
1991 struct rte_mbuf *buf;
1992 int queue_type;
1993
1994 if (hw->vqs == NULL)
1995 return;
1996
1997 for (i = 0; i < nr_vq; i++) {
1998 vq = hw->vqs[i];
1999 if (!vq)
2000 continue;
2001
2002 queue_type = virtio_get_queue_type(hw, i);
2003 if (queue_type == VTNET_RQ)
2004 type = "rxq";
2005 else if (queue_type == VTNET_TQ)
2006 type = "txq";
2007 else
2008 continue;
2009
2010 PMD_INIT_LOG(DEBUG,
2011 "Before freeing %s[%d] used and unused buf",
2012 type, i);
2013 VIRTQUEUE_DUMP(vq);
2014
2015 while ((buf = virtqueue_detach_unused(vq)) != NULL) {
2016 rte_pktmbuf_free(buf);
2017 mbuf_num++;
2018 }
2019
2020 PMD_INIT_LOG(DEBUG,
2021 "After freeing %s[%d] used and unused buf",
2022 type, i);
2023 VIRTQUEUE_DUMP(vq);
2024 }
2025
2026 PMD_INIT_LOG(DEBUG, "%d mbufs freed", mbuf_num);
2027}
