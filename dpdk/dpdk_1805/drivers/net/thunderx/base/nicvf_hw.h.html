111static inline void
112nicvf_reg_write(struct nicvf *nic, uint32_t offset, uint64_t val)
113{
114 nicvf_addr_write(nic->reg_base + offset, val);
115}
116
117static inline uint64_t
118nicvf_reg_read(struct nicvf *nic, uint32_t offset)
119{
120 return nicvf_addr_read(nic->reg_base + offset);
121}
122
123static inline uintptr_t
124nicvf_qset_base(struct nicvf *nic, uint32_t qidx)
125{
126 return nic->reg_base + (qidx << NIC_Q_NUM_SHIFT);
127}
128
129static inline void
130nicvf_queue_reg_write(struct nicvf *nic, uint32_t offset, uint32_t qidx,
131 uint64_t val)
132{
133 nicvf_addr_write(nicvf_qset_base(nic, qidx) + offset, val);
134}
135
136static inline uint64_t
137nicvf_queue_reg_read(struct nicvf *nic, uint32_t offset, uint32_t qidx)
138{
139 return nicvf_addr_read(nicvf_qset_base(nic, qidx) + offset);
140}

142static inline void
143nicvf_disable_all_interrupts(struct nicvf *nic)
144{
145 nicvf_reg_write(nic, NIC_VF_ENA_W1C, NICVF_INTR_ALL_MASK);
146 nicvf_reg_write(nic, NIC_VF_INT, NICVF_INTR_ALL_MASK);
147}
