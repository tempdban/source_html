13/* PF <--> VF Mailbox communication
14 * Two 64bit registers are shared between PF and VF for each VF
15 * Writing into second register means end of message.
16 */
17
18/* PF <--> VF mailbox communication */
19#define NIC_PF_VF_MAILBOX_SIZE 2
20#define NIC_MBOX_MSG_TIMEOUT 2000 /* ms */
21
22/* Mailbox message types */
23#define NIC_MBOX_MSG_INVALID 0x00 /* Invalid message */
24#define NIC_MBOX_MSG_READY 0x01 /* Is PF ready to rcv msgs */
25#define NIC_MBOX_MSG_ACK 0x02 /* ACK the message received */
26#define NIC_MBOX_MSG_NACK 0x03 /* NACK the message received */
27#define NIC_MBOX_MSG_QS_CFG 0x04 /* Configure Qset */
28#define NIC_MBOX_MSG_RQ_CFG 0x05 /* Configure receive queue */
29#define NIC_MBOX_MSG_SQ_CFG 0x06 /* Configure Send queue */
30#define NIC_MBOX_MSG_RQ_DROP_CFG 0x07 /* Configure receive queue */
31#define NIC_MBOX_MSG_SET_MAC 0x08 /* Add MAC ID to DMAC filter */
32#define NIC_MBOX_MSG_SET_MAX_FRS 0x09 /* Set max frame size */
33#define NIC_MBOX_MSG_CPI_CFG 0x0A /* Config CPI, RSSI */
34#define NIC_MBOX_MSG_RSS_SIZE 0x0B /* Get RSS indir_tbl size */
35#define NIC_MBOX_MSG_RSS_CFG 0x0C /* Config RSS table */
36#define NIC_MBOX_MSG_RSS_CFG_CONT 0x0D /* RSS config continuation */
37#define NIC_MBOX_MSG_RQ_BP_CFG 0x0E /* RQ backpressure config */
38#define NIC_MBOX_MSG_RQ_SW_SYNC 0x0F /* Flush inflight pkts to RQ */
39#define NIC_MBOX_MSG_BGX_LINK_CHANGE 0x11 /* BGX:LMAC link status */
40#define NIC_MBOX_MSG_ALLOC_SQS 0x12 /* Allocate secondary Qset */
41#define NIC_MBOX_MSG_LOOPBACK 0x16 /* Set interface in loopback */
42#define NIC_MBOX_MSG_RESET_STAT_COUNTER 0x17 /* Reset statistics counters */
43#define NIC_MBOX_MSG_CFG_DONE 0xF0 /* VF configuration done */
44#define NIC_MBOX_MSG_SHUTDOWN 0xF1 /* VF is being shutdown */
45#define NIC_MBOX_MSG_MAX 0x100 /* Maximum number of messages */
46
47/* Get vNIC VF configuration */
48struct nic_cfg_msg {
49 uint8_t msg;
50 uint8_t vf_id;
51 uint8_t node_id;
52 bool tns_mode:1;
53 bool sqs_mode:1;
54 bool loopback_supported:1;
55 uint8_t mac_addr[NICVF_MAC_ADDR_SIZE];
56};
57
58/* Qset configuration */
59struct qs_cfg_msg {
60 uint8_t msg;
61 uint8_t num;
62 uint8_t sqs_count;
63 uint64_t cfg;
64};
65
66/* Receive queue configuration */
67struct rq_cfg_msg {
68 uint8_t msg;
69 uint8_t qs_num;
70 uint8_t rq_num;
71 uint64_t cfg;
72};
73
74/* Send queue configuration */
75struct sq_cfg_msg {
76 uint8_t msg;
77 uint8_t qs_num;
78 uint8_t sq_num;
79 bool sqs_mode;
80 uint64_t cfg;
81};
82
83/* Set VF's MAC address */
84struct set_mac_msg {
85 uint8_t msg;
86 uint8_t vf_id;
87 uint8_t mac_addr[NICVF_MAC_ADDR_SIZE];
88};
89
90/* Set Maximum frame size */
91struct set_frs_msg {
92 uint8_t msg;
93 uint8_t vf_id;
94 uint16_t max_frs;
95};
96
97/* Set CPI algorithm type */
98struct cpi_cfg_msg {
99 uint8_t msg;
100 uint8_t vf_id;
101 uint8_t rq_cnt;
102 uint8_t cpi_alg;
103};
104
105/* Get RSS table size */
106struct rss_sz_msg {
107 uint8_t msg;
108 uint8_t vf_id;
109 uint16_t ind_tbl_size;
110};
111
112/* Set RSS configuration */
113struct rss_cfg_msg {
114 uint8_t msg;
115 uint8_t vf_id;
116 uint8_t hash_bits;
117 uint8_t tbl_len;
118 uint8_t tbl_offset;
119#define RSS_IND_TBL_LEN_PER_MBX_MSG 8
120 uint8_t ind_tbl[RSS_IND_TBL_LEN_PER_MBX_MSG];
121};
122
123/* Physical interface link status */
124struct bgx_link_status {
125 uint8_t msg;
126 uint8_t mac_type;
127 uint8_t link_up;
128 uint8_t duplex;
129 uint32_t speed;
130};
131
132/* Allocate additional SQS to VF */
133struct sqs_alloc {
134 uint8_t msg;
135 uint8_t spec;
136 uint8_t qs_count;
137 uint8_t svf[MAX_SQS_PER_VF];
138};
139
140/* Set interface in loopback mode */
141struct set_loopback {
142 uint8_t msg;
143 uint8_t vf_id;
144 bool enable;
145};
146
147/* Reset statistics counters */
148struct reset_stat_cfg {
149 uint8_t msg;
150 /* Bitmap to select NIC_PF_VNIC(vf_id)_RX_STAT(0..13) */
151 uint16_t rx_stat_mask;
152 /* Bitmap to select NIC_PF_VNIC(vf_id)_TX_STAT(0..4) */
153 uint8_t tx_stat_mask;
154 /* Bitmap to select NIC_PF_QS(0..127)_RQ(0..7)_STAT(0..1)
155 * bit14, bit15 NIC_PF_QS(vf_id)_RQ7_STAT(0..1)
156 * bit12, bit13 NIC_PF_QS(vf_id)_RQ6_STAT(0..1)
157 * ..
158 * bit2, bit3 NIC_PF_QS(vf_id)_RQ1_STAT(0..1)
159 * bit0, bit1 NIC_PF_QS(vf_id)_RQ0_STAT(0..1)
160 */
161 uint16_t rq_stat_mask;
162 /* Bitmap to select NIC_PF_QS(0..127)_SQ(0..7)_STAT(0..1)
163 * bit14, bit15 NIC_PF_QS(vf_id)_SQ7_STAT(0..1)
164 * bit12, bit13 NIC_PF_QS(vf_id)_SQ6_STAT(0..1)
165 * ..
166 * bit2, bit3 NIC_PF_QS(vf_id)_SQ1_STAT(0..1)
167 * bit0, bit1 NIC_PF_QS(vf_id)_SQ0_STAT(0..1)
168 */
169 uint16_t sq_stat_mask;
170};
171
172struct nic_mbx {
173/* 128 bit shared memory between PF and each VF */
174union {
175 struct { uint8_t msg; } msg;
176 struct nic_cfg_msg nic_cfg;
177 struct qs_cfg_msg qs;
178 struct rq_cfg_msg rq;
179 struct sq_cfg_msg sq;
180 struct set_mac_msg mac;
181 struct set_frs_msg frs;
182 struct cpi_cfg_msg cpi_cfg;
183 struct rss_sz_msg rss_size;
184 struct rss_cfg_msg rss_cfg;
185 struct bgx_link_status link_status;
186 struct sqs_alloc sqs_alloc;
187 struct set_loopback lbk;
188 struct reset_stat_cfg reset_stat;
189};
190};
