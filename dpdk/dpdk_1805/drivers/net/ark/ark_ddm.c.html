11int
12ark_ddm_verify(struct ark_ddm_t *ddm)
13{
14 if (sizeof(struct ark_ddm_t) != ARK_DDM_EXPECTED_SIZE) {
15 PMD_DRV_LOG(ERR, "ARK: DDM structure looks incorrect %d vs %zd\n",
16 ARK_DDM_EXPECTED_SIZE, sizeof(struct ark_ddm_t));
17 return -1;
18 }
19
20 if (ddm->cfg.const0 != ARK_DDM_CONST) {
21 PMD_DRV_LOG(ERR, "ARK: DDM module not found as expected 0x%08x\n",
22 ddm->cfg.const0);
23 return -1;
24 }
25 return 0;
26}
27
28void
29ark_ddm_start(struct ark_ddm_t *ddm)
30{
31 ddm->cfg.command = 1;
32}
33
34int
35ark_ddm_stop(struct ark_ddm_t *ddm, const int wait)
36{
37 int cnt = 0;
38
39 ddm->cfg.command = 2;
40 while (wait && (ddm->cfg.stop_flushed & 0x01) == 0) {
41 if (cnt++ > 1000)
42 return 1;
43
44 usleep(10);
45 }
46 return 0;
47}
48
49void
50ark_ddm_reset(struct ark_ddm_t *ddm)
51{
52 int status;
53
54 /* reset only works if ddm has stopped properly. */
55 status = ark_ddm_stop(ddm, 1);
56
57 if (status != 0) {
58 PMD_DEBUG_LOG(INFO, "%s stop failed doing forced reset\n",
59 __func__);
60 ddm->cfg.command = 4;
61 usleep(10);
62 }
63 ddm->cfg.command = 3;
64}

67ark_ddm_setup(struct ark_ddm_t *ddm, rte_iova_t cons_addr, uint32_t interval)

87void
88ark_ddm_dump_stats(struct ark_ddm_t *ddm, const char *msg)
89{
90 struct ark_ddm_stats_t *stats = &ddm->stats;
91
92 PMD_STATS_LOG(INFO, "DDM Stats: %s"
93 ARK_SU64 ARK_SU64 ARK_SU64
94 "\n", msg,
95 "Bytes:", stats->tx_byte_count,
96 "Packets:", stats->tx_pkt_count,
97 "MBufs", stats->tx_mbuf_count);
98}

100int
101ark_ddm_is_stopped(struct ark_ddm_t *ddm)
