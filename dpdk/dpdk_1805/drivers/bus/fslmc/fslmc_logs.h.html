7#ifndef _FSLMC_LOGS_H_
8#define _FSLMC_LOGS_H_
9
10extern int dpaa2_logtype_bus;
11
12#define DPAA2_BUS_LOG(level, fmt, args...) \
13 rte_log(RTE_LOG_ ## level, dpaa2_logtype_bus, "fslmc: " fmt "\n", \
14 ##args)
15
16/* Debug logs are with Function names */
17#define DPAA2_BUS_DEBUG(fmt, args...) \
18 rte_log(RTE_LOG_DEBUG, dpaa2_logtype_bus, "fslmc: %s(): " fmt "\n", \
19 __func__, ##args)
20
21#define BUS_INIT_FUNC_TRACE() DPAA2_BUS_DEBUG(" >>")
22
23#define DPAA2_BUS_INFO(fmt, args...) \
24 DPAA2_BUS_LOG(INFO, fmt, ## args)
25#define DPAA2_BUS_ERR(fmt, args...) \
26 DPAA2_BUS_LOG(ERR, fmt, ## args)
27#define DPAA2_BUS_WARN(fmt, args...) \
28 DPAA2_BUS_LOG(WARNING, fmt, ## args)
29
30/* DP Logs, toggled out at compile time if level lower than current level */
31#define DPAA2_BUS_DP_LOG(level, fmt, args...) \
32 RTE_LOG_DP(level, PMD, fmt, ## args)
33
34#define DPAA2_BUS_DP_DEBUG(fmt, args...) \
35 DPAA2_BUS_DP_LOG(DEBUG, fmt, ## args)
36#define DPAA2_BUS_DP_INFO(fmt, args...) \
37 DPAA2_BUS_DP_LOG(INFO, fmt, ## args)
38#define DPAA2_BUS_DP_WARN(fmt, args...) \
39 DPAA2_BUS_DP_LOG(WARNING, fmt, ## args)
40
41#endif /* _FSLMC_LOGS_H_ */
