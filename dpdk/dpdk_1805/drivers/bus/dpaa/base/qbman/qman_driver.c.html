27static __thread int qmfd = -1;
28static __thread struct qm_portal_config qpcfg;
29static __thread struct dpaa_ioctl_portal_map map = {
30 .type = dpaa_portal_qman
31};

33static int fsl_qman_portal_init(uint32_t index, int is_shared)
34{
35 cpu_set_t cpuset;
36 struct qman_portal *portal;
37 int loop, ret;
38 struct dpaa_ioctl_irq_map irq_map;
39
40 /* Verify the thread's cpu-affinity */
41 ret = pthread_getaffinity_np(pthread_self(), sizeof(cpu_set_t),
42 &cpuset);
43 if (ret) {
44 error(0, ret, "pthread_getaffinity_np()");
45 return ret;
46 }
47 qpcfg.cpu = -1;
48 for (loop = 0; loop < CPU_SETSIZE; loop++)
49 if (CPU_ISSET(loop, &cpuset)) {
50 if (qpcfg.cpu != -1) {
51 pr_err("Thread is not affine to 1 cpu\n");
52 return -EINVAL;
53 }
54 qpcfg.cpu = loop;
55 }
56 if (qpcfg.cpu == -1) {
57 pr_err("Bug in getaffinity handling!\n");
58 return -EINVAL;
59 }
60
61 /* Allocate and map a qman portal */
62 map.index = index;
63 ret = process_portal_map(&map);
64 if (ret) {
65 error(0, ret, "process_portal_map()");
66 return ret;
67 }
68 qpcfg.channel = map.channel;
69 qpcfg.pools = map.pools;
70 qpcfg.index = map.index;
71
72 /* Make the portal's cache-[enabled|inhibited] regions */
73 qpcfg.addr_virt[DPAA_PORTAL_CE] = map.addr.cena;
74 qpcfg.addr_virt[DPAA_PORTAL_CI] = map.addr.cinh;
75
76 qmfd = open(QMAN_PORTAL_IRQ_PATH, O_RDONLY);
77 if (qmfd == -1) {
78 pr_err("QMan irq init failed\n");
79 process_portal_unmap(&map.addr);
80 return -EBUSY;
81 }
82
83 qpcfg.is_shared = is_shared;
84 qpcfg.node = NULL;
85 qpcfg.irq = qmfd;
86
87 portal = qman_create_affine_portal(&qpcfg, NULL, 0);
88 if (!portal) {
89 pr_err("Qman portal initialisation failed (%d)\n",
90 qpcfg.cpu);
91 process_portal_unmap(&map.addr);
92 return -EBUSY;
93 }
94
95 irq_map.type = dpaa_portal_qman;
96 irq_map.portal_cinh = map.addr.cinh;
97 process_portal_irq_map(qmfd, &irq_map);
98 return 0;
99}
100
101static int fsl_qman_portal_finish(void)
102{
103 __maybe_unused const struct qm_portal_config *cfg;
104 int ret;
105
106 process_portal_irq_unmap(qmfd);
107
108 cfg = qman_destroy_affine_portal(NULL);
109 DPAA_BUG_ON(cfg != &qpcfg);
110 ret = process_portal_unmap(&map.addr);
111 if (ret)
112 error(0, ret, "process_portal_unmap()");
113 return ret;
114}

116int qman_thread_init(void)
117{
118 /* Convert from contiguous/virtual cpu numbering to real cpu when
119 * calling into the code that is dependent on the device naming.
120 */
121 return fsl_qman_portal_init(QBMAN_ANY_PORTAL_IDX, 0);
122}
123
124int qman_thread_finish(void)
125{
126 return fsl_qman_portal_finish();
127}
128
129void qman_thread_irq(void)
130{
131 qbman_invoke_irq(qpcfg.irq);
132
133 /* Now we need to uninhibit interrupts. This is the only code outside
134 * the regular portal driver that manipulates any portal register, so
135 * rather than breaking that encapsulation I am simply hard-coding the
136 * offset to the inhibit register here.
137 */
138 out_be32(qpcfg.addr_virt[DPAA_PORTAL_CI] + 0xe0c, 0);
139}

141struct qman_portal *fsl_qman_portal_create(void)
142{
143 cpu_set_t cpuset;
144 struct qman_portal *res;
145
146 struct qm_portal_config *q_pcfg;
147 int loop, ret;
148 struct dpaa_ioctl_irq_map irq_map;
149 struct dpaa_ioctl_portal_map q_map = {0};
150 int q_fd;
151
152 q_pcfg = kzalloc((sizeof(struct qm_portal_config)), 0);
153 if (!q_pcfg) {
154 error(0, -1, "q_pcfg kzalloc failed");
155 return NULL;
156 }
157
158 /* Verify the thread's cpu-affinity */
159 ret = pthread_getaffinity_np(pthread_self(), sizeof(cpu_set_t),
160 &cpuset);
161 if (ret) {
162 error(0, ret, "pthread_getaffinity_np()");
163 kfree(q_pcfg);
164 return NULL;
165 }
166
167 q_pcfg->cpu = -1;
168 for (loop = 0; loop < CPU_SETSIZE; loop++)
169 if (CPU_ISSET(loop, &cpuset)) {
170 if (q_pcfg->cpu != -1) {
171 pr_err("Thread is not affine to 1 cpu\n");
172 kfree(q_pcfg);
173 return NULL;
174 }
175 q_pcfg->cpu = loop;
176 }
177 if (q_pcfg->cpu == -1) {
178 pr_err("Bug in getaffinity handling!\n");
179 kfree(q_pcfg);
180 return NULL;
181 }
182
183 /* Allocate and map a qman portal */
184 q_map.type = dpaa_portal_qman;
185 q_map.index = QBMAN_ANY_PORTAL_IDX;
186 ret = process_portal_map(&q_map);
187 if (ret) {
188 error(0, ret, "process_portal_map()");
189 kfree(q_pcfg);
190 return NULL;
191 }
192 q_pcfg->channel = q_map.channel;
193 q_pcfg->pools = q_map.pools;
194 q_pcfg->index = q_map.index;
195
196 /* Make the portal's cache-[enabled|inhibited] regions */
197 q_pcfg->addr_virt[DPAA_PORTAL_CE] = q_map.addr.cena;
198 q_pcfg->addr_virt[DPAA_PORTAL_CI] = q_map.addr.cinh;
199
200 q_fd = open(QMAN_PORTAL_IRQ_PATH, O_RDONLY);
201 if (q_fd == -1) {
202 pr_err("QMan irq init failed\n");
203 goto err1;
204 }
205
206 q_pcfg->irq = q_fd;
207
208 res = qman_create_affine_portal(q_pcfg, NULL, true);
209 if (!res) {
210 pr_err("Qman portal initialisation failed (%d)\n",
211 q_pcfg->cpu);
212 goto err2;
213 }
214
215 irq_map.type = dpaa_portal_qman;
216 irq_map.portal_cinh = q_map.addr.cinh;
217 process_portal_irq_map(q_fd, &irq_map);
218
219 return res;
220err2:
221 close(q_fd);
222err1:
223 process_portal_unmap(&q_map.addr);
224 kfree(q_pcfg);
225 return NULL;
226}

228int fsl_qman_portal_destroy(struct qman_portal *qp)
229{
230 const struct qm_portal_config *cfg;
231 struct dpaa_portal_map addr;
232 int ret;
233
234 cfg = qman_destroy_affine_portal(qp);
235 kfree(qp);
236
237 process_portal_irq_unmap(cfg->irq);
238
239 addr.cena = cfg->addr_virt[DPAA_PORTAL_CE];
240 addr.cinh = cfg->addr_virt[DPAA_PORTAL_CI];
241
242 ret = process_portal_unmap(&addr);
243 if (ret)
244 pr_err("process_portal_unmap() (%d)\n", ret);
245
246 kfree((void *)cfg);
247
248 return ret;
249}

251int qman_global_init(void)
252{
253 const struct device_node *dt_node;
254 size_t lenp;
255 const u32 *chanid;
256 static int ccsr_map_fd;
257 const uint32_t *qman_addr;
258 uint64_t phys_addr;
259 uint64_t regs_size;
260 const u32 *clk;
261
262 static int done;
263
264 if (done)
265 return -EBUSY;
266
267 /* Use the device-tree to determine IP revision until something better
268 * is devised.
269 */
270 dt_node = of_find_compatible_node(NULL, NULL, "fsl,qman-portal");
271 if (!dt_node) {
272 pr_err("No qman portals available for any CPU\n");
273 return -ENODEV;
274 }
275 if (of_device_is_compatible(dt_node, "fsl,qman-portal-1.0") ||
276 of_device_is_compatible(dt_node, "fsl,qman-portal-1.0.0"))
277 pr_err("QMan rev1.0 on P4080 rev1 is not supported!\n");
278 else if (of_device_is_compatible(dt_node, "fsl,qman-portal-1.1") ||
279 of_device_is_compatible(dt_node, "fsl,qman-portal-1.1.0"))
280 qman_ip_rev = QMAN_REV11;
281 else if (of_device_is_compatible(dt_node, "fsl,qman-portal-1.2") ||
282 of_device_is_compatible(dt_node, "fsl,qman-portal-1.2.0"))
283 qman_ip_rev = QMAN_REV12;
284 else if (of_device_is_compatible(dt_node, "fsl,qman-portal-2.0") ||
285 of_device_is_compatible(dt_node, "fsl,qman-portal-2.0.0"))
286 qman_ip_rev = QMAN_REV20;
287 else if (of_device_is_compatible(dt_node, "fsl,qman-portal-3.0.0") ||
288 of_device_is_compatible(dt_node, "fsl,qman-portal-3.0.1"))
289 qman_ip_rev = QMAN_REV30;
290 else if (of_device_is_compatible(dt_node, "fsl,qman-portal-3.1.0") ||
291 of_device_is_compatible(dt_node, "fsl,qman-portal-3.1.1") ||
292 of_device_is_compatible(dt_node, "fsl,qman-portal-3.1.2") ||
293 of_device_is_compatible(dt_node, "fsl,qman-portal-3.1.3"))
294 qman_ip_rev = QMAN_REV31;
295 else if (of_device_is_compatible(dt_node, "fsl,qman-portal-3.2.0") ||
296 of_device_is_compatible(dt_node, "fsl,qman-portal-3.2.1"))
297 qman_ip_rev = QMAN_REV32;
298 else
299 qman_ip_rev = QMAN_REV11;
300
301 if (!qman_ip_rev) {
302 pr_err("Unknown qman portal version\n");
303 return -ENODEV;
304 }
305 if ((qman_ip_rev & 0xFF00) >= QMAN_REV30) {
306 qm_channel_pool1 = QMAN_CHANNEL_POOL1_REV3;
307 qm_channel_caam = QMAN_CHANNEL_CAAM_REV3;
308 qm_channel_pme = QMAN_CHANNEL_PME_REV3;
309 }
310
311 dt_node = of_find_compatible_node(NULL, NULL, "fsl,pool-channel-range");
312 if (!dt_node) {
313 pr_err("No qman pool channel range available\n");
314 return -ENODEV;
315 }
316 chanid = of_get_property(dt_node, "fsl,pool-channel-range", &lenp);
317 if (!chanid) {
318 pr_err("Can not get pool-channel-range property\n");
319 return -EINVAL;
320 }
321
322 /* get ccsr base */
323 dt_node = of_find_compatible_node(NULL, NULL, "fsl,qman");
324 if (!dt_node) {
325 pr_err("No qman device node available\n");
326 return -ENODEV;
327 }
328 qman_addr = of_get_address(dt_node, 0, &regs_size, NULL);
329 if (!qman_addr) {
330 pr_err("of_get_address cannot return qman address\n");
331 return -EINVAL;
332 }
333 phys_addr = of_translate_address(dt_node, qman_addr);
334 if (!phys_addr) {
335 pr_err("of_translate_address failed\n");
336 return -EINVAL;
337 }
338
339 ccsr_map_fd = open("/dev/mem", O_RDWR);
340 if (unlikely(ccsr_map_fd < 0)) {
341 pr_err("Can not open /dev/mem for qman ccsr map\n");
342 return ccsr_map_fd;
343 }
344
345 qman_ccsr_map = mmap(NULL, regs_size, PROT_READ | PROT_WRITE,
346 MAP_SHARED, ccsr_map_fd, phys_addr);
347 if (qman_ccsr_map == MAP_FAILED) {
348 pr_err("Can not map qman ccsr base\n");
349 return -EINVAL;
350 }
351
352 clk = of_get_property(dt_node, "clock-frequency", NULL);
353 if (!clk)
354 pr_warn("Can't find Qman clock frequency\n");
355 else
356 qman_clk = be32_to_cpu(*clk);
357
358#ifdef CONFIG_FSL_QMAN_FQ_LOOKUP
359 return qman_setup_fq_lookup_table(CONFIG_FSL_QMAN_FQ_LOOKUP_MAX);
360#endif
361 return 0;
362}
