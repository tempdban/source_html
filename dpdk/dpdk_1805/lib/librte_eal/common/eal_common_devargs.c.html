29int
30rte_eal_parse_devargs_str(const char *devargs_str,
31 char **drvname, char **drvargs)
32{
33 char *sep;
34
35 if ((devargs_str) == NULL || (drvname) == NULL || (drvargs == NULL))
36 return -1;
37
38 *drvname = strdup(devargs_str);
39 if (*drvname == NULL)
40 return -1;
41
42 /* set the first ',' to '\0' to split name and arguments */
43 sep = strchr(*drvname, ',');
44 if (sep != NULL) {
45 sep[0] = '\0';
46 *drvargs = strdup(sep + 1);
47 } else {
48 *drvargs = strdup("");
49 }
50
51 if (*drvargs == NULL) {
52 free(*drvname);
53 *drvname = NULL;
54 return -1;
55 }
56 return 0;
57}

59static int
60bus_name_cmp(const struct rte_bus *bus, const void *name)
61{
62 return strncmp(bus->name, name, strlen(bus->name));
63}

65int __rte_experimental
66rte_devargs_parse(struct rte_devargs *da, const char *format, ...)
67{
68 struct rte_bus *bus = NULL;
69 va_list ap;
70 va_start(ap, format);
71 char dev[vsnprintf(NULL, 0, format, ap) + 1];
72 const char *devname;
73 const size_t maxlen = sizeof(da->name);
74 size_t i;
75
76 va_end(ap);
77 if (da == NULL)
78 return -EINVAL;
79
80 va_start(ap, format);
81 vsnprintf(dev, sizeof(dev), format, ap);
82 va_end(ap);
83 /* Retrieve eventual bus info */
84 do {
85 devname = dev;
86 bus = rte_bus_find(bus, bus_name_cmp, dev);
87 if (bus == NULL)
88 break;
89 devname = dev + strlen(bus->name) + 1;
90 if (rte_bus_find_by_device_name(devname) == bus)
91 break;
92 } while (1);
93 /* Store device name */
94 i = 0;
95 while (devname[i] != '\0' && devname[i] != ',') {
96 da->name[i] = devname[i];
97 i++;
98 if (i == maxlen) {
99 fprintf(stderr, "WARNING: Parsing \"%s\": device name should be shorter than %zu\n",
100 dev, maxlen);
101 da->name[i - 1] = '\0';
102 return -EINVAL;
103 }
104 }
105 da->name[i] = '\0';
106 if (bus == NULL) {
107 bus = rte_bus_find_by_device_name(da->name);
108 if (bus == NULL) {
109 fprintf(stderr, "ERROR: failed to parse device \"%s\"\n",
110 da->name);
111 return -EFAULT;
112 }
113 }
114 da->bus = bus;
115 /* Parse eventual device arguments */
116 if (devname[i] == ',')
117 da->args = strdup(&devname[i + 1]);
118 else
119 da->args = strdup("");
120 if (da->args == NULL) {
121 fprintf(stderr, "ERROR: not enough memory to parse arguments\n");
122 return -ENOMEM;
123 }
124 return 0;
125}

127int __rte_experimental
128rte_devargs_insert(struct rte_devargs *da)
129{
130 int ret;
131
132 ret = rte_devargs_remove(da->bus->name, da->name);
133 if (ret < 0)
134 return ret;
135 TAILQ_INSERT_TAIL(&devargs_list, da, next);
136 return 0;
137}

139/* store a whitelist parameter for later parsing */
140__rte_experimental
141int
142rte_devargs_add(enum rte_devtype devtype, const char *devargs_str)
143{
144 struct rte_devargs *devargs = NULL;
145 struct rte_bus *bus = NULL;
146 const char *dev = devargs_str;
147
148 /* use calloc instead of rte_zmalloc as it's called early at init */
149 devargs = calloc(1, sizeof(*devargs));
150 if (devargs == NULL)
151 goto fail;
152
153 if (rte_devargs_parse(devargs, "%s", dev))
154 goto fail;
155 devargs->type = devtype;
156 bus = devargs->bus;
157 if (devargs->type == RTE_DEVTYPE_BLACKLISTED_PCI)
158 devargs->policy = RTE_DEV_BLACKLISTED;
159 if (bus->conf.scan_mode == RTE_BUS_SCAN_UNDEFINED) {
160 if (devargs->policy == RTE_DEV_WHITELISTED)
161 bus->conf.scan_mode = RTE_BUS_SCAN_WHITELIST;
162 else if (devargs->policy == RTE_DEV_BLACKLISTED)
163 bus->conf.scan_mode = RTE_BUS_SCAN_BLACKLIST;
164 }
165 TAILQ_INSERT_TAIL(&devargs_list, devargs, next);
166 return 0;
167
168fail:
169 if (devargs) {
170 free(devargs->args);
171 free(devargs);
172 }
173
174 return -1;
175}

177int __rte_experimental
178rte_devargs_remove(const char *busname, const char *devname)
179{
180 struct rte_devargs *d;
181 void *tmp;
182
183 TAILQ_FOREACH_SAFE(d, &devargs_list, next, tmp) {
184 if (strcmp(d->bus->name, busname) == 0 &&
185 strcmp(d->name, devname) == 0) {
186 TAILQ_REMOVE(&devargs_list, d, next);
187 free(d->args);
188 free(d);
189 return 0;
190 }
191 }
192 return 1;
193}

195/* count the number of devices of a specified type */
196__rte_experimental
197unsigned int
198rte_devargs_type_count(enum rte_devtype devtype)
199{
200 struct rte_devargs *devargs;
201 unsigned int count = 0;
202
203 TAILQ_FOREACH(devargs, &devargs_list, next) {
204 if (devargs->type != devtype)
205 continue;
206 count++;
207 }
208 return count;
209}

211/* dump the user devices on the console */
212__rte_experimental
213void
214rte_devargs_dump(FILE *f)
215{
216 struct rte_devargs *devargs;
217
218 fprintf(f, "User device list:\n");
219 TAILQ_FOREACH(devargs, &devargs_list, next) {
220 fprintf(f, " [%s]: %s %s\n",
221 (devargs->bus ? devargs->bus->name : "??"),
222 devargs->name, devargs->args);
223 }
224}

226/* bus-aware rte_devargs iterator. */
227__rte_experimental
228struct rte_devargs *
229rte_devargs_next(const char *busname, const struct rte_devargs *start)
230{
231 struct rte_devargs *da;
232
233 if (start != NULL)
234 da = TAILQ_NEXT(start, next);
235 else
236 da = TAILQ_FIRST(&devargs_list);
237 while (da != NULL) {
238 if (busname == NULL ||
239 (strcmp(busname, da->bus->name) == 0))
240 return da;
241 da = TAILQ_NEXT(da, next);
242 }
243 return NULL;
244}
